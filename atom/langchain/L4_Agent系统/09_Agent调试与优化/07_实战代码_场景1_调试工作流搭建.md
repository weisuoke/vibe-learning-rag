# Agent调试与优化 - 实战代码场景1：调试工作流搭建

> 从零搭建完整的 Agent 调试环境，包含回调追踪、日志记录和问题定位。

---

## 场景描述

你构建了一个多工具 Agent，但它经常：
- 调用了错误的工具
- 返回了不完整的答案
- 偶尔陷入死循环

你需要搭建一套调试工作流，快速定位这些问题。

---

## 完整代码

```python
"""
Agent 调试工作流搭建
演示：从零构建完整的 Agent 调试环境

包含：
1. 分层调试回调系统
2. 结构化日志记录
3. 执行过程可视化
4. 问题自动检测
"""

import os
import json
import time
from datetime import datetime
from typing import Any, Optional

from dotenv import load_dotenv
load_dotenv()

from langchain_openai import ChatOpenAI
from langchain_core.callbacks import BaseCallbackHandler
from langchain_core.tools import tool
from langchain_core.messages import HumanMessage

# ===== 1. 构建分层调试回调 =====
print("=" * 60)
print("1. 构建分层调试回调系统")
print("=" * 60)


class AgentDebugHandler(BaseCallbackHandler):
    """
    完整的 Agent 调试回调处理器

    功能：
    - 记录每一步的时间、输入、输出
    - 自动检测异常模式（死循环、重复调用）
    - 生成调试摘要报告
    """

    def __init__(self, verbose_level: int = 1):
        """
        Args:
            verbose_level: 详细程度
                0 = 只记录错误
                1 = 记录关键事件（默认）
                2 = 记录所有细节
        """
        self.verbose_level = verbose_level
        self.events = []          # 所有事件记录
        self.tool_calls = []      # 工具调用记录
        self.errors = []          # 错误记录
        self.start_time = None
        self.step_count = 0
        self.llm_call_count = 0
        self.total_tokens = 0

    def _log(self, level: int, emoji: str, message: str):
        """分级输出日志"""
        if self.verbose_level >= level:
            timestamp = datetime.now().strftime("%H:%M:%S.%f")[:-3]
            print(f"  [{timestamp}] {emoji} {message}")

    # ----- LLM 事件 -----
    def on_llm_start(self, serialized: dict, prompts: list, **kwargs):
        self.llm_call_count += 1
        if self.start_time is None:
            self.start_time = time.time()

        prompt_preview = prompts[0][:80] if prompts else "N/A"
        self._log(1, "🤖", f"LLM 调用 #{self.llm_call_count}")
        self._log(2, "  ", f"提示词预览: {prompt_preview}...")
        self.events.append({
            "type": "llm_start",
            "step": self.llm_call_count,
            "time": time.time()
        })

    def on_llm_end(self, response, **kwargs):
        # 统计 token 使用
        if hasattr(response, "llm_output") and response.llm_output:
            usage = response.llm_output.get("token_usage", {})
            tokens = usage.get("total_tokens", 0)
            self.total_tokens += tokens
            self._log(2, "📊", f"Token 使用: {tokens}")

    def on_llm_error(self, error, **kwargs):
        self.errors.append({"type": "llm_error", "error": str(error)})
        self._log(0, "❌", f"LLM 错误: {error}")

    # ----- Agent 事件 -----
    def on_agent_action(self, action, **kwargs):
        self.step_count += 1
        tool_name = action.tool
        tool_input = str(action.tool_input)[:100]

        self._log(1, "🎯", f"步骤 {self.step_count}: 使用工具 [{tool_name}]")
        self._log(2, "  ", f"输入: {tool_input}")

        self.tool_calls.append({
            "step": self.step_count,
            "tool": tool_name,
            "input": action.tool_input,
            "time": time.time()
        })

        # 自动检测：重复调用同一工具
        if len(self.tool_calls) >= 3:
            recent = [tc["tool"] for tc in self.tool_calls[-3:]]
            if len(set(recent)) == 1:
                self._log(0, "⚠️", f"警告: 连续 3 次调用 [{recent[0]}]，可能陷入循环！")

    def on_agent_finish(self, finish, **kwargs):
        elapsed = time.time() - self.start_time if self.start_time else 0
        self._log(1, "✅", f"Agent 完成 (用时 {elapsed:.2f}s)")

    # ----- 工具事件 -----
    def on_tool_start(self, serialized, input_str, **kwargs):
        self._log(2, "🔧", f"工具开始执行: {serialized.get('name', 'unknown')}")

    def on_tool_end(self, output, **kwargs):
        output_preview = str(output)[:100]
        self._log(2, "📤", f"工具返回: {output_preview}")

    def on_tool_error(self, error, **kwargs):
        self.errors.append({"type": "tool_error", "error": str(error)})
        self._log(0, "❌", f"工具错误: {error}")

    # ----- Chain 事件 -----
    def on_chain_start(self, serialized, inputs, **kwargs):
        name = serialized.get("name", serialized.get("id", ["unknown"])[-1])
        self._log(2, "⛓️", f"链开始: {name}")

    def on_chain_error(self, error, **kwargs):
        self.errors.append({"type": "chain_error", "error": str(error)})
        self._log(0, "❌", f"链错误: {error}")

    # ----- 调试报告 -----
    def get_summary(self) -> dict:
        """生成调试摘要报告"""
        elapsed = time.time() - self.start_time if self.start_time else 0
        tool_counts = {}
        for tc in self.tool_calls:
            tool_counts[tc["tool"]] = tool_counts.get(tc["tool"], 0) + 1

        return {
            "total_time_sec": round(elapsed, 2),
            "llm_calls": self.llm_call_count,
            "total_steps": self.step_count,
            "total_tokens": self.total_tokens,
            "tool_usage": tool_counts,
            "errors": len(self.errors),
            "error_details": self.errors
        }

    def print_summary(self):
        """打印格式化的调试摘要"""
        summary = self.get_summary()
        print("\n" + "=" * 50)
        print("📋 调试摘要报告")
        print("=" * 50)
        print(f"  总用时:     {summary['total_time_sec']}s")
        print(f"  LLM 调用:  {summary['llm_calls']} 次")
        print(f"  Agent 步骤: {summary['total_steps']} 步")
        print(f"  Token 使用: {summary['total_tokens']}")
        print(f"  工具使用:   {summary['tool_usage']}")
        print(f"  错误数量:   {summary['errors']}")
        if summary["errors"] > 0:
            print(f"  错误详情:")
            for err in summary["error_details"]:
                print(f"    - [{err['type']}] {err['error'][:80]}")
        print("=" * 50)


# ===== 2. 定义测试工具 =====
print("\n" + "=" * 60)
print("2. 定义测试工具")
print("=" * 60)


@tool
def search_web(query: str) -> str:
    """在互联网上搜索信息。输入搜索关键词，返回搜索结果摘要。"""
    # 模拟搜索结果
    return f"搜索 '{query}' 的结果: LangChain 是一个用于构建 LLM 应用的框架，支持 Agent、RAG 等模式。"


@tool
def calculate(expression: str) -> str:
    """计算数学表达式。输入数学表达式字符串，返回计算结果。"""
    try:
        result = eval(expression)
        return f"计算结果: {expression} = {result}"
    except Exception as e:
        return f"计算错误: {e}"


@tool
def get_current_time() -> str:
    """获取当前时间。不需要输入参数。"""
    return f"当前时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}"


tools = [search_web, calculate, get_current_time]
print(f"  已注册 {len(tools)} 个工具: {[t.name for t in tools]}")


# ===== 3. 创建 Agent 并挂载调试 =====
print("\n" + "=" * 60)
print("3. 创建 Agent 并运行调试")
print("=" * 60)

llm = ChatOpenAI(
    model="gpt-4o-mini",
    temperature=0,
    max_retries=2,
)

# 绑定工具
llm_with_tools = llm.bind_tools(tools)

# 创建调试回调（verbose_level=1 显示关键事件）
debug_handler = AgentDebugHandler(verbose_level=1)

# ===== 4. 简化的 Agent 执行循环 =====
print("\n--- 运行 Agent ---")

from langchain_core.messages import AIMessage, ToolMessage

def run_agent_with_debug(query: str, handler: AgentDebugHandler):
    """带调试的简化 Agent 执行循环"""
    messages = [HumanMessage(content=query)]
    max_iterations = 5

    for i in range(max_iterations):
        # 调用 LLM
        response = llm_with_tools.invoke(
            messages,
            config={"callbacks": [handler]}
        )
        messages.append(response)

        # 检查是否有工具调用
        if not response.tool_calls:
            print(f"\n  💬 最终回答: {response.content[:200]}")
            return response.content

        # 执行工具调用
        for tool_call in response.tool_calls:
            tool_name = tool_call["name"]
            tool_args = tool_call["args"]

            # 记录 Agent 决策
            from langchain_core.agents import AgentAction
            action = AgentAction(tool=tool_name, tool_input=tool_args, log="")
            handler.on_agent_action(action)

            # 查找并执行工具
            tool_map = {t.name: t for t in tools}
            if tool_name in tool_map:
                try:
                    result = tool_map[tool_name].invoke(tool_args)
                    handler.on_tool_end(result)
                except Exception as e:
                    result = f"工具执行失败: {e}"
                    handler.on_tool_error(e)
            else:
                result = f"未找到工具: {tool_name}"

            messages.append(ToolMessage(
                content=str(result),
                tool_call_id=tool_call["id"]
            ))

    print("  ⚠️ 达到最大迭代次数")
    return "达到最大迭代次数，请简化问题重试。"


# 执行测试查询
query = "请帮我搜索 LangChain 的最新信息，然后计算 2024 * 3 + 100"
print(f"\n  用户问题: {query}\n")

result = run_agent_with_debug(query, debug_handler)

# 打印调试摘要
debug_handler.print_summary()


# ===== 5. 日志持久化 =====
print("\n" + "=" * 60)
print("5. 调试日志持久化")
print("=" * 60)

# 将调试报告保存为 JSON
summary = debug_handler.get_summary()
log_file = f"debug_log_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
print(f"  调试报告已生成（可保存到 {log_file}）")
print(f"  报告内容:")
print(f"  {json.dumps(summary, indent=2, ensure_ascii=False)}")


# ===== 6. 快速诊断工具函数 =====
print("\n" + "=" * 60)
print("6. 快速诊断工具")
print("=" * 60)


def diagnose_agent(summary: dict) -> list[str]:
    """根据调试摘要自动诊断问题"""
    issues = []

    # 检查：LLM 调用过多
    if summary["llm_calls"] > 10:
        issues.append(
            f"⚠️ LLM 调用 {summary['llm_calls']} 次过多，"
            f"考虑精简 prompt 或减少工具数量"
        )

    # 检查：有错误发生
    if summary["errors"] > 0:
        issues.append(
            f"❌ 发现 {summary['errors']} 个错误，"
            f"请检查工具实现和输入格式"
        )

    # 检查：耗时过长
    if summary["total_time_sec"] > 30:
        issues.append(
            f"🐌 总耗时 {summary['total_time_sec']}s 过长，"
            f"考虑启用缓存或并行执行"
        )

    # 检查：工具使用不均衡
    if summary["tool_usage"]:
        max_calls = max(summary["tool_usage"].values())
        if max_calls >= 3:
            tool_name = [k for k, v in summary["tool_usage"].items() if v == max_calls][0]
            issues.append(
                f"🔄 工具 [{tool_name}] 被调用 {max_calls} 次，"
                f"可能存在重复调用，检查工具描述和 prompt"
            )

    if not issues:
        issues.append("✅ 未发现明显问题")

    return issues


# 执行诊断
print("\n  🏥 自动诊断结果:")
issues = diagnose_agent(summary)
for issue in issues:
    print(f"  {issue}")

print("\n" + "=" * 60)
print("✅ 调试工作流搭建完成！")
print("=" * 60)
```

## 关键技巧总结

| 技巧 | 说明 |
|------|------|
| `verbose_level` 分级 | 0=错误, 1=关键, 2=全部，按需调节 |
| 自动循环检测 | 连续 3 次调用同一工具自动告警 |
| 调试摘要报告 | 执行后一键生成统计信息 |
| 自动诊断函数 | 根据指标自动发现问题 |
| 日志持久化 | JSON 格式保存，方便后续分析 |
