```typescript
/**
 * 诊断工具集
 */

import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export interface DiagnosticResult {
  component: string;
  status: 'ok' | 'warning' | 'error';
  message: string;
  details?: any;
}

export class Diagnostics {
  /**
   * 检查 Gateway 状态
   */
  async checkGateway(): Promise<DiagnosticResult> {
    try {
      const { stdout } = await execAsync('openclaw gateway status');

      if (stdout.includes('Running')) {
        return {
          component: 'Gateway',
          status: 'ok',
          message: 'Gateway is running',
          details: this.parseGatewayStatus(stdout)
        };
      }

      return {
        component: 'Gateway',
        status: 'error',
        message: 'Gateway is not running'
      };
    } catch (error) {
      return {
        component: 'Gateway',
        status: 'error',
        message: 'Failed to check Gateway status',
        details: error
      };
    }
  }

  /**
   * 检查配对状态
   */
  async checkPairings(): Promise<DiagnosticResult> {
    try {
      const { stdout } = await execAsync('openclaw pairing list');
      const pairings = this.parsePairingList(stdout);

      if (pairings.length === 0) {
        return {
          component: 'Pairing',
          status: 'warning',
          message: 'No active pairings found'
        };
      }

      const activePairings = pairings.filter(p => p.status === 'active');

      if (activePairings.length === 0) {
        return {
          component: 'Pairing',
          status: 'error',
          message: 'All pairings are inactive',
          details: pairings
        };
      }

      return {
        component: 'Pairing',
        status: 'ok',
        message: `${activePairings.length} active pairing(s)`,
        details: pairings
      };
    } catch (error) {
      return {
        component: 'Pairing',
        status: 'error',
        message: 'Failed to check pairings',
        details: error
      };
    }
  }

  /**
   * 测试消息发送
   */
  async testMessageSend(channel?: string): Promise<DiagnosticResult> {
    try {
      const testMessage = `Test message at ${new Date().toISOString()}`;
      const command = channel
        ? `openclaw message send "${testMessage}" --channel ${channel}`
        : `openclaw message send "${testMessage}"`;

      const { stdout } = await execAsync(command);

      if (stdout.includes('Message sent')) {
        return {
          component: 'Message Send',
          status: 'ok',
          message: 'Message sent successfully',
          details: { channel, testMessage }
        };
      }

      return {
        component: 'Message Send',
        status: 'error',
        message: 'Message send failed'
      };
    } catch (error) {
      return {
        component: 'Message Send',
        status: 'error',
        message: 'Failed to send test message',
        details: error
      };
    }
  }

  /**
   * 检查网络连接
   */
  async checkNetwork(): Promise<DiagnosticResult> {
    try {
      // 测试网络连接
      const { stdout } = await execAsync('ping -c 1 8.8.8.8');

      if (stdout.includes('1 packets transmitted, 1 received')) {
        return {
          component: 'Network',
          status: 'ok',
          message: 'Network connection is working'
        };
      }

      return {
        component: 'Network',
        status: 'error',
        message: 'Network connection failed'
      };
    } catch (error) {
      return {
        component: 'Network',
        status: 'error',
        message: 'Network check failed',
        details: error
      };
    }
  }

  /**
   * 运行完整诊断
   */
  async runFullDiagnostics(): Promise<DiagnosticResult[]> {
    console.log('🔍 Running full diagnostics...\n');

    const results: DiagnosticResult[] = [];

    // 检查 Gateway
    console.log('Checking Gateway...');
    results.push(await this.checkGateway());

    // 检查配对
    console.log('Checking Pairings...');
    results.push(await this.checkPairings());

    // 检查网络
    console.log('Checking Network...');
    results.push(await this.checkNetwork());

    // 测试消息发送
    console.log('Testing Message Send...');
    results.push(await this.testMessageSend());

    return results;
  }

  /**
   * 显示诊断结果
   */
  displayResults(results: DiagnosticResult[]): void {
    console.log('\n' + '='.repeat(60));
    console.log('📊 Diagnostic Results');
    console.log('='.repeat(60));

    for (const result of results) {
      const icon = result.status === 'ok' ? '✅' :
                   result.status === 'warning' ? '⚠️' : '❌';

      console.log(`\n${icon} ${result.component}: ${result.status.toUpperCase()}`);
      console.log(`   ${result.message}`);

      if (result.details) {
        console.log(`   Details: ${JSON.stringify(result.details, null, 2)}`);
      }
    }

    console.log('\n' + '='.repeat(60));

    // 总结
    const okCount = results.filter(r => r.status === 'ok').length;
    const warningCount = results.filter(r => r.status === 'warning').length;
    const errorCount = results.filter(r => r.status === 'error').length;

    console.log(`\nSummary: ${okCount} OK, ${warningCount} Warning, ${errorCount} Error`);

    if (errorCount > 0) {
      console.log('\n⚠️  Issues detected. See recommendations below.');
      this.displayRecommendations(results);
    } else {
      console.log('\n✅ All systems operational!');
    }
  }

  /**
   * 显示修复建议
   */
  private displayRecommendations(results: DiagnosticResult[]): void {
    console.log('\n📋 Recommendations:\n');

    for (const result of results) {
      if (result.status === 'error' || result.status === 'warning') {
        console.log(`${result.component}:`);

        switch (result.component) {
          case 'Gateway':
            console.log('  - Run: openclaw gateway start');
            console.log('  - Check logs: openclaw gateway logs');
            break;
          case 'Pairing':
            console.log('  - Run: openclaw pairing start');
            console.log('  - Verify pairing: openclaw pairing list');
            break;
          case 'Network':
            console.log('  - Check internet connection');
            console.log('  - Verify firewall settings');
            break;
          case 'Message Send':
            console.log('  - Verify Gateway is running');
            console.log('  - Check pairing status');
            console.log('  - Review error logs');
            break;
        }
        console.log();
      }
    }
  }

  /**
   * 解析 Gateway 状态
   */
  private parseGatewayStatus(output: string): any {
    const portMatch = output.match(/Port: (\d+)/);
    const uptimeMatch = output.match(/Uptime: (\d+)/);

    return {
      port: portMatch ? parseInt(portMatch[1]) : null,
      uptime: uptimeMatch ? parseInt(uptimeMatch[1]) : null
    };
  }

  /**
   * 解析配对列表
   */
  private parsePairingList(output: string): Array<{
    channel: string;
    user: string;
    status: string;
  }> {
    const lines = output.split('\n').filter(line => line.includes('│'));
    const pairings: any[] = [];

    for (const line of lines) {
      if (line.includes('Channel') || line.includes('─')) continue;

      const parts = line.split('│').map(p => p.trim()).filter(p => p);
      if (parts.length >= 3) {
        pairings.push({
          channel: parts[0],
          user: parts[1],
          status: parts[2].includes('Active') ? 'active' : 'inactive'
        });
      }
    }

    return pairings;
  }
}
```

### 2. src/health-check.ts

```typescript
/**
 * 健康检查服务
 */

import { Diagnostics } from './diagnostics.js';

export class HealthCheckService {
  private diagnostics: Diagnostics;
  private checkInterval: NodeJS.Timeout | null = null;

  constructor() {
    this.diagnostics = new Diagnostics();
  }

  /**
   * 启动定期健康检查
   */
  startPeriodicCheck(intervalMs: number = 60000): void {
    console.log(`🏥 Starting periodic health check (every ${intervalMs / 1000}s)...\n`);

    // 立即执行一次
    this.runCheck();

    // 定期执行
    this.checkInterval = setInterval(() => {
      this.runCheck();
    }, intervalMs);
  }

  /**
   * 停止定期检查
   */
  stopPeriodicCheck(): void {
    if (this.checkInterval) {
      clearInterval(this.checkInterval);
      this.checkInterval = null;
      console.log('\n🛑 Stopped periodic health check');
    }
  }

  /**
   * 运行单次检查
   */
  private async runCheck(): Promise<void> {
    const timestamp = new Date().toISOString();
    console.log(`\n[${timestamp}] Running health check...`);

    const results = await this.diagnostics.runFullDiagnostics();
    this.diagnostics.displayResults(results);

    // 检查是否有严重问题
    const hasErrors = results.some(r => r.status === 'error');
    if (hasErrors) {
      console.log('\n⚠️  Critical issues detected! Taking action...');
      await this.attemptAutoFix(results);
    }
  }

  /**
   * 尝试自动修复
   */
  private async attemptAutoFix(results: any[]): Promise<void> {
    for (const result of results) {
      if (result.status === 'error') {
        console.log(`\nAttempting to fix: ${result.component}...`);

        switch (result.component) {
          case 'Gateway':
            await this.fixGateway();
            break;
          case 'Pairing':
            console.log('  ⚠️  Pairing issues require manual intervention');
            break;
        }
      }
    }
  }

  /**
   * 修复 Gateway
   */
  private async fixGateway(): Promise<void> {
    try {
      const { exec } = await import('child_process');
      const { promisify } = await import('util');
      const execAsync = promisify(exec);

      console.log('  - Restarting Gateway...');
      await execAsync('openclaw gateway restart');
      console.log('  ✅ Gateway restarted');
    } catch (error) {
      console.log('  ❌ Failed to restart Gateway');
    }
  }
}
```

### 3. src/index.ts

```typescript
/**
 * OpenClaw 故障排查示例
 */

import 'dotenv/config';
import { Diagnostics } from './diagnostics.js';
import { HealthCheckService } from './health-check.js';

async function main() {
  console.log('🎯 OpenClaw Troubleshooting Example\n');

  const args = process.argv.slice(2);
  const mode = args[0] || 'once';

  if (mode === 'watch') {
    // 持续监控模式
    const healthCheck = new HealthCheckService();
    healthCheck.startPeriodicCheck(30000); // 每30秒检查一次

    // 监听退出信号
    process.on('SIGINT', () => {
      healthCheck.stopPeriodicCheck();
      process.exit(0);
    });
  } else {
    // 单次诊断模式
    const diagnostics = new Diagnostics();
    const results = await diagnostics.runFullDiagnostics();
    diagnostics.displayResults(results);
  }
}

main().catch(console.error);
```

## 常见问题修复

### 问题 1: Gateway 无法启动

**症状**：
```
❌ Gateway: ERROR
   Gateway is not running
```

**解决方案**：
```bash
# 检查端口占用
lsof -ti:3000 | xargs kill -9

# 重启 Gateway
openclaw gateway restart

# 查看日志
openclaw gateway logs
```

### 问题 2: 配对失效

**症状**：
```
❌ Pairing: ERROR
   All pairings are inactive
```

**解决方案**：
```bash
# 查看配对列表
openclaw pairing list

# 重新配对
openclaw pairing start

# 删除失效配对
openclaw pairing delete <pairing-id>
```

### 问题 3: 消息发送失败

**症状**：
```
❌ Message Send: ERROR
   Failed to send test message
```

**解决方案**：
```bash
# 1. 检查 Gateway
openclaw gateway status

# 2. 检查配对
openclaw pairing list

# 3. 测试发送
openclaw message send "Test" --channel whatsapp
```
