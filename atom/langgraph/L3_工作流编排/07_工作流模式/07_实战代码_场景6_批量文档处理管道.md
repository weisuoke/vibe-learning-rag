> LangGraph L3_工作流编排 | 实战场景 6/6

---

## 场景描述

构建一个**批量文档处理管道**：接收一批文档，动态并行处理每个文档（分块→向量化→存储），最后汇总处理报告。

模式组合：**Map-Reduce + 顺序链式**

```
文档列表 → Splitter → ┌→ [分块→向量化→存储](doc1) ─┐
                       ├→ [分块→向量化→存储](doc2) ─┤→ 汇总报告 → END
                       └→ [分块→向量化→存储](doc3) ─┘
```

---

## 完整代码

```python
"""
批量文档处理管道
模式组合：Map-Reduce + 顺序链式

每个文档经过：分块 → 向量化 → 存储（顺序链式）
多个文档并行处理（Map-Reduce）
最后汇总处理报告
"""
import operator
import hashlib
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Send


# ===== 1. 定义状态 =====
class PipelineState(TypedDict):
    """全局状态"""
    documents: list[dict]                              # 输入文档列表
    processing_reports: Annotated[list[dict], operator.add]  # 处理报告
    summary: dict                                      # 汇总报告


class DocProcessState(TypedDict):
    """单个文档的处理状态"""
    doc_id: str
    doc_name: str
    doc_content: str
    chunks: list[str]
    vectors: list[list[float]]
    stored_count: int
    status: str
    error: str


# ===== 2. 文档验证与分发 =====
def validate_and_dispatch(state: PipelineState) -> dict:
    """验证文档列表"""
    docs = state["documents"]
    valid_docs = [d for d in docs if d.get("content")]
    print(f"[验证] 收到 {len(docs)} 个文档，{len(valid_docs)} 个有效")
    return {"documents": valid_docs}


def fan_out_documents(state: PipelineState):
    """动态分发：为每个文档创建处理任务"""
    return [
        Send("process_document", {
            "doc_id": doc.get("id", f"doc_{i}"),
            "doc_name": doc.get("name", f"文档{i+1}"),
            "doc_content": doc["content"],
            "chunks": [],
            "vectors": [],
            "stored_count": 0,
            "status": "pending",
            "error": "",
        })
        for i, doc in enumerate(state["documents"])
    ]


# ===== 3. 单文档处理（顺序链式：分块→向量化→存储）=====
def process_document(state: DocProcessState) -> dict:
    """处理单个文档：分块 → 向量化 → 存储"""
    doc_name = state["doc_name"]
    content = state["doc_content"]

    try:
        # 步骤1：分块
        chunks = chunk_text(content, chunk_size=100)
        print(f"  [分块] {doc_name}: {len(chunks)} 个块")

        # 步骤2：向量化
        vectors = vectorize_chunks(chunks)
        print(f"  [向量化] {doc_name}: {len(vectors)} 个向量")

        # 步骤3：存储
        stored = store_vectors(state["doc_id"], chunks, vectors)
        print(f"  [存储] {doc_name}: 存储了 {stored} 条记录")

        return {
            "processing_reports": [{
                "doc_id": state["doc_id"],
                "doc_name": doc_name,
                "chunks_count": len(chunks),
                "vectors_count": len(vectors),
                "stored_count": stored,
                "status": "success",
                "error": "",
            }]
        }

    except Exception as e:
        print(f"  [错误] {doc_name}: {e}")
        return {
            "processing_reports": [{
                "doc_id": state["doc_id"],
                "doc_name": doc_name,
                "chunks_count": 0,
                "vectors_count": 0,
                "stored_count": 0,
                "status": "failed",
                "error": str(e),
            }]
        }


def chunk_text(text: str, chunk_size: int = 100) -> list[str]:
    """简单分块：按字符数切分"""
    chunks = []
    for i in range(0, len(text), chunk_size):
        chunk = text[i:i + chunk_size].strip()
        if chunk:
            chunks.append(chunk)
    return chunks


def vectorize_chunks(chunks: list[str]) -> list[list[float]]:
    """模拟向量化：生成伪向量"""
    vectors = []
    for chunk in chunks:
        # 用 hash 模拟 embedding（实际项目用 OpenAI/sentence-transformers）
        hash_bytes = hashlib.md5(chunk.encode()).digest()
        vector = [b / 255.0 for b in hash_bytes[:8]]  # 8维伪向量
        vectors.append(vector)
    return vectors


def store_vectors(doc_id: str, chunks: list[str], vectors: list[list[float]]) -> int:
    """模拟向量存储"""
    # 实际项目中写入 ChromaDB/Milvus/FAISS
    return len(vectors)


# ===== 4. 汇总报告 =====
def generate_summary(state: PipelineState) -> dict:
    """汇总所有文档的处理报告"""
    reports = state["processing_reports"]

    success_count = sum(1 for r in reports if r["status"] == "success")
    failed_count = sum(1 for r in reports if r["status"] == "failed")
    total_chunks = sum(r["chunks_count"] for r in reports)
    total_stored = sum(r["stored_count"] for r in reports)

    summary = {
        "total_documents": len(reports),
        "success": success_count,
        "failed": failed_count,
        "total_chunks": total_chunks,
        "total_stored": total_stored,
        "details": reports,
    }

    print(f"\n[汇总] 处理完成:")
    print(f"  总文档数: {len(reports)}")
    print(f"  成功: {success_count}")
    print(f"  失败: {failed_count}")
    print(f"  总块数: {total_chunks}")
    print(f"  总存储: {total_stored}")

    return {"summary": summary}


# ===== 5. 构建图 =====
builder = StateGraph(PipelineState)

builder.add_node("validate", validate_and_dispatch)
builder.add_node("process_document", process_document)
builder.add_node("summarize", generate_summary)

builder.add_edge(START, "validate")

# Map-Reduce：动态分发文档处理任务
builder.add_conditional_edges(
    "validate",
    fan_out_documents,
    ["process_document"],
)

builder.add_edge("process_document", "summarize")
builder.add_edge("summarize", END)

doc_pipeline = builder.compile()


# ===== 6. 测试 =====
if __name__ == "__main__":
    # 准备测试文档
    test_documents = [
        {
            "id": "doc_001",
            "name": "LangGraph入门指南.pdf",
            "content": "LangGraph是一个用于构建状态化工作流的框架。"
                       "它基于图结构，支持节点、边和条件路由。"
                       "LangGraph的核心优势是支持持久化和人机循环。"
        },
        {
            "id": "doc_002",
            "name": "RAG最佳实践.md",
            "content": "RAG（检索增强生成）是提升LLM准确性的关键技术。"
                       "核心流程包括：文档加载、分块、向量化、检索和生成。"
        },
        {
            "id": "doc_003",
            "name": "向量数据库对比.docx",
            "content": "常见的向量数据库包括ChromaDB、Milvus和FAISS。"
                       "ChromaDB适合快速原型，Milvus适合生产环境。"
                       "FAISS是Facebook开源的高性能向量检索库。"
        },
    ]

    print("=" * 60)
    print("批量文档处理管道")
    print("=" * 60)

    result = doc_pipeline.invoke({
        "documents": test_documents,
    })

    print(f"\n{'='*60}")
    print("处理报告详情:")
    print("=" * 60)
    for report in result["summary"]["details"]:
        status_icon = "✅" if report["status"] == "success" else "❌"
        print(f"  {status_icon} {report['doc_name']}: "
              f"{report['chunks_count']}块, "
              f"{report['stored_count']}条存储")
```

## 学习要点

1. **Map-Reduce + 顺序链式的组合**：每个 Worker 内部是顺序处理（分块→向量化→存储）
2. **错误隔离**：单个文档处理失败不影响其他文档
3. **汇总报告设计**：统计成功/失败数量，提供详细的每文档报告
4. **生产级考虑**：实际项目中替换模拟函数为真实的 embedding 和向量库调用即可
