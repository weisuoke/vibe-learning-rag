## 场景描述

**构建一个自适应规划系统，支持动态任务分解、优先级调整和智能重规划。当某个步骤失败或出现新信息时，系统能自动调整后续计划。**

### 与场景1的区别

| 维度 | 场景1（基础 P&E） | 场景2（自适应规划） |
|------|-------------------|-------------------|
| 规划 | 一次性生成 | 动态分解 + 优先级 |
| 重规划 | 只在全部完成后 | 每步都可触发 |
| 失败处理 | 无 | 自动重试 + 替代方案 |
| 并行执行 | 不支持 | 识别可并行任务 |

## 完整实现

### 第一步：增强状态和模型定义

```python
from typing import List, TypedDict, Annotated, Literal, Optional
from langgraph.graph import StateGraph, END
from langchain_openai import ChatOpenAI
from langchain_core.tools import tool
from pydantic import BaseModel, Field
from enum import Enum
import operator
from dotenv import load_dotenv

load_dotenv()

# === 子任务定义 ===
class SubTask(BaseModel):
    """结构化子任务"""
    id: int = Field(description="任务编号")
    title: str = Field(description="任务标题")
    tool_name: str = Field(description="使用的工具: search/analyze/write")
    tool_input: str = Field(description="工具输入参数")
    dependencies: List[int] = Field(
        default_factory=list,
        description="依赖的前置任务编号"
    )
    priority: int = Field(default=1, description="优先级 1-5")
    status: str = Field(default="pending")
    result: Optional[str] = Field(default=None)
    retry_count: int = Field(default=0)

class AdaptivePlan(BaseModel):
    """自适应计划"""
    tasks: List[SubTask] = Field(description="子任务列表")
    parallel_groups: List[List[int]] = Field(
        default_factory=list,
        description="可并行执行的任务组，如 [[1,2], [3]]"
    )

class ReplanAction(BaseModel):
    """重规划动作"""
    action: Literal["continue", "retry", "replace", "insert", "complete"]
    reason: str
    new_tasks: List[SubTask] = Field(default_factory=list)
    final_answer: str = Field(default="")

# === 状态 ===
class AdaptiveState(TypedDict):
    input: str
    tasks: List[dict]              # 序列化的 SubTask 列表
    current_task_id: int
    completed_results: Annotated[List[tuple], operator.add]
    response: str
    replan_count: int

# === 模型和工具 ===
planner_llm = ChatOpenAI(model="gpt-4o", temperature=0)
executor_llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

@tool
def search(query: str) -> str:
    """搜索网络信息"""
    return f"搜索 '{query}' 的结果：找到相关信息。"

@tool
def analyze(data: str) -> str:
    """分析数据"""
    return f"分析结果：{data[:100]} 的深度分析。"

@tool
def write(content: str) -> str:
    """编写内容"""
    return f"已编写：{content[:100]}..."

tool_map = {"search": search, "analyze": analyze, "write": write}
```

### 第二步：自适应规划节点

```python
async def adaptive_plan_node(state: AdaptiveState):
    """自适应规划：生成带优先级和依赖的任务列表"""
    planner = planner_llm.with_structured_output(AdaptivePlan)

    plan = await planner.ainvoke(f"""
请为以下任务制定自适应执行计划。

要求：
1. 分解为 3-7 个子任务
2. 每个子任务指定：工具(search/analyze/write)、优先级(1最高)
3. 标注任务间的依赖关系
4. 识别可以并行执行的任务组

任务：{state['input']}
""")

    # 转为字典列表存入状态
    tasks = [t.model_dump() for t in plan.tasks]

    print(f"📋 自适应计划：{len(tasks)} 个子任务")
    for t in tasks:
        deps = f" (依赖: {t['dependencies']})" if t['dependencies'] else ""
        print(f"  [{t['priority']}] #{t['id']} {t['title']}{deps}")
    if plan.parallel_groups:
        print(f"  ⚡ 可并行: {plan.parallel_groups}")

    # 找第一个可执行的任务
    first_id = _find_next_task(tasks)

    return {
        "tasks": tasks,
        "current_task_id": first_id,
        "replan_count": 0,
    }

def _find_next_task(tasks: List[dict]) -> int:
    """找到下一个可执行的任务（依赖已满足且未完成）"""
    completed_ids = {t["id"] for t in tasks if t["status"] == "completed"}

    for t in sorted(tasks, key=lambda x: x["priority"]):
        if t["status"] == "pending":
            deps_met = all(d in completed_ids for d in t["dependencies"])
            if deps_met:
                return t["id"]
    return -1  # 没有可执行的任务
```

### 第三步：带重试的执行节点

```python
async def adaptive_execute_node(state: AdaptiveState):
    """执行当前任务，支持失败重试"""
    tasks = state["tasks"]
    task_id = state["current_task_id"]

    # 找到当前任务
    current = next(t for t in tasks if t["id"] == task_id)
    current["status"] = "running"

    print(f"\n⚡ 执行 #{task_id}: {current['title']}")
    print(f"  工具: {current['tool_name']}({current['tool_input'][:50]})")

    try:
        # 注入已有结果作为上下文
        context = current["tool_input"]
        if state["completed_results"]:
            context += "\n上下文: " + "; ".join(
                f"{k}={v[:100]}" for k, v in state["completed_results"][-3:]
            )

        # 调用工具
        if current["tool_name"] in tool_map:
            result = tool_map[current["tool_name"]].invoke(context)
        else:
            result = await executor_llm.ainvoke(
                f"执行: {current['title']}\n输入: {context}"
            )
            result = result.content

        # 标记完成
        current["status"] = "completed"
        current["result"] = result
        print(f"  ✅ 完成: {result[:100]}...")

        return {
            "tasks": tasks,
            "completed_results": [(current["title"], result)],
        }

    except Exception as e:
        # 失败处理
        current["retry_count"] += 1
        if current["retry_count"] <= 2:
            current["status"] = "pending"  # 重新排队
            print(f"  ⚠️ 失败，重试 {current['retry_count']}/2")
        else:
            current["status"] = "failed"
            print(f"  ❌ 失败: {e}")

        return {"tasks": tasks}
```

### 第四步：智能重规划节点

```python
async def adaptive_replan_node(state: AdaptiveState):
    """智能重规划：根据执行情况动态调整"""
    tasks = state["tasks"]
    replan_count = state.get("replan_count", 0)

    # 检查是否有失败的任务
    failed = [t for t in tasks if t["status"] == "failed"]
    completed = [t for t in tasks if t["status"] == "completed"]
    pending = [t for t in tasks if t["status"] == "pending"]

    print(f"\n🔄 评估: 完成={len(completed)}, "
          f"待执行={len(pending)}, 失败={len(failed)}")

    # 防止无限重规划
    if replan_count >= 3:
        print("  ⚠️ 达到最大重规划次数，强制完成")
        answer = await planner_llm.ainvoke(
            f"目标：{state['input']}\n"
            f"已有结果：{state['completed_results']}\n"
            f"请基于已有信息生成最佳答案。"
        )
        return {"response": answer.content, "replan_count": replan_count + 1}

    # 如果没有待执行和失败的任务，任务完成
    if not pending and not failed:
        replanner = planner_llm.with_structured_output(ReplanAction)
        decision = await replanner.ainvoke(
            f"目标：{state['input']}\n"
            f"执行结果：{state['completed_results']}\n"
            f"所有步骤已完成，请判断目标是否达成。"
        )

        if decision.action == "complete":
            print(f"  ✅ 任务完成: {decision.reason}")
            return {"response": decision.final_answer}
        elif decision.action == "insert":
            # 需要补充步骤
            new_tasks = [t.model_dump() for t in decision.new_tasks]
            print(f"  📝 插入 {len(new_tasks)} 个补充任务")
            tasks.extend(new_tasks)
            next_id = _find_next_task(tasks)
            return {
                "tasks": tasks,
                "current_task_id": next_id,
                "replan_count": replan_count + 1,
            }

    # 有失败任务，尝试替代方案
    if failed:
        replanner = planner_llm.with_structured_output(ReplanAction)
        decision = await replanner.ainvoke(
            f"任务 #{failed[0]['id']} '{failed[0]['title']}' 执行失败。\n"
            f"请提供替代方案。"
        )
        if decision.new_tasks:
            new_tasks = [t.model_dump() for t in decision.new_tasks]
            tasks.extend(new_tasks)

    # 找下一个可执行的任务
    next_id = _find_next_task(tasks)
    if next_id == -1:
        # 没有可执行的任务了
        answer = await planner_llm.ainvoke(
            f"目标：{state['input']}\n"
            f"结果：{state['completed_results']}\n"
            f"请生成最终答案。"
        )
        return {"response": answer.content}

    return {
        "tasks": tasks,
        "current_task_id": next_id,
        "replan_count": replan_count + 1,
    }
```

### 第五步：构建图

```python
def adaptive_route(state: AdaptiveState):
    if state.get("response"):
        return "end"

    tasks = state["tasks"]
    next_id = _find_next_task(tasks)

    if next_id != -1:
        return "execute"
    return "replan"

# 构建图
graph = StateGraph(AdaptiveState)
graph.add_node("planner", adaptive_plan_node)
graph.add_node("executor", adaptive_execute_node)
graph.add_node("replanner", adaptive_replan_node)

graph.set_entry_point("planner")
graph.add_edge("planner", "executor")
graph.add_conditional_edges("executor", adaptive_route, {
    "execute": "executor",
    "replan": "replanner",
    "end": END,
})
graph.add_conditional_edges("replanner", adaptive_route, {
    "execute": "executor",
    "replan": "replanner",
    "end": END,
})

app = graph.compile()
```

### 第六步：运行

```python
import asyncio

async def main():
    result = await app.ainvoke({
        "input": "帮我调研 2026 年最流行的 3 个向量数据库，"
                 "分析它们在 RAG 项目中的表现，给出选型建议。"
    })
    print("\n" + "="*60)
    print("📊 最终结果：")
    print(result["response"])

asyncio.run(main())
```

## 与场景1的关键区别

```
场景1（基础）：
  Plan → Execute1 → Execute2 → ... → Replan → Done
  （线性执行，只在末尾重规划）

场景2（自适应）：
  Plan → Execute1 → 检查 → Execute2 → 失败！
    → Replan（替换失败步骤） → Execute2' → 检查
    → 发现遗漏 → Insert（补充步骤） → Execute → Done
  （每步都可触发重规划，支持失败重试和计划插入）
```

---

## 关键设计亮点

| 特性 | 实现方式 |
|------|---------|
| 依赖感知 | `_find_next_task()` 检查依赖是否满足 |
| 自动重试 | `retry_count` 限制重试次数 |
| 智能重规划 | 根据失败原因生成替代方案 |
| 防无限循环 | `replan_count >= 3` 强制终止 |
| 任务插入 | 支持在执行中动态添加新任务 |
