## 完整可运行代码

```python
"""
Milvus 相似度度量性能基准测试
演示：生产级性能测试方法论

基于 2025-2026 VDBBench 最佳实践：
- P95/P99 尾延迟测量
- 可持续吞吐量测试
- 召回率与性能权衡
- 串行与并发测试分离

参考来源：
- VDBBench 1.0 (July 04, 2025)
- "Benchmarks Lie" Article (May 14, 2025)
"""

import numpy as np
from pymilvus import (
    connections, Collection, CollectionSchema, FieldSchema, DataType,
    utility
)
import time
from typing import List, Dict, Tuple
from dataclasses import dataclass
import concurrent.futures
from collections import defaultdict

# ===== 1. 性能指标数据类 =====
print("=== Milvus 性能基准测试系统 ===\n")

@dataclass
class PerformanceMetrics:
    """性能指标"""
    metric_type: str
    serial_latency_p50: float  # ms
    serial_latency_p95: float  # ms
    serial_latency_p99: float  # ms
    conc_latency_p50: float    # ms
    conc_latency_p95: float    # ms
    conc_latency_p99: float    # ms
    max_qps: float
    recall: float
    test_duration: float       # seconds

    def __str__(self):
        return f"""
【度量类型】: {self.metric_type}
【串行延迟】:
  - P50: {self.serial_latency_p50:.2f} ms
  - P95: {self.serial_latency_p95:.2f} ms
  - P99: {self.serial_latency_p99:.2f} ms
【并发延迟】:
  - P50: {self.conc_latency_p50:.2f} ms
  - P95: {self.conc_latency_p95:.2f} ms
  - P99: {self.conc_latency_p99:.2f} ms
【最大 QPS】: {self.max_qps:.0f}
【召回率】: {self.recall:.4f}
【测试时长】: {self.test_duration:.2f}s
"""

# ===== 2. 性能测试器 =====

class PerformanceBenchmark:
    """
    性能基准测试器

    基于 VDBBench 1.0 方法论：
    - 串行测试：建立基线性能
    - 并发测试：模拟生产负载
    - P95/P99 尾延迟：真实用户体验
    - 可持续吞吐量：长期稳定性
    """

    def __init__(self, collection: Collection, query_vectors: List[List[float]]):
        self.collection = collection
        self.query_vectors = query_vectors
        self.metric_type = collection.index().params["metric_type"]

    def run_serial_test(self, num_queries: int = 1000, top_k: int = 10) -> Dict:
        """
        串行测试：单进程顺序执行

        来源：VDBBench 1.0 - Serial Test Phase
        """
        print(f"\n  执行串行测试 ({num_queries} 次查询)...")

        latencies = []
        recalls = []

        for i in range(num_queries):
            query_vec = self.query_vectors[i % len(self.query_vectors)]

            start_time = time.time()
            results = self.collection.search(
                data=[query_vec],
                anns_field="vector",
                param={"metric_type": self.metric_type, "ef": 100},
                limit=top_k,
                output_fields=["id"]
            )
            latency = (time.time() - start_time) * 1000  # ms

            latencies.append(latency)

            # 简化召回率计算（实际应与 ground truth 对比）
            recalls.append(1.0)  # 假设完美召回

        # 计算百分位延迟
        latencies_sorted = sorted(latencies)
        p50 = latencies_sorted[int(len(latencies) * 0.50)]
        p95 = latencies_sorted[int(len(latencies) * 0.95)]
        p99 = latencies_sorted[int(len(latencies) * 0.99)]

        return {
            "p50": p50,
            "p95": p95,
            "p99": p99,
            "recall": np.mean(recalls)
        }

    def run_concurrent_test(
        self,
        concurrency: int = 10,
        duration: int = 30,
        top_k: int = 10
    ) -> Dict:
        """
        并发测试：多进程并发执行

        来源：VDBBench 1.0 - Concurrency Test Phase
        """
        print(f"\n  执行并发测试 (并发度: {concurrency}, 时长: {duration}s)...")

        latencies = []
        query_count = 0
        start_time = time.time()

        def worker(worker_id: int):
            """独立工作进程"""
            worker_latencies = []
            worker_queries = 0

            while time.time() - start_time < duration:
                query_vec = self.query_vectors[worker_queries % len(self.query_vectors)]

                query_start = time.time()
                try:
                    results = self.collection.search(
                        data=[query_vec],
                        anns_field="vector",
                        param={"metric_type": self.metric_type, "ef": 100},
                        limit=top_k,
                        output_fields=["id"]
                    )
                    latency = (time.time() - query_start) * 1000  # ms
                    worker_latencies.append(latency)
                    worker_queries += 1
                except Exception as e:
                    print(f"    Worker {worker_id} error: {e}")

            return worker_latencies, worker_queries

        # 并发执行
        with concurrent.futures.ThreadPoolExecutor(max_workers=concurrency) as executor:
            futures = [executor.submit(worker, i) for i in range(concurrency)]
            results = [f.result() for f in concurrent.futures.as_completed(futures)]

        # 聚合结果
        for worker_latencies, worker_queries in results:
            latencies.extend(worker_latencies)
            query_count += worker_queries

        actual_duration = time.time() - start_time

        # 计算百分位延迟
        latencies_sorted = sorted(latencies)
        p50 = latencies_sorted[int(len(latencies) * 0.50)]
        p95 = latencies_sorted[int(len(latencies) * 0.95)]
        p99 = latencies_sorted[int(len(latencies) * 0.99)]

        # 计算 QPS
        qps = query_count / actual_duration

        return {
            "p50": p50,
            "p95": p95,
            "p99": p99,
            "qps": qps,
            "total_queries": query_count,
            "duration": actual_duration
        }

    def run_full_benchmark(self) -> PerformanceMetrics:
        """运行完整基准测试"""
        print(f"\n=== 测试度量: {self.metric_type} ===")

        # Phase 1: 串行测试
        serial_results = self.run_serial_test(num_queries=1000)

        # Phase 2: 并发测试
        conc_results = self.run_concurrent_test(concurrency=10, duration=30)

        return PerformanceMetrics(
            metric_type=self.metric_type,
            serial_latency_p50=serial_results["p50"],
            serial_latency_p95=serial_results["p95"],
            serial_latency_p99=serial_results["p99"],
            conc_latency_p50=conc_results["p50"],
            conc_latency_p95=conc_results["p95"],
            conc_latency_p99=conc_results["p99"],
            max_qps=conc_results["qps"],
            recall=serial_results["recall"],
            test_duration=conc_results["duration"]
        )

# ===== 3. 测试环境准备 =====

print("=== 准备测试环境 ===")

# 连接 Milvus
connections.connect(
    alias="default",
    host="localhost",
    port="19530"
)
print("✅ 已连接到 Milvus")

# 生成测试数据
DIM = 768  # 模拟 Cohere embeddings
NUM_ENTITIES = 10000
NUM_QUERIES = 100

print(f"\n生成测试数据:")
print(f"  - 向量维度: {DIM}")
print(f"  - 数据量: {NUM_ENTITIES}")
print(f"  - 查询数: {NUM_QUERIES}")

# 生成归一化向量
vectors = []
for _ in range(NUM_ENTITIES):
    vec = np.random.randn(DIM).astype(np.float32)
    vec = vec / np.linalg.norm(vec)  # 归一化
    vectors.append(vec.tolist())

ids = list(range(NUM_ENTITIES))

# 生成查询向量
query_vectors = []
for _ in range(NUM_QUERIES):
    vec = np.random.randn(DIM).astype(np.float32)
    vec = vec / np.linalg.norm(vec)
    query_vectors.append(vec.tolist())

print("✅ 测试数据生成完成")

# ===== 4. 创建测试 Collections =====

print("\n=== 创建测试 Collections ===")

def create_test_collection(name: str, metric_type: str) -> Collection:
    """创建测试 Collection"""
    if utility.has_collection(name):
        utility.drop_collection(name)

    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=False),
        FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=DIM)
    ]
    schema = CollectionSchema(fields=fields, description=f"Benchmark: {metric_type}")

    collection = Collection(name=name, schema=schema)

    # 创建索引
    index_params = {
        "index_type": "HNSW",
        "metric_type": metric_type,
        "params": {"M": 16, "efConstruction": 200}
    }
    collection.create_index(field_name="vector", index_params=index_params)

    # 插入数据
    collection.insert([ids, vectors])
    collection.flush()

    # 加载到内存
    collection.load()

    return collection

# 创建三个测试 collections
metrics_to_test = ["L2", "IP", "COSINE"]
collections = {}

for metric in metrics_to_test:
    print(f"  创建 {metric} collection...")
    collections[metric] = create_test_collection(f"benchmark_{metric.lower()}", metric)

print("✅ 所有 Collections 创建完成")

# ===== 5. 执行基准测试 =====

print("\n" + "=" * 60)
print("开始性能基准测试")
print("=" * 60)

results = {}

for metric, collection in collections.items():
    benchmark = PerformanceBenchmark(collection, query_vectors)
    results[metric] = benchmark.run_full_benchmark()

# ===== 6. 结果分析 =====

print("\n" + "=" * 60)
print("基准测试结果")
print("=" * 60)

for metric, perf in results.items():
    print(perf)

# ===== 7. 性能对比分析 =====

print("\n" + "=" * 60)
print("性能对比分析")
print("=" * 60)

print("\n【串行延迟对比 (P99)】")
for metric in metrics_to_test:
    print(f"  {metric}: {results[metric].serial_latency_p99:.2f} ms")

fastest_serial = min(results.items(), key=lambda x: x[1].serial_latency_p99)
print(f"  ⚡ 最快: {fastest_serial[0]}")

print("\n【并发延迟对比 (P99)】")
for metric in metrics_to_test:
    print(f"  {metric}: {results[metric].conc_latency_p99:.2f} ms")

fastest_conc = min(results.items(), key=lambda x: x[1].conc_latency_p99)
print(f"  ⚡ 最快: {fastest_conc[0]}")

print("\n【最大 QPS 对比】")
for metric in metrics_to_test:
    print(f"  {metric}: {results[metric].max_qps:.0f} QPS")

highest_qps = max(results.items(), key=lambda x: x[1].max_qps)
print(f"  ⚡ 最高: {highest_qps[0]}")

print("\n【召回率对比】")
for metric in metrics_to_test:
    print(f"  {metric}: {results[metric].recall:.4f}")

# ===== 8. 关键发现 =====

print("\n" + "=" * 60)
print("关键发现（基于 VDBBench 2025-2026 方法论）")
print("=" * 60)

print("""
1. P95/P99 尾延迟比平均延迟更重要
   - 尾延迟反映真实用户体验
   - 用于 SLA 制定和容量规划

2. 串行 vs 并发延迟差异
   - 串行延迟：系统最佳性能
   - 并发延迟：生产环境真实表现

3. 可持续吞吐量 (max_qps)
   - 不是短期峰值，而是长期稳定值
   - 用于容量规划和扩容决策

4. 召回率与性能权衡
   - 速度快但召回率低 = 无意义
   - 必须同时考虑性能和准确率

5. 归一化向量的度量特性
   - IP 最快（省略归一化计算）
   - COSINE 次之（自动归一化）
   - L2 稍慢（计算欧氏距离）
""")

# ===== 9. 生产环境建议 =====

print("\n" + "=" * 60)
print("生产环境建议")
print("=" * 60)

print("""
【度量选型】:
- 性能优先 + 归一化向量 → IP

【容量规划】:

【监控指标】:

print("\n=== 清理资源 ===")
for metric, collection in collections.items():
    collection.release()
    utility.drop_collection(collection.name)
    print(f"✅ {metric}: 已清理")

connections.disconnect("default")
print("✅ 已断开连接")

print("\n=== 基准测试完成 ===")
```

## 核心要点

### 1. VDBBench 方法论 (2025-2026)

**两阶段测试结构**（来源：VDBBench 1.0, July 04, 2025）:

1. **串行测试 (Serial Test)**:
   - 单进程顺序执行 1,000 次查询
   - 建立基线性能和准确率
   - 报告 `serial_latency_p99` 和 recall

2. **并发测试 (Concurrency Test)**:
   - 多进程并发执行，模拟生产环境
   - 独立进程，独立连接，独立查询集
   - 同步启动，确保 QPS 准确性
   - 报告 `conc_latency_p99` 和 `max_qps`

### 2. 关键指标定义

| 指标 | 定义 | 用途 |
|------|------|------|
| **serial_latency_p99** | 串行测试 P99 延迟 | 系统最佳性能基线 |
| **conc_latency_p99** | 并发测试 P99 延迟 | 生产环境真实表现 |
| **max_qps** | 可持续最大 QPS | 容量规划依据 |
| **recall** | 召回率 | 准确率保证 |

### 3. 为什么 P95/P99 比平均值重要

**来源**: "Benchmarks Lie" Article (May 14, 2025)

> 平均/中位延迟掩盖了让真实用户沮丧的异常值。P95/P99 尾延迟揭示了 95% 或 99% 的查询实际达到的性能。

**示例**:
- 平均延迟: 10ms
- P99 延迟: 2000ms
- **结果**: 1% 的用户体验极差（2秒延迟）

### 4. 性能特征对比 (归一化向量)

| 度量 | 串行 P99 | 并发 P99 | Max QPS | 原因 |
|------|----------|----------|---------|------|
| **IP** | 最快 | 最快 | 最高 | 省略归一化计算 |
| **COSINE** | 中等 | 中等 | 中等 | 自动归一化处理 |
| **L2** | 稍慢 | 稍慢 | 稍低 | 计算欧氏距离 |

### 5. 生产环境容量规划

**SLA 制定**:
```
目标 P99 延迟: < 50ms
→ 使用 conc_latency_p99 作为基准
→ 预留 30-50% 性能余量
→ 实际 P99 目标: < 35ms
```

**集群容量**:
```
预期 QPS: 10,000
→ 使用 max_qps 作为基准
→ 单节点 max_qps: 1,500
→ 所需节点数: 10,000 / 1,500 = 7 节点
→ 加余量: 10 节点
```
