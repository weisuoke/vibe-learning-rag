# 实战代码 - 场景4：嵌套子图与流式监控

## 场景说明

本示例展示如何构建 **三层嵌套子图**（parent -> child -> grandchild）并使用 `subgraphs=True` 流式监控所有层级的执行过程。

**实际应用场景：多层数据处理管道**
- 父图：接收原始数据 -> 调用验证子图 -> 调用转换子图 -> 输出最终结果
- 子图 A（数据验证）：包含内嵌的「格式检查」孙子图
- 子图 B（数据转换）：执行数据标准化和增强
- 使用 `stream(subgraphs=True)` 实时监控所有层级的执行状态

**核心技术点：**
- 三层图嵌套：parent -> child -> grandchild
- `stream_mode="updates"` + `subgraphs=True` 流式监控
- namespace tuple 标识执行层级
- 每层子图有独立的状态空间

## 完整可运行代码

```python
"""
LangGraph 嵌套子图与流式监控实战
场景：多层数据处理管道

演示：
1. 三层嵌套：parent -> child (validate) -> grandchild (format_check)
2. stream(subgraphs=True) 流式监控所有层级
3. namespace tuple 标识执行层级
4. 每层子图有独立的状态空间
"""

from typing import TypedDict
from langgraph.graph import StateGraph, START, END


# ============================================================
# 第一部分：定义各层状态结构
# ============================================================

# --- 父图状态 ---
class ParentState(TypedDict):
    """父图状态：管理整个数据处理流程"""
    raw_data: dict              # 输入的原始数据
    validation_result: str      # 验证子图的输出
    transformed_data: dict      # 转换子图的输出
    final_report: str           # 最终报告


# --- 子图 A 状态（验证子图） ---
class ValidatorState(TypedDict):
    """验证子图内部状态"""
    raw_data: dict              # 与父图共享键（自动传递）
    basic_check: str            # 基础检查结果
    format_check_result: str    # 格式检查结果（孙子图产出）
    validation_result: str      # 最终验证结果（与父图共享键）


# --- 孙子图状态（格式检查子图） ---
class FormatCheckState(TypedDict):
    """格式检查孙子图内部状态"""
    raw_data: dict              # 从父级传入的数据
    type_check: str             # 类型检查结果
    range_check: str            # 范围检查结果
    format_check_result: str    # 最终格式检查结果（与子图 A 共享键）


# --- 子图 B 状态（转换子图） ---
class TransformerState(TypedDict):
    """转换子图内部状态"""
    raw_data: dict              # 与父图共享键
    validation_result: str      # 与父图共享键（获取验证结果）
    normalized: dict            # 标准化后的数据
    transformed_data: dict      # 与父图共享键（返回转换结果）


# ============================================================
# 第二部分：孙子图节点函数（格式检查）
# ============================================================

def check_type(state: FormatCheckState) -> dict:
    """
    类型检查节点：验证数据字段的类型

    这是最内层（孙子图）的第一个节点。
    """
    data = state["raw_data"]
    print(f"      [孙子图/类型检查] 检查数据类型...")

    errors = []
    # 检查必需字段
    required_fields = {"name": str, "age": (int, float), "score": (int, float)}
    for field, expected_type in required_fields.items():
        if field not in data:
            errors.append(f"缺少字段: {field}")
        elif not isinstance(data[field], expected_type):
            errors.append(f"字段 {field} 类型错误: 期望 {expected_type}, 实际 {type(data[field])}")

    result = "PASS" if not errors else f"FAIL: {'; '.join(errors)}"
    print(f"      [孙子图/类型检查] 结果: {result}")

    return {"type_check": result}


def check_range(state: FormatCheckState) -> dict:
    """
    范围检查节点：验证数值字段的范围

    孙子图的第二个节点，依赖类型检查通过后执行。
    """
    data = state["raw_data"]
    type_result = state["type_check"]
    print(f"      [孙子图/范围检查] 检查数值范围...")

    # 如果类型检查失败，跳过范围检查
    if type_result.startswith("FAIL"):
        result = "SKIP (类型检查未通过)"
        print(f"      [孙子图/范围检查] 结果: {result}")
        return {"range_check": result, "format_check_result": f"FORMAT_FAIL: {type_result}"}

    errors = []
    if "age" in data:
        if not (0 <= data["age"] <= 150):
            errors.append(f"age={data['age']} 超出范围 [0, 150]")
    if "score" in data:
        if not (0 <= data["score"] <= 100):
            errors.append(f"score={data['score']} 超出范围 [0, 100]")

    result = "PASS" if not errors else f"FAIL: {'; '.join(errors)}"
    format_result = "FORMAT_OK" if not errors else f"FORMAT_FAIL: {'; '.join(errors)}"

    print(f"      [孙子图/范围检查] 结果: {result}")

    return {
        "range_check": result,
        "format_check_result": format_result,  # 共享键，传回子图 A
    }


# ============================================================
# 第三部分：子图 A 节点函数（数据验证）
# ============================================================

def check_basic(state: ValidatorState) -> dict:
    """
    基础检查节点：验证数据非空

    子图 A 的第一步，做最基本的非空校验。
    """
    data = state["raw_data"]
    print(f"    [子图A/基础检查] 检查数据是否有效...")

    if not data:
        result = "FAIL: 数据为空"
    elif not isinstance(data, dict):
        result = "FAIL: 数据不是字典类型"
    else:
        result = f"PASS: 包含 {len(data)} 个字段"

    print(f"    [子图A/基础检查] 结果: {result}")
    return {"basic_check": result}


def mark_validated(state: ValidatorState) -> dict:
    """
    标记验证结果节点：汇总所有检查结果

    子图 A 的最后一步，综合基础检查和格式检查的结果。
    """
    basic = state["basic_check"]
    fmt = state.get("format_check_result", "未执行")

    print(f"    [子图A/汇总] 基础检查: {basic}")
    print(f"    [子图A/汇总] 格式检查: {fmt}")

    if basic.startswith("PASS") and fmt == "FORMAT_OK":
        validation = "VALIDATED"
    else:
        validation = f"REJECTED (basic={basic}, format={fmt})"

    print(f"    [子图A/汇总] 最终验证: {validation}")

    return {"validation_result": validation}  # 共享键，传回父图


# ============================================================
# 第四部分：子图 B 节点函数（数据转换）
# ============================================================

def normalize_data(state: TransformerState) -> dict:
    """
    标准化节点：对数据进行格式标准化

    将字段名统一为小写，字符串值去除首尾空格。
    """
    data = state["raw_data"]
    validation = state.get("validation_result", "UNKNOWN")
    print(f"    [子图B/标准化] 验证状态: {validation}")
    print(f"    [子图B/标准化] 开始标准化数据...")

    # 标准化处理
    normalized = {}
    for key, value in data.items():
        norm_key = key.lower().strip()
        if isinstance(value, str):
            norm_value = value.strip()
        else:
            norm_value = value
        normalized[norm_key] = norm_value

    print(f"    [子图B/标准化] 完成: {normalized}")
    return {"normalized": normalized}


def enrich_data(state: TransformerState) -> dict:
    """
    增强节点：添加衍生字段

    基于标准化后的数据添加计算字段。
    """
    normalized = state["normalized"]
    print(f"    [子图B/增强] 添加衍生字段...")

    # 添加衍生数据
    enriched = dict(normalized)
    if "age" in enriched:
        enriched["age_group"] = (
            "少年" if enriched["age"] < 18
            else "青年" if enriched["age"] < 35
            else "中年" if enriched["age"] < 60
            else "老年"
        )
    if "score" in enriched:
        enriched["grade"] = (
            "A" if enriched["score"] >= 90
            else "B" if enriched["score"] >= 80
            else "C" if enriched["score"] >= 70
            else "D" if enriched["score"] >= 60
            else "F"
        )

    enriched["processed"] = True

    print(f"    [子图B/增强] 完成: {enriched}")
    return {"transformed_data": enriched}  # 共享键，传回父图


# ============================================================
# 第五部分：父图节点函数
# ============================================================

def receive_data(state: ParentState) -> dict:
    """
    接收节点：记录输入数据

    父图的入口节点。
    """
    data = state["raw_data"]
    print(f"\n[父图/接收] 收到原始数据: {data}")
    return {}


def output_result(state: ParentState) -> dict:
    """
    输出节点：生成最终处理报告

    汇总验证和转换结果，生成最终报告。
    """
    validation = state.get("validation_result", "N/A")
    transformed = state.get("transformed_data", {})

    print(f"\n[父图/输出] 验证结果: {validation}")
    print(f"[父图/输出] 转换结果: {transformed}")

    report = (
        f"处理报告:\n"
        f"  验证状态: {validation}\n"
        f"  转换字段数: {len(transformed)}\n"
        f"  最终数据: {transformed}"
    )

    print(f"[父图/输出] {report}")
    return {"final_report": report}


# ============================================================
# 第六部分：构建三层图
# ============================================================

def build_format_check_grandchild():
    """
    构建孙子图：格式检查

    层级：grandchild（最内层）
    功能：类型检查 -> 范围检查
    """
    builder = StateGraph(FormatCheckState)
    builder.add_node("check_type", check_type)
    builder.add_node("check_range", check_range)

    builder.add_edge(START, "check_type")
    builder.add_edge("check_type", "check_range")
    builder.add_edge("check_range", END)

    return builder.compile()


def build_validator_child():
    """
    构建子图 A：数据验证

    层级：child
    功能：基础检查 -> 格式检查（孙子图） -> 汇总
    """
    # 先编译孙子图
    format_checker = build_format_check_grandchild()

    builder = StateGraph(ValidatorState)
    builder.add_node("check_basic", check_basic)
    builder.add_node("format_check", format_checker)      # 孙子图作为节点
    builder.add_node("mark_validated", mark_validated)

    builder.add_edge(START, "check_basic")
    builder.add_edge("check_basic", "format_check")       # 基础检查 -> 孙子图
    builder.add_edge("format_check", "mark_validated")     # 孙子图 -> 汇总
    builder.add_edge("mark_validated", END)

    return builder.compile()


def build_transformer_child():
    """
    构建子图 B：数据转换

    层级：child
    功能：标准化 -> 增强
    """
    builder = StateGraph(TransformerState)
    builder.add_node("normalize", normalize_data)
    builder.add_node("enrich", enrich_data)

    builder.add_edge(START, "normalize")
    builder.add_edge("normalize", "enrich")
    builder.add_edge("enrich", END)

    return builder.compile()


def build_parent_graph():
    """
    构建父图：完整的数据处理管道

    层级：parent（最外层）
    流程：接收 -> 验证子图 -> 转换子图 -> 输出
    """
    # 编译子图
    validator = build_validator_child()
    transformer = build_transformer_child()

    builder = StateGraph(ParentState)

    # 添加节点
    builder.add_node("receive", receive_data)
    builder.add_node("validate", validator)           # 子图 A（内含孙子图）
    builder.add_node("transform", transformer)        # 子图 B
    builder.add_node("output", output_result)

    # 线性流程
    builder.add_edge(START, "receive")
    builder.add_edge("receive", "validate")
    builder.add_edge("validate", "transform")
    builder.add_edge("transform", "output")
    builder.add_edge("output", END)

    return builder.compile()


# ============================================================
# 第七部分：流式监控运行
# ============================================================

def explain_namespace(namespace: tuple) -> str:
    """
    解释 namespace tuple 的含义

    namespace 是一个元组，表示当前执行所在的层级路径：
    - ()                                    -> 父图
    - ("validate:xxx",)                     -> 子图 A
    - ("validate:xxx", "format_check:yyy",) -> 孙子图
    """
    if len(namespace) == 0:
        return "父图 (Parent)"
    elif len(namespace) == 1:
        node_name = namespace[0].split(":")[0]  # 去掉实例 ID
        return f"子图 (Child: {node_name})"
    elif len(namespace) == 2:
        child = namespace[0].split(":")[0]
        grandchild = namespace[1].split(":")[0]
        return f"孙子图 (Grandchild: {child}/{grandchild})"
    else:
        return f"第 {len(namespace)} 层"


if __name__ == "__main__":
    # 构建三层嵌套图
    graph = build_parent_graph()

    # ---- 测试用例 1: 正常数据 ----
    print("=" * 70)
    print("测试用例 1: 正常数据")
    print("=" * 70)

    test_data_1 = {
        "raw_data": {
            "name": "Alice",
            "age": 28,
            "score": 92,
        }
    }

    print("\n>>> 流式监控（subgraphs=True）")
    print("-" * 70)

    for namespace, step in graph.stream(
        test_data_1,
        stream_mode="updates",
        subgraphs=True,
    ):
        level = explain_namespace(namespace)
        depth = len(namespace)
        indent = "  " * depth

        for node_name, update in step.items():
            # 显示层级、节点名和更新的键
            updated_keys = list(update.keys())
            print(f"{indent}[{level}] {node_name} -> {updated_keys}")

            # 显示关键值
            for key in updated_keys:
                value = update[key]
                if isinstance(value, str) and len(value) < 80:
                    print(f"{indent}  {key} = {value}")
                elif isinstance(value, dict):
                    print(f"{indent}  {key} = {value}")

    print()

    # ---- 测试用例 2: 异常数据 ----
    print("=" * 70)
    print("测试用例 2: 异常数据（age 超出范围）")
    print("=" * 70)

    test_data_2 = {
        "raw_data": {
            "name": "Bob",
            "age": 200,      # 超出范围
            "score": 85,
        }
    }

    print("\n>>> 流式监控（subgraphs=True）")
    print("-" * 70)

    for namespace, step in graph.stream(
        test_data_2,
        stream_mode="updates",
        subgraphs=True,
    ):
        level = explain_namespace(namespace)
        depth = len(namespace)
        indent = "  " * depth

        for node_name, update in step.items():
            updated_keys = list(update.keys())
            print(f"{indent}[{level}] {node_name} -> {updated_keys}")

    print()

    # ---- 对比：不开启 subgraphs=True ----
    print("=" * 70)
    print("对比：不开启 subgraphs=True（只能看到父图层级）")
    print("=" * 70)

    print("\n>>> 普通流式（subgraphs=False，默认）")
    print("-" * 70)

    for step in graph.stream(test_data_1, stream_mode="updates"):
        for node_name, update in step.items():
            updated_keys = list(update.keys())
            print(f"[父图] {node_name} -> {updated_keys}")
```

## 关键机制解析

### 1. Namespace Tuple 的含义

```python
# 流式输出格式：(namespace_tuple, {node_name: state_update})

# namespace 示例：
()                                           # 父图节点
("validate:abc123",)                         # 子图 A 的节点
("validate:abc123", "format_check:def456",)  # 孙子图的节点
```

| namespace | 含义 | 层级 |
|-----------|------|------|
| `()` | 父图 | 第 0 层 |
| `("validate:id1",)` | validate 子图 | 第 1 层 |
| `("validate:id1", "format_check:id2",)` | format_check 孙子图 | 第 2 层 |

- 元组的每个元素格式为 `"节点名:实例ID"`
- 实例 ID 是 LangGraph 运行时生成的唯一标识
- 元组长度 = 嵌套深度

### 2. 共享键实现跨层状态传递

```
父图 ParentState          子图 ValidatorState       孙子图 FormatCheckState
├── raw_data ◄──────────► raw_data ◄──────────► raw_data
├── validation_result ◄──► validation_result
├── transformed_data       format_check_result ◄──► format_check_result
└── final_report           basic_check
                                                   type_check
                                                   range_check
```

- 共享键（同名字段）自动在相邻层级间传递
- 私有键只在当前图的作用域内可见
- 子图返回时，只有共享键会更新到父图

### 3. subgraphs=True vs 默认模式

```python
# 默认模式：只看到父图级别的更新
for step in graph.stream(input, stream_mode="updates"):
    # step = {node_name: state_update}
    # 子图被当作黑盒，只看到最终输出

# subgraphs=True：看到所有层级的内部更新
for namespace, step in graph.stream(input, stream_mode="updates", subgraphs=True):
    # namespace = () 或 ("child:id",) 或 ("child:id", "grandchild:id",)
    # step = {node_name: state_update}
    # 可以观察每一层的每一步
```

| 场景 | 父图 | 子图 | 孙子图 | 监控用途 |
|------|------|------|--------|----------|

| Agent 系统 | 调度器 | 子 Agent | Agent 内部工具调用 | 追踪决策链路 |

## 常见陷阱

1. **共享键命名冲突**：父图和子图的同名键会自动传递。如果子图有与父图同名但含义不同的字段，数据会被意外覆盖。解决方案：使用不同的键名，或通过包装函数（状态转换模式）隔离
2. **subgraphs=True 改变返回格式**：默认返回 `dict`，开启后返回 `tuple(namespace, dict)`。切换时注意解包方式
3. **namespace 中的实例 ID 不稳定**：`"validate:abc123"` 中的 `abc123` 每次运行都不同，不要硬编码匹配。用 `split(":")[0]` 提取节点名
4. **孙子图的状态不会直接传回父图**：孙子图 -> 子图 -> 父图，每层只传递共享键。如果需要把孙子图的结果传到父图，必须在子图中设置对应的共享键作为中转
