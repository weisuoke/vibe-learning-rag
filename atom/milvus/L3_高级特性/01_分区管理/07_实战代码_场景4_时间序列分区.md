> **适用场景**：历史数据管理、热冷数据分离、按时间范围查询

---

## 场景描述

在 RAG 系统中，文档数据通常具有时间属性。随着时间推移，历史数据的访问频率会降低，而最新数据的访问频率较高。

本场景演示如何使用手动分区实现时间序列数据管理：
- 按月份创建分区
- 热数据（最近 3 个月）保持加载
- 冷数据（历史数据）按需加载
- 自动归档和清理

---

## 完整代码示例

```python
"""
Milvus 时间序列分区实战
场景：新闻文档知识库 - 按月份分区管理

功能：
1. 按月份创建分区
2. 热冷数据管理
3. 按时间范围查询
4. 自动归档历史数据

依赖：
- pymilvus
- numpy
- datetime
"""

from pymilvus import (
    connections,
    Collection,
    CollectionSchema,
    FieldSchema,
    DataType,
    utility
)
import numpy as np
from typing import List, Dict, Any
from datetime import datetime, timedelta
import time


class TimeSeriesPartitionManager:
    """时间序列分区管理器"""

    def __init__(self, collection_name: str = "news_knowledge_base"):
        self.collection_name = collection_name
        self.collection = None
        self.hot_months = 3  # 热数据保留月数

    def connect_to_milvus(self, host: str = "localhost", port: str = "19530"):
        """连接到 Milvus"""
        print(f"连接到 Milvus: {host}:{port}")
        connections.connect(alias="default", host=host, port=port)
        print("连接成功")

    def create_collection(self):
        """创建 Collection"""
        if utility.has_collection(self.collection_name):
            print(f"Collection '{self.collection_name}' 已存在，删除后重新创建")
            utility.drop_collection(self.collection_name)

        # 定义 Schema
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="title", dtype=DataType.VARCHAR, max_length=512),
            FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
            FieldSchema(name="publish_date", dtype=DataType.INT64),  # Unix 时间戳
            FieldSchema(name="category", dtype=DataType.VARCHAR, max_length=64),
            FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=128)
        ]

        schema = CollectionSchema(
            fields=fields,
            description="新闻知识库 - 按月份分区"
        )

        self.collection = Collection(name=self.collection_name, schema=schema)
        print(f"Collection '{self.collection_name}' 创建成功")

    def get_partition_name(self, date: datetime) -> str:
        """根据日期生成分区名称"""
        return f"news_{date.year}_{date.month:02d}"

    def create_monthly_partitions(self, start_date: datetime, months: int):
        """创建多个月份的分区"""
        print(f"\n创建 {months} 个月份的分区...")
        for i in range(months):
            date = start_date + timedelta(days=30 * i)
            partition_name = self.get_partition_name(date)

            if not self.collection.has_partition(partition_name):
                self.collection.create_partition(partition_name)
                print(f"创建分区: {partition_name}")

    def insert_documents(
        self,
        documents: List[Dict[str, Any]]
    ):
        """
        插入文档到对应的月份分区

        Args:
            documents: 文档列表，每个文档包含 title, content, publish_date, category
        """
        print(f"\n插入 {len(documents)} 个文档...")

        # 按月份分组
        monthly_docs = {}
        for doc in documents:
            publish_date = datetime.fromtimestamp(doc['publish_date'])
            partition_name = self.get_partition_name(publish_date)

            if partition_name not in monthly_docs:
                monthly_docs[partition_name] = []
            monthly_docs[partition_name].append(doc)

        # 为每个月份创建分区并插入数据
        for partition_name, docs in monthly_docs.items():
            # 确保分区存在
            if not self.collection.has_partition(partition_name):
                self.collection.create_partition(partition_name)
                print(f"创建分区: {partition_name}")

            # 准备数据
            embeddings = np.random.rand(len(docs), 128).tolist()
            data = [
                [doc['title'] for doc in docs],
                [doc['content'] for doc in docs],
                [doc['publish_date'] for doc in docs],
                [doc['category'] for doc in docs],
                embeddings
            ]

            # 插入数据
            result = self.collection.insert(data, partition_name=partition_name)
            print(f"向分区 '{partition_name}' 插入 {result.insert_count} 个文档")

    def create_index(self):
        """创建索引"""
        print("\n创建索引...")
        index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "L2",
            "params": {"nlist": 128}
        }
        self.collection.create_index(field_name="embedding", index_params=index_params)
        print("索引创建成功")

    def load_hot_partitions(self):
        """加载热数据分区（最近 N 个月）"""
        print(f"\n加载热数据分区（最近 {self.hot_months} 个月）...")

        # 计算热数据分区名称
        hot_partitions = []
        current_date = datetime.now()
        for i in range(self.hot_months):
            date = current_date - timedelta(days=30 * i)
            partition_name = self.get_partition_name(date)
            if self.collection.has_partition(partition_name):
                hot_partitions.append(partition_name)

        if hot_partitions:
            self.collection.load(partition_names=hot_partitions)
            time.sleep(2)
            print(f"已加载分区: {hot_partitions}")
        else:
            print("没有找到热数据分区")

    def search_by_time_range(
        self,
        query_embedding: List[float],
        start_date: datetime,
        end_date: datetime,
        top_k: int = 5
    ):
        """按时间范围搜索"""
        print(f"\n搜索时间范围: {start_date.date()} 到 {end_date.date()}")

        # 计算涉及的分区
        partitions = []
        current = start_date
        while current <= end_date:
            partition_name = self.get_partition_name(current)
            if self.collection.has_partition(partition_name):
                partitions.append(partition_name)
            current += timedelta(days=30)

        if not partitions:
            print("没有找到相关分区")
            return []

        print(f"搜索分区: {partitions}")

        # 确保分区已加载
        for partition in partitions:
            state = utility.load_state(
                collection_name=self.collection_name,
                partition_names=[partition]
            )
            if state.name != "Loaded":
                print(f"加载分区: {partition}")
                self.collection.load(partition_names=[partition])
                time.sleep(1)

        # 搜索
        search_params = {"metric_type": "L2", "params": {"nprobe": 10}}

        # 构建时间过滤表达式
        start_ts = int(start_date.timestamp())
        end_ts = int(end_date.timestamp())
        expr = f"publish_date >= {start_ts} && publish_date <= {end_ts}"

        results = self.collection.search(
            data=[query_embedding],
            anns_field="embedding",
            param=search_params,
            limit=top_k,
            partition_names=partitions,
            expr=expr,
            output_fields=["title", "content", "publish_date", "category"]
        )

        print(f"找到 {len(results[0])} 个结果")
        for i, hit in enumerate(results[0], 1):
            pub_date = datetime.fromtimestamp(hit.entity.get('publish_date'))
            print(f"{i}. {hit.entity.get('title')} ({pub_date.date()})")

        return results

    def get_partition_stats(self):
        """获取所有分区的统计信息"""
        print("\n分区统计信息:")
        partitions = self.collection.partitions

        stats = []
        for partition in partitions:
            if partition.name == "_default":
                continue

            num_entities = partition.num_entities
            stats.append({
                "partition": partition.name,
                "documents": num_entities
            })

        # 按分区名称排序
        stats.sort(key=lambda x: x['partition'])

        for stat in stats:
            print(f"- {stat['partition']}: {stat['documents']} 个文档")

        return stats

    def archive_old_partitions(self, months_to_keep: int = 6):
        """归档旧分区（释放内存）"""
        print(f"\n归档 {months_to_keep} 个月之前的数据...")

        cutoff_date = datetime.now() - timedelta(days=30 * months_to_keep)
        partitions_to_release = []

        for partition in self.collection.partitions:
            if partition.name == "_default":
                continue

            # 解析分区名称中的日期
            try:
                parts = partition.name.split('_')
                year = int(parts[1])
                month = int(parts[2])
                partition_date = datetime(year, month, 1)

                if partition_date < cutoff_date:
                    partitions_to_release.append(partition.name)
            except:
                continue

        if partitions_to_release:
            self.collection.release(partition_names=partitions_to_release)
            print(f"已释放分区: {partitions_to_release}")
        else:
            print("没有需要归档的分区")


def main():
    """主函数 - 演示时间序列分区"""

    # 1. 初始化
    manager = TimeSeriesPartitionManager()
    manager.connect_to_milvus()
    manager.create_collection()

    # 2. 创建最近 6 个月的分区
    start_date = datetime.now() - timedelta(days=150)
    manager.create_monthly_partitions(start_date, months=6)

    # 3. 插入不同月份的文档
    documents = []

    # 最近 1 个月的文档
    for i in range(5):
        documents.append({
            "title": f"最新新闻 {i+1}",
            "content": f"这是最新的新闻内容 {i+1}...",
            "publish_date": int((datetime.now() - timedelta(days=i)).timestamp()),
            "category": "科技"
        })

    # 2 个月前的文档
    for i in range(3):
        documents.append({
            "title": f"两月前新闻 {i+1}",
            "content": f"这是两个月前的新闻内容 {i+1}...",
            "publish_date": int((datetime.now() - timedelta(days=60+i)).timestamp()),
            "category": "财经"
        })

    # 5 个月前的文档
    for i in range(2):
        documents.append({
            "title": f"历史新闻 {i+1}",
            "content": f"这是历史新闻内容 {i+1}...",
            "publish_date": int((datetime.now() - timedelta(days=150+i)).timestamp()),
            "category": "体育"
        })

    manager.insert_documents(documents)

    # 4. 创建索引
    manager.create_index()

    # 5. 加载热数据分区
    manager.load_hot_partitions()

    # 6. 按时间范围搜索
    query_embedding = np.random.rand(128).tolist()

    # 搜索最近 1 个月
    manager.search_by_time_range(
        query_embedding,
        start_date=datetime.now() - timedelta(days=30),
        end_date=datetime.now(),
        top_k=3
    )

    # 搜索 2-3 个月前
    manager.search_by_time_range(
        query_embedding,
        start_date=datetime.now() - timedelta(days=90),
        end_date=datetime.now() - timedelta(days=60),
        top_k=3
    )

    # 7. 获取分区统计
    manager.get_partition_stats()

    # 8. 归档旧数据
    manager.archive_old_partitions(months_to_keep=3)

    print("\n演示完成")


if __name__ == "__main__":
    main()
```
