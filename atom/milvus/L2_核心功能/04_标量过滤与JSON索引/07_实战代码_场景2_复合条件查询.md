from pymilvus import MilvusClient, DataType
from sentence_transformers import SentenceTransformer
import random

# 初始化
client = MilvusClient("http://localhost:19530")
model = SentenceTransformer('all-MiniLM-L6-v2')

## 场景1：AND 组合查询

### 数据准备

```python
# 创建集合
collection_name = "products_complex"
if client.has_collection(collection_name):
    client.drop_collection(collection_name)

schema = client.create_schema(auto_id=True)
schema.add_field("id", DataType.INT64, is_primary=True)
schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
schema.add_field("price", DataType.DOUBLE)
schema.add_field("stock", DataType.INT32)
schema.add_field("brand", DataType.VARCHAR, max_length=100)
schema.add_field("category", DataType.VARCHAR, max_length=100)
schema.add_field("rating", DataType.DOUBLE)

client.create_collection(collection_name=collection_name, schema=schema)

# 准备数据
brands = ["Apple", "Samsung", "Huawei", "Xiaomi", "OPPO"]
categories = ["phone", "laptop", "tablet", "watch", "earbuds"]
data = []

for i in range(10000):
    text = f"Product {i}"
    vector = model.encode(text).tolist()
    data.append({
        "vector": vector,
        "price": random.uniform(100, 10000),
        "stock": random.randint(0, 100),
        "brand": random.choice(brands),
        "category": random.choice(categories),
        "rating": random.uniform(1.0, 5.0)
    })

client.insert(collection_name, data)
print(f"插入 {len(data)} 条数据")
```

## 场景2：OR 组合查询

### OR 查询示例

```python
# 示例1：多品牌选择
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="brand == 'Apple' OR brand == 'Samsung' OR brand == 'Huawei'",
    limit=10,
    output_fields=["brand", "price"]
)

print("\n查询结果（Apple/Samsung/Huawei）：")
for hits in results:
    for hit in hits:
        print(f"  Brand: {hit['entity']['brand']}, Price: {hit['entity']['price']:.2f}")

# 示例2：使用 IN 替代 OR（推荐）
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="brand IN ['Apple', 'Samsung', 'Huawei']",
    limit=10,
    output_fields=["brand", "price"]
)

print("\n查询结果（使用IN）：")
for hits in results:
    for hit in hits:
        print(f"  Brand: {hit['entity']['brand']}")
```

### 性能对比

```python
import time

# OR 操作符
start = time.time()
client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="brand == 'Apple' OR brand == 'Samsung' OR brand == 'Huawei'",
    limit=10
)
time_or = (time.time() - start) * 1000

# IN 操作符
start = time.time()
client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="brand IN ['Apple', 'Samsung', 'Huawei']",
    limit=10
)
time_in = (time.time() - start) * 1000

print(f"\n性能对比：")
print(f"OR 操作符：{time_or:.2f} ms")
print(f"IN 操作符：{time_in:.2f} ms")
print(f"性能提升：{time_or / time_in:.2f}x")
```

## 场景5：优先级控制

### 括号优先级

```python
# 错误：没有括号，优先级不明确
# filter = "brand == 'Apple' OR brand == 'Samsung' AND price >= 5000"
# 实际执行：brand == 'Apple' OR (brand == 'Samsung' AND price >= 5000)

# 正确：使用括号明确优先级
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="(brand == 'Apple' OR brand == 'Samsung') AND price >= 5000",
    limit=10,
    output_fields=["brand", "price"]
)

print("\n查询结果（明确优先级）：")
for hits in results:
    for hit in hits:
        print(f"  Brand: {hit['entity']['brand']}, Price: {hit['entity']['price']:.2f}")
```

## 性能优化

### 优化1：选择性高的条件在前

```python
import time

# 不好的写法（选择性低的条件在前）
start = time.time()
client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="price >= 100 AND price <= 9000 AND stock > 50",
    limit=10
)
time_bad = (time.time() - start) * 1000

# 好的写法（选择性高的条件在前）
start = time.time()
client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="stock > 50 AND price >= 100 AND price <= 9000",
    limit=10
)
time_good = (time.time() - start) * 1000

print(f"\n性能对比（条件顺序）：")
print(f"选择性低在前：{time_bad:.2f} ms")
print(f"选择性高在前：{time_good:.2f} ms")
print(f"性能提升：{time_bad / time_good:.2f}x")
```

### 优化2：使用 IN 替代多个 OR

```python
# 不推荐
filter_or = "brand == 'Apple' OR brand == 'Samsung' OR brand == 'Huawei' OR brand == 'Xiaomi'"

# 推荐
filter_in = "brand IN ['Apple', 'Samsung', 'Huawei', 'Xiaomi']"

# 性能测试
start = time.time()
client.search(collection_name, [query_vector], filter=filter_or, limit=10)
time_or = (time.time() - start) * 1000

start = time.time()
client.search(collection_name, [query_vector], filter=filter_in, limit=10)
time_in = (time.time() - start) * 1000

print(f"\nOR vs IN 性能对比：")
print(f"OR：{time_or:.2f} ms")
print(f"IN：{time_in:.2f} ms")
```

### 优化3：避免 NOT 操作符

```python
# 不推荐：使用 NOT
filter_not = "NOT (age >= 60)"

# 推荐：使用直接操作符
filter_direct = "age < 60"

# NOT 可能导致全表扫描，性能较差
```

## 常见问题

### Q1: 括号不匹配

**问题**：
```python
# 错误：括号不匹配
filter = "(brand == 'Apple' OR brand == 'Samsung' AND price >= 5000"
# SyntaxError
```

**解决方案**：
```python
# 正确：括号匹配
filter = "(brand == 'Apple' OR brand == 'Samsung') AND price >= 5000"
```

### Q2: 优先级错误

**问题**：
```python
# 错误：没有括号，优先级不明确
filter = "brand == 'Apple' OR brand == 'Samsung' AND price >= 5000"
# 实际执行：brand == 'Apple' OR (brand == 'Samsung' AND price >= 5000)
```

**解决方案**：
```python
# 正确：使用括号明确优先级
filter = "(brand == 'Apple' OR brand == 'Samsung') AND price >= 5000"
```

### Q3: 字符串引号错误

**问题**：
```python
# 错误：引号不匹配
filter = "brand == 'Apple" AND price >= 5000"
```

**解决方案**：
```python
# 正确：引号匹配
filter = "brand == 'Apple' AND price >= 5000"
# 或使用双引号
filter = 'brand == "Apple" AND price >= 5000'
```

## 完整示例代码

```python
from pymilvus import MilvusClient, DataType
from sentence_transformers import SentenceTransformer
import random
import time

def main():
    # 初始化
    client = MilvusClient("http://localhost:19530")
    model = SentenceTransformer('all-MiniLM-L6-v2')

    # 创建集合
    collection_name = "complex_query_demo"
    if client.has_collection(collection_name):
        client.drop_collection(collection_name)

    schema = client.create_schema(auto_id=True)
    schema.add_field("id", DataType.INT64, is_primary=True)
    schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
    schema.add_field("price", DataType.DOUBLE)
    schema.add_field("stock", DataType.INT32)
    schema.add_field("brand", DataType.VARCHAR, max_length=100)
    schema.add_field("rating", DataType.DOUBLE)

    client.create_collection(collection_name=collection_name, schema=schema)

    # 准备数据
    brands = ["Apple", "Samsung", "Huawei"]
    data = []
    for i in range(10000):
        text = f"Product {i}"
        vector = model.encode(text).tolist()
        data.append({
            "vector": vector,
            "price": random.uniform(100, 10000),
            "stock": random.randint(0, 100),
            "brand": random.choice(brands),
            "rating": random.uniform(1.0, 5.0)
        })

    client.insert(collection_name, data)

    # 创建索引
    index_params = client.prepare_index_params()
    index_params.add_index("vector", "AUTOINDEX", "COSINE")
    index_params.add_index("price", "INVERTED")
    index_params.add_index("stock", "INVERTED")
    index_params.add_index("brand", "BITMAP")
    index_params.add_index("rating", "INVERTED")
    client.create_index(collection_name, index_params)

    query_vector = model.encode("Product query").tolist()

    # AND 查询
    print("=== AND 查询 ===")
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="price >= 1000 AND price <= 5000 AND stock > 10",
        limit=5,
        output_fields=["price", "stock"]
    )
    for hits in results:
        for hit in hits:
            print(f"Price: {hit['entity']['price']:.2f}, Stock: {hit['entity']['stock']}")

    # OR 查询（使用IN）
    print("\n=== OR 查询（IN） ===")
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="brand IN ['Apple', 'Samsung']",
        limit=5,
        output_fields=["brand"]
    )
    for hits in results:
        for hit in hits:
            print(f"Brand: {hit['entity']['brand']}")

    # 复合查询
    print("\n=== 复合查询 ===")
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="(brand == 'Apple' AND price >= 5000) OR (brand == 'Samsung' AND rating >= 4.5)",
        limit=5,
        output_fields=["brand", "price", "rating"]
    )
    for hits in results:
        for hit in hits:
            print(f"Brand: {hit['entity']['brand']}, "
                  f"Price: {hit['entity']['price']:.2f}, "
                  f"Rating: {hit['entity']['rating']:.2f}")

if __name__ == "__main__":
    main()
```

## 类比理解

### 前端开发类比

**AND = JavaScript &&**：
```javascript
// JavaScript
const filtered = products.filter(p =>
  p.price >= 1000 && p.price <= 5000 && p.stock > 10
);

// Milvus
filter = "price >= 1000 AND price <= 5000 AND stock > 10"
```

**OR = JavaScript ||**：
```javascript
// JavaScript
const filtered = products.filter(p =>
  p.brand === 'Apple' || p.brand === 'Samsung'
);

// Milvus
filter = "brand IN ['Apple', 'Samsung']"
```

### 日常生活类比

**AND = 同时满足**：
```
找工作的条件：
- 薪资 > 20k（必须）
- 地点在北京（必须）
- 五险一金（必须）

所有条件都要满足 = AND
```

**OR = 满足任一**：
```
交通方式选择：
- 地铁
- 公交
- 打车

任选一种即可 = OR
```
