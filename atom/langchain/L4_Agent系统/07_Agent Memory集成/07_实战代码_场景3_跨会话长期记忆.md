# 实战代码 场景 3：跨会话长期记忆

> LangGraph Store + 语义搜索 — 让 Agent 在新会话中也能记住用户

---

## 场景描述

**问题：** Checkpointer 只管「同一 thread 内」的对话。用户开新对话后，Agent 完全不认识他了。

**目标：** 创建一个能跨会话记住用户偏好、历史事实的 Agent。

**需求：**
- 自动从对话中提取用户信息并存储
- 新会话开始时自动加载用户的长期记忆
- 支持语义搜索（根据当前话题找相关记忆）
- 短期记忆 + 长期记忆协同工作

---

## 方案 1：Store 基础用法

```python
"""
场景 3.1：使用 InMemoryStore 实现跨会话记忆
演示 Store 的基本 put / get / search 操作
"""
from langchain.embeddings import init_embeddings
from langgraph.store.memory import InMemoryStore


# ========== 1. 创建带语义搜索的 Store ==========
embeddings = init_embeddings("openai:text-embedding-3-small")

store = InMemoryStore(
    index={
        "embed": embeddings,     # Embedding 模型
        "dims": 1536,            # text-embedding-3-small 的维度
    }
)


# ========== 2. 存入用户信息 ==========
# namespace: (用户ID, 记忆类别)
# key: 唯一标识
# value: 记忆内容

# Alice 的偏好
store.put(
    ("user-alice", "preferences"), "lang",
    {"text": "喜欢 Python，偏好简洁代码风格"}
)
store.put(
    ("user-alice", "preferences"), "topic",
    {"text": "正在学习 RAG 开发，对向量数据库特别感兴趣"}
)
store.put(
    ("user-alice", "preferences"), "level",
    {"text": "有 3 年编程经验，但 AI/ML 是初学者"}
)

# Alice 的事实
store.put(
    ("user-alice", "facts"), "name",
    {"text": "用户名叫 Alice"}
)
store.put(
    ("user-alice", "facts"), "job",
    {"text": "在一家互联网公司做后端开发"}
)


# ========== 3. 检索用户信息 ==========

# 方式 1：精确获取
item = store.get(("user-alice", "preferences"), "lang")
print(f"精确获取: {item.value['text']}")
# → "喜欢 Python，偏好简洁代码风格"

# 方式 2：语义搜索（根据查询找相关记忆）
results = store.search(
    ("user-alice", "preferences"),
    query="用户的技术水平如何？",
    limit=2,
)
print("\n语义搜索结果：")
for item in results:
    print(f"  - {item.value['text']} (相似度: {item.score:.3f})")
# → "有 3 年编程经验，但 AI/ML 是初学者" (相似度最高)
# → "正在学习 RAG 开发..." (相似度次之)

# 方式 3：列出所有记忆
all_prefs = store.search(("user-alice", "preferences"), limit=10)
print(f"\nAlice 的所有偏好 ({len(all_prefs)} 条)：")
for item in all_prefs:
    print(f"  [{item.key}] {item.value['text']}")
```

### Store 数据模型

```
Store 内部结构：

store = {
    ("user-alice", "preferences"): {    ← namespace（命名空间）
        "lang":  {"text": "喜欢Python..."},     ← key: value
        "topic": {"text": "正在学习RAG..."},
        "level": {"text": "有3年经验..."},
    },
    ("user-alice", "facts"): {
        "name": {"text": "用户名叫Alice"},
        "job":  {"text": "后端开发"},
    },
    ("user-bob", "preferences"): {
        "lang": {"text": "喜欢Java..."},
    },
}

命名空间设计建议：
  (user_id, "preferences")  → 用户偏好
  (user_id, "facts")        → 用户事实
  (user_id, "history")      → 重要历史事件
  (user_id, "feedback")     → 用户反馈
```

## 方案 2：Store + Agent 完整集成

```python
"""
场景 3.2：将 Store 集成到 Agent 中
Agent 自动从 Store 加载长期记忆，并在对话中更新
"""
from langchain.agents import create_agent
from langchain.agents.middleware import before_model, after_model
from langchain.embeddings import init_embeddings
from langchain_core.messages import SystemMessage
from langchain_core.tools import tool
from langgraph.checkpoint.memory import InMemorySaver
from langgraph.store.memory import InMemoryStore


# ========== 1. 设置 Store ==========
embeddings = init_embeddings("openai:text-embedding-3-small")
store = InMemoryStore(index={"embed": embeddings, "dims": 1536})

# 预存一些用户记忆（模拟之前的对话已经提取了这些信息）
store.put(
    ("user-alice", "memories"), "m1",
    {"text": "Alice 是 Python 后端开发者，有 3 年经验"}
)
store.put(
    ("user-alice", "memories"), "m2",
    {"text": "Alice 正在学习 RAG 开发，目标是做智能文档问答系统"}
)
store.put(
    ("user-alice", "memories"), "m3",
    {"text": "Alice 的公司使用 PostgreSQL 和 Redis"}
)


# ========== 2. 中间件：自动注入长期记忆 ==========
@before_model
def inject_long_term_memory(state, runtime):
    """在模型调用前，从 Store 搜索相关的长期记忆并注入"""
    messages = state["messages"]

    # 获取用户最新消息作为搜索查询
    user_messages = [m for m in messages if m.type == "human"]
    if not user_messages:
        return None

    latest_query = user_messages[-1].content

    # 从 runtime.store 搜索相关记忆
    # 注意：需要从 config 中获取 user_id
    user_id = runtime.config.get("configurable", {}).get("user_id", "unknown")
    namespace = (user_id, "memories")

    try:
        memories = runtime.store.search(
            namespace,
            query=latest_query,
            limit=3,  # 最多注入 3 条相关记忆
        )
    except Exception:
        memories = []

    if memories:
        memory_text = "\n".join(
            f"- {m.value['text']}" for m in memories
        )
        # 将长期记忆注入为系统消息
        memory_msg = SystemMessage(
            content=f"【用户长期记忆】\n{memory_text}\n请参考这些信息个性化回答。"
        )
        # 注入到消息列表中（system prompt 之后）
        return {"messages": [memory_msg]}

    return None


# ========== 3. 工具：手动存储记忆 ==========
@tool
def save_user_memory(memory_text: str) -> str:
    """保存用户的重要信息到长期记忆中。
    当用户告诉你他们的偏好、背景、需求等重要信息时使用。
    """
    # 注意：在实际实现中，store 和 user_id 会通过 runtime 获取
    # 这里为了演示简化处理
    import uuid
    key = f"m-{uuid.uuid4().hex[:8]}"
    store.put(("user-alice", "memories"), key, {"text": memory_text})
    return f"已保存到长期记忆: {memory_text}"


# ========== 4. 创建 Agent ==========
agent = create_agent(
    model="gpt-4.1",
    tools=[save_user_memory],
    system_prompt=(
        "你是一个个性化学习助手。"
        "你能记住用户的偏好和背景信息，并在回答中体现。"
        "当用户分享重要的个人信息时，使用 save_user_memory 工具保存。"
        "重要信息包括：姓名、职业、技术栈、学习目标、偏好等。"
    ),
    middleware=[inject_long_term_memory],
    checkpointer=InMemorySaver(),
    store=store,
)


# ========== 5. 测试跨会话记忆 ==========
if __name__ == "__main__":
    # --- 会话 1 ---
    print("=" * 50)
    print("会话 1（已有长期记忆）")
    print("=" * 50)

    config1 = {
        "configurable": {
            "thread_id": "alice-session-new-001",
            "user_id": "user-alice",
        }
    }

    # 新会话，但 Agent 应该从 Store 获取用户信息
    resp = agent.invoke(
        {"messages": [{"role": "user", "content": "你好！你还记得我吗？"}]},
        config1,
    )
    print(f"🤖 {resp['messages'][-1].content}")
    # 期望：Agent 能从 Store 获取信息，知道 Alice 的背景

    # 添加新的长期记忆
    resp = agent.invoke(
        {"messages": [{"role": "user", "content": "我最近开始使用 ChromaDB 做向量存储"}]},
        config1,
    )
    print(f"\n🤖 {resp['messages'][-1].content}")
    # 期望：Agent 使用 save_user_memory 保存这个新信息

    # --- 会话 2（全新的 thread，但仍然记得用户） ---
    print("\n" + "=" * 50)
    print("会话 2（新 thread，测试长期记忆）")
    print("=" * 50)

    config2 = {
        "configurable": {
            "thread_id": "alice-session-new-002",  # 新的 thread！
            "user_id": "user-alice",                # 同一个用户
        }
    }

    resp = agent.invoke(
        {"messages": [{"role": "user", "content": "推荐一些适合我的学习资源"}]},
        config2,
    )
    print(f"🤖 {resp['messages'][-1].content}")
    # 期望：即使是新 thread，Agent 仍能从 Store 获取 Alice 的背景
    # 推荐应该考虑她是 Python 开发者、在学 RAG、用 ChromaDB 等
```

## 方案 3：自动记忆提取（从对话中自动学习）

```python
"""
场景 3.3：Agent 自动从对话中提取和存储用户信息
无需用户主动要求保存，Agent 自动识别重要信息
"""
from langchain.agents import create_agent
from langchain.agents.middleware import after_model
from langchain.embeddings import init_embeddings
from langchain_openai import ChatOpenAI
from langgraph.checkpoint.memory import InMemorySaver
from langgraph.store.memory import InMemoryStore


# Store 设置
embeddings = init_embeddings("openai:text-embedding-3-small")
store = InMemoryStore(index={"embed": embeddings, "dims": 1536})

# 用于提取记忆的小模型
extraction_llm = ChatOpenAI(model="gpt-4.1-mini", temperature=0)


@after_model
def auto_extract_memories(state, runtime):
    """对话结束后，自动从最新对话中提取用户信息"""
    messages = state["messages"]

    # 获取最近的用户消息和 AI 回复
    recent_human = None
    recent_ai = None
    for msg in reversed(messages):
        if msg.type == "ai" and not recent_ai:
            recent_ai = msg.content
        elif msg.type == "human" and not recent_human:
            recent_human = msg.content
        if recent_human and recent_ai:
            break

    if not recent_human:
        return None

    # 用 LLM 判断是否包含值得记住的信息
    extraction_prompt = f"""分析以下对话，提取用户透露的重要个人信息。
只提取明确的事实，不要推测。
如果没有值得记住的信息，回复"无"。

用户: {recent_human}
助手: {recent_ai or ''}

提取的信息（每条一行，简洁描述）："""

    try:
        result = extraction_llm.invoke(extraction_prompt)
        extracted = result.content.strip()

        if extracted and extracted != "无":
            # 存入 Store
            user_id = runtime.config.get("configurable", {}).get(
                "user_id", "unknown"
            )
            import uuid
            for line in extracted.split("\n"):
                line = line.strip().lstrip("- •")
                if line and len(line) > 5:  # 过滤太短的内容
                    key = f"auto-{uuid.uuid4().hex[:8]}"
                    runtime.store.put(
                        (user_id, "memories"), key,
                        {"text": line, "source": "auto_extracted"}
                    )
                    print(f"  💾 自动保存记忆: {line}")
    except Exception as e:
        print(f"  ⚠️ 记忆提取失败: {e}")

    return None


# 创建 Agent
agent = create_agent(
    model="gpt-4.1",
    tools=[],
    system_prompt="你是一个友好的助手。自然地与用户对话，了解他们的需求。",
    middleware=[auto_extract_memories],
    checkpointer=InMemorySaver(),
    store=store,
)


# 测试
if __name__ == "__main__":
    config = {
        "configurable": {
            "thread_id": "auto-extract-test",
            "user_id": "user-charlie",
        }
    }

    conversations = [
        "你好！我叫 Charlie，我在深圳做全栈开发",
        "我最近在用 FastAPI + React 做项目",
        "我想在项目中加入 AI 功能，特别是文档搜索",
        "对了，我们团队有 5 个人，都是 Python 技术栈",
    ]

    for msg in conversations:
        print(f"\n👤 {msg}")
        resp = agent.invoke(
            {"messages": [{"role": "user", "content": msg}]},
            config,
        )
        print(f"🤖 {resp['messages'][-1].content[:150]}...")

    # 查看自动提取的记忆
    print("\n" + "=" * 50)
    print("📋 自动提取的 Charlie 的记忆：")
    print("=" * 50)
    all_memories = store.search(("user-charlie", "memories"), limit=20)
    for m in all_memories:
        print(f"  [{m.key}] {m.value['text']}")
```

## 三种长期记忆方案对比

```
方案 1: Store 基础用法（手动读写）
  ├── 优点：完全可控，简单明了
  ├── 缺点：需要手动管理
  └── 适合：数据量小、需求明确

方案 2: Store + Agent 集成（middleware 注入）
  ├── 优点：自动注入相关记忆，用户体验好
  ├── 缺点：需要配置 middleware
  └── 适合：个性化助手、学习系统

方案 3: 自动记忆提取（AI 驱动）
  ├── 优点：全自动，无需用户操作
  ├── 缺点：有额外 LLM 调用成本，可能提取不准确
  └── 适合：长期关系型应用（如私人助理）

推荐组合：方案 2 + 方案 3 = 自动提取 + 自动注入
```

## Checkpointer + Store 协同工作流

```
用户开始新会话：

  1. 创建新的 thread_id（短期记忆为空）

  2. before_model 中间件触发：
     → 从 Store 搜索用户的长期记忆
     → 注入到 system prompt 中

  3. Agent 推理：
     → 利用长期记忆个性化回答
     → Checkpointer 自动保存对话

  4. after_model 中间件触发：
     → 分析对话内容
     → 提取新的用户信息
     → 存入 Store

  5. 下次新会话：
     → 重复步骤 1-4
     → Store 中的记忆越来越丰富

┌────────────────────────────────────────────┐
│                                            │
│   Store (长期)                              │
│   ┌──────────────────────┐                 │
│   │ Alice: Python开发     │ ←── 存入 ──┐   │
│   │ Alice: 学习RAG        │            │   │
│   │ Alice: 用ChromaDB     │            │   │
│   └──────┬───────────────┘            │   │
│          │ 读取                        │   │
│          ↓                            │   │
│   ┌──────────────────────┐            │   │
│   │ before_model:        │            │   │
│   │ 注入相关记忆到prompt  │            │   │
│   └──────┬───────────────┘            │   │
│          ↓                            │   │
│   ┌──────────────────────┐            │   │
│   │ Agent 推理           │            │   │
│   │ (利用长期记忆回答)    │            │   │
│   └──────┬───────────────┘            │   │
│          ↓                            │   │
│   ┌──────────────────────┐            │   │
│   │ after_model:         │────────────┘   │
│   │ 提取新信息存入Store   │               │
│   └──────────────────────┘               │
│                                            │
│   Checkpointer (短期)                      │
│   thread-001: [msg1, msg2, msg3...]        │
│   thread-002: [msg1, msg2...]              │
│                                            │
└────────────────────────────────────────────┘
```

## 与 RAG 开发的关系

| 长期记忆技术 | RAG 应用 |
|------------|---------|
| Store 语义搜索 | 记住用户常检索的主题，优化推荐 |
| 自动记忆提取 | 自动学习用户的知识领域和水平 |
| before_model 注入 | 个性化 RAG 检索策略（根据用户背景调整） |
| 用户偏好存储 | 记住用户偏好的回答风格和详细程度 |
