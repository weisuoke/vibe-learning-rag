# 图可视化 - 实战代码: 场景3 子图深度探索

## 场景描述

在开发包含多层嵌套子图的复杂工作流时,需要使用 xray 参数深入探索子图结构,理解子图的内部逻辑和连接关系。

## 完整代码

```python
"""
场景3: 子图深度探索
演示: 使用 xray 参数探索嵌套子图结构
"""

from langgraph.graph import StateGraph, START, END
from typing import TypedDict
from IPython.display import Image, display, Markdown

# ===== 1. 定义状态 =====
class State(TypedDict):
    """工作流状态定义"""
    query: str
    documents: list[str]
    answer: str
    step: int

# ===== 2. 创建子图1: 查询处理 =====
print("=== 创建子图1: 查询处理 ===")

def analyze_query(state: State) -> State:
    """分析查询"""
    print(f"[Analyze] 分析查询: {state['query']}")
    return {**state, "step": state.get("step", 0) + 1}

def rewrite_query(state: State) -> State:
    """改写查询"""
    print(f"[Rewrite] 改写查询: {state['query']}")
    new_query = f"{state['query']} (改写)"
    return {**state, "query": new_query, "step": state.get("step", 0) + 1}

# 创建查询处理子图
query_workflow = StateGraph(State)
query_workflow.add_node("analyze", analyze_query)
query_workflow.add_node("rewrite", rewrite_query)
query_workflow.add_edge(START, "analyze")
query_workflow.add_edge("analyze", "rewrite")
query_workflow.add_edge("rewrite", END)

query_app = query_workflow.compile()
print("查询处理子图创建完成")

# ===== 3. 创建子图2: 文档处理 =====
print("\n=== 创建子图2: 文档处理 ===")

def retrieve_docs(state: State) -> State:
    """检索文档"""
    print(f"[Retrieve] 检索文档: {state['query']}")
    documents = [f"文档{i}: {state['query']}" for i in range(1, 4)]
    return {**state, "documents": documents, "step": state.get("step", 0) + 1}

def grade_docs(state: State) -> State:
    """评分文档"""
    print(f"[Grade] 评分 {len(state['documents'])} 个文档")
    relevant_docs = state['documents'][:2]  # 保留前2个
    return {**state, "documents": relevant_docs, "step": state.get("step", 0) + 1}

# 创建文档处理子图
doc_workflow = StateGraph(State)
doc_workflow.add_node("retrieve", retrieve_docs)
doc_workflow.add_node("grade", grade_docs)
doc_workflow.add_edge(START, "retrieve")
doc_workflow.add_edge("retrieve", "grade")
doc_workflow.add_edge("grade", END)

doc_app = doc_workflow.compile()
print("文档处理子图创建完成")

# ===== 4. 创建主图 =====
print("\n=== 创建主图 ===")

def generate_answer(state: State) -> State:
    """生成答案"""
    print(f"[Generate] 基于 {len(state['documents'])} 个文档生成答案")
    answer = f"答案: 基于 {len(state['documents'])} 个文档..."
    return {**state, "answer": answer, "step": state.get("step", 0) + 1}

# 创建主图
main_workflow = StateGraph(State)
main_workflow.add_node("query_processing", query_app)  # 嵌入子图1
main_workflow.add_node("doc_processing", doc_app)      # 嵌入子图2
main_workflow.add_node("generate", generate_answer)

main_workflow.add_edge(START, "query_processing")
main_workflow.add_edge("query_processing", "doc_processing")
main_workflow.add_edge("doc_processing", "generate")
main_workflow.add_edge("generate", END)

main_app = main_workflow.compile()
print("主图创建完成")

# ===== 5. 可视化 - 不展开子图 (xray=False) =====
print("\n=== 可视化 - 不展开子图 (xray=False) ===")
graph_simple = main_app.get_graph(xray=False)

print(f"节点数量: {len(graph_simple.nodes)}")
print("节点列表:")
for node in graph_simple.nodes:
    print(f"  - {node.id}")

# 在 Jupyter 中显示
display(Markdown("### 简单视图 (xray=False)"))
display(Image(graph_simple.draw_mermaid_png()))

# ===== 6. 可视化 - 展开一层子图 (xray=1) =====
print("\n=== 可视化 - 展开一层子图 (xray=1) ===")
graph_one_level = main_app.get_graph(xray=1)

print(f"节点数量: {len(graph_one_level.nodes)}")
print("节点列表:")
for node in graph_one_level.nodes:
    print(f"  - {node.id}")

# 在 Jupyter 中显示
display(Markdown("### 展开一层 (xray=1)"))
display(Image(graph_one_level.draw_mermaid_png()))

# ===== 7. 可视化 - 展开所有子图 (xray=True) =====
print("\n=== 可视化 - 展开所有子图 (xray=True) ===")
graph_full = main_app.get_graph(xray=True)

print(f"节点数量: {len(graph_full.nodes)}")
print("节点列表:")
for node in graph_full.nodes:
    print(f"  - {node.id}")

# 在 Jupyter 中显示
display(Markdown("### 完整视图 (xray=True)"))
display(Image(graph_full.draw_mermaid_png()))

# ===== 8. 对比不同 xray 参数 =====
print("\n=== 对比不同 xray 参数 ===")

xray_configs = [
    (False, "不展开"),
    (1, "展开一层"),
    (2, "展开两层"),
    (True, "展开所有")
]

for xray_value, description in xray_configs:
    graph = main_app.get_graph(xray=xray_value)
    mermaid_text = graph.draw_mermaid()
    
    print(f"\n{description} (xray={xray_value}):")
    print(f"  节点数: {len(graph.nodes)}")
    print(f"  边数: {len(graph.edges)}")
    print(f"  Mermaid 文本长度: {len(mermaid_text)} 字符")

# ===== 9. 单独可视化子图 =====
print("\n=== 单独可视化子图 ===")

# 子图1
display(Markdown("### 子图1: 查询处理"))
display(Image(query_app.get_graph().draw_mermaid_png()))

# 子图2
display(Markdown("### 子图2: 文档处理"))
display(Image(doc_app.get_graph().draw_mermaid_png()))

# ===== 10. 执行工作流 =====
print("\n=== 执行工作流 ===")

test_input = {
    "query": "LangGraph",
    "documents": [],
    "answer": "",
    "step": 0
}

result = main_app.invoke(test_input)

print(f"\n最终结果:")
print(f"  查询: {result['query']}")
print(f"  文档数: {len(result['documents'])}")
print(f"  答案: {result['answer']}")
print(f"  总步骤数: {result['step']}")

# ===== 11. 分析子图结构 =====
print("\n=== 分析子图结构 ===")

def analyze_graph_structure(app, name, xray=False):
    """分析图结构"""
    graph = app.get_graph(xray=xray)
    
    print(f"\n{name} (xray={xray}):")
    print(f"  节点数: {len(graph.nodes)}")
    print(f"  边数: {len(graph.edges)}")
    
    # 统计节点类型
    start_nodes = [n for n in graph.nodes if n.id == "__start__"]
    end_nodes = [n for n in graph.nodes if n.id == "__end__"]
    regular_nodes = [n for n in graph.nodes if n.id not in ("__start__", "__end__")]
    
    print(f"  START 节点: {len(start_nodes)}")
    print(f"  END 节点: {len(end_nodes)}")
    print(f"  普通节点: {len(regular_nodes)}")
    
    # 统计边类型
    conditional_edges = [e for e in graph.edges if e.conditional]
    regular_edges = [e for e in graph.edges if not e.conditional]
    
    print(f"  条件边: {len(conditional_edges)}")
    print(f"  普通边: {len(regular_edges)}")

# 分析主图
analyze_graph_structure(main_app, "主图", xray=False)
analyze_graph_structure(main_app, "主图", xray=True)

# 分析子图
analyze_graph_structure(query_app, "查询处理子图")
analyze_graph_structure(doc_app, "文档处理子图")

# ===== 12. 保存不同视图 =====
print("\n=== 保存不同视图 ===")

# 保存简单视图
png_simple = main_app.get_graph(xray=False).draw_mermaid_png()
with open("graph_simple.png", "wb") as f:
    f.write(png_simple)
print("简单视图已保存: graph_simple.png")

# 保存详细视图
png_detailed = main_app.get_graph(xray=True).draw_mermaid_png()
with open("graph_detailed.png", "wb") as f:
    f.write(png_detailed)
print("详细视图已保存: graph_detailed.png")

# 保存 Mermaid 文本
mermaid_simple = main_app.get_graph(xray=False).draw_mermaid()
with open("graph_simple.mmd", "w") as f:
    f.write(mermaid_simple)
print("简单视图 Mermaid 文本已保存: graph_simple.mmd")

mermaid_detailed = main_app.get_graph(xray=True).draw_mermaid()
with open("graph_detailed.mmd", "w") as f:
    f.write(mermaid_detailed)
print("详细视图 Mermaid 文本已保存: graph_detailed.mmd")

print("\n=== 子图深度探索完成 ===")
```

## 关键点说明

### 1. xray 参数的三种用法

```python
# 不展开子图
graph = app.get_graph(xray=False)

# 展开一层子图
graph = app.get_graph(xray=1)

# 展开所有子图
graph = app.get_graph(xray=True)
```

### 2. 子图嵌套

```python
# 创建子图
sub_app = sub_workflow.compile()

# 将子图嵌入主图
main_workflow.add_node("sub_graph", sub_app)
```

### 3. 单独可视化子图

```python
# 单独可视化子图
sub_graph = sub_app.get_graph()
display(Image(sub_graph.draw_mermaid_png()))
```

### 4. 对比不同视图

```python
# 对比节点数和边数
for xray_value in [False, 1, True]:
    graph = app.get_graph(xray=xray_value)
    print(f"xray={xray_value}: {len(graph.nodes)} 节点, {len(graph.edges)} 边")
```

## 常见问题

### Q1: 如何判断是否需要使用 xray?

**A**: 检查是否有子图:
```python
subgraphs = list(app.get_subgraphs())
if subgraphs:
    print(f"包含 {len(subgraphs)} 个子图,建议使用 xray")
    graph = app.get_graph(xray=True)
else:
    print("不包含子图,无需使用 xray")
    graph = app.get_graph()
```

### Q2: xray=True 导致图表过于复杂怎么办?

**A**: 使用 xray=1 或 xray=2 控制展开深度:
```python
# 只展开一层
graph = app.get_graph(xray=1)
```

### Q3: 如何单独查看某个子图?

**A**: 获取子图并单独可视化:
```python
for name, sub_app in app.get_subgraphs():
    print(f"子图: {name}")
    sub_graph = sub_app.get_graph()
    display(Image(sub_graph.draw_mermaid_png()))
```
