# 实战代码 - 场景3：混合检索实战

> 实现 Dense + Sparse 混合检索，对比 RRFRanker 和 WeightedRanker

---

## 场景描述

构建一个**智能文档搜索系统**，结合稠密向量（语义相似度）和稀疏向量（关键词匹配）的优势，实现更准确的检索。

### 技术需求
- 同时使用稠密向量和稀疏向量
- 对比 RRFRanker 和 WeightedRanker 两种融合策略
- 评估混合检索的性能提升

---

## 完整代码

```python
"""
场景3：混合检索实战
演示：Dense + Sparse 混合检索，对比不同融合策略

数据来源：
- reference/context7_pymilvus_01.md - PyMilvus 官方文档
- reference/search_hybrid_weight_01.md - 混合检索权重调优
"""

import os
from pymilvus import (
    connections, FieldSchema, CollectionSchema, DataType, Collection,
    Function, FunctionType, AnnSearchRequest, RRFRanker, WeightedRanker, utility
)
import numpy as np
import time

print("=== 混合检索实战 ===\n")

# ===== 1. 连接 Milvus =====
connections.connect(host=os.getenv("MILVUS_HOST", "localhost"), port="19530")
print("✓ 连接 Milvus")

# ===== 2. 创建 Collection（Dense + Sparse）=====
fields = [
    FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
    FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
    FieldSchema(name="dense_vector", dtype=DataType.FLOAT_VECTOR, dim=128),
    FieldSchema(name="sparse_vector", dtype=DataType.SPARSE_FLOAT_VECTOR)
]

bm25_fn = Function(
    name="bm25", input_field_names=["content"],
    output_field_names=["sparse_vector"], function_type=FunctionType.BM25
)

collection_name = "hybrid_search_demo"
if utility.has_collection(collection_name):
    utility.drop_collection(collection_name)

schema = CollectionSchema(fields=fields, functions=[bm25_fn])
collection = Collection(name=collection_name, schema=schema)
print(f"✓ 创建 Collection: {collection_name}")

# ===== 3. 插入测试数据 =====
docs = [
    "Python 是一种高级编程语言，广泛用于数据科学和机器学习",
    "机器学习是人工智能的一个分支，使用算法从数据中学习",
    "深度学习是机器学习的子集，使用神经网络处理复杂问题",
    "自然语言处理 NLP 是 AI 的重要应用，处理文本和语音",
    "计算机视觉使用深度学习识别图像中的物体和场景",
    "强化学习通过奖励机制训练智能体做出决策",
    "数据科学结合统计学和编程，从数据中提取洞察",
    "神经网络模拟人脑结构，是深度学习的基础",
    "卷积神经网络 CNN 专门用于图像处理任务",
    "循环神经网络 RNN 适合处理序列数据如文本和时间序列"
]

data = [{
    "content": doc,
    "dense_vector": np.random.rand(128).tolist()
} for doc in docs]

collection.insert(data)
collection.flush()
print(f"✓ 插入 {len(docs)} 条文档")

# ===== 4. 创建索引 =====
# Dense 向量索引
collection.create_index(
    field_name="dense_vector",
    index_params={"index_type": "FLAT", "metric_type": "L2"}
)

# Sparse 向量索引
collection.create_index(
    field_name="sparse_vector",
    index_params={
        "index_type": "SPARSE_INVERTED_INDEX",
        "metric_type": "BM25",
        "params": {"bm25_k1": 1.2, "bm25_b": 0.75}
    }
)

collection.load()
print("✓ 创建索引并加载")

# ===== 5. 执行混合检索 =====
query_text = "深度学习和神经网络"
query_dense = np.random.rand(128).tolist()

print(f"\n查询：{query_text}\n")

# 5.1 单独的 Dense 检索
print("--- Dense 检索 ---")
start = time.time()
dense_results = collection.search(
    data=[query_dense], anns_field="dense_vector",
    param={"metric_type": "L2"}, limit=3, output_fields=["content"]
)
print(f"耗时：{(time.time()-start)*1000:.2f}ms")
for i, hit in enumerate(dense_results[0]):
    print(f"  {i+1}. {hit.entity.get('content')[:40]}... (score: {hit.score:.4f})")

# 5.2 单独的 Sparse 检索
print("\n--- Sparse (BM25) 检索 ---")
start = time.time()
sparse_results = collection.search(
    data=[query_text], anns_field="sparse_vector",
    param={"metric_type": "BM25"}, limit=3, output_fields=["content"]
)
print(f"耗时：{(time.time()-start)*1000:.2f}ms")
for i, hit in enumerate(sparse_results[0]):
    print(f"  {i+1}. {hit.entity.get('content')[:40]}... (score: {hit.score:.4f})")

# 5.3 混合检索 - RRFRanker
print("\n--- 混合检索 (RRFRanker) ---")
dense_req = AnnSearchRequest(
    data=[query_dense], anns_field="dense_vector",
    param={"metric_type": "L2"}, limit=10
)
sparse_req = AnnSearchRequest(
    data=[query_text], anns_field="sparse_vector",
    param={"metric_type": "BM25"}, limit=10
)

start = time.time()
rrf_results = collection.hybrid_search(
    reqs=[dense_req, sparse_req],
    ranker=RRFRanker(k=60),
    limit=3,
    output_fields=["content"]
)
print(f"耗时：{(time.time()-start)*1000:.2f}ms")
for i, hit in enumerate(rrf_results[0]):
    print(f"  {i+1}. {hit.entity.get('content')[:40]}... (score: {hit.score:.4f})")

# 5.4 混合检索 - WeightedRanker（不同权重）
weight_configs = [
    (0.7, 0.3, "Dense 优先"),
    (0.5, 0.5, "平衡"),
    (0.3, 0.7, "Sparse 优先")
]

for w1, w2, name in weight_configs:
    print(f"\n--- 混合检索 (WeightedRanker: {name}) ---")
    start = time.time()
    weighted_results = collection.hybrid_search(
        reqs=[dense_req, sparse_req],
        ranker=WeightedRanker(w1, w2),
        limit=3,
        output_fields=["content"]
    )
    print(f"权重：Dense={w1}, Sparse={w2}")
    print(f"耗时：{(time.time()-start)*1000:.2f}ms")
    for i, hit in enumerate(weighted_results[0]):
        print(f"  {i+1}. {hit.entity.get('content')[:40]}... (score: {hit.score:.4f})")

# ===== 6. 性能对比实验 =====
print("\n=== 性能对比实验 ===\n")

queries = [
    ("机器学习算法", np.random.rand(128).tolist()),
    ("图像识别技术", np.random.rand(128).tolist()),
    ("文本处理方法", np.random.rand(128).tolist())
]

results_summary = []

for query_text, query_vec in queries:
    # Dense 检索
    start = time.time()
    collection.search(data=[query_vec], anns_field="dense_vector",
                     param={"metric_type": "L2"}, limit=5)
    dense_time = time.time() - start

    # Sparse 检索
    start = time.time()
    collection.search(data=[query_text], anns_field="sparse_vector",
                     param={"metric_type": "BM25"}, limit=5)
    sparse_time = time.time() - start

    # 混合检索
    start = time.time()
    collection.hybrid_search(
        reqs=[
            AnnSearchRequest(data=[query_vec], anns_field="dense_vector",
                           param={"metric_type": "L2"}, limit=10),
            AnnSearchRequest(data=[query_text], anns_field="sparse_vector",
                           param={"metric_type": "BM25"}, limit=10)
        ],
        ranker=RRFRanker(k=60),
        limit=5
    )
    hybrid_time = time.time() - start

    results_summary.append({
        "query": query_text,
        "dense_ms": dense_time * 1000,
        "sparse_ms": sparse_time * 1000,
        "hybrid_ms": hybrid_time * 1000
    })

print("查询性能对比：")
print(f"{'查询':<20} {'Dense':<12} {'Sparse':<12} {'Hybrid':<12}")
print("-" * 60)
for r in results_summary:
    print(f"{r['query']:<20} {r['dense_ms']:<12.2f} {r['sparse_ms']:<12.2f} {r['hybrid_ms']:<12.2f}")

# ===== 7. 清理 =====
collection.release()
connections.disconnect("default")
print("\n✓ 完成")
```

## 关键点解析

### 1. RRFRanker vs WeightedRanker

**RRFRanker（Reciprocal Rank Fusion）**：
- 公式：`RRF(d) = Σ 1/(k + rank_i(d))`
- 优点：无需手动调整权重，鲁棒性好
- 适用：不确定各向量字段的相对重要性

**WeightedRanker（加权融合）**：
- 公式：`Score = w1 * score1 + w2 * score2`
- 优点：精确控制权重
- 适用：明确知道各向量字段的重要性

**数据来源**：`reference/context7_pymilvus_01.md`

### 2. 混合检索的优势

| 特性 | Dense | Sparse | Hybrid |
|------|-------|--------|--------|
| 语义理解 | ✓✓✓ | ✗ | ✓✓✓ |
| 关键词匹配 | ✗ | ✓✓✓ | ✓✓✓ |
| 准确率 | 中 | 中 | 高 |
| 召回率 | 中 | 中 | 高 |

### 3. 权重调优策略

**实验驱动**：
1. 准备测试集（查询 + 相关文档）
2. 尝试不同权重组合
3. 评估准确率和召回率
4. 选择最优权重

**经验值**：
- 通用场景：Dense=0.5, Sparse=0.5
- 语义理解优先：Dense=0.7, Sparse=0.3
- 关键词匹配优先：Dense=0.3, Sparse=0.7

**数据来源**：`reference/search_hybrid_weight_01.md`

### 4. 性能优化

**索引优化**：
- Dense：使用 HNSW 或 IVF_FLAT
- Sparse：使用 SPARSE_WAND（大规模数据）

**搜索优化**：
- 增加 limit 参数（如 10-20）
- 使用 drop_ratio_search 参数
