```python
from typing import List
from pathlib import Path
import pypdf

class DocumentLoader:
    """文档加载器"""
    
    @staticmethod
    def load_pdf(file_path: str) -> str:
        """加载 PDF 文件"""
        text = ""
        with open(file_path, 'rb') as file:
            pdf_reader = pypdf.PdfReader(file)
            for page in pdf_reader.pages:
                text += page.extract_text()
        return text
    
    @staticmethod
    def load_txt(file_path: str) -> str:
        """加载 TXT 文件"""
        with open(file_path, 'r', encoding='utf-8') as file:
            return file.read()
    
    @staticmethod
    def load_markdown(file_path: str) -> str:
        """加载 Markdown 文件"""
        with open(file_path, 'r', encoding='utf-8') as file:
            return file.read()
    
    @classmethod
    def load_document(cls, file_path: str) -> str:
        """根据文件类型加载文档"""
        path = Path(file_path)
        suffix = path.suffix.lower()
        
        if suffix == '.pdf':
            return cls.load_pdf(file_path)
        elif suffix == '.txt':
            return cls.load_txt(file_path)
        elif suffix in ['.md', '.markdown']:
            return cls.load_markdown(file_path)
        else:
            raise ValueError(f"不支持的文件类型: {suffix}")

print(f"✅ 文档加载成功，长度: {len(document_text)} 字符")
```

```python
from typing import List

class TextChunker:
    """文本分块器"""
    
    def __init__(self, chunk_size: int = 500, chunk_overlap: int = 50):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
    
    def split_text(self, text: str) -> List[str]:
        """分块文本"""
        chunks = []
        start = 0
        
        while start < len(text):
            end = start + self.chunk_size
            chunk = text[start:end]
            
            # 尝试在句子边界分割
            if end < len(text):
                last_period = chunk.rfind('。')
                last_newline = chunk.rfind('\n')
                split_point = max(last_period, last_newline)
                
                if split_point > 0:
                    chunk = chunk[:split_point + 1]
                    end = start + split_point + 1
            
            chunks.append(chunk.strip())
            start = end - self.chunk_overlap
        
        return [c for c in chunks if c]  # 过滤空块

# 测试文本分块
chunker = TextChunker(chunk_size=500, chunk_overlap=50)

print(f"✅ 文本分块完成，共 {len(chunks)} 个块")
```

### 步骤3：Milvus 混合检索管理器

```python
from pymilvus import (
    MilvusClient,
    DataType,
    Function,
    FunctionType,
    AnnSearchRequest,
    RRFRanker
)
from sentence_transformers import SentenceTransformer
from typing import List, Dict
import os
from dotenv import load_dotenv

load_dotenv()

class MilvusHybridRetriever:
    """Milvus 混合检索器"""
    
    def __init__(
        self,
        collection_name: str = "rag_documents",
        uri: str = "http://localhost:19530",
        token: str = "root:Milvus"
    ):
        self.collection_name = collection_name
        self.client = MilvusClient(uri=uri, token=token)
        self.model = SentenceTransformer('sentence-transformers/all-MiniLM-L6-v2')
        
        # 创建 Collection
        self._create_collection()
    
    def _create_collection(self):
        """创建 Collection"""
        # 删除已存在的 Collection
        if self.client.has_collection(self.collection_name):
            self.client.drop_collection(self.collection_name)
        
        # 创建 Schema
        schema = self.client.create_schema(auto_id=False)
        schema.add_field(field_name="id", datatype=DataType.INT64, is_primary=True)
        schema.add_field(field_name="text", datatype=DataType.VARCHAR, 
                         max_length=2000, enable_analyzer=True)
        schema.add_field(field_name="text_dense", datatype=DataType.FLOAT_VECTOR, dim=384)
        schema.add_field(field_name="text_sparse", datatype=DataType.SPARSE_FLOAT_VECTOR)
        
        # BM25 函数
        bm25_function = Function(
            name="text_bm25_emb",
            input_field_names=["text"],
            output_field_names=["text_sparse"],
            function_type=FunctionType.BM25,
        )
        schema.add_function(bm25_function)
        
        # 配置索引
        index_params = self.client.prepare_index_params()
        index_params.add_index(field_name="text_dense", index_type="AUTOINDEX", metric_type="IP")
        index_params.add_index(field_name="text_sparse", index_type="SPARSE_INVERTED_INDEX",
                               metric_type="BM25")
        
        # 创建 Collection
        self.client.create_collection(
            collection_name=self.collection_name,
            schema=schema,
            index_params=index_params
        )
        
        print(f"✅ Collection '{self.collection_name}' 创建成功")
    
    def add_documents(self, chunks: List[str]):
        """添加文档块"""
        # 生成稠密向量
        dense_vectors = self.model.encode(chunks).tolist()
        
        # 构造数据
        data = [
            {
                "id": i,
                "text": chunk,
                "text_dense": dense_vectors[i]
            }
            for i, chunk in enumerate(chunks)
        ]
        
        # 插入数据
        self.client.insert(collection_name=self.collection_name, data=data)
        print(f"✅ 插入 {len(chunks)} 个文档块")
    
    def hybrid_search(self, query: str, top_k: int = 5) -> List[Dict]:
        """混合检索"""
        # 生成查询向量
        query_dense = self.model.encode([query])[0].tolist()
        
        # 创建检索请求
        request_dense = AnnSearchRequest(
            data=[query_dense],
            anns_field="text_dense",
            param={"nprobe": 10},
            limit=top_k * 2  # 召回更多候选
        )
        
        request_sparse = AnnSearchRequest(
            data=[query],
            anns_field="text_sparse",
            limit=top_k * 2
        )
        
        # RRF 重排序
        ranker = RRFRanker(k=60)
        
        # 执行混合检索
        results = self.client.hybrid_search(
            collection_name=self.collection_name,
            reqs=[request_dense, request_sparse],
            ranker=ranker,
            limit=top_k,
            output_fields=["text"]
        )
        
        # 提取结果
        retrieved_docs = []
        for hits in results:
            for hit in hits:
                retrieved_docs.append({
                    "text": hit["entity"]["text"],
                    "score": hit["distance"]
                })
        
        return retrieved_docs

# 初始化检索器
retriever = MilvusHybridRetriever()
retriever.add_documents(chunks)
```

### 步骤4：LLM 生成器

```python
from openai import OpenAI
from typing import List, Dict

class LLMGenerator:
    """LLM 生成器"""
    
    def __init__(self, model: str = "gpt-4"):
        self.client = OpenAI(
            api_key=os.getenv("OPENAI_API_KEY"),
            base_url=os.getenv("OPENAI_BASE_URL", "https://api.openai.com/v1")
        )
        self.model = model
    
    def generate_answer(
        self,
        query: str,
        context_docs: List[Dict],
        max_tokens: int = 500
    ) -> str:
        """生成答案"""
        # 构造上下文
        context = "\n\n".join([
            f"文档 {i+1}:\n{doc['text']}"
            for i, doc in enumerate(context_docs)
        ])
        
        # 构造 Prompt
        prompt = f"""基于以下文档内容回答问题。如果文档中没有相关信息，请说"根据提供的文档无法回答该问题"。

文档内容：
{context}

问题：{query}

请提供详细且准确的答案："""
        
        # 调用 LLM
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": "你是一个专业的文档问答助手，基于提供的文档内容回答问题。"},
                {"role": "user", "content": prompt}
            ],
            max_tokens=max_tokens,
            temperature=0.3
        )
        
        return response.choices[0].message.content

# 初始化生成器
generator = LLMGenerator(model="gpt-4")
```

### 步骤5：RAG 管道

```python
class RAGPipeline:
    """RAG 管道"""
    
    def __init__(
        self,
        retriever: MilvusHybridRetriever,
        generator: LLMGenerator,
        top_k: int = 3
    ):
        self.retriever = retriever
        self.generator = generator
        self.top_k = top_k
    
    def query(self, question: str) -> Dict:
        """查询"""
        print(f"\n🔍 查询: {question}")
        
        # 1. 混合检索
        print("  ⏳ 检索相关文档...")
        retrieved_docs = self.retriever.hybrid_search(question, top_k=self.top_k)
        
        print(f"  ✅ 检索到 {len(retrieved_docs)} 个相关文档")
        for i, doc in enumerate(retrieved_docs):
            print(f"    文档 {i+1} (得分: {doc['score']:.4f}): {doc['text'][:100]}...")
        
        # 2. LLM 生成
        print("  ⏳ 生成答案...")
        answer = self.generator.generate_answer(question, retrieved_docs)
        
        print(f"  ✅ 答案生成完成\n")
        
        return {
            "question": question,
            "answer": answer,
            "retrieved_docs": retrieved_docs
        }

# 创建 RAG 管道
rag_pipeline = RAGPipeline(
    retriever=retriever,
    generator=generator,
    top_k=3
)
```

```python
class MilvusHybridRetrieverWithMetadata(MilvusHybridRetriever):
    """带元数据过滤的混合检索器"""
    
    def _create_collection(self):
        """创建带元数据的 Collection"""
        if self.client.has_collection(self.collection_name):
            self.client.drop_collection(self.collection_name)
        
        schema = self.client.create_schema(auto_id=False)
        schema.add_field(field_name="id", datatype=DataType.INT64, is_primary=True)
        schema.add_field(field_name="text", datatype=DataType.VARCHAR, 
                         max_length=2000, enable_analyzer=True)
        schema.add_field(field_name="source", datatype=DataType.VARCHAR, max_length=200)
        schema.add_field(field_name="page", datatype=DataType.INT64)
        schema.add_field(field_name="text_dense", datatype=DataType.FLOAT_VECTOR, dim=384)
        schema.add_field(field_name="text_sparse", datatype=DataType.SPARSE_FLOAT_VECTOR)
        
        bm25_function = Function(
            name="text_bm25_emb",
            input_field_names=["text"],
            output_field_names=["text_sparse"],
            function_type=FunctionType.BM25,
        )
        schema.add_function(bm25_function)
        
        index_params = self.client.prepare_index_params()
        index_params.add_index(field_name="text_dense", index_type="AUTOINDEX", metric_type="IP")
        index_params.add_index(field_name="text_sparse", index_type="SPARSE_INVERTED_INDEX",
                               metric_type="BM25")
        index_params.add_index(field_name="source", index_type="INVERTED")
        
        self.client.create_collection(
            collection_name=self.collection_name,
            schema=schema,
            index_params=index_params
        )
    
    def hybrid_search_with_filter(
        self,
        query: str,
        source_filter: str = None,
        page_filter: int = None,
        top_k: int = 5
    ) -> List[Dict]:
        """带过滤的混合检索"""
        query_dense = self.model.encode([query])[0].tolist()
        
        # 构造过滤表达式
        expr = []
        if source_filter:
            expr.append(f'source == "{source_filter}"')
        if page_filter:
            expr.append(f'page == {page_filter}')
        filter_expr = " AND ".join(expr) if expr else None
        
        request_dense = AnnSearchRequest(
            data=[query_dense],
            anns_field="text_dense",
            expr=filter_expr,
            limit=top_k * 2
        )
        
        request_sparse = AnnSearchRequest(
            data=[query],
            anns_field="text_sparse",
            expr=filter_expr,
            limit=top_k * 2
        )
        
        ranker = RRFRanker(k=60)
        
        results = self.client.hybrid_search(
            collection_name=self.collection_name,
            reqs=[request_dense, request_sparse],
            ranker=ranker,
            limit=top_k,
            output_fields=["text", "source", "page"]
        )
        
        retrieved_docs = []
        for hits in results:
            for hit in hits:
                retrieved_docs.append({
                    "text": hit["entity"]["text"],
                    "source": hit["entity"]["source"],
                    "page": hit["entity"]["page"],
                    "score": hit["distance"]
                })
        
        return retrieved_docs
```

### 2. 流式生成

```python
class StreamingLLMGenerator(LLMGenerator):
    """流式 LLM 生成器"""
    
    def generate_answer_stream(
        self,
        query: str,
        context_docs: List[Dict],
        max_tokens: int = 500
    ):
        """流式生成答案"""
        context = "\n\n".join([
            f"文档 {i+1}:\n{doc['text']}"
            for i, doc in enumerate(context_docs)
        ])
        
        prompt = f"""基于以下文档内容回答问题。

文档内容：
{context}

问题：{query}

请提供详细且准确的答案："""
        
        stream = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": "你是一个专业的文档问答助手。"},
                {"role": "user", "content": prompt}
            ],
            max_tokens=max_tokens,
            temperature=0.3,
            stream=True
        )
        
        for chunk in stream:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content
```
