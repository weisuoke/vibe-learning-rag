# 实战代码 场景 5：生产级持久化部署

> PostgresSaver + PostgresStore + 环境分离 — 可上线的 Agent 记忆系统

---

## 场景描述

**目标：** 构建一个可以部署到生产环境的 Agent 记忆系统。

**需求：**
- 记忆持久化（重启不丢失）
- 多进程/多 Worker 共享记忆
- 环境分离（开发/测试/生产用不同后端）
- 连接池管理
- 监控和清理

## 方案 1：环境自适应的 Checkpointer 工厂

```python
"""
场景 5.1：根据环境自动选择持久化后端
开发用 InMemory，测试用 SQLite，生产用 PostgreSQL
"""
import os
from typing import Any


def create_checkpointer(env: str | None = None) -> Any:
    """根据环境创建合适的 Checkpointer

    Args:
        env: 环境名称。如果不传，从环境变量 ENV 读取。
             可选值：development, testing, staging, production

    Returns:
        对应环境的 Checkpointer 实例
    """
    env = env or os.environ.get("ENV", "development")

    if env == "development":
        # 开发环境：内存存储，零配置
        from langgraph.checkpoint.memory import InMemorySaver
        print("✅ 使用 InMemorySaver（开发模式）")
        return InMemorySaver()

    elif env == "testing":
        # 测试环境：SQLite 文件，持久化但轻量
        from langgraph.checkpoint.sqlite import SqliteSaver
        db_path = os.environ.get("SQLITE_PATH", "./test_agent.db")
        print(f"✅ 使用 SqliteSaver（测试模式）→ {db_path}")
        return SqliteSaver(db_path=db_path)

    elif env in ("staging", "production"):
        # 生产环境：PostgreSQL，高可用
        from langgraph.checkpoint.postgres import PostgresSaver
        db_url = os.environ.get("DATABASE_URL")
        if not db_url:
            raise ValueError(
                "生产环境必须设置 DATABASE_URL 环境变量！\n"
                "格式：postgresql://user:password@host:port/dbname"
            )
        print(f"✅ 使用 PostgresSaver（{env}模式）")
        return PostgresSaver(conn_string=db_url)

    else:
        raise ValueError(f"未知环境：{env}")


def create_store(env: str | None = None) -> Any:
    """根据环境创建合适的 Store（长期记忆）"""
    from langchain.embeddings import init_embeddings

    env = env or os.environ.get("ENV", "development")
    embeddings = init_embeddings("openai:text-embedding-3-small")

    if env in ("development", "testing"):
        from langgraph.store.memory import InMemoryStore
        return InMemoryStore(index={"embed": embeddings, "dims": 1536})

    elif env in ("staging", "production"):
        # 生产环境使用 PostgresStore
        from langgraph.store.postgres import PostgresStore
        db_url = os.environ.get("DATABASE_URL")
        return PostgresStore(
            conn_string=db_url,
            index={"embed": embeddings, "dims": 1536},
        )

    raise ValueError(f"未知环境：{env}")


# 使用示例
if __name__ == "__main__":
    # 开发环境
    os.environ["ENV"] = "development"
    checkpointer = create_checkpointer()
    store = create_store()
    print(f"Checkpointer: {type(checkpointer).__name__}")
    print(f"Store: {type(store).__name__}")
```

## 方案 2：PostgreSQL 完整配置

```python
"""
场景 5.2：PostgreSQL 生产级配置
包含连接池、错误处理、健康检查
"""
import os
from langchain.agents import create_agent
from langchain.agents.middleware import SummarizationMiddleware
from langchain.embeddings import init_embeddings
from langchain_core.tools import tool
from langgraph.checkpoint.postgres import PostgresSaver
from langgraph.store.postgres import PostgresStore


# ========== 1. 数据库配置 ==========
DATABASE_URL = os.environ.get(
    "DATABASE_URL",
    "postgresql://agent_user:secure_password@localhost:5432/agent_production"
)


# ========== 2. 创建 Checkpointer（带连接池建议） ==========
def create_production_checkpointer():
    """创建生产级 PostgresSaver"""
    return PostgresSaver(
        conn_string=DATABASE_URL,
        # 连接池配置建议（取决于具体实现版本）：
        # pool_size=20,          # 连接池大小 = Worker 数 × 2
        # max_overflow=10,       # 最大溢出连接
        # pool_timeout=30,       # 获取连接超时（秒）
        # pool_recycle=3600,     # 连接回收时间（秒），防止过期
    )


# ========== 3. 创建 Store（长期记忆） ==========
def create_production_store():
    """创建生产级 PostgresStore"""
    embeddings = init_embeddings("openai:text-embedding-3-small")
    return PostgresStore(
        conn_string=DATABASE_URL,
        index={"embed": embeddings, "dims": 1536},
    )


# ========== 4. 工具 ==========
@tool
def search_docs(query: str) -> str:
    """搜索文档知识库（模拟）"""
    return f"关于「{query}」的检索结果：这是一段相关的技术文档内容..."


# ========== 5. 组装生产 Agent ==========
def create_production_agent():
    """创建生产级 Agent"""
    checkpointer = create_production_checkpointer()
    store = create_production_store()

    agent = create_agent(
        model="gpt-4.1",
        tools=[search_docs],
        system_prompt=(
            "你是一个企业级智能助手。"
            "请根据用户的问题，搜索知识库并给出准确的回答。"
        ),
        middleware=[
            SummarizationMiddleware(
                model="gpt-4.1-mini",
                trigger=("tokens", 6000),
                keep=("messages", 30),
            ),
        ],
        checkpointer=checkpointer,
        store=store,
    )
    return agent


# ========== 6. FastAPI 集成示例 ==========
"""
# app.py - FastAPI 集成
from fastapi import FastAPI, Request
from pydantic import BaseModel

app = FastAPI()
agent = create_production_agent()

class ChatRequest(BaseModel):
    message: str
    user_id: str
    session_id: str

class ChatResponse(BaseModel):
    reply: str
    session_id: str

@app.post("/chat", response_model=ChatResponse)
async def chat(req: ChatRequest):
    config = {
        "configurable": {
            "thread_id": f"{req.user_id}-{req.session_id}",
            "user_id": req.user_id,
        }
    }

    response = agent.invoke(
        {"messages": [{"role": "user", "content": req.message}]},
        config,
    )

    return ChatResponse(
        reply=response["messages"][-1].content,
        session_id=req.session_id,
    )

# 启动：uvicorn app:app --host 0.0.0.0 --port 8000 --workers 4
# 4 个 Worker 进程共享同一个 PostgreSQL → 记忆一致 ✓
"""
```

### 多 Worker 部署示意

```
                        用户请求
                           │
                    ┌──────┴──────┐
                    │  Nginx/LB   │
                    └──────┬──────┘
                           │
              ┌────────────┼────────────┐
              ↓            ↓            ↓
        ┌──────────┐ ┌──────────┐ ┌──────────┐
        │ Worker 1 │ │ Worker 2 │ │ Worker 3 │
        │ Agent    │ │ Agent    │ │ Agent    │
        └────┬─────┘ └────┬─────┘ └────┬─────┘
             │            │            │
             └────────────┼────────────┘
                          │
                   ┌──────┴──────┐
                   │ PostgreSQL  │
                   │             │
                   │ checkpoints │  ← 所有 Worker 共享
                   │ store_items │  ← 记忆一致
                   └─────────────┘

  InMemorySaver 的问题：
    Worker 1 的记忆 ≠ Worker 2 的记忆
    用户第一次请求到 Worker 1，第二次到 Worker 2
    → 记忆不一致！

  PostgresSaver 的解决：
    所有 Worker 读写同一个数据库
    → 记忆完全一致 ✓
```

## 方案 3：定时清理和监控

```python
"""
场景 5.3：生产环境的记忆维护
定时清理过期数据 + 使用量监控
"""
import asyncio
from datetime import datetime, timedelta


async def cleanup_expired_threads(
    checkpointer,
    max_age_days: int = 30,
    dry_run: bool = True,
):
    """清理超过 N 天未活跃的对话线程

    Args:
        checkpointer: PostgresSaver 实例
        max_age_days: 最大保留天数
        dry_run: 如果 True，只统计不删除
    """
    cutoff_date = datetime.now() - timedelta(days=max_age_days)

    # 使用 SQL 查询过期记录
    # 注意：实际 SQL 取决于 LangGraph 的表结构
    query = """
        SELECT thread_id, MAX(created_at) as last_active
        FROM checkpoints
        GROUP BY thread_id
        HAVING MAX(created_at) < %s
    """

    async with checkpointer.conn.cursor() as cur:
        await cur.execute(query, (cutoff_date,))
        expired_threads = await cur.fetchall()

    print(f"找到 {len(expired_threads)} 个过期线程（{max_age_days} 天未活跃）")

    if dry_run:
        print("（DRY RUN 模式，不执行删除）")
        for thread_id, last_active in expired_threads[:5]:
            print(f"  - {thread_id}: 最后活跃 {last_active}")
        return

    # 执行删除
    delete_query = """
        DELETE FROM checkpoints
        WHERE thread_id IN (
            SELECT thread_id FROM checkpoints
            GROUP BY thread_id
            HAVING MAX(created_at) < %s
        )
    """
    async with checkpointer.conn.cursor() as cur:
        await cur.execute(delete_query, (cutoff_date,))
        deleted = cur.rowcount
        await checkpointer.conn.commit()

    print(f"✅ 已清理 {deleted} 条过期记录")


async def get_memory_stats(checkpointer):
    """获取记忆系统的使用统计"""
    stats_queries = {
        "总线程数": "SELECT COUNT(DISTINCT thread_id) FROM checkpoints",
        "总检查点数": "SELECT COUNT(*) FROM checkpoints",
        "最早记录": "SELECT MIN(created_at) FROM checkpoints",
        "最新记录": "SELECT MAX(created_at) FROM checkpoints",
    }

    print("📊 记忆系统统计：")
    async with checkpointer.conn.cursor() as cur:
        for name, query in stats_queries.items():
            try:
                await cur.execute(query)
                result = await cur.fetchone()
                print(f"  {name}: {result[0]}")
            except Exception as e:
                print(f"  {name}: 查询失败 ({e})")


# 定时任务示例
"""
# 使用 APScheduler 或 Celery 定时执行

from apscheduler.schedulers.asyncio import AsyncIOScheduler

scheduler = AsyncIOScheduler()

# 每天凌晨 3 点清理过期记忆
scheduler.add_job(
    cleanup_expired_threads,
    'cron',
    hour=3,
    args=[checkpointer],
    kwargs={'max_age_days': 30, 'dry_run': False},
)

# 每小时输出一次统计
scheduler.add_job(
    get_memory_stats,
    'interval',
    hours=1,
    args=[checkpointer],
)

scheduler.start()
"""
```

## 方案 4：Redis 高性能方案

```python
"""
场景 5.4：Redis 高性能持久化（适合实时对话场景）
"""
import os
from langchain.agents import create_agent
from langgraph.checkpoint.redis import RedisSaver


# ========== Redis 配置 ==========
REDIS_URL = os.environ.get(
    "REDIS_URL",
    "redis://:password@localhost:6379/0"
)

checkpointer = RedisSaver(
    url=REDIS_URL,
    # ttl=86400,  # 对话记忆 24 小时后自动过期
    #              适合客服场景（会话结束自动清理）
)

agent = create_agent(
    model="gpt-4.1-mini",
    tools=[],
    system_prompt="你是一个实时客服助手。",
    checkpointer=checkpointer,
)


# ========== Redis 持久化配置说明 ==========
"""
# redis.conf 关键配置

# ===== RDB 快照（定时备份） =====
save 900 1       # 900秒内有1次写入 → 快照
save 300 10      # 300秒内有10次写入 → 快照
save 60 10000    # 60秒内有10000次写入 → 快照

# ===== AOF 日志（实时备份） =====
appendonly yes
appendfsync everysec   # 每秒同步，平衡性能和安全

# ===== 内存管理 =====
maxmemory 2gb
maxmemory-policy allkeys-lru   # 满了删最不常用的

# ===== 推荐策略 =====
# 实时对话（短期）：TTL=24h + RDB 快照即可
# 重要数据（长期）：AOF + RDB 双重备份
# 高性能优先：只用 RDB，接受最多几分钟的数据丢失
"""
```

## 四种后端对比选型

```
┌─────────────┬──────────┬──────────┬──────────┬──────────┐
│             │ InMemory │ SQLite   │Postgres  │ Redis    │
├─────────────┼──────────┼──────────┼──────────┼──────────┤
│ 持久化      │ ❌       │ ✅ 文件   │ ✅ ACID  │ ⚠️ 需配置 │
│ 多进程      │ ❌       │ ⚠️ 受限   │ ✅       │ ✅       │
│ 延迟        │ ~0ms     │ ~1ms     │ ~5ms     │ ~0.5ms   │
│ 容量        │ 内存     │ 磁盘     │ 磁盘     │ 内存     │
│ TTL 自动过期│ ❌       │ ❌       │ 需实现   │ ✅ 原生   │
│ 部署难度    │ ⭐       │ ⭐       │ ⭐⭐⭐    │ ⭐⭐     │
│ 适用场景    │ 开发     │ 个人项目 │ 正式产品 │ 高性能   │
├─────────────┼──────────┼──────────┼──────────┼──────────┤
│ 推荐使用    │ 原型     │ 内部工具 │ 首选     │ 实时     │
└─────────────┴──────────┴──────────┴──────────┴──────────┘

选型决策：
  不知道选什么 → PostgreSQL（最安全的选择）
  需要极低延迟 → Redis
  个人/小团队 → SQLite
  只是试试    → InMemory
```

## Docker Compose 一键部署

```yaml
# docker-compose.yml
# 一键启动 Agent 服务 + PostgreSQL + Redis

version: '3.8'

services:
  # Agent API 服务
  agent-api:
    build: .
    ports:
      - "8000:8000"
    environment:
      - ENV=production
      - DATABASE_URL=postgresql://agent:secret@postgres:5432/agent_db
      - REDIS_URL=redis://redis:6379/0
      - OPENAI_API_KEY=${OPENAI_API_KEY}
    depends_on:
      - postgres
      - redis
    deploy:
      replicas: 3  # 3 个 Worker
    command: uvicorn app:app --host 0.0.0.0 --port 8000

  # PostgreSQL（持久化存储）
  postgres:
    image: postgres:16
    environment:
      POSTGRES_USER: agent
      POSTGRES_PASSWORD: secret
      POSTGRES_DB: agent_db
    volumes:
      - pgdata:/var/lib/postgresql/data
    ports:
      - "5432:5432"

  # Redis（高性能缓存，可选）
  redis:
    image: redis:7
    command: redis-server --appendonly yes
    volumes:
      - redisdata:/data
    ports:
      - "6379:6379"

volumes:
  pgdata:
  redisdata:
```

```bash
# 启动
docker-compose up -d

# 查看状态
docker-compose ps

# 查看日志
docker-compose logs -f agent-api
```

## 生产环境检查清单

```
部署前检查：

  □ 数据库连接
    □ DATABASE_URL 环境变量已设置
    □ 数据库可访问（telnet host port）
    □ 用户权限正确（CREATE TABLE, INSERT, SELECT, DELETE）

  □ 连接池
    □ pool_size 设置合理（Worker数 × 2）
    □ max_overflow 不超过数据库 max_connections

  □ 记忆管理
    □ SummarizationMiddleware 已配置
    □ Token 阈值设置合理（模型上下文的 30-50%）
    □ 定时清理任务已部署

  □ 监控
    □ 数据库连接数监控
    □ 记忆表大小监控
    □ API 延迟监控

  □ 安全
    □ 数据库密码不在代码中硬编码
    □ API 有认证（不允许未授权访问记忆）
    □ thread_id 包含 user_id（防止越权访问）

  □ 备份
    □ PostgreSQL 定时 pg_dump
    □ 或配置流复制（主从）
```
