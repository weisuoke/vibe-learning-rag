### 2. 文档加载器

```python
class DocumentLoader:
    """文档加载器：支持 TXT, PDF, Markdown"""

    @staticmethod
    def load_txt(file_path: str) -> str:
        """加载 TXT 文件"""
        with open(file_path, 'r', encoding='utf-8') as f:
            return f.read()

    @staticmethod
    def load_pdf(file_path: str) -> str:
        """加载 PDF 文件"""
        text = ""
        with open(file_path, 'rb') as f:
            pdf_reader = pypdf.PdfReader(f)
            for page in pdf_reader.pages:
                text += page.extract_text() + "\n"
        return text

    @staticmethod
    def load_markdown(file_path: str) -> str:
        """加载 Markdown 文件"""
        with open(file_path, 'r', encoding='utf-8') as f:
            return f.read()

    @classmethod
    def load(cls, file_path: str) -> str:
        """自动识别文件类型并加载"""
        ext = file_path.lower().split('.')[-1]
        if ext == 'txt':
            return cls.load_txt(file_path)
        elif ext == 'pdf':
            return cls.load_pdf(file_path)
        elif ext in ['md', 'markdown']:
            return cls.load_markdown(file_path)
        else:
            raise ValueError(f"不支持的文件类型: {ext}")
```

### 3. 文本分块器

```python
class TextChunker:
    """文本分块器：将长文本分割成小块"""

    def __init__(self, chunk_size: int = 500, chunk_overlap: int = 50):
        """
        参数:
            chunk_size: 每块的字符数
            chunk_overlap: 块之间的重叠字符数
        """
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap

    def split_text(self, text: str) -> List[str]:
        """分割文本"""
        # 清理文本
        text = re.sub(r'\n+', '\n', text)
        text = re.sub(r'\s+', ' ', text)

        chunks = []
        start = 0

        while start < len(text):
            end = start + self.chunk_size
            chunk = text[start:end]

            # 尝试在句子边界分割
            if end < len(text):
                last_period = chunk.rfind('.')
                last_newline = chunk.rfind('\n')
                split_point = max(last_period, last_newline)

                if split_point > self.chunk_size * 0.5:
                    chunk = chunk[:split_point + 1]
                    end = start + split_point + 1

            chunks.append(chunk.strip())
            start = end - self.chunk_overlap

        return [c for c in chunks if len(c) > 50]  # 过滤太短的块
```

### 4. RAG 系统核心类

```python
class RAGSystem:
    """RAG 系统核心类"""

    def __init__(self):
        self.collection_name = COLLECTION_NAME
        self._create_collection()

    def _create_collection(self):
        """创建 Collection"""
        if milvus_client.has_collection(self.collection_name):
            milvus_client.drop_collection(self.collection_name)

        schema = milvus_client.create_schema()

        schema.add_field("id", DataType.INT64, is_primary=True, auto_id=True)
        schema.add_field("text", DataType.VARCHAR, max_length=2000)
        schema.add_field("source", DataType.VARCHAR, max_length=500)
        schema.add_field("dense", DataType.FLOAT_VECTOR, dim=1536)

        # 配置 Embedding Function
        schema.add_function(
            Function(
                name="openai_embedding",
                function_type=FunctionType.TEXTEMBEDDING,
                input_field_names=["text"],
                output_field_names=["dense"],
                params={
                    "provider": "openai",
                    "model_name": "text-embedding-3-small",
                    "api_key": OPENAI_API_KEY,
                    "base_url": OPENAI_BASE_URL
                }
            )
        )

        index_params = milvus_client.prepare_index_params()
        index_params.add_index(
            field_name="dense",
            index_type="AUTOINDEX",
            metric_type="COSINE"
        )

        milvus_client.create_collection(
            collection_name=self.collection_name,
            schema=schema,
            index_params=index_params
        )

        print(f"✅ Collection 创建成功: {self.collection_name}")

    def ingest_document(self, file_path: str, chunk_size: int = 500):
        """
        导入文档到 RAG 系统

        参数:
            file_path: 文档路径
            chunk_size: 分块大小
        """
        print(f"\n【文档导入】{file_path}")

        # 1. 加载文档
        print("  步骤 1: 加载文档...")
        text = DocumentLoader.load(file_path)
        print(f"  ✅ 文档长度: {len(text)} 字符")

        # 2. 文本分块
        print("  步骤 2: 文本分块...")
        chunker = TextChunker(chunk_size=chunk_size, chunk_overlap=50)
        chunks = chunker.split_text(text)
        print(f"  ✅ 分块数量: {len(chunks)}")

        # 3. 插入 Milvus（自动向量化）
        print("  步骤 3: 插入 Milvus...")
        documents = [
            {"text": chunk, "source": file_path}
            for chunk in chunks
        ]

        result = milvus_client.insert(self.collection_name, documents)
        print(f"  ✅ 插入成功: {result['insert_count']} 条")

        return result

    def search(self, query: str, top_k: int = 3) -> List[Dict]:
        """
        语义检索

        参数:
            query: 查询文本
            top_k: 返回前 K 个结果

        返回:
            检索结果列表
        """
        results = milvus_client.search(
            collection_name=self.collection_name,
            data=[query],
            anns_field="dense",
            limit=top_k,
            output_fields=["text", "source"]
        )

        return [
            {
                "text": hit['entity']['text'],
                "source": hit['entity']['source'],
                "score": hit['distance']
            }
            for hit in results[0]
        ]

    def answer_question(self, question: str, top_k: int = 3) -> Dict:
        """
        RAG 问答

        参数:
            question: 用户问题
            top_k: 检索文档数量

        返回:
            包含答案和来源的字典
        """
        print(f"\n💬 问题: {question}")

        # 1. 检索相关文档
        print(f"  步骤 1: 检索 Top-{top_k} 文档...")
        search_results = self.search(question, top_k=top_k)

        print(f"  ✅ 检索完成")
        for i, result in enumerate(search_results, 1):
            print(f"    {i}. 相似度: {result['score']:.4f}")
            print(f"       文本: {result['text'][:60]}...")

        # 2. 构建上下文
        contexts = [r['text'] for r in search_results]
        context_text = "\n\n".join(contexts)

        # 3. 构建 Prompt
        prompt = f"""基于以下上下文回答问题。如果上下文中没有相关信息，请说"我不知道"。

上下文:
{context_text}

问题: {question}

答案:"""

        # 4. 调用 LLM 生成答案
        print("  步骤 2: LLM 生成答案...")
        response = openai_client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[
                {"role": "system", "content": "你是一个专业的技术助手，基于提供的上下文回答问题。"},
                {"role": "user", "content": prompt}
            ],
            temperature=0.3,
            max_tokens=500
        )

        answer = response.choices[0].message.content

        print(f"  ✅ 答案生成完成")

        return {
            "question": question,
            "answer": answer,
            "sources": [r['source'] for r in search_results],
            "contexts": contexts
        }
```

def main():
    """主函数：完整的 RAG 应用演示"""

    print("=" * 60)
    print("端到端 RAG 应用演示")
    print("=" * 60)

    # 初始化 RAG 系统
    rag = RAGSystem()

    # 示例 1: 导入文档
    print("\n【示例 1: 导入文档】")

    # 创建示例文档
    sample_doc = """
# Milvus Vector Database

Milvus is an open-source vector database built for AI applications.
It provides high-performance vector similarity search and supports
billion-scale vector data.

## Key Features

1. Embedding Functions: Milvus 2.6 introduces automatic vectorization.
   You can insert raw text directly, and Milvus calls embedding APIs
   automatically.

2. Hybrid Search: Combine dense vectors (semantic understanding) and
   sparse vectors (keyword matching) for better retrieval quality.

3. Scalability: Milvus supports billion-scale vector data with
   distributed architecture.

## Use Cases

- Document Q&A systems
- Semantic search engines
- Recommendation systems
- RAG (Retrieval-Augmented Generation) applications
"""

    # 保存示例文档
    with open("sample_doc.txt", "w") as f:
        f.write(sample_doc)

    # 导入文档
    rag.ingest_document("sample_doc.txt", chunk_size=300)

    # 示例 2: 问答测试
    print("\n【示例 2: 问答测试】")

    questions = [
        "What is Milvus?",
        "What are the key features of Milvus 2.6?",
        "What is hybrid search?"
    ]

    for question in questions:
        result = rag.answer_question(question, top_k=3)

        print(f"\n✅ 答案:")
        print(result['answer'])
        print(f"\n📚 来源: {', '.join(set(result['sources']))}")
        print("-" * 60)

    # 清理
    os.remove("sample_doc.txt")

    print("\n✅ 演示完成！")

if __name__ == "__main__":
    main()

def safe_answer(rag: RAGSystem, question: str) -> Dict:
    """带错误处理的问答"""
    try:
        return rag.answer_question(question)
    except Exception as e:
        return {
            "question": question,
            "answer": f"抱歉，处理问题时出错: {str(e)}",
            "sources": [],
            "contexts": []
        }

## 扩展功能

### 1. 对话式 RAG

```python
class ConversationalRAG(RAGSystem):
    """对话式 RAG：支持多轮对话"""

    def __init__(self):
        super().__init__()
        self.history = []

    def chat(self, question: str) -> str:
        """多轮对话"""
        # 结合历史上下文
        context = "\n".join([f"Q: {h['q']}\nA: {h['a']}" for h in self.history[-3:]])
        full_question = f"{context}\n\nQ: {question}" if context else question

        result = self.answer_question(full_question)

        # 保存历史
        self.history.append({"q": question, "a": result['answer']})

        return result['answer']
```
