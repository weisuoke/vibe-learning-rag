# 实战代码 - 场景3：状态迁移与版本管理

## 场景说明

本场景演示生产系统中最常见的痛点——Schema 变更后如何保持旧检查点可用。你将学会：

1. 用版本号标记 Schema，让新旧状态可区分
2. 添加新字段时保持向后兼容（Optional + 默认值）
3. 编写迁移函数处理不兼容变更（字段重命名、类型变更）
4. 在节点入口处自动迁移旧状态

核心思路：把状态当成数据库表——加列容易，改列名要写迁移脚本。LangGraph 内部的 `_migrate_checkpoint()` 就是这个模式（v1→v2→v3→v4 逐级迁移）。

[来源: reference/source_state_management_01.md | 检查点版本迁移机制]
[来源: reference/context7_langgraph_01.md | LangGraph 官方文档]

## 完整代码

```python
"""
状态迁移与版本管理实战
演示：Schema 版本化、向后兼容添加、迁移函数、旧检查点恢复

核心知识点：
- _schema_version 字段标记状态版本
- Optional + .get() 实现向后兼容
- 迁移函数链：v1 → v2 → v3 逐级升级
- 节点入口处自动检测并迁移旧状态

运行环境：Python 3.13+, langgraph
安装依赖：uv add langgraph
"""

from typing import Annotated, Optional
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langgraph.checkpoint.memory import MemorySaver


# ============================================================
# 1. V1 Schema —— 初始版本（最简单的 RAG 状态）
# ============================================================

class StateV1(TypedDict):
    """V1: 只有消息和查询"""
    messages: Annotated[list, add_messages]
    query: str


# ============================================================
# 2. V2 Schema —— 向后兼容添加（新增 language 字段）
# ============================================================

class StateV2(TypedDict):
    """V2: 新增 language 字段，支持多语言"""
    messages: Annotated[list, add_messages]
    query: str
    language: str           # 新增：用户语言偏好
    _schema_version: int    # 新增：版本标记


# ============================================================
# 3. V3 Schema —— 不兼容变更（query 重命名为 question）
# ============================================================

class StateV3(TypedDict):
    """V3: query 重命名为 question，新增 source"""
    messages: Annotated[list, add_messages]
    question: str           # 重命名：query → question
    language: str
    source: str             # 新增：查询来源
    _schema_version: int


# ============================================================
# 4. 迁移函数 —— 逐级升级，像 LangGraph 内部一样
# ============================================================

def migrate_v1_to_v2(state: dict) -> dict:
    """V1 → V2：添加 language 默认值"""
    state["language"] = state.get("language", "zh")
    state["_schema_version"] = 2
    return state


def migrate_v2_to_v3(state: dict) -> dict:
    """V2 → V3：query 重命名为 question，添加 source"""
    if "query" in state and "question" not in state:
        state["question"] = state.pop("query")
    state["source"] = state.get("source", "unknown")
    state["_schema_version"] = 3
    return state


def migrate_to_latest(state: dict) -> dict:
    """自动检测版本并迁移到最新"""
    version = state.get("_schema_version", 1)

    if version < 2:
        state = migrate_v1_to_v2(state)
    if version < 3:
        state = migrate_v2_to_v3(state)

    return state


# ============================================================
# 5. 构建 V1 图并创建检查点
# ============================================================

def build_v1_graph():
    """构建 V1 版本的图"""
    def process_v1(state: StateV1) -> dict:
        return {"messages": [f"[V1] 处理查询: {state['query']}"]}

    builder = StateGraph(StateV1)
    builder.add_node("process", process_v1)
    builder.add_edge(START, "process")
    builder.add_edge("process", END)
    return builder


# ============================================================
# 6. 构建 V3 图（兼容旧状态）
# ============================================================

def build_v3_graph():
    """构建 V3 版本的图，节点内部处理旧状态兼容"""
    def process_v3(state: StateV3) -> dict:
        # 安全获取字段 —— 旧检查点可能缺少这些字段
        question = state.get("question", state.get("query", ""))
        language = state.get("language", "zh")
        source = state.get("source", "unknown")

        msg = f"[V3] question={question}, lang={language}, source={source}"
        return {
            "messages": [msg],
            "_schema_version": 3,
        }

    builder = StateGraph(StateV3)
    builder.add_node("process", process_v3)
    builder.add_edge(START, "process")
    builder.add_edge("process", END)
    return builder


# ============================================================
# 7. 演示完整迁移过程
# ============================================================

checkpointer = MemorySaver()
config_v1 = {"configurable": {"thread_id": "migration-demo"}}

# ----- 阶段 1：用 V1 Schema 运行并保存检查点 -----
print("=" * 55)
print("  阶段 1：V1 Schema 运行")
print("=" * 55)

graph_v1 = build_v1_graph().compile(checkpointer=checkpointer)
result_v1 = graph_v1.invoke(
    {"query": "什么是RAG？", "messages": []},
    config_v1,
)
print(f"V1 结果: {result_v1['messages'][-1].content}")
print(f"V1 状态字段: {[k for k in result_v1 if not k.startswith('_')]}")

# ----- 阶段 2：模拟从旧检查点恢复并迁移 -----
print()
print("=" * 55)
print("  阶段 2：迁移旧状态到 V3")
print("=" * 55)

# 模拟从检查点恢复的 V1 状态
old_state = {
    "query": "什么是RAG？",
    "messages": result_v1["messages"],
}

print(f"\n迁移前 (V1):")
print(f"  字段: {list(old_state.keys())}")
print(f"  版本: v{old_state.get('_schema_version', 1)}")

# 执行迁移链
migrated = migrate_to_latest(old_state.copy())

print(f"\n迁移后 (V3):")
print(f"  字段: {list(migrated.keys())}")
print(f"  版本: v{migrated['_schema_version']}")
print(f"  question: {migrated.get('question')}")
print(f"  language: {migrated.get('language')}")
print(f"  source: {migrated.get('source')}")
print(f"  query 已移除: {'query' not in migrated}")

# ----- 阶段 3：用 V3 Schema 运行迁移后的状态 -----
print()
print("=" * 55)
print("  阶段 3：V3 Schema 运行（使用迁移后的状态）")
print("=" * 55)

config_v3 = {"configurable": {"thread_id": "migration-v3"}}
graph_v3 = build_v3_graph().compile(checkpointer=checkpointer)

result_v3 = graph_v3.invoke(
    {
        "question": migrated["question"],
        "language": migrated["language"],
        "source": "migrated_from_v1",
        "messages": [],
        "_schema_version": 3,
    },
    config_v3,
)
print(f"V3 结果: {result_v3['messages'][-1].content}")

# ----- 阶段 4：验证 V3 节点直接处理 V1 风格输入 -----
print()
print("=" * 55)
print("  阶段 4：V3 节点兼容性验证")
print("=" * 55)

config_compat = {"configurable": {"thread_id": "compat-test"}}
# 故意传入缺少 V3 字段的输入，验证 .get() 兜底
result_compat = graph_v3.invoke(
    {
        "question": "Embedding是什么？",
        "messages": [],
        # 故意不传 language、source、_schema_version
    },
    config_compat,
)
print(f"兼容结果: {result_compat['messages'][-1].content}")
print(f"  → 缺失字段自动使用默认值，不会崩溃")


# ============================================================
# 8. 迁移策略总结
# ============================================================

print()
print("=" * 55)
print("  迁移策略速查表")
print("=" * 55)

print("""
变更类型          安全等级    做法
─────────────────────────────────────────────────
添加字段          安全       .get(field, default) 兜底
删除字段          中等       三步走：停写 → 停读 → 删定义
重命名字段        危险       迁移函数 + 过渡期双字段共存
修改字段类型      危险       迁移函数 + 类型检测转换

核心原则：
  1. 用 _schema_version 标记版本
  2. 迁移函数逐级升级（v1→v2→v3）
  3. 节点内用 .get() 防御性读取
  4. 永远不要假设字段一定存在
""")
```

## 关键知识点解析

### 1. 迁移函数链 vs 直接转换

```python
# 推荐：逐级迁移（每步变更小，容易测试）
v1 → migrate_v1_to_v2() → v2 → migrate_v2_to_v3() → v3

# 不推荐：直接跳跃（逻辑复杂，容易遗漏）
v1 → migrate_v1_to_v3() → v3
```

LangGraph 内部的 `_migrate_checkpoint()` 就是逐级迁移模式。每个版本只处理一步变更，组合起来覆盖所有历史版本。

### 2. 向后兼容的三板斧

| 手段 | 代码 | 适用场景 |
|------|------|----------|
| `.get()` 默认值 | `state.get("language", "zh")` | 新增字段 |
| `Optional` 类型 | `language: Optional[str]` | 类型提示层面标记可选 |
| `total=False` | `class State(TypedDict, total=False)` | 所有字段都可选 |

### 3. `_schema_version` 的作用

```python
# 没有版本号 → 只能猜测状态是哪个版本
state = {"query": "hello", "messages": []}  # V1? V2?

# 有版本号 → 精确判断，精确迁移
state = {"query": "hello", "messages": [], "_schema_version": 1}  # 明确是 V1
```

## 在 RAG 系统中的应用

| 场景 | 迁移需求 | 策略 |
|------|----------|------|
| 新增 rerank 功能 | 添加 `rerank_scores` 字段 | `.get()` + 默认空列表 |
| documents 从 `list[str]` 改为 `list[dict]` | 类型变更 | 迁移函数 + `isinstance` 检测 |
| 废弃 `debug_info` 字段 | 删除字段 | 三步删除法 |
| 支持多语言 | 添加 `language` 字段 | Optional + 默认 "zh" |
