## 示例 1：OpenAI 基础使用（单文本 + 批量）

### 场景说明

最基础的 Embedding 使用场景：嵌入单个查询和批量嵌入文档。这是所有 RAG 应用的起点。

### 完整代码

```python
"""
示例 1：OpenAI Embeddings 基础使用
演示单文本和批量文本的嵌入操作
"""

import os
from dotenv import load_dotenv
from langchain_openai import OpenAIEmbeddings

# 加载环境变量
load_dotenv()

def basic_embedding_demo():
    """基础 Embedding 演示"""

    # 1. 初始化 Embeddings 模型
    embeddings = OpenAIEmbeddings(
        model="text-embedding-3-small",  # 使用 small 模型（性价比高）
        # model="text-embedding-3-large",  # 或使用 large 模型（性能更好）
    )

    print("=" * 60)
    print("示例 1：OpenAI Embeddings 基础使用")
    print("=" * 60)

    # 2. 嵌入单个查询（用于检索场景）
    query = "什么是向量数据库？"
    query_vector = embeddings.embed_query(query)

    print(f"\n【单文本嵌入】")
    print(f"查询文本: {query}")
    print(f"向量维度: {len(query_vector)}")
    print(f"向量前 10 个值: {query_vector[:10]}")

    # 3. 批量嵌入文档（用于索引构建）
    documents = [
        "向量数据库是专门用于存储和检索向量数据的数据库系统。",
        "常见的向量数据库包括 Milvus、Pinecone、Weaviate 等。",
        "向量数据库支持高效的相似度搜索和语义检索。",
        "RAG 系统通常使用向量数据库来存储文档的 Embedding。"
    ]

    doc_vectors = embeddings.embed_documents(documents)

    print(f"\n【批量文档嵌入】")
    print(f"文档数量: {len(documents)}")
    print(f"生成向量数: {len(doc_vectors)}")
    print(f"每个向量维度: {len(doc_vectors[0])}")

    # 4. 验证向量维度一致性
    assert len(query_vector) == len(doc_vectors[0]), "查询和文档向量维度不一致！"
    print(f"\n✅ 向量维度一致性检查通过")

    # 5. 计算简单的余弦相似度（演示用）
    import numpy as np

    def cosine_similarity(vec1, vec2):
        """计算余弦相似度"""
        vec1 = np.array(vec1)
        vec2 = np.array(vec2)
        return np.dot(vec1, vec2) / (np.linalg.norm(vec1) * np.linalg.norm(vec2))

    print(f"\n【相似度计算】")
    for i, doc in enumerate(documents):
        similarity = cosine_similarity(query_vector, doc_vectors[i])
        print(f"文档 {i+1} 相似度: {similarity:.4f} - {doc[:30]}...")

if __name__ == "__main__":
    basic_embedding_demo()
```

### 关键点说明

1. **方法分离设计**：
   - `embed_query()` - 单个查询嵌入（检索时使用）
   - `embed_documents()` - 批量文档嵌入（索引构建时使用）
   - 虽然底层可能相同，但接口分离提供了语义清晰性

2. **模型选择**：
   - `text-embedding-3-small` (1536 维) - 性价比高，适合大多数场景
   - `text-embedding-3-large` (3072 维) - 性能最好，成本较高

3. **向量维度**：
   - 默认维度：small=1536, large=3072
   - 可通过 `dimensions` 参数降低维度以节省成本

4. **API 调用次数**：
   - 单文本：1 次 API 调用
   - 批量文档：1 次 API 调用（内部自动批处理）

## 示例 2：批量处理优化

### 场景说明

处理大量文档时，需要控制批量大小和并发数，避免触发 API 速率限制，同时提升处理效率。

### 完整代码

```python
"""
示例 2：批量处理优化
演示如何高效处理大量文档的嵌入
"""

import os
import time
from dotenv import load_dotenv
from langchain_openai import OpenAIEmbeddings

load_dotenv()

def batch_processing_demo():
    """批量处理优化演示"""

    print("=" * 60)
    print("示例 2：批量处理优化")
    print("=" * 60)

    # 1. 创建测试数据（模拟大量文档）
    documents = [
        f"这是第 {i+1} 个测试文档，用于演示批量处理。内容包括向量数据库、RAG 系统、语义检索等主题。"
        for i in range(50)  # 50 个文档
    ]

    print(f"\n【测试数据】")
    print(f"文档总数: {len(documents)}")

    # 2. 默认批量处理（chunk_size=1000）
    print(f"\n【方案 1：默认批量大小】")
    embeddings_default = OpenAIEmbeddings(
        model="text-embedding-3-small",
        chunk_size=1000,  # 默认值
    )

    start_time = time.time()
    vectors_default = embeddings_default.embed_documents(documents)
    time_default = time.time() - start_time

    print(f"处理时间: {time_default:.2f} 秒")
    print(f"生成向量数: {len(vectors_default)}")
    print(f"平均速度: {len(documents)/time_default:.2f} 文档/秒")

    # 3. 小批量处理（适合速率限制严格的场景）
    print(f"\n【方案 2：小批量处理】")
    embeddings_small = OpenAIEmbeddings(
        model="text-embedding-3-small",
        chunk_size=10,  # 每批 10 个文档
        show_progress_bar=True,  # 显示进度条
    )

    start_time = time.time()
    vectors_small = embeddings_small.embed_documents(documents)
    time_small = time.time() - start_time

    print(f"处理时间: {time_small:.2f} 秒")
    print(f"生成向量数: {len(vectors_small)}")
    print(f"平均速度: {len(documents)/time_small:.2f} 文档/秒")

    # 4. 对比结果
    print(f"\n【性能对比】")
    print(f"默认批量: {time_default:.2f} 秒")
    print(f"小批量: {time_small:.2f} 秒")
    print(f"速度差异: {(time_small/time_default - 1)*100:.1f}%")

    # 5. 验证结果一致性
    assert len(vectors_default) == len(vectors_small), "向量数量不一致！"
    print(f"\n✅ 批量处理结果一致性检查通过")

if __name__ == "__main__":
    batch_processing_demo()
```

### 关键点说明

1. **chunk_size 参数**：
   - 控制每批发送到 API 的文档数量
   - 默认 1000，适合大多数场景
   - 速率限制严格时可降低到 10-100

2. **性能权衡**：
   - 大批量：API 调用次数少，速度快，但可能触发速率限制
   - 小批量：API 调用次数多，速度慢，但更稳定

3. **进度条**：
   - `show_progress_bar=True` 可显示处理进度

4. **错误处理**：
   - LangChain 内置重试机制（max_retries=2）
   - 自动处理临时网络错误和速率限制

## 示例 3：添加缓存（CacheBackedEmbeddings）

### 场景说明

在开发和测试阶段，经常需要重复嵌入相同的文档。使用缓存可以避免重复的 API 调用，节省成本和时间。

### 完整代码

```python
"""
示例 3：使用缓存优化 Embedding
演示如何使用 CacheBackedEmbeddings 减少 API 调用
"""

import os
import time
from dotenv import load_dotenv
from langchain_openai import OpenAIEmbeddings
from langchain.embeddings import CacheBackedEmbeddings
from langchain.storage import LocalFileStore

load_dotenv()

def cache_embedding_demo():
    """缓存 Embedding 演示"""

    print("=" * 60)
    print("示例 3：使用缓存优化 Embedding")
    print("=" * 60)

    # 1. 创建测试数据
    documents = [
        "向量数据库是专门用于存储和检索向量数据的数据库系统。",
        "常见的向量数据库包括 Milvus、Pinecone、Weaviate 等。",
        "向量数据库支持高效的相似度搜索和语义检索。",
        "RAG 系统通常使用向量数据库来存储文档的 Embedding。"
    ]

    # 2. 创建底层 Embeddings 模型
    underlying_embeddings = OpenAIEmbeddings(
        model="text-embedding-3-small"
    )

    # 3. 创建本地文件存储（缓存位置）
    store = LocalFileStore("./embedding_cache/")

    # 4. 创建缓存包装器
    cached_embeddings = CacheBackedEmbeddings.from_bytes_store(
        underlying_embeddings,
        store,
        namespace="openai-3-small"  # 命名空间，区分不同模型
    )

    # 5. 第一次嵌入（会调用 API）
    print(f"\n【第一次嵌入 - 无缓存】")
    start_time = time.time()
    vectors_1 = cached_embeddings.embed_documents(documents)
    time_1 = time.time() - start_time

    print(f"处理时间: {time_1:.3f} 秒")
    print(f"生成向量数: {len(vectors_1)}")
    print(f"向量维度: {len(vectors_1[0])}")

    # 6. 第二次嵌入（使用缓存）
    print(f"\n【第二次嵌入 - 使用缓存】")
    start_time = time.time()
    vectors_2 = cached_embeddings.embed_documents(documents)
    time_2 = time.time() - start_time

    print(f"处理时间: {time_2:.3f} 秒")
    print(f"生成向量数: {len(vectors_2)}")
    print(f"向量维度: {len(vectors_2[0])}")

    # 7. 性能对比
    print(f"\n【性能对比】")
    print(f"第一次（API 调用）: {time_1:.3f} 秒")
    print(f"第二次（缓存读取）: {time_2:.3f} 秒")
    print(f"速度提升: {(time_1/time_2):.1f}x")
    print(f"时间节省: {((time_1-time_2)/time_1*100):.1f}%")

    # 8. 验证结果一致性
    import numpy as np
    for i in range(len(vectors_1)):
        assert np.allclose(vectors_1[i], vectors_2[i]), f"向量 {i} 不一致！"

    print(f"\n✅ 缓存结果一致性检查通过")

    # 9. 查看缓存文件
    import os
    cache_dir = "./embedding_cache/openai-3-small"
    if os.path.exists(cache_dir):
        cache_files = os.listdir(cache_dir)
        print(f"\n【缓存信息】")
        print(f"缓存目录: {cache_dir}")
        print(f"缓存文件数: {len(cache_files)}")

if __name__ == "__main__":
    cache_embedding_demo()
```

### 关键点说明

1. **缓存机制**：
   - 基于文本内容的哈希值作为缓存键
   - 相同文本会命中缓存，避免重复 API 调用
   - 适合开发、测试、增量更新场景

2. **存储后端**：
   - `LocalFileStore` - 本地文件存储（适合单机）
   - 也支持 Redis、数据库等后端（适合分布式）

3. **命名空间**：
   - 用于区分不同模型或配置的缓存
   - 避免不同模型的缓存混淆

4. **性能提升**：
   - 缓存命中时速度提升 100-300 倍
   - 节省 API 调用成本
   - 特别适合重复处理相同文档的场景

## 示例 4：异步批量处理

### 场景说明

在高并发场景下，使用异步 API 可以显著提升吞吐量。LangChain 提供了异步方法支持。

### 完整代码

```python
"""
示例 4：异步批量处理
演示如何使用异步 API 提升并发性能
"""

import os
import time
import asyncio
from dotenv import load_dotenv
from langchain_openai import OpenAIEmbeddings

load_dotenv()

async def async_embedding_demo():
    """异步 Embedding 演示"""

    print("=" * 60)
    print("示例 4：异步批量处理")
    print("=" * 60)

    # 1. 创建测试数据（模拟多个独立任务）
    task_groups = [
        [f"任务组 1 - 文档 {i+1}" for i in range(10)],
        [f"任务组 2 - 文档 {i+1}" for i in range(10)],
        [f"任务组 3 - 文档 {i+1}" for i in range(10)],
        [f"任务组 4 - 文档 {i+1}" for i in range(10)],
    ]

    embeddings = OpenAIEmbeddings(
        model="text-embedding-3-small"
    )

    # 2. 同步处理（串行）
    print(f"\n【方案 1：同步串行处理】")
    start_time = time.time()

    sync_results = []
    for group in task_groups:
        vectors = embeddings.embed_documents(group)
        sync_results.append(vectors)

    time_sync = time.time() - start_time
    print(f"处理时间: {time_sync:.2f} 秒")
    print(f"任务组数: {len(task_groups)}")
    print(f"总文档数: {sum(len(g) for g in task_groups)}")

    # 3. 异步处理（并行）
    print(f"\n【方案 2：异步并行处理】")
    start_time = time.time()

    # 创建异步任务
    async_tasks = [
        embeddings.aembed_documents(group)
        for group in task_groups
    ]

    # 并行执行
    async_results = await asyncio.gather(*async_tasks)

    time_async = time.time() - start_time
    print(f"处理时间: {time_async:.2f} 秒")
    print(f"任务组数: {len(task_groups)}")
    print(f"总文档数: {sum(len(g) for g in task_groups)}")

    # 4. 性能对比
    print(f"\n【性能对比】")
    print(f"同步处理: {time_sync:.2f} 秒")
    print(f"异步处理: {time_async:.2f} 秒")
    print(f"速度提升: {(time_sync/time_async):.1f}x")
    print(f"时间节省: {((time_sync-time_async)/time_sync*100):.1f}%")

    # 5. 验证结果一致性
    assert len(sync_results) == len(async_results), "结果数量不一致！"
    for i in range(len(sync_results)):
        assert len(sync_results[i]) == len(async_results[i]), f"任务组 {i} 向量数不一致！"

    print(f"\n✅ 异步处理结果一致性检查通过")

def run_async_demo():
    """运行异步演示（包装器）"""
    asyncio.run(async_embedding_demo())

if __name__ == "__main__":
    run_async_demo()
```

### 关键点说明

1. **异步方法**：
   - `aembed_query()` - 异步单文本嵌入
   - `aembed_documents()` - 异步批量嵌入
   - 与同步方法接口一致，只是返回协程

2. **并发执行**：
   - 使用 `asyncio.gather()` 并行执行多个任务
   - 适合多个独立的嵌入任务
   - 显著提升吞吐量

3. **性能提升**：
   - 并发度越高，提升越明显

   - 受限于 API 速率限制和网络带宽

4. **适用场景**：
   - 高并发 Web 服务
   - 批量文档处理
   - 实时查询响应
