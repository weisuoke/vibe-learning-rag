# 实战代码 - 场景7：生产级状态管理

> 完整可运行的生产级状态管理示例

## 场景描述

在生产环境中，状态管理需要考虑性能、可靠性、监控等多个方面。本场景演示如何构建生产级的状态管理系统。

**应用场景**：
- 高并发多用户系统
- 长时间运行的工作流
- 需要监控和告警的系统
- 需要状态恢复的关键业务

## 核心知识点

1. **状态大小控制**：避免状态无限增长
2. **性能优化**：减少序列化开销
3. **错误处理**：优雅处理失败
4. **监控告警**：实时监控状态
5. **清理策略**：定期清理过期状态

## 完整代码：生产级状态管理系统

```python
from typing import TypedDict, Annotated, Optional
import operator
import time
import logging
from dataclasses import dataclass
from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.postgres import PostgresSaver
from psycopg_pool import ConnectionPool

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# 1. 定义状态 Schema（带大小控制）
def trim_messages(existing: list, new: list) -> list:
    """限制消息数量，只保留最近100条"""
    all_messages = existing + new
    if len(all_messages) > 100:
        logger.warning(f"Trimming messages from {len(all_messages)} to 100")
        return all_messages[-100:]
    return all_messages

def merge_metadata(existing: dict, new: dict) -> dict:
    """合并元数据，限制键数量"""
    merged = {**existing, **new}
    if len(merged) > 50:
        logger.warning(f"Metadata has {len(merged)} keys, trimming to 50")
        # 保留最新的50个键
        keys = list(merged.keys())[-50:]
        return {k: merged[k] for k in keys}
    return merged

class State(TypedDict):
    user_id: str
    session_id: str
    messages: Annotated[list, trim_messages]
    metadata: Annotated[dict, merge_metadata]
    error_count: int
    last_updated: float

# 2. 状态监控类
@dataclass
class StateMetrics:
    """状态指标"""
    total_states: int = 0
    active_states: int = 0
    error_states: int = 0
    avg_message_count: float = 0.0
    avg_metadata_size: float = 0.0

class StateMonitor:
    """状态监控器"""

    def __init__(self):
        self.metrics = StateMetrics()

    def record_state(self, state: State):
        """记录状态指标"""
        self.metrics.total_states += 1

        # 检查是否有错误
        if state.get('error_count', 0) > 0:
            self.metrics.error_states += 1

        # 记录消息数量
        msg_count = len(state.get('messages', []))
        self.metrics.avg_message_count = (
            (self.metrics.avg_message_count * (self.metrics.total_states - 1) + msg_count)
            / self.metrics.total_states
        )

        # 记录元数据大小
        metadata_size = len(state.get('metadata', {}))
        self.metrics.avg_metadata_size = (
            (self.metrics.avg_metadata_size * (self.metrics.total_states - 1) + metadata_size)
            / self.metrics.total_states
        )

        # 检查阈值
        if msg_count > 80:
            logger.warning(f"State {state['session_id']} has {msg_count} messages (threshold: 80)")

        if metadata_size > 40:
            logger.warning(f"State {state['session_id']} has {metadata_size} metadata keys (threshold: 40)")

    def get_metrics(self) -> StateMetrics:
        """获取指标"""
        return self.metrics

    def reset_metrics(self):
        """重置指标"""
        self.metrics = StateMetrics()

# 3. 状态清理器
class StateCleaner:
    """状态清理器"""

    def __init__(self, app, max_age_seconds: int = 3600):
        self.app = app
        self.max_age_seconds = max_age_seconds

    def cleanup_old_states(self, thread_ids: list[str]):
        """清理过期状态"""
        current_time = time.time()
        cleaned_count = 0

        for thread_id in thread_ids:
            try:
                config = {"configurable": {"thread_id": thread_id}}
                state = self.app.get_state(config)

                if state and state.values:
                    last_updated = state.values.get('last_updated', 0)

                    # 如果状态过旧，标记为需要清理
                    if current_time - last_updated > self.max_age_seconds:
                        logger.info(f"Cleaning up old state for {thread_id}")
                        # 注意：实际清理需要数据库支持
                        cleaned_count += 1
            except Exception as e:
                logger.error(f"Error cleaning up {thread_id}: {e}")

        logger.info(f"Cleaned up {cleaned_count} old states")
        return cleaned_count

# 4. 错误处理节点
def safe_process(state: State) -> dict:
    """带错误处理的处理节点"""
    try:
        # 模拟处理逻辑
        user_msg = state['messages'][-1] if state['messages'] else ""
        response = f"Processed: {user_msg}"

        return {
            "messages": [response],
            "metadata": {"last_process_time": time.time()},
            "error_count": 0,  # 重置错误计数
            "last_updated": time.time()
        }
    except Exception as e:
        logger.error(f"Error processing state: {e}")

        # 增加错误计数
        error_count = state.get('error_count', 0) + 1

        # 如果错误次数过多，触发告警
        if error_count >= 3:
            logger.critical(f"State {state['session_id']} has {error_count} errors!")

        return {
            "metadata": {"last_error": str(e), "error_time": time.time()},
            "error_count": error_count,
            "last_updated": time.time()
        }

# 5. 构建生产级图
def create_production_graph(db_url: str):
    """创建生产级图"""

    # 创建连接池
    pool = ConnectionPool(
        db_url,
        min_size=2,
        max_size=10,
        timeout=30
    )

    # 创建 checkpointer
    checkpointer = PostgresSaver(pool)

    # 创建图
    graph = StateGraph(State)
    graph.add_node("process", safe_process)
    graph.add_edge(START, "process")
    graph.add_edge("process", END)

    # 编译
    app = graph.compile(checkpointer=checkpointer)

    return app

# 6. 生产级使用示例
if __name__ == "__main__":
    # 数据库连接
    db_url = "postgresql://user:password@localhost:5432/langgraph"

    # 创建图
    app = create_production_graph(db_url)

    # 创建监控器
    monitor = StateMonitor()

    # 创建清理器
    cleaner = StateCleaner(app, max_age_seconds=3600)

    # 模拟多用户使用
    user_ids = [f"user_{i}" for i in range(10)]

    for user_id in user_ids:
        config = {"configurable": {"thread_id": user_id}}

        try:
            # 执行
            result = app.invoke({
                "user_id": user_id,
                "session_id": user_id,
                "messages": [f"Hello from {user_id}"],
                "metadata": {},
                "error_count": 0,
                "last_updated": time.time()
            }, config=config)

            # 记录指标
            monitor.record_state(result)

            logger.info(f"Processed {user_id}: {len(result['messages'])} messages")

        except Exception as e:
            logger.error(f"Error processing {user_id}: {e}")

    # 打印指标
    metrics = monitor.get_metrics()
    logger.info(f"Total states: {metrics.total_states}")
    logger.info(f"Error states: {metrics.error_states}")
    logger.info(f"Avg message count: {metrics.avg_message_count:.2f}")
    logger.info(f"Avg metadata size: {metrics.avg_metadata_size:.2f}")

    # 清理过期状态
    cleaner.cleanup_old_states(user_ids)
```

### 2. 批量操作

```python
def batch_process(app, states: list, configs: list):
    """批量处理状态"""
    results = []

    for state, config in zip(states, configs):
        try:
            result = app.invoke(state, config=config)
            results.append(result)
        except Exception as e:
            logger.error(f"Batch process error: {e}")
            results.append(None)

    return results
```

### 3. 异步处理

```python
import asyncio
from langgraph.checkpoint.postgres import AsyncPostgresSaver

async def async_process(app, state, config):
    """异步处理"""
    try:
        result = await app.ainvoke(state, config=config)
        return result
    except Exception as e:
        logger.error(f"Async process error: {e}")
        return None

async def batch_async_process(app, states, configs):
    """批量异步处理"""
    tasks = [async_process(app, state, config) for state, config in zip(states, configs)]
    results = await asyncio.gather(*tasks)
    return results
```

## 监控告警

### 1. Prometheus 指标

```python
from prometheus_client import Counter, Histogram, Gauge

# 定义指标
state_operations = Counter('state_operations_total', 'Total state operations')
state_errors = Counter('state_errors_total', 'Total state errors')
state_size = Histogram('state_size_bytes', 'State size in bytes')
active_states = Gauge('active_states', 'Number of active states')

def monitored_process(state: State) -> dict:
    """带监控的处理"""
    state_operations.inc()

    try:
        result = safe_process(state)

        # 记录状态大小
        import sys
        size = sys.getsizeof(result)
        state_size.observe(size)

        return result
    except Exception as e:
        state_errors.inc()
        raise
```

## 常见陷阱

### 陷阱1：状态无限增长

```python
# 错误：不限制状态大小
class State(TypedDict):
    messages: Annotated[list, operator.add]  # 无限增长

# 正确：限制状态大小
def trim_messages(a: list, b: list) -> list:
    return (a + b)[-100:]  # 只保留最近100条

class State(TypedDict):
    messages: Annotated[list, trim_messages]
```

### 陷阱2：不处理连接失败

```python
# 错误：不处理连接失败
result = app.invoke(state, config=config)

# 正确：处理连接失败
try:
    result = app.invoke(state, config=config)
except Exception as e:
    logger.error(f"Connection failed: {e}")
    # 重试或降级处理
```

### 陷阱3：忘记清理过期状态

```python
# 错误：不清理过期状态
# 状态会无限累积

# 正确：定期清理
cleaner = StateCleaner(app, max_age_seconds=3600)
cleaner.cleanup_old_states(thread_ids)
```
