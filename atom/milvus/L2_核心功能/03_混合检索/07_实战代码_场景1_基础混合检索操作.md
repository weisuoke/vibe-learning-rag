## 完整代码实现

### 步骤1：创建 Collection Schema

```python
from pymilvus import (
    MilvusClient,
    DataType,
    Function,
    FunctionType
)

# 连接 Milvus
client = MilvusClient(
    uri="http://localhost:19530",
    token="root:Milvus"
)

# 创建 Schema
schema = client.create_schema(auto_id=False)

# 添加字段
schema.add_field(
    field_name="id",
    datatype=DataType.INT64,
    is_primary=True,
    description="产品ID"
)

schema.add_field(
    field_name="text",
    datatype=DataType.VARCHAR,
    max_length=1000,
    enable_analyzer=True,  # 启用分析器用于 BM25
    description="产品描述"
)

schema.add_field(
    field_name="text_dense",
    datatype=DataType.FLOAT_VECTOR,
    dim=384,  # sentence-transformers/all-MiniLM-L6-v2 维度
    description="文本稠密向量"
)

schema.add_field(
    field_name="text_sparse",
    datatype=DataType.SPARSE_FLOAT_VECTOR,
    description="文本稀疏向量（BM25自动生成）"
)

# 添加 BM25 函数
bm25_function = Function(
    name="text_bm25_emb",
    input_field_names=["text"],
    output_field_names=["text_sparse"],
    function_type=FunctionType.BM25,
)
schema.add_function(bm25_function)

print("✅ Schema 创建成功")
```

---

### 步骤2：配置索引

```python
# 准备索引参数
index_params = client.prepare_index_params()

# 稠密向量索引（AUTOINDEX）
index_params.add_index(
    field_name="text_dense",
    index_name="text_dense_index",
    index_type="AUTOINDEX",
    metric_type="IP"  # Inner Product
)

# 稀疏向量索引（BM25）
index_params.add_index(
    field_name="text_sparse",
    index_name="text_sparse_index",
    index_type="SPARSE_INVERTED_INDEX",
    metric_type="BM25",
    params={
        "inverted_index_algo": "DAAT_MAXSCORE",
        "bm25_k1": 1.2,
        "bm25_b": 0.75
    }
)

print("✅ 索引参数配置成功")
```

---

### 步骤3：创建 Collection

```python
# 创建 Collection
client.create_collection(
    collection_name="product_search",
    schema=schema,
    index_params=index_params
)

print("✅ Collection 创建成功")
```

---

### 步骤4：准备数据并插入

```python
from sentence_transformers import SentenceTransformer

# 加载 Embedding 模型
model = SentenceTransformer('sentence-transformers/all-MiniLM-L6-v2')

# 准备产品数据
products = [
    {"id": 1, "text": "红色棉质圆领T恤，舒适透气"},
    {"id": 2, "text": "无线降噪头戴式耳机，安静舒适"},
    {"id": 3, "text": "不锈钢保温水杯，500ml容量"},
    {"id": 4, "text": "白色运动鞋，轻便透气"},
    {"id": 5, "text": "蓝牙音箱，便携式设计"},
]

# 生成稠密向量
texts = [p["text"] for p in products]
dense_vectors = model.encode(texts).tolist()

# 构造插入数据（稀疏向量由 BM25 函数自动生成）
data = [
    {
        "id": p["id"],
        "text": p["text"],
        "text_dense": dense_vectors[i]
    }
    for i, p in enumerate(products)
]

# 插入数据
res = client.insert(
    collection_name="product_search",
    data=data
)

print(f"✅ 插入 {len(data)} 条数据成功")
```

---

### 步骤5：执行混合检索

```python
from pymilvus import AnnSearchRequest, RRFRanker

# 查询文本
query_text = "白色耳机，安静舒适"

# 生成查询的稠密向量
query_dense = model.encode([query_text])[0].tolist()

# 创建稠密向量检索请求
request_dense = AnnSearchRequest(
    data=[query_dense],
    anns_field="text_dense",
    param={"nprobe": 10},
    limit=5
)

# 创建稀疏向量检索请求（BM25）
request_sparse = AnnSearchRequest(
    data=[query_text],  # 直接使用文本，BM25 自动处理
    anns_field="text_sparse",
    limit=5
)

# 创建 RRF 重排序器
ranker = RRFRanker(k=60)

# 执行混合检索
results = client.hybrid_search(
    collection_name="product_search",
    reqs=[request_dense, request_sparse],
    ranker=ranker,
    limit=3,
    output_fields=["id", "text"]
)

# 打印结果
print("\n🔍 混合检索结果：")
print(f"查询：{query_text}\n")

for i, hits in enumerate(results):
    for j, hit in enumerate(hits):
        print(f"Top {j+1}:")
        print(f"  ID: {hit['id']}")
        print(f"  文本: {hit['entity']['text']}")
        print(f"  距离: {hit['distance']:.4f}")
        print()
```

## 代码解析

### 1. 为什么 Top 1 是耳机？

**稠密向量检索**（语义理解）：
- "白色耳机，安静舒适" → 理解为"降噪耳机"
- 匹配到"无线降噪头戴式耳机，安静舒适"

**稀疏向量检索**（关键词匹配）：
- "安静" → 匹配"降噪"（BM25 理解为相关）
- "舒适" → 精确匹配

**RRF 融合**：
- 耳机在两路检索中都排名靠前
- 综合得分最高

---

### 2. 为什么 Top 2 是白色运动鞋？

**稠密向量检索**：
- "白色" → 语义相关性较低

**稀疏向量检索**：
- "白色" → 精确匹配关键词
- 提升了排名

**RRF 融合**：
- 虽然语义相关性不高，但关键词匹配提升了排名

---

### 3. RRF 如何工作？

```python
# RRF 公式
RRF_score(d) = Σ(i=1 to N) 1 / (k + rank_i(d))

# 示例计算（k=60）
# 耳机：
#   稠密向量排名: 1 → 1/(60+1) = 0.0164
#   稀疏向量排名: 1 → 1/(60+1) = 0.0164
#   总分: 0.0164 + 0.0164 = 0.0328

# 白色运动鞋：
#   稠密向量排名: 5 → 1/(60+5) = 0.0154
#   稀疏向量排名: 2 → 1/(60+2) = 0.0161
#   总分: 0.0154 + 0.0161 = 0.0315
```

## 参数调优

### 1. 调整 RRF 参数 k

```python
# k 值越小，排名靠前的结果权重越大
ranker_k30 = RRFRanker(k=30)   # 更激进
ranker_k60 = RRFRanker(k=60)   # 默认
ranker_k100 = RRFRanker(k=100) # 更平滑
```

**建议**：
- k=30：强调排名靠前的结果
- k=60：平衡（推荐）
- k=100：更公平地对待所有结果

---

### 2. 调整 BM25 参数

```python
# 修改索引参数
index_params.add_index(
    field_name="text_sparse",
    index_name="text_sparse_index",
    index_type="SPARSE_INVERTED_INDEX",
    metric_type="BM25",
    params={
        "inverted_index_algo": "DAAT_MAXSCORE",
        "bm25_k1": 1.5,  # 增加词频权重
        "bm25_b": 0.5    # 减少文档长度归一化
    }
)
```

**参数说明**：
- **bm25_k1** (1.2-2.0)：控制词频饱和度
  - 越大：词频越重要
  - 越小：词频影响越小
- **bm25_b** (0-1)：控制文档长度归一化
  - 1：完全归一化
  - 0：不归一化

## 完整可运行代码

```python
"""
基础混合检索示例
功能：稠密向量 + 稀疏向量混合检索
"""

from pymilvus import (
    MilvusClient,
    DataType,
    Function,
    FunctionType,
    AnnSearchRequest,
    RRFRanker
)
from sentence_transformers import SentenceTransformer

def main():
    # 1. 连接 Milvus
    client = MilvusClient(
        uri="http://localhost:19530",
        token="root:Milvus"
    )
    
    # 2. 删除已存在的 Collection
    if client.has_collection("product_search"):
        client.drop_collection("product_search")
    
    # 3. 创建 Schema
    schema = client.create_schema(auto_id=False)
    schema.add_field(field_name="id", datatype=DataType.INT64, is_primary=True)
    schema.add_field(field_name="text", datatype=DataType.VARCHAR, 
                     max_length=1000, enable_analyzer=True)
    schema.add_field(field_name="text_dense", datatype=DataType.FLOAT_VECTOR, dim=384)
    schema.add_field(field_name="text_sparse", datatype=DataType.SPARSE_FLOAT_VECTOR)
    
    # 4. 添加 BM25 函数
    bm25_function = Function(
        name="text_bm25_emb",
        input_field_names=["text"],
        output_field_names=["text_sparse"],
        function_type=FunctionType.BM25,
    )
    schema.add_function(bm25_function)
    
    # 5. 配置索引
    index_params = client.prepare_index_params()
    index_params.add_index(field_name="text_dense", index_type="AUTOINDEX", metric_type="IP")
    index_params.add_index(field_name="text_sparse", index_type="SPARSE_INVERTED_INDEX",
                           metric_type="BM25", params={"inverted_index_algo": "DAAT_MAXSCORE"})
    
    # 6. 创建 Collection
    client.create_collection(collection_name="product_search", schema=schema, index_params=index_params)
    
    # 7. 准备数据
    model = SentenceTransformer('sentence-transformers/all-MiniLM-L6-v2')
    products = [
        {"id": 1, "text": "红色棉质圆领T恤，舒适透气"},
        {"id": 2, "text": "无线降噪头戴式耳机，安静舒适"},
        {"id": 3, "text": "不锈钢保温水杯，500ml容量"},
        {"id": 4, "text": "白色运动鞋，轻便透气"},
        {"id": 5, "text": "蓝牙音箱，便携式设计"},
    ]
    
    texts = [p["text"] for p in products]
    dense_vectors = model.encode(texts).tolist()
    
    data = [
        {"id": p["id"], "text": p["text"], "text_dense": dense_vectors[i]}
        for i, p in enumerate(products)
    ]
    
    # 8. 插入数据
    client.insert(collection_name="product_search", data=data)
    
    # 9. 执行混合检索
    query_text = "白色耳机，安静舒适"
    query_dense = model.encode([query_text])[0].tolist()
    
    request_dense = AnnSearchRequest(data=[query_dense], anns_field="text_dense", limit=5)
    request_sparse = AnnSearchRequest(data=[query_text], anns_field="text_sparse", limit=5)
    ranker = RRFRanker(k=60)
    
    results = client.hybrid_search(
        collection_name="product_search",
        reqs=[request_dense, request_sparse],
        ranker=ranker,
        limit=3,
        output_fields=["id", "text"]
    )
    
    # 10. 打印结果
    print(f"\n🔍 混合检索结果：\n查询：{query_text}\n")
    for hits in results:
        for i, hit in enumerate(hits):
            print(f"Top {i+1}: {hit['entity']['text']} (距离: {hit['distance']:.4f})")

if __name__ == "__main__":
    main()
```
