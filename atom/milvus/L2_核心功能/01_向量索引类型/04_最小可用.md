## 最小可用代码（50行）

```python
from pymilvus import connections, Collection, FieldSchema, CollectionSchema, DataType
import numpy as np
import time

# 1. 连接 Milvus
connections.connect("default", host="localhost", port="19530")

# 2. 定义 Schema
fields = [
    FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
    FieldSchema(name="embeddings", dtype=DataType.FLOAT_VECTOR, dim=128)
]
schema = CollectionSchema(fields, description="Index comparison demo")

# 3. 创建三个 Collection（不同索引）
collections = {}

# FLAT 索引
collections["flat"] = Collection("index_flat", schema)
collections["flat"].create_index("embeddings", {
    "index_type": "FLAT",
    "metric_type": "L2"
})

# HNSW 索引
collections["hnsw"] = Collection("index_hnsw", schema)
collections["hnsw"].create_index("embeddings", {
    "index_type": "HNSW",
    "metric_type": "L2",
    "params": {"M": 16, "efConstruction": 200}
})

# IVF_FLAT 索引
collections["ivf"] = Collection("index_ivf", schema)
collections["ivf"].create_index("embeddings", {
    "index_type": "IVF_FLAT",
    "metric_type": "L2",
    "params": {"nlist": 128}
})

# 4. 插入相同数据
vectors = np.random.rand(10000, 128).astype(np.float32).tolist()
for coll in collections.values():
    coll.insert([vectors])
    coll.load()

# 5. 查询性能对比
query_vector = np.random.rand(1, 128).astype(np.float32).tolist()

for name, coll in collections.items():
    start = time.time()
    results = coll.search(query_vector, "embeddings", {"metric_type": "L2"}, limit=10)
    latency = (time.time() - start) * 1000
    print(f"{name.upper()}: {latency:.2f}ms")
```

### 1. 连接 Milvus

```python
connections.connect("default", host="localhost", port="19530")
```

**说明：**
- `default`：连接别名
- `host`：Milvus 服务地址
- `port`：Milvus 服务端口（默认 19530）

---

### 2. 定义 Schema

```python
fields = [
    FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
    FieldSchema(name="embeddings", dtype=DataType.FLOAT_VECTOR, dim=128)
]
schema = CollectionSchema(fields, description="Index comparison demo")
```

**说明：**
- `id`：主键字段，自动生成
- `embeddings`：向量字段，128维
- `schema`：定义 Collection 的结构

---

### 3. 创建索引

#### FLAT 索引（精确搜索）

```python
collections["flat"].create_index("embeddings", {
    "index_type": "FLAT",
    "metric_type": "L2"
})
```

**特点：**
- ✅ 100% 准确率
- ❌ O(n) 时间复杂度
- 🎯 适合：< 1M 向量

#### HNSW 索引（高性能）

```python
collections["hnsw"].create_index("embeddings", {
    "index_type": "HNSW",
    "metric_type": "L2",
    "params": {"M": 16, "efConstruction": 200}
})
```

**参数说明：**
- `M=16`：每个节点最多 16 个连接
- `efConstruction=200`：构建质量参数

**特点：**
- ✅ 95-99% 准确率
- ✅ O(log n) 时间复杂度
- 🎯 适合：1M-100M 向量

#### IVF_FLAT 索引（平衡性能）

```python
collections["ivf"].create_index("embeddings", {
    "index_type": "IVF_FLAT",
    "metric_type": "L2",
    "params": {"nlist": 128}
})
```

**参数说明：**
- `nlist=128`：聚类数量

**特点：**
- ✅ 95-99% 准确率
- ✅ O(nprobe * n/nlist) 时间复杂度
- 🎯 适合：1M-10M 向量

## 常见问题

### Q1：为什么 HNSW 比 FLAT 快这么多？

**A：** HNSW 使用多层图结构，时间复杂度 O(log n)，而 FLAT 是暴力搜索 O(n)。

**类比：**
- **FLAT**：逐页翻字典（O(n)）
- **HNSW**：使用字母索引快速定位（O(log n)）

---

### Q2：HNSW 的准确率为什么不是 100%？

**A：** HNSW 是近似算法，使用贪婪搜索可能陷入局部最优。

**权衡：**
- ✅ 速度提升 20 倍
- ❌ 准确率损失 1-2%

**实际影响：**
- 98% 准确率意味着 Top-10 结果中可能有 1-2 个不是真实最近邻
- 对大多数应用（RAG、推荐系统）影响很小

---

### Q3：如何选择索引类型？

**决策树：**

```
数据规模？
  ├─ < 1M → FLAT（简单直接）
  ├─ 1M-10M → IVF_FLAT（平衡性能）
  ├─ 10M-100M → HNSW（生产标准）
  └─ > 100M → GPU_CAGRA 或 RaBitQ（高性能/成本优化）
```

---

### Q4：索引参数如何调优？

**HNSW 参数：**
```python
# 平衡配置（推荐）
{"M": 16, "efConstruction": 200}

# 高性能配置
{"M": 32, "efConstruction": 400}

# 低内存配置
{"M": 8, "efConstruction": 100}
```

**IVF_FLAT 参数：**
```python
# nlist 推荐值
nlist = int(sqrt(num_vectors))

# 例如：1M 向量
nlist = int(sqrt(1000000)) = 1000
```

## 完整示例代码

```python
"""
Milvus 向量索引类型对比示例
演示 FLAT、HNSW、IVF_FLAT 三种索引的性能差异
"""

from pymilvus import connections, Collection, FieldSchema, CollectionSchema, DataType, utility
import numpy as np
import time

def create_collection_with_index(name, index_type, index_params=None):
    """创建带索引的 Collection"""
    # 删除已存在的 Collection
    if utility.has_collection(name):
        utility.drop_collection(name)

    # 定义 Schema
    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
        FieldSchema(name="embeddings", dtype=DataType.FLOAT_VECTOR, dim=128)
    ]
    schema = CollectionSchema(fields, description=f"{index_type} index demo")

    # 创建 Collection
    collection = Collection(name, schema)

    # 创建索引
    index_config = {
        "index_type": index_type,
        "metric_type": "L2",
        "params": index_params or {}
    }
    collection.create_index("embeddings", index_config)

    return collection

def benchmark_index(collection, query_vectors, num_queries=100):
    """测试索引性能"""
    latencies = []

    for i in range(num_queries):
        start = time.time()
        results = collection.search(
            data=[query_vectors[i]],
            anns_field="embeddings",
            param={"metric_type": "L2"},
            limit=10
        )
        latency = (time.time() - start) * 1000
        latencies.append(latency)

    avg_latency = np.mean(latencies)
    p95_latency = np.percentile(latencies, 95)
    p99_latency = np.percentile(latencies, 99)

    return {
        "avg": avg_latency,
        "p95": p95_latency,
        "p99": p99_latency
    }

def main():
    # 连接 Milvus
    connections.connect("default", host="localhost", port="19530")

    # 生成测试数据
    num_vectors = 10000
    dim = 128
    vectors = np.random.rand(num_vectors, dim).astype(np.float32).tolist()
    query_vectors = np.random.rand(100, dim).astype(np.float32).tolist()

    # 创建不同索引的 Collection
    indexes = {
        "FLAT": ("index_flat", "FLAT", {}),
        "HNSW": ("index_hnsw", "HNSW", {"M": 16, "efConstruction": 200}),
        "IVF_FLAT": ("index_ivf", "IVF_FLAT", {"nlist": 128})
    }

    results = {}

    for name, (coll_name, index_type, params) in indexes.items():
        print(f"\n测试 {name} 索引...")

        # 创建 Collection
        collection = create_collection_with_index(coll_name, index_type, params)

        # 插入数据
        print(f"  插入 {num_vectors} 个向量...")
        collection.insert([vectors])

        # 加载索引
        print(f"  加载索引...")
        collection.load()

        # 性能测试
        print(f"  执行 100 次查询...")
        stats = benchmark_index(collection, query_vectors)
        results[name] = stats

        print(f"  平均延迟: {stats['avg']:.2f}ms")
        print(f"  P95 延迟: {stats['p95']:.2f}ms")
        print(f"  P99 延迟: {stats['p99']:.2f}ms")

    # 输出对比结果
    print("\n" + "="*60)
    print("性能对比总结")
    print("="*60)
    print(f"{'索引类型':<15} {'平均延迟':<15} {'P95延迟':<15} {'P99延迟':<15}")
    print("-"*60)

    for name, stats in results.items():
        print(f"{name:<15} {stats['avg']:>10.2f}ms {stats['p95']:>10.2f}ms {stats['p99']:>10.2f}ms")

    print("="*60)

    # 清理
    for coll_name, _, _ in indexes.values():
        utility.drop_collection(coll_name)

    connections.disconnect("default")

if __name__ == "__main__":
    main()
```
