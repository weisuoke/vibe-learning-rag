```python
def auto_select_index_type(field_name, data):
    """根据数据特征自动选择索引类型"""

    # 计算基数
    unique_values = len(set([d[field_name] for d in data]))
    total_values = len(data)
    cardinality = unique_values / total_values

    # 判断数据类型
    sample_value = data[0][field_name]
    data_type = type(sample_value).__name__

    # 选择索引
    if cardinality < 0.01:  # 低基数
        return "BITMAP"
    elif data_type in ["int", "float"]:
        return "INVERTED"
    elif data_type == "str":
        # 检查是否有前缀匹配需求
        if has_prefix_queries(field_name):
            return "TRIE"
        else:
            return "INVERTED"
    else:
        return "AUTOINDEX"

# 使用示例
data = [{"age": 25}, {"age": 30}, {"age": 25}]

print(f"推荐索引类型: {index_type}")
```

### 技巧2：查询计划优化

**场景**：根据查询模式优化过滤条件顺序

```python
def optimize_filter_order(filters, statistics):
    """根据统计信息优化过滤条件顺序"""

    # 计算每个条件的选择性
    selectivity = {}
    for field, condition in filters.items():
        # 从统计信息中获取选择性
        selectivity[field] = statistics.get(field, {}).get("selectivity", 1.0)

    # 按选择性排序（选择性高的在前）
    sorted_filters = sorted(
        filters.items(),
        key=lambda x: selectivity[x[0]]
    )

    # 构建优化后的过滤表达式
    filter_expr = " AND ".join([
        f"{field} {condition}" for field, condition in sorted_filters
    ])

    return filter_expr

# 使用示例
filters = {
    "price": ">= 100 AND price <= 500",  # 选择性: 0.8
    "stock": "> 0",  # 选择性: 0.1
    "category": "== 'electronics'"  # 选择性: 0.2
}

statistics = {
    "price": {"selectivity": 0.8},
    "stock": {"selectivity": 0.1},
    "category": {"selectivity": 0.2}
}

optimized_filter = optimize_filter_order(filters, statistics)
print(f"优化后的过滤条件: {optimized_filter}")
# 输出: stock > 0 AND category == 'electronics' AND price >= 100 AND price <= 500
```

```python
class JSONPathIndexCache:
    """JSON Path 索引缓存管理器"""

    def __init__(self, client, collection_name):
        self.client = client
        self.collection_name = collection_name
        self.cache = {}
        self._load_existing_indexes()

    def _load_existing_indexes(self):
        """加载已存在的索引"""
        indexes = self.client.list_indexes(self.collection_name)
        for idx in indexes:
            if idx.get("field_name") == "metadata":
                json_path = idx.get("params", {}).get("json_path")
                if json_path:
                    self.cache[json_path] = idx["index_name"]

    def ensure_index(self, json_path, json_cast_type):
        """确保索引存在，如果不存在则创建"""
        if json_path in self.cache:
            return self.cache[json_path]

        # 创建新索引
        index_name = f"json_index_{len(self.cache)}"
        index_params = self.client.prepare_index_params()
        index_params.add_index(
            field_name="metadata",
            index_name=index_name,
            index_type="INVERTED",
            params={
                "json_path": json_path,
                "json_cast_type": json_cast_type
            }
        )
        self.client.create_index(self.collection_name, index_params)

        # 更新缓存
        self.cache[json_path] = index_name
        return index_name

# 使用示例
cache = JSONPathIndexCache(client, "products")
index_name = cache.ensure_index("metadata['category']", "varchar")
print(f"索引名称: {index_name}")
```

**场景**：使用分区优化大规模数据查询

```python
def partition_by_time(client, collection_name, data, time_field="created_at"):
    """按时间分区插入数据"""
    from datetime import datetime

    # 按月份分组
    partitions = {}
    for item in data:
        timestamp = item[time_field]
        month = datetime.fromtimestamp(timestamp).strftime("%Y_%m")

        if month not in partitions:
            partitions[month] = []
        partitions[month].append(item)

    # 为每个月份创建分区并插入数据
    for month, items in partitions.items():
        partition_name = f"partition_{month}"

        # 创建分区（如果不存在）
        if not client.has_partition(collection_name, partition_name):
            client.create_partition(collection_name, partition_name)

        # 插入数据到分区
        client.insert(
            collection_name=collection_name,
            data=items,
            partition_name=partition_name
        )

    return list(partitions.keys())

# 查询时指定分区
def query_with_partition(client, collection_name, query_vector, time_range):
    """查询指定时间范围的分区"""
    start_month, end_month = time_range
    partition_names = [f"partition_{month}" for month in [start_month, end_month]]

    results = client.search(
        collection_name=collection_name,
        partition_names=partition_names,
        data=[query_vector],
        limit=10
    )

    return results
```

## 深层原理

### 原理1：索引内部结构

**INVERTED 索引结构**：
```
倒排索引 (Inverted Index):
┌─────────────────────────────────────┐
│ Term Dictionary (词典)              │
├─────────────────────────────────────┤
│ "electronics" → Posting List 1      │
│ "books"       → Posting List 2      │
│ "clothing"    → Posting List 3      │
└─────────────────────────────────────┘

Posting List 1 (文档列表):
[1, 5, 10, 15, 20, 25, ...]

查询流程:
1. 在 Term Dictionary 中查找 "electronics"
2. 获取 Posting List
3. 返回文档 ID 列表
时间复杂度: O(log n)
```

**BITMAP 索引结构**：
```
位图索引 (Bitmap Index):
┌─────────────────────────────────────┐
│ Value → Bitmap                      │
├─────────────────────────────────────┤
│ "active"  → [1,0,1,0,1,0,1,0,...]  │
│ "pending" → [0,1,0,1,0,1,0,1,...]  │
│ "deleted" → [0,0,0,0,0,0,0,0,...]  │
└─────────────────────────────────────┘

查询流程:
1. 获取对应值的 Bitmap
2. 位运算找到匹配的文档
时间复杂度: O(1)

位运算优化:
AND: bitmap1 & bitmap2
OR:  bitmap1 | bitmap2
NOT: ~bitmap1
```

### 原理2：查询执行计划

**Pre-filtering 执行计划**：
```
执行计划:
1. 标量过滤阶段
   ├─ 读取索引
   ├─ 应用过滤条件
   └─ 生成候选集 (Candidate Set)

2. 向量检索阶段
   ├─ 在候选集中构建 HNSW 图
   ├─ 执行 ANN 搜索
   └─ 返回 Top-K 结果

成本分析:
- 标量过滤: O(log n)
- 向量检索: O(log m)，其中 m = 候选集大小
- 总成本: O(log n + log m)

适用条件:
- 选择性 > 50%
- 候选集 < 10万条
```

**Post-filtering 执行计划**：
```
执行计划:
1. 向量检索阶段
   ├─ 在全集中执行 ANN 搜索
   ├─ 获取 Top-K' 结果 (K' > K)
   └─ 生成候选集

2. 标量过滤阶段
   ├─ 应用过滤条件
   ├─ 过滤候选集
   └─ 返回 Top-K 结果

成本分析:
- 向量检索: O(log n)
- 标量过滤: O(K')
- 总成本: O(log n + K')

适用条件:
- 选择性 < 50%
- 候选集 > 50万条
```

```python
class IndexManager:
    """生产环境索引管理器"""

    def __init__(self, client, collection_name):
        self.client = client
        self.collection_name = collection_name
        self.index_stats = {}

    def analyze_query_patterns(self, query_logs):
        """分析查询模式"""
        field_usage = {}

        for log in query_logs:
            filter_expr = log.get("filter")
            if filter_expr:
                # 提取使用的字段
                fields = self._extract_fields(filter_expr)
                for field in fields:
                    field_usage[field] = field_usage.get(field, 0) + 1

        return field_usage

    def recommend_indexes(self, field_usage, threshold=100):
        """推荐需要创建的索引"""
        recommendations = []

        for field, count in field_usage.items():
            if count > threshold:
                # 分析字段特征
                cardinality = self._analyze_cardinality(field)
                index_type = "BITMAP" if cardinality < 0.01 else "INVERTED"

                recommendations.append({
                    "field": field,
                    "index_type": index_type,
                    "usage_count": count,
                    "priority": "high" if count > 1000 else "medium"
                })

        return recommendations

    def create_indexes_batch(self, recommendations):
        """批量创建索引"""
        index_params = self.client.prepare_index_params()

        for rec in recommendations:
            index_params.add_index(
                field_name=rec["field"],
                index_type=rec["index_type"]
            )

        self.client.create_index(self.collection_name, index_params)

    def monitor_index_performance(self):
        """监控索引性能"""
        indexes = self.client.list_indexes(self.collection_name)

        for idx in indexes:
            field_name = idx["field_name"]
            # 收集性能指标
            self.index_stats[field_name] = {
                "hit_rate": self._calculate_hit_rate(field_name),
                "avg_latency": self._calculate_avg_latency(field_name)
            }

        return self.index_stats
```

```python
class QueryPerformanceMonitor:
    """查询性能监控器"""

    def __init__(self):
        self.metrics = []

    def record_query(self, query_info):
        """记录查询信息"""
        self.metrics.append({
            "timestamp": time.time(),
            "filter": query_info.get("filter"),
            "latency": query_info.get("latency"),
            "result_count": query_info.get("result_count")
        })

    def analyze_slow_queries(self, threshold_ms=100):
        """分析慢查询"""
        slow_queries = [
            m for m in self.metrics
            if m["latency"] > threshold_ms
        ]

        # 按延迟排序
        slow_queries.sort(key=lambda x: x["latency"], reverse=True)

        return slow_queries

    def generate_report(self):
        """生成性能报告"""
        if not self.metrics:
            return {}

        latencies = [m["latency"] for m in self.metrics]

        return {
            "total_queries": len(self.metrics),
            "avg_latency": statistics.mean(latencies),
            "p50_latency": statistics.median(latencies),
            "p95_latency": self._percentile(latencies, 0.95),
            "p99_latency": self._percentile(latencies, 0.99),
            "slow_query_count": len(self.analyze_slow_queries())
        }

    def _percentile(self, data, percentile):
        """计算百分位数"""
        sorted_data = sorted(data)
        index = int(len(sorted_data) * percentile)
        return sorted_data[index]

# 使用示例
monitor = QueryPerformanceMonitor()

# 记录查询
monitor.record_query({
    "filter": "age > 30",
    "latency": 50,
    "result_count": 100
})

# 生成报告
report = monitor.generate_report()
print(f"平均延迟: {report['avg_latency']:.2f} ms")
print(f"P95 延迟: {report['p95_latency']:.2f} ms")
```

```python
    """数据预处理优化索引性能"""

    processed_data = []

    for item in data:
        processed_item = item.copy()

        # 1. 扁平化高频访问字段
```

```python
class ResilientQueryExecutor:
    """容错查询执行器"""

    def __init__(self, client, collection_name):
        self.client = client
        self.collection_name = collection_name

    def execute_with_fallback(self, query_vector, filter_expr, limit=10):
        """带降级策略的查询执行"""

        try:
            # 尝试使用完整过滤条件
            results = self.client.search(
                collection_name=self.collection_name,
                data=[query_vector],
                filter=filter_expr,
                limit=limit,
                timeout=5000  # 5秒超时
            )
            return results

        except TimeoutError:
            # 降级策略1：简化过滤条件
            simplified_filter = self._simplify_filter(filter_expr)
            try:
                results = self.client.search(
                    collection_name=self.collection_name,
                    data=[query_vector],
                    filter=simplified_filter,
                    limit=limit * 2  # 增加返回数量
                )
                # 后处理过滤
                return self._post_filter(results, filter_expr)

            except Exception:
                # 降级策略2：无过滤查询
                results = self.client.search(
                    collection_name=self.collection_name,
                    data=[query_vector],
                    limit=limit * 5
                )
                return self._post_filter(results, filter_expr)

        except Exception as e:
            # 记录错误并返回空结果
            print(f"查询失败: {e}")
            return []

    def _simplify_filter(self, filter_expr):
        """简化过滤条件（保留最重要的条件）"""
        # 提取第一个条件
        if " AND " in filter_expr:
            return filter_expr.split(" AND ")[0]
        return filter_expr

    def _post_filter(self, results, filter_expr):
        """后处理过滤"""
        # 在应用层应用完整过滤条件
        # 这里简化实现
        return results
```

```python
import concurrent.futures

def parallel_query(client, collection_name, queries, max_workers=10):
    """并发查询"""

    def single_query(query_info):
        query_vector, filter_expr = query_info
        return client.search(
            collection_name=collection_name,
            data=[query_vector],
            filter=filter_expr,
            limit=10
        )

    with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = [executor.submit(single_query, q) for q in queries]
        results = [f.result() for f in concurrent.futures.as_completed(futures)]

    return results

# 使用示例
```

### 技巧3：缓存策略

```python
from functools import lru_cache
import hashlib

class QueryCache:
    """查询结果缓存"""

    def __init__(self, max_size=1000, ttl=3600):
        self.cache = {}
        self.max_size = max_size
        self.ttl = ttl

    def _generate_key(self, query_vector, filter_expr):
        """生成缓存键"""
        vector_hash = hashlib.md5(str(query_vector).encode()).hexdigest()
        filter_hash = hashlib.md5(filter_expr.encode()).hexdigest()
        return f"{vector_hash}_{filter_hash}"

    def get(self, query_vector, filter_expr):
        """获取缓存"""
        key = self._generate_key(query_vector, filter_expr)

        if key in self.cache:
            cached_data, timestamp = self.cache[key]
            if time.time() - timestamp < self.ttl:
                return cached_data
            else:
                del self.cache[key]

        return None

    def set(self, query_vector, filter_expr, results):
        """设置缓存"""
        if len(self.cache) >= self.max_size:
            # 删除最旧的缓存
            oldest_key = min(self.cache.keys(), key=lambda k: self.cache[k][1])
            del self.cache[oldest_key]

        key = self._generate_key(query_vector, filter_expr)
        self.cache[key] = (results, time.time())

# 使用示例
cache = QueryCache()

def cached_search(client, collection_name, query_vector, filter_expr):
    """带缓存的查询"""
    # 尝试从缓存获取
    cached_result = cache.get(query_vector, filter_expr)
    if cached_result:
        return cached_result

    # 执行查询
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter=filter_expr,
        limit=10
    )

    # 缓存结果
    cache.set(query_vector, filter_expr, results)

    return results
```
