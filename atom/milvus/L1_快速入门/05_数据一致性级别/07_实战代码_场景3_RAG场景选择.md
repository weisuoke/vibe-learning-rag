```python
# RAG场景一致性级别选择

from pymilvus import connections, Collection, CollectionSchema, FieldSchema, DataType, ConsistencyLevel, utility
import numpy as np
from openai import OpenAI
from datetime import datetime
from enum import Enum

# 连接Milvus
connections.connect(alias="default", host="localhost", port="19530")

# 初始化OpenAI客户端
client = OpenAI()

# RAG场景枚举
class RAGScenario(Enum):
    DOCUMENT_QA = "document_qa"           # 文档问答
    REALTIME_KB = "realtime_kb"           # 实时知识库
    BATCH_PROCESSING = "batch_processing" # 批量处理
    INTERACTIVE_CHAT = "interactive_chat" # 交互式聊天

# 场景1: 文档问答系统（Bounded一致性）
class DocumentQASystem:
    def __init__(self):
        self.collection_name = "document_qa"
        self.collection = self._create_collection()

    def _create_collection(self):
        if utility.has_collection(self.collection_name):
            return Collection(self.collection_name)

        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=1536),
            FieldSchema(name="document_id", dtype=DataType.VARCHAR, max_length=100),
            FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
            FieldSchema(name="metadata", dtype=DataType.JSON)
        ]
        schema = CollectionSchema(fields=fields)

        # 使用Bounded一致性（默认，适合大多数场景）
        collection = Collection(
            name=self.collection_name,
            schema=schema,
            consistency_level="Bounded"
        )

        index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "COSINE",
            "params": {"nlist": 128}
        }
        collection.create_index(field_name="vector", index_params=index_params)
        collection.load()

        return collection

    def add_documents(self, documents):
        """批量添加文档"""
        entities = []
        for doc in documents:
            # 生成embedding
            embedding = self._get_embedding(doc["content"])
            entities.append({
                "vector": embedding,
                "document_id": doc["id"],
                "content": doc["content"],
                "metadata": doc.get("metadata", {})
            })

        self.collection.insert(entities)
        self.collection.flush()
        print(f"✅ Added {len(documents)} documents")

    def query(self, question, top_k=5):
        """查询文档并生成答案"""
        # 1. 生成问题embedding
        query_embedding = self._get_embedding(question)

        # 2. 向量检索（使用Bounded一致性）
        results = self.collection.search(
            data=[query_embedding],
            anns_field="vector",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=top_k,
            output_fields=["content", "document_id"],
            consistency_level=ConsistencyLevel.Bounded  # 明确指定
        )

        # 3. 提取上下文
        contexts = [hit.entity.get("content") for hit in results[0]]
        context_text = "\n\n".join(contexts)

        # 4. 生成答案
        answer = self._generate_answer(question, context_text)

        return {
            "question": question,
            "answer": answer,
            "sources": [hit.entity.get("document_id") for hit in results[0]],
            "num_sources": len(results[0])
        }

    def _get_embedding(self, text):
        response = client.embeddings.create(
            model="text-embedding-3-small",
            input=text
        )
        return response.data[0].embedding

    def _generate_answer(self, question, context):
        response = client.chat.completions.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are a helpful assistant. Answer based on the provided context."},
                {"role": "user", "content": f"Context:\n{context}\n\nQuestion: {question}\n\nAnswer:"}
            ]
        )
        return response.choices[0].message.content

# 场景2: 实时知识库（Strong/Session一致性）
class RealtimeKnowledgeBase:
    def __init__(self):
        self.collection_name = "realtime_kb"
        self.collection = self._create_collection()

    def _create_collection(self):
        if utility.has_collection(self.collection_name):
            return Collection(self.collection_name)

        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=1536),
            FieldSchema(name="kb_id", dtype=DataType.VARCHAR, max_length=100),
            FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
            FieldSchema(name="updated_at", dtype=DataType.INT64)
        ]
        schema = CollectionSchema(fields=fields)

        # 使用Session一致性（交互式场景）
        collection = Collection(
            name=self.collection_name,
            schema=schema,
            consistency_level="Session"
        )

        index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "COSINE",
            "params": {"nlist": 128}
        }
        collection.create_index(field_name="vector", index_params=index_params)
        collection.load()

        return collection

    def update_knowledge(self, kb_id, content):
        """更新知识库（用户立即可见）"""
        embedding = self._get_embedding(content)

        # 使用upsert确保更新
        self.collection.upsert([{
            "kb_id": kb_id,
            "vector": embedding,
            "content": content,
            "updated_at": int(datetime.now().timestamp() * 1000)
        }])

        print(f"✅ Updated knowledge: {kb_id}")

    def query_immediately(self, question):
        """立即查询（Session一致性保证能看到刚更新的内容）"""
        query_embedding = self._get_embedding(question)

        # 使用Session一致性
        results = self.collection.search(
            data=[query_embedding],
            anns_field="vector",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=3,
            output_fields=["content", "kb_id", "updated_at"],
            consistency_level=ConsistencyLevel.Session  # 会话内一致
        )

        return [{
            "content": hit.entity.get("content"),
            "kb_id": hit.entity.get("kb_id"),
            "updated_at": hit.entity.get("updated_at"),
            "score": hit.distance
        } for hit in results[0]]

    def _get_embedding(self, text):
        response = client.embeddings.create(
            model="text-embedding-3-small",
            input=text
        )
        return response.data[0].embedding

# 场景3: 批量文档处理（Eventually一致性）
class BatchDocumentProcessor:
    def __init__(self):
        self.collection_name = "batch_docs"
        self.collection = self._create_collection()

    def _create_collection(self):
        if utility.has_collection(self.collection_name):
            return Collection(self.collection_name)

        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=1536),
            FieldSchema(name="doc_id", dtype=DataType.VARCHAR, max_length=100),
            FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000)
        ]
        schema = CollectionSchema(fields=fields)

        # 使用Eventually一致性（批量场景，性能优先）
        collection = Collection(
            name=self.collection_name,
            schema=schema,
            consistency_level="Eventually"
        )

        index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "COSINE",
            "params": {"nlist": 128}
        }
        collection.create_index(field_name="vector", index_params=index_params)
        collection.load()

        return collection

    def batch_import(self, documents, batch_size=1000):
        """批量导入文档（高性能）"""
        total = len(documents)
        print(f"📦 Importing {total} documents...")

        for i in range(0, total, batch_size):
            batch = documents[i:i+batch_size]
            entities = []

            for doc in batch:
                embedding = self._get_embedding(doc["content"])
                entities.append({
                    "vector": embedding,
                    "doc_id": doc["id"],
                    "content": doc["content"]
                })

            self.collection.insert(entities)

            if (i + batch_size) % 5000 == 0:
                print(f"   Progress: {min(i+batch_size, total)}/{total}")

        self.collection.flush()
        print(f"✅ Import complete: {total} documents")

    def batch_search(self, queries, top_k=10):
        """批量搜索（Eventually一致性，最高性能）"""
        query_embeddings = [self._get_embedding(q) for q in queries]

        results = self.collection.search(
            data=query_embeddings,
            anns_field="vector",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=top_k,
            output_fields=["content", "doc_id"],
            consistency_level=ConsistencyLevel.Eventually  # 最高性能
        )

        return results

    def _get_embedding(self, text):
        response = client.embeddings.create(
            model="text-embedding-3-small",
            input=text
        )
        return response.data[0].embedding

class SmartRAGSystem:
    def __init__(self):
        self.collection_name = "smart_rag"
        self.collection = self._create_collection()

    def _create_collection(self):
        if utility.has_collection(self.collection_name):
            return Collection(self.collection_name)

        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=1536),
            FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
            FieldSchema(name="priority", dtype=DataType.INT64)
        ]
        schema = CollectionSchema(fields=fields)

        # 默认使用Bounded
        collection = Collection(
            name=self.collection_name,
            schema=schema,
            consistency_level="Bounded"
        )

        index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "COSINE",
            "params": {"nlist": 128}
        }
        collection.create_index(field_name="vector", index_params=index_params)
        collection.load()

        return collection

    def select_consistency_level(self, scenario: RAGScenario, is_critical: bool = False):
        """根据场景自动选择一致性级别"""
        if is_critical:
            return ConsistencyLevel.Strong

        consistency_map = {
            RAGScenario.DOCUMENT_QA: ConsistencyLevel.Bounded,
            RAGScenario.REALTIME_KB: ConsistencyLevel.Session,
            RAGScenario.BATCH_PROCESSING: ConsistencyLevel.Eventually,
            RAGScenario.INTERACTIVE_CHAT: ConsistencyLevel.Session
        }

        return consistency_map.get(scenario, ConsistencyLevel.Bounded)

    def smart_query(self, question, scenario: RAGScenario, is_critical: bool = False):
        """智能查询（自动选择一致性级别）"""
        # 自动选择一致性级别
        consistency_level = self.select_consistency_level(scenario, is_critical)

        print(f"🤖 Scenario: {scenario.value}")
        print(f"🎯 Selected consistency: {consistency_level.name}")

        # 生成embedding
        query_embedding = self._get_embedding(question)

        # 执行搜索
        results = self.collection.search(
            data=[query_embedding],
            anns_field="vector",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=5,
            output_fields=["content", "priority"],
            consistency_level=consistency_level  # 动态选择
        )

        return {
            "scenario": scenario.value,
            "consistency_level": consistency_level.name,
            "results": [{
                "content": hit.entity.get("content"),
                "priority": hit.entity.get("priority"),
                "score": hit.distance
            } for hit in results[0]]
        }

    def _get_embedding(self, text):
        response = client.embeddings.create(
            model="text-embedding-3-small",
            input=text
        )
        return response.data[0].embedding
```

## 场景选择决策树

```python
def choose_consistency_level(scenario_type, requirements):
    """
    一致性级别选择决策树

    参数:
        scenario_type: 场景类型
        requirements: 需求字典 {
            'update_frequency': 'high'|'medium'|'low',
            'latency_requirement': 'strict'|'moderate'|'relaxed',
            'data_criticality': 'critical'|'important'|'normal'
        }

    返回:
        推荐的一致性级别
    """

    # 关键数据优先使用Strong
    if requirements.get('data_criticality') == 'critical':
        return 'Strong'

    # 根据场景类型选择
    if scenario_type == 'document_qa':
        # 文档问答：Bounded（默认）
        return 'Bounded'

    elif scenario_type == 'realtime_kb':
        # 实时知识库：根据更新频率选择
        if requirements.get('update_frequency') == 'high':
            return 'Session'  # 高频更新，用户需要立即看到
        else:
            return 'Bounded'

    elif scenario_type == 'batch_processing':
        # 批量处理：Eventually（性能优先）
        return 'Eventually'

    elif scenario_type == 'interactive_chat':
        # 交互式聊天：Session（用户体验）
        return 'Session'

    else:
        # 默认：Bounded
        return 'Bounded'

# 使用示例
print(choose_consistency_level('document_qa', {
    'update_frequency': 'low',
    'latency_requirement': 'moderate',
    'data_criticality': 'normal'
}))
# 输出: Bounded

print(choose_consistency_level('realtime_kb', {
    'update_frequency': 'high',
    'latency_requirement': 'strict',
    'data_criticality': 'critical'
}))
# 输出: Strong
```

## 总结

### 场景与一致性级别映射

| RAG场景 | 推荐级别 | 理由 |
|---------|---------|------|
| **文档问答** | Bounded | 平衡性能与准确性 |
| **实时知识库** | Session | 用户立即看到更新 |
| **批量处理** | Eventually | 最高性能 |
| **交互式聊天** | Session | 用户体验优先 |
| **关键数据** | Strong | 准确性优先 |

### 关键要点

1. **默认使用Bounded**：适合70%的RAG场景
2. **交互式用Session**：用户写后立即读
3. **批量用Eventually**：性能优先
4. **关键数据用Strong**：准确性优先
