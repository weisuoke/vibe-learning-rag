# 实战代码：场景6 - 索引选型实战

## 1. 场景概述

### 业务场景
**端到端索引选型决策系统**

某技术团队需要为多个AI项目选择合适的向量索引，项目特点各异：
- 项目A：小规模文档问答（10万向量）
- 项目B：中型推荐系统（500万向量）
- 项目C：大型图像搜索（5000万向量）
- 项目D：超大规模知识库（10亿向量）

### 为什么需要索引选型？

1. **性能差异巨大**：不同索引在不同场景下性能差异可达10x+
2. **成本影响显著**：错误选择可能导致成本增加5-10倍
3. **迁移代价高**：索引切换需要重建，耗时且影响业务
4. **权衡复杂**：需要在速度、准确率、内存、成本间平衡

### 索引对比总览

| 索引类型 | 适用规模 | 内存占用 | QPS | 召回率 | 成本 | 适用场景 |
|---------|---------|---------|-----|--------|------|---------|
| **FLAT** | <10万 | 100% | 低 | 100% | 低 | 小规模、高精度 |
| **IVF_FLAT** | 10万-500万 | 100% | 中 | 95%+ | 中 | 中等规模、平衡 |
| **IVF_SQ8** | 100万-1000万 | 25% | 高 | 90%+ | 低 | 大规模、内存受限 |
| **IVF_PQ** | 500万-5000万 | 12.5% | 高 | 70-90% | 低 | 超大规模、成本优先 |
| **HNSW** | 100万-5000万 | 高 | 极高 | 98%+ | 高 | 高QPS、高精度 |
| **GPU_CAGRA** | 1000万-1亿 | 1.8x | 极高 | 95%+ | 中 | GPU加速、高吞吐 |
| **RaBitQ** | 10亿+ | 28% | 极高 | 95%+ | 极低 | 十亿级、成本敏感 |

class IndexSelector:
    """索引选型决策引擎"""

    @staticmethod
    def select_index(
        num_vectors: int,
        qps_requirement: int,
        recall_requirement: float,
        memory_budget_gb: Optional[float] = None,
        has_gpu: bool = False,
        cost_sensitive: bool = False
    ) -> Dict:
        """
        基于业务需求自动选择最优索引

        参数:
            num_vectors: 向量数量
            qps_requirement: QPS要求
            recall_requirement: 召回率要求 (0-1)
            memory_budget_gb: 内存预算(GB)，None表示无限制
            has_gpu: 是否有GPU
            cost_sensitive: 是否成本敏感

        返回:
            推荐的索引配置
        """

        # 计算基础内存需求（假设768维float32）
        dimension = 768
        base_memory_gb = num_vectors * dimension * 4 / (1024**3)

        # 决策逻辑
        if num_vectors < 100_000:
            # 小规模：FLAT
            return {
                "index_type": "FLAT",
                "reason": "数据量<10万，FLAT提供100%召回率且无需调参",
                "params": {},
                "search_params": {},
                "estimated_memory_gb": base_memory_gb,
                "estimated_qps": 50,
                "estimated_recall": 1.0
            }

        elif num_vectors < 1_000_000:
            # 中等规模：IVF_FLAT或HNSW
            if qps_requirement > 500:
                return {
                    "index_type": "HNSW",
                    "reason": "QPS>500，HNSW提供最高吞吐量",
                    "params": {"M": 16, "efConstruction": 200},
                    "search_params": {"ef": 64},
                    "estimated_memory_gb": base_memory_gb * 1.5,
                    "estimated_qps": 800,
                    "estimated_recall": 0.98
                }
            else:
                return {
                    "index_type": "IVF_FLAT",
                    "reason": "中等规模，IVF_FLAT平衡性能和成本",
                    "params": {"nlist": 1024},
                    "search_params": {"nprobe": 16},
                    "estimated_memory_gb": base_memory_gb,
                    "estimated_qps": 300,
                    "estimated_recall": 0.95
                }

        elif num_vectors < 10_000_000:
            # 大规模：根据内存和召回率选择
            if memory_budget_gb and base_memory_gb > memory_budget_gb:
                # 内存受限
                return {
                    "index_type": "IVF_SQ8",
                    "reason": "内存受限，IVF_SQ8减少75%内存占用",
                    "params": {"nlist": 2048},
                    "search_params": {"nprobe": 32},
                    "estimated_memory_gb": base_memory_gb * 0.25,
                    "estimated_qps": 400,
                    "estimated_recall": 0.92
                }
            elif qps_requirement > 500 and not cost_sensitive:
                return {
                    "index_type": "HNSW",
                    "reason": "高QPS需求，HNSW提供最佳性能",
                    "params": {"M": 32, "efConstruction": 400},
                    "search_params": {"ef": 128},
                    "estimated_memory_gb": base_memory_gb * 1.8,
                    "estimated_qps": 1000,
                    "estimated_recall": 0.98
                }
            else:
                return {
                    "index_type": "IVF_FLAT",
                    "reason": "大规模平衡方案",
                    "params": {"nlist": 2048},
                    "search_params": {"nprobe": 32},
                    "estimated_memory_gb": base_memory_gb,
                    "estimated_qps": 350,
                    "estimated_recall": 0.95
                }

        elif num_vectors < 100_000_000:
            # 超大规模：GPU或量化
            if has_gpu:
                return {
                    "index_type": "GPU_CAGRA",
                    "reason": "有GPU，GPU_CAGRA提供12-15x构建加速",
                    "params": {
                        "intermediate_graph_degree": 64,
                        "graph_degree": 32,
                        "build_algo": "IVF_PQ"
                    },
                    "search_params": {"itopk_size": 128, "search_width": 4},
                    "estimated_memory_gb": base_memory_gb * 1.8,
                    "estimated_qps": 1200,
                    "estimated_recall": 0.95
                }
            elif cost_sensitive:
                return {
                    "index_type": "IVF_PQ",
                    "reason": "成本敏感，IVF_PQ大幅降低内存",
                    "params": {"nlist": 4096, "m": 16, "nbits": 8},
                    "search_params": {"nprobe": 64},
                    "estimated_memory_gb": base_memory_gb * 0.125,
                    "estimated_qps": 500,
                    "estimated_recall": 0.85
                }
            else:
                return {
                    "index_type": "IVF_SQ8",
                    "reason": "超大规模平衡方案",
                    "params": {"nlist": 4096},
                    "search_params": {"nprobe": 64},
                    "estimated_memory_gb": base_memory_gb * 0.25,
                    "estimated_qps": 450,
                    "estimated_recall": 0.92
                }

        else:
            # 十亿级：RaBitQ
            return {
                "index_type": "IVF_RABITQ",
                "reason": "十亿级规模，RaBitQ提供72%内存节省和4x性能提升",
                "params": {
                    "nlist": 8192,
                    "nbits": 1,
                    "refine": True,
                    "refine_type": "SQ8"
                },
                "search_params": {"nprobe": 128},
                "estimated_memory_gb": base_memory_gb * 0.28,
                "estimated_qps": 900,
                "estimated_recall": 0.95
            }

# 2. 性能基准测试框架
# ============================================================================

class IndexBenchmark:
    """索引性能基准测试"""

    def __init__(self, dimension: int = 768):
        self.dimension = dimension
        self.results = {}

    def create_test_collection(self, index_config: Dict, collection_name: str):
        """创建测试collection"""
        if utility.has_collection(collection_name):
            utility.drop_collection(collection_name)

        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="embeddings", dtype=DataType.FLOAT_VECTOR, dim=self.dimension)
        ]

        schema = CollectionSchema(fields=fields)
        collection = Collection(name=collection_name, schema=schema)

        # 创建索引
        index_params = {
            "index_type": index_config["index_type"],
            "metric_type": "L2",
            "params": index_config["params"]
        }

        collection.create_index(field_name="embeddings", index_params=index_params)
        return collection

    def benchmark(
        self,
        index_config: Dict,
        num_vectors: int,
        num_queries: int = 100
    ) -> Dict:
        """执行基准测试"""

        collection_name = f"benchmark_{index_config['index_type'].lower()}"

        # 创建collection
        collection = self.create_test_collection(index_config, collection_name)

        # 生成测试数据
        embeddings = np.random.rand(num_vectors, self.dimension).astype('float32')

        # 插入数据
        insert_start = time.time()
        batch_size = 10000
        for i in range(0, num_vectors, batch_size):
            end_idx = min(i + batch_size, num_vectors)
            collection.insert([embeddings[i:end_idx].tolist()])
        collection.flush()
        insert_time = time.time() - insert_start

        # 加载collection
        collection.load()

        # 查询测试
        query_vectors = embeddings[:num_queries]
        search_params = {"metric_type": "L2", **index_config["search_params"]}

        latencies = []
        for i in range(num_queries):
            query_start = time.time()
            collection.search(
                data=[query_vectors[i].tolist()],
                anns_field="embeddings",
                param=search_params,
                limit=10
            )
            latencies.append((time.time() - query_start) * 1000)

        qps = num_queries / sum(latencies) * 1000

        # 清理
        collection.release()
        utility.drop_collection(collection_name)

        return {
            "index_type": index_config["index_type"],
            "insert_time": insert_time,
            "avg_latency": np.mean(latencies),
            "p95_latency": np.percentile(latencies, 95),
            "qps": qps
        }

class IndexMigration:
    """索引迁移策略"""

    @staticmethod
    def suggest_migration(
        current_index: str,
        num_vectors: int,
        current_qps: int,
        target_qps: int
    ) -> Dict:
        """建议索引迁移策略"""

        migrations = {
            "FLAT": {
                "trigger": num_vectors > 100_000,
                "target": "IVF_FLAT",
                "reason": "数据量超过10万，FLAT性能下降",
                "steps": [
                    "1. 创建新collection with IVF_FLAT索引",
                    "2. 使用bulk_insert迁移数据",
                    "3. 验证召回率（应>95%）",
                    "4. 切换流量到新collection",
                    "5. 删除旧collection"
                ]
            },
            "IVF_FLAT": {
                "trigger": current_qps < target_qps * 0.5,
                "target": "HNSW",
                "reason": "QPS不足，需要更高性能索引",
                "steps": [
                    "1. 评估内存预算（HNSW需要1.5-2x内存）",
                    "2. 创建新collection with HNSW索引",
                    "3. 迁移数据并调优M和ef参数",
                    "4. 性能测试确认QPS提升",
                    "5. 灰度切换流量"
                ]
            },
            "IVF_SQ8": {
                "trigger": num_vectors > 50_000_000,
                "target": "RaBitQ",
                "reason": "数据量超过5000万，需要更激进的压缩",
                "steps": [
                    "1. 升级Milvus到2.6+版本",
                    "2. 创建IVF_RABITQ索引",
                    "3. 启用SQ8精炼保证召回率",
                    "4. 验证内存节省（应达72%）",
                    "5. 逐步迁移数据"
                ]
            }
        }

        if current_index in migrations:
            migration = migrations[current_index]
            if migration["trigger"]:
                return {
                    "should_migrate": True,
                    "current_index": current_index,
                    "target_index": migration["target"],
                    "reason": migration["reason"],
                    "steps": migration["steps"]
                }

        return {
            "should_migrate": False,
            "message": "当前索引适合当前规模"
        }
