## 完整代码

```python
"""
LangGraph 实战代码 - 场景6：审批流程实战
演示：构建人机协作的文档审批系统
"""

from langgraph.graph import StateGraph, START, END
from typing import Literal
from typing_extensions import TypedDict


# ===== 1. 定义状态 =====
class ApprovalState(TypedDict):
    """审批状态"""
    document_id: str        # 文档ID
    document_title: str     # 文档标题
    document_content: str   # 文档内容
    risk_score: float       # 风险分数
    approval_status: str    # 审批状态
    approver: str           # 审批人
    approval_comment: str   # 审批意见
    approval_level: int     # 审批级别
    log: list[str]          # 操作日志


# ===== 2. 定义节点函数 =====
def analyze_document(state: ApprovalState) -> dict:
    """分析文档风险"""
    print(f"[文档分析] 文档ID: {state['document_id']}")
    print(f"[文档分析] 标题: {state['document_title']}")

    # 模拟风险评分（基于内容长度和关键词）
    content = state["document_content"]
    risk_score = 0.0

    # 长度评分
    if len(content) > 500:
        risk_score += 0.3
    elif len(content) > 200:
        risk_score += 0.2
    else:
        risk_score += 0.1

    # 关键词评分
    high_risk_keywords = ["机密", "敏感", "重要"]
    for keyword in high_risk_keywords:
        if keyword in content:
            risk_score += 0.3

    # 确保分数在 0-1 范围内
    risk_score = min(1.0, risk_score)

    print(f"[文档分析] 风险分数: {risk_score:.2f}")

    return {
        "risk_score": risk_score,
        "approval_status": "pending",
        "log": state.get("log", []) + [f"文档分析完成，风险分数: {risk_score:.2f}"]
    }


def auto_approve(state: ApprovalState) -> dict:
    """自动批准"""
    print("[自动批准] 低风险文档，自动批准")

    return {
        "approval_status": "approved",
        "approver": "system",
        "approval_comment": "低风险文档，系统自动批准",
        "log": state.get("log", []) + ["文档已自动批准"]
    }


def manager_review(state: ApprovalState) -> dict:
    """经理审批"""
    print("[经理审批] 中等风险文档，需要经理审批")

    # 模拟经理审批（这里简化为自动批准）
    # 实际应用中，这里应该是人工审批的中断点
    approval_decision = "approved"  # 实际应该由人工输入

    print(f"[经理审批] 审批决策: {approval_decision}")

    return {
        "approval_status": approval_decision,
        "approver": "manager",
        "approval_comment": "经理审批通过",
        "approval_level": 1,
        "log": state.get("log", []) + ["经理审批完成"]
    }


def director_review(state: ApprovalState) -> dict:
    """总监审批"""
    print("[总监审批] 高风险文档，需要总监审批")

    # 模拟总监审批（这里简化为自动批准）
    # 实际应用中，这里应该是人工审批的中断点
    approval_decision = "approved"  # 实际应该由人工输入

    print(f"[总监审批] 审批决策: {approval_decision}")

    return {
        "approval_status": approval_decision,
        "approver": "director",
        "approval_comment": "总监审批通过",
        "approval_level": 2,
        "log": state.get("log", []) + ["总监审批完成"]
    }


def reject_document(state: ApprovalState) -> dict:
    """拒绝文档"""
    print("[拒绝文档] 文档被拒绝")

    return {
        "approval_status": "rejected",
        "approval_comment": "文档不符合要求",
        "log": state.get("log", []) + ["文档已拒绝"]
    }


# ===== 3. 定义路由函数 =====
def route_by_risk(state: ApprovalState) -> Literal["auto_approve", "manager_review", "director_review"]:
    """根据风险分数路由"""
    risk_score = state["risk_score"]

    print(f"[路由决策] 风险分数: {risk_score:.2f}")

    # 低风险：自动批准
    if risk_score < 0.3:
        print("[路由决策] → auto_approve (低风险)")
        return "auto_approve"

    # 中等风险：经理审批
    elif risk_score < 0.7:
        print("[路由决策] → manager_review (中等风险)")
        return "manager_review"

    # 高风险：总监审批
    else:
        print("[路由决策] → director_review (高风险)")
        return "director_review"


def route_after_approval(state: ApprovalState) -> Literal["reject", END]:
    """审批后的路由决策"""
    approval_status = state["approval_status"]

    print(f"[路由决策] 审批状态: {approval_status}")

    if approval_status == "approved":
        print("[路由决策] → END (已批准)")
        return END
    else:
        print("[路由决策] → reject (拒绝)")
        return "reject"


# ===== 4. 构建图 =====
def create_approval_graph():
    """创建审批流程图"""
    # 创建状态图
    graph = StateGraph(ApprovalState)

    # 添加节点
    graph.add_node("analyze", analyze_document)
    graph.add_node("auto_approve", auto_approve)
    graph.add_node("manager_review", manager_review)
    graph.add_node("director_review", director_review)
    graph.add_node("reject", reject_document)

    # 添加边
    graph.add_edge(START, "analyze")

    # 第一层条件路由：根据风险分数决定审批路径
    graph.add_conditional_edges(
        "analyze",
        route_by_risk,
        {
            "auto_approve": "auto_approve",
            "manager_review": "manager_review",
            "director_review": "director_review"
        }
    )

    # 自动批准后结束
    graph.add_edge("auto_approve", END)

    # 经理审批后的路由
    graph.add_conditional_edges(
        "manager_review",
        route_after_approval,
        {
            "reject": "reject",
            END: END
        }
    )

    # 总监审批后的路由
    graph.add_conditional_edges(
        "director_review",
        route_after_approval,
        {
            "reject": "reject",
            END: END
        }
    )

    # 拒绝后结束
    graph.add_edge("reject", END)

    # 编译图
    app = graph.compile()

    return app


# ===== 5. 测试不同场景 =====
def test_low_risk():
    """测试低风险文档"""
    print("=" * 60)
    print("测试场景 1：低风险文档")
    print("=" * 60)

        "approval_level": 0,

    print(f"  审批人: {result['approver']}")
    print(f"  审批意见: {result['approval_comment']}")

def test_medium_risk():
    """测试中等风险文档"""
    print("=" * 60)

    print("=" * 60)
    print()

    initial_state = {
        "document_id": "DOC002",

        "risk_score": 0.0,
        "approval_status": "",
        "approver": "",

    }

    result = app.invoke(initial_state)

    print()

    print(f"  文档ID: {result['document_id']}")

    print(f"  审批级别: {result['approval_level']}")

def test_high_risk():
    """测试高风险文档"""
    print("=" * 60)
    print("测试场景 3：高风险文档")
    print("=" * 60)
    print()

    app = create_approval_graph()

    initial_state = {
        "document_id": "DOC003",
        "document_title": "机密技术文档",
        "document_content": "这是一份机密的技术文档，包含敏感的技术信息和重要的商业机密。" * 10,
        "risk_score": 0.0,
        "approval_status": "",
        "approver": "",
        "approval_comment": "",
        "approval_level": 0,
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("最终结果：")
    print(f"  文档ID: {result['document_id']}")

    print(f"  审批级别: {result['approval_level']}")

# ===== 6. 主函数 =====
def main():
    """主函数"""
    print("=" * 60)
    print("LangGraph 实战 - 场景6：审批流程实战")
    print("=" * 60)

    test_low_risk()

if __name__ == "__main__":
    main()
```

## 代码详解

### 1. 多级审批路由

```python
# 第一层：根据风险分数决定审批路径
graph.add_conditional_edges(
    "analyze",
    route_by_risk,
    {
        "auto_approve": "auto_approve",      # 低风险 → 自动批准
        "manager_review": "manager_review",  # 中等风险 → 经理审批
        "director_review": "director_review" # 高风险 → 总监审批
    }
)

# 第二层：审批后的决策
graph.add_conditional_edges(
    "manager_review",
    route_after_approval,
    {
        "reject": "reject",
        END: END
    }
)
```

**关键点**：
- **多级审批**：根据风险级别路由到不同审批人
- **审批决策**：审批后根据结果决定是否拒绝
- **人机协作**：审批节点是人工介入点

---

### 2. 执行流程图

```
START
  ↓
analyze (文档分析)
  ↓
route_by_risk
  ↓
┌─────┼─────┼─────┐
↓     ↓     ↓
auto_approve manager_review director_review
(自动批准)  (经理审批)    (总监审批)
↓           ↓             ↓
END    route_after_approval route_after_approval
            ↓             ↓
       ┌────┼────┐   ┌────┼────┐
       ↓    ↓    ↓   ↓    ↓    ↓
    reject END  reject END
    (拒绝) (批准) (拒绝) (批准)
       ↓         ↓
      END       END
```

**执行路径**：
- **低风险**：analyze → auto_approve → END
- **中等风险**：analyze → manager_review → END
- **高风险**：analyze → director_review → END

```python
class ExpenseState(TypedDict):
```

### 2. 审批历史记录

```python
def record_approval(state: ApprovalState) -> dict:
    """记录审批历史"""
    approval_record = {
        "timestamp": datetime.now(),
        "approver": state["approver"],
        "decision": state["approval_status"],
        "comment": state["approval_comment"]
    }

    return {
        "log": state.get("log", []) + [approval_record]
    }
```

### 3. 审批超时处理

```python
def route_with_timeout(state: ApprovalState) -> Literal["escalate", "auto_reject", END]:
    """带超时的审批路由"""
    approval_time = state.get("approval_time", 0)

    # 超时升级
    if approval_time > 24:  # 24小时
        return "escalate"

    # 审批完成
    if state["approval_status"] == "approved":
        return END

    return "auto_reject"
```
