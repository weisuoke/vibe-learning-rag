### 1. src/channel-selector.ts

```typescript
/**
 * 通道选择器
 * 根据策略选择要发送的通道
 */

export interface ChannelInfo {
  type: string;
  status: 'active' | 'inactive' | 'error';
  priority: number;
  lastUsed: number;
}

export type SelectionStrategy = 'all' | 'primary' | 'fallback' | 'round-robin';

export class ChannelSelector {
  private channels: Map<string, ChannelInfo> = new Map();
  private roundRobinIndex = 0;

  /**
   * 注册通道
   */
  registerChannel(type: string, info: Omit<ChannelInfo, 'type'>): void {
    this.channels.set(type, { type, ...info });
  }

  /**
   * 选择通道
   */
  selectChannels(strategy: SelectionStrategy): string[] {
    const activeChannels = Array.from(this.channels.values())
      .filter(ch => ch.status === 'active')
      .sort((a, b) => a.priority - b.priority);

    switch (strategy) {
      case 'all':
        return activeChannels.map(ch => ch.type);

      case 'primary':
        return activeChannels.length > 0 ? [activeChannels[0].type] : [];

      case 'fallback':
        // 尝试主通道，失败则使用备用通道
        return activeChannels.map(ch => ch.type);

      case 'round-robin':
        if (activeChannels.length === 0) return [];
        const selected = activeChannels[this.roundRobinIndex % activeChannels.length];
        this.roundRobinIndex++;
        return [selected.type];

      default:
        return [];
    }
  }

  /**
   * 更新通道状态
   */
  updateChannelStatus(type: string, status: ChannelInfo['status']): void {
    const channel = this.channels.get(type);
    if (channel) {
      channel.status = status;
      channel.lastUsed = Date.now();
    }
  }

  /**
   * 获取通道信息
   */
  getChannelInfo(type: string): ChannelInfo | undefined {
    return this.channels.get(type);
  }

  /**
   * 获取所有通道
   */
  getAllChannels(): ChannelInfo[] {
    return Array.from(this.channels.values());
  }
}
```

### 2. src/result-aggregator.ts

```typescript
/**
 * 结果聚合器
 * 收集和分析多通道发送结果
 */

export interface SendResult {
  channel: string;
  success: boolean;
  messageId?: string;
  error?: string;
  timestamp: number;
  duration: number;
}

export interface AggregatedResult {
  totalChannels: number;
  successCount: number;
  failureCount: number;
  results: SendResult[];
  summary: string;
}

export class ResultAggregator {
  /**
   * 聚合结果
   */
  aggregate(results: SendResult[]): AggregatedResult {
    const successCount = results.filter(r => r.success).length;
    const failureCount = results.length - successCount;

    return {
      totalChannels: results.length,
      successCount,
      failureCount,
      results,
      summary: this.generateSummary(results, successCount, failureCount)
    };
  }

  /**
   * 生成摘要
   */
  private generateSummary(
    results: SendResult[],
    successCount: number,
    failureCount: number
  ): string {
    const total = results.length;
    const successRate = ((successCount / total) * 100).toFixed(1);

    let summary = `发送完成：${successCount}/${total} 成功 (${successRate}%)`;

    if (failureCount > 0) {
      const failedChannels = results
        .filter(r => !r.success)
        .map(r => r.channel)
        .join(', ');
      summary += `\n失败通道：${failedChannels}`;
    }

    return summary;
  }

  /**
   * 显示详细结果
   */
  displayResults(aggregated: AggregatedResult): void {
    console.log('\n' + '='.repeat(60));
    console.log('📊 多通道发送结果');
    console.log('='.repeat(60));
    console.log(aggregated.summary);
    console.log('\n详细结果：');

    for (const result of aggregated.results) {
      const icon = result.success ? '✅' : '❌';
      const status = result.success ? '成功' : '失败';
      const duration = `${result.duration}ms`;

      console.log(`${icon} ${result.channel}: ${status} (${duration})`);

      if (result.success && result.messageId) {
        console.log(`   消息ID: ${result.messageId}`);
      }

      if (!result.success && result.error) {
        console.log(`   错误: ${result.error}`);
      }
    }

    console.log('='.repeat(60) + '\n');
  }
}
```

### 3. src/multi-channel-sender.ts

```typescript
/**
 * 多通道发送器
 * 核心发送逻辑
 */

import { exec } from 'child_process';
import { promisify } from 'util';
import { ChannelSelector, SelectionStrategy } from './channel-selector.js';
import { ResultAggregator, SendResult } from './result-aggregator.js';

const execAsync = promisify(exec);

export interface SendOptions {
  text: string;
  strategy?: SelectionStrategy;
  parallel?: boolean;
  retryOnFailure?: boolean;
  maxRetries?: number;
}

export class MultiChannelSender {
  private selector: ChannelSelector;
  private aggregator: ResultAggregator;

  constructor() {
    this.selector = new ChannelSelector();
    this.aggregator = new ResultAggregator();
  }

  /**
   * 初始化通道
   */
  async initialize(): Promise<void> {
    console.log('🔄 初始化通道...\n');

    // 获取已配对的通道
    const { stdout } = await execAsync('openclaw pairing list');
    const channels = this.parsePairingList(stdout);

    // 注册通道
    channels.forEach((channel, index) => {
      this.selector.registerChannel(channel, {
        status: 'active',
        priority: index + 1,
        lastUsed: 0
      });
    });

    console.log(`✅ 已加载 ${channels.length} 个通道\n`);
  }

  /**
   * 发送消息到多个通道
   */
  async send(options: SendOptions): Promise<void> {
    const {
      text,
      strategy = 'all',
      parallel = true,
      retryOnFailure = true,
      maxRetries = 3
    } = options;

    // 选择通道
    const selectedChannels = this.selector.selectChannels(strategy);

    if (selectedChannels.length === 0) {
      console.log('⚠️  没有可用的通道');
      return;
    }

    console.log(`📤 发送消息到 ${selectedChannels.length} 个通道...`);
    console.log(`策略: ${strategy}`);
    console.log(`并行: ${parallel ? '是' : '否'}\n`);

    // 发送消息
    const results = parallel
      ? await this.sendParallel(text, selectedChannels, retryOnFailure, maxRetries)
      : await this.sendSequential(text, selectedChannels, retryOnFailure, maxRetries);

    // 聚合和显示结果
    const aggregated = this.aggregator.aggregate(results);
    this.aggregator.displayResults(aggregated);
  }

  /**
   * 并行发送
   */
  private async sendParallel(
    text: string,
    channels: string[],
    retry: boolean,
    maxRetries: number
  ): Promise<SendResult[]> {
    const promises = channels.map(channel =>
      this.sendToChannel(text, channel, retry, maxRetries)
    );

    return await Promise.all(promises);
  }

  /**
   * 顺序发送
   */
  private async sendSequential(
    text: string,
    channels: string[],
    retry: boolean,
    maxRetries: number
  ): Promise<SendResult[]> {
    const results: SendResult[] = [];

    for (const channel of channels) {
      const result = await this.sendToChannel(text, channel, retry, maxRetries);
      results.push(result);

      // 短暂延迟避免频率限制
      await this.delay(500);
    }

    return results;
  }

  /**
   * 发送到单个通道
   */
  private async sendToChannel(
    text: string,
    channel: string,
    retry: boolean,
    maxRetries: number
  ): Promise<SendResult> {
    const startTime = Date.now();
    let lastError: string | undefined;

    const attempts = retry ? maxRetries : 1;

    for (let i = 0; i < attempts; i++) {
      try {
        const { stdout } = await execAsync(
          `openclaw message send "${text}" --channel ${channel}`
        );

        // 提取消息ID
        const messageIdMatch = stdout.match(/Message ID: (.+)/);
        const messageId = messageIdMatch ? messageIdMatch[1] : undefined;

        // 更新通道状态
        this.selector.updateChannelStatus(channel, 'active');

        return {
          channel,
          success: true,
          messageId,
          timestamp: Date.now(),
          duration: Date.now() - startTime
        };
      } catch (error) {
        lastError = error instanceof Error ? error.message : 'Unknown error';

        if (i < attempts - 1) {
          // 重试前等待
          await this.delay(1000 * (i + 1));
        }
      }
    }

    // 所有重试都失败
    this.selector.updateChannelStatus(channel, 'error');

    return {
      channel,
      success: false,
      error: lastError,
      timestamp: Date.now(),
      duration: Date.now() - startTime
    };
  }

  /**
   * 解析配对列表
   */
  private parsePairingList(output: string): string[] {
    const lines = output.split('\n').filter(line => line.includes('│'));
    const channels: string[] = [];

    for (const line of lines) {
      if (line.includes('Channel') || line.includes('─')) {
        continue;
      }

      const parts = line.split('│').map(p => p.trim()).filter(p => p);
      if (parts.length >= 1) {
        channels.push(parts[0].toLowerCase());
      }
    }

    return channels;
  }

  /**
   * 获取通道选择器
   */
  getSelector(): ChannelSelector {
    return this.selector;
  }

  /**
   * 延迟函数
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

```typescript
/**
 * OpenClaw 多通道消息发送示例
 */

import 'dotenv/config';
import { MultiChannelSender } from './multi-channel-sender.js';

async function main() {
  console.log('🎯 OpenClaw Multi-Channel Messaging Example\n');

  const sender = new MultiChannelSender();

  try {
    // 初始化
    await sender.initialize();

    // 场景1：发送到所有通道（并行）
    console.log('场景1：发送到所有通道（并行）');
    console.log('-'.repeat(60));
    await sender.send({
      text: 'Hello from all channels! 👋',
      strategy: 'all',
      parallel: true
    });

    // 等待一段时间
    await delay(2000);

    // 场景2：只发送到主通道
    console.log('\n场景2：只发送到主通道');
    console.log('-'.repeat(60));
    await sender.send({
      text: 'Primary channel message',
      strategy: 'primary'
    });

    await delay(2000);

    // 场景3：轮询发送
    console.log('\n场景3：轮询发送（3条消息）');
    console.log('-'.repeat(60));
    for (let i = 1; i <= 3; i++) {
      await sender.send({
        text: `Round-robin message ${i}`,
        strategy: 'round-robin'
      });
      await delay(1000);
    }

    // 场景4：顺序发送（带重试）
    console.log('\n场景4：顺序发送（带重试）');
    console.log('-'.repeat(60));
    await sender.send({
      text: 'Sequential message with retry',
      strategy: 'all',
      parallel: false,
      retryOnFailure: true,
      maxRetries: 3
    });

    console.log('\n✅ 所有场景执行完成！');
  } catch (error) {
    console.error('\n❌ 错误:', error);
    process.exit(1);
  }
}

function delay(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms));
}

// 运行主函数
main().catch(console.error);
```

### 3. 性能监控

```typescript
class PerformanceMonitor {
  private metrics: Map<string, number[]> = new Map();

  recordDuration(channel: string, duration: number): void {
    if (!this.metrics.has(channel)) {
      this.metrics.set(channel, []);
    }
    this.metrics.get(channel)!.push(duration);
  }

  getAverageDuration(channel: string): number {
    const durations = this.metrics.get(channel) || [];
    if (durations.length === 0) return 0;
    return durations.reduce((a, b) => a + b, 0) / durations.length;
  }

  displayReport(): void {
    console.log('\n📈 性能报告：\n');
    for (const [channel, durations] of this.metrics) {
      const avg = this.getAverageDuration(channel);
      console.log(`${channel}: 平均 ${avg.toFixed(0)}ms (${durations.length} 次)`);
    }
  }
}
```
