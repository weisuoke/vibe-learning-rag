# 实战代码 - 场景4：BM25 混合检索

> **场景说明**：实现 BM25 稀疏向量 + 稠密向量的混合检索，兼顾语义理解和精确关键词匹配

## 场景概述

### 应用场景
- **文档问答系统**：需要同时支持语义搜索和关键词精确匹配
- **知识库检索**：用户查询可能包含专业术语，需要精确匹配
- **企业搜索**：结合语义理解和关键词匹配提升搜索质量
- **RAG 系统**：提升检索召回率和准确率

### 核心优势
- **语义理解**：稠密向量捕获查询的语义含义
- **精确匹配**：BM25 稀疏向量实现关键词精确匹配
- **互补增强**：两种检索方式互补，提升整体检索质量
- **自动化**：BM25 函数自动生成稀疏向量，无需手动处理

### 技术要点
1. 使用 BM25 函数自动生成稀疏向量
2. 定义稠密向量和稀疏向量两个字段
3. 使用 RRFRanker 融合两种检索结果
4. 文本字段需要启用 `enable_analyzer=True`

## 完整代码实现

```python
"""
BM25 + 稠密向量混合检索实战
场景：文档问答系统，支持语义搜索和关键词精确匹配
"""

from pymilvus import (
    MilvusClient,
    DataType,
    Function,
    FunctionType,
    AnnSearchRequest,
    RRFRanker,
    WeightedRanker,
)
import numpy as np
from typing import List, Dict, Any
import os
from dotenv import load_dotenv

# 加载环境变量
load_dotenv()

# ============================================================================
# 第一部分：Collection 创建与 BM25 函数配置
# ============================================================================

class BM25HybridSearchSystem:
    """BM25 混合检索系统"""

    def __init__(self, uri: str = "http://localhost:19530"):
        """初始化系统

        Args:
            uri: Milvus 服务地址
        """
        self.client = MilvusClient(uri=uri)
        self.collection_name = "bm25_hybrid_search_demo"
        self.dense_dim = 768  # 稠密向量维度

    def create_collection_with_bm25(self):
        """创建支持 BM25 的 Collection

        关键点：
        1. 文本字段需要启用 enable_analyzer=True
        2. 定义稠密向量字段和稀疏向量字段
        3. 添加 BM25 函数自动生成稀疏向量
        4. 为两个向量字段创建索引
        """
        # 删除已存在的 Collection
        if self.client.has_collection(self.collection_name):
            self.client.drop_collection(self.collection_name)
            print(f"已删除旧的 Collection: {self.collection_name}")

        # 定义 Schema
        schema = self.client.create_schema(
            auto_id=True,
            enable_dynamic_field=False,
        )

        # 1. 主键字段
        schema.add_field(
            field_name="id",
            datatype=DataType.INT64,
            is_primary=True,
            auto_id=True,
        )

        # 2. 文本字段（需要启用 analyzer）
        schema.add_field(
            field_name="text",
            datatype=DataType.VARCHAR,
            max_length=65535,
            enable_analyzer=True,  # 关键：启用分词器
        )

        # 3. 稠密向量字段
        schema.add_field(
            field_name="dense_vector",
            datatype=DataType.FLOAT_VECTOR,
            dim=self.dense_dim,
        )

        # 4. 稀疏向量字段（由 BM25 函数生成）
        schema.add_field(
            field_name="sparse_vector",
            datatype=DataType.SPARSE_FLOAT_VECTOR,
        )

        # 5. 元数据字段
        schema.add_field(
            field_name="doc_id",
            datatype=DataType.VARCHAR,
            max_length=100,
        )

        schema.add_field(
            field_name="category",
            datatype=DataType.VARCHAR,
            max_length=50,
        )

        # 6. 添加 BM25 函数
        bm25_function = Function(
            name="bm25_fn",
            function_type=FunctionType.BM25,
            input_field_names=["text"],  # 输入：文本字段
            output_field_names=["sparse_vector"],  # 输出：稀疏向量字段
            params={},
        )
        schema.add_function(bm25_function)

        # 7. 创建索引配置
        index_params = self.client.prepare_index_params()

        # 稠密向量索引（HNSW）
        index_params.add_index(
            field_name="dense_vector",
            index_type="HNSW",
            metric_type="IP",
            params={"M": 16, "efConstruction": 200},
        )

        # 稀疏向量索引（SPARSE_INVERTED_INDEX）
        index_params.add_index(
            field_name="sparse_vector",
            index_type="SPARSE_INVERTED_INDEX",
            metric_type="IP",
        )

        # 8. 创建 Collection
        self.client.create_collection(
            collection_name=self.collection_name,
            schema=schema,
            index_params=index_params,
        )

        print(f"✓ 创建 Collection: {self.collection_name}")
        print(f"  - 稠密向量维度: {self.dense_dim}")
        print(f"  - BM25 函数: text -> sparse_vector")
        print(f"  - 索引类型: HNSW (dense) + SPARSE_INVERTED_INDEX (sparse)")


# ============================================================================
# 第二部分：数据插入（自动生成 BM25 向量）
# ============================================================================

    def insert_documents(self, documents: List[Dict[str, Any]]):
        """插入文档数据

        Args:
            documents: 文档列表，每个文档包含 text, dense_vector, doc_id, category

        关键点：
        1. 只需提供 text 和 dense_vector
        2. sparse_vector 由 BM25 函数自动生成
        3. 插入后自动触发 BM25 计算
        """
        # 准备插入数据
        data = []
        for doc in documents:
            data.append({
                "text": doc["text"],
                "dense_vector": doc["dense_vector"],
                "doc_id": doc["doc_id"],
                "category": doc["category"],
                # 注意：不需要提供 sparse_vector，由 BM25 函数自动生成
            })

        # 插入数据
        result = self.client.insert(
            collection_name=self.collection_name,
            data=data,
        )

        print(f"✓ 插入 {result['insert_count']} 条文档")
        return result


# ============================================================================
# 第三部分：BM25 混合检索实现
# ============================================================================

    def hybrid_search_with_bm25(
        self,
        query_text: str,
        query_dense_vector: List[float],
        limit: int = 5,
        use_rrf: bool = True,
        sparse_weight: float = 1.0,
        dense_weight: float = 1.0,
    ) -> List[Dict[str, Any]]:
        """BM25 + 稠密向量混合检索

        Args:
            query_text: 查询文本（用于 BM25 检索）
            query_dense_vector: 查询稠密向量
            limit: 返回结果数量
            use_rrf: 是否使用 RRFRanker（否则使用 WeightedRanker）
            sparse_weight: 稀疏向量权重（仅 WeightedRanker 使用）
            dense_weight: 稠密向量权重（仅 WeightedRanker 使用）

        Returns:
            检索结果列表
        """
        # 1. 定义 BM25 稀疏向量检索请求
        sparse_request = AnnSearchRequest(
            data=[query_text],  # 直接传入文本，Milvus 自动调用 BM25
            anns_field="sparse_vector",
            param={"metric_type": "BM25"},
            limit=limit,
        )

        # 2. 定义稠密向量检索请求
        dense_request = AnnSearchRequest(
            data=[query_dense_vector],
            anns_field="dense_vector",
            param={"metric_type": "IP"},
            limit=limit,
        )

        # 3. 选择 Ranker
        if use_rrf:
            ranker = RRFRanker()
            ranker_name = "RRFRanker"
        else:
            ranker = WeightedRanker(sparse_weight, dense_weight)
            ranker_name = f"WeightedRanker({sparse_weight}, {dense_weight})"

        # 4. 执行混合检索
        results = self.client.hybrid_search(
            collection_name=self.collection_name,
            reqs=[sparse_request, dense_request],
            ranker=ranker,
            limit=limit,
            output_fields=["text", "doc_id", "category"],
        )

        print(f"\n{'='*60}")
        print(f"混合检索结果 (Ranker: {ranker_name})")
        print(f"{'='*60}")

        # 5. 格式化结果
        formatted_results = []
        for i, hits in enumerate(results):
            for j, hit in enumerate(hits):
                result = {
                    "rank": j + 1,
                    "score": hit["distance"],
                    "doc_id": hit["entity"]["doc_id"],
                    "category": hit["entity"]["category"],
                    "text": hit["entity"]["text"][:100] + "...",
                }
                formatted_results.append(result)

                print(f"\n[{j+1}] Score: {hit['distance']:.4f}")
                print(f"    Doc ID: {hit['entity']['doc_id']}")
                print(f"    Category: {hit['entity']['category']}")
                print(f"    Text: {hit['entity']['text'][:100]}...")

        return formatted_results


# ============================================================================
# 第四部分：对比实验（单独检索 vs 混合检索）
# ============================================================================

    def compare_search_methods(
        self,
        query_text: str,
        query_dense_vector: List[float],
        limit: int = 3,
    ):
        """对比不同检索方法的效果

        对比：
        1. 仅 BM25 稀疏向量检索
        2. 仅稠密向量检索
        3. RRFRanker 混合检索
        4. WeightedRanker 混合检索
        """
        print(f"\n{'='*60}")
        print(f"检索对比实验")
        print(f"查询文本: {query_text}")
        print(f"{'='*60}")

        # 1. 仅 BM25 检索
        print(f"\n【方法1】仅 BM25 稀疏向量检索")
        print("-" * 60)
        sparse_results = self.client.search(
            collection_name=self.collection_name,
            data=[query_text],
            anns_field="sparse_vector",
            search_params={"metric_type": "BM25"},
            limit=limit,
            output_fields=["text", "doc_id", "category"],
        )
        self._print_results(sparse_results[0])

        # 2. 仅稠密向量检索
        print(f"\n【方法2】仅稠密向量检索")
        print("-" * 60)
        dense_results = self.client.search(
            collection_name=self.collection_name,
            data=[query_dense_vector],
            anns_field="dense_vector",
            search_params={"metric_type": "IP"},
            limit=limit,
            output_fields=["text", "doc_id", "category"],
        )
        self._print_results(dense_results[0])

        # 3. RRFRanker 混合检索
        print(f"\n【方法3】RRFRanker 混合检索")
        print("-" * 60)
        self.hybrid_search_with_bm25(
            query_text=query_text,
            query_dense_vector=query_dense_vector,
            limit=limit,
            use_rrf=True,
        )

        # 4. WeightedRanker 混合检索（BM25 权重更高）
        print(f"\n【方法4】WeightedRanker 混合检索 (BM25权重=1.5, 稠密=1.0)")
        print("-" * 60)
        self.hybrid_search_with_bm25(
            query_text=query_text,
            query_dense_vector=query_dense_vector,
            limit=limit,
            use_rrf=False,
            sparse_weight=1.5,
            dense_weight=1.0,
        )

    def _print_results(self, hits):
        """打印检索结果"""
        for i, hit in enumerate(hits):
            print(f"\n[{i+1}] Score: {hit['distance']:.4f}")
            print(f"    Doc ID: {hit['entity']['doc_id']}")
            print(f"    Category: {hit['entity']['category']}")
            print(f"    Text: {hit['entity']['text'][:100]}...")


# ============================================================================
# 第五部分：模拟 Embedding 函数（实际应用中使用真实模型）
# ============================================================================

def generate_mock_embedding(text: str, dim: int = 768) -> List[float]:
    """生成模拟的稠密向量

    实际应用中应使用：
    - OpenAI text-embedding-ada-002
    - sentence-transformers
    - 其他 Embedding 模型
    """
    np.random.seed(hash(text) % (2**32))
    embedding = np.random.randn(dim).astype(np.float32)
    # 归一化（用于 IP 距离）
    embedding = embedding / np.linalg.norm(embedding)
    return embedding.tolist()


# ============================================================================
# 第六部分：完整示例
# ============================================================================

def main():
    """完整示例：BM25 混合检索系统"""

    # 1. 初始化系统
    print("=" * 60)
    print("BM25 混合检索系统 - 完整示例")
    print("=" * 60)

    system = BM25HybridSearchSystem(uri="http://localhost:19530")

    # 2. 创建 Collection
    system.create_collection_with_bm25()

    # 3. 准备示例文档
    documents = [
        {
            "text": "Python is a high-level programming language known for its simplicity and readability.",
            "dense_vector": generate_mock_embedding("Python programming language"),
            "doc_id": "doc_001",
            "category": "programming",
        },
        {
            "text": "Machine learning is a subset of artificial intelligence that enables systems to learn from data.",
            "dense_vector": generate_mock_embedding("Machine learning AI"),
            "doc_id": "doc_002",
            "category": "ai",
        },
        {
            "text": "Vector databases like Milvus are optimized for similarity search on high-dimensional vectors.",
            "dense_vector": generate_mock_embedding("Vector database Milvus"),
            "doc_id": "doc_003",
            "category": "database",
        },
        {
            "text": "Natural language processing (NLP) helps computers understand and generate human language.",
            "dense_vector": generate_mock_embedding("NLP natural language"),
            "doc_id": "doc_004",
            "category": "ai",
        },
        {
            "text": "RAG (Retrieval-Augmented Generation) combines retrieval and generation for better AI responses.",
            "dense_vector": generate_mock_embedding("RAG retrieval generation"),
            "doc_id": "doc_005",
            "category": "ai",
        },
    ]

    # 4. 插入文档
    system.insert_documents(documents)

    # 5. 执行混合检索
    query_text = "vector database similarity search"
    query_vector = generate_mock_embedding(query_text)

    # 6. 对比不同检索方法
    system.compare_search_methods(
        query_text=query_text,
        query_dense_vector=query_vector,
        limit=3,
    )

    print(f"\n{'='*60}")
    print("示例完成")
    print(f"{'='*60}")


if __name__ == "__main__":
    main()
```

## 关键要点总结

### 1. BM25 函数优势
- **自动化**：无需手动计算稀疏向量
- **高效**：Milvus 内置优化的 BM25 实现
- **准确**：基于经典的 BM25 算法

### 2. 混合检索优势
- **语义理解**：稠密向量捕获语义关系
- **精确匹配**：BM25 实现关键词精确匹配
- **互补增强**：两种方式互补，提升检索质量

### 3. Ranker 选择
- **RRFRanker**：不确定权重时使用，自动平衡
- **WeightedRanker**：已知权重时使用，精确控制

### 4. 实际应用建议
- 使用真实的 Embedding 模型（OpenAI、sentence-transformers）
- 根据业务场景调整权重
- 监控检索质量，持续优化
