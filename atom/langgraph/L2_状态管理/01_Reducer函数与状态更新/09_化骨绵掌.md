## 技巧 1: 使用工厂函数创建参数化 Reducer

### 适用场景
需要创建可配置的 Reducer，如保留最新 N 个元素、限制列表长度等。

### 实现原理
使用闭包（closure）捕获配置参数，返回符合 Reducer 签名的函数。

### 代码示例

```python
from typing import Annotated, TypeVar
from typing_extensions import TypedDict
from langgraph.graph import StateGraph

T = TypeVar('T')

def keep_last_n(n: int):
    """工厂函数: 保留最新 N 个元素"""
    def reducer(old: list[T], new: list[T]) -> list[T]:
        combined = old + new
        return combined[-n:]
    return reducer

def limit_length(max_length: int):
    """工厂函数: 限制字符串长度"""
    def reducer(old: str, new: str) -> str:
        combined = old + new
        return combined[:max_length]
    return reducer

def weighted_sum(weight_old: float, weight_new: float):
    """工厂函数: 加权求和"""
    def reducer(old: float, new: float) -> float:
        return old * weight_old + new * weight_new
    return reducer

class State(TypedDict):
    # 保留最新 5 个元素
    recent_items: Annotated[list, keep_last_n(5)]
    
    # 限制日志长度为 1000 字符
    log: Annotated[str, limit_length(1000)]
    
    # 加权平均分数
    score: Annotated[float, weighted_sum(0.7, 0.3)]

# 使用示例
def node_a(state: State) -> dict:
    return {
        "recent_items": [1, 2, 3, 4, 5, 6],
        "log": "A" * 500,
        "score": 0.8
    }

def node_b(state: State) -> dict:
    return {
        "recent_items": [7, 8],
        "log": "B" * 600,
        "score": 0.9
    }

builder = StateGraph(State)
builder.add_node("a", node_a)
builder.add_node("b", node_b)
builder.add_edge("__start__", "a")
builder.add_edge("a", "b")
builder.add_edge("b", "__end__")

graph = builder.compile()
result = graph.invoke({"recent_items": [], "log": "", "score": 0.0})
print(result)
# {
#     'recent_items': [4, 5, 6, 7, 8],  # 只保留最新 5 个
#     'log': 'AAA...BBB...',  # 限制为 1000 字符
#     'score': 0.83  # 0.8 * 0.7 + 0.9 * 0.3
# }
```

### 最佳实践
1. **参数验证**: 在工厂函数中验证参数有效性
2. **文档字符串**: 为工厂函数和内部 Reducer 添加清晰的文档
3. **类型注解**: 使用泛型保持类型安全
4. **命名规范**: 工厂函数使用动词，如 `keep_last_n`, `limit_length`

## 技巧 2: 实现条件 Reducer（根据状态动态选择合并策略）

### 适用场景
需要根据状态的其他字段或外部条件动态选择合并策略。

### 实现原理
在 Reducer 内部检查条件，根据不同条件执行不同的合并逻辑。

### 代码示例

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph

class State(TypedDict):
    mode: str  # "append" 或 "replace"
    items: Annotated[list, lambda old, new: old]  # 占位符

def conditional_reducer(old: list, new: list) -> list:
    """根据 mode 字段选择合并策略"""
    # 注意: 这里无法直接访问 state，需要通过其他方式传递
    # 实际应用中，可以使用自定义 Channel 或在节点中处理
    return old + new

# 更实用的方案: 在节点中处理条件逻辑
class State(TypedDict):
    mode: str
    items: list

def smart_append(state: State) -> dict:
    """根据 mode 智能追加"""
    mode = state["mode"]
    items = state["items"]
    new_items = [1, 2, 3]
    
    if mode == "append":
        return {"items": items + new_items}
    elif mode == "replace":
        return {"items": new_items}
    elif mode == "unique":
        unique_items = items.copy()
        for item in new_items:
            if item not in unique_items:
                unique_items.append(item)
        return {"items": unique_items}
    else:
        return {"items": items}

# 使用示例
builder = StateGraph(State)
builder.add_node("append", smart_append)
builder.set_entry_point("append")
builder.set_finish_point("append")

graph = builder.compile()

# 测试不同模式
result1 = graph.invoke({"mode": "append", "items": [0]})
print(result1)  # {'mode': 'append', 'items': [0, 1, 2, 3]}

result2 = graph.invoke({"mode": "replace", "items": [0]})
print(result2)  # {'mode': 'replace', 'items': [1, 2, 3]}

result3 = graph.invoke({"mode": "unique", "items": [0, 1]})
print(result3)  # {'mode': 'unique', 'items': [0, 1, 2, 3]}
```

### 最佳实践
1. **节点处理优先**: 条件逻辑优先在节点中处理，而非 Reducer
2. **状态标志**: 使用状态字段作为条件标志
3. **清晰命名**: 使用清晰的模式名称（如 "append", "replace"）
4. **默认行为**: 提供合理的默认行为

## 技巧 3: 使用 Overwrite 实现状态重置模式

### 适用场景
需要在特定条件下重置状态，如清空缓存、重新开始对话等。

### 实现原理
使用 Overwrite 机制跳过 Reducer，直接替换状态值。

### 代码示例

```python
import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph
from langgraph.types import Overwrite
from langgraph.graph.message import add_messages, REMOVE_ALL_MESSAGES, RemoveMessage

class ChatState(TypedDict):
    messages: Annotated[list, add_messages]
    context: Annotated[dict, operator.or_]
    session_id: str

def check_reset_condition(state: ChatState) -> dict:
    """检查是否需要重置"""
    messages = state["messages"]
    
    # 条件 1: 消息数量超过 50
    if len(messages) > 50:
        return {
            "messages": [RemoveMessage(id=REMOVE_ALL_MESSAGES)],
            "context": Overwrite({})
        }
    
    # 条件 2: 用户明确要求重置
    last_message = messages[-1] if messages else None
    if last_message and "重置" in last_message.content:
        return {
            "messages": [RemoveMessage(id=REMOVE_ALL_MESSAGES)],
            "context": Overwrite({})
        }
    
    return {}

def chatbot(state: ChatState) -> dict:
    """聊天机器人节点"""
    return {"messages": [("assistant", "Hello!")]}

# 构建图
builder = StateGraph(ChatState)
builder.add_node("check_reset", check_reset_condition)
builder.add_node("chatbot", chatbot)
builder.add_edge("__start__", "check_reset")
builder.add_edge("check_reset", "chatbot")
builder.add_edge("chatbot", "__end__")

graph = builder.compile()

# 测试重置
result = graph.invoke({
    "messages": [("user", "重置")],
    "context": {"key": "value"},
    "session_id": "123"
})
print(result)
# {
#     'messages': [AIMessage(content='Hello!')],  # 历史被清空
#     'context': {},  # 上下文被重置
#     'session_id': '123'
# }
```

### 性能对比

1. **明确重置条件**: 清晰定义何时需要重置
2. **保留必要信息**: 重置时保留 session_id 等关键信息
3. **使用 Overwrite**: 优先使用 Overwrite 而非手动清空
4. **日志记录**: 记录重置事件用于调试

## 技巧 4: 实现幂等 Reducer（避免重复处理）

### 适用场景
需要确保相同的值不会被重复处理，如去重、幂等操作等。

### 实现原理
在 Reducer 中检查值是否已存在，只添加新值。

### 代码示例

```python
from typing import Annotated, Any
from typing_extensions import TypedDict
from langgraph.graph import StateGraph
import hashlib
import json

def idempotent_list_reducer(old: list, new: list) -> list:
    """幂等列表 Reducer: 去重追加"""
    result = old.copy()
    for item in new:
        if item not in result:
            result.append(item)
    return result

def idempotent_dict_reducer(old: dict, new: dict) -> dict:
    """幂等字典 Reducer: 基于哈希去重"""
    result = old.copy()
    
    # 为每个值计算哈希
    old_hashes = {_hash_value(v) for v in old.values()}
    
    for key, value in new.items():
        value_hash = _hash_value(value)
        if value_hash not in old_hashes:
            result[key] = value
            old_hashes.add(value_hash)
    
    return result

def _hash_value(value: Any) -> str:
    """计算值的哈希"""
    try:
        value_str = json.dumps(value, sort_keys=True)
        return hashlib.md5(value_str.encode()).hexdigest()
    except:
        return str(hash(str(value)))

class State(TypedDict):
    unique_items: Annotated[list, idempotent_list_reducer]
    unique_configs: Annotated[dict, idempotent_dict_reducer]

# 使用示例
def node_a(state: State) -> dict:
    return {
        "unique_items": [1, 2, 3],
        "unique_configs": {"a": {"timeout": 30}}
    }

def node_b(state: State) -> dict:
    return {
        "unique_items": [2, 3, 4],  # 2, 3 重复
        "unique_configs": {"b": {"timeout": 30}}  # 值重复
    }

builder = StateGraph(State)
builder.add_node("a", node_a)
builder.add_node("b", node_b)
builder.add_edge("__start__", "a")
builder.add_edge("a", "b")
builder.add_edge("b", "__end__")

graph = builder.compile()
result = graph.invoke({"unique_items": [], "unique_configs": {}})
print(result)
# {
#     'unique_items': [1, 2, 3, 4],  # 去重
#     'unique_configs': {'a': {'timeout': 30}}  # 基于值去重
# }
```

### 性能对比

```python
# 方案 1: 列表遍历（O(n*m)）
def slow_dedup(old: list, new: list) -> list:
    result = old.copy()
    for item in new:
        if item not in result:  # O(n) 查找
            result.append(item)
    return result

# 方案 2: 集合优化（O(n+m)）
def fast_dedup(old: list, new: list) -> list:
    seen = set(old)
    result = old.copy()
    for item in new:
        if item not in seen:  # O(1) 查找
            result.append(item)
            seen.add(item)
    return result

# 性能测试
# 列表遍历: O(n*m) - 慢
# 集合优化: O(n+m) - 快
```

### 最佳实践
1. **使用集合**: 对于简单类型，使用集合优化查找
2. **哈希函数**: 对于复杂对象，实现稳定的哈希函数
3. **性能权衡**: 考虑去重的性能开销
4. **保持顺序**: 如需保持顺序，使用列表而非集合

## 技巧 5: 调试 Reducer 的技巧

### 适用场景
Reducer 行为不符合预期，需要调试和排查问题。

### 实现原理
添加日志、断言和类型检查，帮助定位问题。

### 代码示例

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph
import logging

# 配置日志
logging.basicConfig(level=logging.DEBUG)
logger = logging.getLogger(__name__)

def debug_reducer(old: list, new: list) -> list:
    """带调试信息的 Reducer"""
    logger.debug(f"Reducer called: old={old}, new={new}")
    
    # 类型检查
    if not isinstance(old, list):
        logger.error(f"Invalid old type: {type(old)}")
        raise TypeError(f"Expected list, got {type(old)}")
    
    if not isinstance(new, list):
        logger.error(f"Invalid new type: {type(new)}")
        raise TypeError(f"Expected list, got {type(new)}")
    
    # 执行合并
    result = old + new
    logger.debug(f"Reducer result: {result}")
    
    # 断言检查
    assert len(result) >= len(old), "Result should not be shorter than old"
    assert len(result) >= len(new), "Result should not be shorter than new"
    
    return result

class State(TypedDict):
    items: Annotated[list, debug_reducer]

# 使用示例
def node_a(state: State) -> dict:
    logger.info("Node A executing")
    return {"items": [1, 2, 3]}

def node_b(state: State) -> dict:
    logger.info("Node B executing")
    return {"items": [4, 5, 6]}

builder = StateGraph(State)
builder.add_node("a", node_a)
builder.add_node("b", node_b)
builder.add_edge("__start__", "a")
builder.add_edge("a", "b")
builder.add_edge("b", "__end__")

graph = builder.compile()
result = graph.invoke({"items": []})

# 输出日志:
# DEBUG:__main__:Reducer called: old=[], new=[1, 2, 3]
# DEBUG:__main__:Reducer result: [1, 2, 3]
# INFO:__main__:Node A executing
# INFO:__main__:Node B executing
# DEBUG:__main__:Reducer called: old=[1, 2, 3], new=[4, 5, 6]
# DEBUG:__main__:Reducer result: [1, 2, 3, 4, 5, 6]
```

```python
import functools
import time

def debug_reducer_decorator(func):
    """Reducer 调试装饰器"""
    @functools.wraps(func)
    def wrapper(old, new):
        start_time = time.time()
        logger.debug(f"[{func.__name__}] Input: old={old}, new={new}")
        
        try:
            result = func(old, new)
            elapsed = time.time() - start_time
            logger.debug(f"[{func.__name__}] Output: {result} (took {elapsed:.4f}s)")
            return result
        except Exception as e:
            logger.error(f"[{func.__name__}] Error: {e}")
            raise
    
    return wrapper

@debug_reducer_decorator
def my_reducer(old: list, new: list) -> list:
    return old + new

class State(TypedDict):
    items: Annotated[list, my_reducer]
```

## 技巧 6: 生产环境的 Reducer 最佳实践

### 适用场景
在生产环境中使用 Reducer，需要考虑性能、可靠性和可维护性。

### 实现原理
遵循最佳实践，确保 Reducer 的健壮性和性能。

### 代码示例

```python
from typing import Annotated, TypeVar, Generic
from typing_extensions import TypedDict
from langgraph.graph import StateGraph
import operator
from langgraph.graph.message import add_messages

T = TypeVar('T')

class ProductionReducer(Generic[T]):
    """生产级 Reducer 基类"""
    
    def __init__(self, max_size: int = 1000):
        self.max_size = max_size
    
    def __call__(self, old: list[T], new: list[T]) -> list[T]:
        # 1. 输入验证
        if not isinstance(old, list):
            old = []
        if not isinstance(new, list):
            new = []
        
        # 2. 执行合并
        result = old + new
        
        # 3. 限制大小（防止内存溢出）
        if len(result) > self.max_size:
            result = result[-self.max_size:]
        
        # 4. 返回结果
        return result

class State(TypedDict):
    # 使用内置 Reducer（推荐）
    messages: Annotated[list, add_messages]
    config: Annotated[dict, operator.or_]
    
    # 使用生产级自定义 Reducer
    logs: Annotated[list, ProductionReducer(max_size=500)]

# 完整示例
def node_a(state: State) -> dict:
    return {
        "messages": [("user", "Hello")],
        "config": {"timeout": 30},
        "logs": ["Log A"]
    }

def node_b(state: State) -> dict:
    return {
        "messages": [("assistant", "Hi!")],
        "config": {"retries": 3},
        "logs": ["Log B"]
    }

builder = StateGraph(State)
builder.add_node("a", node_a)
builder.add_node("b", node_b)
builder.add_edge("__start__", "a")
builder.add_edge("a", "b")
builder.add_edge("b", "__end__")

graph = builder.compile()
result = graph.invoke({
    "messages": [],
    "config": {},
    "logs": []
})
print(result)
```

```python
class GoodReducer:
    """生产级 Reducer 示例"""
    
    def __call__(self, old: list, new: list) -> list:
        # 1. 输入验证
        if not isinstance(old, list):
            old = []
        if not isinstance(new, list):
            new = []
        
        # 2. 边界检查
        if len(old) > 10000:
            old = old[-5000:]  # 保留最新的一半
        
        # 3. 执行合并
        result = old.copy()  # 不修改输入
        result.extend(new)
        
        # 4. 去重（如需要）
        seen = set()
        unique_result = []
        for item in result:
            if item not in seen:
                unique_result.append(item)
                seen.add(item)
        
        return unique_result
```

```python
# 技巧 1: 使用生成器（节省内存）
def memory_efficient_reducer(old: list, new: list) -> list:
    """内存高效的 Reducer"""
    # 使用生成器表达式
    result = [item for item in old if item is not None]
    result.extend(item for item in new if item is not None)
    return result

# 技巧 2: 批量处理（提高性能）
def batch_reducer(old: list, new: list) -> list:
    """批量处理 Reducer"""
    # 一次性分配内存
    result = [None] * (len(old) + len(new))
    result[:len(old)] = old
    result[len(old):] = new
    return result

# 技巧 3: 延迟计算（按需处理）
class LazyReducer:
    """延迟计算 Reducer"""
    
    def __init__(self):
        self.chunks = []
    
    def __call__(self, old: list, new: list) -> list:
        # 只记录，不立即合并
        self.chunks.append((old, new))
        
        # 达到阈值时才合并
        if len(self.chunks) > 10:
            return self._merge_all()
        
        return old + new
    
    def _merge_all(self):
        result = []
        for old, new in self.chunks:
            result.extend(old)
            result.extend(new)
        self.chunks.clear()
        return result
```
