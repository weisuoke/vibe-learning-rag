```python
"""
Milvus 2.6 批量数据导入实战
演示：高效批量插入、分批处理、性能优化
"""

from pymilvus import MilvusClient
import numpy as np
import time

# ===== 1. 连接 Milvus =====
print("=== 1. 连接 Milvus ===")
client = MilvusClient(uri="http://localhost:19530")
print("✓ 连接成功")

# ===== 2. 创建 Collection =====
print("\n=== 2. 创建 Collection ===")
collection_name = "bulk_import_demo"

# 删除已存在的 Collection
if client.has_collection(collection_name):
    client.drop_collection(collection_name)
    print(f"✓ 删除旧 Collection: {collection_name}")

# 创建新 Collection
client.create_collection(
    collection_name=collection_name,
    dimension=128,
    metric_type="L2",
    auto_id=False,
)
print(f"✓ 创建 Collection: {collection_name}")

# ===== 3. 准备大量数据 =====
print("\n=== 3. 准备大量数据 ===")

def generate_data(start_id: int, count: int) -> List[Dict]:
    """生成测试数据"""
    rng = np.random.default_rng(seed=42 + start_id)
    data = []
    for i in range(count):
        doc_id = start_id + i
        data.append({
            "id": doc_id,
            "vector": rng.random(128).tolist(),
            "title": f"Document {doc_id}",
            "content": f"This is the content of document {doc_id}",
            "category": ["tech", "business", "science", "health"][doc_id % 4],
            "score": float(doc_id % 100),
            "timestamp": int(time.time()) + doc_id
        })
    return data

# 生成 10,000 条数据
total_count = 10000
print(f"生成 {total_count} 条测试数据...")
all_data = generate_data(1, total_count)
print(f"✓ 数据生成完成: {len(all_data)} 条")

# ===== 4. 方法1：单条插入（不推荐）=====
print("\n=== 4. 方法1：单条插入（不推荐）===")

# 创建测试 Collection
test_collection_1 = "single_insert_test"
if client.has_collection(test_collection_1):
    client.drop_collection(test_collection_1)
client.create_collection(
    collection_name=test_collection_1,
    dimension=128,
    metric_type="L2",
    auto_id=False,
)

# 单条插入（仅测试前100条）
test_data = all_data[:100]
start_time = time.time()
for item in test_data:
    client.insert(test_collection_1, [item])
single_insert_time = time.time() - start_time

print(f"✓ 单条插入 {len(test_data)} 条数据")
print(f"  耗时: {single_insert_time:.2f} 秒")
print(f"  平均速度: {len(test_data) / single_insert_time:.2f} 条/秒")

# 清理测试 Collection
client.drop_collection(test_collection_1)

# ===== 5. 方法2：批量插入（推荐）=====
print("\n=== 5. 方法2：批量插入（推荐）===")

# 批量插入（测试前100条）
test_collection_2 = "batch_insert_test"
if client.has_collection(test_collection_2):
    client.drop_collection(test_collection_2)
client.create_collection(
    collection_name=test_collection_2,
    dimension=128,
    metric_type="L2",
    auto_id=False,
)

start_time = time.time()
result = client.insert(test_collection_2, test_data)
batch_insert_time = time.time() - start_time

print(f"✓ 批量插入 {result['insert_count']} 条数据")
print(f"  耗时: {batch_insert_time:.2f} 秒")
print(f"  平均速度: {result['insert_count'] / batch_insert_time:.2f} 条/秒")
print(f"  性能提升: {single_insert_time / batch_insert_time:.2f}x")

# 清理测试 Collection
client.drop_collection(test_collection_2)

# ===== 6. 方法3：分批插入（生产推荐）=====
print("\n=== 6. 方法3：分批插入（生产推荐）===")

def batch_insert_with_progress(
    client: MilvusClient,
    collection_name: str,
    data: List[Dict],
    batch_size: int = 1000
) -> Dict:
    """
    分批插入数据，显示进度

    Args:
        client: Milvus 客户端
        collection_name: Collection 名称
        data: 要插入的数据
        batch_size: 每批数据量

    Returns:
        插入统计信息
    """
    total_count = len(data)
    inserted_count = 0
    failed_count = 0
    start_time = time.time()

    print(f"开始分批插入 {total_count} 条数据，每批 {batch_size} 条...")

    for i in range(0, total_count, batch_size):
        batch_data = data[i:i + batch_size]
        try:
            result = client.insert(collection_name, batch_data)
            inserted_count += result['insert_count']

            # 显示进度
            progress = (i + len(batch_data)) / total_count * 100
            print(f"  进度: {progress:.1f}% ({inserted_count}/{total_count})")

        except Exception as e:
            failed_count += len(batch_data)
            print(f"  ✗ 批次 {i//batch_size + 1} 插入失败: {e}")

    elapsed_time = time.time() - start_time

    return {
        "total": total_count,
        "inserted": inserted_count,
        "failed": failed_count,
        "time": elapsed_time,
        "speed": inserted_count / elapsed_time if elapsed_time > 0 else 0
    }

# 执行分批插入
start_time = time.time()
stats = batch_insert_with_progress(
    client=client,
    collection_name=collection_name,
    data=all_data,
    batch_size=1000
)
total_time = time.time() - start_time

print(f"\n✓ 分批插入完成")
print(f"  总数据量: {stats['total']} 条")
print(f"  成功插入: {stats['inserted']} 条")
print(f"  失败数量: {stats['failed']} 条")
print(f"  总耗时: {stats['time']:.2f} 秒")
print(f"  平均速度: {stats['speed']:.2f} 条/秒")

# ===== 7. 验证数据 =====
print("\n=== 7. 验证数据 ===")

# 查询总数
result = client.query(
    collection_name=collection_name,
    filter="id >= 0",
    output_fields=["id"],
    limit=1
)
print(f"✓ Collection 中有数据（查询到至少 {len(result)} 条）")

# 随机抽样验证
sample_ids = [1, 1000, 5000, 9999]
sample_result = client.query(
    collection_name=collection_name,
    ids=sample_ids,
    output_fields=["id", "title", "category"]
)
print(f"✓ 随机抽样验证 {len(sample_result)} 条:")
for item in sample_result:
    print(f"  - ID: {item['id']}, Title: {item['title']}, Category: {item['category']}")

# ===== 8. 性能对比总结 =====
print("\n=== 8. 性能对比总结 ===")
print(f"{'方法':<20} {'数据量':<10} {'耗时(秒)':<12} {'速度(条/秒)':<15}")
print("-" * 60)
print(f"{'单条插入':<20} {len(test_data):<10} {single_insert_time:<12.2f} {len(test_data)/single_insert_time:<15.2f}")
print(f"{'批量插入':<20} {len(test_data):<10} {batch_insert_time:<12.2f} {len(test_data)/batch_insert_time:<15.2f}")
print(f"{'分批插入':<20} {stats['total']:<10} {stats['time']:<12.2f} {stats['speed']:<15.2f}")

# ===== 9. 清理资源 =====
print("\n=== 9. 清理资源 ===")
client.drop_collection(collection_name)
print(f"✓ 删除 Collection: {collection_name}")

print("\n=== 完成 ===")
```

## 关键知识点

### 1. 批量插入的性能优势

**性能对比**：
- 单条插入：~20 条/秒
- 批量插入：~600-800 条/秒
- 性能提升：30-40倍

**原因**：
- 减少网络往返次数
- 批量处理减少开销
- 更高效的资源利用

### 2. 最佳批次大小

**推荐批次大小**：
- 小数据集（< 10万）：1000-5000 条/批
- 中数据集（10万-100万）：5000-10000 条/批
- 大数据集（> 100万）：10000-50000 条/批

**影响因素**：
- 向量维度（维度越高，批次越小）
- 网络带宽
- 服务器内存
- 数据复杂度

### 3. 分批插入的优势

**优势**：
- 避免单次请求过大导致超时
- 可以显示进度
- 失败后可以重试部分数据
- 更好的错误处理

### 4. 错误处理策略

**策略1：失败重试**
```python
def insert_with_retry(client, collection_name, data, max_retries=3):
    for attempt in range(max_retries):
        try:
            return client.insert(collection_name, data)
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            print(f"重试 {attempt + 1}/{max_retries}...")
            time.sleep(2 ** attempt)  # 指数退避
```

**策略2：记录失败数据**
```python
failed_batches = []
for i in range(0, len(data), batch_size):
    batch = data[i:i + batch_size]
    try:
        client.insert(collection_name, batch)
    except Exception as e:
        failed_batches.append((i, batch, str(e)))

# 稍后重试失败的批次
for i, batch, error in failed_batches:
    print(f"重试批次 {i}: {error}")
    client.insert(collection_name, batch)
```

## 最佳实践

### 1. 数据预处理

```python
def preprocess_data(raw_data: List[Dict]) -> List[Dict]:
    """数据预处理"""
    processed = []
    for item in raw_data:
        # 验证必需字段
        if "id" not in item or "vector" not in item:
            continue

        # 数据清洗
        item["title"] = item.get("title", "").strip()
        item["content"] = item.get("content", "").strip()

        # 类型转换
        item["score"] = float(item.get("score", 0))

        processed.append(item)

    return processed
```

### 2. 进度监控

```python
from tqdm import tqdm

def batch_insert_with_tqdm(client, collection_name, data, batch_size=1000):
    """使用 tqdm 显示进度条"""
    total_batches = (len(data) + batch_size - 1) // batch_size

    with tqdm(total=len(data), desc="插入数据") as pbar:
        for i in range(0, len(data), batch_size):
            batch = data[i:i + batch_size]
            result = client.insert(collection_name, batch)
            pbar.update(result['insert_count'])
```

### 3. 内存优化

```python
def batch_insert_from_file(client, collection_name, file_path, batch_size=1000):
    """从文件流式读取并插入，避免一次性加载所有数据"""
    import json

    batch = []
    with open(file_path, 'r') as f:
        for line in f:
            item = json.loads(line)
            batch.append(item)

            if len(batch) >= batch_size:
                client.insert(collection_name, batch)
                batch = []

        # 插入剩余数据
        if batch:
            client.insert(collection_name, batch)
```

### 4. 并发插入（高级）

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

def parallel_batch_insert(client, collection_name, data, batch_size=1000, workers=4):
    """并发批量插入"""
    batches = [data[i:i + batch_size] for i in range(0, len(data), batch_size)]

    with ThreadPoolExecutor(max_workers=workers) as executor:
        futures = [
            executor.submit(client.insert, collection_name, batch)
            for batch in batches
        ]

        for future in as_completed(futures):
            result = future.result()
            print(f"插入 {result['insert_count']} 条")
```

## 常见问题

### Q1: 批量插入时内存不足怎么办？

**解决方案**：
1. 减小批次大小
2. 使用流式读取（不要一次性加载所有数据）
3. 增加服务器内存

### Q2: 批量插入失败如何恢复？

**解决方案**：
1. 记录失败的批次
2. 使用事务日志
3. 实现断点续传

### Q3: 如何提高批量插入速度？

**优化方法**：
1. 增大批次大小（在内存允许的情况下）
2. 使用并发插入
3. 优化网络配置
4. 使用 SSD 存储
