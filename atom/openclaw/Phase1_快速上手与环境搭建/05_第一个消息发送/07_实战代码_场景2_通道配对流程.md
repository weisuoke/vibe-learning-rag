### 3. src/qr-handler.ts

```typescript
/**
 * QR 码处理模块
 */

import qrcode from 'qrcode-terminal';
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export class QRHandler {
  /**
   * 显示 QR 码
   */
  displayQR(qrData: string): void {
    console.log('\n📱 请使用手机扫描以下二维码：\n');
    qrcode.generate(qrData, { small: true });
    console.log('\n');
  }

  /**
   * 等待 QR 码扫描
   */
  async waitForScan(timeout: number = 60000): Promise<boolean> {
    const startTime = Date.now();

    while (Date.now() - startTime < timeout) {
      // 检查配对状态
      const status = await this.checkPairingStatus();
      if (status === 'connected') {
        return true;
      }

      // 每秒检查一次
      await this.delay(1000);

      // 显示倒计时
      const remaining = Math.ceil((timeout - (Date.now() - startTime)) / 1000);
      process.stdout.write(`\r⏳ 等待扫描... (${remaining}秒)`);
    }

    console.log('\n');
    return false;
  }

  /**
   * 检查配对状态
   */
  private async checkPairingStatus(): Promise<string> {
    try {
      const { stdout } = await execAsync('openclaw pairing status');
      if (stdout.includes('Connected')) {
        return 'connected';
      }
      return 'pending';
    } catch (error) {
      return 'error';
    }
  }

  /**
   * 延迟函数
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

### 4. src/pairing-flow.ts

```typescript
/**
 * 配对流程管理
 */

import { exec } from 'child_process';
import { promisify } from 'util';
import inquirer from 'inquirer';
import { getChannelConfig, ChannelConfig } from './channel-config.js';
import { QRHandler } from './qr-handler.js';

const execAsync = promisify(exec);

export interface PairingResult {
  success: boolean;
  channelType: string;
  channelId?: string;
  user?: string;
  error?: string;
}

export class PairingFlow {
  private qrHandler: QRHandler;

  constructor() {
    this.qrHandler = new QRHandler();
  }

  /**
   * 启动配对流程
   */
  async start(channelType?: string): Promise<PairingResult> {
    try {
      // 1. 选择通道
      const selectedChannel = channelType || await this.selectChannel();
      const config = getChannelConfig(selectedChannel);

      console.log(`\n🔗 开始配对 ${config.name}...\n`);

      // 2. 显示设置说明
      this.displayInstructions(config);

      // 3. 根据通道类型执行配对
      let result: PairingResult;

      if (config.type === 'qr') {
        result = await this.pairWithQR(selectedChannel, config);
      } else if (config.type === 'token') {
        result = await this.pairWithToken(selectedChannel, config);
      } else {
        throw new Error(`Unsupported pairing type: ${config.type}`);
      }

      // 4. 显示结果
      this.displayResult(result);

      return result;
    } catch (error) {
      console.error('❌ 配对失败:', error);
      return {
        success: false,
        channelType: channelType || 'unknown',
        error: error instanceof Error ? error.message : 'Unknown error'
      };
    }
  }

  /**
   * 选择通道
   */
  private async selectChannel(): Promise<string> {
    const { channel } = await inquirer.prompt([
      {
        type: 'list',
        name: 'channel',
        message: '请选择要配对的通道:',
        choices: [
          { name: '📱 WhatsApp', value: 'whatsapp' },
          { name: '✈️ Telegram', value: 'telegram' },
          { name: '🎮 Discord', value: 'discord' }
        ]
      }
    ]);
    return channel;
  }

  /**
   * 显示设置说明
   */
  private displayInstructions(config: ChannelConfig): void {
    console.log('📋 设置步骤：');
    config.setupInstructions.forEach(instruction => {
      console.log(`   ${instruction}`);
    });
    console.log();
  }

  /**
   * QR 码配对
   */
  private async pairWithQR(
    channelType: string,
    config: ChannelConfig
  ): Promise<PairingResult> {
    console.log('🔄 生成二维码...\n');

    // 启动配对命令
    const { stdout } = await execAsync(
      `openclaw pairing start --channel ${channelType}`
    );

    // 提取 QR 码数据
    const qrMatch = stdout.match(/QR Code: (.+)/);
    if (!qrMatch) {
      throw new Error('Failed to generate QR code');
    }

    const qrData = qrMatch[1];
    this.qrHandler.displayQR(qrData);

    // 等待扫描
    const scanned = await this.qrHandler.waitForScan();

    if (!scanned) {
      throw new Error('QR code scan timeout');
    }

    console.log('\n✅ 扫描成功！正在建立连接...\n');

    // 等待连接建立
    await this.delay(2000);

    // 获取配对信息
    const pairingInfo = await this.getPairingInfo(channelType);

    return {
      success: true,
      channelType,
      ...pairingInfo
    };
  }

  /**
   * Token 配对
   */
  private async pairWithToken(
    channelType: string,
    config: ChannelConfig
  ): Promise<PairingResult> {
    // 提示输入 Token
    const { token } = await inquirer.prompt([
      {
        type: 'password',
        name: 'token',
        message: `请输入 ${config.name} Bot Token:`,
        mask: '*'
      }
    ]);

    if (!token) {
      throw new Error('Token is required');
    }

    console.log('\n🔄 验证 Token...\n');

    // 设置环境变量
    const envVar = `${channelType.toUpperCase()}_BOT_TOKEN`;
    process.env[envVar] = token;

    // 启动配对
    const { stdout } = await execAsync(
      `openclaw pairing start --channel ${channelType} --token ${token}`
    );

    // 检查结果
    if (!stdout.includes('Pairing successful')) {
      throw new Error('Token validation failed');
    }

    console.log('✅ Token 验证成功！\n');

    // 获取配对信息
    const pairingInfo = await this.getPairingInfo(channelType);

    return {
      success: true,
      channelType,
      ...pairingInfo
    };
  }

  /**
   * 获取配对信息
   */
  private async getPairingInfo(channelType: string): Promise<{
    channelId: string;
    user: string;
  }> {
    const { stdout } = await execAsync('openclaw pairing list');

    // 解析配对列表
    const lines = stdout.split('\n');
    for (const line of lines) {
      if (line.includes(channelType)) {
        const parts = line.split('│').map(p => p.trim()).filter(p => p);
        if (parts.length >= 3) {
          return {
            channelId: parts[2],
            user: parts[1]
          };
        }
      }
    }

    throw new Error('Failed to get pairing info');
  }

  /**
   * 显示配对结果
   */
  private displayResult(result: PairingResult): void {
    console.log('\n' + '='.repeat(50));

    if (result.success) {
      console.log('✅ 配对成功！\n');
      console.log(`📱 通道: ${result.channelType}`);
      console.log(`👤 用户: ${result.user}`);
      console.log(`🆔 ID: ${result.channelId}`);
      console.log('\n💡 现在可以使用以下命令发送消息：');
      console.log(`   openclaw message send "Hello!" --channel ${result.channelType}`);
    } else {
      console.log('❌ 配对失败\n');
      console.log(`错误: ${result.error}`);
      console.log('\n💡 请检查：');
      console.log('   1. 网络连接是否正常');
      console.log('   2. Token 是否正确');
      console.log('   3. Gateway 是否正在运行');
    }

    console.log('='.repeat(50) + '\n');
  }

  /**
   * 延迟函数
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

### 5. src/index.ts

```typescript
/**
 * OpenClaw 通道配对示例
 * 演示完整的配对流程
 */

import 'dotenv/config';
import { PairingFlow } from './pairing-flow.js';

async function main() {
  console.log('🎯 OpenClaw Channel Pairing Example\n');

  // 解析命令行参数
  const args = process.argv.slice(2);
  const channelIndex = args.indexOf('--channel');
  const channelType = channelIndex >= 0 ? args[channelIndex + 1] : undefined;

  // 启动配对流程
  const pairingFlow = new PairingFlow();
  const result = await pairingFlow.start(channelType);

  // 退出
  process.exit(result.success ? 0 : 1);
}

// 运行主函数
main().catch(console.error);
```

## 常见问题

### Q1: QR 码扫描超时

**错误**：`QR code scan timeout`

**解决方案**：
```bash
# 1. 检查 Gateway 状态
openclaw gateway status

# 2. 重启 Gateway
openclaw gateway restart

# 3. 重新尝试配对
npm run pair:whatsapp
```

### Q2: Token 验证失败

**错误**：`Token validation failed`

**解决方案**：
1. 确认 Token 是否正确复制（无多余空格）
2. 检查 Bot 是否已创建成功
3. 验证 Token 是否已过期

### Q3: 配对成功但无法发送消息

**解决方案**：
```bash
# 检查配对状态
openclaw pairing list

# 测试连接
openclaw message send "Test" --channel whatsapp

# 查看日志
openclaw gateway logs
```
