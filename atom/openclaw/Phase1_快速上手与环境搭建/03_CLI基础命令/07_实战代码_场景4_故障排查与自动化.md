// src/diagnostics/health-checker.ts
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export interface HealthStatus {
  gateway: boolean;
  channels: boolean;
  config: boolean;
  overall: boolean;
  issues: string[];
}

export class HealthChecker {
  async check(): Promise<HealthStatus> {
    const issues: string[] = [];

    // 检查 Gateway
    const gatewayHealthy = await this.checkGateway();
    if (!gatewayHealthy) {
      issues.push('Gateway is not running');
    }

    // 检查通道
    const channelsHealthy = await this.checkChannels();
    if (!channelsHealthy) {
      issues.push('No channels connected');
    }

    // 检查配置
    const configHealthy = await this.checkConfig();
    if (!configHealthy) {
      issues.push('Configuration is invalid');
    }

    return {
      gateway: gatewayHealthy,
      channels: channelsHealthy,
      config: configHealthy,
      overall: gatewayHealthy && channelsHealthy && configHealthy,
      issues,
    };
  }

  private async checkGateway(): Promise<boolean> {
    try {
      await execAsync('openclaw gateway status');
      return true;
    } catch {
      return false;
    }
  }

  private async checkChannels(): Promise<boolean> {
    try {
      const { stdout } = await execAsync('openclaw channels list --json');
      const result = JSON.parse(stdout);
      const connected = result.channels.filter((c: any) => c.status === 'connected');
      return connected.length > 0;
    } catch {
      return false;
    }
  }

  private async checkConfig(): Promise<boolean> {
    try {
      await execAsync('openclaw config get');
      return true;
    } catch {
      return false;
    }
  }
}

// src/diagnostics/auto-healer.ts
export class AutoHealer {
  private checker = new HealthChecker();

  async heal(): Promise<void> {
    const health = await this.checker.check();

    if (!health.overall) {
      console.log('🔧 System unhealthy, attempting auto-heal...');

      if (!health.gateway) {
        await this.healGateway();
      }

      if (!health.channels) {
        await this.healChannels();
      }

      if (!health.config) {
        await this.healConfig();
      }

      // 验证修复
      const newHealth = await this.checker.check();
      if (newHealth.overall) {
        console.log('✅ Auto-heal successful');
      } else {
        console.error('❌ Auto-heal failed, manual intervention required');
      }
    }
  }

  private async healGateway(): Promise<void> {
    console.log('🔧 Restarting Gateway...');
    try {
      await execAsync('openclaw gateway restart');
      await this.delay(3000);
    } catch (error) {
      console.error('Failed to restart Gateway:', error);
    }
  }

  private async healChannels(): Promise<void> {
    console.log('🔧 Reconnecting channels...');
    try {
      const { stdout } = await execAsync('openclaw channels list --json');
      const result = JSON.parse(stdout);

      for (const channel of result.channels) {
        if (channel.status === 'disconnected') {
          console.log(`Reconnecting ${channel.id}...`);
          await execAsync(`openclaw channels logout ${channel.id}`);
          await this.delay(2000);
          await execAsync(`openclaw channels add ${channel.type}`);
        }
      }
    } catch (error) {
      console.error('Failed to heal channels:', error);
    }
  }

  private async healConfig(): Promise<void> {
    console.log('🔧 Resetting configuration...');
    try {
      await execAsync('openclaw reset');
      await execAsync('openclaw onboard --non-interactive');
    } catch (error) {
      console.error('Failed to heal config:', error);
    }
  }

  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// src/diagnostics/log-analyzer.ts
import * as fs from 'fs/promises';
import * as path from 'path';

export interface LogEntry {
  timestamp: string;
  level: string;
  message: string;
}

export class LogAnalyzer {
  private logPath: string;

  constructor() {
    const homeDir = process.env.HOME || '';
    this.logPath = path.join(homeDir, '.openclaw', 'logs', 'gateway.log');
  }

  async analyze(): Promise<{
    errors: LogEntry[];
    warnings: LogEntry[];
    summary: string;
  }> {
    const content = await fs.readFile(this.logPath, 'utf-8');
    const lines = content.split('\n');

    const errors: LogEntry[] = [];
    const warnings: LogEntry[] = [];

    for (const line of lines) {
      if (line.includes('ERROR')) {
        errors.push(this.parseLine(line));
      } else if (line.includes('WARN')) {
        warnings.push(this.parseLine(line));
      }
    }

    const summary = this.generateSummary(errors, warnings);

    return { errors, warnings, summary };
  }

  private parseLine(line: string): LogEntry {
    const match = line.match(/\[(.*?)\] \[(.*?)\] (.*)/);
    if (match) {
      return {
        timestamp: match[1],
        level: match[2],
        message: match[3],
      };
    }
    return {
      timestamp: '',
      level: '',
      message: line,
    };
  }

  private generateSummary(errors: LogEntry[], warnings: LogEntry[]): string {
    return `
Log Analysis Summary:
- Total Errors: ${errors.length}
- Total Warnings: ${warnings.length}
- Most Recent Error: ${errors[errors.length - 1]?.message || 'None'}
- Most Recent Warning: ${warnings[warnings.length - 1]?.message || 'None'}
    `.trim();
  }
}

// src/diagnostics/monitor.ts
import { EventEmitter } from 'events';

export class SystemMonitor extends EventEmitter {
  private checker = new HealthChecker();
  private healer = new AutoHealer();
  private analyzer = new LogAnalyzer();
  private interval: NodeJS.Timeout | null = null;

  start(checkInterval = 60000): void {
    this.interval = setInterval(async () => {
      await this.monitor();
    }, checkInterval);

    console.log('📡 System monitoring started');
  }

  stop(): void {
    if (this.interval) {
      clearInterval(this.interval);
      this.interval = null;
      console.log('📡 System monitoring stopped');
    }
  }

  private async monitor(): Promise<void> {
    try {
      // 健康检查
      const health = await this.checker.check();

      if (!health.overall) {
        this.emit('unhealthy', health);

        // 尝试自动修复
        await this.healer.heal();

        // 分析日志
        const logAnalysis = await this.analyzer.analyze();
        this.emit('log-analysis', logAnalysis);
      } else {
        this.emit('healthy', health);
      }
    } catch (error) {
      this.emit('error', error);
    }
  }
}

#!/bin/bash
# scripts/log-rotation.sh

LOG_DIR="$HOME/.openclaw/logs"
BACKUP_DIR="$HOME/.openclaw/logs/backup"

mkdir -p "$BACKUP_DIR"

# 压缩并归档旧日志
find "$LOG_DIR" -name "*.log" -mtime +7 -exec gzip {} \;
find "$LOG_DIR" -name "*.log.gz" -exec mv {} "$BACKUP_DIR" \;

async function sendAlert(message: string) {
  const webhookUrl = process.env.SLACK_WEBHOOK_URL;
  if (!webhookUrl) return;

  await fetch(webhookUrl, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ text: message }),
  });
}
