# 实战代码场景6：字段添加实战

> 本文档演示 Milvus 2.6 AddCollectionField 的完整流程，包括添加字段、设置默认值、并发控制和错误处理

---

## 一、场景描述

### 1.1 业务需求

**场景**：RAG 知识库上线后，业务需求变化，需要为已有 Collection 添加新的标量字段（如文档分类、优先级），同时保证已有数据和查询不受影响。

**核心挑战**：
- 不能重建 Collection（已有大量数据）
- 添加字段期间需要处理并发写入
- 已有数据的新字段需要合理的默认值
- 添加后需要验证所有查询类型正常

### 1.2 技术目标

1. **基础字段添加**：使用 AddCollectionField 添加 nullable 字段
2. **默认值设置**：为新字段设置默认值
3. **生产安全流程**：暂停写入 → 添加 → reload → 验证 → 恢复
4. **错误处理**：处理常见的生产问题

---

## 二、完整实战代码

```python
"""
AddCollectionField 字段添加实战
演示：添加字段、默认值、生产安全流程、错误处理
"""

import numpy as np
import time
from pymilvus import MilvusClient, DataType
from pymilvus.exceptions import MilvusException

# ===== 配置 =====
MILVUS_URI = "http://localhost:19530"
DIM = 128
COLLECTION_NAME = "add_field_demo"
client = MilvusClient(MILVUS_URI)
np.random.seed(42)


# ============================================================
# 1. 创建初始 Collection 并插入数据
# ============================================================
print("=" * 60)
print("1. 创建初始 Collection")
print("=" * 60)

if client.has_collection(COLLECTION_NAME):
    client.drop_collection(COLLECTION_NAME)

# 初始 Schema：只有基础字段
schema = client.create_schema(auto_id=False, enable_dynamic_field=True)
schema.add_field("id", DataType.INT64, is_primary=True)
schema.add_field("vector", DataType.FLOAT_VECTOR, dim=DIM)
schema.add_field("text", DataType.VARCHAR, max_length=1024)
schema.add_field("department", DataType.VARCHAR, max_length=64)

index_params = client.prepare_index_params()
index_params.add_index("vector", index_type="HNSW", metric_type="COSINE",
                       params={"M": 16, "efConstruction": 200})
index_params.add_index("department", index_type="INVERTED")

client.create_collection(COLLECTION_NAME, schema=schema, index_params=index_params)

# 插入初始数据（模拟已有数据）
initial_data = [
    {
        "id": i,
        "vector": np.random.randn(DIM).astype(np.float32).tolist(),
        "text": f"document_{i}",
        "department": np.random.choice(["tech", "legal", "finance"]),
    }
    for i in range(500)
]
client.insert(COLLECTION_NAME, initial_data)
print(f"初始 Collection 创建完成，插入 500 条数据")
print(f"当前字段: id, vector, text, department")


# ============================================================
# 2. 添加 nullable 字段（基础用法）
# ============================================================
print("\n" + "=" * 60)
print("2. 添加 nullable 字段")
print("=" * 60)

# 添加 category 字段（nullable，无默认值）
client.add_collection_field(
    collection_name=COLLECTION_NAME,
    field_name="category",
    data_type=DataType.VARCHAR,
    max_length=100,
    nullable=True  # 必须为 True
)
print("✅ 添加字段: category (VARCHAR, nullable)")

# 验证：已有数据的 category 为 NULL
results = client.query(
    COLLECTION_NAME,
    filter="id < 5",
    output_fields=["id", "text", "department", "category"]
)
print("\n已有数据的 category 字段值:")
for r in results:
    print(f"  id={r['id']}, category={r.get('category', 'NULL')}")


# ============================================================
# 3. 添加带默认值的字段
# ============================================================
print("\n" + "=" * 60)
print("3. 添加带默认值的字段")
print("=" * 60)

# 添加 priority 字段（nullable + 默认值）
client.add_collection_field(
    collection_name=COLLECTION_NAME,
    field_name="priority",
    data_type=DataType.VARCHAR,
    max_length=20,
    nullable=True,
    default_value="normal"  # 默认值
)
print('✅ 添加字段: priority (VARCHAR, nullable, default="normal")')

# 添加 version 字段（INT64 + 默认值）
client.add_collection_field(
    collection_name=COLLECTION_NAME,
    field_name="version",
    data_type=DataType.INT64,
    nullable=True,
    default_value=1
)
print("✅ 添加字段: version (INT64, nullable, default=1)")


# ============================================================
# 4. 重新加载 Collection（关键步骤！）
# ============================================================
print("\n" + "=" * 60)
print("4. 重新加载 Collection")
print("=" * 60)

# 添加字段后必须 reload，否则查询可能异常
client.release_collection(COLLECTION_NAME)
print("  Collection 已释放")

client.load_collection(COLLECTION_NAME)
print("  Collection 已重新加载")

# 等待加载完成
time.sleep(1)
print("✅ Collection reload 完成")


# ============================================================
# 5. 验证所有查询类型
# ============================================================
print("\n" + "=" * 60)
print("5. 验证查询")
print("=" * 60)

query_vector = np.random.randn(DIM).astype(np.float32).tolist()

# 5.1 普通搜索
print("\n5.1 普通搜索:")
results = client.search(
    COLLECTION_NAME,
    data=[query_vector],
    limit=3,
    output_fields=["text", "department", "category", "priority", "version"]
)
for r in results[0]:
    entity = r["entity"]
    print(f"  id={r['id']}, dept={entity.get('department')}, "
          f"category={entity.get('category', 'NULL')}, "
          f"priority={entity.get('priority', 'NULL')}, "
          f"version={entity.get('version', 'NULL')}")

# 5.2 使用新字段过滤
print("\n5.2 过滤新字段（category is null）:")
results = client.search(
    COLLECTION_NAME,
    data=[query_vector],
    limit=3,
    filter="category is null",  # 过滤 NULL 值
    output_fields=["text", "category"]
)
print(f"  找到 {len(results[0])} 条 category 为 NULL 的记录")

# 5.3 使用默认值过滤
print("\n5.3 过滤默认值（priority == 'normal'）:")
results = client.search(
    COLLECTION_NAME,
    data=[query_vector],
    limit=3,
    filter='priority == "normal"',
    output_fields=["text", "priority"]
)
for r in results[0]:
    print(f"  id={r['id']}, priority={r['entity'].get('priority')}")


# ============================================================
# 6. 插入新数据（使用新字段）
# ============================================================
print("\n" + "=" * 60)
print("6. 插入新数据（使用新字段）")
print("=" * 60)

new_data = [
    {
        "id": 1000 + i,
        "vector": np.random.randn(DIM).astype(np.float32).tolist(),
        "text": f"new_document_{i}",
        "department": "tech",
        "category": np.random.choice(["tutorial", "api_doc", "changelog"]),
        "priority": np.random.choice(["high", "normal", "low"]),
        "version": np.random.choice([1, 2, 3]),
    }
    for i in range(50)
]
client.insert(COLLECTION_NAME, new_data)
print(f"插入 50 条新数据（包含新字段值）")

# 验证新数据
results = client.query(
    COLLECTION_NAME,
    filter="id >= 1000 and id < 1005",
    output_fields=["id", "text", "category", "priority", "version"]
)
print("\n新数据验证:")
for r in results:
    print(f"  id={r['id']}, category={r.get('category')}, "
          f"priority={r.get('priority')}, version={r.get('version')}")


# ============================================================
# 7. 错误处理示例
# ============================================================
print("\n" + "=" * 60)
print("7. 错误处理")
print("=" * 60)

# 7.1 重复添加字段
print("\n7.1 重复添加字段:")
try:
    client.add_collection_field(
        collection_name=COLLECTION_NAME,
        field_name="category",  # 已存在
        data_type=DataType.VARCHAR,
        max_length=100,
        nullable=True
    )
except MilvusException as e:
    print(f"  ❌ 预期错误: {e}")

# 7.2 尝试添加向量字段
print("\n7.2 尝试添加向量字段:")
try:
    client.add_collection_field(
        collection_name=COLLECTION_NAME,
        field_name="extra_vector",
        data_type=DataType.FLOAT_VECTOR,
        dim=64,
        nullable=True
    )
except MilvusException as e:
    print(f"  ❌ 预期错误: 不能添加向量字段")

# 7.3 尝试添加非 nullable 字段
print("\n7.3 尝试添加非 nullable 字段:")
try:
    client.add_collection_field(
        collection_name=COLLECTION_NAME,
        field_name="required_field",
        data_type=DataType.VARCHAR,
        max_length=100,
        nullable=False  # 不允许
    )
except MilvusException as e:
    print(f"  ❌ 预期错误: 必须设置 nullable=True")


# ============================================================
# 8. 总结
# ============================================================
print("\n" + "=" * 60)
print("8. 总结")
print("=" * 60)

# 查看最终 Schema
desc = client.describe_collection(COLLECTION_NAME)
print(f"\n最终 Collection Schema:")
print(f"  Collection: {COLLECTION_NAME}")
print(f"  字段列表:")
for field in desc.get("fields", []):
    nullable_str = ", nullable" if field.get("params", {}).get("nullable") else ""
    default_str = ""
    if "default_value" in field.get("params", {}):
        default_str = f", default={field['params']['default_value']}"
    print(f"    - {field['name']} ({field['type']}{nullable_str}{default_str})")

print(f"""
生产环境 AddCollectionField 检查清单:
  ✅ 暂停写入操作
  ✅ 添加字段（nullable=True）
  ✅ 设置合理默认值
  ✅ Release + Load Collection
  ✅ 验证普通搜索
  ✅ 验证 HybridSearch
  ✅ 验证过滤查询
  ✅ 插入新数据验证
  ✅ 恢复写入操作
""")

# 清理
client.drop_collection(COLLECTION_NAME)
print("✅ 测试完成，Collection 已清理")
```
