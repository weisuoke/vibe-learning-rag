# 实战代码 - 场景3：等待边的同步机制（多个并行任务完成后汇聚）

## 场景说明

本示例展示如何使用等待边（Waiting Edge）实现多个并行任务的同步机制。

**实际应用场景：多数据源并行处理系统**
- 输入：需要从多个数据源并行获取数据
- 使用等待边确保所有数据源都完成后再汇总
- 展示 AND 逻辑的同步机制
- 最后统一处理所有结果

**核心特性：**
- 使用 `add_edge([node1, node2, ...], target)` 创建等待边
- 等待所有起始节点完成后才触发目标节点
- 内部使用 `NamedBarrierValue` 通道同步
- 适合需要汇聚多个并行任务结果的场景

## 完整可运行代码

```python
"""
LangGraph 等待边同步机制实战
场景：多数据源并行处理系统

演示：
1. 使用等待边实现多节点同步
2. 并行从多个数据源获取数据
3. 等待所有数据源完成后汇总
4. 展示 AND 逻辑的同步机制
"""

from typing import TypedDict, Annotated, List, Dict
import operator
import time
from langgraph.graph import StateGraph, START, END


# ===== 1. 定义状态结构 =====

class DataSourceState(TypedDict):
    """整体工作流状态"""
    # 各数据源的数据（使用 add reducer 自动合并）
    api_data: Annotated[List[Dict], operator.add]
    database_data: Annotated[List[Dict], operator.add]
    file_data: Annotated[List[Dict], operator.add]

    # 汇总结果
    merged_data: List[Dict]
    summary: str


# ===== 2. 定义节点函数 =====

def fetch_from_api(state: DataSourceState) -> Dict:
    """
    节点1：从 API 获取数据

    模拟从外部 API 获取数据的过程
    """
    print("\n" + "="*60)
    print("🌐 [API 数据源] 开始获取数据...")
    print("="*60)

    # 模拟 API 调用延迟
    time.sleep(1.5)

    # 模拟获取的数据
    api_data = [
        {"source": "api", "id": 1, "type": "user", "value": "Alice"},
        {"source": "api", "id": 2, "type": "user", "value": "Bob"},
        {"source": "api", "id": 3, "type": "user", "value": "Charlie"},
    ]

    print(f"✅ [API 数据源] 获取完成，共 {len(api_data)} 条记录")
    for item in api_data:
        print(f"   - {item}")

    return {"api_data": api_data}


def fetch_from_database(state: DataSourceState) -> Dict:
    """
    节点2：从数据库获取数据

    模拟从数据库查询数据的过程
    """
    print("\n" + "="*60)
    print("🗄️  [数据库数据源] 开始查询数据...")
    print("="*60)

    # 模拟数据库查询延迟
    time.sleep(2.0)

    # 模拟查询的数据
    database_data = [
        {"source": "database", "id": 101, "type": "order", "value": "Order-A"},
        {"source": "database", "id": 102, "type": "order", "value": "Order-B"},
    ]

    print(f"✅ [数据库数据源] 查询完成，共 {len(database_data)} 条记录")
    for item in database_data:
        print(f"   - {item}")

    return {"database_data": database_data}


def fetch_from_file(state: DataSourceState) -> Dict:
    """
    节点3：从文件读取数据

    模拟从文件系统读取数据的过程
    """
    print("\n" + "="*60)
    print("📁 [文件数据源] 开始读取数据...")
    print("="*60)

    # 模拟文件读取延迟
    time.sleep(1.0)

    # 模拟读取的数据
    file_data = [
        {"source": "file", "id": 201, "type": "config", "value": "setting-1"},
        {"source": "file", "id": 202, "type": "config", "value": "setting-2"},
        {"source": "file", "id": 203, "type": "config", "value": "setting-3"},
        {"source": "file", "id": 204, "type": "config", "value": "setting-4"},
    ]

    print(f"✅ [文件数据源] 读取完成，共 {len(file_data)} 条记录")
    for item in file_data:
        print(f"   - {item}")

    return {"file_data": file_data}


def merge_all_data(state: DataSourceState) -> Dict:
    """
    节点4：汇总所有数据源的数据

    这个节点会等待所有数据源节点完成后才执行
    展示等待边的同步机制
    """
    print("\n" + "="*60)
    print("🔄 [数据汇总] 开始合并所有数据源...")
    print("="*60)

    # 获取所有数据源的数据
    api_data = state.get("api_data", [])
    database_data = state.get("database_data", [])
    file_data = state.get("file_data", [])

    print(f"\n📊 数据源统计：")
    print(f"   - API 数据源: {len(api_data)} 条")
    print(f"   - 数据库数据源: {len(database_data)} 条")
    print(f"   - 文件数据源: {len(file_data)} 条")

    # 合并所有数据
    merged_data = api_data + database_data + file_data

    print(f"\n✅ 合并完成，总计 {len(merged_data)} 条记录")

    # 生成摘要
    summary = f"成功从 3 个数据源获取并合并了 {len(merged_data)} 条记录"

    return {
        "merged_data": merged_data,
        "summary": summary
    }


def generate_report(state: DataSourceState) -> Dict:
    """
    节点5：生成最终报告

    基于汇总的数据生成报告
    """
    print("\n" + "="*60)
    print("📝 [报告生成] 开始生成最终报告...")
    print("="*60)

    merged_data = state.get("merged_data", [])

    # 按数据源分组统计
    source_stats = {}
    for item in merged_data:
        source = item["source"]
        source_stats[source] = source_stats.get(source, 0) + 1

    print(f"\n📈 最终报告：")
    print(f"   总记录数: {len(merged_data)}")
    print(f"   数据源分布:")
    for source, count in source_stats.items():
        print(f"      - {source}: {count} 条")

    # 按类型分组统计
    type_stats = {}
    for item in merged_data:
        item_type = item["type"]
        type_stats[item_type] = type_stats.get(item_type, 0) + 1

    print(f"   数据类型分布:")
    for item_type, count in type_stats.items():
        print(f"      - {item_type}: {count} 条")

    print(f"\n✅ 报告生成完成")
    print(f"   摘要: {state.get('summary', '')}")

    return {}


# ===== 3. 构建图（使用等待边） =====

def create_multi_source_graph():
    """
    创建多数据源并行处理图

    图结构：
                    START
                      |
            +---------+---------+
            |         |         |
            v         v         v
        [API]    [Database]  [File]
            |         |         |
            +---------+---------+
                      |
                (等待边：等待所有数据源完成)
                      |
                      v
                  [Merge]
                      |
                      v
                  [Report]
                      |
                      v
                    END

    关键点：
    - 使用 add_edge([node1, node2, node3], target) 创建等待边
    - merge_all_data 节点会等待所有数据源节点完成
    - 展示 AND 逻辑的同步机制
    """
    # 创建图
    graph = StateGraph(DataSourceState)

    # 添加节点
    graph.add_node("fetch_api", fetch_from_api)
    graph.add_node("fetch_database", fetch_from_database)
    graph.add_node("fetch_file", fetch_from_file)
    graph.add_node("merge_data", merge_all_data)
    graph.add_node("generate_report", generate_report)

    # 添加普通边：从 START 到三个数据源节点
    graph.add_edge(START, "fetch_api")
    graph.add_edge(START, "fetch_database")
    graph.add_edge(START, "fetch_file")

    # 添加等待边：等待所有数据源节点完成后才执行 merge_data
    # 这是本示例的核心！
    graph.add_edge(["fetch_api", "fetch_database", "fetch_file"], "merge_data")

    # 添加普通边：merge_data -> generate_report -> END
    graph.add_edge("merge_data", "generate_report")
    graph.add_edge("generate_report", END)

    return graph.compile()


# ===== 4. 运行示例 =====

def main():
    """主函数：运行多数据源并行处理示例"""
    print("\n" + "="*60)
    print("🚀 LangGraph 等待边同步机制实战")
    print("="*60)
    print("\n场景：多数据源并行处理系统")
    print("特性：使用等待边确保所有数据源完成后再汇总\n")

    # 创建图
    app = create_multi_source_graph()

    # 初始状态
    initial_state = {
        "api_data": [],
        "database_data": [],
        "file_data": [],
        "merged_data": [],
        "summary": ""
    }

    # 运行图
    print("="*60)
    print("开始执行工作流...")
    print("="*60)

    start_time = time.time()

    # 执行图
    result = app.invoke(initial_state)

    end_time = time.time()

    # 输出最终结果
    print("\n" + "="*60)
    print("🎉 工作流执行完成")
    print("="*60)
    print(f"\n⏱️  总耗时: {end_time - start_time:.2f} 秒")
    print(f"\n📊 最终状态：")
    print(f"   - API 数据: {len(result.get('api_data', []))} 条")
    print(f"   - 数据库数据: {len(result.get('database_data', []))} 条")
    print(f"   - 文件数据: {len(result.get('file_data', []))} 条")
    print(f"   - 合并数据: {len(result.get('merged_data', []))} 条")
    print(f"   - 摘要: {result.get('summary', '')}")


if __name__ == "__main__":
    main()
```

## 核心知识点

### 1. 等待边的语法

```python
# 等待边：等待多个节点全部完成后才执行目标节点
graph.add_edge(["node1", "node2", "node3"], "target_node")
```

**关键特性：**
- 使用列表指定多个起始节点
- 目标节点会等待所有起始节点完成
- 实现 AND 逻辑的同步机制

### 2. 等待边 vs 普通边

```python
# 普通边：单个起始节点
graph.add_edge("node1", "node2")  # node1 完成后立即触发 node2

# 等待边：多个起始节点
graph.add_edge(["node1", "node2"], "node3")  # 等待 node1 和 node2 都完成后才触发 node3
```

### 3. 内部实现机制

等待边使用 `NamedBarrierValue` 通道实现同步：

```python
# 编译时创建 barrier 通道
channel_name = f"join:{'+'.join(starts)}:{end}"
self.channels[channel_name] = NamedBarrierValue(str, set(starts))

# 每个起始节点完成时写入通道
for start in starts:
    self.nodes[start].writers.append(
        ChannelWrite((ChannelWriteEntry(channel_name, start),))
    )

# 目标节点订阅通道，等待所有起始节点完成
self.nodes[end].triggers.append(channel_name)
```

### 4. 适用场景

**适合使用等待边的场景：**
- 多数据源并行获取后汇总
- 多个独立任务完成后统一处理
- 需要 AND 逻辑的同步机制
- 并行计算后合并结果

**不适合使用等待边的场景：**
- 需要 OR 逻辑（任一节点完成即可）
- 需要动态决定并行任务数量（使用 Send）
- 需要条件分支（使用条件边）

## 扩展练习

### 练习1：添加超时机制

为等待边添加超时机制，如果某个数据源超时则跳过：

```python
def fetch_with_timeout(state: DataSourceState) -> Dict:
    """带超时的数据获取"""
    try:
        # 设置超时时间
        result = fetch_data(timeout=5.0)
        return {"data": result}
    except TimeoutError:
        print("数据源超时，使用默认值")
        return {"data": []}
```

### 练习2：添加错误处理

为每个数据源添加错误处理，确保部分失败不影响整体流程：

```python
def fetch_with_error_handling(state: DataSourceState) -> Dict:
    """带错误处理的数据获取"""
    try:
        result = fetch_data()
        return {"data": result, "status": "success"}
    except Exception as e:
        print(f"数据源错误: {e}")
        return {"data": [], "status": "failed"}
```

### 练习3：添加进度追踪

追踪每个数据源的完成进度：

```python
class ProgressState(TypedDict):
    """带进度追踪的状态"""
    completed_sources: Annotated[List[str], operator.add]
    total_sources: int

def track_progress(state: ProgressState) -> Dict:
    """追踪进度"""
    completed = len(state.get("completed_sources", []))
    total = state.get("total_sources", 0)
    print(f"进度: {completed}/{total} ({completed/total*100:.1f}%)")
    return {}
```

## 总结

本示例展示了 LangGraph 等待边的核心特性：

1. **同步机制**：使用 `add_edge([node1, node2, ...], target)` 实现多节点同步
2. **AND 逻辑**：等待所有起始节点完成后才触发目标节点
3. **并行执行**：多个数据源节点并行执行，提升效率
4. **数据汇聚**：使用 `operator.add` reducer 自动合并多个节点的输出

**关键要点：**
- 等待边适合需要汇聚多个并行任务结果的场景
- 内部使用 `NamedBarrierValue` 通道实现同步
- 与普通边的区别在于起始节点是列表而非单个节点
- 可以与其他边类型（普通边、条件边）组合使用
