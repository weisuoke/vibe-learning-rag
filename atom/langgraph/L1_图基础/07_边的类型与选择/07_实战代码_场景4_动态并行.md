# 实战代码 - 场景4：Send 对象的动态并行执行

## 场景说明

本示例展示如何使用 `Send` 对象实现动态并行执行（fan-out）模式。

**实际应用场景：多文档智能分析系统**
- 输入：多个文档需要并行处理
- 使用 Send 动态创建并行任务
- 每个文档独立分析（提取摘要、关键词、情感）
- 最后汇总所有结果

**核心特性：**
- 动态决定并行任务数量
- 每个任务接收自定义状态
- 并行执行提升效率
- 自动汇总结果

---

## 完整可运行代码

```python
"""
LangGraph Send 对象动态并行执行实战
场景：多文档智能分析系统

演示：
1. 使用 Send 实现动态 fan-out
2. 并行处理多个文档
3. 每个文档独立分析
4. 自动汇总结果
"""

from typing import TypedDict, Annotated, List
from typing_extensions import Literal
import operator
from langgraph.graph import StateGraph, START, END
from langgraph.types import Send


# ===== 1. 定义状态结构 =====

class DocumentState(TypedDict):
    """单个文档的状态"""
    doc_id: str
    content: str
    summary: str
    keywords: List[str]
    sentiment: str


class OverallState(TypedDict):
    """整体工作流状态"""
    documents: List[dict]  # 输入的文档列表
    results: Annotated[List[dict], operator.add]  # 汇总的结果列表（使用 add reducer）


# ===== 2. 定义节点函数 =====

def fan_out_documents(state: OverallState) -> List[Send]:
    """
    Fan-out 节点：动态创建并行任务

    功能：
    - 遍历所有文档
    - 为每个文档创建一个 Send 对象
    - 每个 Send 指向 "analyze_document" 节点
    - 传递该文档的特定状态

    返回：Send 对象列表
    """
    print("\n" + "="*60)
    print("📤 Fan-out: 开始分发文档到并行任务")
    print("="*60)

    documents = state["documents"]
    print(f"📊 总文档数: {len(documents)}")

    # 为每个文档创建一个 Send 对象
    send_list = []
    for doc in documents:
        print(f"  ➡️  分发文档: {doc['doc_id']}")

        # 创建 Send 对象
        # 参数1: 目标节点名称
        # 参数2: 传递给目标节点的状态
        send_obj = Send(
            "analyze_document",  # 目标节点
            {
                "doc_id": doc["doc_id"],
                "content": doc["content"],
                "summary": "",
                "keywords": [],
                "sentiment": ""
            }
        )
        send_list.append(send_obj)

    print(f"✅ 创建了 {len(send_list)} 个并行任务")
    return send_list


def analyze_document(state: DocumentState) -> dict:
    """
    文档分析节点：处理单个文档

    功能：
    - 提取文档摘要
    - 提取关键词
    - 分析情感倾向

    注意：这里使用简单的规则模拟，实际应用中可以调用 LLM
    """
    doc_id = state["doc_id"]
    content = state["content"]

    print(f"\n🔍 [{doc_id}] 开始分析...")

    # 模拟文档分析（实际应用中可以调用 LLM）

    # 1. 生成摘要（取前50个字符）
    summary = content[:50] + "..." if len(content) > 50 else content
    print(f"  📝 [{doc_id}] 摘要: {summary}")

    # 2. 提取关键词（简单分词）
    words = content.split()
    keywords = [w for w in words if len(w) > 4][:3]  # 取长度>4的前3个词
    print(f"  🔑 [{doc_id}] 关键词: {keywords}")

    # 3. 情感分析（简单规则）
    positive_words = ["good", "great", "excellent", "amazing", "wonderful"]
    negative_words = ["bad", "terrible", "awful", "poor", "horrible"]

    content_lower = content.lower()
    pos_count = sum(1 for w in positive_words if w in content_lower)
    neg_count = sum(1 for w in negative_words if w in content_lower)

    if pos_count > neg_count:
        sentiment = "positive"
    elif neg_count > pos_count:
        sentiment = "negative"
    else:
        sentiment = "neutral"

    print(f"  😊 [{doc_id}] 情感: {sentiment}")

    # 返回结果（会被添加到 OverallState.results 中）
    return {
        "results": [{
            "doc_id": doc_id,
            "summary": summary,
            "keywords": keywords,
            "sentiment": sentiment
        }]
    }


# ===== 3. 构建图 =====

def create_parallel_analysis_graph():
    """创建多文档并行分析图"""

    # 创建状态图
    builder = StateGraph(OverallState)

    # 添加节点
    builder.add_node("analyze_document", analyze_document)

    # 添加条件边：从 START 到 fan_out_documents
    # fan_out_documents 返回 Send 列表，自动创建并行执行
    builder.add_conditional_edges(
        START,
        fan_out_documents  # 路由函数返回 Send 列表
    )

    # 添加普通边：所有 analyze_document 完成后到 END
    builder.add_edge("analyze_document", END)

    # 编译图
    graph = builder.compile()

    return graph


# ===== 4. 测试示例 =====

def example_1_basic_parallel():
    """示例1：基础并行文档分析"""
    print("\n" + "="*70)
    print("示例1：基础并行文档分析")
    print("="*70)

    # 创建图
    graph = create_parallel_analysis_graph()

    # 准备输入数据
    input_state = {
        "documents": [
            {
                "doc_id": "doc_001",
                "content": "This is a great product with excellent quality and amazing features."
            },
            {
                "doc_id": "doc_002",
                "content": "The service was terrible and the experience was awful."
            },
            {
                "doc_id": "doc_003",
                "content": "The product works as expected with standard performance."
            }
        ],
        "results": []
    }

    # 执行图
    result = graph.invoke(input_state)

    # 输出结果
    print("\n" + "="*60)
    print("📊 最终汇总结果")
    print("="*60)

    for item in result["results"]:
        print(f"\n📄 文档: {item['doc_id']}")
        print(f"  摘要: {item['summary']}")
        print(f"  关键词: {item['keywords']}")
        print(f"  情感: {item['sentiment']}")


def example_2_large_scale_parallel():
    """示例2：大规模并行处理"""
    print("\n" + "="*70)
    print("示例2：大规模并行处理（10个文档）")
    print("="*70)

    # 创建图
    graph = create_parallel_analysis_graph()

    # 生成10个文档
    documents = []
    sentiments = ["positive", "negative", "neutral"]

    for i in range(10):
        sentiment_type = sentiments[i % 3]

        if sentiment_type == "positive":
            content = f"Document {i+1}: This is an excellent and wonderful product with great features."
        elif sentiment_type == "negative":
            content = f"Document {i+1}: This is a terrible and awful product with poor quality."
        else:
            content = f"Document {i+1}: This is a standard product with normal features."

        documents.append({
            "doc_id": f"doc_{i+1:03d}",
            "content": content
        })

    # 准备输入
    input_state = {
        "documents": documents,
        "results": []
    }

    # 执行图
    result = graph.invoke(input_state)

    # 统计结果
    print("\n" + "="*60)
    print("📊 统计分析")
    print("="*60)

    sentiment_counts = {"positive": 0, "negative": 0, "neutral": 0}
    for item in result["results"]:
        sentiment_counts[item["sentiment"]] += 1

    print(f"总文档数: {len(result['results'])}")
    print(f"正面情感: {sentiment_counts['positive']}")
    print(f"负面情感: {sentiment_counts['negative']}")
    print(f"中性情感: {sentiment_counts['neutral']}")


def example_3_conditional_fanout():
    """示例3：条件性 Fan-out（根据文档类型分发）"""
    print("\n" + "="*70)
    print("示例3：条件性 Fan-out（根据文档类型分发）")
    print("="*70)

    # 定义扩展状态
    class ExtendedDocState(TypedDict):
        doc_id: str
        content: str
        doc_type: str
        summary: str
        keywords: List[str]
        sentiment: str

    class ExtendedOverallState(TypedDict):
        documents: List[dict]
        results: Annotated[List[dict], operator.add]

    # 条件性 fan-out 函数
    def conditional_fan_out(state: ExtendedOverallState) -> List[Send]:
        """根据文档类型决定是否处理"""
        print("\n" + "="*60)
        print("📤 条件性 Fan-out: 只处理特定类型文档")
        print("="*60)

        documents = state["documents"]
        send_list = []

        for doc in documents:
            doc_type = doc.get("doc_type", "unknown")

            # 只处理 "review" 和 "article" 类型
            if doc_type in ["review", "article"]:
                print(f"  ✅ 处理文档: {doc['doc_id']} (类型: {doc_type})")
                send_list.append(Send(
                    "analyze_document",
                    {
                        "doc_id": doc["doc_id"],
                        "content": doc["content"],
                        "doc_type": doc_type,
                        "summary": "",
                        "keywords": [],
                        "sentiment": ""
                    }
                ))
            else:
                print(f"  ⏭️  跳过文档: {doc['doc_id']} (类型: {doc_type})")

        print(f"✅ 创建了 {len(send_list)} 个并行任务")
        return send_list

    # 创建图
    builder = StateGraph(ExtendedOverallState)
    builder.add_node("analyze_document", analyze_document)
    builder.add_conditional_edges(START, conditional_fan_out)
    builder.add_edge("analyze_document", END)
    graph = builder.compile()

    # 准备输入（包含不同类型的文档）
    input_state = {
        "documents": [
            {
                "doc_id": "doc_001",
                "doc_type": "review",
                "content": "This is a great product review with excellent feedback."
            },
            {
                "doc_id": "doc_002",
                "doc_type": "spam",
                "content": "Buy now! Click here for amazing deals!"
            },
            {
                "doc_id": "doc_003",
                "doc_type": "article",
                "content": "This article discusses important topics in technology."
            },
            {
                "doc_id": "doc_004",
                "doc_type": "advertisement",
                "content": "Limited time offer! Get 50% off today!"
            }
        ],
        "results": []
    }

    # 执行图
    result = graph.invoke(input_state)

    # 输出结果
    print("\n" + "="*60)
    print("📊 处理结果")
    print("="*60)
    print(f"输入文档数: 4")
    print(f"实际处理数: {len(result['results'])}")
    print(f"跳过文档数: {4 - len(result['results'])}")


def example_4_nested_fanout():
    """示例4：嵌套 Fan-out（两层并行）"""
    print("\n" + "="*70)
    print("示例4：嵌套 Fan-out（文档 -> 段落两层并行）")
    print("="*70)

    # 定义段落状态
    class ParagraphState(TypedDict):
        doc_id: str
        para_id: str
        content: str
        word_count: int

    class DocWithParasState(TypedDict):
        doc_id: str
        paragraphs: List[str]
        para_results: Annotated[List[dict], operator.add]

    class NestedOverallState(TypedDict):
        documents: List[dict]
        results: Annotated[List[dict], operator.add]

    # 第一层 fan-out：文档级别
    def fanout_documents(state: NestedOverallState) -> List[Send]:
        print("\n📤 第一层 Fan-out: 分发文档")
        documents = state["documents"]
        return [
            Send("process_document", {
                "doc_id": doc["doc_id"],
                "paragraphs": doc["paragraphs"],
                "para_results": []
            })
            for doc in documents
        ]

    # 第二层 fan-out：段落级别
    def fanout_paragraphs(state: DocWithParasState) -> List[Send]:
        doc_id = state["doc_id"]
        paragraphs = state["paragraphs"]
        print(f"\n  📤 第二层 Fan-out: 文档 {doc_id} 分发 {len(paragraphs)} 个段落")

        return [
            Send("analyze_paragraph", {
                "doc_id": doc_id,
                "para_id": f"{doc_id}_p{i+1}",
                "content": para,
                "word_count": 0
            })
            for i, para in enumerate(paragraphs)
        ]

    # 段落分析节点
    def analyze_paragraph(state: ParagraphState) -> dict:
        para_id = state["para_id"]
        content = state["content"]
        word_count = len(content.split())

        print(f"    🔍 分析段落: {para_id} (字数: {word_count})")

        return {
            "para_results": [{
                "para_id": para_id,
                "word_count": word_count
            }]
        }

    # 文档处理节点（触发第二层 fan-out）
    def process_document(state: DocWithParasState) -> dict:
        doc_id = state["doc_id"]

        # 先执行第二层 fan-out
        sends = fanout_paragraphs(state)

        # 注意：这里需要手动执行段落分析
        # 在实际 LangGraph 中，这会自动处理
        para_results = []
        for send in sends:
            result = analyze_paragraph(send.arg)
            para_results.extend(result["para_results"])

        # 汇总文档结果
        total_words = sum(p["word_count"] for p in para_results)
        print(f"  ✅ 文档 {doc_id} 完成: {len(para_results)} 个段落, 总字数: {total_words}")

        return {
            "results": [{
                "doc_id": doc_id,
                "paragraph_count": len(para_results),
                "total_words": total_words
            }]
        }

    # 创建图
    builder = StateGraph(NestedOverallState)
    builder.add_node("process_document", process_document)
    builder.add_conditional_edges(START, fanout_documents)
    builder.add_edge("process_document", END)
    graph = builder.compile()

    # 准备输入
    input_state = {
        "documents": [
            {
                "doc_id": "doc_A",
                "paragraphs": [
                    "First paragraph with some content here.",
                    "Second paragraph with more detailed information.",
                    "Third paragraph concluding the document."
                ]
            },
            {
                "doc_id": "doc_B",
                "paragraphs": [
                    "Opening paragraph for document B.",
                    "Middle section with key points."
                ]
            }
        ],
        "results": []
    }

    # 执行图
    result = graph.invoke(input_state)

    # 输出结果
    print("\n" + "="*60)
    print("📊 嵌套并行处理结果")
    print("="*60)
    for item in result["results"]:
        print(f"文档: {item['doc_id']}")
        print(f"  段落数: {item['paragraph_count']}")
        print(f"  总字数: {item['total_words']}")


# ===== 5. 主函数 =====

if __name__ == "__main__":
    print("\n" + "="*70)
    print("🚀 LangGraph Send 对象动态并行执行实战")
    print("="*70)

    # 运行所有示例
    example_1_basic_parallel()
    example_2_large_scale_parallel()
    example_3_conditional_fanout()
    example_4_nested_fanout()

    print("\n" + "="*70)
    print("✅ 所有示例执行完成")
    print("="*70)
```

## 核心知识点总结

### 1. Send 对象的创建

```python
Send(
    "target_node",  # 目标节点名称
    {"key": "value"}  # 传递给目标节点的状态
)
```

### 2. Fan-out 模式实现

```python
def fan_out_function(state: OverallState) -> List[Send]:
    """返回 Send 对象列表"""
    return [
        Send("process_node", {"data": item})
        for item in state["items"]
    ]

# 在图中使用
builder.add_conditional_edges(START, fan_out_function)
```

### 3. 结果汇总

使用 `Annotated[List, operator.add]` 自动汇总结果：

```python
class State(TypedDict):
    results: Annotated[List[dict], operator.add]
```

### 4. 条件性 Fan-out

```python
def conditional_fan_out(state):
    return [
        Send("node", data)
        for data in state["items"]
        if condition(data)  # 添加条件判断
    ]
```

## 实际应用场景

1. **批量文档处理**：并行分析多个文档
2. **多任务执行**：同时执行多个独立任务
3. **数据并行处理**：对数据集进行并行计算
4. **分布式爬虫**：并行抓取多个网页
5. **批量 API 调用**：并行调用多个 API 接口

---

## 性能优势

- **并行执行**：多个任务同时运行，提升效率
- **动态扩展**：根据输入动态决定并行数量
- **资源优化**：充分利用多核 CPU
- **灵活控制**：可以条件性地创建并行任务

---

## 注意事项

1. **状态隔离**：每个 Send 接收独立的状态副本
2. **结果汇总**：使用 reducer（如 `operator.add`）自动汇总
3. **错误处理**：单个任务失败不影响其他任务
4. **资源限制**：注意并行任务数量，避免资源耗尽
