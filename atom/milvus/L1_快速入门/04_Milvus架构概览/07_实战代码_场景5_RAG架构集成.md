## 一、RAG 架构概览

### 1.1 什么是 RAG？

**RAG（Retrieval-Augmented Generation）** 是一种结合检索和生成的 AI 架构，通过从知识库检索相关信息来增强大语言模型的回答质量。

**RAG 工作流程：**
```
用户问题 → 向量化 → 向量检索 → 上下文注入 → LLM 生成 → 回答
           ↓           ↓
        Embedding   Milvus
         Model    (向量数据库)
```

### 1.2 为什么选择 Milvus？

**Milvus 的优势：**
- ✅ **高性能**：毫秒级向量检索
- ✅ **大规模**：支持数十亿向量
- ✅ **分布式**：水平扩展，高可用
- ✅ **零磁盘**：Milvus 2.6 降低成本 80%+
- ✅ **易集成**：与 LangChain、LlamaIndex 深度集成

### 2.2 完整 RAG 示例（LangChain）

```python
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain.vectorstores import Milvus
from langchain.chains import RetrievalQA
from langchain_community.document_loaders import TextLoader
import os

# 设置 OpenAI API Key
os.environ["OPENAI_API_KEY"] = "your-api-key"

def build_rag_with_langchain():
    """使用 LangChain 构建 RAG 系统"""

    print("Building RAG System with LangChain + Milvus")
    print("=" * 60)

    # 1. 加载文档
    print("\n[Step 1] Loading documents...")
    loader = TextLoader("knowledge_base.txt")
    documents = loader.load()
    print(f"✅ Loaded {len(documents)} documents")

    # 2. 文档分块
    print("\n[Step 2] Splitting documents...")
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=500,
        chunk_overlap=50
    )
    chunks = text_splitter.split_documents(documents)
    print(f"✅ Split into {len(chunks)} chunks")

    # 3. 创建 Embedding 模型
    print("\n[Step 3] Creating embedding model...")
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    print("✅ Embedding model ready")

    # 4. 创建 Milvus 向量存储
    print("\n[Step 4] Creating Milvus vector store...")
    vector_store = Milvus.from_documents(
        documents=chunks,
        embedding=embeddings,
        connection_args={
            "host": "localhost",
            "port": "19530"
        },
        collection_name="langchain_rag_kb",
        drop_old=True  # 删除旧的 Collection
    )
    print("✅ Vector store created")

    # 5. 创建 LLM
    print("\n[Step 5] Creating LLM...")
    llm = ChatOpenAI(model="gpt-4", temperature=0)
    print("✅ LLM ready")

    # 6. 创建 RAG Chain
    print("\n[Step 6] Creating RAG chain...")
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        chain_type="stuff",
        retriever=vector_store.as_retriever(search_kwargs={"k": 3}),
        return_source_documents=True
    )
    print("✅ RAG chain ready")

    # 7. 测试查询
    print("\n[Step 7] Testing RAG system...")
    query = "What is Milvus 2.6?"

    result = qa_chain({"query": query})

    print(f"\nQuery: {query}")
    print(f"Answer: {result['result']}")
    print(f"\nSource documents:")
    for i, doc in enumerate(result['source_documents']):
        print(f"  {i+1}. {doc.page_content[:100]}...")

    print("\n" + "=" * 60)
    print("RAG System Built Successfully!")
    print("=" * 60)

    return qa_chain

# 运行
if __name__ == "__main__":
    qa_chain = build_rag_with_langchain()

    # 交互式查询
    while True:
        query = input("\nEnter your question (or 'quit' to exit): ")
        if query.lower() == 'quit':
            break

        result = qa_chain({"query": query})
        print(f"\nAnswer: {result['result']}")
```

### 2.3 高级 RAG 功能（LangChain）

**混合检索（向量 + 关键词）：**

```python
from langchain.retrievers import EnsembleRetriever
from langchain.retrievers import BM25Retriever

def create_hybrid_retriever(vector_store, documents):
    """创建混合检索器"""

    # 1. 向量检索器
    vector_retriever = vector_store.as_retriever(search_kwargs={"k": 5})

    # 2. BM25 关键词检索器
    bm25_retriever = BM25Retriever.from_documents(documents)
    bm25_retriever.k = 5

    # 3. 混合检索器
    ensemble_retriever = EnsembleRetriever(
        retrievers=[vector_retriever, bm25_retriever],
        weights=[0.7, 0.3]  # 向量检索权重 70%，关键词检索权重 30%
    )

    return ensemble_retriever

# 使用混合检索器
hybrid_retriever = create_hybrid_retriever(vector_store, chunks)

qa_chain = RetrievalQA.from_chain_type(
    llm=llm,
    chain_type="stuff",
    retriever=hybrid_retriever,
    return_source_documents=True
)
```

### 3.2 完整 RAG 示例（LlamaIndex）

```python
from llama_index.core import VectorStoreIndex, SimpleDirectoryReader, StorageContext
from llama_index.vector_stores.milvus import MilvusVectorStore
from llama_index.core.node_parser import SimpleNodeParser
from llama_index.llms.openai import OpenAI
from llama_index.embeddings.openai import OpenAIEmbedding
import os

# 设置 OpenAI API Key
os.environ["OPENAI_API_KEY"] = "your-api-key"

def build_rag_with_llamaindex():
    """使用 LlamaIndex 构建 RAG 系统"""

    print("Building RAG System with LlamaIndex + Milvus")
    print("=" * 60)

    # 1. 加载文档
    print("\n[Step 1] Loading documents...")
    documents = SimpleDirectoryReader("./data").load_data()
    print(f"✅ Loaded {len(documents)} documents")

    # 2. 创建节点解析器
    print("\n[Step 2] Parsing nodes...")
    parser = SimpleNodeParser.from_defaults(chunk_size=500, chunk_overlap=50)
    nodes = parser.get_nodes_from_documents(documents)
    print(f"✅ Parsed {len(nodes)} nodes")

    # 3. 创建 Milvus 向量存储
    print("\n[Step 3] Creating Milvus vector store...")
    vector_store = MilvusVectorStore(
        uri="http://localhost:19530",
        collection_name="llamaindex_rag_kb",
        dim=1536,  # OpenAI embedding dimension
        overwrite=True
    )
    print("✅ Vector store created")

    # 4. 创建存储上下文
    storage_context = StorageContext.from_defaults(vector_store=vector_store)

    # 5. 创建索引
    print("\n[Step 4] Building index...")
    index = VectorStoreIndex(
        nodes,
        storage_context=storage_context,
        embed_model=OpenAIEmbedding(model="text-embedding-3-small")
    )
    print("✅ Index built")

    # 6. 创建查询引擎
    print("\n[Step 5] Creating query engine...")
    query_engine = index.as_query_engine(
        llm=OpenAI(model="gpt-4", temperature=0),
        similarity_top_k=3
    )
    print("✅ Query engine ready")

    # 7. 测试查询
    print("\n[Step 6] Testing RAG system...")
    query = "What is Milvus 2.6?"

    response = query_engine.query(query)

    print(f"\nQuery: {query}")
    print(f"Answer: {response.response}")
    print(f"\nSource nodes:")
    for i, node in enumerate(response.source_nodes):
        print(f"  {i+1}. Score: {node.score:.4f}")
        print(f"     {node.text[:100]}...")

    print("\n" + "=" * 60)
    print("RAG System Built Successfully!")
    print("=" * 60)

    return query_engine

# 运行
if __name__ == "__main__":
    query_engine = build_rag_with_llamaindex()

    # 交互式查询
    while True:
        query = input("\nEnter your question (or 'quit' to exit): ")
        if query.lower() == 'quit':
            break

        response = query_engine.query(query)
        print(f"\nAnswer: {response.response}")
```

```python
def create_query_engine_with_rerank(index):
    """创建带重排序的查询引擎"""

    # 创建重排序器
```

## 四、完整 RAG 系统实现

### 4.1 端到端 RAG 系统

**完整实现（包含文档管理、查询、更新）：**

```python
from pymilvus import connections, Collection, FieldSchema, CollectionSchema, DataType, utility
from sentence_transformers import SentenceTransformer
import openai
import numpy as np

class MilvusRAGSystem:
    """完整的 RAG 系统"""

    def __init__(self, collection_name="rag_system"):
        self.collection_name = collection_name
        self.embedding_model = SentenceTransformer('all-MiniLM-L6-v2')

        # 连接到 Milvus
        connections.connect(host="localhost", port="19530")

        # 创建 Collection
        self._create_collection()

    def _create_collection(self):
        """创建 Collection"""

        if utility.has_collection(self.collection_name):
            utility.drop_collection(self.collection_name)

        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=65535),
            FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=384),
            FieldSchema(name="metadata", dtype=DataType.JSON)
        ]

        schema = CollectionSchema(fields, description="RAG knowledge base")
        collection = Collection(self.collection_name, schema)

        # 创建索引
        index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "L2",
            "params": {"nlist": 128}
        }
        collection.create_index("embedding", index_params)

        print(f"✅ Collection '{self.collection_name}' created")

    def add_documents(self, documents):
        """添加文档"""

        collection = Collection(self.collection_name)

        texts = [doc["text"] for doc in documents]
        embeddings = self.embedding_model.encode(texts).tolist()
        metadata = [doc.get("metadata", {}) for doc in documents]

        collection.insert([texts, embeddings, metadata])
        collection.flush()

        print(f"✅ Added {len(documents)} documents")

    def search(self, query, top_k=3):
        """检索相关文档"""

        collection = Collection(self.collection_name)
        collection.load()

        # 生成查询向量
        query_embedding = self.embedding_model.encode([query])[0].tolist()

        # 检索
        results = collection.search(
            data=[query_embedding],
            anns_field="embedding",
            param={"metric_type": "L2", "params": {"nprobe": 10}},
            limit=top_k,
            output_fields=["text", "metadata"]
        )

        documents = []
        for hit in results[0]:
            documents.append({
                "text": hit.entity.get("text"),
                "metadata": hit.entity.get("metadata"),
                "distance": hit.distance
            })

        return documents

    def generate_answer(self, query, context_docs):
        """生成回答"""

        # 构建上下文
        context = "\n\n".join([doc["text"] for doc in context_docs])

        # 构建 Prompt
        prompt = f"""Based on the following context, answer the question.

Context:
{context}

Question: {query}

Answer:"""

        # 调用 LLM
        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are a helpful assistant."},
                {"role": "user", "content": prompt}
            ],
            temperature=0
        )

        return response.choices[0].message.content

    def query(self, question, top_k=3):
        """完整的 RAG 查询流程"""

        print(f"\nQuery: {question}")
        print("-" * 60)

        # 1. 检索相关文档
        print("Retrieving relevant documents...")
        docs = self.search(question, top_k)

        print(f"✅ Retrieved {len(docs)} documents")
        for i, doc in enumerate(docs):
            print(f"  {i+1}. Distance: {doc['distance']:.4f}")
            print(f"     {doc['text'][:80]}...")

        # 2. 生成回答
        print("\nGenerating answer...")
        answer = self.generate_answer(question, docs)

        print(f"✅ Answer generated")
        print(f"\nAnswer: {answer}")

        return {
            "answer": answer,
            "source_documents": docs
        }

# 使用示例
def main():
    # 1. 创建 RAG 系统
    rag = MilvusRAGSystem()

    # 2. 添加文档
    documents = [
        {
            "text": "Milvus 2.6 introduces Woodpecker, a zero-disk WAL implementation.",
            "metadata": {"source": "milvus_docs", "version": "2.6"}
        },
        {
            "text": "MixCoord merges RootCoord, DataCoord, and QueryCoord into one component.",
            "metadata": {"source": "milvus_docs", "version": "2.6"}
        },
        {
            "text": "Streaming Node handles real-time data streams in Milvus 2.6.",
            "metadata": {"source": "milvus_docs", "version": "2.6"}
        }
    ]

    rag.add_documents(documents)

    # 3. 查询
    result = rag.query("What is Woodpecker in Milvus 2.6?")

    # 4. 交互式查询
    while True:
        question = input("\n\nEnter your question (or 'quit' to exit): ")
        if question.lower() == 'quit':
            break

        rag.query(question)

if __name__ == "__main__":
    main()
```

### 4.2 生产环境优化

**添加缓存和监控：**

```python
from functools import lru_cache
import time

class ProductionRAGSystem(MilvusRAGSystem):
    """生产环境 RAG 系统"""

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.query_count = 0
        self.total_latency = 0

    @lru_cache(maxsize=100)
    def search_cached(self, query, top_k=3):
        """带缓存的检索"""
        return tuple(self.search(query, top_k))

    def query(self, question, top_k=3):
        """带监控的查询"""

        start_time = time.time()

        # 执行查询
        result = super().query(question, top_k)

        # 记录指标
        latency = time.time() - start_time
        self.query_count += 1
        self.total_latency += latency

        print(f"\n📊 Metrics:")
        print(f"   Latency: {latency:.2f}s")
        print(f"   Total queries: {self.query_count}")
        print(f"   Avg latency: {self.total_latency / self.query_count:.2f}s")

        return result
```

## 五、总结

### 5.1 RAG 集成的核心价值

1. **提升回答质量**：基于知识库的事实性回答
2. **降低幻觉**：LLM 基于检索到的上下文生成
3. **实时更新**：知识库可随时更新，无需重新训练
4. **可追溯**：每个回答都有来源文档

### 5.2 Milvus 在 RAG 中的优势

- **高性能**：毫秒级检索，支持实时 RAG
- **大规模**：支持数十亿向量，适合企业级应用
- **易集成**：与 LangChain、LlamaIndex 无缝集成
- **低成本**：Milvus 2.6 零磁盘架构，成本降低 80%+

### 5.3 最佳实践

1. **文档分块**：500-1000 字符，overlap 50-100 字符
2. **混合检索**：向量检索 + 关键词检索
3. **ReRank**：使用重排序模型提升相关性
4. **缓存**：缓存常见查询，降低延迟
5. **监控**：记录查询延迟、准确率等指标
