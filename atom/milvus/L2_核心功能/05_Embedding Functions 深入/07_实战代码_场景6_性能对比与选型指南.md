## 维度1:性能基准测试

### 1.1 测试方法论

根据Milvus官方基准测试,关键性能指标包括:
- **延迟**: 单次API调用时间
- **吞吐量**: 每秒处理文本数
- **批量效率**: 批量处理加速比
- **稳定性**: P95/P99延迟

**来源**: `reference/fetch_benchmarks.md:50-100`

### 1.2 实现代码

```python
"""
Provider性能基准测试
对比不同Provider的延迟和吞吐量
"""

import os
import time
from typing import List, Dict
from dataclasses import dataclass
import statistics
from pymilvus import MilvusClient, Function, FunctionType, CollectionSchema, FieldSchema, DataType

@dataclass
class BenchmarkResult:
    """基准测试结果"""
    provider: str
    model: str
    batch_size: int
    total_texts: int
    total_time: float
    avg_latency: float
    p50_latency: float
    p95_latency: float
    p99_latency: float
    throughput: float

class ProviderBenchmark:
    """Provider性能基准测试"""
    
    def __init__(self):
        self.providers = [
            {"name": "openai", "model": "text-embedding-3-small", "dim": 1536, "batch": 128},
            {"name": "voyageai", "model": "voyage-3-large", "dim": 1024, "batch": 128},
            {"name": "cohere", "model": "embed-english-v3.0", "dim": 1024, "batch": 96},
            {"name": "dashscope", "model": "text-embedding-v3", "dim": 1024, "batch": 6},
        ]
        self.client = MilvusClient(uri="http://localhost:19530")
    
    def create_test_collection(self, provider: dict, collection_name: str):
        """创建测试Collection"""
        params = {
            "provider": provider["name"],
            "model_name": provider["model"],
            "dim": provider["dim"]
        }
        
        # 添加API密钥
        if provider["name"] == "openai":
            params["api_key"] = os.getenv("OPENAI_API_KEY")
        elif provider["name"] == "voyageai":
            params["api_key"] = os.getenv("VOYAGEAI_API_KEY")
        elif provider["name"] == "cohere":
            params["api_key"] = os.getenv("COHERE_API_KEY")
        elif provider["name"] == "dashscope":
            params["api_key"] = os.getenv("DASHSCOPE_API_KEY")
        
        embedding_func = Function(
            name=f"{provider['name']}_ef",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["text"],
            output_field_names=["vector"],
            params=params
        )
        
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=65535),
            FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=provider["dim"])
        ]
        
        schema = CollectionSchema(
            fields=fields,
            functions=[embedding_func],
            description=f"Benchmark collection for {provider['name']}"
        )
        
        if self.client.has_collection(collection_name):
            self.client.drop_collection(collection_name)
        
        self.client.create_collection(collection_name, schema=schema)
    
    def benchmark_provider(self, provider: dict, texts: List[str]) -> BenchmarkResult:
        """测试单个Provider"""
        collection_name = f"benchmark_{provider['name']}"
        
        print(f"\n{'='*60}")
        print(f"测试Provider: {provider['name']} ({provider['model']})")
        print(f"{'='*60}")
        
        # 创建Collection
        self.create_test_collection(provider, collection_name)
        
        # 分批测试
        batch_size = provider["batch"]
        batches = [texts[i:i + batch_size] for i in range(0, len(texts), batch_size)]
        
        latencies = []
        start_time = time.time()
        
        for i, batch in enumerate(batches):
            batch_start = time.time()
            
            data = [{"text": text} for text in batch]
            self.client.insert(collection_name, data)
            
            batch_latency = time.time() - batch_start
            latencies.append(batch_latency)
            
            print(f"  批次 {i+1}/{len(batches)}: {len(batch)}条, {batch_latency:.3f}s")
        
        total_time = time.time() - start_time
        
        # 计算统计指标
        avg_latency = statistics.mean(latencies)
        p50_latency = statistics.median(latencies)
        p95_latency = statistics.quantiles(latencies, n=20)[18] if len(latencies) > 1 else latencies[0]
        p99_latency = statistics.quantiles(latencies, n=100)[98] if len(latencies) > 1 else latencies[0]
        throughput = len(texts) / total_time
        
        result = BenchmarkResult(
            provider=provider["name"],
            model=provider["model"],
            batch_size=batch_size,
            total_texts=len(texts),
            total_time=total_time,
            avg_latency=avg_latency,
            p50_latency=p50_latency,
            p95_latency=p95_latency,
            p99_latency=p99_latency,
            throughput=throughput
        )
        
        print(f"\n📊 性能指标:")
        print(f"  总耗时: {total_time:.2f}s")
        print(f"  平均延迟: {avg_latency:.3f}s")
        print(f"  P50延迟: {p50_latency:.3f}s")
        print(f"  P95延迟: {p95_latency:.3f}s")
        print(f"  P99延迟: {p99_latency:.3f}s")
        print(f"  吞吐量: {throughput:.2f} texts/s")
        
        return result
    
    def run_benchmark(self, num_texts: int = 500) -> List[BenchmarkResult]:
        """运行完整基准测试"""
        # 生成测试数据
        texts = [f"Document {i}: This is a benchmark test document for embedding performance evaluation." for i in range(num_texts)]
        
        print(f"\n🚀 开始基准测试")
        print(f"测试数据: {num_texts}条文本")
        print(f"测试Provider: {len(self.providers)}个")
        
        results = []
        for provider in self.providers:
            try:
                result = self.benchmark_provider(provider, texts)
                results.append(result)
            except Exception as e:
                print(f"❌ {provider['name']} 测试失败: {e}")
        
        return results
    
    def print_comparison(self, results: List[BenchmarkResult]):
        """打印对比结果"""
        print(f"\n{'='*80}")
        print(f"性能对比总结")
        print(f"{'='*80}")
        
        # 按吞吐量排序
        sorted_results = sorted(results, key=lambda x: x.throughput, reverse=True)
        
        print(f"\n{'Provider':<15} {'Model':<30} {'吞吐量':<15} {'P95延迟':<15}")
        print(f"{'-'*80}")
        
        for result in sorted_results:
            print(f"{result.provider:<15} {result.model:<30} {result.throughput:>10.2f} t/s {result.p95_latency:>10.3f}s")
        
        # 最佳Provider
        best_throughput = sorted_results[0]
        best_latency = min(results, key=lambda x: x.p95_latency)
        
        print(f"\n🏆 最佳吞吐量: {best_throughput.provider} ({best_throughput.throughput:.2f} texts/s)")
        print(f"⚡ 最低延迟: {best_latency.provider} (P95: {best_latency.p95_latency:.3f}s)")

# 使用示例
def demo_benchmark():
    """演示性能基准测试"""
    benchmark = ProviderBenchmark()
    results = benchmark.run_benchmark(num_texts=200)
    benchmark.print_comparison(results)

if __name__ == "__main__":
    demo_benchmark()
```

### 2.2 实现代码

```python
"""
成本效益分析工具
计算不同Provider的总拥有成本(TCO)
"""

from dataclasses import dataclass
from typing import Dict, List

@dataclass
class CostModel:
    """成本模型"""
    provider: str
    cost_per_1m_tokens: float
    quality_score: float  # 0-10
    speed_score: float  # 0-10
    setup_cost: float = 0  # 一次性设置成本
    maintenance_cost_monthly: float = 0  # 月度维护成本

class TCOCalculator:
    """总拥有成本计算器"""
    
    def __init__(self):
        self.models = [
            CostModel("openai", 0.13, 9.5, 7.0),
            CostModel("voyageai", 0.12, 9.3, 9.0),
            CostModel("cohere", 0.10, 9.0, 9.0),
            CostModel("vertexai", 0.025, 8.5, 8.0),
            CostModel("dashscope", 0.07, 8.0, 6.0),
            CostModel("tei", 0.02, 7.5, 9.5, setup_cost=500, maintenance_cost_monthly=100),
        ]
    
    def calculate_monthly_cost(self, provider: str, monthly_tokens: int) -> Dict:
        """计算月度成本"""
        model = next((m for m in self.models if m.provider == provider), None)
        if not model:
            return {}
        
        # API调用成本
        api_cost = (monthly_tokens / 1_000_000) * model.cost_per_1m_tokens
        
        # 总成本
        total_cost = api_cost + model.maintenance_cost_monthly
        
        # 性价比评分 (质量 * 速度 / 成本)
        value_score = (model.quality_score * model.speed_score) / (total_cost + 1)
        
        return {
            "provider": provider,
            "api_cost": api_cost,
            "maintenance_cost": model.maintenance_cost_monthly,
            "total_cost": total_cost,
            "quality_score": model.quality_score,
            "speed_score": model.speed_score,
            "value_score": value_score
        }
    
    def calculate_12month_tco(self, provider: str, monthly_tokens: int) -> Dict:
        """计算12个月TCO"""
        model = next((m for m in self.models if m.provider == provider), None)
        if not model:
            return {}
        
        monthly_cost = self.calculate_monthly_cost(provider, monthly_tokens)
        
        # 12个月总成本
        tco_12m = (monthly_cost["total_cost"] * 12) + model.setup_cost
        
        return {
            "provider": provider,
            "setup_cost": model.setup_cost,
            "monthly_cost": monthly_cost["total_cost"],
            "tco_12m": tco_12m,
            "avg_monthly_cost": tco_12m / 12,
            "quality_score": model.quality_score,
            "speed_score": model.speed_score
        }
    
    def compare_scenarios(self, scenarios: List[Dict]):
        """对比不同场景"""
        print(f"\n{'='*80}")
        print(f"成本效益对比分析")
        print(f"{'='*80}")
        
        for scenario in scenarios:
            name = scenario["name"]
            monthly_tokens = scenario["monthly_tokens"]
            
            print(f"\n场景: {name} (月度tokens: {monthly_tokens:,})")
            print(f"{'-'*80}")
            print(f"{'Provider':<15} {'月度成本':<15} {'12月TCO':<15} {'质量':<10} {'速度':<10}")
            print(f"{'-'*80}")
            
            results = []
            for model in self.models:
                tco = self.calculate_12month_tco(model.provider, monthly_tokens)
                results.append(tco)
            
            # 按12月TCO排序
            results.sort(key=lambda x: x["tco_12m"])
            
            for result in results:
                print(f"{result['provider']:<15} ${result['monthly_cost']:>10.2f} ${result['tco_12m']:>12.2f} {result['quality_score']:>8.1f} {result['speed_score']:>8.1f}")
            
            # 推荐
            best_value = results[0]
            print(f"\n💡 推荐: {best_value['provider']} (12月TCO: ${best_value['tco_12m']:.2f})")

# 使用示例
def demo_cost_analysis():
    """演示成本分析"""
    calculator = TCOCalculator()
    
    scenarios = [
        {"name": "小型应用", "monthly_tokens": 1_000_000},
        {"name": "中型应用", "monthly_tokens": 50_000_000},
        {"name": "大型应用", "monthly_tokens": 500_000_000},
    ]
    
    calculator.compare_scenarios(scenarios)

if __name__ == "__main__":
    demo_cost_analysis()
```

### 3.1 决策矩阵

根据不同场景的优先级选择Provider:

**场景1: 原型验证**
- 优先级: 速度 > 成本 > 质量
- 推荐: VoyageAI, Cohere

**场景2: 生产环境**
- 优先级: 质量 > 稳定性 > 成本
- 推荐: OpenAI, VoyageAI

**场景3: 大规模部署**
- 优先级: 成本 > 吞吐量 > 质量
- 推荐: TEI (self-hosted), VertexAI

**场景4: 多云环境**
- 优先级: 灵活性 > 稳定性 > 成本
- 推荐: 多Provider混合策略

### 3.2 实现代码

```python
"""
Provider选型决策引擎
基于场景需求自动推荐最优Provider
"""

from enum import Enum
from typing import List, Dict
from dataclasses import dataclass

class Priority(Enum):
    """优先级"""
    SPEED = "speed"
    COST = "cost"
    QUALITY = "quality"
    STABILITY = "stability"
    FLEXIBILITY = "flexibility"

@dataclass
class ScenarioRequirement:
    """场景需求"""
    name: str
    priorities: List[Priority]
    monthly_tokens: int
    max_latency_ms: float = None
    min_quality_score: float = None
    max_monthly_budget: float = None

class ProviderSelector:
    """Provider选择器"""
    
    def __init__(self):
        self.providers = {
            "openai": {
                "quality": 9.5, "speed": 7.0, "cost": 0.13,
                "stability": 9.0, "flexibility": 8.0
            },
            "voyageai": {
                "quality": 9.3, "speed": 9.0, "cost": 0.12,
                "stability": 8.5, "flexibility": 8.5
            },
            "cohere": {
                "quality": 9.0, "speed": 9.0, "cost": 0.10,
                "stability": 8.5, "flexibility": 8.0
            },
            "vertexai": {
                "quality": 8.5, "speed": 8.0, "cost": 0.025,
                "stability": 9.5, "flexibility": 7.0
            },
            "tei": {
                "quality": 7.5, "speed": 9.5, "cost": 0.02,
                "stability": 8.0, "flexibility": 9.5
            }
        }
    
    def score_provider(self, provider: str, priorities: List[Priority]) -> float:
        """计算Provider评分"""
        metrics = self.providers[provider]
        
        # 权重分配
        weights = {
            Priority.QUALITY: 0.3,
            Priority.SPEED: 0.25,
            Priority.COST: 0.2,
            Priority.STABILITY: 0.15,
            Priority.FLEXIBILITY: 0.1
        }
        
        # 根据优先级调整权重
        adjusted_weights = {}
        for i, priority in enumerate(priorities):
            # 第一优先级权重最高
            weight_multiplier = 1.5 - (i * 0.2)
            adjusted_weights[priority] = weights[priority] * weight_multiplier
        
        # 归一化权重
        total_weight = sum(adjusted_weights.values())
        normalized_weights = {k: v / total_weight for k, v in adjusted_weights.items()}
        
        # 计算总分
        score = 0
        for priority, weight in normalized_weights.items():
            if priority == Priority.QUALITY:
                score += metrics["quality"] * weight
            elif priority == Priority.SPEED:
                score += metrics["speed"] * weight
            elif priority == Priority.COST:
                # 成本越低分数越高
                score += (10 - metrics["cost"] * 50) * weight
            elif priority == Priority.STABILITY:
                score += metrics["stability"] * weight
            elif priority == Priority.FLEXIBILITY:
                score += metrics["flexibility"] * weight
        
        return score
    
    def select_provider(self, requirement: ScenarioRequirement) -> Dict:
        """选择最优Provider"""
        print(f"\n{'='*60}")
        print(f"场景: {requirement.name}")
        print(f"{'='*60}")
        print(f"优先级: {', '.join([p.value for p in requirement.priorities])}")
        print(f"月度tokens: {requirement.monthly_tokens:,}")
        
        # 计算每个Provider的评分
        scores = {}
        for provider in self.providers.keys():
            score = self.score_provider(provider, requirement.priorities)
            scores[provider] = score
        
        # 排序
        sorted_providers = sorted(scores.items(), key=lambda x: x[1], reverse=True)
        
        print(f"\nProvider评分:")
        print(f"{'-'*60}")
        for provider, score in sorted_providers:
            print(f"  {provider:<15} {score:.2f}")
        
        # 推荐
        recommended = sorted_providers[0]
        print(f"\n💡 推荐Provider: {recommended[0]} (评分: {recommended[1]:.2f})")
        
        return {
            "recommended": recommended[0],
            "score": recommended[1],
            "alternatives": [p[0] for p in sorted_providers[1:3]]
        }

# 使用示例
def demo_provider_selection():
    """演示Provider选择"""
    selector = ProviderSelector()
    
    scenarios = [
        ScenarioRequirement(
            name="原型验证",
            priorities=[Priority.SPEED, Priority.COST, Priority.QUALITY],
            monthly_tokens=1_000_000
        ),
        ScenarioRequirement(
            name="生产环境",
            priorities=[Priority.QUALITY, Priority.STABILITY, Priority.COST],
            monthly_tokens=50_000_000
        ),
        ScenarioRequirement(
            name="大规模部署",
            priorities=[Priority.COST, Priority.SPEED, Priority.QUALITY],
            monthly_tokens=500_000_000
        ),
        ScenarioRequirement(
            name="多云环境",
            priorities=[Priority.FLEXIBILITY, Priority.STABILITY, Priority.COST],
            monthly_tokens=100_000_000
        )
    ]
    
    for scenario in scenarios:
        selector.select_provider(scenario)

if __name__ == "__main__":
    demo_provider_selection()
```
