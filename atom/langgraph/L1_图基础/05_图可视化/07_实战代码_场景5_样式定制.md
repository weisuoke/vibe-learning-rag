# 图可视化 - 实战代码: 场景5 样式定制

## 场景描述

在生成图表时,需要自定义图表的样式,包括节点颜色、边样式、背景颜色等,以满足不同的文档和展示需求。

## 完整代码

```python
"""
场景5: 样式定制
演示: 自定义图表样式,包括节点颜色、边样式、背景颜色等
"""

from langgraph.graph import StateGraph, START, END
from typing import TypedDict
from langchain_core.runnables.graph import CurveStyle, NodeStyles
from IPython.display import Image, display, Markdown

# ===== 1. 定义状态 =====
class State(TypedDict):
    """工作流状态定义"""
    messages: list[str]

# ===== 2. 创建基础工作流 =====
print("=== 创建基础工作流 ===")

workflow = StateGraph(State)
workflow.add_node("node1", lambda x: {"messages": x["messages"] + ["node1"]})
workflow.add_node("node2", lambda x: {"messages": x["messages"] + ["node2"]})
workflow.add_node("node3", lambda x: {"messages": x["messages"] + ["node3"]})

workflow.add_edge(START, "node1")
workflow.add_edge("node1", "node2")
workflow.add_edge("node2", "node3")
workflow.add_edge("node3", END)

app = workflow.compile()
print("工作流创建完成")

# ===== 3. 默认样式 =====
print("\n=== 默认样式 ===")

graph = app.get_graph()
mermaid_default = graph.draw_mermaid()
print("默认 Mermaid 文本:")
print(mermaid_default[:500])  # 只打印前500字符

display(Markdown("### 默认样式"))
display(Image(graph.draw_mermaid_png()))

# ===== 4. 自定义节点颜色 =====
print("\n=== 自定义节点颜色 ===")

# 定义自定义颜色
custom_colors = NodeStyles(
    default="#e0f7fa",  # 浅青色
    first="#ffebee",    # 浅红色
    last="#c8e6c9"      # 浅绿色
)

mermaid_custom_colors = graph.draw_mermaid(node_colors=custom_colors)
print("自定义颜色 Mermaid 文本:")
print(mermaid_custom_colors[:500])

display(Markdown("### 自定义节点颜色"))
display(Image(graph.draw_mermaid_png()))

# ===== 5. 自定义曲线样式 =====
print("\n=== 自定义曲线样式 ===")

# 线性样式 (默认)
mermaid_linear = graph.draw_mermaid(curve_style=CurveStyle.LINEAR)
print("线性样式:")
print(mermaid_linear[:200])

# 基础曲线样式
mermaid_basis = graph.draw_mermaid(curve_style=CurveStyle.BASIS)
print("\n基础曲线样式:")
print(mermaid_basis[:200])

# 基数曲线样式
mermaid_cardinal = graph.draw_mermaid(curve_style=CurveStyle.CARDINAL)
print("\n基数曲线样式:")
print(mermaid_cardinal[:200])

display(Markdown("### 线性样式 (LINEAR)"))
display(Image(graph.draw_mermaid_png()))

# ===== 6. 无样式输出 =====
print("\n=== 无样式输出 ===")

mermaid_no_styles = graph.draw_mermaid(with_styles=False)
print("无样式 Mermaid 文本:")
print(mermaid_no_styles)

display(Markdown("### 无样式输出"))
# 注意: 无样式的 Mermaid 文本仍然可以渲染,只是没有颜色

# ===== 7. 自定义 PNG 背景 =====
print("\n=== 自定义 PNG 背景 ===")

# 白色背景 (默认)
png_white = graph.draw_mermaid_png(background_color="white")
print(f"白色背景 PNG 大小: {len(png_white)} 字节")

# 透明背景
png_transparent = graph.draw_mermaid_png(background_color="transparent")
print(f"透明背景 PNG 大小: {len(png_transparent)} 字节")

# 自定义颜色背景
png_custom_bg = graph.draw_mermaid_png(background_color="#f0f0f0")
print(f"自定义背景 PNG 大小: {len(png_custom_bg)} 字节")

display(Markdown("### 白色背景"))
display(Image(png_white))

display(Markdown("### 透明背景"))
display(Image(png_transparent))

# ===== 8. 自定义边距 =====
print("\n=== 自定义边距 ===")

# 默认边距 (10px)
png_default_padding = graph.draw_mermaid_png(padding=10)
print(f"默认边距 PNG 大小: {len(png_default_padding)} 字节")

# 无边距
png_no_padding = graph.draw_mermaid_png(padding=0)
print(f"无边距 PNG 大小: {len(png_no_padding)} 字节")

# 大边距
png_large_padding = graph.draw_mermaid_png(padding=50)
print(f"大边距 PNG 大小: {len(png_large_padding)} 字节")

display(Markdown("### 默认边距 (10px)"))
display(Image(png_default_padding))

display(Markdown("### 大边距 (50px)"))
display(Image(png_large_padding))

# ===== 9. 组合样式 =====
print("\n=== 组合样式 ===")

# 组合多种样式定制
custom_colors_2 = NodeStyles(
    default="#fff3e0",  # 浅橙色
    first="#e1f5fe",    # 浅蓝色
    last="#f3e5f5"      # 浅紫色
)

mermaid_combined = graph.draw_mermaid(
    with_styles=True,
    curve_style=CurveStyle.BASIS,
    node_colors=custom_colors_2,
    wrap_label_n_words=5
)

png_combined = graph.draw_mermaid_png(
    background_color="#fafafa",
    padding=30
)

display(Markdown("### 组合样式"))
display(Image(png_combined))

# ===== 10. 保存不同样式的图表 =====
print("\n=== 保存不同样式的图表 ===")

# 样式1: 默认样式
png_style1 = graph.draw_mermaid_png()
with open("graph_style_default.png", "wb") as f:
    f.write(png_style1)
print("默认样式已保存: graph_style_default.png")

# 样式2: 自定义颜色
png_style2 = graph.draw_mermaid_png(background_color="transparent")
with open("graph_style_transparent.png", "wb") as f:
    f.write(png_style2)
print("透明背景已保存: graph_style_transparent.png")

# 样式3: 大边距
png_style3 = graph.draw_mermaid_png(padding=50)
with open("graph_style_large_padding.png", "wb") as f:
    f.write(png_style3)
print("大边距已保存: graph_style_large_padding.png")

# ===== 11. 样式对比 =====
print("\n=== 样式对比 ===")

styles = [
    ("默认", {}),
    ("透明背景", {"background_color": "transparent"}),
    ("无边距", {"padding": 0}),
    ("大边距", {"padding": 50}),
    ("自定义背景", {"background_color": "#f0f0f0", "padding": 30})
]

for name, kwargs in styles:
    png_data = graph.draw_mermaid_png(**kwargs)
    print(f"{name}: {len(png_data)} 字节")

# ===== 12. Mermaid 文本样式对比 =====
print("\n=== Mermaid 文本样式对比 ===")

mermaid_styles = [
    ("默认", {}),
    ("无样式", {"with_styles": False}),
    ("线性曲线", {"curve_style": CurveStyle.LINEAR}),
    ("基础曲线", {"curve_style": CurveStyle.BASIS}),
    ("自定义颜色", {"node_colors": custom_colors})
]

for name, kwargs in mermaid_styles:
    mermaid_text = graph.draw_mermaid(**kwargs)
    print(f"\n{name}:")
    print(f"  文本长度: {len(mermaid_text)} 字符")
    print(f"  前100字符: {mermaid_text[:100]}")

# ===== 13. 生成样式文档 =====
print("\n=== 生成样式文档 ===")

documentation = f"""
# 图表样式定制文档

## 可用样式选项

### 节点颜色

```python
custom_colors = NodeStyles(
    default="#e0f7fa",  # 普通节点颜色
    first="#ffebee",    # START 节点颜色
    last="#c8e6c9"      # END 节点颜色
)
```

### 曲线样式

- `CurveStyle.LINEAR`: 直线连接
- `CurveStyle.BASIS`: B样条曲线
- `CurveStyle.CARDINAL`: 基数样条曲线

### PNG 样式

- `background_color`: 背景颜色 (white, transparent, #hex)
- `padding`: 边距 (像素)

## 使用示例

```python
# Mermaid 文本样式
mermaid_text = graph.draw_mermaid(
    with_styles=True,
    curve_style=CurveStyle.BASIS,
    node_colors=custom_colors
)

# PNG 图像样式
png_data = graph.draw_mermaid_png(
    background_color="transparent",
    padding=30
)
```

## 样式对比

| 样式 | 文本长度 | PNG 大小 |
|------|----------|----------|
| 默认 | {len(mermaid_default)} 字符 | {len(png_white)} 字节 |
| 无样式 | {len(mermaid_no_styles)} 字符 | - |
| 自定义颜色 | {len(mermaid_custom_colors)} 字符 | - |
"""

with open("style_documentation.md", "w") as f:
    f.write(documentation)
print("样式文档已保存: style_documentation.md")

print("\n=== 样式定制完成 ===")
```

## 常见问题

### Q1: 如何选择合适的节点颜色?

**A**: 考虑以下因素:
- 文档背景颜色
- 品牌颜色
- 可读性和对比度
- 打印效果

### Q2: 透明背景在哪些场景下使用?

**A**:
- 深色主题文档
- 演示文稿
- 网页嵌入
- 需要与背景融合的场景

### Q3: 如何减小 PNG 文件大小?

**A**:
- 使用透明背景
- 减小边距
- 使用 `with_styles=False` 生成简单图表

### Q4: 曲线样式如何选择?

**A**:
- LINEAR: 清晰直观,适合技术文档
- BASIS: 柔和美观,适合演示文稿
- CARDINAL: 平滑过渡,适合复杂图表
