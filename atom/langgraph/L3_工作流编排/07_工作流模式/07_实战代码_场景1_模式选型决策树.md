# 07_工作流模式 - 实战代码 场景1：模式选型决策树

> LangGraph L3_工作流编排 | 实战场景 1/6

---

## 场景描述

构建一个**工作流模式选型工具**：输入任务描述，自动推荐最合适的工作流模式，并生成对应的代码骨架。

这个场景本身就是一个**路由模式**的应用——根据任务特征路由到不同的模式推荐。

---

## 完整代码

```python
"""
工作流模式选型决策树
输入任务描述 → 分析任务特征 → 推荐工作流模式 → 生成代码骨架

模式：路由模式（根据任务特征路由到不同推荐）
"""
from typing import Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


# ===== 1. 定义状态 =====
class State(TypedDict):
    task_description: str       # 用户的任务描述
    has_dependencies: bool      # 步骤间是否有依赖
    needs_classification: bool  # 是否需要分类处理
    has_parallel_tasks: bool    # 是否有可并行的子任务
    dynamic_task_count: bool    # 子任务数量是否动态
    needs_iteration: bool       # 是否需要迭代改进
    recommended_pattern: str    # 推荐的模式
    code_skeleton: str          # 生成的代码骨架
    explanation: str            # 推荐理由


# ===== 2. 分析任务特征 =====
def analyze_task(state: State) -> dict:
    """分析任务描述，提取特征"""
    desc = state["task_description"].lower()

    # 简单规则分析（实际项目中可用 LLM）
    features = {
        "has_dependencies": any(
            kw in desc for kw in ["先", "然后", "之后", "步骤", "流程", "管道"]
        ),
        "needs_classification": any(
            kw in desc for kw in ["分类", "路由", "分流", "判断", "选择"]
        ),
        "has_parallel_tasks": any(
            kw in desc for kw in ["同时", "并行", "并发", "多个", "多维"]
        ),
        "dynamic_task_count": any(
            kw in desc for kw in ["批量", "动态", "不确定", "列表", "每个"]
        ),
        "needs_iteration": any(
            kw in desc for kw in ["优化", "改进", "迭代", "评估", "反馈", "直到"]
        ),
    }

    print(f"[分析] 任务特征: {features}")
    return features


# ===== 3. 决策路由 =====
def decide_pattern(state: State) -> str:
    """根据特征决定推荐哪种模式"""
    if state.get("needs_iteration"):
        return "evaluator_optimizer"
    if state.get("dynamic_task_count"):
        if state.get("needs_classification"):
            return "orchestrator_worker"
        return "map_reduce"
    if state.get("has_parallel_tasks"):
        return "parallel"
    if state.get("needs_classification"):
        return "routing"
    return "sequential"


# ===== 4. 各模式推荐节点 =====
def recommend_sequential(state: State) -> dict:
    """推荐顺序链式模式"""
    return {
        "recommended_pattern": "顺序链式模式（Prompt Chaining）",
        "explanation": (
            "你的任务有明确的先后步骤依赖，适合用顺序链式模式。"
            "每个步骤的输出作为下一步的输入，简单直观。"
        ),
        "code_skeleton": '''
from langgraph.graph import StateGraph, START, END
from typing_extensions import TypedDict

class State(TypedDict):
    input: str
    step1_output: str
    step2_output: str
    final_output: str

def step1(state: State):
    # TODO: 实现步骤1
    return {"step1_output": "..."}

def step2(state: State):
    # TODO: 实现步骤2
    return {"step2_output": "..."}

def step3(state: State):
    # TODO: 实现步骤3
    return {"final_output": "..."}

builder = StateGraph(State)
builder.add_node("step1", step1)
builder.add_node("step2", step2)
builder.add_node("step3", step3)
builder.add_edge(START, "step1")
builder.add_edge("step1", "step2")
builder.add_edge("step2", "step3")
builder.add_edge("step3", END)
graph = builder.compile()
''',
    }


def recommend_routing(state: State) -> dict:
    """推荐路由模式"""
    return {
        "recommended_pattern": "路由模式（Routing）",
        "explanation": (
            "你的任务需要根据输入分类处理，适合用路由模式。"
            "一个路由节点做分类，不同类别走不同处理路径。"
        ),
        "code_skeleton": '''
from langgraph.graph import StateGraph, START, END
from typing_extensions import TypedDict

class State(TypedDict):
    input: str
    category: str
    output: str

def classifier(state: State):
    # TODO: 实现分类逻辑
    return {"category": "type_a"}

def handler_a(state: State):
    return {"output": "处理A"}

def handler_b(state: State):
    return {"output": "处理B"}

def route(state: State):
    return state["category"]

builder = StateGraph(State)
builder.add_node("classifier", classifier)
builder.add_node("handler_a", handler_a)
builder.add_node("handler_b", handler_b)
builder.add_edge(START, "classifier")
builder.add_conditional_edges("classifier", route, {
    "type_a": "handler_a",
    "type_b": "handler_b",
})
builder.add_edge("handler_a", END)
builder.add_edge("handler_b", END)
graph = builder.compile()
''',
    }


def recommend_parallel(state: State) -> dict:
    """推荐并行模式"""
    return {
        "recommended_pattern": "并行模式（Parallelization）",
        "explanation": (
            "你的任务有多个独立子任务可以同时执行，适合用并行模式。"
            "Fan-out 分发任务，Fan-in 汇聚结果，大幅降低延迟。"
        ),
        "code_skeleton": '''
import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END

class State(TypedDict):
    input: str
    results: Annotated[list, operator.add]

def task_a(state): return {"results": ["结果A"]}
def task_b(state): return {"results": ["结果B"]}
def aggregate(state): return {"results": state["results"]}

builder = StateGraph(State)
builder.add_node("task_a", task_a)
builder.add_node("task_b", task_b)
builder.add_node("aggregate", aggregate)
builder.add_edge(START, "task_a")
builder.add_edge(START, "task_b")
builder.add_edge("task_a", "aggregate")
builder.add_edge("task_b", "aggregate")
builder.add_edge("aggregate", END)
graph = builder.compile()
''',
    }


def recommend_map_reduce(state: State) -> dict:
    """推荐 Map-Reduce 模式"""
    return {
        "recommended_pattern": "Map-Reduce 模式",
        "explanation": (
            "你的任务需要对动态数量的数据做相同处理，适合用 Map-Reduce。"
            "Send API 在运行时动态创建并行任务。"
        ),
        "code_skeleton": '''
import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Send

class State(TypedDict):
    items: list[str]
    results: Annotated[list, operator.add]

class WorkerInput(TypedDict):
    item: str

def splitter(state): return state
def worker(state: WorkerInput):
    return {"results": [f"处理: {state['item']}"]}
def reducer(state): return state

def fan_out(state):
    return [Send("worker", {"item": i}) for i in state["items"]]

builder = StateGraph(State)
builder.add_node("splitter", splitter)
builder.add_node("worker", worker)
builder.add_node("reducer", reducer)
builder.add_edge(START, "splitter")
builder.add_conditional_edges("splitter", fan_out, ["worker"])
builder.add_edge("worker", "reducer")
builder.add_edge("reducer", END)
graph = builder.compile()
''',
    }


def recommend_orchestrator_worker(state: State) -> dict:
    """推荐编排者-工作者模式"""
    return {
        "recommended_pattern": "编排者-工作者模式（Orchestrator-Worker）",
        "explanation": (
            "你的任务需要 LLM 动态分解子任务并并行执行，适合编排者-工作者模式。"
        ),
        "code_skeleton": '''
from langgraph.func import task, entrypoint

@task
def orchestrator(topic: str):
    # TODO: LLM 分解任务
    return ["子任务1", "子任务2", "子任务3"]

@task
def worker(subtask: str):
    # TODO: 执行子任务
    return f"完成: {subtask}"

@task
def synthesizer(results: list[str]):
    return "\\n".join(results)

@entrypoint()
def workflow(topic: str):
    subtasks = orchestrator(topic).result()
    futures = [worker(t) for t in subtasks]
    return synthesizer([f.result() for f in futures]).result()
''',
    }


def recommend_evaluator_optimizer(state: State) -> dict:
    """推荐评估者-优化者模式"""
    return {
        "recommended_pattern": "评估者-优化者模式（Evaluator-Optimizer）",
        "explanation": (
            "你的任务需要迭代改进直到满足质量要求，适合评估者-优化者模式。"
        ),
        "code_skeleton": '''
from langgraph.graph import StateGraph, START, END
from typing_extensions import TypedDict

class State(TypedDict):
    input: str
    output: str
    feedback: str
    grade: str
    attempts: int

def generator(state):
    # TODO: 生成/改进内容
    return {"output": "...", "attempts": state.get("attempts", 0) + 1}

def evaluator(state):
    # TODO: 评估质量
    return {"grade": "pass", "feedback": "..."}

def should_continue(state):
    if state["grade"] == "pass": return "done"
    if state["attempts"] >= 3: return "done"
    return "retry"

builder = StateGraph(State)
builder.add_node("generator", generator)
builder.add_node("evaluator", evaluator)
builder.add_edge(START, "generator")
builder.add_edge("generator", "evaluator")
builder.add_conditional_edges("evaluator", should_continue, {
    "done": END, "retry": "generator"
})
graph = builder.compile()
''',
    }


# ===== 5. 构建决策树图 =====
builder = StateGraph(State)

builder.add_node("analyze", analyze_task)
builder.add_node("sequential", recommend_sequential)
builder.add_node("routing", recommend_routing)
builder.add_node("parallel", recommend_parallel)
builder.add_node("map_reduce", recommend_map_reduce)
builder.add_node("orchestrator_worker", recommend_orchestrator_worker)
builder.add_node("evaluator_optimizer", recommend_evaluator_optimizer)

builder.add_edge(START, "analyze")
builder.add_conditional_edges("analyze", decide_pattern, {
    "sequential": "sequential",
    "routing": "routing",
    "parallel": "parallel",
    "map_reduce": "map_reduce",
    "orchestrator_worker": "orchestrator_worker",
    "evaluator_optimizer": "evaluator_optimizer",
})

for node in ["sequential", "routing", "parallel",
             "map_reduce", "orchestrator_worker", "evaluator_optimizer"]:
    builder.add_edge(node, END)

decision_tree = builder.compile()


# ===== 6. 测试 =====
if __name__ == "__main__":
    test_cases = [
        "先提取文本，然后分块，之后生成向量",
        "根据用户问题分类，选择不同的处理方式",
        "同时从多个数据源检索，并行处理",
        "对列表中的每个文档批量生成摘要",
        "生成回答后评估质量，不满意就迭代优化",
    ]

    for desc in test_cases:
        print(f"\n{'='*60}")
        print(f"任务: {desc}")
        result = decision_tree.invoke({"task_description": desc})
        print(f"推荐: {result['recommended_pattern']}")
        print(f"理由: {result['explanation']}")
        print(f"代码骨架:\n{result['code_skeleton'][:200]}...")
```

## 学习要点

1. **路由模式的实际应用**：决策树本身就是路由模式
2. **规则路由 vs LLM 路由**：简单场景用关键词匹配就够了
3. **代码骨架生成**：每种模式都有固定的代码结构模板
4. **模式选型思路**：从任务特征出发，而非从技术出发
