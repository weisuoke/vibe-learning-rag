## 场景1：基础异步节点定义

### 代码示例

```python
"""
基础异步节点实现
展示异步节点的基本定义和使用
"""
import asyncio
from typing import TypedDict
from langgraph.graph import StateGraph, START, END

# 定义状态
class State(TypedDict):
    message: str
    count: int

# 异步节点函数
async def async_process_node(state: State) -> dict:
    """异步处理节点 - 模拟耗时操作"""
    print(f"Processing: {state['message']}")

    # 模拟异步 I/O 操作
    await asyncio.sleep(1)

    return {
        "message": f"Processed: {state['message']}",
        "count": state["count"] + 1
    }

async def async_transform_node(state: State) -> dict:
    """异步转换节点"""
    print(f"Transforming: {state['message']}")

    await asyncio.sleep(0.5)

    return {
        "message": state["message"].upper(),
        "count": state["count"] + 1
    }

# 构建图
def create_async_graph():
    builder = StateGraph(State)

    # 添加异步节点
    builder.add_node("process", async_process_node)
    builder.add_node("transform", async_transform_node)

    # 添加边
    builder.add_edge(START, "process")
    builder.add_edge("process", "transform")
    builder.add_edge("transform", END)

    return builder.compile()

# 运行示例
async def main():
    graph = create_async_graph()

    result = await graph.ainvoke({
        "message": "Hello LangGraph",
        "count": 0
    })

    print(f"Final result: {result}")
    # 输出: {'message': 'PROCESSED: HELLO LANGGRAPH', 'count': 2}

if __name__ == "__main__":
    asyncio.run(main())
```

**关键点**：
- 使用 `async def` 定义异步节点
- 使用 `await` 调用异步操作
- 使用 `ainvoke()` 异步执行图

## 场景2：异步 LLM 调用

### 代码示例

```python
"""
异步 LLM 调用节点
展示如何在节点中异步调用 OpenAI API
"""
import os
from typing import TypedDict
from langgraph.graph import StateGraph, START, END
from openai import AsyncOpenAI

# 初始化 OpenAI 客户端
client = AsyncOpenAI(api_key=os.getenv("OPENAI_API_KEY"))

class LLMState(TypedDict):
    query: str
    response: str
    tokens_used: int

async def llm_call_node(state: LLMState) -> dict:
    """异步调用 LLM"""
    print(f"Calling LLM with query: {state['query']}")

    try:
        # 异步调用 OpenAI API
        response = await client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "You are a helpful assistant."},
                {"role": "user", "content": state["query"]}
            ],
            temperature=0.7,
            max_tokens=500
        )

        return {
            "response": response.choices[0].message.content,
            "tokens_used": response.usage.total_tokens
        }

    except Exception as e:
        print(f"LLM call failed: {e}")
        return {
            "response": f"Error: {str(e)}",
            "tokens_used": 0
        }

async def llm_summarize_node(state: LLMState) -> dict:
    """异步总结响应"""
    print("Summarizing response...")

    try:
        response = await client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "Summarize the following text in one sentence."},
                {"role": "user", "content": state["response"]}
            ],
            temperature=0.5,
            max_tokens=100
        )

        return {
            "response": response.choices[0].message.content,
            "tokens_used": state["tokens_used"] + response.usage.total_tokens
        }

    except Exception as e:
        return {"response": state["response"]}

# 构建图
def create_llm_graph():
    builder = StateGraph(LLMState)

    builder.add_node("llm_call", llm_call_node)
    builder.add_node("summarize", llm_summarize_node)

    builder.add_edge(START, "llm_call")
    builder.add_edge("llm_call", "summarize")
    builder.add_edge("summarize", END)

    return builder.compile()

# 运行示例
async def main():
    graph = create_llm_graph()

    result = await graph.ainvoke({
        "query": "What is LangGraph?",
        "response": "",
        "tokens_used": 0
    })

    print(f"Response: {result['response']}")
    print(f"Total tokens: {result['tokens_used']}")

if __name__ == "__main__":
    import asyncio
    asyncio.run(main())
```

**关键点**：
- 使用 `AsyncOpenAI` 客户端
- 异步调用 `create()` 方法
- 处理 API 异常

## 场景3：异步数据库操作

### 代码示例

```python
"""
异步数据库操作节点
展示如何在节点中进行异步数据库查询
"""
import asyncio
from typing import TypedDict, List
from langgraph.graph import StateGraph, START, END

# 模拟异步数据库客户端
class AsyncDBClient:
    async def query(self, sql: str) -> List[dict]:
        """模拟异步数据库查询"""
        await asyncio.sleep(0.5)  # 模拟网络延迟

        # 模拟查询结果
        if "users" in sql:
            return [
                {"id": 1, "name": "Alice", "email": "alice@example.com"},
                {"id": 2, "name": "Bob", "email": "bob@example.com"}
            ]
        return []

    async def insert(self, table: str, data: dict) -> int:
        """模拟异步插入"""
        await asyncio.sleep(0.3)
        return 1  # 返回插入的 ID

# 初始化数据库客户端
db = AsyncDBClient()

class DBState(TypedDict):
    query: str
    results: List[dict]
    inserted_id: int

async def db_query_node(state: DBState) -> dict:
    """异步查询数据库"""
    print(f"Querying database: {state['query']}")

    try:
        results = await db.query(state["query"])
        return {"results": results}

    except Exception as e:
        print(f"Database query failed: {e}")
        return {"results": []}

async def db_insert_node(state: DBState) -> dict:
    """异步插入数据"""
    if not state["results"]:
        return {}

    print("Inserting data...")

    try:
        # 插入第一条结果
        inserted_id = await db.insert("logs", state["results"][0])
        return {"inserted_id": inserted_id}

    except Exception as e:
        print(f"Database insert failed: {e}")
        return {"inserted_id": 0}

# 构建图
def create_db_graph():
    builder = StateGraph(DBState)

    builder.add_node("query", db_query_node)
    builder.add_node("insert", db_insert_node)

    builder.add_edge(START, "query")
    builder.add_edge("query", "insert")
    builder.add_edge("insert", END)

    return builder.compile()

# 运行示例
async def main():
    graph = create_db_graph()

    result = await graph.ainvoke({
        "query": "SELECT * FROM users",
        "results": [],
        "inserted_id": 0
    })

    print(f"Query results: {result['results']}")
    print(f"Inserted ID: {result['inserted_id']}")

if __name__ == "__main__":
    asyncio.run(main())
```

**关键点**：
- 使用异步数据库客户端
- 处理数据库异常
- 返回部分状态更新

## 场景4：异步 HTTP 请求

### 代码示例

```python
"""
异步 HTTP 请求节点
展示如何在节点中进行异步 HTTP 调用
"""
import asyncio
from typing import TypedDict, Optional
from langgraph.graph import StateGraph, START, END

# 模拟异步 HTTP 客户端
class AsyncHTTPClient:
    async def get(self, url: str) -> dict:
        """模拟异步 GET 请求"""
        await asyncio.sleep(0.5)
        return {
            "status": 200,
            "data": {"message": f"Response from {url}"}
        }

    async def post(self, url: str, data: dict) -> dict:
        """模拟异步 POST 请求"""
        await asyncio.sleep(0.3)
        return {
            "status": 201,
            "data": {"id": 123, "created": True}
        }

# 初始化 HTTP 客户端
http = AsyncHTTPClient()

class HTTPState(TypedDict):
    url: str
    method: str
    response_data: Optional[dict]
    status_code: int

async def http_get_node(state: HTTPState) -> dict:
    """异步 GET 请求"""
    print(f"GET {state['url']}")

    try:
        response = await http.get(state["url"])
        return {
            "response_data": response["data"],
            "status_code": response["status"]
        }

    except Exception as e:
        print(f"HTTP GET failed: {e}")
        return {
            "response_data": {"error": str(e)},
            "status_code": 500
        }

async def http_post_node(state: HTTPState) -> dict:
    """异步 POST 请求"""
    if state["status_code"] != 200:
        return {}

    print(f"POST {state['url']}")

    try:
        response = await http.post(
            state["url"],
            data=state["response_data"]
        )
        return {
            "response_data": response["data"],
            "status_code": response["status"]
        }

    except Exception as e:
        print(f"HTTP POST failed: {e}")
        return {"status_code": 500}

# 构建图
def create_http_graph():
    builder = StateGraph(HTTPState)

    builder.add_node("get", http_get_node)
    builder.add_node("post", http_post_node)

    builder.add_edge(START, "get")
    builder.add_edge("get", "post")
    builder.add_edge("post", END)

    return builder.compile()

# 运行示例
async def main():
    graph = create_http_graph()

    result = await graph.ainvoke({
        "url": "https://api.example.com/data",
        "method": "GET",
        "response_data": None,
        "status_code": 0
    })

    print(f"Final response: {result['response_data']}")
    print(f"Status code: {result['status_code']}")

if __name__ == "__main__":
    asyncio.run(main())
```

**关键点**：
- 使用异步 HTTP 客户端
- 处理 HTTP 异常
- 条件执行（基于状态码）

## 场景5：并发执行多个异步节点

### 代码示例

```python
"""
并发执行异步节点
展示如何并发执行多个独立的异步操作
"""
import asyncio
from typing import TypedDict, List
from langgraph.graph import StateGraph, START, END

class ConcurrentState(TypedDict):
    tasks: List[str]
    results: List[str]

async def concurrent_task_1(state: ConcurrentState) -> dict:
    """并发任务 1"""
    print("Task 1 started")
    await asyncio.sleep(1)
    print("Task 1 completed")
    return {"results": state["results"] + ["Task 1 done"]}

async def concurrent_task_2(state: ConcurrentState) -> dict:
    """并发任务 2"""
    print("Task 2 started")
    await asyncio.sleep(0.5)
    print("Task 2 completed")
    return {"results": state["results"] + ["Task 2 done"]}

async def concurrent_task_3(state: ConcurrentState) -> dict:
    """并发任务 3"""
    print("Task 3 started")
    await asyncio.sleep(0.8)
    print("Task 3 completed")
    return {"results": state["results"] + ["Task 3 done"]}

async def aggregate_node(state: ConcurrentState) -> dict:
    """聚合结果"""
    print(f"Aggregating {len(state['results'])} results")
    return {"results": state["results"]}

# 构建图（使用并行边）
def create_concurrent_graph():
    builder = StateGraph(ConcurrentState)

    # 添加节点
    builder.add_node("task1", concurrent_task_1)
    builder.add_node("task2", concurrent_task_2)
    builder.add_node("task3", concurrent_task_3)
    builder.add_node("aggregate", aggregate_node)

    # 并行执行三个任务
    builder.add_edge(START, "task1")
    builder.add_edge(START, "task2")
    builder.add_edge(START, "task3")

    # 聚合结果
    builder.add_edge("task1", "aggregate")
    builder.add_edge("task2", "aggregate")
    builder.add_edge("task3", "aggregate")
    builder.add_edge("aggregate", END)

    return builder.compile()

# 运行示例
async def main():
    graph = create_concurrent_graph()

    import time
    start = time.time()

    result = await graph.ainvoke({
        "tasks": ["task1", "task2", "task3"],
        "results": []
    })

    elapsed = time.time() - start

    print(f"Results: {result['results']}")
    print(f"Elapsed time: {elapsed:.2f}s")
    # 并发执行约 1s（最长任务的时间），而非 2.3s（串行执行）

if __name__ == "__main__":
    asyncio.run(main())
```

**关键点**：
- 使用多个 `add_edge(START, node)` 实现并发
- 并发执行可显著减少总执行时间
- 使用聚合节点收集结果

## 场景6：异步错误处理与重试

### 代码示例

```python
"""
异步错误处理与重试
展示如何处理异步节点中的错误和配置重试策略
"""
import asyncio
import random
from typing import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import RetryPolicy

class RetryState(TypedDict):
    attempt: int
    success: bool
    error: str

async def unreliable_api_node(state: RetryState) -> dict:
    """模拟不可靠的 API 调用"""
    attempt = state["attempt"] + 1
    print(f"Attempt {attempt}")

    # 50% 概率失败
    if random.random() < 0.5:
        raise Exception("API rate limit exceeded")

    return {
        "attempt": attempt,
        "success": True,
        "error": ""
    }

async def error_handling_node(state: RetryState) -> dict:
    """带错误处理的节点"""
    try:
        # 模拟可能失败的操作
        await asyncio.sleep(0.1)

        if random.random() < 0.3:
            raise ValueError("Invalid data format")

        return {
            "success": True,
            "error": ""
        }

    except ValueError as e:
        # 捕获特定异常
        print(f"Caught ValueError: {e}")
        return {
            "success": False,
            "error": str(e)
        }

    except Exception as e:
        # 捕获其他异常
        print(f"Caught unexpected error: {e}")
        return {
            "success": False,
            "error": f"Unexpected: {str(e)}"
        }

async def cancellable_node(state: RetryState) -> dict:
    """处理取消错误的节点"""
    try:
        await asyncio.sleep(2)
        return {"success": True}

    except asyncio.CancelledError:
        # 特殊处理取消错误
        print("Task was cancelled")
        raise  # 重新抛出以确保正确取消

    except Exception as e:
        return {
            "success": False,
            "error": str(e)
        }

# 自定义重试条件
def retry_on_rate_limit(error: Exception) -> bool:
    """只在遇到 rate limit 错误时重试"""
    return "rate limit" in str(error).lower()

# 构建图（带重试策略）
def create_retry_graph():
    builder = StateGraph(RetryState)

    # 添加节点并配置重试策略
    builder.add_node(
        "unreliable_api",
        unreliable_api_node,
        retry_policy=RetryPolicy(
            initial_interval=0.5,      # 初始重试间隔 0.5s
            backoff_factor=2.0,        # 指数退避因子
            max_interval=5.0,          # 最大重试间隔 5s
            max_attempts=5,            # 最多重试 5 次
            jitter=True,               # 添加随机抖动
            retry_on=retry_on_rate_limit  # 自定义重试条件
        )
    )

    builder.add_node("error_handling", error_handling_node)

    builder.add_edge(START, "unreliable_api")
    builder.add_edge("unreliable_api", "error_handling")
    builder.add_edge("error_handling", END)

    return builder.compile()

# 运行示例
async def main():
    graph = create_retry_graph()

    try:
        result = await graph.ainvoke({
            "attempt": 0,
            "success": False,
            "error": ""
        })

        print(f"Final result: {result}")
        print(f"Success: {result['success']}")
        print(f"Attempts: {result['attempt']}")

    except Exception as e:
        print(f"Graph execution failed: {e}")

if __name__ == "__main__":
    asyncio.run(main())
```

**关键点**：
- 使用 `RetryPolicy` 配置重试策略
- 指数退避算法（`backoff_factor`）
- 自定义重试条件（`retry_on`）
- 处理 `CancelledError`

## 最佳实践总结

### 1. 异步节点定义

```python
# 推荐：使用普通异步函数
async def my_node(state: State) -> dict:
    result = await some_async_operation()
    return {"key": result}

# 不推荐：使用异步生成器（可能有问题）
async def my_node(state: State):
    async for item in some_generator():
        yield {"item": item}
```

[来源: reference/search_节点函数约定_01.md - 异步生成器节点问题]

### 2. 错误处理

- 在节点内部使用 `try-except` 捕获预期异常
- 使用 `RetryPolicy` 处理临时性错误
- 特殊处理 `CancelledError`

### 3. 并发执行

- 使用多个 `add_edge(START, node)` 实现并发
- 确保并发节点之间没有依赖关系
- 使用聚合节点收集并发结果

### 4. 性能优化

- 使用异步 I/O 避免阻塞
- 合理配置重试间隔和次数
- 添加 jitter 避免惊群效应

---

## 注意事项

1. **异步生成器问题**：避免使用 `async function*` 作为节点函数
2. **取消错误处理**：必须重新抛出 `CancelledError` 以确保正确取消
3. **流式错误处理**：在 `astream_events` 循环内处理错误
4. **重试策略配置**：根据实际场景调整重试参数

## 完整示例：异步 RAG 管道

```python
"""
完整的异步 RAG 管道示例
整合 LLM 调用、数据库查询和错误处理
"""
import asyncio
from typing import TypedDict, List
from langgraph.graph import StateGraph, START, END
from langgraph.types import RetryPolicy

class RAGState(TypedDict):
    query: str
    documents: List[str]
    response: str
    error: str

async def retrieve_node(state: RAGState) -> dict:
    """异步检索文档"""
    print(f"Retrieving documents for: {state['query']}")
    await asyncio.sleep(0.5)

    # 模拟检索结果
    docs = [
        "LangGraph is a framework for building stateful workflows.",
        "It supports async nodes and error handling."
    ]

    return {"documents": docs}

async def generate_node(state: RAGState) -> dict:
    """异步生成响应"""
    print("Generating response...")
    await asyncio.sleep(1)

    # 模拟 LLM 生成
    context = "\n".join(state["documents"])
    response = f"Based on the context:\n{context}\n\nAnswer: {state['query']}"

    return {"response": response}

def create_rag_graph():
    builder = StateGraph(RAGState)

    builder.add_node(
        "retrieve",
        retrieve_node,
        retry_policy=RetryPolicy(max_attempts=3)
    )
    builder.add_node(
        "generate",
        generate_node,
        retry_policy=RetryPolicy(max_attempts=3)
    )

    builder.add_edge(START, "retrieve")
    builder.add_edge("retrieve", "generate")
    builder.add_edge("generate", END)

    return builder.compile()

async def main():
    graph = create_rag_graph()

    result = await graph.ainvoke({
        "query": "What is LangGraph?",
        "documents": [],
        "response": "",
        "error": ""
    })

    print(f"\nFinal response:\n{result['response']}")

if __name__ == "__main__":
    asyncio.run(main())
```

**关键点**：
- 完整的异步 RAG 管道
- 每个节点都配置了重试策略
- 清晰的状态流转
