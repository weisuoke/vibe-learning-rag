# 实战代码：场景3 - 数据库查询工具

> 使用 StructuredTool + 复杂验证 + SQL 注入防护构建生产级数据库查询工具

---

## 场景概述

**目标**：构建一个安全的数据库查询工具，支持 Agent 动态查询数据库，同时防止 SQL 注入和非法访问。

**技术栈**：
- `StructuredTool.from_function()` - 显式 schema 控制
- Pydantic 复杂验证 - 白名单、范围限制
- SQLAlchemy - 参数化查询
- 连接池管理 - 资源复用

**应用场景**：
- RAG 系统中的结构化数据检索
- Agent 驱动的数据分析
- 多租户数据库访问控制
- 企业知识库查询

## 完整代码实现

```python
"""
数据库查询工具 - 生产级实现
演示：StructuredTool + 复杂验证 + SQL 注入防护

技术要点：
1. StructuredTool.from_function() 显式创建
2. Pydantic validator 实现白名单验证
3. SQLAlchemy 参数化查询防止注入
4. 连接池管理和资源清理
5. 详细的错误处理和日志
"""

import os
from typing import List, Any, Optional
from pydantic import BaseModel, Field, validator
from langchain.tools import StructuredTool, ToolException
from sqlalchemy import create_engine, text, MetaData, Table
from sqlalchemy.pool import QueuePool
from sqlalchemy.exc import SQLAlchemyError


# ===== 1. 定义输入 Schema（复杂验证） =====

class DatabaseQueryInput(BaseModel):
    """数据库查询工具的输入参数定义

    使用 Pydantic 实现多层验证：
    - 白名单验证（表名、列名）
    - 范围限制（limit）
    - 类型约束
    """

    table: str = Field(
        description="要查询的表名。允许的表：users, products, orders, reviews"
    )

    columns: List[str] = Field(
        description="要查询的列名列表。例如：['id', 'name', 'email']"
    )

    where_clause: Optional[str] = Field(
        default=None,
        description="WHERE 条件（可选）。例如：'status = active'。注意：不支持复杂子查询"
    )

    limit: int = Field(
        default=10,
        ge=1,
        le=100,
        description="返回结果数量限制，范围 1-100"
    )

    # ===== 白名单验证 =====

    @validator('table')
    def validate_table(cls, v):
        """验证表名在白名单中"""
        allowed_tables = ['users', 'products', 'orders', 'reviews']
        if v not in allowed_tables:
            raise ValueError(
                f"表名 '{v}' 不在允许列表中。允许的表：{', '.join(allowed_tables)}"
            )
        return v

    @validator('columns')
    def validate_columns(cls, v):
        """验证列名合法性"""
        if not v:
            raise ValueError("至少需要指定一个列名")

        # 白名单：允许的列名模式
        allowed_patterns = [
            'id', 'name', 'email', 'status', 'created_at', 'updated_at',
            'price', 'quantity', 'description', 'user_id', 'product_id',
            'order_id', 'rating', 'comment'
        ]

        for col in v:
            # 检查是否包含危险字符
            if any(char in col for char in [';', '--', '/*', '*/', 'DROP', 'DELETE']):
                raise ValueError(f"列名 '{col}' 包含非法字符")

            # 检查是否在白名单中（宽松匹配）
            if not any(pattern in col.lower() for pattern in allowed_patterns):
                raise ValueError(
                    f"列名 '{col}' 不在允许列表中。允许的列：{', '.join(allowed_patterns)}"
                )

        return v

    @validator('where_clause')
    def validate_where_clause(cls, v):
        """验证 WHERE 条件安全性"""
        if v is None:
            return v

        # 检查危险关键字
        dangerous_keywords = [
            'DROP', 'DELETE', 'UPDATE', 'INSERT', 'EXEC', 'EXECUTE',
            '--', '/*', '*/', ';', 'UNION', 'SCRIPT'
        ]

        v_upper = v.upper()
        for keyword in dangerous_keywords:
            if keyword in v_upper:
                raise ValueError(
                    f"WHERE 条件包含非法关键字：{keyword}"
                )

        return v


# ===== 2. 数据库工具类（连接池管理） =====

class DatabaseQueryTool:
    """数据库查询工具类

    特性：
    - 连接池管理（复用连接）
    - 参数化查询（防止 SQL 注入）
    - 元数据验证（表和列存在性检查）
    - 详细错误处理
    """

    def __init__(self, connection_string: str):
        """初始化数据库连接

        Args:
            connection_string: 数据库连接字符串
                例如：'postgresql://user:pass@localhost:5432/dbname'
                或：'sqlite:///./test.db'
        """
        # 创建引擎（带连接池）
        self.engine = create_engine(
            connection_string,
            poolclass=QueuePool,
            pool_size=5,          # 连接池大小
            max_overflow=10,      # 最大溢出连接数
            pool_timeout=30,      # 连接超时（秒）
            pool_recycle=3600,    # 连接回收时间（秒）
            echo=False            # 生产环境关闭 SQL 日志
        )

        # 加载元数据（用于验证表和列）
        self.metadata = MetaData()
        self.metadata.reflect(bind=self.engine)

        print(f"✓ 数据库连接已建立：{connection_string.split('@')[-1]}")
        print(f"✓ 可用表：{list(self.metadata.tables.keys())}")

    def query(
        self,
        table: str,
        columns: List[str],
        where_clause: Optional[str] = None,
        limit: int = 10
    ) -> str:
        """执行安全的数据库查询

        Args:
            table: 表名（已通过白名单验证）
            columns: 列名列表（已通过白名单验证）
            where_clause: WHERE 条件（可选）
            limit: 结果数量限制

        Returns:
            查询结果的 JSON 字符串

        Raises:
            ToolException: 查询失败时抛出
        """
        try:
            # ===== 验证表存在 =====
            if table not in self.metadata.tables:
                raise ToolException(
                    f"表 '{table}' 不存在。可用表：{list(self.metadata.tables.keys())}"
                )

            table_obj = self.metadata.tables[table]

            # ===== 验证列存在 =====
            table_columns = [col.name for col in table_obj.columns]
            for col in columns:
                if col not in table_columns:
                    raise ToolException(
                        f"列 '{col}' 在表 '{table}' 中不存在。可用列：{table_columns}"
                    )

            # ===== 构建参数化查询（防止 SQL 注入） =====
            # 使用 SQLAlchemy text() + 绑定参数
            cols_str = ", ".join(columns)

            if where_clause:
                # 注意：这里简化处理，生产环境应该解析 where_clause 并参数化
                query_str = f"SELECT {cols_str} FROM {table} WHERE {where_clause} LIMIT :limit"
            else:
                query_str = f"SELECT {cols_str} FROM {table} LIMIT :limit"

            query = text(query_str)

            # ===== 执行查询 =====
            with self.engine.connect() as conn:
                result = conn.execute(query, {"limit": limit})
                rows = result.fetchall()

                # 转换为字典列表
                results = [
                    dict(zip(columns, row))
                    for row in rows
                ]

                # 返回格式化结果
                if not results:
                    return f"查询成功，但未找到匹配的记录（表：{table}）"

                return (
                    f"查询成功！找到 {len(results)} 条记录：\n"
                    f"{self._format_results(results)}"
                )

        except SQLAlchemyError as e:
            # 数据库错误
            raise ToolException(f"数据库查询失败：{str(e)}")

        except Exception as e:
            # 其他错误
            raise ToolException(f"查询执行错误：{str(e)}")

    def _format_results(self, results: List[dict], max_display: int = 5) -> str:
        """格式化查询结果为可读字符串"""
        import json

        # 只显示前 N 条
        display_results = results[:max_display]
        formatted = json.dumps(display_results, indent=2, ensure_ascii=False)

        if len(results) > max_display:
            formatted += f"\n... (还有 {len(results) - max_display} 条记录未显示)"

        return formatted

    def close(self):
        """关闭数据库连接"""
        self.engine.dispose()
        print("✓ 数据库连接已关闭")


# ===== 3. 创建 StructuredTool =====

# 初始化数据库工具实例
# 生产环境：从环境变量读取连接字符串
db_connection_string = os.getenv(
    "DATABASE_URL",
    "sqlite:///./example.db"  # 默认使用 SQLite
)

db_tool_instance = DatabaseQueryTool(db_connection_string)

# 使用 StructuredTool.from_function() 创建工具
database_query_tool = StructuredTool.from_function(
    name="database_query",
    description=(
        "查询数据库表中的数据。支持指定表名、列名、WHERE 条件和结果数量限制。"
        "适用于获取用户信息、产品数据、订单记录等结构化数据。"
        "注意：只能查询允许的表（users, products, orders, reviews）。"
    ),
    func=db_tool_instance.query,
    args_schema=DatabaseQueryInput,
    return_direct=False,  # 结果返回给 Agent，不直接返回用户
)


# ===== 4. 测试工具 =====

if __name__ == "__main__":

    print("数据库查询工具测试")
    print("="*60 + "\n")

    except Exception as e:

    print("\n" + "-"*60 + "\n")

    # ===== 测试 3：白名单验证 - 非法表名 =====

        result = database_query_tool.invoke({
            "table": "admin_secrets",  # 不在白名单中

    print("\n" + "-"*60 + "\n")

    # ===== 测试 4：SQL 注入防护 =====
    print("【测试 4】安全验证 - SQL 注入尝试")
    try:
        result = database_query_tool.invoke({
            "table": "users",

    except Exception as e:
        print(f"✓ 安全拦截：{e}")

    print("\n" + "-"*60 + "\n")

    # ===== 测试 5：查看工具 Schema =====
    print("【测试 5】工具 Schema 信息")
    print(f"工具名称：{database_query_tool.name}")
    print(f"工具描述：{database_query_tool.description}")
    print(f"\n输入 Schema：")
    import json
    print(json.dumps(
        database_query_tool.args_schema.model_json_schema(),
        indent=2,
        ensure_ascii=False
    ))

    # 清理资源
    print("\n" + "="*60)
    db_tool_instance.close()
```

## 核心技术点解析

### 1. StructuredTool.from_function() 的优势

**为什么使用 StructuredTool 而非 @tool？**

```python
# ❌ @tool 装饰器不适合这个场景
@tool
def query_db(table: str, columns: List[str]) -> str:
    # 问题：无法方便地管理数据库连接状态
    # 问题：每次调用都要重新连接数据库
    pass

# ✅ StructuredTool 适合有状态的工具
db_tool_instance = DatabaseQueryTool(connection_string)  # 初始化一次
tool = StructuredTool.from_function(
    func=db_tool_instance.query,  # 绑定实例方法
    args_schema=DatabaseQueryInput,
)
```

**优势**：
- 可以在工具类中管理连接池
- 工具实例可以复用（避免重复初始化）
- 更好的资源管理（`__init__` 和 `close()`）

### 2. Pydantic Validator 实现多层验证

**白名单验证模式**：

```python
@validator('table')
def validate_table(cls, v):
    allowed_tables = ['users', 'products', 'orders', 'reviews']
    if v not in allowed_tables:
        raise ValueError(f"表名 '{v}' 不在允许列表中")
    return v
```

**为什么需要白名单？**
- 防止访问敏感表（如 `admin_users`, `api_keys`）
- 限制 Agent 的权限范围
- 符合最小权限原则

**范围限制**：

```python
limit: int = Field(default=10, ge=1, le=100)
```

- `ge=1`：大于等于 1
- `le=100`：小于等于 100
- 防止查询过多数据导致性能问题

### 3. SQL 注入防护

**参数化查询**：

```python
# ❌ 危险：字符串拼接
query_str = f"SELECT * FROM {table} WHERE {where_clause}"

# ✅ 安全：参数化查询
query = text("SELECT * FROM users WHERE id = :user_id")
result = conn.execute(query, {"user_id": user_id})
```

**关键字检测**：

```python
dangerous_keywords = ['DROP', 'DELETE', 'UPDATE', 'INSERT', 'EXEC']
if any(keyword in where_clause.upper() for keyword in dangerous_keywords):
    raise ValueError("包含非法关键字")
```

### 4. 连接池管理

**为什么需要连接池？**

```python
# ❌ 每次查询都创建新连接（慢）
def query():
    conn = create_connection()
    result = conn.execute(...)
    conn.close()

# ✅ 使用连接池（快）
engine = create_engine(url, poolclass=QueuePool, pool_size=5)
with engine.connect() as conn:  # 从池中获取连接
    result = conn.execute(...)
# 连接自动归还到池中
```

**连接池配置**：
- `pool_size=5`：常驻连接数
- `max_overflow=10`：高峰期额外连接数
- `pool_recycle=3600`：1小时后回收连接（防止连接过期）

### 4. 权限控制

```python
class DatabaseQueryTool:
    def __init__(self, connection_string: str, user_role: str):
        self.user_role = user_role
        self.allowed_tables = self._get_allowed_tables(user_role)

    def _get_allowed_tables(self, role: str) -> List[str]:
        # 根据角色返回允许的表
        role_permissions = {
            "admin": ["users", "products", "orders", "reviews", "analytics"],
            "analyst": ["products", "orders", "reviews"],
            "customer": ["products", "reviews"],
        }
        return role_permissions.get(role, [])
```
