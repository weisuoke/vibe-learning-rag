# 场景2：RAG 检索容错管道

> **实战代码 - 场景2** | 预计阅读：10分钟
> 多源检索 + Fallback 链 + 降级策略

---

## 场景描述

在 RAG 应用中，检索是关键环节，但可能面临多种失败：
- **向量数据库不可用**：网络故障、服务重启
- **检索超时**：大规模数据集查询慢
- **检索结果为空**：查询词不匹配
- **质量不足**：相似度分数过低

解决方案：**多源检索 + Fallback 链**
- 主检索源失败 → 备用检索源
- 所有检索源失败 → 缓存结果
- 缓存也没有 → 默认响应

---

## 完整代码

```python
"""
场景2：RAG 检索容错管道
演示：多源检索 + Fallback 链 + 降级策略
"""

import os
import time
from typing import TypedDict, List, Dict, Optional
from openai import OpenAI

from langgraph.graph import StateGraph, START, END
from langgraph.types import RetryPolicy

# ===== 1. 定义状态 =====
class State(TypedDict):
    query: str
    documents: List[Dict[str, str]]  # [{"content": "...", "source": "..."}]
    answer: str
    retrieval_source: str  # 记录使用了哪个检索源

# ===== 2. 模拟向量数据库 =====
class VectorDB:
    """模拟向量数据库（可能失败）"""
    def __init__(self, name: str, fail_rate: float = 0.0):
        self.name = name
        self.fail_rate = fail_rate
        self.call_count = 0

    def search(self, query: str, top_k: int = 3) -> List[Dict[str, str]]:
        """模拟检索"""
        self.call_count += 1

        # 模拟失败
        import random
        if random.random() < self.fail_rate:
            raise ConnectionError(f"{self.name} 连接失败")

        # 模拟延迟
        time.sleep(0.1)

        # 返回模拟结果
        return [
            {"content": f"{self.name} 文档1：LangGraph 是工作流框架", "source": self.name},
            {"content": f"{self.name} 文档2：支持状态管理和重试", "source": self.name},
            {"content": f"{self.name} 文档3：适合构建 Agent 系统", "source": self.name},
        ][:top_k]

# 创建多个检索源
primary_db = VectorDB("Milvus主库", fail_rate=0.3)    # 30% 失败率
backup_db = VectorDB("ChromaDB备库", fail_rate=0.1)   # 10% 失败率
cache_db = VectorDB("Redis缓存", fail_rate=0.0)       # 不失败

# ===== 3. 检索节点（带重试） =====
def retrieve_from_primary(state: State) -> dict:
    """从主数据库检索"""
    print(f"\n[检索] 尝试从主库检索...")
    print(f"  查询: {state['query']}")

    try:
        docs = primary_db.search(state['query'])
        print(f"[检索] ✅ 主库成功返回 {len(docs)} 个文档")
        return {
            "documents": docs,
            "retrieval_source": "primary"
        }
    except Exception as e:
        print(f"[检索] ❌ 主库失败: {e}")
        raise  # 让 RetryPolicy 处理

def retrieve_from_backup(state: State) -> dict:
    """从备用数据库检索（Fallback）"""
    print(f"\n[Fallback] 主库失败，尝试备库...")
    print(f"  查询: {state['query']}")

    try:
        docs = backup_db.search(state['query'])
        print(f"[Fallback] ✅ 备库成功返回 {len(docs)} 个文档")
        return {
            "documents": docs,
            "retrieval_source": "backup"
        }
    except Exception as e:
        print(f"[Fallback] ❌ 备库也失败: {e}")
        raise

def retrieve_from_cache(state: State) -> dict:
    """从缓存检索（最终 Fallback）"""
    print(f"\n[Fallback] 所有数据库失败，尝试缓存...")
    print(f"  查询: {state['query']}")

    try:
        docs = cache_db.search(state['query'], top_k=1)
        print(f"[Fallback] ✅ 缓存返回 {len(docs)} 个文档")
        return {
            "documents": docs,
            "retrieval_source": "cache"
        }
    except Exception as e:
        print(f"[Fallback] ❌ 缓存也失败: {e}")
        # 返回默认文档（最终降级）
        print(f"[Fallback] ⚠️  使用默认文档")
        return {
            "documents": [{"content": "抱歉，暂时无法检索相关文档。", "source": "default"}],
            "retrieval_source": "default"
        }

# ===== 4. 生成节点 =====
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

def generate_answer(state: State) -> dict:
    """基于检索结果生成答案"""
    print(f"\n[生成] 基于 {len(state['documents'])} 个文档生成答案...")
    print(f"  检索源: {state['retrieval_source']}")

    # 构建上下文
    context = "\n".join([f"- {doc['content']}" for doc in state['documents']])

    try:
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "你是一个有帮助的助手。"},
                {"role": "user", "content": f"基于以下文档回答问题：\n\n{context}\n\n问题：{state['query']}"}
            ],
            temperature=0.7,
            max_tokens=300,
            timeout=30.0
        )

        answer = response.choices[0].message.content
        print(f"[生成] ✅ 成功生成答案 ({len(answer)} 字符)")

        return {"answer": answer}

    except Exception as e:
        print(f"[生成] ❌ 生成失败: {e}")
        raise

# ===== 5. 路由逻辑 =====
def should_try_backup(state: State) -> str:
    """判断是否需要尝试备库"""
    if state.get("documents"):
        return "generate"  # 已有文档，直接生成
    return "backup"  # 没有文档，尝试备库

def should_try_cache(state: State) -> str:
    """判断是否需要尝试缓存"""
    if state.get("documents"):
        return "generate"  # 已有文档，直接生成
    return "cache"  # 没有文档，尝试缓存

# ===== 6. 配置重试策略 =====
# 主库：多次重试，间隔短
primary_retry = RetryPolicy(
    max_attempts=3,
    initial_interval=0.5,
    backoff_factor=2.0,
    retry_on=ConnectionError
)

# 备库：少量重试
backup_retry = RetryPolicy(
    max_attempts=2,
    initial_interval=0.3,
    retry_on=ConnectionError
)

# 缓存：不重试（直接降级）
# 生成：标准重试
generate_retry = RetryPolicy(max_attempts=3)

# ===== 7. 构建工作流 =====
builder = StateGraph(State)

# 添加节点
builder.add_node("primary", retrieve_from_primary, retry_policy=primary_retry)
builder.add_node("backup", retrieve_from_backup, retry_policy=backup_retry)
builder.add_node("cache", retrieve_from_cache)  # 不重试
builder.add_node("generate", generate_answer, retry_policy=generate_retry)

# 添加边
builder.add_edge(START, "primary")

# 主库 → 判断 → 备库或生成
builder.add_conditional_edges(
    "primary",
    should_try_backup,
    {"backup": "backup", "generate": "generate"}
)

# 备库 → 判断 → 缓存或生成
builder.add_conditional_edges(
    "backup",
    should_try_cache,
    {"cache": "cache", "generate": "generate"}
)

# 缓存 → 生成
builder.add_edge("cache", "generate")

# 生成 → 结束
builder.add_edge("generate", END)

graph = builder.compile()

print("✅ RAG 容错管道构建完成\n")

# ===== 8. 测试场景 =====
print("="*60)
print("测试1：主库成功")
print("="*60)

# 重置计数器
primary_db.call_count = 0
backup_db.call_count = 0
cache_db.call_count = 0

# 临时降低主库失败率
primary_db.fail_rate = 0.0

initial_state = {
    "query": "什么是 LangGraph？",
    "documents": [],
    "answer": "",
    "retrieval_source": ""
}

try:
    result = graph.invoke(initial_state)
    print(f"\n✅ 执行成功")
    print(f"  检索源: {result['retrieval_source']}")
    print(f"  文档数: {len(result['documents'])}")
    print(f"  答案: {result['answer'][:100]}...")
    print(f"\n调用统计:")
    print(f"  主库: {primary_db.call_count} 次")
    print(f"  备库: {backup_db.call_count} 次")
    print(f"  缓存: {cache_db.call_count} 次")
except Exception as e:
    print(f"\n❌ 执行失败: {type(e).__name__}: {e}")

# ===== 9. 测试主库失败场景 =====
print("\n" + "="*60)
print("测试2：主库失败，备库成功")
print("="*60)

# 重置计数器
primary_db.call_count = 0
backup_db.call_count = 0
cache_db.call_count = 0

# 主库必定失败
primary_db.fail_rate = 1.0
backup_db.fail_rate = 0.0

try:
    result = graph.invoke(initial_state)
    print(f"\n✅ 执行成功（通过 Fallback）")
    print(f"  检索源: {result['retrieval_source']}")
    print(f"  文档数: {len(result['documents'])}")
    print(f"  答案: {result['answer'][:100]}...")
    print(f"\n调用统计:")
    print(f"  主库: {primary_db.call_count} 次（全部失败）")
    print(f"  备库: {backup_db.call_count} 次")
    print(f"  缓存: {cache_db.call_count} 次")
except Exception as e:
    print(f"\n❌ 执行失败: {type(e).__name__}: {e}")

# ===== 10. 测试全部失败场景 =====
print("\n" + "="*60)
print("测试3：主库和备库都失败，使用缓存")
print("="*60)

# 重置计数器
primary_db.call_count = 0
backup_db.call_count = 0
cache_db.call_count = 0

# 主库和备库都失败
primary_db.fail_rate = 1.0
backup_db.fail_rate = 1.0

try:
    result = graph.invoke(initial_state)
    print(f"\n✅ 执行成功（通过缓存降级）")
    print(f"  检索源: {result['retrieval_source']}")
    print(f"  文档数: {len(result['documents'])}")
    print(f"  答案: {result['answer'][:100]}...")
    print(f"\n调用统计:")
    print(f"  主库: {primary_db.call_count} 次（全部失败）")
    print(f"  备库: {backup_db.call_count} 次（全部失败）")
    print(f"  缓存: {cache_db.call_count} 次")
except Exception as e:
    print(f"\n❌ 执行失败: {type(e).__name__}: {e}")

# ===== 11. 实际应用建议 =====
print("\n" + "="*60)
print("实际应用建议")
print("="*60)

recommendations = """
1. **多源检索架构**
   - 主数据库（高性能）→ 备用数据库（高可用）→ 缓存（低延迟）→ 默认响应
   - 每一层都是更低成本/更低质量的替代方案
   - 确保系统始终有响应

2. **差异化重试策略**
   - 主库：多次重试（3次），因为它最优
   - 备库：少量重试（2次），快速失败转向缓存
   - 缓存：不重试，直接返回或降级

3. **检索质量评估**
   - 检查相似度分数阈值（如 > 0.7）
   - 检查文档数量（至少 1-3 个）
   - 质量不足时主动触发 Fallback

4. **缓存策略**
   - 缓存热门查询结果（TTL 1小时）
   - 缓存最近查询结果（LRU 淘汰）
   - 缓存默认响应（永久）

5. **监控与告警**
   - 记录每次 Fallback 的原因
   - 统计各检索源的成功率
   - 主库成功率 < 95% 时告警

6. **降级策略**
   - 主库失败 → 备库（完整功能）
   - 备库失败 → 缓存（部分功能）
   - 缓存失败 → 默认响应（最小功能）
   - 生成失败 → 返回检索结果（无生成）
"""

print(recommendations)

print("\n" + "="*60)
print("✅ 场景2 完成")
print("="*60)
```

## 关键要点

### 1. Fallback 链设计

```python
主库（最优） → 备库（次优） → 缓存（可用） → 默认（保底）
   ↓ 失败         ↓ 失败         ↓ 失败         ↓ 总是成功
```

### 2. 差异化重试策略

```python
# 主库：多次重试（因为它最优）
RetryPolicy(max_attempts=3, initial_interval=0.5)

# 备库：少量重试（快速失败）
RetryPolicy(max_attempts=2, initial_interval=0.3)

# 缓存：不重试（直接降级）
# 不传 retry_policy
```

### 3. 条件路由

```python
def should_try_backup(state: State) -> str:
    if state.get("documents"):
        return "generate"  # 已有文档
    return "backup"  # 尝试备库
```

### 4. 最终降级

```python
# 所有检索源都失败时，返回默认文档
return {
    "documents": [{"content": "抱歉，暂时无法检索相关文档。", "source": "default"}],
    "retrieval_source": "default"
}
```
