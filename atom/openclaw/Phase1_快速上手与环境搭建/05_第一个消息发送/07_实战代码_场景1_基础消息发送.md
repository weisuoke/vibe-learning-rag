### 4. src/gateway.ts

```typescript
/**
 * Gateway 管理模块
 * 负责启动、停止和监控 Gateway
 */

import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export class GatewayManager {
  private gatewayProcess: any = null;
  private port: number;
  private host: string;

  constructor(port: number = 3000, host: string = 'localhost') {
    this.port = port;
    this.host = host;
  }

  /**
   * 启动 Gateway
   */
  async start(): Promise<void> {
    console.log('🚀 Starting Gateway...');

    try {
      // 检查 Gateway 是否已经在运行
      const isRunning = await this.isGatewayRunning();
      if (isRunning) {
        console.log('✓ Gateway is already running');
        return;
      }

      // 启动 Gateway
      const { stdout, stderr } = await execAsync(
        `openclaw gateway start --port ${this.port} --host ${this.host}`
      );

      if (stderr && !stderr.includes('Gateway started')) {
        throw new Error(`Failed to start Gateway: ${stderr}`);
      }

      console.log('✓ Gateway started successfully');
      console.log(`✓ Listening on ${this.host}:${this.port}`);

      // 等待 Gateway 完全启动
      await this.waitForGateway();
    } catch (error) {
      console.error('✗ Failed to start Gateway:', error);
      throw error;
    }
  }

  /**
   * 停止 Gateway
   */
  async stop(): Promise<void> {
    console.log('🛑 Stopping Gateway...');

    try {
      await execAsync('openclaw gateway stop');
      console.log('✓ Gateway stopped');
    } catch (error) {
      console.error('✗ Failed to stop Gateway:', error);
      throw error;
    }
  }

  /**
   * 检查 Gateway 状态
   */
  async getStatus(): Promise<GatewayStatus> {
    try {
      const { stdout } = await execAsync('openclaw gateway status');
      return this.parseGatewayStatus(stdout);
    } catch (error) {
      return {
        running: false,
        port: this.port,
        uptime: 0,
        messageCount: 0
      };
    }
  }

  /**
   * 检查 Gateway 是否正在运行
   */
  private async isGatewayRunning(): Promise<boolean> {
    const status = await this.getStatus();
    return status.running;
  }

  /**
   * 等待 Gateway 完全启动
   */
  private async waitForGateway(maxAttempts: number = 10): Promise<void> {
    for (let i = 0; i < maxAttempts; i++) {
      const isRunning = await this.isGatewayRunning();
      if (isRunning) {
        return;
      }
      await this.delay(1000);
    }
    throw new Error('Gateway failed to start within timeout');
  }

  /**
   * 解析 Gateway 状态输出
   */
  private parseGatewayStatus(output: string): GatewayStatus {
    const running = output.includes('Running');
    const portMatch = output.match(/Port: (\d+)/);
    const uptimeMatch = output.match(/Uptime: (\d+)/);
    const messageCountMatch = output.match(/Messages: (\d+)/);

    return {
      running,
      port: portMatch ? parseInt(portMatch[1]) : this.port,
      uptime: uptimeMatch ? parseInt(uptimeMatch[1]) : 0,
      messageCount: messageCountMatch ? parseInt(messageCountMatch[1]) : 0
    };
  }

  /**
   * 延迟函数
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

/**
 * Gateway 状态接口
 */
export interface GatewayStatus {
  running: boolean;
  port: number;
  uptime: number;
  messageCount: number;
}
```

```typescript
/**
 * 配对管理模块
 * 负责通道配对和配对状态管理
 */

import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export class PairingManager {
  /**
   * 启动配对流程
   */
  async startPairing(channelType: string): Promise<PairingResult> {
    console.log(`🔗 Starting pairing for ${channelType}...`);

    try {
      // 启动配对命令
      const { stdout, stderr } = await execAsync(
        `openclaw pairing start --channel ${channelType}`
      );

      if (stderr && !stderr.includes('Pairing successful')) {
        throw new Error(`Pairing failed: ${stderr}`);
      }

      // 解析配对结果
      const result = this.parsePairingOutput(stdout);

      console.log('✓ Pairing successful');
      console.log(`✓ Channel: ${result.channelType}`);
      console.log(`✓ User: ${result.user}`);
      console.log(`✓ Pairing ID: ${result.pairingId}`);

      return result;
    } catch (error) {
      console.error('✗ Pairing failed:', error);
      throw error;
    }
  }

  /**
   * 获取所有配对列表
   */
  async listPairings(): Promise<Pairing[]> {
    try {
      const { stdout } = await execAsync('openclaw pairing list');
      return this.parsePairingList(stdout);
    } catch (error) {
      console.error('✗ Failed to list pairings:', error);
      return [];
    }
  }

  /**
   * 检查是否有活跃的配对
   */
  async hasActivePairing(): Promise<boolean> {
    const pairings = await this.listPairings();
    return pairings.some(p => p.status === 'active');
  }

  /**
   * 获取特定通道的配对
   */
  async getPairingByChannel(channelType: string): Promise<Pairing | null> {
    const pairings = await this.listPairings();
    return pairings.find(p =>
      p.channelType === channelType && p.status === 'active'
    ) || null;
  }

  /**
   * 删除配对
   */
  async deletePairing(pairingId: string): Promise<void> {
    console.log(`🗑️  Deleting pairing ${pairingId}...`);

    try {
      await execAsync(`openclaw pairing delete ${pairingId}`);
      console.log('✓ Pairing deleted');
    } catch (error) {
      console.error('✗ Failed to delete pairing:', error);
      throw error;
    }
  }

  /**
   * 解析配对输出
   */
  private parsePairingOutput(output: string): PairingResult {
    const channelMatch = output.match(/Channel: (\w+)/);
    const userMatch = output.match(/User: (.+)/);
    const pairingIdMatch = output.match(/Pairing ID: (.+)/);

    return {
      channelType: channelMatch ? channelMatch[1] : 'unknown',
      user: userMatch ? userMatch[1] : 'unknown',
      pairingId: pairingIdMatch ? pairingIdMatch[1] : 'unknown',
      status: 'active'
    };
  }

  /**
   * 解析配对列表输出
   */
  private parsePairingList(output: string): Pairing[] {
    const lines = output.split('\n').filter(line => line.includes('│'));
    const pairings: Pairing[] = [];

    for (const line of lines) {
      if (line.includes('Channel') || line.includes('─')) {
        continue;
      }

      const parts = line.split('│').map(p => p.trim()).filter(p => p);
      if (parts.length >= 4) {
        pairings.push({
          channelType: parts[0],
          user: parts[1],
          pairingId: parts[2],
          status: parts[3].includes('Active') ? 'active' : 'inactive'
        });
      }
    }

    return pairings;
  }
}

/**
 * 配对结果接口
 */
export interface PairingResult {
  channelType: string;
  user: string;
  pairingId: string;
  status: 'active' | 'inactive';
}

/**
 * 配对接口
 */
export interface Pairing extends PairingResult {
  // 继承 PairingResult 的所有字段
}
```

```typescript
/**
 * 消息发送模块
 * 负责发送消息和处理响应
 */

import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export class MessageSender {
  /**
   * 发送文本消息
   */
  async sendText(text: string, options?: SendOptions): Promise<SendResult> {
    console.log(`📤 Sending message: "${text}"`);

    try {
      // 构建命令
      let command = `openclaw message send "${text}"`;

      if (options?.channel) {
        command += ` --channel ${options.channel}`;
      }

      if (options?.format) {
        command += ` --format ${options.format}`;
      }

      if (options?.silent) {
        command += ` --silent`;
      }

      // 执行命令
      const { stdout, stderr } = await execAsync(command);

      if (stderr && !stderr.includes('Message sent')) {
        throw new Error(`Failed to send message: ${stderr}`);
      }

      // 解析结果
      const result = this.parseSendOutput(stdout);

      console.log('✓ Message sent successfully');
      console.log(`✓ Channel: ${result.channel}`);
      console.log(`✓ Recipient: ${result.recipient}`);
      console.log(`✓ Message ID: ${result.messageId}`);
      console.log(`✓ Status: ${result.status}`);

      return result;
    } catch (error) {
      console.error('✗ Failed to send message:', error);
      throw error;
    }
  }

  /**
   * 发送带附件的消息
   */
  async sendWithAttachment(
    text: string,
    attachmentPath: string,
    options?: SendOptions
  ): Promise<SendResult> {
    console.log(`📤 Sending message with attachment: "${text}"`);
    console.log(`📎 Attachment: ${attachmentPath}`);

    try {
      let command = `openclaw message send "${text}" --attachment "${attachmentPath}"`;

      if (options?.channel) {
        command += ` --channel ${options.channel}`;
      }

      const { stdout, stderr } = await execAsync(command);

      if (stderr && !stderr.includes('Message sent')) {
        throw new Error(`Failed to send message: ${stderr}`);
      }

      const result = this.parseSendOutput(stdout);

      console.log('✓ Message with attachment sent successfully');

      return result;
    } catch (error) {
      console.error('✗ Failed to send message with attachment:', error);
      throw error;
    }
  }

  /**
   * 批量发送消息
   */
  async sendBatch(messages: string[], options?: SendOptions): Promise<SendResult[]> {
    console.log(`📤 Sending ${messages.length} messages...`);

    const results: SendResult[] = [];

    for (const message of messages) {
      try {
        const result = await this.sendText(message, options);
        results.push(result);

        // 避免频率限制
        await this.delay(1000);
      } catch (error) {
        console.error(`✗ Failed to send message: "${message}"`, error);
        results.push({
          success: false,
          channel: 'unknown',
          recipient: 'unknown',
          messageId: 'failed',
          status: 'failed',
          error: error instanceof Error ? error.message : 'Unknown error'
        });
      }
    }

    console.log(`✓ Sent ${results.filter(r => r.success).length}/${messages.length} messages`);

    return results;
  }

  /**
   * 解析发送输出
   */
  private parseSendOutput(output: string): SendResult {
    const channelMatch = output.match(/Channel: (\w+)/);
    const recipientMatch = output.match(/Recipient: (.+)/);
    const messageIdMatch = output.match(/Message ID: (.+)/);
    const statusMatch = output.match(/Status: (\w+)/);

    return {
      success: true,
      channel: channelMatch ? channelMatch[1] : 'unknown',
      recipient: recipientMatch ? recipientMatch[1] : 'unknown',
      messageId: messageIdMatch ? messageIdMatch[1] : 'unknown',
      status: statusMatch ? statusMatch[1] : 'unknown'
    };
  }

  /**
   * 延迟函数
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

/**
 * 发送选项接口
 */
export interface SendOptions {
  channel?: string;
  format?: 'text' | 'markdown' | 'html';
  silent?: boolean;
}

/**
 * 发送结果接口
 */
export interface SendResult {
  success: boolean;
  channel: string;
  recipient: string;
  messageId: string;
  status: string;
  error?: string;
}
```

### 7. src/index.ts

```typescript
/**
 * OpenClaw 基础消息发送示例
 * 演示如何启动 Gateway、配对通道、发送消息
 */

import 'dotenv/config';
import { GatewayManager } from './gateway.js';
import { PairingManager } from './pairing.js';
import { MessageSender } from './message.js';

async function main() {
  console.log('🎯 OpenClaw Basic Message Sending Example\n');

  // 1. 初始化管理器
  const gateway = new GatewayManager(
    parseInt(process.env.GATEWAY_PORT || '3000'),
    process.env.GATEWAY_HOST || 'localhost'
  );
  const pairing = new PairingManager();
  const message = new MessageSender();

  try {
    // 2. 启动 Gateway
    console.log('Step 1: Starting Gateway');
    console.log('─'.repeat(50));
    await gateway.start();
    console.log();

    // 3. 检查配对状态
    console.log('Step 2: Checking Pairing Status');
    console.log('─'.repeat(50));
    const hasActivePairing = await pairing.hasActivePairing();

    if (!hasActivePairing) {
      console.log('⚠️  No active pairing found');
      console.log('💡 Please run: openclaw pairing start');
      console.log('💡 Then run this script again');
      return;
    }

    const pairings = await pairing.listPairings();
    console.log(`✓ Found ${pairings.length} active pairing(s):`);
    pairings.forEach(p => {
      console.log(`  - ${p.channelType}: ${p.user}`);
    });
    console.log();

    // 4. 发送基础消息
    console.log('Step 3: Sending Basic Message');
    console.log('─'.repeat(50));
    await message.sendText('Hello from OpenClaw! 👋');
    console.log();

    // 5. 发送 Markdown 格式消息
    console.log('Step 4: Sending Markdown Message');
    console.log('─'.repeat(50));
    await message.sendText(
      '**Bold text** and *italic text*\n[Link](https://example.com)',
      { format: 'markdown' }
    );
    console.log();

    // 6. 批量发送消息
    console.log('Step 5: Sending Batch Messages');
    console.log('─'.repeat(50));
    const messages = [
      'Message 1: Testing batch send',
      'Message 2: This is the second message',
      'Message 3: Final message in batch'
    ];
    await message.sendBatch(messages);
    console.log();

    // 7. 显示 Gateway 状态
    console.log('Step 6: Gateway Status');
    console.log('─'.repeat(50));
    const status = await gateway.getStatus();
    console.log(`✓ Gateway is ${status.running ? 'running' : 'stopped'}`);
    console.log(`✓ Port: ${status.port}`);
    console.log(`✓ Uptime: ${status.uptime}s`);
    console.log(`✓ Messages sent: ${status.messageCount}`);
    console.log();

    console.log('✅ All steps completed successfully!');
  } catch (error) {
    console.error('\n❌ Error:', error);
    process.exit(1);
  }
}

// 运行主函数
main().catch(console.error);
```

## 常见问题

### Q1: Gateway 启动失败

**错误**：`Port 3000 already in use`

**解决方案**：
```bash
# 停止占用端口的进程
lsof -ti:3000 | xargs kill -9

# 或使用其他端口
export GATEWAY_PORT=3001
npm start
```

### Q2: 没有活跃的配对

**错误**：`No active pairing found`

**解决方案**：
```bash
# 启动配对流程
openclaw pairing start

# 选择通道并完成配对
# 然后重新运行示例
npm start
```

### Q3: 消息发送失败

**错误**：`Failed to send message`

**解决方案**：
```bash
# 检查 Gateway 状态
openclaw gateway status

# 检查配对状态
openclaw pairing list

# 重启 Gateway
openclaw gateway restart
```

```typescript
async sendTextWithRetry(
  text: string,
  options?: SendOptions,
  maxRetries: number = 3
): Promise<SendResult> {
  let lastError: Error;

  for (let i = 0; i < maxRetries; i++) {
    try {
      return await this.sendText(text, options);
    } catch (error) {
      lastError = error as Error;
      console.log(`⚠️  Retry ${i + 1}/${maxRetries}...`);
      await this.delay(1000 * (i + 1));
    }
  }

  throw lastError!;
}
```

```typescript
class MessageQueue {
  private queue: string[] = [];
  private processing = false;

  async enqueue(message: string): Promise<void> {
    this.queue.push(message);
    if (!this.processing) {
      await this.process();
    }
  }

  private async process(): Promise<void> {
    this.processing = true;
    while (this.queue.length > 0) {
      const message = this.queue.shift()!;
      await this.sendText(message);
      await this.delay(1000);
    }
    this.processing = false;
  }
}
```
