## 最小示例 1：基础标量过滤

### 步骤 1：创建集合

```python
from pymilvus import MilvusClient

# 连接 Milvus
client = MilvusClient("http://localhost:19530")

# 创建集合（自动创建 schema）
client.create_collection(
    collection_name="products",
    dimension=384,  # 向量维度
    auto_id=True
)
```

### 步骤 2：插入数据

```python
# 准备数据
data = [
    {
        "vector": [0.1] * 384,  # 向量
        "price": 99.99,
        "stock": 100,
        "brand": "Apple"
    },
    {
        "vector": [0.2] * 384,
        "price": 199.99,
        "stock": 50,
        "brand": "Samsung"
    },
    {
        "vector": [0.3] * 384,
        "price": 299.99,
        "stock": 0,
        "brand": "Huawei"
    }
]

# 插入数据
client.insert(
    collection_name="products",
    data=data
)
```

### 步骤 3：创建索引

```python
# 为标量字段创建索引
index_params = client.prepare_index_params()

# 向量索引
index_params.add_index(
    field_name="vector",
    index_type="AUTOINDEX",
    metric_type="COSINE"
)

# 标量索引
index_params.add_index(
    field_name="price",
    index_type="INVERTED"
)

index_params.add_index(
    field_name="stock",
    index_type="INVERTED"
)

index_params.add_index(
    field_name="brand",
    index_type="BITMAP"  # 低基数字段使用 BITMAP
)

# 创建索引
client.create_index(
    collection_name="products",
    index_params=index_params
)
```

### 步骤 4：标量过滤查询

```python
# 查询：价格在100-200元，有库存的商品
query_vector = [0.15] * 384

results = client.search(
    collection_name="products",
    data=[query_vector],
    filter="price >= 100 AND price <= 200 AND stock > 0",
    limit=10,
    output_fields=["price", "stock", "brand"]
)

# 打印结果
for hits in results:
    for hit in hits:
        print(f"ID: {hit['id']}, Price: {hit['entity']['price']}, "
              f"Stock: {hit['entity']['stock']}, Brand: {hit['entity']['brand']}")
```

**输出示例**：
```
ID: 2, Price: 199.99, Stock: 50, Brand: Samsung
```

## 最小示例 2：JSON Path Index

### 步骤 1：创建带 JSON 字段的集合

```python
from pymilvus import MilvusClient, DataType

client = MilvusClient("http://localhost:19530")

# 创建 schema
schema = client.create_schema(auto_id=True)

# 添加字段
schema.add_field("id", DataType.INT64, is_primary=True)
schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
schema.add_field("metadata", DataType.JSON)  # JSON 字段

# 创建集合
client.create_collection(
    collection_name="products_json",
    schema=schema
)
```

### 步骤 2：插入 JSON 数据

```python
# 准备数据
data = [
    {
        "vector": [0.1] * 384,
        "metadata": {
            "product": {
                "category": "electronics",
                "brand": "Apple"
            },
            "pricing": {
                "base": 999.99,
                "final": 899.99
            },
            "tags": ["hot", "new", "5G"]
        }
    },
    {
        "vector": [0.2] * 384,
        "metadata": {
            "product": {
                "category": "electronics",
                "brand": "Samsung"
            },
            "pricing": {
                "base": 799.99,
                "final": 699.99
            },
            "tags": ["sale", "4G"]
        }
    }
]

# 插入数据
client.insert(
    collection_name="products_json",
    data=data
)
```

### 步骤 3：创建 JSON Path Index

```python
# 准备索引参数
index_params = client.prepare_index_params()

# 向量索引
index_params.add_index(
    field_name="vector",
    index_type="AUTOINDEX",
    metric_type="COSINE"
)

# JSON Path Index - category
index_params.add_index(
    field_name="metadata",
    index_name="category_index",
    index_type="INVERTED",
    params={
        "json_path": "metadata['product']['category']",
        "json_cast_type": "varchar"
    }
)

# JSON Path Index - brand
index_params.add_index(
    field_name="metadata",
    index_name="brand_index",
    index_type="INVERTED",
    params={
        "json_path": "metadata['product']['brand']",
        "json_cast_type": "varchar"
    }
)

# JSON Path Index - price
index_params.add_index(
    field_name="metadata",
    index_name="price_index",
    index_type="INVERTED",
    params={
        "json_path": "metadata['pricing']['final']",
        "json_cast_type": "double"
    }
)

# JSON Path Index - tags array
index_params.add_index(
    field_name="metadata",
    index_name="tags_index",
    index_type="INVERTED",
    params={
        "json_path": "metadata['tags']",
        "json_cast_type": "array_varchar"
    }
)

# 创建索引
client.create_index(
    collection_name="products_json",
    index_params=index_params
)
```

### 步骤 4：JSON 过滤查询

```python
# 查询：电子产品，价格<800，包含"hot"标签
query_vector = [0.15] * 384

results = client.search(
    collection_name="products_json",
    data=[query_vector],
    filter='''
        metadata['product']['category'] == 'electronics'
        AND metadata['pricing']['final'] < 800
        AND ARRAY_CONTAINS(metadata['tags'], 'hot')
    ''',
    limit=10,
    output_fields=["metadata"]
)

# 打印结果
for hits in results:
    for hit in hits:
        print(f"ID: {hit['id']}")
        print(f"Metadata: {hit['entity']['metadata']}")
```

**输出示例**：
```
ID: 1
Metadata: {'product': {'category': 'electronics', 'brand': 'Apple'},
           'pricing': {'base': 999.99, 'final': 899.99},
           'tags': ['hot', 'new', '5G']}
```

## 最小示例 3：RAG 混合检索

### 完整代码

```python
from pymilvus import MilvusClient
from sentence_transformers import SentenceTransformer

# 1. 初始化
client = MilvusClient("http://localhost:19530")
model = SentenceTransformer('all-MiniLM-L6-v2')

# 2. 创建集合
client.create_collection(
    collection_name="documents",
    dimension=384,
    auto_id=True
)

# 3. 准备文档数据
documents = [
    {
        "text": "Python is a high-level programming language.",
        "topic": "programming",
        "language": "en",
        "quality_score": 0.9
    },
    {
        "text": "Machine learning is a subset of AI.",
        "topic": "AI",
        "language": "en",
        "quality_score": 0.95
    },
    {
        "text": "深度学习是机器学习的一个分支。",
        "topic": "AI",
        "language": "zh",
        "quality_score": 0.85
    }
]

# 4. 生成向量并插入
data = []
for doc in documents:
    vector = model.encode(doc["text"]).tolist()
    data.append({
        "vector": vector,
        "text": doc["text"],
        "topic": doc["topic"],
        "language": doc["language"],
        "quality_score": doc["quality_score"]
    })

client.insert(collection_name="documents", data=data)

# 5. 创建索引
index_params = client.prepare_index_params()
index_params.add_index("vector", "AUTOINDEX", "COSINE")
index_params.add_index("topic", "BITMAP")
index_params.add_index("language", "BITMAP")
index_params.add_index("quality_score", "INVERTED")
client.create_index("documents", index_params)

# 6. 混合检索
query = "What is machine learning?"
query_vector = model.encode(query).tolist()

results = client.search(
    collection_name="documents",
    data=[query_vector],
    filter="topic == 'AI' AND language == 'en' AND quality_score >= 0.9",
    limit=5,
    output_fields=["text", "topic", "quality_score"]
)

# 7. 打印结果
print("查询:", query)
print("\n结果:")
for hits in results:
    for hit in hits:
        print(f"- {hit['entity']['text']}")
        print(f"  Topic: {hit['entity']['topic']}, Score: {hit['entity']['quality_score']}")
        print(f"  Distance: {hit['distance']:.4f}\n")
```

**输出示例**：
```
查询: What is machine learning?

结果:
- Machine learning is a subset of AI.
  Topic: AI, Score: 0.95
  Distance: 0.1234
```

## 常见问题

### Q1: 查询返回空结果？

**原因**：过滤条件太严格，没有匹配的数据。

**解决**：
```python
# 1. 先不加过滤，确认数据存在
results = client.search(
    collection_name="products",
    data=[query_vector],
    limit=10
)

# 2. 逐步添加过滤条件
results = client.search(
    collection_name="products",
    data=[query_vector],
    filter="price >= 100",  # 先测试单个条件
    limit=10
)
```

### Q2: 索引创建失败？

**原因**：字段不存在或类型不匹配。

**解决**：
```python
# 检查集合 schema
schema = client.describe_collection("products")
print(schema)

# 确认字段名称和类型正确
```

### Q3: JSON Path 查询报错？

**原因**：路径不存在或类型转换错误。

**解决**：
```python
# 1. 检查 JSON 数据结构
results = client.query(
    collection_name="products_json",
    filter="id > 0",
    output_fields=["metadata"],
    limit=1
)
print(results[0]["metadata"])

# 2. 确认路径正确
# 错误：metadata.product.category
# 正确：metadata['product']['category']
```

### Q4: 性能慢？

**原因**：没有创建索引。

**解决**：
```python
# 为常用过滤字段创建索引
index_params.add_index("price", "INVERTED")
index_params.add_index("stock", "INVERTED")
index_params.add_index("brand", "BITMAP")
```

## 性能优化技巧

### 1. 索引选择

```python
# 低基数字段（< 100个值）→ BITMAP
index_params.add_index("status", "BITMAP")  # 只有几个状态值

# 中高基数字段 → INVERTED
index_params.add_index("price", "INVERTED")  # 价格范围广

# 不确定 → AUTOINDEX
index_params.add_index("field", "AUTOINDEX")  # 让 Milvus 自动选择
```

### 2. 查询优化

```python
# 选择性高的条件在前
filter = "stock > 0 AND price >= 100 AND price <= 500"
# stock > 0 选择性高（10%），先执行

# 使用 IN 替代多个 OR
filter = "city IN ['北京', '上海', '深圳']"  # 推荐
# 而不是：city == '北京' OR city == '上海' OR city == '深圳'
```

### 3. 批量操作

```python
# 批量插入
client.insert(collection_name="products", data=data_list)  # 一次插入多条

# 批量查询
query_vectors = [vector1, vector2, vector3]
results = client.search(
    collection_name="products",
    data=query_vectors,  # 一次查询多个向量
    limit=10
)
```

## 完整示例代码

### 保存为 `quick_start.py`

```python
from pymilvus import MilvusClient, DataType
from sentence_transformers import SentenceTransformer

def main():
    # 1. 连接 Milvus
    client = MilvusClient("http://localhost:19530")

    # 2. 创建集合
    if client.has_collection("quick_start"):
        client.drop_collection("quick_start")

    schema = client.create_schema(auto_id=True)
    schema.add_field("id", DataType.INT64, is_primary=True)
    schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
    schema.add_field("text", DataType.VARCHAR, max_length=1000)
    schema.add_field("category", DataType.VARCHAR, max_length=100)
    schema.add_field("price", DataType.DOUBLE)
    schema.add_field("metadata", DataType.JSON)

    client.create_collection(
        collection_name="quick_start",
        schema=schema
    )

    # 3. 准备数据
    model = SentenceTransformer('all-MiniLM-L6-v2')

    texts = [
        "Python programming tutorial",
        "Machine learning basics",
        "Deep learning with PyTorch"
    ]

    data = []
    for i, text in enumerate(texts):
        vector = model.encode(text).tolist()
        data.append({
            "vector": vector,
            "text": text,
            "category": "tech",
            "price": 99.99 + i * 50,
            "metadata": {
                "tags": ["programming", "tutorial"],
                "level": "beginner"
            }
        })

    # 4. 插入数据
    client.insert("quick_start", data)

    # 5. 创建索引
    index_params = client.prepare_index_params()
    index_params.add_index("vector", "AUTOINDEX", "COSINE")
    index_params.add_index("category", "BITMAP")
    index_params.add_index("price", "INVERTED")
    index_params.add_index("metadata", "INVERTED", params={
        "json_path": "metadata['level']",
        "json_cast_type": "varchar"
    })
    client.create_index("quick_start", index_params)

    # 6. 查询
    query = "learn machine learning"
    query_vector = model.encode(query).tolist()

    results = client.search(
        collection_name="quick_start",
        data=[query_vector],
        filter="category == 'tech' AND price < 200 AND metadata['level'] == 'beginner'",
        limit=5,
        output_fields=["text", "price", "metadata"]
    )

    # 7. 打印结果
    print(f"查询: {query}\n")
    for hits in results:
        for hit in hits:
            print(f"文本: {hit['entity']['text']}")
            print(f"价格: {hit['entity']['price']}")
            print(f"元数据: {hit['entity']['metadata']}")
            print(f"距离: {hit['distance']:.4f}\n")

if __name__ == "__main__":
    main()
```

### 运行

```bash
python quick_start.py
```
