```go
// 一致性级别枚举
type ConsistencyLevel int

const (
    Strong ConsistencyLevel = iota  // 0: 强一致性
    Bounded                          // 1: 有界一致性
    Eventually                       // 2: 最终一致性
    Session                          // 3: 会话一致性
)

// WAL接口定义
type WAL interface {
    // 读取操作（根据一致性级别）
    Read(ctx context.Context, level ConsistencyLevel) (*Record, error)

    // 写入操作
    Write(ctx context.Context, record *Record) error

    // 获取当前同步时间戳
    GetCurrentSyncTimestamp() uint64

    // 获取最新时间戳
    GetLatestTimestamp() uint64
}

// 记录结构
type Record struct {
    Timestamp uint64
    Data      []byte
    Metadata  map[string]string
}
```

```go
// Strong一致性：等待所有节点同步到最新时间戳
func (w *WAL) readWithSync(ctx context.Context) (*Record, error) {
    // 1. 获取最新时间戳
    latestTs := w.GetLatestTimestamp()

    // 2. 等待所有节点同步到最新时间戳
    if err := w.waitForSync(ctx, latestTs); err != nil {
        return nil, fmt.Errorf("wait for sync failed: %w", err)
    }

    // 3. 读取数据
    record, err := w.readRecord(ctx, latestTs)
    if err != nil {
        return nil, fmt.Errorf("read record failed: %w", err)
    }

    return record, nil
}

// 等待同步到目标时间戳
func (w *WAL) waitForSync(ctx context.Context, targetTs uint64) error {
    ticker := time.NewTicker(10 * time.Millisecond)
    defer ticker.Stop()

    timeout := time.After(5 * time.Second)  // 5秒超时

    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-timeout:
            return fmt.Errorf("wait for sync timeout")
        case <-ticker.C:
            // 检查当前同步进度
            currentTs := w.GetCurrentSyncTimestamp()
            if currentTs >= targetTs {
                return nil  // 同步完成
            }
            // 继续等待
        }
    }
}
```

```go
// Bounded一致性：等待时间窗口内的时间戳
func (w *WAL) readWithBoundedStaleness(ctx context.Context, stalenessBound time.Duration) (*Record, error) {
    // 1. 计算目标时间戳（当前时间 - staleness_bound）
    now := time.Now()
    targetTime := now.Add(-stalenessBound)
    targetTs := uint64(targetTime.UnixNano())

    // 2. 等待同步到目标时间戳（不需要等到最新）
    if err := w.waitForTimestamp(ctx, targetTs); err != nil {
        return nil, fmt.Errorf("wait for timestamp failed: %w", err)
    }

    // 3. 读取数据
    record, err := w.readRecord(ctx, targetTs)
    if err != nil {
        return nil, fmt.Errorf("read record failed: %w", err)
    }

    return record, nil
}

// 等待到达目标时间戳（Bounded版本）
func (w *WAL) waitForTimestamp(ctx context.Context, targetTs uint64) error {
    ticker := time.NewTicker(10 * time.Millisecond)
    defer ticker.Stop()

    timeout := time.After(2 * time.Second)  // 2秒超时（比Strong短）

    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-timeout:
            return fmt.Errorf("wait for timestamp timeout")
        case <-ticker.C:
            currentTs := w.GetCurrentSyncTimestamp()
            if currentTs >= targetTs {
                return nil  // 达到目标时间戳
            }
        }
    }
}
```

```go
func (w *WAL) readAsync(ctx context.Context) (*Record, error) {

    record, err := w.readRecordImmediate(ctx)

// 立即读取（不检查时间戳）
func (w *WAL) readRecordImmediate(ctx context.Context) (*Record, error) {
    w.mu.RLock()
    defer w.mu.RUnlock()

    // 从本地缓存读取（可能是旧数据）
    if w.localCache == nil || w.localCache.IsEmpty() {
        return nil, fmt.Errorf("no data available")
    }

    record := w.localCache.GetLatest()
    return record, nil
}
```

## 2. Woodpecker WAL机制分析

### 2.1 Checkpoint管理器

```go
// 基于 temp/source_code/woodpecker_wal_impl.md 的分析

// Checkpoint管理器
type CheckpointManager struct {
    currentCheckpoint uint64        // 当前checkpoint位置
    checkpointTime    time.Time     // checkpoint时间
    localData         *DataCache    // 本地数据缓存
    mu                sync.RWMutex  // 读写锁
}

// Strong一致性：等待checkpoint更新到最新
func (cm *CheckpointManager) WaitForCheckpoint(ctx context.Context, targetCp uint64) error {
    cm.mu.RLock()
    current := cm.currentCheckpoint
    cm.mu.RUnlock()

    if current >= targetCp {
        return nil  // 已经同步
    }

    // 等待checkpoint更新
    ticker := time.NewTicker(10 * time.Millisecond)
    defer ticker.Stop()

    timeout := time.After(5 * time.Second)

    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-timeout:
            return fmt.Errorf("wait for checkpoint timeout")
        case <-ticker.C:
            cm.mu.RLock()
            current = cm.currentCheckpoint
            cm.mu.RUnlock()

            if current >= targetCp {
                return nil  // 同步完成
            }
        }
    }
}

// Bounded一致性：等待checkpoint在时间窗口内
func (cm *CheckpointManager) WaitForBoundedCheckpoint(ctx context.Context, stalenessBound time.Duration) error {
    targetTime := time.Now().Add(-stalenessBound)

    cm.mu.RLock()
    currentCpTime := cm.checkpointTime
    cm.mu.RUnlock()

    if currentCpTime.After(targetTime) || currentCpTime.Equal(targetTime) {
        return nil  // checkpoint在时间窗口内
    }

    // 等待checkpoint更新到时间窗口内
    ticker := time.NewTicker(10 * time.Millisecond)
    defer ticker.Stop()

    timeout := time.After(2 * time.Second)

    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-timeout:
            return fmt.Errorf("wait for bounded checkpoint timeout")
        case <-ticker.C:
            cm.mu.RLock()
            currentCpTime = cm.checkpointTime
            cm.mu.RUnlock()

            if currentCpTime.After(targetTime) || currentCpTime.Equal(targetTime) {
                return nil  // checkpoint在时间窗口内
            }
        }
    }
}

// Eventually一致性：不等待checkpoint
func (cm *CheckpointManager) ReadWithoutWait(ctx context.Context) (*Data, error) {
    cm.mu.RLock()
    defer cm.mu.RUnlock()

    // 直接从本地数据读取（可能未完全同步）
    if cm.localData == nil {
        return nil, fmt.Errorf("no local data available")
    }

    data := cm.localData.Get()
    return data, nil
}
```

### 2.2 后台同步进程

```go
// 后台同步进程（保证最终一致）
func (cm *CheckpointManager) BackgroundSync(ctx context.Context) {
    ticker := time.NewTicker(1 * time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            // 定期同步数据
            if err := cm.syncData(ctx); err != nil {
                log.Printf("sync data failed: %v", err)
            }
        }
    }
}

// 同步数据
func (cm *CheckpointManager) syncData(ctx context.Context) error {
    // 1. 获取远程最新数据
    remoteData, err := cm.fetchRemoteData(ctx)
    if err != nil {
        return fmt.Errorf("fetch remote data failed: %w", err)
    }

    // 2. 更新本地数据
    cm.mu.Lock()
    cm.localData.Update(remoteData)
    cm.checkpointTime = time.Now()
    cm.currentCheckpoint = remoteData.Checkpoint
    cm.mu.Unlock()

    return nil
}
```

## 3. GuaranteeTs时间戳机制

### 3.1 GuaranteeTs管理器

```go
// GuaranteeTs管理器
type GuaranteeTsManager struct {
    latestTs       uint64        // 最新时间戳
    stalenessBound time.Duration // staleness bound
    mu             sync.RWMutex  // 读写锁
}

// 获取GuaranteeTs（根据一致性级别）
func (gtm *GuaranteeTsManager) GetGuaranteeTs(level ConsistencyLevel) uint64 {
    gtm.mu.RLock()
    defer gtm.mu.RUnlock()

    switch level {
    case Strong:
        // Strong: 使用最新时间戳
        return gtm.latestTs

    case Bounded:
        // Bounded: 最新时间戳 - staleness_bound
        boundedTs := gtm.latestTs - uint64(gtm.stalenessBound.Nanoseconds())
        return boundedTs

    case Eventually:
        // Eventually: 无保证，返回0
        return 0

    case Session:
        // Session: 会话内最大时间戳（简化实现）
        return gtm.latestTs

    default:
        return gtm.latestTs
    }
}

// 更新最新时间戳
func (gtm *GuaranteeTsManager) UpdateLatestTs(ts uint64) {
    gtm.mu.Lock()
    defer gtm.mu.Unlock()

    if ts > gtm.latestTs {
        gtm.latestTs = ts
    }
}
```

```go
    // 3. 执行搜索

        return nil, fmt.Errorf("execute search failed: %w", err)
    }

    return result, nil
}

// 等待ServiceTime满足GuaranteeTs

    ticker := time.NewTicker(10 * time.Millisecond)
    defer ticker.Stop()

    timeout := time.After(5 * time.Second)

    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-timeout:
            return fmt.Errorf("wait for service time timeout")
        case <-ticker.C:
            serviceTime := qn.GetServiceTime()

                return nil  // ServiceTime满足GuaranteeTs
            }
        }
    }
```

```python
import time
import threading
from enum import Enum
from typing import Optional, Dict, Any
from dataclasses import dataclass
from datetime import datetime, timedelta

class ConsistencyLevel(Enum):
    STRONG = "Strong"
    BOUNDED = "Bounded"
    EVENTUALLY = "Eventually"
    SESSION = "Session"

@dataclass
class Record:
    timestamp: int
    data: bytes
    metadata: Dict[str, Any]

class WALSimulator:
    """模拟Milvus WAL实现"""

    def __init__(self):
        self.latest_ts = 0
        self.current_sync_ts = 0
        self.local_cache = []
        self.lock = threading.RLock()
        self.staleness_bound = timedelta(seconds=0)  # Bounded默认0秒

        # 启动后台同步线程
        self.sync_thread = threading.Thread(target=self._background_sync, daemon=True)
        self.sync_thread.start()

    def write(self, data: bytes, metadata: Dict[str, Any] = None) -> int:
        """写入数据"""
        with self.lock:
            # 生成时间戳
            ts = int(time.time() * 1000000)  # 微秒
            self.latest_ts = ts

            # 创建记录
            record = Record(timestamp=ts, data=data, metadata=metadata or {})
            self.local_cache.append(record)

            print(f"✍️  Write: ts={ts}, data_len={len(data)}")
            return ts

    def read(self, level: ConsistencyLevel) -> Optional[Record]:
        """读取数据（根据一致性级别）"""
        if level == ConsistencyLevel.STRONG:
            return self._read_with_sync()
        elif level == ConsistencyLevel.BOUNDED:
            return self._read_with_bounded_staleness()
        elif level == ConsistencyLevel.EVENTUALLY:
            return self._read_async()
        elif level == ConsistencyLevel.SESSION:
            return self._read_with_sync()  # 简化实现
        else:
            return self._read_with_bounded_staleness()

    def _read_with_sync(self) -> Optional[Record]:
        """Strong一致性：等待同步到最新"""
        print(f"🔍 Strong read: waiting for sync to {self.latest_ts}")

        # 等待同步到最新时间戳
        start_time = time.time()
        while time.time() - start_time < 5:  # 5秒超时
            with self.lock:
                if self.current_sync_ts >= self.latest_ts:
                    # 同步完成，读取最新数据
                    if self.local_cache:
                        record = self.local_cache[-1]
                        print(f"✅ Strong read: found record at ts={record.timestamp}")
                        return record
                    break
            time.sleep(0.01)  # 10ms

        print(f"❌ Strong read: timeout")
        return None

    def _read_with_bounded_staleness(self) -> Optional[Record]:
        """Bounded一致性：等待时间窗口内"""
        target_time = datetime.now() - self.staleness_bound
        target_ts = int(target_time.timestamp() * 1000000)

        print(f"🔍 Bounded read: waiting for sync to {target_ts}")

        # 等待同步到目标时间戳
        start_time = time.time()
        while time.time() - start_time < 2:  # 2秒超时
            with self.lock:
                if self.current_sync_ts >= target_ts:
                    # 同步到时间窗口内，读取数据
                    if self.local_cache:
                        record = self.local_cache[-1]
                        print(f"✅ Bounded read: found record at ts={record.timestamp}")
                        return record
                    break
            time.sleep(0.01)

        print(f"❌ Bounded read: timeout")
        return None

    def _read_async(self) -> Optional[Record]:
        """Eventually一致性：不等待同步"""
        print(f"🔍 Eventually read: immediate")

        with self.lock:
            if self.local_cache:
                record = self.local_cache[-1]
                print(f"✅ Eventually read: found record at ts={record.timestamp}")
                return record

        print(f"❌ Eventually read: no data")
        return None

    def _background_sync(self):
        """后台同步线程"""
        while True:
            time.sleep(0.1)  # 100ms同步一次

            with self.lock:
                # 模拟同步进度（逐步追赶latest_ts）
                if self.current_sync_ts < self.latest_ts:
                    # 每次同步前进一点
                    gap = self.latest_ts - self.current_sync_ts
                    increment = max(1, gap // 10)  # 每次同步10%的gap
                    self.current_sync_ts += increment

                    if self.current_sync_ts >= self.latest_ts:
                        self.current_sync_ts = self.latest_ts
```

## 5. 关键洞察

### 5.1 实现差异

| 一致性级别 | 等待策略 | 超时时间 | 性能影响 |
|-----------|---------|---------|---------|
| **Strong** | 等待最新时间戳 | 5秒 | 最高延迟 |
| **Bounded** | 等待时间窗口内 | 2秒 | 中等延迟 |
| **Eventually** | 不等待 | 无 | 最低延迟 |

### 5.2 性能优化点

1. **Strong**: 减少等待时间 → 优化同步速度
2. **Bounded**: 调整staleness_bound → 平衡性能与一致性
3. **Eventually**: 无需优化 → 已是最优性能

### 5.3 生产环境建议

- 监控同步延迟（current_sync_ts vs latest_ts）
- 设置合理的超时时间
- 使用Bounded作为默认级别
- 关键场景临时提升到Strong
