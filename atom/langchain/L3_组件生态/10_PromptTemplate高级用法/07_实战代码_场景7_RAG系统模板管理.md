## 场景概述

### 为什么需要模板管理？

在 RAG 系统中，我们需要：
- **多语言支持**：中文、英文、日文等不同语言的提示词
- **模板复用**：系统提示、检索提示、生成提示可以组合使用
- **动态注入**：根据用户、时间、上下文动态调整提示词
- **版本管理**：A/B 测试不同的提示词版本

[来源: reference/search_partial_01.md | LangChain OpenTutorial 2025]

### 双重类比

**前端类比**：
- 模板管理 = React 组件库 + i18n 国际化
- Partial Variables = React Context（全局状态）
- 模板组合 = 组件组合（Composition）

**日常生活类比**：
- 模板管理 = 餐厅菜单管理系统
- Partial Variables = 固定的调料配方
- 模板组合 = 套餐组合（前菜 + 主菜 + 甜点）

class Language(Enum):
    """支持的语言"""
    ZH_CN = "zh_cn"  # 简体中文
    EN_US = "en_us"  # 英文
    JA_JP = "ja_jp"  # 日文


class RAGTemplateManager:
    """RAG 模板管理器"""

    def __init__(self):
        # 系统提示词模板（多语言）
        self.system_templates = {
            Language.ZH_CN: """你是一个专业的 AI 助手，名字是 {assistant_name}。
当前时间：{current_time}
你的任务是根据检索到的上下文回答用户问题。

重要规则：
1. 只使用提供的上下文信息回答
2. 如果上下文中没有相关信息，明确告知用户
3. 保持回答简洁、准确、专业""",

            Language.EN_US: """You are a professional AI assistant named {assistant_name}.
Current time: {current_time}
Your task is to answer user questions based on retrieved context.

Important rules:
1. Only use the provided context information
2. If no relevant information in context, clearly inform the user
3. Keep answers concise, accurate, and professional""",

            Language.JA_JP: """あなたは{assistant_name}という名前のプロフェッショナルなAIアシスタントです。
現在時刻：{current_time}
検索されたコンテキストに基づいてユーザーの質問に答えてください。

重要なルール：
1. 提供されたコンテキスト情報のみを使用する
2. コンテキストに関連情報がない場合は、ユーザーに明確に伝える
3. 回答は簡潔、正確、専門的に保つ"""
        }

        # 检索提示词模板（多语言）
        self.retrieval_templates = {
            Language.ZH_CN: """基于以下上下文信息回答问题。

上下文：
{context}

问题：{question}

回答：""",

            Language.EN_US: """Answer the question based on the following context.

Context:
{context}

Question: {question}

Answer:""",

            Language.JA_JP: """以下のコンテキスト情報に基づいて質問に答えてください。

コンテキスト：
{context}

質問：{question}

回答："""
        }

    def get_system_prompt(
        self,
        language: Language,
        assistant_name: str = "RAG Assistant"
    ) -> PromptTemplate:
        """获取系统提示词（带 partial variables）"""
        template = self.system_templates[language]

        # 使用 partial_variables 预填充固定值和动态函数
        return PromptTemplate(
            template=template,
            input_variables=[],
            partial_variables={
                "assistant_name": assistant_name,
                "current_time": lambda: datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            }
        )

    def get_retrieval_prompt(
        self,
        language: Language
    ) -> PromptTemplate:
        """获取检索提示词"""
        template = self.retrieval_templates[language]
        return PromptTemplate.from_template(template)

    def create_rag_chain_prompt(
        self,
        language: Language,
        assistant_name: str = "RAG Assistant"
    ) -> ChatPromptTemplate:
        """创建完整的 RAG 链提示词（组合系统提示 + 检索提示）"""
        system_prompt = self.get_system_prompt(language, assistant_name)
        retrieval_prompt = self.get_retrieval_prompt(language)

        # 使用 ChatPromptTemplate 组合
        return ChatPromptTemplate.from_messages([
            ("system", system_prompt.template),
            ("human", retrieval_prompt.template)
        ])

class DynamicRAGTemplate:
    """支持动态变量注入的 RAG 模板"""

    def __init__(self):
        self.base_template = PromptTemplate(
            template="""你是 {assistant_name}，当前时间：{current_time}
用户信息：{user_info}
系统状态：{system_status}

基于以下上下文回答问题：
{context}

问题：{question}

回答：""",
            input_variables=["context", "question"],
            partial_variables={
                "assistant_name": "RAG Pro",
                "current_time": self._get_current_time,
                "user_info": self._get_user_info,
                "system_status": self._get_system_status
            }
        )

    @staticmethod
    def _get_current_time() -> str:
        """动态获取当前时间"""
        return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    @staticmethod
    def _get_user_info() -> str:
        """动态获取用户信息（模拟）"""
        return "VIP用户 | 已使用 50 次查询"

    @staticmethod
    def _get_system_status() -> str:
        """动态获取系统状态（模拟）"""
        return "正常运行 | 响应时间 < 100ms"

    def format(self, context: str, question: str) -> str:
        """格式化模板"""
        return self.base_template.format(
            context=context,
            question=question
        )

class ProductionRAGSystem:
    """生产级 RAG 系统"""

    def __init__(
        self,
        language: Language = Language.ZH_CN,
        assistant_name: str = "RAG Pro",
        model_name: str = "gpt-4o-mini"
    ):
        self.language = language
        self.template_manager = RAGTemplateManager()
        self.llm = ChatOpenAI(model=model_name, temperature=0)

        # 创建 RAG 链
        self.prompt = self.template_manager.create_rag_chain_prompt(
            language, assistant_name
        )
        self.chain = self.prompt | self.llm | StrOutputParser()

    def query(
        self,
        question: str,
        context: str,
        **kwargs
    ) -> str:
        """执行 RAG 查询"""
        return self.chain.invoke({
            "question": question,
            "context": context,
            **kwargs
        })

class VersionedTemplate:
    """带版本管理的模板"""

    def __init__(self, template: str, version: str, description: str):
        self.template = template
        self.version = version
        self.description = description
        self.created_at = datetime.now()

    def to_prompt_template(self) -> PromptTemplate:
        """转换为 PromptTemplate"""
        return PromptTemplate.from_template(self.template)


class TemplateVersionManager:
    """模板版本管理器"""

    def __init__(self):
        self.templates: Dict[str, List[VersionedTemplate]] = {}

    def register(
        self,
        name: str,
        template: str,
        version: str,
        description: str
    ):
        """注册模板版本"""
        if name not in self.templates:
            self.templates[name] = []

        versioned = VersionedTemplate(template, version, description)
        self.templates[name].append(versioned)

    def get(
        self,
        name: str,
        version: Optional[str] = None
    ) -> PromptTemplate:
        """获取指定版本的模板"""
        if name not in self.templates:
            raise ValueError(f"Template '{name}' not found")

        versions = self.templates[name]

        if version is None:
            # 返回最新版本
            latest = max(versions, key=lambda v: v.created_at)
            return latest.to_prompt_template()

        # 返回指定版本
        for v in versions:
            if v.version == version:
                return v.to_prompt_template()

        raise ValueError(f"Version '{version}' not found for template '{name}'")

## 关键技术点

### 1. Partial Variables 的两种用法

**字符串固定值**：
```python
partial_variables={"assistant_name": "RAG Pro"}
```

**函数动态值**：
```python
partial_variables={"current_time": lambda: datetime.now().strftime("%Y-%m-%d %H:%M:%S")}
```

[来源: reference/search_partial_01.md]

### 2. 模板组合的注意事项

- 使用 `+` 操作符组合
- 自动合并 `input_variables`
- 检查 `partial_variables` 冲突
- 要求 `template_format` 一致

[来源: reference/source_prompttemplate_01.md]

### 3. 多语言模板管理

- 使用枚举管理语言类型
- 字典存储不同语言的模板
- 统一的接口获取模板
