# OpenClaw 安装与配置 - 核心概念 2：环境要求

**核心概念：** Node.js 22+ 特性、版本管理和环境检测

本文档深入讲解 OpenClaw 为什么需要 Node.js 22+，以及如何管理和检测 Node.js 环境。

## 1. Node.js 22+ 的关键特性

### 1.1 ESM 同步 require() 支持（稳定）

**Node.js 20:**
```javascript
// 需要 --experimental-require-module flag
node --experimental-require-module app.js

// 或者使用动态 import
const module = await import('./esm-module.mjs');
```

**Node.js 22:**
```javascript
// 直接 require ESM 模块（无需 flag）
const esmModule = require('./esm-module.mjs');

// 要求：
// 1. 模块必须是同步的（无 top-level await）
// 2. 模块必须明确标记为 ESM（.mjs 或 "type": "module"）
```

**OpenClaw 的使用场景:**
```typescript
// OpenClaw 配置加载器
// config-loader.ts
export function loadConfig(): Config {
  // 可以同步 require ESM 配置文件
  const userConfig = require('~/.openclaw/config.mjs');
  return userConfig.default;
}
```

### 1.2 内置 WebSocket 客户端（默认启用）

**Node.js 20:**
```javascript
// 需要 --experimental-websocket flag
node --experimental-websocket app.js

// 或使用第三方库
const WebSocket = require('ws');
```

**Node.js 22:**
```javascript
// 内置 WebSocket，无需外部依赖
const ws = new WebSocket('ws://localhost:3000');

ws.onopen = () => {
  console.log('Connected');
  ws.send('Hello');
};

ws.onmessage = (event) => {
  console.log('Received:', event.data);
};

ws.onerror = (error) => {
  console.error('Error:', error);
};

ws.onclose = () => {
  console.log('Disconnected');
};
```

**OpenClaw 的使用场景:**
```typescript
// OpenClaw Gateway WebSocket 服务器
// gateway.ts
import { WebSocketServer } from 'ws'; // 使用内置 WebSocket

export class Gateway {
  private wss: WebSocketServer;

  constructor(port: number) {
    this.wss = new WebSocketServer({ port });

    this.wss.on('connection', (ws) => {
      console.log('Client connected');

      ws.on('message', (data) => {
        // 处理客户端消息
        this.handleMessage(data);
      });
    });
  }

  private handleMessage(data: Buffer): void {
    // 路由到不同的 AI 通道
    const message = JSON.parse(data.toString());
    const channel = this.getChannel(message.model);
    channel.send(message);
  }
}
```

### 1.3 Maglev 编译器（性能提升）

**Maglev** 是 V8 的中层 JIT 编译器，介于 Ignition（解释器）和 TurboFan（优化编译器）之间

**性能对比:**
```typescript
// 测试脚本：CLI 启动时间
import { performance } from 'perf_hooks';

const start = performance.now();

// 模拟 OpenClaw CLI 启动
import('./cli.js').then(() => {
  const end = performance.now();
  console.log(`启动时间: ${(end - start).toFixed(2)}ms`);
});

// Node.js 20（无 Maglev）：
// 启动时间: 450.23ms

// Node.js 22（有 Maglev）：
// 启动时间: 315.67ms
// 性能提升: 30%
```

**为什么 Maglev 对 CLI 工具重要？**
- CLI 工具频繁启动和退出
- Maglev 快速生成中等优化的代码
- 减少启动延迟，提升用户体验

### 1.4 V8 引擎 12.4（新语法支持）

**新特性：**

**1. Array.fromAsync()**
```typescript
// 异步迭代器转数组
async function* asyncGenerator() {
  yield 1;
  yield 2;
  yield 3;
}

// Node.js 20: 需要手动实现
const arr20 = [];
for await (const item of asyncGenerator()) {
  arr20.push(item);
}

// Node.js 22: 使用 Array.fromAsync()
const arr22 = await Array.fromAsync(asyncGenerator());
console.log(arr22); // [1, 2, 3]
```

**OpenClaw 使用场景:**
```typescript
// 批量加载通道配置
async function* loadChannelConfigs() {
  for (const channelName of ['claude', 'gpt', 'gemini']) {
    yield await loadConfig(channelName);
  }
}

const configs = await Array.fromAsync(loadChannelConfigs());
```

**2. Set methods**
```typescript
// Set 集合操作
const setA = new Set([1, 2, 3]);
const setB = new Set([2, 3, 4]);

// Node.js 20: 需要手动实现
const union20 = new Set([...setA, ...setB]);
const intersection20 = new Set([...setA].filter(x => setB.has(x)));

// Node.js 22: 使用内置方法
const union22 = setA.union(setB);           // {1, 2, 3, 4}
const intersection22 = setA.intersection(setB); // {2, 3}
const difference22 = setA.difference(setB);     // {1}
const symmetricDifference22 = setA.symmetricDifference(setB); // {1, 4}
```

**OpenClaw 使用场景:**
```typescript
// 通道权限管理
const enabledChannels = new Set(['claude', 'gpt']);
const requestedChannels = new Set(['gpt', 'gemini']);

// 找出用户请求但未启用的通道
const unavailable = requestedChannels.difference(enabledChannels);
console.log(unavailable); // {'gemini'}
```

**使用 nvm:**

```bash
# 安装 Node.js 22

nvm alias default 22

# 查看已安装版本
nvm list

# 查看可用版本
nvm list-remote
```

**项目级版本管理:**
```bash
# 在项目根目录创建 .nvmrc
echo "22" > .nvmrc

# 自动使用项目指定版本
nvm use
# Found '/path/to/project/.nvmrc' with version <22>
# Now using node v22.18.0
```

**TypeScript 自动化脚本:**
```typescript
import { execSync } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';

/**
 * 确保使用正确的 Node.js 版本
 */
function ensureNodeVersion(requiredVersion: number): void {
  // 1. 检查当前版本
  const currentVersion = process.version;
  const currentMajor = parseInt(currentVersion.slice(1).split('.')[0]);

  if (currentMajor >= requiredVersion) {
    console.log(`✅ Node.js ${currentVersion} (满足要求)`);
    return;
  }

  console.log(`❌ Node.js ${currentVersion} (需要 ${requiredVersion}+)`);

  // 2. 检查是否安装了 nvm
  try {
    execSync('command -v nvm', { stdio: 'ignore' });
  } catch {
    console.error('❌ 未安装 nvm');
    console.error('   安装方法: https://github.com/nvm-sh/nvm');
    process.exit(1);
  }

  // 3. 使用 nvm 切换版本
  try {
    console.log(`📦 切换到 Node.js ${requiredVersion}...`);
    execSync(`nvm use ${requiredVersion}`, { stdio: 'inherit' });
    console.log(`✅ 已切换到 Node.js ${requiredVersion}`);
  } catch {
    console.log(`📦 安装 Node.js ${requiredVersion}...`);
    execSync(`nvm install ${requiredVersion}`, { stdio: 'inherit' });
    execSync(`nvm use ${requiredVersion}`, { stdio: 'inherit' });
    console.log(`✅ 已安装并切换到 Node.js ${requiredVersion}`);
  }
}

// 使用示例
ensureNodeVersion(22);
```

```typescript
/**
 * 使用 asdf 管理 Node.js 版本
 */
function ensureNodeVersionAsdf(requiredVersion: string): void {
  // 1. 检查是否安装了 asdf
  try {
    execSync('command -v asdf', { stdio: 'ignore' });
  } catch {
    console.error('❌ 未安装 asdf');
    console.error('   安装方法: https://asdf-vm.com/guide/getting-started.html');

    execSync('asdf plugin list | grep nodejs', { stdio: 'ignore' });

    console.log('📦 安装 nodejs 插件...');
    execSync('asdf plugin add nodejs', { stdio: 'inherit' });
  }

  // 3. 安装指定版本
  try {
    execSync(`asdf list nodejs | grep ${requiredVersion}`, { stdio: 'ignore' });
    console.log(`✅ Node.js ${requiredVersion} 已安装`);
  } catch {
    console.log(`📦 安装 Node.js ${requiredVersion}...`);
    execSync(`asdf install nodejs ${requiredVersion}`, { stdio: 'inherit' });
  }

  // 4. 设置全局版本
  execSync(`asdf global nodejs ${requiredVersion}`, { stdio: 'inherit' });
  console.log(`✅ 已设置 Node.js ${requiredVersion} 为全局版本`);
}

// 使用示例
ensureNodeVersionAsdf('22.18.0');
```

## 3. 环境检测脚本

### 3.1 完整的环境检测

```typescript
import { execSync } from 'child_process';
import * as os from 'os';

/**
 * 环境检测结果
 */
interface EnvironmentCheck {
  platform: string;
  nodeVersion: string;
  nodeMajor: number;
  npmVersion: string;
  hasNvm: boolean;
  hasAsdf: boolean;
  hasVolta: boolean;
  v8Version: string;
  v8Major: number;
  features: {
    esmRequire: boolean;
    webSocket: boolean;
    maglev: boolean;
    arrayFromAsync: boolean;
    setMethods: boolean;
  };
  passed: boolean;
}

/**
 * 检测 Node.js 环境
 */
function checkEnvironment(): EnvironmentCheck {
  // 1. 基本信息
  const platform = os.platform();
  const nodeVersion = process.version;
  const nodeMajor = parseInt(nodeVersion.slice(1).split('.')[0]);

  // 2. npm 版本
  const npmVersion = execSync('npm --version', { encoding: 'utf-8' }).trim();

  // 3. 版本管理工具
  const hasNvm = checkCommand('nvm');
  const hasAsdf = checkCommand('asdf');
  const hasVolta = checkCommand('volta');

  // 4. V8 版本
  const v8Version = process.versions.v8;
  const v8Major = parseInt(v8Version.split('.')[0]);

  // 5. 特性检测
  const features = {
    esmRequire: nodeMajor >= 22,
    webSocket: typeof WebSocket !== 'undefined',
    maglev: nodeMajor >= 22,
    arrayFromAsync: typeof Array.fromAsync === 'function',
    setMethods: typeof Set.prototype.union === 'function'
  };

  // 6. 是否通过检查
  const passed = nodeMajor >= 22 && Object.values(features).every(Boolean);

  return {
    platform,
    nodeVersion,
    nodeMajor,
    npmVersion,
    hasNvm,
    hasAsdf,
    hasVolta,
    v8Version,
    v8Major,
    features,
    passed
  };
}

/**
 * 检查命令是否存在
 */
function checkCommand(command: string): boolean {
  try {
    execSync(`command -v ${command}`, { stdio: 'ignore' });
    return true;
  } catch {
    return false;
  }
}

/**
 * 打印环境检测结果
 */
function printEnvironmentCheck(): void {
  const check = checkEnvironment();

  console.log('=== OpenClaw 环境检测 ===\n');

  // 基本信息
  console.log('基本信息:');
  console.log(`  平台: ${check.platform}`);
  console.log(`  Node.js: ${check.nodeVersion} ${check.nodeMajor >= 22 ? '✅' : '❌'}`);
  console.log(`  npm: ${check.npmVersion}`);
  console.log(`  V8: ${check.v8Version} ${check.v8Major >= 12 ? '✅' : '❌'}`);
  console.log('');

  // 版本管理工具
  console.log('版本管理工具:');
  console.log(`  nvm: ${check.hasNvm ? '✅ 已安装' : '❌ 未安装'}`);
  console.log(`  asdf: ${check.hasAsdf ? '✅ 已安装' : '❌ 未安装'}`);
  console.log(`  volta: ${check.hasVolta ? '✅ 已安装' : '❌ 未安装'}`);
  console.log('');

  // 特性检测
  console.log('Node.js 22 特性:');
  console.log(`  ESM 同步 require: ${check.features.esmRequire ? '✅' : '❌'}`);
  console.log(`  内置 WebSocket: ${check.features.webSocket ? '✅' : '❌'}`);
  console.log(`  Maglev 编译器: ${check.features.maglev ? '✅' : '❌'}`);
  console.log(`  Array.fromAsync: ${check.features.arrayFromAsync ? '✅' : '❌'}`);
  console.log(`  Set methods: ${check.features.setMethods ? '✅' : '❌'}`);
  console.log('');

  // 总结
  console.log('='.repeat(50));
  if (check.passed) {
    console.log('✅ 环境检测通过！可以安装 OpenClaw。');
  } else {
    console.log('❌ 环境检测失败！');
    console.log('\n升级方法:');

    if (check.hasNvm) {
      console.log('  nvm install 22');
      console.log('  nvm use 22');
    } else if (check.hasAsdf) {
      console.log('  asdf install nodejs 22.18.0');
      console.log('  asdf global nodejs 22.18.0');
    } else if (check.hasVolta) {
      console.log('  volta install node@22');
    } else {
      console.log('  访问 https://nodejs.org 下载 Node.js 22+');
    }
  }
}

// 运行检测
printEnvironmentCheck();
```

## 4. CI/CD 环境配置

### 4.1 GitHub Actions

```yaml
# .github/workflows/ci.yml
name: CI

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest

    strategy:
      matrix:
        node-version: [22.x]

    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js ${{ matrix.node-version }}
        uses: actions/setup-node@v3
        with:
          node-version: ${{ matrix.node-version }}

      - name: Verify Node.js version
        run: |
          node --version
          npm --version

      - name: Install OpenClaw
        run: npm install -g openclaw@latest

      - name: Verify installation
        run: openclaw --version

      - name: Run tests
        run: |
          openclaw gateway start &
          sleep 5
          curl http://localhost:3000/health
```
