## 示例 1：企业级 RAG 系统类型架构

### 1.1 类型系统设计

```python
from typing_extensions import TypedDict, Annotated, NotRequired
from pydantic import BaseModel, Field, validator
from langgraph.graph import StateGraph
from datetime import datetime
import uuid

# ============ 边界层：Pydantic 验证 ============

class APIQueryInput(BaseModel):
    """API 输入验证"""
    query: str = Field(..., min_length=1, max_length=1000, description="用户查询")
    user_id: str = Field(..., pattern=r'^[a-zA-Z0-9_-]+$', description="用户ID")
    session_id: str = Field(default_factory=lambda: str(uuid.uuid4()), description="会话ID")
    max_results: int = Field(default=5, ge=1, le=20, description="最大结果数")
    filters: dict = Field(default_factory=dict, description="过滤条件")
    
    @validator('query')
    def validate_query(cls, v):
        if not v.strip():
            raise ValueError("查询不能为空")
        return v.strip()
    
    class Config:
        json_schema_extra = {
            "example": {
                "query": "什么是 LangGraph?",
                "user_id": "user_123",
                "max_results": 5
            }
        }

class APIQueryOutput(BaseModel):
    """API 输出验证"""
    answer: str = Field(..., description="生成的答案")
    sources: list[dict] = Field(default_factory=list, description="来源文档")
    confidence: float = Field(..., ge=0.0, le=1.0, description="置信度")
    session_id: str = Field(..., description="会话ID")
    processing_time_ms: float = Field(..., description="处理时间(毫秒)")
    
    class Config:
        json_schema_extra = {
            "example": {
                "answer": "LangGraph 是一个用于构建状态化 AI 应用的框架",
                "sources": [{"doc_id": "123", "score": 0.95}],
                "confidence": 0.92,
                "session_id": "abc-123",
                "processing_time_ms": 245.5
            }
        }

# ============ 内部层：TypedDict 高性能状态 ============

def add_messages(left: list, right: list) -> list:
    """消息累积 Reducer"""
    return left + right

def merge_metadata(left: dict, right: dict) -> dict:
    """元数据合并 Reducer"""
    return {**left, **right}

class InternalState(TypedDict):
    """内部状态：高性能 TypedDict"""
    # 累积字段
    messages: Annotated[list[dict], add_messages]
    retrieved_docs: Annotated[list[dict], add_messages]
    
    # 覆盖字段
    query: str
    user_id: str
    session_id: str
    
    # 元数据
    metadata: Annotated[dict, merge_metadata]
    
    # 可选字段
    answer: NotRequired[str]
    confidence: NotRequired[float]
    error: NotRequired[str]

class ProcessingContext(TypedDict):
    """不可变上下文"""
    start_time: float
    max_results: int
    filters: dict
    model_config: dict

# ============ 转换层：边界与内部转换 ============

class TypeConverter:
    """类型转换器"""
    
    @staticmethod
    def api_input_to_internal(api_input: APIQueryInput) -> InternalState:
        """API 输入 -> 内部状态"""
        return {
            "messages": [{"role": "user", "content": api_input.query}],
            "retrieved_docs": [],
            "query": api_input.query,
            "user_id": api_input.user_id,
            "session_id": api_input.session_id,
            "metadata": {
                "created_at": datetime.now().isoformat(),
                "max_results": api_input.max_results,
                "filters": api_input.filters
            }
        }
    
    @staticmethod
    def internal_to_api_output(
        state: InternalState,
        processing_time_ms: float
    ) -> APIQueryOutput:
        """内部状态 -> API 输出"""
        return APIQueryOutput(
            answer=state.get("answer", ""),
            sources=[
                {"doc_id": doc.get("id"), "score": doc.get("score")}
                for doc in state.get("retrieved_docs", [])
            ],
            confidence=state.get("confidence", 0.0),
            session_id=state["session_id"],
            processing_time_ms=processing_time_ms
        )

### 1.2 图构建

```python
from langgraph.graph import StateGraph, END
import time

# 创建图
graph = StateGraph(
    state_schema=InternalState,
    context_schema=ProcessingContext
)

# 节点函数
def retrieve_node(state: InternalState) -> InternalState:
    """检索节点"""
    query = state["query"]
    max_results = state["metadata"]["max_results"]
    
    # 模拟检索
    docs = [
        {"id": f"doc_{i}", "content": f"内容 {i}", "score": 0.9 - i * 0.1}
        for i in range(max_results)
    ]
    
    return {
        "retrieved_docs": docs,
        "metadata": {"retrieval_count": len(docs)}
    }

def generate_node(state: InternalState) -> InternalState:
    """生成节点"""
    query = state["query"]
    docs = state.get("retrieved_docs", [])
    
    # 模拟生成
    context = "\n".join([doc["content"] for doc in docs[:3]])
    answer = f"基于 {len(docs)} 个文档，回答：{query}"
    
    return {
        "answer": answer,
        "confidence": 0.85,
        "messages": [{"role": "assistant", "content": answer}]
    }

# 添加节点
graph.add_node("retrieve", retrieve_node)
graph.add_node("generate", generate_node)

# 添加边
graph.set_entry_point("retrieve")
graph.add_edge("retrieve", "generate")
graph.add_edge("generate", END)

# 编译
app = graph.compile()

### 1.3 完整流程

```python
def process_query(api_input: APIQueryInput) -> APIQueryOutput:
    """处理查询的完整流程"""
    start_time = time.time()
    
    try:
        # 1. 边界验证（Pydantic 自动验证）
        # api_input 已经通过 Pydantic 验证
        
        # 2. 转换为内部状态
        internal_state = TypeConverter.api_input_to_internal(api_input)
        
        # 3. 创建上下文
        context = ProcessingContext(
            start_time=start_time,
            max_results=api_input.max_results,
            filters=api_input.filters,
            model_config={"temperature": 0.7}
        )
        
        # 4. 执行图（内部使用 TypedDict，高性能）
        result = app.invoke(internal_state, config={"context": context})
        
        # 5. 转换为 API 输出
        processing_time_ms = (time.time() - start_time) * 1000
        api_output = TypeConverter.internal_to_api_output(result, processing_time_ms)
        
        # 6. 输出验证（Pydantic 自动验证）
        return api_output
        
    except Exception as e:
        # 错误处理
        raise ValueError(f"处理失败: {str(e)}")

# 使用示例
if __name__ == "__main__":
    # 创建输入（自动验证）
    query_input = APIQueryInput(
        query="什么是 LangGraph?",
        user_id="user_123",
        max_results=5
    )
    
    # 处理查询
    result = process_query(query_input)
    
    print(f"答案: {result.answer}")
    print(f"来源数: {len(result.sources)}")
    print(f"置信度: {result.confidence}")
    print(f"处理时间: {result.processing_time_ms:.2f}ms")
```

## 示例 2：多租户系统类型架构

### 2.1 类型系统设计

```python
from typing_extensions import TypedDict, Annotated, NotRequired, Required
from pydantic import BaseModel, Field, root_validator
from enum import Enum
import hashlib

# ============ 枚举类型 ============

class TenantTier(str, Enum):
    """租户等级"""
    FREE = "free"
    PRO = "pro"
    ENTERPRISE = "enterprise"

class ResourceType(str, Enum):
    """资源类型"""
    QUERY = "query"
    DOCUMENT = "document"
    STORAGE = "storage"

# ============ 边界层：租户验证 ============

class TenantConfig(BaseModel):
    """租户配置"""
    tenant_id: str = Field(..., pattern=r'^tenant_[a-zA-Z0-9_-]+$')
    tier: TenantTier
    max_queries_per_day: int = Field(..., gt=0)
    max_documents: int = Field(..., gt=0)
    max_storage_mb: int = Field(..., gt=0)
    allowed_models: list[str] = Field(default_factory=list)
    
    @root_validator
    def validate_tier_limits(cls, values):
        tier = values.get('tier')
        max_queries = values.get('max_queries_per_day')
        
        # 根据等级验证限制
        if tier == TenantTier.FREE and max_queries > 100:
            raise ValueError("免费版每天最多 100 次查询")
        elif tier == TenantTier.PRO and max_queries > 10000:
            raise ValueError("专业版每天最多 10000 次查询")
        
        return values

class TenantRequest(BaseModel):
    """租户请求"""
    tenant_id: str
    user_id: str
    request_id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    query: str = Field(..., min_length=1, max_length=2000)
    namespace: str = Field(default="default")
    
    def get_cache_key(self) -> str:
        """生成缓存键"""
        content = f"{self.tenant_id}:{self.query}:{self.namespace}"
        return hashlib.md5(content.encode()).hexdigest()

# ============ 内部层：租户状态 ============

def add_usage(left: dict, right: dict) -> dict:
    """使用量累积"""
    result = left.copy()
    for key, value in right.items():
        result[key] = result.get(key, 0) + value
    return result

class TenantState(TypedDict):
    """租户状态"""
    # 租户信息
    tenant_id: Required[str]
    user_id: Required[str]
    request_id: Required[str]
    namespace: Required[str]
    
    # 查询信息
    query: Required[str]
    cache_key: Required[str]
    
    # 资源使用
    usage: Annotated[dict, add_usage]
    
    # 结果
    answer: NotRequired[str]
    cached: NotRequired[bool]
    error: NotRequired[str]

class TenantContext(TypedDict):
    """租户上下文"""
    config: TenantConfig
    quota_remaining: dict
    rate_limit_reset: float

# ============ 配额管理器 ============

class QuotaManager:
    """配额管理器"""
    
    def __init__(self):
        self.usage_db = {}  # 模拟数据库
    
    def check_quota(self, tenant_id: str, resource: ResourceType, amount: int = 1) -> bool:
        """检查配额"""
        # 模拟配额检查
        current_usage = self.usage_db.get(tenant_id, {}).get(resource.value, 0)
        # 实际应该从配置中获取限制
        return current_usage + amount <= 1000
    
    def consume_quota(self, tenant_id: str, resource: ResourceType, amount: int = 1):
        """消耗配额"""
        if tenant_id not in self.usage_db:
            self.usage_db[tenant_id] = {}
        
        current = self.usage_db[tenant_id].get(resource.value, 0)
        self.usage_db[tenant_id][resource.value] = current + amount
    
    def get_remaining(self, tenant_id: str, resource: ResourceType) -> int:
        """获取剩余配额"""
        current_usage = self.usage_db.get(tenant_id, {}).get(resource.value, 0)
        return max(0, 1000 - current_usage)

# ============ 缓存管理器 ============

class CacheManager:
    """缓存管理器"""
    
    def __init__(self):
        self.cache = {}
    
    def get(self, key: str) -> str | None:
        """获取缓存"""
        return self.cache.get(key)
    
    def set(self, key: str, value: str, ttl: int = 3600):
        """设置缓存"""
        self.cache[key] = value

### 2.2 图构建

```python
# 全局管理器
quota_manager = QuotaManager()
cache_manager = CacheManager()

# 创建图
tenant_graph = StateGraph(
    state_schema=TenantState,
    context_schema=TenantContext
)

def check_quota_node(state: TenantState, context: TenantContext) -> TenantState:
    """检查配额节点"""
    tenant_id = state["tenant_id"]
    
    # 检查查询配额
    if not quota_manager.check_quota(tenant_id, ResourceType.QUERY):
        return {
            "error": "查询配额已用完",
            "usage": {ResourceType.QUERY.value: 0}
        }
    
    # 消耗配额
    quota_manager.consume_quota(tenant_id, ResourceType.QUERY)
    
    return {
        "usage": {ResourceType.QUERY.value: 1}
    }

def check_cache_node(state: TenantState) -> TenantState:
    """检查缓存节点"""
    cache_key = state["cache_key"]
    
    # 检查缓存
    cached_answer = cache_manager.get(cache_key)
    
    if cached_answer:
        return {
            "answer": cached_answer,
            "cached": True
        }
    
    return {"cached": False}

def process_query_node(state: TenantState, context: TenantContext) -> TenantState:
    """处理查询节点"""
    if state.get("cached"):
        return {}  # 已缓存，跳过
    
    query = state["query"]
    namespace = state["namespace"]
    
    # 模拟查询处理
    answer = f"[{namespace}] 处理查询: {query}"
    
    # 缓存结果
    cache_manager.set(state["cache_key"], answer)
    
    return {
        "answer": answer
    }

# 添加节点
tenant_graph.add_node("check_quota", check_quota_node)
tenant_graph.add_node("check_cache", check_cache_node)
tenant_graph.add_node("process_query", process_query_node)

# 添加边
tenant_graph.set_entry_point("check_quota")

def should_continue(state: TenantState) -> str:
    """条件路由"""
    if state.get("error"):
        return END
    if state.get("cached"):
        return END
    return "process_query"

tenant_graph.add_conditional_edges(
    "check_quota",
    lambda s: "check_cache" if not s.get("error") else END
)
tenant_graph.add_conditional_edges("check_cache", should_continue)
tenant_graph.add_edge("process_query", END)

# 编译
tenant_app = tenant_graph.compile()

### 2.3 完整流程

```python
class TenantService:
    """租户服务"""
    
    def __init__(self):
        self.configs = {}  # 租户配置缓存
    
    def load_tenant_config(self, tenant_id: str) -> TenantConfig:
        """加载租户配置"""
        # 模拟从数据库加载
        if tenant_id not in self.configs:
            self.configs[tenant_id] = TenantConfig(
                tenant_id=tenant_id,
                tier=TenantTier.PRO,
                max_queries_per_day=1000,
                max_documents=10000,
                max_storage_mb=1000,
                allowed_models=["gpt-4", "gpt-3.5-turbo"]
            )
        return self.configs[tenant_id]
    
    def process_tenant_request(self, request: TenantRequest) -> dict:
        """处理租户请求"""
        # 1. 加载租户配置
        config = self.load_tenant_config(request.tenant_id)
        
        # 2. 创建内部状态
        state = TenantState(
            tenant_id=request.tenant_id,
            user_id=request.user_id,
            request_id=request.request_id,
            namespace=request.namespace,
            query=request.query,
            cache_key=request.get_cache_key(),
            usage={}
        )
        
        # 3. 创建上下文
        context = TenantContext(
            config=config,
            quota_remaining={
                ResourceType.QUERY.value: quota_manager.get_remaining(
                    request.tenant_id, ResourceType.QUERY
                )
            },
            rate_limit_reset=time.time() + 86400
        )
        
        # 4. 执行图
        result = tenant_app.invoke(state, config={"context": context})
        
        # 5. 返回结果
        return {
            "request_id": result["request_id"],
            "answer": result.get("answer"),
            "cached": result.get("cached", False),
            "error": result.get("error"),
            "usage": result.get("usage", {}),
            "quota_remaining": context["quota_remaining"]
        }

# 使用示例
if __name__ == "__main__":
    service = TenantService()
    
    # 创建请求
    request = TenantRequest(
        tenant_id="tenant_abc123",
        user_id="user_456",
        query="什么是多租户系统?",
        namespace="knowledge_base"
    )
    
    # 处理请求
    result = service.process_tenant_request(request)
    
    print(f"请求ID: {result['request_id']}")
    print(f"答案: {result['answer']}")
    print(f"缓存命中: {result['cached']}")
    print(f"使用量: {result['usage']}")
    print(f"剩余配额: {result['quota_remaining']}")
```

## 最佳实践总结

### 1. 类型系统分层

```
API 边界 (Pydantic)
    ↓ 验证 + 转换
内部状态 (TypedDict)
    ↓ 高性能处理
API 边界 (Pydantic)
    ↓ 验证 + 输出
```

### 2. 验证策略

- **输入验证**：Pydantic Field + validator
- **业务验证**：自定义验证函数
- **输出验证**：Pydantic 自动验证

### 3. 性能优化

- **边界验证**：只在边界验证，内部信任数据
- **类型转换**：避免深拷贝，直接引用
- **缓存策略**：验证结果缓存，配置缓存

### 4. 错误处理

- **验证错误**：Pydantic ValidationError
- **业务错误**：自定义异常
- **系统错误**：统一错误处理
