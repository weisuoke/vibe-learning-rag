# 实战代码 - 场景5：复杂分支合并

> 混合使用条件边和 Send API 实现动态决策与并行执行结合的复杂工作流

---

## 场景概述

本场景演示如何构建一个 **Deep Research Agent**，它能够：
1. 根据用户查询动态决定研究方向
2. 并行执行多个研究任务
3. 智能合并研究结果
4. 支持条件分支和动态路由

这是 LangGraph 中最复杂的并行执行模式，结合了：
- **条件边**：动态决策研究方向
- **Send API**：并行执行研究任务
- **状态合并**：智能聚合研究结果
- **多层路由**：支持嵌套的分支逻辑

## 完整代码实现

```python
"""
LangGraph 复杂分支合并实战：Deep Research Agent

演示场景：
1. 用户提出研究问题
2. 系统分析问题并生成研究计划
3. 根据计划动态创建并行研究任务
4. 每个任务独立执行研究
5. 智能合并所有研究结果
6. 生成最终研究报告

技术要点：
- 条件边：动态决策研究方向
- Send API：并行执行研究任务
- 状态合并：使用 Reducer 聚合结果
- 多层路由：支持复杂的分支逻辑
"""

import os
from typing import List, Literal, TypedDict, Annotated
from typing_extensions import TypedDict
import operator
from dotenv import load_dotenv

from langgraph.graph import StateGraph, START, END
from langgraph.types import Send

# 加载环境变量
load_dotenv()

# ===== 1. 定义状态结构 =====

class ResearchTask(TypedDict):
    """单个研究任务的状态"""
    topic: str  # 研究主题
    query: str  # 具体查询
    result: str  # 研究结果


class OverallState(TypedDict):
    """整体工作流状态"""
    # 用户输入
    user_query: str

    # 研究计划
    research_plan: List[str]  # 研究方向列表

    # 研究任务（使用 operator.add 合并）
    research_tasks: Annotated[List[ResearchTask], operator.add]

    # 最终报告
    final_report: str

    # 控制标志
    needs_deep_research: bool  # 是否需要深度研究


# ===== 2. 节点函数定义 =====

def analyze_query(state: OverallState) -> dict:
    """
    分析用户查询，决定研究策略

    这是第一个决策点：
    - 简单问题：直接回答
    - 复杂问题：生成研究计划
    """
    query = state["user_query"]

    print(f"\n=== 分析查询 ===")
    print(f"用户问题: {query}")

    # 模拟 LLM 分析（实际应用中调用 LLM）
    # 这里简化为关键词检测
    complex_keywords = ["比较", "分析", "深入", "全面", "研究"]
    is_complex = any(keyword in query for keyword in complex_keywords)

    if is_complex:
        print("判断：需要深度研究")
        return {
            "needs_deep_research": True,
            "research_plan": []  # 将在下一步生成
        }
    else:
        print("判断：简单问题，直接回答")
        return {
            "needs_deep_research": False,
            "final_report": f"简单回答：关于 '{query}' 的基本信息..."
        }


def generate_research_plan(state: OverallState) -> dict:
    """
    生成研究计划

    根据用户查询，分解为多个研究方向
    """
    query = state["user_query"]

    print(f"\n=== 生成研究计划 ===")
    print(f"为问题生成研究方向: {query}")

    # 模拟 LLM 生成研究计划（实际应用中调用 LLM）
    # 这里简化为预定义的研究方向
    if "Python" in query and "JavaScript" in query:
        plan = [
            "Python 语言特性和生态系统",
            "JavaScript 语言特性和生态系统",
            "性能对比和基准测试",
            "应用场景和最佳实践"
        ]
    elif "RAG" in query:
        plan = [
            "RAG 核心原理和架构",
            "主流 RAG 框架对比",
            "RAG 性能优化策略",
            "RAG 实际应用案例"
        ]
    else:
        plan = [
            f"{query} - 基础概念",
            f"{query} - 技术实现",
            f"{query} - 最佳实践"
        ]

    print(f"生成 {len(plan)} 个研究方向:")
    for i, direction in enumerate(plan, 1):
        print(f"  {i}. {direction}")

    return {"research_plan": plan}


def conduct_research(state: ResearchTask) -> dict:
    """
    执行单个研究任务

    这是 Map 阶段的核心函数，会被并行调用多次
    """
    topic = state["topic"]
    query = state["query"]

    print(f"\n  [研究任务] 正在研究: {topic}")

    # 模拟研究过程（实际应用中调用搜索 API、LLM 等）
    # 这里简化为生成模拟结果
    result = f"""
    研究主题: {topic}
    查询内容: {query}

    研究发现:
    - 关键点 1: {topic} 的核心特征是...
    - 关键点 2: 在实际应用中，{topic} 主要用于...
    - 关键点 3: 相比其他方案，{topic} 的优势在于...

    参考资料:
    - 官方文档
    - 技术博客
    - 实际案例
    """

    print(f"  [研究任务] 完成: {topic}")

    return {
        "research_tasks": [{
            "topic": topic,
            "query": query,
            "result": result
        }]
    }


def synthesize_report(state: OverallState) -> dict:
    """
    合成最终研究报告

    这是 Reduce 阶段，聚合所有研究结果
    """
    print(f"\n=== 合成研究报告 ===")

    tasks = state["research_tasks"]
    query = state["user_query"]

    print(f"合并 {len(tasks)} 个研究任务的结果")

    # 构建最终报告
    report_sections = [
        f"# 关于 '{query}' 的深度研究报告\n",
        f"## 研究概述\n",
        f"本报告基于 {len(tasks)} 个研究方向的深入分析。\n"
    ]

    for i, task in enumerate(tasks, 1):
        report_sections.append(f"\n## {i}. {task['topic']}\n")
        report_sections.append(task['result'])

    report_sections.append("\n## 总结\n")
    report_sections.append("综合以上研究，我们可以得出以下结论...\n")

    final_report = "\n".join(report_sections)

    print("报告生成完成")

    return {"final_report": final_report}


# ===== 3. 条件路由函数 =====

def should_do_research(state: OverallState) -> Literal["research", "end"]:
    """
    决定是否需要深度研究

    这是第一个条件边：
    - 需要研究 -> 进入研究流程
    - 不需要研究 -> 直接结束
    """
    if state["needs_deep_research"]:
        return "research"
    else:
        return "end"


def continue_to_research_tasks(state: OverallState) -> List[Send]:
    """
    创建并行研究任务

    这是第二个条件边，使用 Send API 实现动态并行：
    - 根据研究计划动态创建任务
    - 每个任务并行执行
    - 自动等待所有任务完成
    """
    plan = state["research_plan"]
    query = state["user_query"]

    print(f"\n=== 创建并行研究任务 ===")
    print(f"将创建 {len(plan)} 个并行任务")

    # 为每个研究方向创建一个 Send 对象
    # 这些任务会并行执行
    return [
        Send(
            "conduct_research",  # 目标节点
            {  # 发送的状态
                "topic": direction,
                "query": f"{query} - {direction}",
                "result": ""
            }
        )
        for direction in plan
    ]


# ===== 4. 构建图 =====

def create_research_graph():
    """构建 Deep Research Agent 图"""

    # 创建状态图
    builder = StateGraph(OverallState)

    # 添加节点
    builder.add_node("analyze_query", analyze_query)
    builder.add_node("generate_research_plan", generate_research_plan)
    builder.add_node("conduct_research", conduct_research)
    builder.add_node("synthesize_report", synthesize_report)

    # 添加边
    # 1. 开始 -> 分析查询
    builder.add_edge(START, "analyze_query")

    # 2. 分析查询 -> 条件分支
    builder.add_conditional_edges(
        "analyze_query",
        should_do_research,
        {
            "research": "generate_research_plan",  # 需要研究
            "end": END  # 不需要研究，直接结束
        }
    )

    # 3. 生成研究计划 -> 并行研究任务
    builder.add_conditional_edges(
        "generate_research_plan",
        continue_to_research_tasks,  # 返回 Send 对象列表
        ["conduct_research"]  # 可能的目标节点
    )

    # 4. 研究任务 -> 合成报告
    builder.add_edge("conduct_research", "synthesize_report")

    # 5. 合成报告 -> 结束
    builder.add_edge("synthesize_report", END)

    # 编译图
    graph = builder.compile()

    return graph


# ===== 5. 测试示例 =====

def test_simple_query():
    """测试简单查询（不需要深度研究）"""
    print("\n" + "="*60)
    print("测试 1: 简单查询")
    print("="*60)

    print("\n=== 最终结果 ===")

# ===== 6. 可视化图结构 =====

def visualize_graph():
    """可视化图结构（需要安装 graphviz）"""
    try:
        from IPython.display import Image, display

        graph = create_research_graph()

        # 生成 Mermaid 图

    except ImportError:
        print("提示：安装 graphviz 和 IPython 可以可视化图结构")


# ===== 7. 主函数 =====

if __name__ == "__main__":
    print("LangGraph 复杂分支合并实战：Deep Research Agent")
    print("="*60)

    # 运行测试
    test_simple_query()
    test_complex_query()
    test_rag_query()

    # 可视化图结构
    visualize_graph()
```

## 核心技术点解析

### 1. 多层条件路由

```python
# 第一层：决定是否需要深度研究
def should_do_research(state: OverallState) -> Literal["research", "end"]:
    if state["needs_deep_research"]:
        return "research"  # 进入研究流程
    else:
        return "end"  # 直接结束

# 第二层：动态创建并行任务
def continue_to_research_tasks(state: OverallState) -> List[Send]:
    return [
        Send("conduct_research", {"topic": direction, ...})
        for direction in state["research_plan"]
    ]
```

**关键点**：
- 第一层使用字符串返回值（简单路由）
- 第二层使用 Send 对象列表（动态并行）
- 两层路由可以灵活组合

### 2. 状态合并策略

```python
class OverallState(TypedDict):
    # 使用 operator.add 自动合并并行任务的结果
    research_tasks: Annotated[List[ResearchTask], operator.add]
```

**工作原理**：
1. 每个并行任务返回一个 `research_tasks` 列表
2. LangGraph 自动使用 `operator.add` 合并所有列表
3. 最终得到包含所有任务结果的完整列表

### 3. Send API 的灵活性

```python
# 可以发送不同的状态到同一个节点
Send("conduct_research", {"topic": "主题1", "query": "查询1"})
Send("conduct_research", {"topic": "主题2", "query": "查询2"})

# 也可以发送到不同的节点
Send("node_a", {...})
Send("node_b", {...})
```

## 性能优化建议

### 1. 控制并行度

```python
def continue_to_research_tasks(state: OverallState) -> List[Send]:
    plan = state["research_plan"]

    # 限制最大并行任务数
    MAX_PARALLEL = 5
    if len(plan) > MAX_PARALLEL:
        plan = plan[:MAX_PARALLEL]

    return [Send("conduct_research", {...}) for direction in plan]
```

### 2. 批处理策略

```python
def continue_to_research_tasks(state: OverallState) -> List[Send]:
    plan = state["research_plan"]

    # 将任务分批
    BATCH_SIZE = 3
    batches = [plan[i:i+BATCH_SIZE] for i in range(0, len(plan), BATCH_SIZE)]

    # 每批作为一个任务
    return [
        Send("conduct_batch_research", {"topics": batch})
        for batch in batches
    ]
```

### 3. 超时控制

```python
import asyncio

async def conduct_research_with_timeout(state: ResearchTask) -> dict:
    """带超时的研究任务"""
    try:
        # 设置5秒超时
```
