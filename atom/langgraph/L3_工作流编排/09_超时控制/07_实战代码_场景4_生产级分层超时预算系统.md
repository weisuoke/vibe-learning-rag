# 实战代码 · 场景 4：生产级分层超时预算系统

> **目标**：把 node timeout、step timeout、fallback、retry-like 设计、runtime context 放在同一个完整示例里
> **适用**：真正准备把 LangGraph agent / workflow 上到生产环境

---

## 场景说明

这个示例展示一个“问题回答工作流”：

1. `retrieve` 节点尝试调用慢速远程检索；
2. 节点内部根据 context 里的 budget 做 timeout；
3. 超时后写入状态，不直接炸整图；
4. 条件路由到 `fallback_answer`；
5. 若没超时则进入 `draft_answer`；
6. 整个图仍有外层 `step_timeout`。

这样可以清楚看到：
- 预算如何注入；
- timeout 如何被业务化；
- 为什么图外层 budget 仍然必要。

## 完整代码

```python
"""
场景4：生产级分层超时预算系统

运行方式：
    python example_budgeted_timeout_system.py

依赖：
    pip install langgraph pydantic
"""

from __future__ import annotations

import asyncio
from typing import Literal

from pydantic import BaseModel
from typing_extensions import TypedDict

from langgraph.graph import END, START, StateGraph
from langgraph.runtime import Runtime


class Budgets(BaseModel):
    retrieve_timeout_s: float = 1.0
    step_timeout_s: float = 2.0
    product_name: str = "LangGraph Tutor"


class State(TypedDict):
    question: str
    retrieved: str
    answer: str
    source: Literal["remote", "fallback", "none"]
    retrieve_timed_out: bool


async def fake_remote_retrieval(question: str, delay_s: float) -> str:
    await asyncio.sleep(delay_s)
    return f"remote_context_for:{question}"


async def retrieve(state: State, runtime: Runtime[Budgets]) -> State:
    budgets = runtime.context
    print(f"[retrieve] timeout={budgets.retrieve_timeout_s}s")

    try:
        # 模拟远程检索需要 1.8 秒，但 context 只给 1 秒
        docs = await asyncio.wait_for(
            fake_remote_retrieval(state["question"], delay_s=1.8),
            timeout=budgets.retrieve_timeout_s,
        )
        return {
            "question": state["question"],
            "retrieved": docs,
            "answer": "",
            "source": "remote",
            "retrieve_timed_out": False,
        }
    except asyncio.TimeoutError:
        print("[retrieve] timeout -> 标记状态并走 fallback")
        return {
            "question": state["question"],
            "retrieved": "",
            "answer": "",
            "source": "none",
            "retrieve_timed_out": True,
        }


def route_after_retrieve(state: State) -> str:
    return "fallback_answer" if state["retrieve_timed_out"] else "draft_answer"


def fallback_answer(state: State, runtime: Runtime[Budgets]) -> State:
    return {
        "question": state["question"],
        "retrieved": state["retrieved"],
        "answer": (
            f"[{runtime.context.product_name}] 远程检索超时，"
            f"先返回保底答案：关于“{state['question']}”，请稍后重试以获得更完整上下文。"
        ),
        "source": "fallback",
        "retrieve_timed_out": state["retrieve_timed_out"],
    }


def draft_answer(state: State, runtime: Runtime[Budgets]) -> State:
    return {
        "question": state["question"],
        "retrieved": state["retrieved"],
        "answer": (
            f"[{runtime.context.product_name}] 根据检索结果生成回答：{state['retrieved']}"
        ),
        "source": "remote",
        "retrieve_timed_out": state["retrieve_timed_out"],
    }


def build_graph():
    builder = StateGraph(state_schema=State, context_schema=Budgets)

    builder.add_node("retrieve", retrieve)
    builder.add_node("fallback_answer", fallback_answer)
    builder.add_node("draft_answer", draft_answer)

    builder.add_edge(START, "retrieve")
    builder.add_conditional_edges(
        "retrieve",
        route_after_retrieve,
        {
            "fallback_answer": "fallback_answer",
            "draft_answer": "draft_answer",
        },
    )
    builder.add_edge("fallback_answer", END)
    builder.add_edge("draft_answer", END)

    graph = builder.compile()
    return graph


async def run_case() -> None:
    graph = build_graph()

    budgets = Budgets(
        retrieve_timeout_s=1.0,
        step_timeout_s=2.0,
        product_name="LangGraph Tutor",
    )

    # 外层 graph budget 与 context 分离：context 用于节点内决策，
    # graph.step_timeout 用于 orchestration 兜底。
    graph.step_timeout = budgets.step_timeout_s

    initial_state: State = {
        "question": "step_timeout 和 http timeout 有什么区别？",
        "retrieved": "",
        "answer": "",
        "source": "none",
        "retrieve_timed_out": False,
    }

    result = await graph.ainvoke(initial_state, context=budgets)

    print("\n=== 最终结果 ===")
    for key, value in result.items():
        print(f"{key}: {value}")


if __name__ == "__main__":
    asyncio.run(run_case())
```

## 这个示例体现了哪些生产思路？

### 1. budget 配置从代码中抽离

使用 `context_schema=Budgets`，让预算成为可注入上下文，而不是 scattered magic numbers。

### 2. node timeout 先于 graph timeout

最贴近慢依赖的层最先决定“不等了”。

### 3. timeout 变成状态，而不是只有异常

`retrieve_timed_out=True` 让后续节点可以做业务级决策。

### 4. fallback 友好于用户

即使远程检索没拿到，用户仍然得到一个有解释的结果，而不是裸异常。

### 5. graph.step_timeout 仍然是兜底

如果整个流程因为别的原因拖住，外层 budget 仍会起作用。

## 你可以继续怎样扩展？

### 1. 加 RetryPolicy
对 `retrieve` 节点做有限重试。

### 2. 加 metrics
统计 timeout 次数、fallback 命中率、平均恢复耗时。

### 3. 加 streaming
把 timeout 前后的状态增量流式发给前端。

### 4. 加 async run + polling
如果整个流程可能分钟级，改成交互层异步化。

## 变体实现：参数化辅助层

```python
from __future__ import annotations

import asyncio
import time
from dataclasses import dataclass
from typing import Any

@dataclass
class TimeoutBudgets:
    node_timeout_s: float = 1.0
    step_timeout_s: float = 2.0
    consumer_timeout_s: float = 0.5

async def production_budget_demo_remote_call(delay_s: float) -> dict[str, Any]:
    await asyncio.sleep(delay_s)
    return {'ok': True, 'delay_s': delay_s}

async def production_budget_demo_guarded_call(delay_s: float, budgets: TimeoutBudgets) -> dict[str, Any]:
    started_at = time.perf_counter()
    try:
        result = await asyncio.wait_for(production_budget_demo_remote_call(delay_s), timeout=budgets.node_timeout_s)
        result["timed_out"] = False
        result["elapsed_s"] = round(time.perf_counter() - started_at, 3)
        return result
    except asyncio.TimeoutError:
        return {
            "ok": False,
            "timed_out": True,
            "delay_s": delay_s,
            "elapsed_s": round(time.perf_counter() - started_at, 3),
        }

async def production_budget_demo_main() -> None:
    budgets = TimeoutBudgets()
    result = await production_budget_demo_guarded_call(delay_s=1.8, budgets=budgets)
    if result["timed_out"]:
        print("[生产级分层超时预算系统] 使用 fallback 或降级路径")
    else:
        print("[生产级分层超时预算系统] 使用主路径结果")
    print(result)

if __name__ == "__main__":
    asyncio.run(production_budget_demo_main())
```
