# 实战代码 - 场景1：基础条件路由

> 本文档演示 LangGraph 条件分支策略的基础用法：根据问题类型路由到不同处理节点，并对比三种路由写法

---

## 场景描述

**场景**：构建一个问答分类系统，用户输入问题后，系统自动判断问题类型（事实型 / 观点型 / 计算型），路由到对应的专属处理节点生成回答。

**特点**：
- 经典的"分类 → 路由 → 处理"三段式流程
- 对比三种路由写法：无类型注解、Literal 类型注解、path_map 映射
- 不依赖外部 API，使用模拟数据，可直接运行
- 展示条件分支的核心 API：`add_conditional_edges`

**流程图**：

```
START
  ↓
classify（分类问题类型）
  ↓
route_question（路由决策）
  ├── "factual"     → handle_factual（事实型回答）→ END
  ├── "opinion"     → handle_opinion（观点型回答）→ END
  └── "calculation" → handle_calculation（计算型回答）→ END
```

## 完整代码

```python
"""
条件分支策略 - 实战场景1：基础条件路由
演示：根据问题类型路由到不同处理节点

三种路由写法对比：
  写法1：无类型注解（不推荐）
  写法2：Literal 类型注解（推荐）
  写法3：path_map 映射（复杂场景推荐）
"""

from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END


# ============================================================
# 第一部分：状态定义
# ============================================================

class QuestionState(TypedDict):
    """问答系统状态

    字段说明：
    - question: 用户输入的原始问题
    - question_type: 分类结果（factual/opinion/calculation）
    - answer: 最终生成的回答
    - log: 处理日志，记录每一步操作
    """
    question: str
    question_type: str
    answer: str
    log: list[str]


# ============================================================
# 第二部分：节点函数
# ============================================================

def classify_question(state: QuestionState) -> dict:
    """分类节点：根据关键词判断问题类型

    这是路由前的"准备节点"——先分类，再路由。
    实际项目中这里通常会调用 LLM 做意图识别，
    这里用关键词匹配模拟，保证代码可直接运行。
    """
    question = state["question"].lower()

    # 关键词匹配规则
    calc_keywords = ["计算", "多少", "加", "减", "乘", "除", "等于", "求"]
    fact_keywords = ["什么是", "谁是", "哪里", "何时", "定义", "解释"]

    if any(w in question for w in calc_keywords):
        q_type = "calculation"
    elif any(w in question for w in fact_keywords):
        q_type = "factual"
    else:
        q_type = "opinion"

    print(f"[分类] 问题: {state['question']}")
    print(f"[分类] 类型: {q_type}")

    return {
        "question_type": q_type,
        "log": state.get("log", []) + [f"分类完成 → {q_type}"],
    }


def handle_factual(state: QuestionState) -> dict:
    """事实型处理节点：返回客观事实"""
    answer = f"[事实型回答] 关于「{state['question']}」的事实：这是一个有明确答案的问题，需要查阅知识库获取准确信息。"
    print(f"[事实型] 生成回答")
    return {
        "answer": answer,
        "log": state.get("log", []) + ["事实型节点处理完成"],
    }


def handle_opinion(state: QuestionState) -> dict:
    """观点型处理节点：返回多角度分析"""
    answer = f"[观点型回答] 关于「{state['question']}」的观点：这是一个开放性问题，不同人可能有不同看法，需要从多个角度分析。"
    print(f"[观点型] 生成回答")
    return {
        "answer": answer,
        "log": state.get("log", []) + ["观点型节点处理完成"],
    }


def handle_calculation(state: QuestionState) -> dict:
    """计算型处理节点：返回计算结果"""
    answer = f"[计算型回答] 关于「{state['question']}」的计算：这是一个需要数学运算的问题，需要解析表达式并计算结果。"
    print(f"[计算型] 生成回答")
    return {
        "answer": answer,
        "log": state.get("log", []) + ["计算型节点处理完成"],
    }


# ============================================================
# 第三部分：三种路由写法对比
# ============================================================

# ------ 写法1：无类型注解（不推荐） ------
def route_question_v1(state: QuestionState):
    """路由函数 - 无类型注解

    问题：LangGraph 无法推断可能的返回值，
    图可视化会显示从 classify 到所有节点的边（全连接），
    失去参考价值。
    """
    return state["question_type"]


# ------ 写法2：Literal 类型注解（推荐） ------
def route_question_v2(state: QuestionState) -> Literal["factual", "opinion", "calculation"]:
    """路由函数 - Literal 类型注解

    优势：
    1. LangGraph 自动从 Literal 推断所有可能路径
    2. 图可视化只显示声明的三条边
    3. 编译时可验证目标节点是否存在
    4. 零额外代码，只需一行类型注解
    """
    return state["question_type"]


# ------ 写法3：path_map 映射（复杂场景推荐） ------
def route_question_v3(state: QuestionState) -> str:
    """路由函数 - 配合 path_map 使用

    返回语义化的键（"fact"/"opinion"/"calc"），
    通过 path_map 映射到实际节点名。
    优势：路由逻辑和节点名完全解耦。
    """
    type_map = {
        "factual": "fact",
        "opinion": "opinion",
        "calculation": "calc",
    }
    return type_map.get(state["question_type"], "opinion")


# ============================================================
# 第四部分：构建图 - 写法2（Literal 类型注解，推荐）
# ============================================================

def build_graph_v2() -> object:
    """使用 Literal 类型注解构建图（推荐写法）"""
    builder = StateGraph(QuestionState)

    # 添加节点
    builder.add_node("classify", classify_question)
    builder.add_node("factual", handle_factual)
    builder.add_node("opinion", handle_opinion)
    builder.add_node("calculation", handle_calculation)

    # 添加边
    builder.add_edge(START, "classify")

    # 条件边：使用 Literal 注解的路由函数
    # LangGraph 自动从 Literal["factual", "opinion", "calculation"]
    # 推断出 path_map = {"factual": "factual", "opinion": "opinion", "calculation": "calculation"}
    builder.add_conditional_edges("classify", route_question_v2)

    # 所有分支汇聚到 END
    builder.add_edge("factual", END)
    builder.add_edge("opinion", END)
    builder.add_edge("calculation", END)

    return builder.compile()


# ============================================================
# 第五部分：构建图 - 写法3（path_map 映射）
# ============================================================

def build_graph_v3() -> object:
    """使用 path_map 映射构建图"""
    builder = StateGraph(QuestionState)

    # 添加节点
    builder.add_node("classify", classify_question)
    builder.add_node("factual", handle_factual)
    builder.add_node("opinion", handle_opinion)
    builder.add_node("calculation", handle_calculation)

    # 添加边
    builder.add_edge(START, "classify")

    # 条件边：使用 path_map 映射
    # 路由函数返回 "fact"/"opinion"/"calc"（语义化键）
    # path_map 将它们映射到实际节点名
    builder.add_conditional_edges(
        "classify",
        route_question_v3,
        {
            "fact": "factual",        # "fact" → factual 节点
            "opinion": "opinion",     # "opinion" → opinion 节点
            "calc": "calculation",    # "calc" → calculation 节点
        },
    )

    builder.add_edge("factual", END)
    builder.add_edge("opinion", END)
    builder.add_edge("calculation", END)

    return builder.compile()


# ============================================================
# 第六部分：测试
# ============================================================

def run_tests():
    """测试三种问题类型"""
    test_questions = [
        "什么是机器学习？",          # → factual
        "你觉得 AI 会取代人类吗？",   # → opinion
        "1+1 等于多少？",            # → calculation
        "解释一下 RAG 的定义",       # → factual
        "Python 和 Java 哪个好？",   # → opinion
        "求 100 除以 5",            # → calculation
    ]

    print("=" * 60)
    print("写法2 测试：Literal 类型注解（推荐）")
    print("=" * 60)

    graph = build_graph_v2()

    for q in test_questions:
        result = graph.invoke({
            "question": q,
            "question_type": "",
            "answer": "",
            "log": [],
        })
        print(f"  回答: {result['answer'][:50]}...")
        print(f"  日志: {result['log']}")
        print()

    print("=" * 60)
    print("写法3 测试：path_map 映射")
    print("=" * 60)

    graph_v3 = build_graph_v3()

    for q in test_questions[:3]:
        result = graph_v3.invoke({
            "question": q,
            "question_type": "",
            "answer": "",
            "log": [],
        })
        print(f"  回答: {result['answer'][:50]}...")
        print()


if __name__ == "__main__":
    run_tests()
```

## 代码详解

### 1. 三种路由写法对比

| 维度 | 写法1：无注解 | 写法2：Literal 注解 | 写法3：path_map |
|------|-------------|-------------------|----------------|
| 代码量 | 最少 | 多一行注解 | 多一个 dict |
| 可视化 | 显示所有可能节点（全连接） | 只显示声明的三条边 | 只显示映射的三条边 |
| 编译检查 | 无法验证目标节点 | 可验证目标存在 | 可验证目标存在 |
| 键值解耦 | 不支持 | 不支持 | 支持 |
| 推荐度 | 不推荐 | 推荐（简单场景） | 推荐（复杂场景） |

### 2. 路由函数的核心约定

```python
# 路由函数的签名
def route_fn(state: StateType) -> Literal["node_a", "node_b", ...]:
    # 1. 只读取状态，不修改状态
    # 2. 返回值必须是目标节点名（或 path_map 的键）
    # 3. 返回值必须与 Literal 声明一致
    return state["some_field"]
```

**关键点**：
- 路由函数是**纯决策函数**，只负责"选路"，不负责"干活"
- 返回值是字符串，对应目标节点名或 path_map 的键
- 使用 `Literal` 注解让 LangGraph 在编译时就知道所有可能路径

### 3. 执行流程详解

```
初始状态:
{
  "question": "什么是机器学习？",
  "question_type": "",
  "answer": "",
  "log": []
}

↓ classify 节点（分类）

{
  "question": "什么是机器学习？",
  "question_type": "factual",      ← 新增
  "answer": "",
  "log": ["分类完成 → factual"]    ← 新增
}

↓ route_question_v2 路由函数
↓ 读取 question_type = "factual" → 返回 "factual"

↓ handle_factual 节点（事实型处理）

{
  "question": "什么是机器学习？",
  "question_type": "factual",
  "answer": "[事实型回答] ...",                          ← 新增
  "log": ["分类完成 → factual", "事实型节点处理完成"]    ← 追加
}

↓ END
```

### 4. 可视化效果对比

```
写法1（无注解）的 Mermaid 图：
┌──────────┐
│ classify │
└────┬─────┘
     │ （到所有节点的边，包括不相关的）
     ├──→ factual     ──→ END
     ├──→ opinion     ──→ END
     ├──→ calculation ──→ END
     └──→ __end__     ← 多余的边！

写法2（Literal 注解）的 Mermaid 图：
┌──────────┐
│ classify │
└────┬─────┘
     │ （只有声明的三条边）
     ├──→ factual     ──→ END

     └──→ calculation ──→ END

写法3（path_map）的 Mermaid 图：
┌──────────┐
│ classify │
└────┬─────┘
     │ （映射后的三条边，标签是语义化的）
     ├─fact────→ factual     ──→ END
     ├─opinion─→ opinion     ──→ END
```

## 最佳实践

### 1. 选择哪种写法？

```
你的路由场景是什么？
│
├── 返回值就是节点名（简单场景）
│   └── 用 Literal 注解（写法2）
│       最简洁，可视化正确
│
├── 返回值和节点名不同（需要解耦）
│   └── 用 path_map（写法3）
│       支持键值解耦
│
└── 任何情况都不要
    └── 无注解 + 无 path_map（写法1）
        可视化会乱
```

### 2. 分类节点 vs 路由函数的职责分离

```python
# ✅ 推荐：分类节点负责"判断"，路由函数负责"选路"
def classify(state):
    """节点：分析并写入 question_type"""
    return {"question_type": "factual"}

def route(state) -> Literal["factual", "opinion"]:
    """路由：读取 question_type，返回目标"""
    return state["question_type"]

# ❌ 不推荐：把分类逻辑塞进路由函数
def route(state) -> Literal["factual", "opinion"]:
    """路由函数不应该包含复杂的分类逻辑"""
    if "什么是" in state["question"]:  # 不要在路由函数里做分类
        return "factual"
    return "opinion"
```

### 3. 处理未知类型的兜底策略

```python
def classify_question(state: QuestionState) -> dict:
    """分类时确保返回已知类型"""
    q_type = classify(state["question"])

    # 兜底：未知类型归入 opinion
    if q_type not in ("factual", "opinion", "calculation"):
        q_type = "opinion"

    return {"question_type": q_type}
```

## 与 RAG 的关联

这个"分类 → 路由"模式在 RAG 系统中非常常见：

| RAG 场景 | 分类依据 | 路由目标 |
|----------|---------|---------|
| Adaptive RAG | 问题复杂度 | 简单检索 / 多步推理 / 网络搜索 |
| Self-RAG | 检索质量评分 | 直接生成 / 重新检索 / 回退 |
| 多知识库 RAG | 问题领域 | 技术文档库 / FAQ 库 / 通用库 |
| 混合检索 | 查询类型 | 向量检索 / 关键词检索 / 混合检索 |

**核心思路一致**：先分析，再路由，最后处理。

---

## 总结

本场景展示了条件分支的最基础用法——"分类 → 路由 → 处理"三段式流程。三种路由写法中，Literal 类型注解是简单场景的首选（零额外代码，可视化精准），path_map 映射适合需要键值解耦的复杂场景。核心原则：路由函数只负责"选路"，不负责"干活"。
