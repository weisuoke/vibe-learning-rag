# 实战代码：场景5 - 动态模板组合

> **目标**：掌握 ChatPromptTemplate 的动态组合技术，使用 + 操作符、append()、extend() 方法构建灵活的对话模板

---

## 场景概述

**适用场景**：
- 根据用户权限动态添加系统消息
- 条件式添加 Few-shot 示例
- 运行时构建复杂对话流程
- 模块化模板管理
- RAG 系统的动态上下文注入

**技术栈**：
- `langchain_core.prompts.ChatPromptTemplate`
- `langchain_openai.ChatOpenAI`
- `langchain_core.output_parsers.StrOutputParser`

**核心概念**：
- **+ 操作符**: 组合两个模板，返回新模板
- **append()**: 在模板末尾添加单条消息（原地修改）
- **extend()**: 在模板末尾添加多条消息（原地修改）

**源码参考**: `chat.py:1006-1044` (+ 操作符), `chat.py:1260-1276` (append/extend)

## 实战1：使用 + 操作符组合模板

### 代码实现

```python
"""
实战1：使用 + 操作符组合模板
演示如何使用 + 操作符组合多个模板片段
"""

from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv

# 加载环境变量
load_dotenv()

def demo_plus_operator():
    """演示 + 操作符组合模板"""
    print("=== 实战1：使用 + 操作符组合模板 ===\n")

    # 1. 创建基础模板片段
    base_template = ChatPromptTemplate.from_messages([
        ("system", "你是一个专业的 AI 助手。")
    ])
    print("1. 创建基础模板")
    print(f"   消息数量: {len(base_template.messages)}")

    # 2. 创建规则模板片段
    rules_template = ChatPromptTemplate.from_messages([
        ("system", "规则:\n1. 回答要简洁\n2. 引用来源\n3. 不确定时说明")
    ])
    print("\n2. 创建规则模板")
    print(f"   消息数量: {len(rules_template.messages)}")

    # 3. 使用 + 操作符组合
    combined_template = base_template + rules_template
    print("\n3. 使用 + 操作符组合")
    print(f"   组合后消息数量: {len(combined_template.messages)}")

    # 4. 添加用户输入
    final_template = combined_template + [("human", "{question}")]
    print("\n4. 添加用户输入")
    print(f"   最终消息数量: {len(final_template.messages)}")
    print(f"   输入变量: {final_template.input_variables}")

    # 5. 测试模板
    print("\n5. 测试模板:")
    messages = final_template.format_messages(question="什么是 Python?")
    for i, msg in enumerate(messages, 1):
        print(f"   [{i}] {msg.__class__.__name__}: {msg.content[:50]}...")

    # 6. 完整链式调用
    print("\n6. 完整链式调用:")
    model = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    chain = final_template | model | StrOutputParser()

    try:
        result = chain.invoke({"question": "什么是 LangChain?"})
        print(f"   回答: {result[:200]}...")
    except Exception as e:
        print(f"   错误: {str(e)}")

if __name__ == "__main__":
    demo_plus_operator()
```

### 代码解析

**核心要点**：
1. **+ 操作符返回新模板**: 不修改原模板，返回新的组合模板
2. **支持多种类型**: 可以组合 ChatPromptTemplate、列表、字符串
3. **保留 partial_variables**: 组合时会合并两个模板的 partial_variables
4. **链式组合**: 可以连续使用 + 操作符

## 实战2：使用 append() 动态添加消息

### 代码实现

```python
"""
实战2：使用 append() 动态添加消息
演示如何使用 append() 方法动态构建模板
"""

from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv

load_dotenv()

def demo_append_method():
    """演示 append() 方法"""
    print("=== 实战2：使用 append() 动态添加消息 ===\n")

    # 1. 创建基础模板
    template = ChatPromptTemplate.from_messages([
        ("system", "你是一个 AI 助手。")
    ])
    print("1. 创建基础模板")
    print(f"   初始消息数量: {len(template.messages)}")

    # 2. 根据条件动态添加消息
    user_level = "premium"  # 模拟用户等级

    if user_level == "premium":
        template.append(("system", "你拥有高级功能访问权限。"))
        print("\n2. 检测到高级用户，添加权限消息")

    print(f"   当前消息数量: {len(template.messages)}")

    # 3. 添加 Few-shot 示例
    add_examples = True  # 模拟是否添加示例

    if add_examples:
        template.append(("human", "你好"))
        template.append(("ai", "你好！我是 AI 助手，有什么可以帮你的？"))
        print("\n3. 添加 Few-shot 示例")

    print(f"   当前消息数量: {len(template.messages)}")

    # 4. 添加用户输入
    template.append(("human", "{question}"))
    print("\n4. 添加用户输入占位符")
    print(f"   最终消息数量: {len(template.messages)}")
    print(f"   输入变量: {template.input_variables}")

    # 5. 测试模板
    print("\n5. 测试模板:")
    messages = template.format_messages(question="介绍一下你自己")
    for i, msg in enumerate(messages, 1):
        print(f"   [{i}] {msg.__class__.__name__}: {msg.content[:40]}...")

    # 6. 实际调用
    print("\n6. 实际调用:")
    model = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    chain = template | model | StrOutputParser()

    try:
        result = chain.invoke({"question": "你有什么特殊功能？"})
        print(f"   回答: {result[:150]}...")
    except Exception as e:
        print(f"   错误: {str(e)}")

if __name__ == "__main__":
    demo_append_method()
```

## 实战3：使用 extend() 批量添加消息

### 代码实现

```python
"""
实战3：使用 extend() 批量添加消息
演示如何使用 extend() 方法批量添加消息
"""

from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv

load_dotenv()

def demo_extend_method():
    """演示 extend() 方法"""
    print("=== 实战3：使用 extend() 批量添加消息 ===\n")

    # 1. 创建基础模板
    template = ChatPromptTemplate.from_messages([
        ("system", "你是一个翻译助手。")
    ])
    print("1. 创建基础模板")
    print(f"   初始消息数量: {len(template.messages)}")

    # 2. 准备 Few-shot 示例
    fewshot_examples = [
        ("human", "Hello"),
        ("ai", "你好"),
        ("human", "Thank you"),
        ("ai", "谢谢"),
        ("human", "Goodbye"),
        ("ai", "再见")
    ]

    # 3. 使用 extend() 批量添加
    template.extend(fewshot_examples)
    print("\n2. 使用 extend() 添加 Few-shot 示例")
    print(f"   添加了 {len(fewshot_examples)} 条消息")
    print(f"   当前消息数量: {len(template.messages)}")

    # 4. 添加用户输入
    template.append(("human", "{input}"))
    print("\n3. 添加用户输入")
    print(f"   最终消息数量: {len(template.messages)}")

    # 5. 测试模板
    print("\n4. 测试模板:")
    messages = template.format_messages(input="Good morning")
    print(f"   生成了 {len(messages)} 条消息")
    print(f"   最后一条: {messages[-1].content}")

    # 6. 实际调用
    print("\n5. 实际调用:")
    model = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    chain = template | model | StrOutputParser()

    test_cases = ["Good night", "How are you?", "See you later"]

    for test_input in test_cases:
        try:
            result = chain.invoke({"input": test_input})
            print(f"   输入: {test_input:20s} -> 输出: {result}")
        except Exception as e:
            print(f"   错误: {str(e)}")

if __name__ == "__main__":
    demo_extend_method()
```

## 实战4：动态构建 RAG 对话模板

### 代码实现

```python
"""
实战4：动态构建 RAG 对话模板
演示如何根据不同场景动态构建 RAG 模板
"""

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_openai import ChatOpenAI
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv
from typing import Dict, List

load_dotenv()

class DynamicRAGTemplateBuilder:
    """动态 RAG 模板构建器"""

    def __init__(self):
        self.template = ChatPromptTemplate.from_messages([
            ("system", "你是一个基于知识库的问答助手。")
        ])

    def add_rules(self, strict_mode: bool = False):
        """添加规则"""
        if strict_mode:
            rules = """严格模式规则:
1. 仅基于提供的上下文回答
2. 不确定时必须说"无法回答"
3. 必须引用具体的文档片段
4. 不要推测或编造信息"""
        else:
            rules = """标准模式规则:
1. 优先使用提供的上下文
2. 可以结合常识回答
3. 尽量引用来源"""

        self.template.append(("system", rules))
        return self

    def add_context(self):
        """添加上下文占位符"""
        self.template.append(("system", "参考文档:\n{context}"))
        return self

    def add_history(self):
        """添加对话历史"""
        self.template.append(MessagesPlaceholder("chat_history", optional=True))
        return self

    def add_examples(self, num_examples: int = 0):
        """添加 Few-shot 示例"""
        if num_examples >= 1:
            examples = [
                ("system", "示例上下文: Python 是一种编程语言。"),
                ("human", "什么是 Python?"),
                ("ai", "根据文档，Python 是一种编程语言。[来源: 示例上下文]")
            ]
            self.template.extend(examples[:num_examples * 3])
        return self

    def add_query(self):
        """添加用户查询"""
        self.template.append(("human", "{question}"))
        return self

    def build(self) -> ChatPromptTemplate:
        """构建最终模板"""
        return self.template

def demo_dynamic_rag_builder():
    """演示动态 RAG 模板构建"""
    print("=== 实战4：动态构建 RAG 对话模板 ===\n")

    # 场景1: 简单 RAG
    print("场景1: 简单 RAG（无历史，无示例）")
    simple_template = (
        DynamicRAGTemplateBuilder()
        .add_rules(strict_mode=False)
        .add_context()
        .add_query()
        .build()
    )
    print(f"   消息数量: {len(simple_template.messages)}")
    print(f"   输入变量: {simple_template.input_variables}")

    # 场景2: 严格模式 RAG
    print("\n场景2: 严格模式 RAG（严格规则 + 示例）")
    strict_template = (
        DynamicRAGTemplateBuilder()
        .add_rules(strict_mode=True)
        .add_context()
        .add_examples(num_examples=1)
        .add_query()
        .build()
    )
    print(f"   消息数量: {len(strict_template.messages)}")
    print(f"   输入变量: {strict_template.input_variables}")

    # 场景3: 对话式 RAG
    print("\n场景3: 对话式 RAG（带历史）")
    conversational_template = (
        DynamicRAGTemplateBuilder()
        .add_rules(strict_mode=False)
        .add_context()
        .add_history()
        .add_query()
        .build()
    )
    print(f"   消息数量: {len(conversational_template.messages)}")
    print(f"   输入变量: {conversational_template.input_variables}")

    # 测试场景1
    print("\n测试场景1（简单 RAG）:")
    model = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    chain = simple_template | model | StrOutputParser()

    try:
        result = chain.invoke({
            "context": "LangChain 是一个用于构建 LLM 应用的框架，由 Harrison Chase 创建。",
            "question": "谁创建了 LangChain?"
        })
        print(f"   回答: {result[:150]}...")
    except Exception as e:
        print(f"   错误: {str(e)}")

if __name__ == "__main__":
    demo_dynamic_rag_builder()
```

## 最佳实践总结

### 1. 选择合适的组合方法

```python
# 场景1: 需要保留原模板 → 使用 + 操作符
base = ChatPromptTemplate.from_messages([("system", "基础")])
new_template = base + [("human", "{input}")]  # base 不变

# 场景2: 原地修改模板 → 使用 append/extend
template = ChatPromptTemplate.from_messages([("system", "基础")])
template.append(("human", "{input}"))  # template 被修改

# 场景3: 批量添加 → 使用 extend
template.extend([
    ("human", "示例1"),
    ("ai", "回答1"),
    ("human", "示例2"),
    ("ai", "回答2")
])
```

### 2. + 操作符支持的类型

```python
template = ChatPromptTemplate.from_messages([("system", "基础")])

# 类型1: ChatPromptTemplate
template2 = ChatPromptTemplate.from_messages([("human", "问题")])
combined = template + template2

# 类型2: 列表
combined = template + [("human", "{input}")]

# 类型3: 字符串（自动转为 HumanMessage）
combined = template + "{input}"

# 类型4: 单个消息模板
from langchain_core.prompts import HumanMessagePromptTemplate
human_template = HumanMessagePromptTemplate.from_template("{input}")
combined = template + human_template
```

### 3. 动态模板构建模式

```python
def build_template(
    include_rules: bool = True,
    include_examples: bool = False,
    include_history: bool = False
) -> ChatPromptTemplate:
    """条件式构建模板"""
    template = ChatPromptTemplate.from_messages([
        ("system", "你是助手。")
    ])

    if include_rules:
        template.append(("system", "规则: ..."))

    if include_examples:
        template.extend([
            ("human", "示例问题"),
            ("ai", "示例回答")
        ])

    if include_history:
        template.append(MessagesPlaceholder("history"))

    template.append(("human", "{input}"))
    return template
```

### 4. 性能考虑

```python
# ✅ 好的做法 - 批量添加
template.extend([msg1, msg2, msg3, msg4, msg5])

# ❌ 不好的做法 - 多次单独添加
template.append(msg1)
template.append(msg2)
template.append(msg3)
template.append(msg4)
template.append(msg5)
```

### 5. 模块化模板管理

```python
# 定义可复用的模板片段
SYSTEM_BASE = ChatPromptTemplate.from_messages([
    ("system", "你是 AI 助手。")
])

STRICT_RULES = ChatPromptTemplate.from_messages([
    ("system", "严格规则: ...")
])

FEWSHOT_EXAMPLES = ChatPromptTemplate.from_messages([
    ("human", "示例1"),
    ("ai", "回答1")
])

# 按需组合
template = SYSTEM_BASE + STRICT_RULES + FEWSHOT_EXAMPLES + [("human", "{input}")]
```

## 常见问题

### Q1: + 操作符会修改原模板吗？

**答案**: 不会。+ 操作符返回新模板，不修改原模板。

```python
template1 = ChatPromptTemplate.from_messages([("system", "A")])
template2 = template1 + [("human", "B")]

print(len(template1.messages))  # 1（未改变）
print(len(template2.messages))  # 2（新模板）
```

### Q2: append 和 extend 的区别？

**答案**:
- `append(message)`: 添加单条消息
- `extend(messages)`: 添加多条消息（列表）

```python
template.append(("human", "单条"))
template.extend([("human", "多条1"), ("ai", "多条2")])
```

### Q3: 如何在组合时保留 partial_variables？

**答案**: + 操作符会自动合并两个模板的 partial_variables。

```python
template1 = ChatPromptTemplate.from_messages([...]).partial(var1="value1")
template2 = ChatPromptTemplate.from_messages([...]).partial(var2="value2")
combined = template1 + template2  # 包含 var1 和 var2
```
