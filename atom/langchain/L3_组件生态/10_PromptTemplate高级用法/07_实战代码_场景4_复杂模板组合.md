## 源码原理深入解析

### partial_variables 合并机制

**来源：** [sourcecode/langchain/libs/core/langchain_core/prompts/prompt.py:142-184]

```python
# 5. 合并 partial_variables（检查冲突）
partial_variables = dict(self.partial_variables.items())
for k, v in other.partial_variables.items():
    if k in partial_variables:
        msg = "Cannot have same variable partialed twice."
        raise ValueError(msg)
    partial_variables[k] = v
```

**关键特性：**
1. **冲突检查**：如果两个模板有相同的 partial_variables 键，会抛出 ValueError
2. **合并策略**：先复制第一个模板的 partial_variables，再逐个添加第二个模板的
3. **不可覆盖**：不允许后面的模板覆盖前面的 partial_variables

**为什么这样设计？**
- 避免意外覆盖预设值
- 确保模板组合的可预测性
- 强制开发者显式处理冲突

## 完整代码示例

```python
"""
场景4：复杂模板组合实战
演示：多层次组合、partial_variables 合并、错误处理
"""

import os
from datetime import datetime
from dotenv import load_dotenv
from langchain_core.prompts import PromptTemplate
from langchain_openai import ChatOpenAI

# 加载环境变量
load_dotenv()

print("=" * 80)
print("场景4：复杂模板组合实战")
print("=" * 80)

# ===== 1. 多层次模板组合 =====
print("\n【示例1】多层次模板组合")
print("-" * 80)

# 第一层：角色定义
role_template = PromptTemplate.from_template(
    "You are a {role}.\n"
)

# 第二层：能力说明
capability_template = PromptTemplate.from_template(
    "Your capabilities include: {capabilities}\n"
)

# 第三层：任务描述
task_template = PromptTemplate.from_template(
    "Current task: {task}\n"
)

# 第四层：输出要求
output_template = PromptTemplate.from_template(
    "Output format: {format}\n"
)

# 多层次组合
level1 = role_template + capability_template
level2 = level1 + task_template
final_prompt = level2 + output_template

print("✓ 多层次组合后的模板：")
print(final_prompt.template)
print(f"\n✓ 合并的变量：{final_prompt.input_variables}")

# 格式化输出
result = final_prompt.format(
    role="Python Expert",
    capabilities="code review, optimization, debugging",
    task="Review the following code for best practices",
    format="Markdown with code examples"
)
print(f"\n✓ 格式化结果：\n{result}")

# ===== 2. partial_variables 基础合并 =====
print("\n【示例2】partial_variables 基础合并")
print("-" * 80)

# 创建带 partial_variables 的模板
template1 = PromptTemplate(
    template="System: {system_name}\nDate: {date}\n",
    input_variables=["system_name"],
    partial_variables={"date": "2026-02-26"}
)

template2 = PromptTemplate(
    template="User: {user_name}\nTask: {task}\n",
    input_variables=["user_name", "task"],
    partial_variables={"version": "1.0"}
)

# 组合（partial_variables 会合并）
combined = template1 + template2

print("✓ 组合后的模板：")
print(combined.template)
print(f"\n✓ input_variables: {combined.input_variables}")
print(f"✓ partial_variables: {combined.partial_variables}")

# 格式化（只需提供 input_variables）
result = combined.format(
    system_name="RAG System",
    user_name="Alice",
    task="Search documents"
)
print(f"\n✓ 格式化结果：\n{result}")

# ===== 3. partial_variables 冲突检测 =====
print("\n【示例3】partial_variables 冲突检测")
print("-" * 80)

try:
    # 创建两个模板，都有相同的 partial_variables 键
    template_a = PromptTemplate(
        template="Part A: {content}\n",
        input_variables=["content"],
        partial_variables={"timestamp": "2026-02-26 10:00"}
    )

    template_b = PromptTemplate(
        template="Part B: {data}\n",
        input_variables=["data"],
        partial_variables={"timestamp": "2026-02-26 11:00"}  # 冲突！
    )

    # 尝试组合（会失败）
    combined = template_a + template_b

except ValueError as e:
    print(f"✗ 检测到冲突：{e}")
    print("\n✓ 解决方案1：使用不同的键名")

    template_b_fixed = PromptTemplate(
        template="Part B: {data}\n",
        input_variables=["data"],
        partial_variables={"timestamp_b": "2026-02-26 11:00"}
    )

    combined = template_a + template_b_fixed
    print(f"✓ 修复后的 partial_variables: {combined.partial_variables}")

# ===== 4. 动态 partial_variables 合并 =====
print("\n【示例4】动态 partial_variables 合并")
print("-" * 80)

# 定义动态函数
def get_current_time():
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

def get_system_version():
    return "v2.0.1"

# 创建带函数的 partial_variables
template1 = PromptTemplate(
    template="System Version: {version}\n",
    input_variables=[],
    partial_variables={"version": get_system_version}
)

template2 = PromptTemplate(
    template="Current Time: {time}\nQuery: {query}\n",
    input_variables=["query"],
    partial_variables={"time": get_current_time}
)

# 组合（函数会被保留）
combined = template1 + template2

print("✓ 组合后的 partial_variables（函数）：")
print(f"  - version: {type(combined.partial_variables['version'])}")
print(f"  - time: {type(combined.partial_variables['time'])}")

# 格式化（函数会被调用）
result = combined.format(query="What is RAG?")
print(f"\n✓ 格式化结果（函数已执行）：\n{result}")

# ===== 5. 错误处理：格式不一致 =====
print("\n【示例5】错误处理：格式不一致")
print("-" * 80)

def safe_combine_templates(template1, template2):
    """安全组合模板，带错误处理"""
    try:
        return template1 + template2
    except ValueError as e:
        error_msg = str(e)
        if "different formats" in error_msg:
            print(f"✗ 格式不一致错误：{e}")
            print("✓ 降级策略：转换为统一格式")
            # 降级：将第二个模板转换为第一个模板的格式
            template2_converted = PromptTemplate.from_template(
                template2.template,
                template_format=template1.template_format
            )
            return template1 + template2_converted
        elif "partialed twice" in error_msg:
            print(f"✗ partial_variables 冲突：{e}")
            print("✓ 降级策略：移除冲突的 partial_variables")
            # 降级：创建不带冲突 partial_variables 的新模板
            return None
        else:
            raise

# 测试格式不一致
fstring_template = PromptTemplate.from_template(
    "Hello {name}",
    template_format="f-string"
)

mustache_template = PromptTemplate.from_template(
    "Age: {{age}}",
    template_format="mustache"
)

result = safe_combine_templates(fstring_template, mustache_template)
if result:
    print(f"✓ 组合成功：{result.template}")

# ===== 6. RAG 系统复杂提示词构建 =====
print("\n【示例6】RAG 系统复杂提示词构建")
print("-" * 80)

# 系统角色（带 partial_variables）
system_role = PromptTemplate(
    template="You are a RAG assistant for {domain}.\n",
    input_variables=["domain"],
    partial_variables={
        "system_version": "RAG v2.0",
        "model": "gpt-4o-mini"
    }
)

# 检索上下文
context_section = PromptTemplate.from_template(
    "Retrieved Context:\n{context}\n\n"
)

# 用户问题
question_section = PromptTemplate.from_template(
    "User Question: {question}\n\n"
)

# 输出指令（带 partial_variables）
instruction_section = PromptTemplate(
    template="Instructions:\n{instructions}\n",
    input_variables=[],
    partial_variables={
        "instructions": (
            "1. Answer based only on the provided context\n"
            "2. If context is insufficient, say 'I don't have enough information'\n"
            "3. Cite sources when possible"
        )
    }
)

# 多层次组合
rag_prompt = system_role + context_section + question_section + instruction_section

print("✓ RAG 提示词模板：")
print(rag_prompt.template)
print(f"\n✓ input_variables: {rag_prompt.input_variables}")
print(f"✓ partial_variables keys: {list(rag_prompt.partial_variables.keys())}")

# 实际使用
result = rag_prompt.format(
    domain="Python programming",
    context="Python decorators are functions that modify other functions...",
    question="What is a Python decorator?"
)
print(f"\n✓ 格式化结果：\n{result}")

# ===== 7. 与 LLM 集成的完整 RAG 流程 =====
print("\n【示例7】与 LLM 集成的完整 RAG 流程")
print("-" * 80)

# 构建完整 RAG 链
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
rag_chain = rag_prompt | llm

print("✓ 调用 RAG 链...")
response = rag_chain.invoke({
    "domain": "Machine Learning",
    "context": (
        "Overfitting occurs when a model learns the training data too well, "
        "including noise and outliers, resulting in poor generalization to new data."
    ),
    "question": "What is overfitting in machine learning?"
})

print(f"\n✓ RAG 响应：\n{response.content}")

# ===== 8. 生产级错误处理封装 =====
print("\n【示例8】生产级错误处理封装")
print("-" * 80)

class PromptComposer:
    """生产级提示词组合器"""

    @staticmethod
    def compose(*templates, strict=True):
        """
        组合多个模板，带完整错误处理

        Args:
            *templates: 要组合的模板列表
            strict: 是否严格模式（遇到错误抛出异常）

        Returns:
            组合后的模板，或 None（非严格模式下）
        """
        if not templates:
            raise ValueError("At least one template is required")

        result = templates[0]

        for i, template in enumerate(templates[1:], 1):
            try:
                result = result + template
            except ValueError as e:
                error_msg = str(e)

                if "different formats" in error_msg:
                    if strict:
                        raise ValueError(
                            f"Template {i} has incompatible format. "
                            f"Expected: {result.template_format}, "
                            f"Got: {template.template_format}"
                        )
                    else:
                        print(f"⚠ Warning: Skipping template {i} due to format mismatch")
                        continue

                elif "partialed twice" in error_msg:
                    if strict:
                        raise ValueError(
                            f"Template {i} has conflicting partial_variables"
                        )
                    else:
                        print(f"⚠ Warning: Skipping template {i} due to partial_variables conflict")
                        continue

                else:
                    raise

        return result

# 测试生产级组合器
templates = [
    PromptTemplate.from_template("Part 1: {a}\n"),
    PromptTemplate.from_template("Part 2: {b}\n"),
    PromptTemplate.from_template("Part 3: {c}\n"),
]

composed = PromptComposer.compose(*templates)
print("✓ 生产级组合成功")
print(f"✓ 变量列表: {composed.input_variables}")

# 测试错误处理（非严格模式）
templates_with_error = [
    PromptTemplate.from_template("Part 1: {a}\n", template_format="f-string"),
    PromptTemplate.from_template("Part 2: {{b}}\n", template_format="mustache"),
    PromptTemplate.from_template("Part 3: {c}\n", template_format="f-string"),
]

composed = PromptComposer.compose(*templates_with_error, strict=False)
if composed:
    print(f"\n✓ 非严格模式组合结果: {composed.input_variables}")

print("\n" + "=" * 80)
print("✓ 复杂模板组合实战完成！")
print("=" * 80)
```

## 核心要点总结

### 1. partial_variables 合并规则

| 场景 | 行为 | 说明 |
|------|------|------|
| 无冲突 | 自动合并 | 两个模板的 partial_variables 键不重复 |
| 有冲突 | 抛出 ValueError | 相同的键会导致错误 |
| 函数值 | 保留函数 | 函数在格式化时才执行 |

### 2. 错误处理策略

**严格模式（推荐生产环境）：**
- 遇到错误立即抛出异常
- 确保模板组合的正确性
- 便于调试和问题定位

**宽松模式（开发调试）：**
- 跳过有问题的模板
- 记录警告信息
- 继续组合剩余模板

### 3. 最佳实践

**设计原则：**
- ✅ 使用唯一的 partial_variables 键名
- ✅ 在组合前验证模板格式一致性
- ✅ 为复杂组合提供降级策略
- ✅ 记录组合过程的关键信息

**性能优化：**
- ✅ 缓存常用的组合结果
- ✅ 避免过深的嵌套层次（建议 ≤ 5 层）
- ✅ 使用函数 partial_variables 实现延迟计算
