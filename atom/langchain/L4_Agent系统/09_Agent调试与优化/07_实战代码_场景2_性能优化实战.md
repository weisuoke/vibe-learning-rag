## 场景描述

你的 Agent 在生产环境中运行良好，但面临三个问题：
1. **太贵**：每天 API 费用超预算 50%
2. **太慢**：复杂问题响应时间超过 30 秒
3. **浪费**：很多重复查询在反复调用 API

你需要在不降低质量的前提下优化成本和性能。

```python
"""
Agent 性能优化实战
演示：缓存 + Token 监控 + 模型路由 + 批处理

包含：
1. LLM 缓存系统搭建
2. Token 使用量追踪
3. 智能模型路由
4. 批量处理优化
"""

import os
import time
from typing import Any

from dotenv import load_dotenv
load_dotenv()

from langchain_openai import ChatOpenAI
from langchain_core.callbacks import BaseCallbackHandler
from langchain_core.messages import HumanMessage, SystemMessage

# ===== 1. Token 使用量追踪 =====
print("=" * 60)
print("1. Token 使用量追踪回调")
print("=" * 60)


class CostTrackerHandler(BaseCallbackHandler):
    """
    追踪 LLM 调用的 Token 使用量和成本

    自动计算每次调用和累计的：
    - 输入/输出 Token 数量
    - 按模型定价计算费用
    """

    # 模型定价（每百万 Token，2026 年价格）
    PRICING = {
        "gpt-4o": {"input": 2.5, "output": 10.0},
        "gpt-4o-mini": {"input": 0.15, "output": 0.6},
        "gpt-4": {"input": 30.0, "output": 60.0},
    }

    def __init__(self):
        self.calls = []
        self.total_input_tokens = 0
        self.total_output_tokens = 0
        self.total_cost = 0.0

    def on_llm_end(self, response, **kwargs):
        """LLM 调用结束时记录 Token 使用"""
        if not hasattr(response, "llm_output") or not response.llm_output:
            return

        usage = response.llm_output.get("token_usage", {})
        model = response.llm_output.get("model_name", "unknown")
        input_tokens = usage.get("prompt_tokens", 0)
        output_tokens = usage.get("completion_tokens", 0)

        # 计算成本
        pricing = self.PRICING.get(model, {"input": 5.0, "output": 15.0})
        cost = (
            input_tokens * pricing["input"] / 1_000_000
            + output_tokens * pricing["output"] / 1_000_000
        )

        self.total_input_tokens += input_tokens
        self.total_output_tokens += output_tokens
        self.total_cost += cost

        self.calls.append({
            "model": model,
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
            "cost": cost,
        })

        print(f"  📊 [{model}] 输入: {input_tokens}, "
              f"输出: {output_tokens}, 费用: ${cost:.6f}")

    def get_report(self) -> dict:
        """生成成本报告"""
        return {
            "total_calls": len(self.calls),
            "total_input_tokens": self.total_input_tokens,
            "total_output_tokens": self.total_output_tokens,
            "total_cost": round(self.total_cost, 6),
            "avg_cost_per_call": round(
                self.total_cost / len(self.calls), 6
            ) if self.calls else 0,
        }

    def print_report(self):
        report = self.get_report()
        print(f"\n  💰 成本报告:")
        print(f"     总调用: {report['total_calls']} 次")
        print(f"     总 Token: {report['total_input_tokens'] + report['total_output_tokens']}")
        print(f"     总费用: ${report['total_cost']}")
        print(f"     平均每次: ${report['avg_cost_per_call']}")


# ===== 2. LLM 缓存系统 =====
print("\n" + "=" * 60)
print("2. LLM 缓存系统")
print("=" * 60)

from langchain_core.globals import set_llm_cache
from langchain_core.caches import InMemoryCache

# 启用内存缓存
set_llm_cache(InMemoryCache())
print("  ✅ InMemoryCache 已启用")

# 演示缓存效果
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
cost_tracker = CostTrackerHandler()

print("\n  --- 缓存效果演示 ---")

# 第一次调用（缓存未命中）
start = time.time()
result1 = llm.invoke(
    "用一句话解释什么是 RAG",
    config={"callbacks": [cost_tracker]}
)
time1 = time.time() - start
print(f"  第1次调用: {time1:.3f}s (缓存未命中)")

# 第二次调用（缓存命中）
start = time.time()
result2 = llm.invoke(
    "用一句话解释什么是 RAG",
    config={"callbacks": [cost_tracker]}
)
time2 = time.time() - start
print(f"  第2次调用: {time2:.3f}s (缓存命中)")
print(f"  ⚡ 加速比: {time1/max(time2, 0.001):.0f}x")


# ===== 3. 智能模型路由 =====
print("\n" + "=" * 60)
print("3. 智能模型路由")
print("=" * 60)

# 清除之前的缓存以获得准确的成本数据
set_llm_cache(None)

# 两个模型
cheap_llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
strong_llm = ChatOpenAI(model="gpt-4o", temperature=0)


def classify_complexity(query: str) -> str:
    """
    简单的任务复杂度分类器

    规则：
    - 包含分析/对比/推理等关键词 → 复杂
    - 问题长度 > 100 字 → 复杂
    - 其他 → 简单
    """
    complex_keywords = [
        "分析", "对比", "推理", "评估", "设计",
        "为什么", "如何实现", "有什么区别", "优缺点",
        "architecture", "design", "compare", "analyze"
    ]

    if any(kw in query for kw in complex_keywords):
        return "complex"
    if len(query) > 100:
        return "complex"
    return "simple"


def route_to_model(query: str) -> ChatOpenAI:
    """根据复杂度路由到合适的模型"""
    complexity = classify_complexity(query)
    if complexity == "complex":
        print(f"  🧠 复杂任务 → GPT-4o (${2.5}/M input)")
        return strong_llm
    else:
        print(f"  ⚡ 简单任务 → GPT-4o-mini (${0.15}/M input)")
        return cheap_llm


# 演示模型路由
queries = [
    "今天星期几？",                              # 简单
    "请分析 RAG 和 Fine-tuning 的优缺点对比",    # 复杂
    "Python 的 list 怎么排序？",                  # 简单
    "请设计一个多步推理 Agent 的架构方案",          # 复杂
]

route_tracker = CostTrackerHandler()

for query in queries:
    print(f"\n  问题: {query}")
    model = route_to_model(query)
    result = model.invoke(
        query,
        config={"callbacks": [route_tracker]}
    )
    print(f"  回答: {result.content[:60]}...")

route_tracker.print_report()


# ===== 4. 批量处理优化 =====
print("\n" + "=" * 60)
print("4. 批量处理优化")
print("=" * 60)

batch_tracker = CostTrackerHandler()

# 准备批量输入
batch_queries = [
    "什么是向量数据库？",
    "什么是 Embedding？",
    "什么是 Prompt Engineering？",
    "什么是 Fine-tuning？",
    "什么是 Token？",
]

# 方法1：逐个调用（基线）
print("\n  --- 方法1: 逐个调用 ---")
start = time.time()
results_sequential = []
for q in batch_queries:
    result = cheap_llm.invoke(q, config={"callbacks": [batch_tracker]})
    results_sequential.append(result.content)
time_sequential = time.time() - start
print(f"  逐个调用总耗时: {time_sequential:.2f}s")

# 方法2：批量调用（利用并发）
print("\n  --- 方法2: 批量调用 (batch) ---")
batch_tracker2 = CostTrackerHandler()
start = time.time()
results_batch = cheap_llm.batch(
    batch_queries,
    config={"callbacks": [batch_tracker2], "max_concurrency": 3}
)
time_batch = time.time() - start
print(f"  批量调用总耗时: {time_batch:.2f}s")
print(f"  ⚡ 加速比: {time_sequential/max(time_batch, 0.001):.1f}x")


# ===== 5. 综合优化效果对比 =====
print("\n" + "=" * 60)
print("5. 综合优化效果对比")
print("=" * 60)

print("""
  ┌─────────────────────────────────────────────────────┐
  │              优化策略效果总结                          │
  ├──────────────────┬──────────┬───────────────────────┤
  │ 优化策略          │ 节省比例  │ 适用场景              │
  ├──────────────────┼──────────┼───────────────────────┤
  │ LLM 缓存          │ ~100%    │ 开发测试/重复查询     │
  │ 模型路由           │ ~60-80%  │ 混合复杂度任务       │
  │ 批量处理           │ ~50-70%  │ 多个独立任务         │
  │ Prompt 精简        │ ~20-40%  │ 所有场景             │
  │ max_iterations    │ 防超限    │ 所有 Agent           │
  └──────────────────┴──────────┴───────────────────────┘

  组合使用效果：
  - 缓存 + 模型路由 = 日常开发节省 80%+
  - 模型路由 + 批处理 = 批量任务节省 70%+
  - 全部组合 = 综合节省 60-90%
""")


# ===== 6. 最佳实践速查 =====
print("=" * 60)
print("6. 成本优化决策树")
print("=" * 60)

print("""
  需要优化成本？
  │
  ├── 开发/测试阶段？
  │   └── ✅ 开启 InMemoryCache（零成本重复调用）
  │

      ├── ✅ 考虑自定义更小的模型
      └── ✅ 评估是否需要 Agent（也许 Chain 就够）
""")

print("\n✅ 性能优化实战完成！")
```

## 关键技巧总结

| 优化手段 | 实现方式 | 节省比例 |
|---------|---------|---------|
| LLM 缓存 | `set_llm_cache(InMemoryCache())` | 重复查询 100% |
| 模型路由 | 关键词分类 + 不同模型 | 60-80% |
| 批量处理 | `llm.batch(queries, max_concurrency=3)` | 时间 50-70% |
| Token 追踪 | `CostTrackerHandler` 回调 | 发现浪费 |
