## 场景 1：loopback 绑定（本地开发）

### 原理

loopback 模式只监听 127.0.0.1，只允许本机访问，是最安全的绑定模式，适合本地开发和单机部署。

### 实现

```typescript
// scripts/config-loopback.ts
/**
 * loopback 绑定配置脚本
 * 用途：本地开发环境，最高安全性
 */

import { spawn } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

interface LoopbackConfig {
  port?: number;
  auth?: {
    mode: 'token' | 'password' | 'none';
    token?: string;
    password?: string;
  };
}

/**
 * 配置 loopback 绑定
 */
async function configureLoopback(config: LoopbackConfig = {}): Promise<void> {
  const {
    port = 18789,
    auth = { mode: 'token' }
  } = config;

  console.log('🔒 Configuring Gateway for loopback binding...');
  console.log(`   Port: ${port}`);
  console.log(`   Auth Mode: ${auth.mode}`);
  console.log('');

  // 生成配置
  const gatewayConfig = {
    gateway: {
      mode: 'local',
      port,
      bind: 'loopback',
      auth: auth.mode === 'token' ? {
        mode: 'token',
        token: auth.token || '${OPENCLAW_GATEWAY_TOKEN}'
      } : auth.mode === 'password' ? {
        mode: 'password',
        password: auth.password || '${OPENCLAW_GATEWAY_PASSWORD}'
      } : {
        mode: 'none'
      }
    }
  };

  // 写入配置文件
  const configPath = path.join(os.homedir(), '.openclaw', 'openclaw.json');
  const existingConfig = fs.existsSync(configPath)
    ? JSON.parse(fs.readFileSync(configPath, 'utf-8'))
    : {};

  const mergedConfig = {
    ...existingConfig,
    gateway: {
      ...existingConfig.gateway,
      ...gatewayConfig.gateway
    }
  };

  fs.writeFileSync(configPath, JSON.stringify(mergedConfig, null, 2));
  console.log('✅ Configuration written to:', configPath);

  // 验证配置
  console.log('');
  console.log('🔍 Verifying configuration...');
  await verifyConfig();

  // 启动 Gateway
  console.log('');
  console.log('🚀 Starting Gateway with loopback binding...');
  await startGateway();
}

/**
 * 验证配置
 */
async function verifyConfig(): Promise<void> {
  return new Promise((resolve, reject) => {
    const doctor = spawn('openclaw', ['doctor'], {
      stdio: 'inherit',
      shell: true
    });

    doctor.on('exit', (code) => {
      if (code === 0) {
        console.log('✅ Configuration is valid');
        resolve();
      } else {
        console.log('⚠️  Configuration has issues');
        reject(new Error('Configuration validation failed'));
      }
    });
  });
}

/**
 * 启动 Gateway
 */
async function startGateway(): Promise<void> {
  return new Promise((resolve) => {
    const gateway = spawn('openclaw', ['gateway', '--verbose'], {
      stdio: 'inherit',
      shell: true
    });

    gateway.on('error', (error) => {
      console.error('❌ Failed to start Gateway:', error);
      process.exit(1);
    });

    // Ctrl+C 处理
    process.on('SIGINT', () => {
      console.log('\n🛑 Stopping Gateway...');
      gateway.kill('SIGINT');
      process.exit(0);
    });
  });
}

// 使用示例
if (require.main === module) {
  const token = process.env.OPENCLAW_GATEWAY_TOKEN || 'dev-token-' + Math.random().toString(36).substring(7);

  configureLoopback({
    port: 18789,
    auth: {
      mode: 'token',
      token
    }
  }).catch((error) => {
    console.error('Configuration failed:', error);
    process.exit(1);
  });
}

export { configureLoopback };
```

## 场景 2：lan 绑定（局域网访问）

### 原理

lan 模式监听所有接口（0.0.0.0），允许局域网访问，必须启用认证，适合家庭网络或办公室内部使用。

### 实现

```typescript
// scripts/config-lan.ts
/**
 * lan 绑定配置脚本
 * 用途：局域网访问，强制认证
 */

import { spawn } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import * as crypto from 'crypto';

interface LanConfig {
  port?: number;
  token?: string;
  generateToken?: boolean;
}

/**
 * 配置 lan 绑定
 */
async function configureLan(config: LanConfig = {}): Promise<void> {
  const {
    port = 18789,
    generateToken = true
  } = config;

  let { token } = config;

  console.log('🌐 Configuring Gateway for LAN binding...');
  console.log(`   Port: ${port}`);
  console.log('');

  // 生成强随机 token
  if (generateToken && !token) {
    token = crypto.randomBytes(32).toString('base64');
    console.log('🔑 Generated secure token (save this!)');
    console.log(`   Token: ${token}`);
    console.log('');
  }

  if (!token) {
    throw new Error('Token is required for LAN binding');
  }

  // 生成配置
  const gatewayConfig = {
    gateway: {
      mode: 'local',
      port,
      bind: 'lan',
      auth: {
        mode: 'token',
        token: `\${OPENCLAW_GATEWAY_TOKEN}`
      }
    }
  };

  // 写入配置文件
  const configPath = path.join(os.homedir(), '.openclaw', 'openclaw.json');
  const existingConfig = fs.existsSync(configPath)
    ? JSON.parse(fs.readFileSync(configPath, 'utf-8'))
    : {};

  const mergedConfig = {
    ...existingConfig,
    gateway: {
      ...existingConfig.gateway,
      ...gatewayConfig.gateway
    }
  };

  fs.writeFileSync(configPath, JSON.stringify(mergedConfig, null, 2));
  console.log('✅ Configuration written to:', configPath);

  // 写入环境变量
  const envPath = path.join(os.homedir(), '.openclaw', '.env');
  const envContent = `OPENCLAW_GATEWAY_TOKEN=${token}\n`;
  fs.writeFileSync(envPath, envContent);
  console.log('✅ Token saved to:', envPath);

  // 安全审计
  console.log('');
  console.log('🔍 Running security audit...');
  await securityAudit();

  // 显示连接信息
  console.log('');
  console.log('📡 Gateway will be accessible at:');
  console.log(`   ws://127.0.0.1:${port} (localhost)`);
  console.log(`   ws://<your-lan-ip>:${port} (LAN)`);
  console.log('');
  console.log('⚠️  Security reminders:');
  console.log('   - Keep your token secret');
  console.log('   - Do not expose to public internet');
  console.log('   - Use firewall rules to restrict access');
}

/**
 * 安全审计
 */
async function securityAudit(): Promise<void> {
  return new Promise((resolve) => {
    const audit = spawn('openclaw', ['security', 'audit'], {
      stdio: 'inherit',
      shell: true
    });

    audit.on('exit', (code) => {
      if (code === 0) {
        console.log('✅ Security audit passed');
      } else {
        console.log('⚠️  Security audit found issues');
      }
      resolve();
    });
  });
}

// 使用示例
if (require.main === module) {
  configureLan({
    port: 18789,
    generateToken: true
  }).catch((error) => {
    console.error('Configuration failed:', error);
    process.exit(1);
  });
}

export { configureLan };
```

## 场景 3：tailnet 绑定（Tailscale 远程访问）

### 原理

tailnet 模式只监听 Tailscale 接口，通过加密隧道实现安全远程访问，适合跨网络远程访问。

### 实现

```typescript
// scripts/config-tailnet.ts
/**
 * tailnet 绑定配置脚本
 * 用途：Tailscale 远程访问，端到端加密
 */

import { spawn, exec } from 'child_process';
import { promisify } from 'util';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

const execAsync = promisify(exec);

interface TailnetConfig {
  port?: number;
  mode?: 'serve' | 'funnel';
  authMode?: 'trusted-proxy' | 'token';
}

/**
 * 配置 tailnet 绑定
 */
async function configureTailnet(config: TailnetConfig = {}): Promise<void> {
  const {
    port = 18789,
    mode = 'serve',
    authMode = 'trusted-proxy'
  } = config;

  console.log('🔐 Configuring Gateway for Tailscale binding...');
  console.log(`   Port: ${port}`);
  console.log(`   Tailscale Mode: ${mode}`);
  console.log(`   Auth Mode: ${authMode}`);
  console.log('');

  // 检查 Tailscale 状态
  console.log('🔍 Checking Tailscale status...');
  const tailscaleStatus = await checkTailscale();

  if (!tailscaleStatus.running) {
    throw new Error('Tailscale is not running. Please start Tailscale first.');
  }

  console.log('✅ Tailscale is running');
  console.log(`   IP: ${tailscaleStatus.ip}`);
  console.log('');

  // 生成配置
  const gatewayConfig = {
    gateway: {
      mode: 'local',
      port,
      bind: 'tailnet',
      auth: authMode === 'trusted-proxy' ? {
        mode: 'trusted-proxy'
      } : {
        mode: 'token',
        token: '${OPENCLAW_GATEWAY_TOKEN}'
      },
      tailscale: {
        mode
      }
    }
  };

  // 写入配置文件
  const configPath = path.join(os.homedir(), '.openclaw', 'openclaw.json');
  const existingConfig = fs.existsSync(configPath)
    ? JSON.parse(fs.readFileSync(configPath, 'utf-8'))
    : {};

  const mergedConfig = {
    ...existingConfig,
    gateway: {
      ...existingConfig.gateway,
      ...gatewayConfig.gateway
    }
  };

  fs.writeFileSync(configPath, JSON.stringify(mergedConfig, null, 2));
  console.log('✅ Configuration written to:', configPath);

  // 安全警告
  if (mode === 'funnel') {
    console.log('');
    console.log('⚠️  WARNING: Funnel mode exposes Gateway to public internet!');
    console.log('   Only use Funnel if you understand the security implications.');
    console.log('');
  }

  // 显示连接信息
  console.log('');
  console.log('📡 Gateway will be accessible at:');
  console.log(`   ws://${tailscaleStatus.ip}:${port} (Tailscale)`);

  if (mode === 'serve') {
    console.log('');
    console.log('✅ Tailscale Serve provides:');
    console.log('   - End-to-end encryption');
    console.log('   - Identity-based authentication');
    console.log('   - No public internet exposure');
  }
}

/**
 * 检查 Tailscale 状态
 */
async function checkTailscale(): Promise<{ running: boolean; ip?: string }> {
  try {
    const { stdout } = await execAsync('tailscale status --json');
    const status = JSON.parse(stdout);

    return {
      running: true,
      ip: status.Self?.TailscaleIPs?.[0]
    };
  } catch (error) {
    return { running: false };
  }
}

// 使用示例
if (require.main === module) {
  configureTailnet({
    port: 18789,
    mode: 'serve',
    authMode: 'trusted-proxy'
  }).catch((error) => {
    console.error('Configuration failed:', error);
    process.exit(1);
  });
}

export { configureTailnet };
```

## 故障排查

### 问题 1：端口已被占用

```bash
# 检查端口占用
lsof -i :18789

# 强制释放端口
openclaw gateway --force
```

### 问题 2：lan 绑定拒绝启动

```bash
# 错误：Refusing to bind gateway to lan without auth
# 解决：添加认证
openclaw gateway --bind lan --token <long-random-token>
```

### 问题 3：Tailscale 连接失败

```bash
# 检查 Tailscale 状态
tailscale status

# 重启 Tailscale
sudo tailscale down
sudo tailscale up
```

## 总结

端口与绑定配置是 Gateway 网络安全的核心：

- **loopback**：本地开发，最高安全性
- **lan**：局域网访问，强制认证
- **tailnet**：远程访问，端到端加密
- **安全原则**：默认最安全，按需开放

掌握这些配置可以确保 Gateway 在不同场景下的安全性。
