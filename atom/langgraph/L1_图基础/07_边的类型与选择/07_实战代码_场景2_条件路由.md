# 实战代码：条件路由

> 本文档展示 LangGraph 条件边的实战应用，包括二分支决策、多分支决策、循环与退出等场景

---

## 代码说明

本示例展示了 LangGraph 中条件边的四种典型应用场景：

1. **场景1：二分支决策** - 根据数值大小进行路由
2. **场景2：多分支决策** - 根据类别进行多路路由
3. **场景3：循环与退出** - 迭代优化场景
4. **场景4：实际应用** - 文本分类路由系统

---

## 完整代码

```python
"""
LangGraph 条件路由实战示例
演示：二分支决策、多分支决策、循环与退出、实际应用
"""

from typing import Literal, TypedDict
from langgraph.graph import StateGraph, START, END


# ===== 场景1：二分支决策 =====
print("=" * 60)
print("场景1：二分支决策 - 根据数值大小路由")
print("=" * 60)


class BinaryState(TypedDict):
    """二分支决策的状态"""
    value: int
    path_taken: str
    result: int


def process_input(state: BinaryState) -> dict:
    """处理输入数据"""
    print(f"[process_input] 接收到值: {state['value']}")
    return {"value": state["value"]}


def route_by_value(state: BinaryState) -> Literal["high_path", "low_path"]:
    """
    路由函数：根据数值大小决定路径
    - value > 50 -> high_path
    - value <= 50 -> low_path
    """
    if state["value"] > 50:
        print(f"[route_by_value] 值 {state['value']} > 50，路由到 high_path")
        return "high_path"
    else:
        print(f"[route_by_value] 值 {state['value']} <= 50，路由到 low_path")
        return "low_path"


def high_handler(state: BinaryState) -> dict:
    """高值处理：乘以2"""
    result = state["value"] * 2
    print(f"[high_handler] 高值处理: {state['value']} * 2 = {result}")
    return {"path_taken": "high", "result": result}


def low_handler(state: BinaryState) -> dict:
    """低值处理：加10"""
    result = state["value"] + 10
    print(f"[low_handler] 低值处理: {state['value']} + 10 = {result}")
    return {"path_taken": "low", "result": result}


# 构建二分支决策图
binary_builder = StateGraph(BinaryState)
binary_builder.add_node("process", process_input)
binary_builder.add_node("high_path", high_handler)
binary_builder.add_node("low_path", low_handler)

binary_builder.add_edge(START, "process")
binary_builder.add_conditional_edges(
    "process",
    route_by_value,
    {"high_path": "high_path", "low_path": "low_path"}
)
binary_builder.add_edge("high_path", END)
binary_builder.add_edge("low_path", END)

binary_graph = binary_builder.compile()

# 测试二分支决策
print("\n测试1：高值路径 (value=80)")
result1 = binary_graph.invoke({"value": 80})
print(f"结果: {result1}\n")

print("测试2：低值路径 (value=30)")
result2 = binary_graph.invoke({"value": 30})
print(f"结果: {result2}\n")


# ===== 场景2：多分支决策 =====
print("=" * 60)
print("场景2：多分支决策 - 根据类别进行多路路由")
print("=" * 60)


class MultiState(TypedDict):
    """多分支决策的状态"""
    category: str
    data: str
    handler: str
    result: str


def classify_input(state: MultiState) -> dict:
    """分类输入数据"""
    print(f"[classify_input] 接收到类别: {state['category']}, 数据: {state['data']}")
    return {"category": state["category"], "data": state["data"]}


def route_by_category(state: MultiState) -> Literal["text_handler", "image_handler", "audio_handler"]:
    """
    路由函数：根据类别决定处理器
    - text -> text_handler
    - image -> image_handler
    - audio -> audio_handler
    """
    category = state["category"]
    print(f"[route_by_category] 类别 '{category}' 路由到 {category}_handler")
    return f"{category}_handler"


def text_handler(state: MultiState) -> dict:
    """文本处理器"""
    result = f"文本处理: {state['data'].upper()}"
    print(f"[text_handler] {result}")
    return {"handler": "text", "result": result}


def image_handler(state: MultiState) -> dict:
    """图像处理器"""
    result = f"图像处理: 调整大小 {state['data']}"
    print(f"[image_handler] {result}")
    return {"handler": "image", "result": result}


def audio_handler(state: MultiState) -> dict:
    """音频处理器"""
    result = f"音频处理: 转码 {state['data']}"
    print(f"[audio_handler] {result}")
    return {"handler": "audio", "result": result}


# 构建多分支决策图
multi_builder = StateGraph(MultiState)
multi_builder.add_node("classify", classify_input)
multi_builder.add_node("text_handler", text_handler)
multi_builder.add_node("image_handler", image_handler)
multi_builder.add_node("audio_handler", audio_handler)

multi_builder.add_edge(START, "classify")
multi_builder.add_conditional_edges(
    "classify",
    route_by_category,
    {
        "text_handler": "text_handler",
        "image_handler": "image_handler",
        "audio_handler": "audio_handler"
    }
)
multi_builder.add_edge("text_handler", END)
multi_builder.add_edge("image_handler", END)
multi_builder.add_edge("audio_handler", END)

multi_graph = multi_builder.compile()

# 测试多分支决策
print("\n测试1：文本处理")
result1 = multi_graph.invoke({"category": "text", "data": "hello world"})
print(f"结果: {result1}\n")

print("测试2：图像处理")
result2 = multi_graph.invoke({"category": "image", "data": "photo.jpg"})
print(f"结果: {result2}\n")

print("测试3：音频处理")
result3 = multi_graph.invoke({"category": "audio", "data": "song.mp3"})
print(f"结果: {result3}\n")


# ===== 场景3：循环与退出 =====
print("=" * 60)
print("场景3：循环与退出 - 迭代优化场景")
print("=" * 60)


class LoopState(TypedDict):
    """循环决策的状态"""
    value: int
    iteration: int
    max_iterations: int
    history: list[int]


def initialize(state: LoopState) -> dict:
    """初始化状态"""
    print(f"[initialize] 初始值: {state['value']}, 最大迭代次数: {state['max_iterations']}")
    return {
        "value": state["value"],
        "iteration": 0,
        "max_iterations": state["max_iterations"],
        "history": [state["value"]]
    }


def optimize(state: LoopState) -> dict:
    """优化处理：每次迭代值减半"""
    new_value = state["value"] // 2
    new_iteration = state["iteration"] + 1
    new_history = state["history"] + [new_value]

    print(f"[optimize] 迭代 {new_iteration}: {state['value']} -> {new_value}")

    return {
        "value": new_value,
        "iteration": new_iteration,
        "history": new_history
    }


def route_loop(state: LoopState) -> Literal["continue", "exit"]:
    """
    路由函数：决定是否继续循环
    - 达到最大迭代次数 -> exit
    - 值小于等于1 -> exit
    - 否则 -> continue
    """
    if state["iteration"] >= state["max_iterations"]:
        print(f"[route_loop] 达到最大迭代次数 {state['max_iterations']}，退出循环")
        return "exit"
    elif state["value"] <= 1:
        print(f"[route_loop] 值已降至 {state['value']}，退出循环")
        return "exit"
    else:
        print(f"[route_loop] 继续优化 (迭代 {state['iteration']}/{state['max_iterations']})")
        return "continue"


def finalize(state: LoopState) -> dict:
    """完成处理"""
    print(f"[finalize] 优化完成，最终值: {state['value']}, 历史: {state['history']}")
    return {}


# 构建循环决策图
loop_builder = StateGraph(LoopState)
loop_builder.add_node("initialize", initialize)
loop_builder.add_node("optimize", optimize)
loop_builder.add_node("finalize", finalize)

loop_builder.add_edge(START, "initialize")
loop_builder.add_conditional_edges(
    "initialize",
    route_loop,
    {
        "continue": "optimize",
        "exit": "finalize"
    }
)
loop_builder.add_conditional_edges(
    "optimize",
    route_loop,
    {
        "continue": "optimize",  # 循环
        "exit": "finalize"
    }
)
loop_builder.add_edge("finalize", END)

loop_graph = loop_builder.compile()

# 测试循环决策
print("\n测试1：正常迭代")
result1 = loop_graph.invoke({"value": 100, "max_iterations": 5})
print(f"结果: {result1}\n")

print("测试2：快速收敛")
result2 = loop_graph.invoke({"value": 8, "max_iterations": 10})
print(f"结果: {result2}\n")


# ===== 场景4：实际应用 - 文本分类路由系统 =====
print("=" * 60)
print("场景4：实际应用 - 文本分类路由系统")
print("=" * 60)


class TextClassificationState(TypedDict):
    """文本分类路由的状态"""
    text: str
    intent: str
    confidence: float
    response: str


def analyze_text(state: TextClassificationState) -> dict:
    """分析文本意图（简化版，实际应用中会使用 LLM）"""
    text = state["text"].lower()

    # 简单的关键词匹配
    if any(word in text for word in ["天气", "温度", "下雨"]):
        intent = "weather"
        confidence = 0.9
    elif any(word in text for word in ["新闻", "资讯", "消息"]):
        intent = "news"
        confidence = 0.85
    elif any(word in text for word in ["帮助", "怎么", "如何"]):
        intent = "help"
        confidence = 0.8
    else:
        intent = "general"
        confidence = 0.5

    print(f"[analyze_text] 文本: '{state['text']}'")
    print(f"[analyze_text] 意图: {intent}, 置信度: {confidence}")

    return {"text": state["text"], "intent": intent, "confidence": confidence}


def route_by_intent(state: TextClassificationState) -> Literal["weather_handler", "news_handler", "help_handler", "general_handler"]:
    """
    路由函数：根据意图决定处理器
    """
    intent = state["intent"]
    print(f"[route_by_intent] 路由到 {intent}_handler")
    return f"{intent}_handler"


def weather_handler(state: TextClassificationState) -> dict:
    """天气查询处理器"""
    response = f"天气查询: 今天晴天，温度 25°C"
    print(f"[weather_handler] {response}")
    return {"response": response}


def news_handler(state: TextClassificationState) -> dict:
    """新闻查询处理器"""
    response = f"新闻资讯: 今日头条新闻..."
    print(f"[news_handler] {response}")
    return {"response": response}


def help_handler(state: TextClassificationState) -> dict:
    """帮助处理器"""
    response = f"帮助信息: 我可以帮您查询天气、新闻等信息"
    print(f"[help_handler] {response}")
    return {"response": response}


def general_handler(state: TextClassificationState) -> dict:
    """通用处理器"""
    response = f"通用回复: 您说的是 '{state['text']}'，我理解了"
    print(f"[general_handler] {response}")
    return {"response": response}


# 构建文本分类路由图
text_builder = StateGraph(TextClassificationState)
text_builder.add_node("analyze", analyze_text)
text_builder.add_node("weather_handler", weather_handler)
text_builder.add_node("news_handler", news_handler)
text_builder.add_node("help_handler", help_handler)
text_builder.add_node("general_handler", general_handler)

text_builder.add_edge(START, "analyze")
text_builder.add_conditional_edges(
    "analyze",
    route_by_intent,
    {
        "weather_handler": "weather_handler",
        "news_handler": "news_handler",
        "help_handler": "help_handler",
        "general_handler": "general_handler"
    }
)
text_builder.add_edge("weather_handler", END)
text_builder.add_edge("news_handler", END)
text_builder.add_edge("help_handler", END)
text_builder.add_edge("general_handler", END)

text_graph = text_builder.compile()

# 测试文本分类路由
print("\n测试1：天气查询")
result1 = text_graph.invoke({"text": "今天天气怎么样？"})
print(f"结果: {result1}\n")

print("测试2：新闻查询")
result2 = text_graph.invoke({"text": "有什么新闻吗？"})
print(f"结果: {result2}\n")

print("测试3：帮助请求")
result3 = text_graph.invoke({"text": "你能帮我做什么？"})
print(f"结果: {result3}\n")

print("测试4：通用对话")
result4 = text_graph.invoke({"text": "你好啊"})
print(f"结果: {result4}\n")


# ===== 总结 =====
print("=" * 60)
print("总结：条件路由的关键要点")
print("=" * 60)
print("""
1. 路由函数设计
   - 使用 Literal 类型注解明确返回值
   - 路由函数应该是纯函数，只读取状态
   - 返回值必须与 path_map 中的键匹配

2. 二分支决策
   - 适用于简单的是/否判断
   - 路由函数返回两个可能的值
   - 常见场景：阈值判断、条件检查

3. 多分支决策
   - 适用于多种类别的分类
   - 路由函数返回多个可能的值
   - 常见场景：意图识别、类型分类

4. 循环与退出
   - 条件边可以指向自身形成循环
   - 需要明确的退出条件
   - 常见场景：迭代优化、重试机制

5. 实际应用
   - 结合业务逻辑设计路由规则
   - 可以集成 LLM 进行智能路由
   - 注意处理边界情况和异常
""")
```

## 代码要点

### 1. 路由函数设计

```python
def route_by_value(state: State) -> Literal["path_a", "path_b"]:
    """
    路由函数的关键要点：
    1. 使用 Literal 类型注解明确返回值
    2. 只读取状态，不修改状态
    3. 返回值必须与 path_map 匹配
    """
    if condition:
        return "path_a"
    return "path_b"
```

### 2. 条件边配置

```python
builder.add_conditional_edges(
    "source_node",           # 源节点
    route_function,          # 路由函数
    {                        # path_map：映射返回值到目标节点
        "path_a": "node_a",
        "path_b": "node_b"
    }
)
```

### 3. 循环实现

```python
# 条件边可以指向自身形成循环
builder.add_conditional_edges(
    "process",
    route_loop,
    {
        "continue": "process",  # 循环
        "exit": END             # 退出
    }
)
```

### 4. 类型安全

```python
# 使用 Literal 类型注解提供类型安全

    return "a"
```
