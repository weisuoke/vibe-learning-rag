# 实战代码 - 场景2：性能优化实战

## 场景说明

一个 LangGraph 工作流运行一段时间后变慢了，需要：

1. 诊断状态膨胀问题 —— 找出哪个字段在"吃"空间
2. 优化消息列表 —— 自定义裁剪 Reducer 控制增长上限
3. 临时数据清理 —— 中间结果用完即清，不污染检查点
4. 对比优化前后的状态大小 —— 用数据说话

**核心问题：** LangGraph 每次节点执行后都保存检查点。如果状态中有无限增长的字段（消息列表、检索结果），检查点会越来越大，序列化/反序列化成本线性增长，最终拖垮整个工作流。

[来源: reference/source_state_management_01.md | LangGraph 源码分析]

## 完整代码

```python
"""
状态膨胀诊断与优化实战
演示：状态大小监控、消息裁剪、临时值优化、优化前后对比

运行环境：Python 3.13+, langgraph
安装依赖：uv add langgraph
"""

import sys
import json
from typing import Annotated, Any
from typing_extensions import TypedDict
from operator import add

from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.memory import InMemorySaver


# ============================================================
# 1. 诊断工具：测量状态各字段大小
# ============================================================

def diagnose_state(label: str, state: dict) -> int:
    """诊断状态大小，返回总字节数"""
    total = len(json.dumps(state, default=str).encode())
    print(f"  [{label}] 总大小: {total:,} bytes ({total/1024:.1f} KB)")

    # 逐字段分析
    fields = []
    for key, value in state.items():
        size = len(json.dumps(value, default=str).encode())
        fields.append((key, size))
    fields.sort(key=lambda x: x[1], reverse=True)

    for key, size in fields[:3]:  # 只显示 Top-3
        pct = size / total * 100 if total > 0 else 0
        bar = "#" * int(pct / 5)
        print(f"    {key:25s}: {size:6,} bytes ({pct:4.1f}%) {bar}")
    return total


# ============================================================
# 2. 问题版本：状态无限膨胀
# ============================================================

print("=" * 55)
print("=== 优化前：状态无限膨胀 ===")
print("=" * 55)


class BloatedState(TypedDict):
    """问题状态：两个字段都用 add reducer，无限增长"""
    messages: Annotated[list[str], add]          # 消息不断累积
    all_results: Annotated[list[dict], add]      # 检索结果不断累积
    intermediate: dict                           # 中间数据也留着
    query: str
    answer: str


def bloated_search(state: BloatedState) -> dict:
    """模拟检索：每次产生大量结果，全部追加到状态"""
    query = state["query"]
    results = [
        {"content": f"关于'{query}'的文档{i}，" + "详细内容 " * 80,
         "score": 0.9 - i * 0.05}
        for i in range(5)
    ]
    return {
        "all_results": results,
        "intermediate": {
            "raw_scores": [0.9, 0.85, 0.8, 0.75, 0.7],
            "debug_log": "检索耗时150ms，扫描10000条记录 " * 20,
        },
        "messages": [f"检索完成：找到5篇关于'{query}'的文档"],
    }


def bloated_generate(state: BloatedState) -> dict:
    """模拟生成：中间数据没清理"""
    docs = state.get("all_results", [])
    answer = f"基于{len(docs)}篇文档的回答：{state['query']}..."
    return {
        "answer": answer,
        "messages": [f"生成完成，引用了{len(docs)}篇文档"],
        # 注意：intermediate 没有清理！
    }


# 构建膨胀版图
builder_bad = StateGraph(BloatedState)
builder_bad.add_node("search", bloated_search)
builder_bad.add_node("generate", bloated_generate)
builder_bad.add_edge(START, "search")
builder_bad.add_edge("search", "generate")
builder_bad.add_edge("generate", END)

mem_bad = InMemorySaver()
graph_bad = builder_bad.compile(checkpointer=mem_bad)

# 模拟多轮对话
bad_sizes = []
for turn in range(10):
    config_bad = {"configurable": {"thread_id": "bloated-demo"}}
    result = graph_bad.invoke(
        {
            "messages": [f"用户第{turn+1}轮提问"],
            "all_results": [],
            "intermediate": {},
            "query": f"第{turn+1}轮问题",
            "answer": "",
        },
        config_bad,
    )
    snapshot = graph_bad.get_state(config_bad)
    size = len(json.dumps(snapshot.values, default=str).encode())
    bad_sizes.append(size)
    if turn in [0, 4, 9]:
        print(f"\n第{turn+1}轮:")
        diagnose_state(f"轮次{turn+1}", snapshot.values)


# ============================================================
# 3. 优化版本：裁剪 + 清理
# ============================================================

print()
print("=" * 55)
print("=== 优化后：裁剪 + 清理 ===")
print("=" * 55)

MAX_MESSAGES = 6


def trimmed_messages(existing: list[str], new: list[str]) -> list[str]:
    """带裁剪的消息 Reducer：只保留最近 N 条

    原理：
    - 先合并新旧消息
    - 超过上限时，只保留最后 MAX_MESSAGES 条
    - 检查点大小因此有上限，不会无限增长

    这是源码中 BinaryOperatorAggregate 的自定义变体。
    """
    combined = existing + new
    if len(combined) > MAX_MESSAGES:
        return combined[-MAX_MESSAGES:]
    return combined


class OptimizedState(TypedDict):
    """优化状态：消息裁剪 + 检索结果覆盖 + 无中间数据"""
    messages: Annotated[list[str], trimmed_messages]  # 自动裁剪
    search_results: list[dict]    # LastValue：只保留最新一轮结果
    query: str
    answer: str


def optimized_search(state: OptimizedState) -> dict:
    """优化检索：结果用覆盖模式，不累积"""
    query = state["query"]
    results = [
        {"content": f"关于'{query}'的文档{i}，" + "详细内容 " * 80,
         "score": 0.9 - i * 0.05}
        for i in range(5)
    ]
    return {
        "search_results": results,  # 覆盖，不追加
        "messages": [f"检索完成：找到5篇关于'{query}'的文档"],
    }


def optimized_generate(state: OptimizedState) -> dict:
    """优化生成：用完结果后清空"""
    docs = state.get("search_results", [])
    answer = f"基于{len(docs)}篇文档的回答：{state['query']}..."
    return {
        "answer": answer,
        "search_results": [],  # 关键：用完即清！
        "messages": [f"生成完成，引用了{len(docs)}篇文档"],
    }


# 构建优化版图
builder_good = StateGraph(OptimizedState)
builder_good.add_node("search", optimized_search)
builder_good.add_node("generate", optimized_generate)
builder_good.add_edge(START, "search")
builder_good.add_edge("search", "generate")
builder_good.add_edge("generate", END)

mem_good = InMemorySaver()
graph_good = builder_good.compile(checkpointer=mem_good)

# 模拟同样的多轮对话
good_sizes = []
for turn in range(10):
    config_good = {"configurable": {"thread_id": "optimized-demo"}}
    result = graph_good.invoke(
        {
            "messages": [f"用户第{turn+1}轮提问"],
            "search_results": [],
            "query": f"第{turn+1}轮问题",
            "answer": "",
        },
        config_good,
    )
    snapshot = graph_good.get_state(config_good)
    size = len(json.dumps(snapshot.values, default=str).encode())
    good_sizes.append(size)
    if turn in [0, 4, 9]:
        print(f"\n第{turn+1}轮:")
        diagnose_state(f"轮次{turn+1}", snapshot.values)


# ============================================================
# 4. 对比总结
# ============================================================

print()
print("=" * 55)
print("=== 优化前后对比 ===")
print("=" * 55)
print(f"\n{'轮次':>6s}  {'优化前':>10s}  {'优化后':>10s}  {'节省':>8s}")
print("-" * 40)
for i in [0, 4, 9]:
    saved = (1 - good_sizes[i] / bad_sizes[i]) * 100 if bad_sizes[i] > 0 else 0
    print(f"  第{i+1:2d}轮  {bad_sizes[i]:>8,} B  {good_sizes[i]:>8,} B  {saved:5.1f}%")

print(f"\n增长趋势:")
print(f"  优化前: 第1轮 {bad_sizes[0]:,}B → 第10轮 {bad_sizes[9]:,}B "
      f"({bad_sizes[9]/bad_sizes[0]:.1f}x 增长)")
print(f"  优化后: 第1轮 {good_sizes[0]:,}B → 第10轮 {good_sizes[9]:,}B "
      f"({good_sizes[9]/good_sizes[0]:.1f}x 增长)")

print(f"\n结论:")
print(f"  优化前：状态线性增长，10轮后已经是初始的数倍")
print(f"  优化后：状态大小稳定，消息有上限，临时数据及时清理")
```

## 优化要点总结

### 三个核心优化手段

| 手段 | 问题 | 解决方案 | 代码关键 |
|------|------|---------|---------|
| 消息裁剪 | `Annotated[list, add]` 无限增长 | 自定义 `trimmed_messages` Reducer | `combined[-MAX:]` |
| 覆盖代替累积 | 检索结果用 `add` 不断追加 | 改用 LastValue（无 Reducer） | 去掉 `Annotated` |
| 临时数据清理 | 中间结果留在检查点中 | 生成节点中显式清空 | `"search_results": []` |

### 源码层面的优化原理

```
BinaryOperatorAggregate.update():
  空更新 → return False → 跳过检查点写入

EphemeralValue.consume():
  读取后 value = MISSING → 不进检查点

channel_versions:
  只有版本号变化的通道才写入 → 增量检查点

BaseChannel.__slots__:
  固定属性列表 → 每个实例省 ~100 bytes
```

### 诊断口诀

```
状态变慢三步查：
1. 量大小 → json.dumps 测总量
2. 找大头 → 逐字段排序看占比
3. 定策略 → 裁剪/覆盖/清理 三选一
```

## 在 RAG 系统中的应用

- **多轮对话**：消息列表是膨胀第一大户，必须用裁剪 Reducer 设上限
- **检索结果**：每轮检索的文档只在当前轮有用，用 LastValue 覆盖而非 `add` 累积
- **Embedding 向量**：768/1536 维浮点数组占用大，绝不存进状态，按需从向量库查
- **调试信息**：`debug_log`、`raw_scores` 等中间数据，生产环境中用完即清
