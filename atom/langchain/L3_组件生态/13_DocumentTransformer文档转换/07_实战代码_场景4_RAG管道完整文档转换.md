## 场景描述

### 业务背景

构建生产级 RAG 问答系统，处理来自网页和 PDF 的混合文档。
从加载到生成回答，中间经历清洗、分块、去重、增强、重排序等多个转换阶段，
每个阶段都依赖 `DocumentTransformer` 接口 —— 它是贯穿整条 RAG 管道的核心抽象。

### 技术挑战

1. **多源异构**：HTML 和 PDF 格式差异大，需要统一清洗
2. **管道编排**：5 个转换阶段需要正确串联
3. **检索优化**：冗余过滤 + 上下文重排序提升生成质量

### 学习目标

- 理解 DocumentTransformer 在 RAG 管道中的 **5 个关键位置**
- 使用 `DocumentCompressorPipeline` 组合多个转换器
- 使用 `ContextualCompressionRetriever` 集成转换管道与检索器
- 掌握推荐顺序：加载 → 清洗 → 分块 → 过滤 → 增强 → 存储 → 检索(重排序) → 生成

## 第二步：文档清洗（DocumentTransformer #1）

> HTML 去噪声，PDF 直接通过。管道中的第一个 DocumentTransformer。

```python
def clean_documents(documents: list[Document]) -> list[Document]:
    """按格式选择清洗策略：HTML → Html2TextTransformer，其他 → 直通"""
    html_docs = [d for d in documents if d.metadata.get("format") == "html"]
    other_docs = [d for d in documents if d.metadata.get("format") != "html"]

    cleaned = []
    if html_docs:
        cleaner = Html2TextTransformer()
        for doc in cleaner.transform_documents(html_docs):
            doc.metadata["cleaner"] = "Html2TextTransformer"
            cleaned.append(doc)

    for doc in other_docs:
        doc.metadata["cleaner"] = "passthrough"
    cleaned.extend(other_docs)
    return cleaned

cleaned_docs = clean_documents(raw_documents)
for doc in cleaned_docs:
    print(f"[{doc.metadata['cleaner']}] {len(doc.page_content)} 字符")
    print(f"  预览: {doc.page_content[:100]}...")
```

## 第五步：元数据增强（DocumentTransformer #4）

> 自定义转换器，只需实现 `transform_documents` 方法即可融入管道。

[来源: reference/source_document_transformer_01.md | BaseDocumentTransformer 接口]

```python
class MetadataEnhancer(BaseDocumentTransformer):
    """自定义转换器：为文档块添加管道处理元数据"""

    def __init__(self, pipeline_name: str = "rag_pipeline_v1"):
        self.pipeline_name = pipeline_name

    def transform_documents(self, documents: list[Document], **kwargs) -> list[Document]:
        enhanced = []
        for idx, doc in enumerate(documents):
            new_doc = Document(
                page_content=doc.page_content,
                metadata={
                    **doc.metadata,
                    "pipeline": self.pipeline_name,
                    "chunk_index": idx,
                    "total_chunks": len(documents),
                    "char_count": len(doc.page_content),
                    "processed_at": datetime.now().isoformat(),
                },
            )
            enhanced.append(new_doc)
        return enhanced

    async def atransform_documents(self, documents: list[Document], **kwargs) -> list[Document]:
        return self.transform_documents(documents, **kwargs)

enhancer = MetadataEnhancer(pipeline_name="rag_full_pipeline_v1")
enhanced_chunks = enhancer.transform_documents(unique_chunks)

print(f"增强后: {len(enhanced_chunks)} 个块")
print(f"示例元数据: {enhanced_chunks[0].metadata}")
```

## 完整封装：RAGPipeline 类

> 将所有阶段封装为可复用的类。

```python
class RAGPipeline:
    """
    完整 RAG 文档转换管道。
    DocumentTransformer 出现在 5 个阶段：
    #1 清洗  #2 分块  #3 过滤  #4 增强  #5 重排序
    """

    def __init__(self, chunk_size=500, chunk_overlap=75, similarity_threshold=0.95):
        self.embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
        self.llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        self.html_cleaner = Html2TextTransformer()                          # Trans #1
        self.splitter = RecursiveCharacterTextSplitter(                     # Trans #2
            chunk_size=chunk_size, chunk_overlap=chunk_overlap,
            add_start_index=True,
            separators=["\n## ", "\n### ", "\n\n", "\n", "。", ".", " ", ""],
        )
        self.redundant_filter = EmbeddingsRedundantFilter(                  # Trans #3
            embeddings=self.embeddings, similarity_threshold=similarity_threshold,
        )
        self.enhancer = MetadataEnhancer()                                  # Trans #4
        self.reordering = LongContextReorder()                              # Trans #5
        self.vectorstore = None

    def ingest(self, documents: list[Document]) -> dict:
        """文档摄入：清洗 → 分块 → 过滤 → 增强 → 存储"""
        stats = {"input": len(documents)}

        # 阶段 1：清洗
        html_docs = [d for d in documents if d.metadata.get("format") == "html"]
        other_docs = [d for d in documents if d.metadata.get("format") != "html"]
        cleaned = self.html_cleaner.transform_documents(html_docs) + other_docs if html_docs else other_docs
        stats["after_clean"] = len(cleaned)

        # 阶段 2-4：分块 → 过滤 → 增强
        chunks = self.splitter.transform_documents(cleaned)
        stats["after_split"] = len(chunks)

        unique = self.redundant_filter.transform_documents(chunks)
        stats["after_filter"] = len(unique)
        stats["removed"] = len(chunks) - len(unique)

        enhanced = self.enhancer.transform_documents(unique)

        # 阶段 5：存储
        self.vectorstore = Chroma.from_documents(
            enhanced, self.embeddings, collection_name="rag_pipeline",
        )
        stats["stored"] = len(enhanced)
        return stats

    def query(self, question: str) -> dict:
        """查询：检索(重排序) → 生成"""
        if not self.vectorstore:
            raise ValueError("请先调用 ingest() 摄入文档")

        # 检索管道（Trans #3 + #5）
        compressor = DocumentCompressorPipeline(
            transformers=[self.redundant_filter, self.reordering]
        )
        retriever = ContextualCompressionRetriever(
            base_compressor=compressor,
            base_retriever=self.vectorstore.as_retriever(search_kwargs={"k": 10}),
        )

        docs = retriever.invoke(question)
        context = "\n\n---\n\n".join(
            f"[来源: {d.metadata.get('source', '未知')}]\n{d.page_content}"
            for d in docs
        )

        prompt = ChatPromptTemplate.from_template(
            "根据以下上下文回答问题。\n\n上下文：\n{context}\n\n问题：{question}\n\n回答："
        )
        answer = (prompt | self.llm | StrOutputParser()).invoke(
            {"context": context, "question": question}
        )
        return {"answer": answer, "sources": [d.metadata.get("source") for d in docs]}


# 使用
rag = RAGPipeline()
stats = rag.ingest(raw_documents)
print(f"摄入统计: {stats}")

result = rag.query("RAG 系统中文档分块的最佳实践是什么？")
print(f"回答: {result['answer']}")
print(f"来源: {result['sources']}")
```

> 对比有无文档转换管道的 RAG 质量差异。

```python
def simple_rag_no_pipeline(documents, query, embeddings, llm):
    """最简 RAG：直接分块 → 存储 → 检索 → 生成（无清洗/过滤/重排序）"""
    splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=75)
    chunks = splitter.split_documents(documents)  # 直接对原始 HTML 分块！

    vs = Chroma.from_documents(chunks, embeddings, collection_name="simple_rag")
    docs = vs.similarity_search(query, k=5)

    context = "\n\n".join(d.page_content for d in docs)
    prompt = ChatPromptTemplate.from_template(
        "根据上下文回答问题。\n上下文：\n{context}\n\n问题：{question}\n\n回答："
    )
    return (prompt | llm | StrOutputParser()).invoke(
        {"context": context, "question": query}
    ), len(chunks), docs
```

## 关键要点

1. **统一接口是关键**：`transform_documents(Sequence[Document]) -> Sequence[Document]` 让所有转换器自由组合
2. **5 个阶段各司其职**：清洗去噪声、分块控粒度、过滤去冗余、增强加元数据、重排序优排列
3. **DocumentCompressorPipeline 是组合利器**：串联多个转换器，配合 `ContextualCompressionRetriever` 在检索时自动执行
4. **2026 最佳实践**：400-512 token 递归分块 + 10-20% 重叠率（NVIDIA 基准测试）
5. **自定义转换器很简单**：继承 `BaseDocumentTransformer`，实现 `transform_documents` 即可

## 常见问题

**Q1：转换器顺序可以调整吗？**

推荐顺序：清洗 → 分块 → 过滤 → 增强 → 重排序。清洗必须在分块前（HTML 标签干扰分块），过滤必须在分块后（需对比块间相似度），重排序必须在检索后。

**Q2：similarity_threshold 怎么调？**

默认 0.95 适合大多数场景。爬虫数据重复率高可降到 0.90，低于 0.85 可能误删语义相近但内容不同的文档。

**Q3：LongContextReorder 原理？**

大模型对上下文中间位置信息关注度最低（"Lost in the Middle"）。LongContextReorder 将最相关文档放在首尾，较低相关性放中间。

**Q4：自定义 DocumentTransformer 需要什么？**

实现 `transform_documents` 方法，输入输出都是 `Sequence[Document]`。异步场景额外实现 `atransform_documents`。
