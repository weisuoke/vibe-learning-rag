## 4.1 安装 OpenClaw

### 核心概念

**全局安装** = 让 `openclaw` 命令在任何目录下都能使用

### 最小步骤

```bash
# 1. 确认 Node.js 版本（必须 22+）
node --version
# 预期输出：v22.x.x

# 2. 全局安装 OpenClaw
npm install -g openclaw@latest

# 3. 验证安装
openclaw --version
# 预期输出：1.2.3
```

### TypeScript 示例：自动化安装检查

```typescript
import { execSync } from 'child_process';

/**
 * 检查并安装 OpenClaw
 */
function installOpenClaw(): void {
  // 1. 检查 Node.js 版本
  const nodeVersion = execSync('node --version', { encoding: 'utf-8' }).trim();
  const majorVersion = parseInt(nodeVersion.slice(1).split('.')[0]);

  if (majorVersion < 22) {
    console.error(`❌ Node.js 版本过低: ${nodeVersion}`);
    console.error('   需要 Node.js 22+');
    console.error('   安装方法: https://nodejs.org');
    process.exit(1);
  }

  console.log(`✅ Node.js 版本: ${nodeVersion}`);

  // 2. 检查是否已安装 OpenClaw
  try {
    const currentVersion = execSync('openclaw --version', {
      encoding: 'utf-8',
      stdio: ['pipe', 'pipe', 'ignore']
    }).trim();
    console.log(`✅ OpenClaw 已安装: ${currentVersion}`);
    return;
  } catch {
    console.log('📦 OpenClaw 未安装，开始安装...');
  }

  // 3. 安装 OpenClaw
  try {
    execSync('npm install -g openclaw@latest', {
      stdio: 'inherit'
    });

    const installedVersion = execSync('openclaw --version', {
      encoding: 'utf-8'
    }).trim();

    console.log(`✅ OpenClaw 安装成功: ${installedVersion}`);
  } catch (error) {
    console.error('❌ 安装失败:', error);
    process.exit(1);
  }
}

// 执行安装
installOpenClaw();
```

**预期输出：**
```
✅ Node.js 版本: v22.18.0
📦 OpenClaw 未安装，开始安装...
[npm install 输出...]
✅ OpenClaw 安装成功: 1.2.3
```

**场景 1：团队新成员入职**

- 运行安装脚本，自动检查和安装

**场景 2：CI/CD 环境**
- 在 Docker 容器中自动安装 OpenClaw

- 在多台开发机器上安装
- 使用脚本确保版本统一

### TypeScript 示例：自动化配置生成

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

/**
 * OpenClaw 配置接口
 */
interface OpenClawConfig {
  gateway: {
    port: number;
    host: string;
  };
  channels: {
    claude?: {
      apiKey: string;
      enabled: boolean;
    };
    gpt?: {
      apiKey: string;
      enabled: boolean;
    };
  };
}

/**
 * 生成 OpenClaw 配置文件
 */
function generateConfig(apiKey: string, port: number = 3000): void {
  const configDir = path.join(os.homedir(), '.openclaw');
  const configPath = path.join(configDir, 'settings.json');

  // 1. 创建配置目录
  if (!fs.existsSync(configDir)) {
    fs.mkdirSync(configDir, { recursive: true });
    console.log(`✅ 创建配置目录: ${configDir}`);
  }

  // 2. 生成配置对象
  const config: OpenClawConfig = {
    gateway: {
      port,
      host: 'localhost'
    },
    channels: {
      claude: {
        apiKey,
        enabled: true
      }
    }
  };

  // 3. 写入配置文件
  fs.writeFileSync(
    configPath,
    JSON.stringify(config, null, 2),
    { mode: 0o600 } // 只有用户可读写
  );

  console.log(`✅ 配置文件已生成: ${configPath}`);
  console.log(`\n下一步:`);
  console.log(`  openclaw gateway start`);
}

// 使用示例
const apiKey = process.env.CLAUDE_API_KEY || 'sk-ant-api03-your-key-here';
generateConfig(apiKey, 3000);
```

```
✅ 配置文件已生成: /Users/username/.openclaw/settings.json

下一步:
  openclaw gateway start
```

## 4.3 启动 Gateway

### 核心概念

**Gateway** = 本地运行的 HTTP 服务器，统一管理多个 AI 通道

### 最小步骤

```bash
# 前台模式（开发调试）
openclaw gateway start

# 守护进程模式（生产环境）
openclaw gateway start -d

# 检查状态
openclaw gateway status
```

### 前台模式 vs 守护进程模式

| 特性 | 前台模式 | 守护进程模式 |
|------|----------|--------------|
| 命令 | `openclaw gateway start` | `openclaw gateway start -d` |
| 日志输出 | 实时显示在终端 | 写入 `~/.openclaw/gateway.log` |
| 进程管理 | Ctrl+C 停止 | `openclaw gateway stop` 停止 |
| 适用场景 | 开发调试 | 生产环境 |
| 终端占用 | 占用终端 | 后台运行 |

### TypeScript 示例：Gateway 启动脚本

```typescript
import { spawn, execSync } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

/**
 * 启动 OpenClaw Gateway
 */
function startGateway(daemon: boolean = false): void {
  const configPath = path.join(os.homedir(), '.openclaw', 'settings.json');

  // 1. 检查配置文件是否存在
  if (!fs.existsSync(configPath)) {
    console.error('❌ 配置文件不存在');
    console.error('   请先运行: openclaw onboard');
    process.exit(1);
  }

  // 2. 读取配置
  const config = JSON.parse(fs.readFileSync(configPath, 'utf-8'));
  const port = config.gateway.port || 3000;

  console.log(`🚀 启动 OpenClaw Gateway...`);
  console.log(`   端口: ${port}`);
  console.log(`   模式: ${daemon ? '守护进程' : '前台'}`);

  if (daemon) {
    // 3a. 守护进程模式
    const logPath = path.join(os.homedir(), '.openclaw', 'gateway.log');
    const logStream = fs.createWriteStream(logPath, { flags: 'a' });

    const child = spawn('openclaw', ['gateway', 'start'], {
      detached: true,
      stdio: ['ignore', logStream, logStream]
    });

    child.unref();

    console.log(`✅ Gateway 已在后台启动 (PID: ${child.pid})`);
    console.log(`   日志文件: ${logPath}`);
    console.log(`   查看日志: tail -f ${logPath}`);
    console.log(`   停止服务: openclaw gateway stop`);
  } else {
    // 3b. 前台模式
    console.log(`\n按 Ctrl+C 停止服务\n`);

    const child = spawn('openclaw', ['gateway', 'start'], {
      stdio: 'inherit'
    });

    child.on('exit', (code) => {
      console.log(`\n✅ Gateway 已停止 (退出码: ${code})`);
    });
  }
}

// 使用示例
const isDaemon = process.argv.includes('--daemon') || process.argv.includes('-d');
startGateway(isDaemon);
```

```
   端口: 3000
   模式: 前台

按 Ctrl+C 停止服务

[2026-02-22 10:00:00] INFO: Gateway started on http://localhost:3000
[2026-02-22 10:00:00] INFO: Claude channel initialized
[2026-02-22 10:00:01] INFO: Ready to accept connections
```

**预期输出（守护进程模式）：**
```
🚀 启动 OpenClaw Gateway...
   端口: 3000

✅ Gateway 已在后台启动 (PID: 12345)
   日志文件: /Users/username/.openclaw/gateway.log
   查看日志: tail -f /Users/username/.openclaw/gateway.log

```

**场景 1：本地开发**

## 4.4 验证安装

### 核心概念

**验证** = 确认 OpenClaw 已正确安装并能正常工作

### 最小步骤

```bash
# 1. 检查版本
openclaw --version

# 2. 检查配置文件
ls -la ~/.openclaw/

# 3. 启动 Gateway（前台模式）
openclaw gateway start

# 4. 新开终端，测试连接
curl http://localhost:3000/health

# 5. 测试 AI 调用（需要有效的 API key）
curl -X POST http://localhost:3000/v1/messages \
  -H "Content-Type: application/json" \
  -d '{
    "model": "claude-3-5-sonnet-20241022",
    "max_tokens": 100,
    "messages": [{"role": "user", "content": "Hello!"}]
  }'
```

### TypeScript 示例：完整验证脚本

```typescript
import { execSync } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

/**
 * 验证 OpenClaw 安装
 */
async function verifyInstallation(): Promise<void> {
  console.log('🔍 验证 OpenClaw 安装...\n');

  let allPassed = true;

  // 1. 检查 openclaw 命令
  try {
    const version = execSync('openclaw --version', {
      encoding: 'utf-8',
      stdio: ['pipe', 'pipe', 'ignore']
    }).trim();
    console.log(`✅ OpenClaw 命令可用: ${version}`);
  } catch {
    console.log('❌ OpenClaw 命令不可用');
    console.log('   解决方案: npm install -g openclaw@latest');
    allPassed = false;
  }

  // 2. 检查配置目录
  const configDir = path.join(os.homedir(), '.openclaw');
  if (fs.existsSync(configDir)) {
    console.log(`✅ 配置目录存在: ${configDir}`);
  } else {
    console.log(`❌ 配置目录不存在: ${configDir}`);
    console.log('   解决方案: openclaw onboard');
    allPassed = false;
  }

  // 3. 检查配置文件
  const configPath = path.join(configDir, 'settings.json');
  if (fs.existsSync(configPath)) {
    console.log(`✅ 配置文件存在: ${configPath}`);

    // 验证配置文件格式
    try {
      const config = JSON.parse(fs.readFileSync(configPath, 'utf-8'));
      if (config.gateway && config.channels) {
        console.log('✅ 配置文件格式正确');
      } else {
        console.log('❌ 配置文件格式错误');
        allPassed = false;
      }
    } catch {
      console.log('❌ 配置文件解析失败');
      allPassed = false;
    }
  } else {
    console.log(`❌ 配置文件不存在: ${configPath}`);
    console.log('   解决方案: openclaw onboard');
    allPassed = false;
  }

  // 4. 检查 Gateway 状态
  try {
    const status = execSync('openclaw gateway status', {
      encoding: 'utf-8',
      stdio: ['pipe', 'pipe', 'ignore']
    }).trim();

    if (status.includes('running')) {
      console.log('✅ Gateway 正在运行');

      // 5. 测试健康检查端点
      try {
        const response = execSync(
          'curl -s http://localhost:3000/health',
          { encoding: 'utf-8', timeout: 5000 }
        );

        const health = JSON.parse(response);
        if (health.status === 'ok') {
          console.log('✅ Gateway 健康检查通过');
        } else {
          console.log('❌ Gateway 健康检查失败');
          allPassed = false;
        }
      } catch {
        console.log('❌ 无法连接到 Gateway');
        allPassed = false;
      }
    } else {
      console.log('⚠️  Gateway 未运行');
      console.log('   启动方法: openclaw gateway start');
    }
  } catch {
    console.log('⚠️  Gateway 未运行');
    console.log('   启动方法: openclaw gateway start');
  }

  // 总结
  console.log('\n' + '='.repeat(50));
  if (allPassed) {
    console.log('✅ 所有检查通过！OpenClaw 已正确安装并运行。');
    console.log('\n下一步:');
    console.log('  - 测试 AI 调用: curl -X POST http://localhost:3000/v1/messages ...');
    console.log('  - 查看文档: https://docs.openclaw.ai');
  } else {
    console.log('❌ 部分检查失败，请按照上述提示解决问题。');
  }
}

// 执行验证
verifyInstallation();
```
