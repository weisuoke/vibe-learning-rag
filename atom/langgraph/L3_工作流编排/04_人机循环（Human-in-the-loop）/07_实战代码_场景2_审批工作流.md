## 场景说明

**业务场景**：企业内部的费用报销审批系统

- 员工提交报销申请
- 主管审批（approve / reject）
- 金额超过阈值时，需要二级审批（总监审批）
- 审批人可以修改金额后批准

这是 HITL 最典型的落地场景之一。我们从简单到复杂，分三个子场景逐步实现。

---

## 子场景 1：基础审批流程（单级审批）

### 思路

最简单的审批：申请 → 审批 → 执行 or 拒绝。

```
[START] → [submit] → [review] → approve → [execute] → [END]
                              → reject  → [cancel]  → [END]
```

### 完整代码

```python
"""
子场景 1：基础审批流程
- 员工提交报销申请
- 主管审批：approve → 执行报销，reject → 取消
- 使用 interrupt() 传递结构化审批信息
- 使用 Command(goto=...) 条件路由
"""

from typing import Literal, Optional, TypedDict
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


# ========== 1. 定义状态 ==========

class ApprovalState(TypedDict):
    """审批流程状态"""
    request_id: str           # 申请编号
    requester: str            # 申请人
    action_details: str       # 报销事由
    amount: float             # 报销金额
    status: str               # 流程状态
    reviewer: Optional[str]   # 审批人
    comment: Optional[str]    # 审批意见


# ========== 2. 定义节点 ==========

def submit_request(state: ApprovalState):
    """提交申请节点：记录申请信息"""
    print(f"\n{'='*50}")
    print(f"[提交] 报销申请 {state['request_id']}")
    print(f"  申请人: {state['requester']}")
    print(f"  事由: {state['action_details']}")
    print(f"  金额: ¥{state['amount']:.2f}")
    print(f"{'='*50}")
    return {"status": "pending_review"}


def review_request(state: ApprovalState) -> Command[Literal["execute", "cancel"]]:
    """审批节点：展示申请信息，等待审批人决策"""

    # interrupt() 传递结构化审批信息给客户端
    decision = interrupt({
        "type": "approval_request",
        "request_id": state["request_id"],
        "requester": state["requester"],
        "action_details": state["action_details"],
        "amount": state["amount"],
        "instruction": "请审批此报销申请，输入 'approve' 或 'reject'",
    })

    # 解析审批决策
    if isinstance(decision, dict):
        action = decision.get("action", "reject")
        reviewer = decision.get("reviewer", "未知")
        comment = decision.get("comment", "")
    else:
        action = str(decision).lower()
        reviewer = "审批人"
        comment = ""

    # 根据决策路由到不同节点
    if action == "approve":
        return Command(
            goto="execute",
            update={
                "status": "approved",
                "reviewer": reviewer,
                "comment": comment or "同意报销",
            },
        )
    else:
        return Command(
            goto="cancel",
            update={
                "status": "rejected",
                "reviewer": reviewer,
                "comment": comment or "不予报销",
            },
        )


def execute_request(state: ApprovalState):
    """执行节点：审批通过，执行报销"""
    print(f"\n[执行] 报销申请 {state['request_id']} 已通过")
    print(f"  审批人: {state['reviewer']}")
    print(f"  意见: {state['comment']}")
    print(f"  报销金额: ¥{state['amount']:.2f}")
    return {"status": "completed"}


def cancel_request(state: ApprovalState):
    """取消节点：审批拒绝"""
    print(f"\n[拒绝] 报销申请 {state['request_id']} 已被拒绝")
    print(f"  审批人: {state['reviewer']}")
    print(f"  原因: {state['comment']}")
    return {"status": "cancelled"}


# ========== 3. 构建图 ==========

builder = StateGraph(ApprovalState)
builder.add_node("submit", submit_request)
builder.add_node("review", review_request)
builder.add_node("execute", execute_request)
builder.add_node("cancel", cancel_request)

builder.add_edge(START, "submit")
builder.add_edge("submit", "review")
# review 节点通过 Command(goto=...) 路由，不需要显式 add_edge
builder.add_edge("execute", END)
builder.add_edge("cancel", END)

graph = builder.compile(checkpointer=MemorySaver())


# ========== 4. 运行演示 ==========

config = {"configurable": {"thread_id": "approval-001"}}

# 第一次调用：提交申请，在 review 节点中断
result = graph.invoke(
    {
        "request_id": "EXP-2026-001",
        "requester": "张三",
        "action_details": "出差北京-机票+酒店",
        "amount": 3500.0,
        "status": "draft",
        "reviewer": None,
        "comment": None,
    },
    config,
)
print("\n图已暂停，等待审批...")

# 查看中断信息
snapshot = graph.get_state(config)
for task in snapshot.tasks:
    for intr in task.interrupts:
        print(f"待审批: {intr.value}")

# 审批人做出决策：批准
result = graph.invoke(
    Command(resume={
        "action": "approve",
        "reviewer": "李经理",
        "comment": "出差合理，同意报销",
    }),
    config,
)
print(f"\n最终状态: {result['status']}")
# 输出: completed
```

**运行输出**：
```
==================================================
[提交] 报销申请 EXP-2026-001
  申请人: 张三
  事由: 出差北京-机票+酒店
  金额: ¥3500.00
==================================================

图已暂停，等待审批...
待审批: {'type': 'approval_request', 'request_id': 'EXP-2026-001', ...}

[执行] 报销申请 EXP-2026-001 已通过
  审批人: 李经理
  意见: 出差合理，同意报销
  报销金额: ¥3500.00

最终状态: completed
```

### 关键点

1. `interrupt()` 传递结构化字典，客户端可以渲染成审批表单
2. `Command(goto="execute")` 根据审批结果路由到不同分支
3. `Command(update={...})` 同时更新状态（审批人、意见等）
4. resume 值可以是简单字符串，也可以是结构化字典

## 子场景 2：多级审批链（金额阈值触发二级审批）

### 思路

增加金额判断逻辑：
- 金额 <= 1000：主管直接审批
- 金额 > 1000：主管审批后，还需要总监二级审批

```
[START] → [submit] → [first_review] → approve & amount<=1000 → [execute] → [END]
                                     → approve & amount>1000  → [second_review] → approve → [execute]
                                     → reject                 → [cancel]  → [END]
```

### 完整代码

```python
"""
子场景 2：多级审批链
- 金额 <= 1000：主管一级审批即可
- 金额 > 1000：主管审批后，还需总监二级审批
- 任何一级拒绝都直接取消
"""

from typing import Literal, Optional, TypedDict
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END

AMOUNT_THRESHOLD = 1000.0  # 二级审批阈值


class MultiApprovalState(TypedDict):
    """多级审批状态"""
    request_id: str
    requester: str
    action_details: str
    amount: float
    status: str
    first_reviewer: Optional[str]
    first_comment: Optional[str]
    second_reviewer: Optional[str]
    second_comment: Optional[str]
    approval_level: int  # 当前审批级别：0=未审批, 1=一级通过, 2=二级通过


def submit_node(state: MultiApprovalState):
    """提交申请"""
    print(f"\n[提交] {state['request_id']} | {state['requester']} | ¥{state['amount']:.2f}")
    level_hint = "需二级审批" if state["amount"] > AMOUNT_THRESHOLD else "一级审批即可"
    print(f"  审批级别提示: {level_hint}")
    return {"status": "pending_first_review", "approval_level": 0}


def first_review_node(state: MultiApprovalState) -> Command[Literal["execute", "second_review", "cancel"]]:
    """一级审批：主管审批"""

    decision = interrupt({
        "type": "first_level_approval",
        "request_id": state["request_id"],
        "requester": state["requester"],
        "action_details": state["action_details"],
        "amount": state["amount"],
        "needs_second_review": state["amount"] > AMOUNT_THRESHOLD,
        "instruction": "一级审批：请输入 'approve' 或 'reject'",
    })

    action = decision.get("action", "reject") if isinstance(decision, dict) else str(decision)
    reviewer = decision.get("reviewer", "主管") if isinstance(decision, dict) else "主管"
    comment = decision.get("comment", "") if isinstance(decision, dict) else ""

    if action == "approve":
        # 金额超过阈值 → 需要二级审批
        if state["amount"] > AMOUNT_THRESHOLD:
            print(f"[一级审批] 通过，金额 ¥{state['amount']:.2f} > ¥{AMOUNT_THRESHOLD}，转二级审批")
            return Command(
                goto="second_review",
                update={
                    "status": "pending_second_review",
                    "first_reviewer": reviewer,
                    "first_comment": comment or "同意，转上级审批",
                    "approval_level": 1,
                },
            )
        else:
            print(f"[一级审批] 通过，金额 ¥{state['amount']:.2f} <= ¥{AMOUNT_THRESHOLD}，直接执行")
            return Command(
                goto="execute",
                update={
                    "status": "approved",
                    "first_reviewer": reviewer,
                    "first_comment": comment or "同意报销",
                    "approval_level": 1,
                },
            )
    else:
        return Command(
            goto="cancel",
            update={
                "status": "rejected_first",
                "first_reviewer": reviewer,
                "first_comment": comment or "不予报销",
            },
        )


def second_review_node(state: MultiApprovalState) -> Command[Literal["execute", "cancel"]]:
    """二级审批：总监审批"""

    decision = interrupt({
        "type": "second_level_approval",
        "request_id": state["request_id"],
        "requester": state["requester"],
        "action_details": state["action_details"],
        "amount": state["amount"],
        "first_reviewer": state["first_reviewer"],
        "first_comment": state["first_comment"],
        "instruction": "二级审批（总监）：请输入 'approve' 或 'reject'",
    })

    action = decision.get("action", "reject") if isinstance(decision, dict) else str(decision)
    reviewer = decision.get("reviewer", "总监") if isinstance(decision, dict) else "总监"
    comment = decision.get("comment", "") if isinstance(decision, dict) else ""

    if action == "approve":
        return Command(
            goto="execute",
            update={
                "status": "approved",
                "second_reviewer": reviewer,
                "second_comment": comment or "同意报销",
                "approval_level": 2,
            },
        )
    else:
        return Command(
            goto="cancel",
            update={
                "status": "rejected_second",
                "second_reviewer": reviewer,
                "second_comment": comment or "不予报销",
            },
        )


def execute_node(state: MultiApprovalState):
    """执行报销"""
    print(f"\n[执行] 报销 {state['request_id']} 已通过所有审批")
    print(f"  一级审批: {state.get('first_reviewer', '-')} | {state.get('first_comment', '-')}")
    if state.get("second_reviewer"):
        print(f"  二级审批: {state['second_reviewer']} | {state['second_comment']}")
    print(f"  报销金额: ¥{state['amount']:.2f}")
    return {"status": "completed"}


def cancel_node(state: MultiApprovalState):
    """取消报销"""
    print(f"\n[拒绝] 报销 {state['request_id']} 被拒绝")
    print(f"  状态: {state['status']}")
    return {"status": "cancelled"}


# 构建图
builder = StateGraph(MultiApprovalState)
builder.add_node("submit", submit_node)
builder.add_node("first_review", first_review_node)
builder.add_node("second_review", second_review_node)
builder.add_node("execute", execute_node)
builder.add_node("cancel", cancel_node)

builder.add_edge(START, "submit")
builder.add_edge("submit", "first_review")
# first_review 和 second_review 通过 Command(goto=...) 路由
builder.add_edge("execute", END)
builder.add_edge("cancel", END)

graph = builder.compile(checkpointer=MemorySaver())


# ===== 演示 A：小额报销（一级审批即可） =====

print("\n" + "=" * 60)
print("演示 A：小额报销（¥500，一级审批即可）")
print("=" * 60)

config_a = {"configurable": {"thread_id": "multi-approval-001"}}

graph.invoke(
    {
        "request_id": "EXP-2026-010",
        "requester": "王五",
        "action_details": "团队午餐",
        "amount": 500.0,
        "status": "draft",
        "first_reviewer": None,
        "first_comment": None,
        "second_reviewer": None,
        "second_comment": None,
        "approval_level": 0,
    },
    config_a,
)

# 主管审批通过 → 直接执行（不需要二级审批）
result_a = graph.invoke(
    Command(resume={"action": "approve", "reviewer": "李经理", "comment": "OK"}),
    config_a,
)
print(f"最终状态: {result_a['status']}")  # completed
print(f"审批级别: {result_a['approval_level']}")  # 1


# ===== 演示 B：大额报销（需要二级审批） =====

print("\n" + "=" * 60)
print("演示 B：大额报销（¥5000，需要二级审批）")
print("=" * 60)

config_b = {"configurable": {"thread_id": "multi-approval-002"}}

graph.invoke(
    {
        "request_id": "EXP-2026-011",
        "requester": "赵六",
        "action_details": "客户招待晚宴",
        "amount": 5000.0,
        "status": "draft",
        "first_reviewer": None,
        "first_comment": None,
        "second_reviewer": None,
        "second_comment": None,
        "approval_level": 0,
    },
    config_b,
)

# 一级审批通过 → 自动转二级审批
graph.invoke(
    Command(resume={"action": "approve", "reviewer": "李经理", "comment": "客户重要，同意"}),
    config_b,
)

# 检查状态：应该在 second_review 节点等待
snapshot = graph.get_state(config_b)
print(f"当前等待节点: {snapshot.next}")  # ('second_review',)

# 二级审批通过
result_b = graph.invoke(
    Command(resume={"action": "approve", "reviewer": "王总监", "comment": "批准"}),
    config_b,
)
print(f"最终状态: {result_b['status']}")  # completed
print(f"审批级别: {result_b['approval_level']}")  # 2
```

### 关键点

1. 金额阈值判断在 `first_review_node` 内部完成，通过 `Command(goto=...)` 动态路由
2. 两个审批节点各自有独立的 `interrupt()`，图会在每个节点分别暂停
3. 二级审批的 interrupt 信息中包含一级审批结果，方便总监参考
4. 同一个 `thread_id` 贯穿整个审批链，状态自动累积

## 子场景 3：审批带修改（审批人可修改金额后批准）

### 思路

审批人不仅可以 approve/reject，还可以修改申请内容（比如调低金额）后批准。

这需要在 resume 值中携带修改信息，并通过 `Command(update=...)` 更新状态。

```
[START] → [submit] → [review] → approve          → [execute] → [END]
                              → approve_modified  → [execute] → [END]  (金额被修改)
                              → reject            → [cancel]  → [END]
```

### 完整代码

```python
"""
子场景 3：审批带修改
- 审批人可以直接批准、拒绝，或修改金额后批准
- 修改后的金额通过 Command(update=...) 写入状态
- 执行节点使用修改后的金额
"""

from typing import Literal, Optional, TypedDict
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


class EditableApprovalState(TypedDict):
    """可编辑审批状态"""
    request_id: str
    requester: str
    action_details: str
    amount: float             # 原始金额
    approved_amount: float    # 批准金额（可能被修改）
    status: str
    reviewer: Optional[str]
    comment: Optional[str]
    modified: bool            # 是否被修改过


def submit_editable(state: EditableApprovalState):
    """提交申请"""
    print(f"\n[提交] {state['request_id']} | {state['requester']} | ¥{state['amount']:.2f}")
    return {"status": "pending", "approved_amount": state["amount"], "modified": False}


def review_editable(state: EditableApprovalState) -> Command[Literal["execute", "cancel"]]:
    """审批节点：支持 approve / reject / modify_approve 三种操作"""

    decision = interrupt({
        "type": "editable_approval",
        "request_id": state["request_id"],
        "requester": state["requester"],
        "action_details": state["action_details"],
        "amount": state["amount"],
        "instruction": (
            "请审批此申请，支持以下操作：\n"
            "  1. {'action': 'approve'} - 直接批准\n"
            "  2. {'action': 'reject', 'comment': '原因'} - 拒绝\n"
            "  3. {'action': 'modify_approve', 'new_amount': 金额} - 修改金额后批准"
        ),
    })

    # 解析决策
    if not isinstance(decision, dict):
        decision = {"action": str(decision)}

    action = decision.get("action", "reject")
    reviewer = decision.get("reviewer", "审批人")
    comment = decision.get("comment", "")

    if action == "approve":
        # 直接批准，金额不变
        return Command(
            goto="execute",
            update={
                "status": "approved",
                "reviewer": reviewer,
                "comment": comment or "同意",
                "approved_amount": state["amount"],
                "modified": False,
            },
        )
    elif action == "modify_approve":
        # 修改金额后批准
        new_amount = decision.get("new_amount", state["amount"])
        original = state["amount"]
        return Command(
            goto="execute",
            update={
                "status": "approved_modified",
                "reviewer": reviewer,
                "comment": comment or f"金额从 ¥{original:.2f} 调整为 ¥{new_amount:.2f}",
                "approved_amount": new_amount,
                "modified": True,
            },
        )
    else:
        # 拒绝
        return Command(
            goto="cancel",
            update={
                "status": "rejected",
                "reviewer": reviewer,
                "comment": comment or "不予批准",
            },
        )


def execute_editable(state: EditableApprovalState):
    """执行报销（使用批准后的金额）"""
    print(f"\n[执行] 报销 {state['request_id']}")
    print(f"  审批人: {state['reviewer']}")
    print(f"  意见: {state['comment']}")
    if state["modified"]:
        print(f"  原始金额: ¥{state['amount']:.2f}")
        print(f"  批准金额: ¥{state['approved_amount']:.2f} (已修改)")
    else:
        print(f"  批准金额: ¥{state['approved_amount']:.2f}")
    return {"status": "completed"}


def cancel_editable(state: EditableApprovalState):
    """取消"""
    print(f"\n[拒绝] {state['request_id']} | {state.get('comment', '')}")
    return {"status": "cancelled"}


# 构建图
builder = StateGraph(EditableApprovalState)
```

### 关键点

1. resume 值是结构化字典，通过 `action` 字段区分三种操作
2. `modify_approve` 操作在 `Command(update=...)` 中同时更新 `approved_amount` 和 `modified` 标记
3. 执行节点根据 `modified` 标记展示不同信息
4. 原始金额 `amount` 保持不变，修改后的金额存在 `approved_amount` 中，方便审计追溯

## 三个场景对比

| 特性 | 场景 1：基础审批 | 场景 2：多级审批 | 场景 3：审批带修改 |
|------|-----------------|-----------------|-------------------|
| 审批级别 | 单级 | 多级（按金额） | 单级 |
| 中断次数 | 1 次 | 1-2 次 | 1 次 |
| 路由方式 | approve/reject | approve/reject + 金额判断 | approve/modify/reject |
| 状态修改 | 只改 status | 累积多级审批信息 | 可修改业务数据 |
| Command 用法 | goto + update | goto + update（多次） | goto + update（含业务数据） |
| 适用场景 | 简单审批 | 分级权限管理 | 需要灵活调整的审批 |

## 核心模式总结

所有审批工作流都遵循同一个模式：

```python
# 1. 在审批节点中使用 interrupt() 暂停，传递审批信息
decision = interrupt({"type": "...", "details": "..."})

# 2. 解析人类决策
action = decision.get("action")

# 3. 根据决策使用 Command 路由 + 更新状态
if action == "approve":
    return Command(goto="execute", update={"status": "approved"})
elif action == "modify_approve":
    return Command(goto="execute", update={"amount": new_amount})
else:
    return Command(goto="cancel", update={"status": "rejected"})
```

掌握这个模式，就能应对绝大多数审批场景。
