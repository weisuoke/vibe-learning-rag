# OpenClaw 安装与配置 - 核心概念 3：配置文件

**核心概念：** ~/.openclaw/ 目录结构、配置文件格式和管理

本文档深入讲解 OpenClaw 配置文件的结构、格式和管理方法。

## 1. ~/.openclaw/ 目录结构

### 1.1 标准目录结构

```
~/.openclaw/
├── settings.json          # 主配置文件
├── gateway.log            # Gateway 日志文件
├── sessions/              # 会话数据目录
│   ├── session-001.jsonl
│   ├── session-002.jsonl
│   └── ...
├── cache/                 # 缓存目录
│   ├── responses/
│   └── embeddings/
└── plugins/               # 插件目录
    ├── plugin-a/
    └── plugin-b/
```

### 1.2 目录创建

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

/**
 * 创建 OpenClaw 配置目录结构
 */
function createOpenClawDirectories(): void {
  const baseDir = path.join(os.homedir(), '.openclaw');

  const directories = [
    baseDir,
    path.join(baseDir, 'sessions'),
    path.join(baseDir, 'cache'),
    path.join(baseDir, 'cache', 'responses'),
    path.join(baseDir, 'cache', 'embeddings'),
    path.join(baseDir, 'plugins')
  ];

  for (const dir of directories) {
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true, mode: 0o700 });
      console.log(`✅ 创建目录: ${dir}`);
    }
  }

  console.log('\n✅ OpenClaw 目录结构创建完成');
}

createOpenClawDirectories();
```

## 3. 配置文件读取与验证

### 3.1 配置读取器

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

/**
 * OpenClaw 完整配置接口
 */
interface OpenClawConfig {
  gateway: GatewayConfig;
  channels: {
    [key: string]: ChannelConfig;
  };
  cache?: CacheConfig;
  logging?: LoggingConfig;
}

/**
 * 配置读取器类
 */
class ConfigLoader {
  private configPath: string;
  private config: OpenClawConfig | null = null;

  constructor() {
    this.configPath = path.join(os.homedir(), '.openclaw', 'settings.json');
  }

  /**
   * 加载配置文件
   */
  load(): OpenClawConfig {
    // 1. 检查配置文件是否存在
    if (!fs.existsSync(this.configPath)) {
      throw new Error(`配置文件不存在: ${this.configPath}`);
    }

    // 2. 读取配置文件
    const configData = fs.readFileSync(this.configPath, 'utf-8');

    // 3. 解析 JSON
    try {
      this.config = JSON.parse(configData);
    } catch (error) {
      throw new Error(`配置文件格式错误: ${error}`);
    }

    // 4. 验证配置
    this.validate();

    // 5. 应用默认值
    this.applyDefaults();

    return this.config!;
  }

  /**
   * 验证配置
   */
  private validate(): void {
    if (!this.config) {
      throw new Error('配置未加载');
    }

    // 验证 Gateway 配置
    if (!this.config.gateway) {
      throw new Error('缺少 gateway 配置');
    }

    if (!this.config.gateway.port || this.config.gateway.port < 1024 || this.config.gateway.port > 65535) {
      throw new Error('gateway.port 必须在 1024-65535 之间');
    }

    // 验证通道配置
    if (!this.config.channels || Object.keys(this.config.channels).length === 0) {
      throw new Error('至少需要配置一个通道');
    }

    // 验证每个通道
    for (const [channelName, channelConfig] of Object.entries(this.config.channels)) {
      if (channelConfig.enabled && !channelConfig.apiKey) {
        throw new Error(`通道 ${channelName} 已启用但缺少 apiKey`);
      }

      if (channelConfig.temperature < 0 || channelConfig.temperature > 1) {
        throw new Error(`通道 ${channelName} 的 temperature 必须在 0-1 之间`);
      }
    }

    console.log('✅ 配置验证通过');
  }

  /**
   * 应用默认值
   */
  private applyDefaults(): void {
    if (!this.config) return;

    // Gateway 默认值
    this.config.gateway.host = this.config.gateway.host || 'localhost';
    this.config.gateway.mode = this.config.gateway.mode || 'development';
    this.config.gateway.logLevel = this.config.gateway.logLevel || 'info';

    // 缓存默认值
    if (!this.config.cache) {
      this.config.cache = {
        enabled: false,
        ttl: 3600,
        maxSize: '100MB'
      };
    }

    // 日志默认值
    if (!this.config.logging) {
      this.config.logging = {
        file: 'gateway.log',
        maxSize: '10MB',
        maxFiles: 5,
        format: 'json'
      };
    }
  }

  /**
   * 保存配置
   */
  save(config: OpenClawConfig): void {
    const configData = JSON.stringify(config, null, 2);
    fs.writeFileSync(this.configPath, configData, { mode: 0o600 });
    console.log(`✅ 配置已保存到: ${this.configPath}`);
  }

  /**
   * 获取特定通道配置
   */
  getChannelConfig(channelName: string): ChannelConfig | null {
    if (!this.config) {
      throw new Error('配置未加载');
    }

    return this.config.channels[channelName] || null;
  }

  /**
   * 获取启用的通道列表
   */
  getEnabledChannels(): string[] {
    if (!this.config) {
      throw new Error('配置未加载');
    }

    return Object.entries(this.config.channels)
      .filter(([_, config]) => config.enabled)
      .map(([name, _]) => name);
  }
}

// 使用示例
const loader = new ConfigLoader();
const config = loader.load();

console.log('Gateway 端口:', config.gateway.port);
console.log('启用的通道:', loader.getEnabledChannels());
```

### 3.2 配置验证器

```typescript
/**
 * 配置验证器
 */
class ConfigValidator {
  /**
   * 验证 API Key 格式
   */
  static validateApiKey(apiKey: string, provider: string): boolean {
    const patterns: { [key: string]: RegExp } = {
      claude: /^sk-ant-api03-[A-Za-z0-9_-]{95}$/,
      gpt: /^sk-[A-Za-z0-9]{48}$/,
      gemini: /^[A-Za-z0-9_-]{39}$/
    };

    const pattern = patterns[provider];
    if (!pattern) {
      console.warn(`未知的 provider: ${provider}`);
      return true; // 未知 provider，跳过验证
    }

    return pattern.test(apiKey);
  }

  /**
   * 验证端口号
   */
  static validatePort(port: number): boolean {
    return Number.isInteger(port) && port >= 1024 && port <= 65535;
  }

  /**
   * 验证温度参数
   */
  static validateTemperature(temperature: number): boolean {
    return temperature >= 0 && temperature <= 1;
  }

  /**
   * 验证文件大小格式
   */
  static validateSize(size: string): boolean {
    return /^\d+(\.\d+)?(KB|MB|GB)$/.test(size);
  }

  /**
   * 完整配置验证
   */
  static validate(config: OpenClawConfig): { valid: boolean; errors: string[] } {
    const errors: string[] = [];

    // 验证 Gateway
    if (!this.validatePort(config.gateway.port)) {
      errors.push(`无效的端口号: ${config.gateway.port}`);
    }

    // 验证通道
    for (const [name, channelConfig] of Object.entries(config.channels)) {
      if (channelConfig.enabled) {
        if (!channelConfig.apiKey) {
          errors.push(`通道 ${name} 缺少 apiKey`);
        } else if (!this.validateApiKey(channelConfig.apiKey, name)) {
          errors.push(`通道 ${name} 的 apiKey 格式无效`);
        }

        if (!this.validateTemperature(channelConfig.temperature)) {
          errors.push(`通道 ${name} 的 temperature 无效`);
        }
      }
    }

    // 验证缓存
    if (config.cache && config.cache.maxSize) {
      if (!this.validateSize(config.cache.maxSize)) {
        errors.push(`无效的缓存大小: ${config.cache.maxSize}`);
      }
    }

    // 验证日志
    if (config.logging && config.logging.maxSize) {
      if (!this.validateSize(config.logging.maxSize)) {
        errors.push(`无效的日志大小: ${config.logging.maxSize}`);
      }
    }

    return {
      valid: errors.length === 0,
      errors
    };
  }
}

// 使用示例
const validation = ConfigValidator.validate(config);
if (!validation.valid) {
  console.error('配置验证失败:');
  validation.errors.forEach(error => console.error(`  - ${error}`));
  process.exit(1);
}
```

## 4. 配置优先级

### 4.1 多层配置系统

```typescript
/**
 * 配置优先级（从高到低）：
 * 1. 命令行参数
 * 2. 环境变量
 * 3. 项目配置文件（.openclawrc）
 * 4. 用户配置文件（~/.openclaw/settings.json）
 * 5. 默认值
 */
class ConfigManager {
  private config: OpenClawConfig;

  constructor() {
    this.config = this.loadConfig();
  }

  /**
   * 加载配置（按优先级合并）
   */
  private loadConfig(): OpenClawConfig {
    // 1. 加载默认配置
    let config = this.getDefaultConfig();

    // 2. 加载用户配置
    const userConfig = this.loadUserConfig();
    if (userConfig) {
      config = this.mergeConfig(config, userConfig);
    }

    // 3. 加载项目配置
    const projectConfig = this.loadProjectConfig();
    if (projectConfig) {
      config = this.mergeConfig(config, projectConfig);
    }

    // 4. 应用环境变量
    config = this.applyEnvVars(config);

    // 5. 应用命令行参数
    config = this.applyCLIArgs(config);

    return config;
  }

  /**
   * 获取默认配置
   */
  private getDefaultConfig(): OpenClawConfig {
    return {
      gateway: {
        port: 3000,
        host: 'localhost',
        mode: 'development',
        logLevel: 'info'
      },
      channels: {}
    };
  }

  /**
   * 加载用户配置
   */
  private loadUserConfig(): OpenClawConfig | null {
    const configPath = path.join(os.homedir(), '.openclaw', 'settings.json');
    if (!fs.existsSync(configPath)) {
      return null;
    }

    try {
      const data = fs.readFileSync(configPath, 'utf-8');
      return JSON.parse(data);
    } catch (error) {
      console.warn(`加载用户配置失败: ${error}`);
      return null;
    }
  }

  /**
   * 加载项目配置
   */
  private loadProjectConfig(): Partial<OpenClawConfig> | null {
    const configPath = path.join(process.cwd(), '.openclawrc');
    if (!fs.existsSync(configPath)) {
      return null;
    }

    try {
      const data = fs.readFileSync(configPath, 'utf-8');
      return JSON.parse(data);
    } catch (error) {
      console.warn(`加载项目配置失败: ${error}`);
      return null;
    }
  }

  /**
   * 应用环境变量
   */
  private applyEnvVars(config: OpenClawConfig): OpenClawConfig {
    // Gateway 端口
    if (process.env.OPENCLAW_PORT) {
      config.gateway.port = parseInt(process.env.OPENCLAW_PORT);
    }

    // Gateway 主机
    if (process.env.OPENCLAW_HOST) {
      config.gateway.host = process.env.OPENCLAW_HOST;
    }

    // 日志级别
    if (process.env.OPENCLAW_LOG_LEVEL) {
      config.gateway.logLevel = process.env.OPENCLAW_LOG_LEVEL as any;
    }

    // Claude API Key
    if (process.env.CLAUDE_API_KEY) {
      config.channels.claude = config.channels.claude || {} as any;
      config.channels.claude.apiKey = process.env.CLAUDE_API_KEY;
    }

    // GPT API Key
    if (process.env.OPENAI_API_KEY) {
      config.channels.gpt = config.channels.gpt || {} as any;
      config.channels.gpt.apiKey = process.env.OPENAI_API_KEY;
    }

    return config;
  }

  /**
   * 应用命令行参数
   */
  private applyCLIArgs(config: OpenClawConfig): OpenClawConfig {
    const args = process.argv.slice(2);

    for (let i = 0; i < args.length; i++) {
      const arg = args[i];

      if (arg === '--port' && args[i + 1]) {
        config.gateway.port = parseInt(args[i + 1]);
        i++;
      } else if (arg === '--host' && args[i + 1]) {
        config.gateway.host = args[i + 1];
        i++;
      } else if (arg === '--log-level' && args[i + 1]) {
        config.gateway.logLevel = args[i + 1] as any;
        i++;
      }
    }

    return config;
  }

  /**
   * 合并配置
   */
  private mergeConfig(base: OpenClawConfig, override: Partial<OpenClawConfig>): OpenClawConfig {
    return {
      gateway: { ...base.gateway, ...override.gateway },
      channels: { ...base.channels, ...override.channels },
      cache: { ...base.cache, ...override.cache },
      logging: { ...base.logging, ...override.logging }
    };
  }

  /**
   * 获取配置
   */
  getConfig(): OpenClawConfig {
    return this.config;
  }
}

// 使用示例
const manager = new ConfigManager();
const config = manager.getConfig();

console.log('最终配置:');
console.log(`  端口: ${config.gateway.port}`);
console.log(`  主机: ${config.gateway.host}`);
console.log(`  日志级别: ${config.gateway.logLevel}`);
```

### 5.2 敏感数据加密

```typescript
import * as crypto from 'crypto';

/**
 * 配置加密器
 */
class ConfigEncryptor {
  private algorithm = 'aes-256-gcm';
  private key: Buffer;

  constructor(password: string) {
    // 从密码派生密钥
    this.key = crypto.scryptSync(password, 'salt', 32);
  }

  /**
   * 加密配置
   */
  encrypt(config: OpenClawConfig): string {
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv(this.algorithm, this.key, iv);

    const configData = JSON.stringify(config);
    let encrypted = cipher.update(configData, 'utf8', 'hex');
    encrypted += cipher.final('hex');

    const authTag = cipher.getAuthTag();

    // 返回：iv + authTag + encrypted
    return iv.toString('hex') + authTag.toString('hex') + encrypted;
  }

  /**
   * 解密配置
   */
  decrypt(encryptedData: string): OpenClawConfig {
    // 解析：iv + authTag + encrypted
    const iv = Buffer.from(encryptedData.slice(0, 32), 'hex');
    const authTag = Buffer.from(encryptedData.slice(32, 64), 'hex');
    const encrypted = encryptedData.slice(64);

    const decipher = crypto.createDecipheriv(this.algorithm, this.key, iv);
    decipher.setAuthTag(authTag);

    let decrypted = decipher.update(encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');

    return JSON.parse(decrypted);
  }
}

// 使用示例
const password = process.env.OPENCLAW_PASSWORD || 'default-password';
const encryptor = new ConfigEncryptor(password);

// 加密配置
const encryptedConfig = encryptor.encrypt(config);
fs.writeFileSync('settings.encrypted', encryptedConfig);

// 解密配置
const decryptedConfig = encryptor.decrypt(encryptedConfig);
```

### 6.1 版本升级迁移

/**
 * 配置迁移器
 */
class ConfigMigrator {
  /**
   * 迁移配置到最新版本
   */
  migrate(oldConfig: any, fromVersion: string, toVersion: string): OpenClawConfig {
    console.log(`迁移配置: ${fromVersion} -> ${toVersion}`);

    let config = oldConfig;

    // 1.0.0 -> 1.1.0
    if (fromVersion === '1.0.0' && toVersion >= '1.1.0') {
      config = this.migrateFrom1_0_0To1_1_0(config);
    }

    // 1.1.0 -> 1.2.0
    if (fromVersion <= '1.1.0' && toVersion >= '1.2.0') {
      config = this.migrateFrom1_1_0To1_2_0(config);
    }

    return config;
  }

  /**
   * 1.0.0 -> 1.1.0 迁移
   */
  private migrateFrom1_0_0To1_1_0(config: any): any {
    console.log('  应用 1.0.0 -> 1.1.0 迁移');

    // 添加缓存配置
    if (!config.cache) {
      config.cache = {
        enabled: false,
        ttl: 3600,
        maxSize: '100MB'
      };
    }

    return config;
  }

  /**
   * 1.1.0 -> 1.2.0 迁移
   */
  private migrateFrom1_1_0To1_2_0(config: any): any {
    console.log('  应用 1.1.0 -> 1.2.0 迁移');

    // 添加日志配置
    if (!config.logging) {
      config.logging = {
        file: 'gateway.log',
        maxSize: '10MB',
        maxFiles: 5,
        format: 'json'
      };
    }

    return config;
  }
}

## 7. 配置备份与恢复

```typescript
/**
 * 配置备份管理器
 */
class ConfigBackupManager {
  private backupDir: string;

  constructor() {
    this.backupDir = path.join(os.homedir(), '.openclaw', 'backups');
    if (!fs.existsSync(this.backupDir)) {
      fs.mkdirSync(this.backupDir, { recursive: true });
    }
  }

  /**
   * 备份配置
   */
  backup(): string {
    const configPath = path.join(os.homedir(), '.openclaw', 'settings.json');
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const backupPath = path.join(this.backupDir, `settings-${timestamp}.json`);

    fs.copyFileSync(configPath, backupPath);
    console.log(`✅ 配置已备份到: ${backupPath}`);

    return backupPath;
  }

  /**
   * 恢复配置
   */
  restore(backupPath: string): void {
    const configPath = path.join(os.homedir(), '.openclaw', 'settings.json');

    // 先备份当前配置
    this.backup();

    // 恢复备份
    fs.copyFileSync(backupPath, configPath);
    console.log(`✅ 配置已恢复from: ${backupPath}`);
  }

  /**
   * 列出所有备份
   */
  listBackups(): string[] {
    const files = fs.readdirSync(this.backupDir);
    return files
      .filter(file => file.startsWith('settings-') && file.endsWith('.json'))
      .sort()
      .reverse();
  }
}

// 使用示例
const backupManager = new ConfigBackupManager();

// 备份配置
backupManager.backup();

// 列出备份
const backups = backupManager.listBackups();
console.log('可用备份:');
backups.forEach(backup => console.log(`  - ${backup}`));

// 恢复最新备份
if (backups.length > 0) {
  const latestBackup = path.join(backupManager['backupDir'], backups[0]);
  backupManager.restore(latestBackup);
}
```
