## 技巧 1：自定义 ExampleSelector 实现

### 技巧名称

**自定义 ExampleSelector 实现**：继承 BaseExampleSelector 实现特定业务逻辑的示例选择器

### 原理

LangChain 提供了 BaseExampleSelector 抽象接口，允许开发者实现自定义的示例选择逻辑。通过继承 BaseExampleSelector 并实现 `add_example()` 和 `select_examples()` 方法，可以实现任意复杂的选择策略。

**核心接口：**
```python
from langchain_core.example_selectors import BaseExampleSelector
from typing import Dict, List

class BaseExampleSelector(ABC):
    @abstractmethod
    def add_example(self, example: Dict[str, str]) -> None:
        """添加示例到存储"""
        pass

    @abstractmethod
    def select_examples(self, input_variables: Dict[str, str]) -> List[Dict]:
        """根据输入选择示例"""
        pass
```

### 代码示例

**场景：基于业务规则的示例选择器**

```python
from langchain_core.example_selectors import BaseExampleSelector
from typing import Dict, List
import re

class BusinessRuleExampleSelector(BaseExampleSelector):
    """基于业务规则的示例选择器
    
    根据输入的业务类型（如订单、支付、退款）选择对应的示例
    """
    
    def __init__(self, examples: List[Dict] = None, k: int = 3):
        """初始化选择器
        
        Args:
            examples: 初始示例列表
            k: 返回的示例数量
        """
        self.examples = examples or []
        self.k = k
        
        # 业务类型关键词映射
        self.business_keywords = {
            "order": ["订单", "下单", "购买", "order", "buy"],
            "payment": ["支付", "付款", "pay", "payment"],
            "refund": ["退款", "退货", "refund", "return"],
            "shipping": ["物流", "配送", "发货", "shipping", "delivery"],
        }
    
    def add_example(self, example: Dict[str, str]) -> None:
        """添加示例"""
        self.examples.append(example)
    
    def _detect_business_type(self, text: str) -> str:
        """检测业务类型"""
        text_lower = text.lower()
        
        for business_type, keywords in self.business_keywords.items():
            for keyword in keywords:
                if keyword in text_lower:
                    return business_type
        
        return "general"
    
    def select_examples(self, input_variables: Dict[str, str]) -> List[Dict]:
        """基于业务规则选择示例"""
        # 1. 提取输入文本
        input_text = " ".join(input_variables.values())
        
        # 2. 检测业务类型
        business_type = self._detect_business_type(input_text)
        
        # 3. 筛选相关示例
        relevant_examples = []
        for example in self.examples:
            example_text = " ".join(example.values())
            example_type = self._detect_business_type(example_text)
            
            if example_type == business_type:
                relevant_examples.append(example)
        
        # 4. 如果没有相关示例，返回通用示例
        if not relevant_examples:
            relevant_examples = [ex for ex in self.examples 
                                if self._detect_business_type(" ".join(ex.values())) == "general"]
        
        # 5. 返回 top-k 个示例
        return relevant_examples[:self.k]

# 使用示例
examples = [
    {"input": "如何创建订单？", "output": "使用 POST /api/orders 接口创建订单"},
    {"input": "如何支付订单？", "output": "使用 POST /api/payments 接口支付订单"},
    {"input": "如何申请退款？", "output": "使用 POST /api/refunds 接口申请退款"},
    {"input": "如何查询物流？", "output": "使用 GET /api/shipping/{id} 接口查询物流"},
]

selector = BusinessRuleExampleSelector(examples=examples, k=2)

# 测试
query = "我想下单购买商品"
selected = selector.select_examples({"input": query})
print(f"查询: {query}")
print(f"选择的示例: {[ex['input'] for ex in selected]}")
```

## 技巧 2：缓存优化策略

### 技巧名称

**缓存优化策略**：通过缓存 Embeddings 和选择结果，显著降低延迟和成本

### 原理

ExampleSelector 的性能瓶颈主要在两个地方：
1. **Embedding 计算**：每次查询都需要调用 Embedding API
2. **向量检索**：在大规模示例库中检索相似示例

通过缓存这两个环节的结果，可以显著提升性能：
- **Embedding 缓存**：缓存查询的 Embedding，避免重复计算
- **结果缓存**：缓存选择结果，避免重复检索

### 代码示例

**方案 1：Embedding 缓存**

```python
from functools import lru_cache
from langchain_openai import OpenAIEmbeddings
from typing import List

class CachedEmbeddings:
    """带缓存的 Embeddings 包装器"""
    
    def __init__(self, embeddings: OpenAIEmbeddings, cache_size: int = 1000):
        """初始化
        
        Args:
            embeddings: 原始 Embeddings 对象
            cache_size: 缓存大小
        """
        self.embeddings = embeddings
        self.cache_size = cache_size
    
    @lru_cache(maxsize=1000)
    def embed_query(self, text: str) -> List[float]:
        """缓存查询 Embeddings"""
        return self.embeddings.embed_query(text)
    
    def embed_documents(self, texts: List[str]) -> List[List[float]]:
        """批量生成 Embeddings（不缓存）"""
        return self.embeddings.embed_documents(texts)

# 使用示例
from langchain_core.example_selectors import SemanticSimilarityExampleSelector
from langchain_chroma import Chroma

# 创建缓存 Embeddings
base_embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
cached_embeddings = CachedEmbeddings(base_embeddings, cache_size=1000)

# 创建选择器
selector = SemanticSimilarityExampleSelector.from_examples(
    examples=examples,
    embeddings=cached_embeddings,  # 使用缓存 Embeddings
    vectorstore_cls=Chroma,
    k=5
)

# 测试缓存效果
import time

query = "如何实现用户登录？"

# 第一次查询（无缓存）
start = time.time()
result1 = selector.select_examples({"input": query})
time1 = time.time() - start

# 第二次查询（有缓存）
start = time.time()
result2 = selector.select_examples({"input": query})
time2 = time.time() - start

print(f"第一次查询时间: {time1:.3f}s")
print(f"第二次查询时间: {time2:.3f}s")
```

**方案 2：结果缓存**

```python
from functools import lru_cache
from langchain_core.example_selectors import BaseExampleSelector
from typing import Dict, List

class CachedExampleSelector(BaseExampleSelector):
    """带结果缓存的示例选择器"""
    
    def __init__(self, selector: BaseExampleSelector, cache_size: int = 100):
        """初始化
        
        Args:
            selector: 原始选择器
            cache_size: 缓存大小
        """
        self.selector = selector
        self.cache_size = cache_size
    
    def add_example(self, example: Dict[str, str]) -> None:
        """添加示例（清除缓存）"""
        self.selector.add_example(example)
        self.select_examples.cache_clear()
    
    @lru_cache(maxsize=100)
    def select_examples(self, input_variables: Dict[str, str]) -> List[Dict]:
        """缓存选择结果
        
        注意：Dict 不能直接作为缓存键，需要转换为 tuple
        """
        # 将 Dict 转换为可哈希的 tuple
        input_tuple = tuple(sorted(input_variables.items()))
        return self._select_examples_impl(input_tuple)
    
    def _select_examples_impl(self, input_tuple: tuple) -> List[Dict]:
        """实际的选择逻辑"""
        input_dict = dict(input_tuple)
        return self.selector.select_examples(input_dict)

# 使用示例
base_selector = SemanticSimilarityExampleSelector.from_examples(
    examples=examples,
    embeddings=OpenAIEmbeddings(),
    vectorstore_cls=Chroma,
    k=5
)

cached_selector = CachedExampleSelector(base_selector, cache_size=100)

# 测试缓存效果
query = "如何实现用户登录？"

# 第一次查询（无缓存）
start = time.time()
result1 = cached_selector.select_examples({"input": query})
time1 = time.time() - start

# 第二次查询（有缓存）
start = time.time()
result2 = cached_selector.select_examples({"input": query})
time2 = time.time() - start

print(f"第一次查询时间: {time1:.3f}s")
print(f"第二次查询时间: {time2:.3f}s")
print(f"性能提升: {(time1 - time2) / time1 * 100:.1f}%")
```

## 技巧 3：批量处理技巧

### 技巧名称

**批量处理技巧**：通过批量处理多个查询，提高吞吐量和资源利用率

### 原理

在实际应用中，经常需要处理大量查询（如批量分类、批量问答）。如果逐个处理，会导致：
1. **API 调用次数过多**：每个查询都调用一次 Embedding API
2. **资源利用率低**：无法充分利用批量处理的优势
3. **总耗时长**：串行处理导致总时间过长

通过批量处理，可以：
- 减少 API 调用次数（批量生成 Embeddings）
- 提高资源利用率（并行处理）
- 降低总耗时（批量操作更快）

### 代码示例

**方案 1：批量 Embedding 生成**

```python
from langchain_core.example_selectors import SemanticSimilarityExampleSelector
from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma
from typing import List, Dict

def batch_select_examples(
    queries: List[str],
    selector: SemanticSimilarityExampleSelector
) -> List[List[Dict]]:
    """批量选择示例
    
    Args:
        queries: 查询列表
        selector: 示例选择器
    
    Returns:
        每个查询对应的示例列表
    """
    # 1. 批量生成 Embeddings（一次 API 调用）
    embeddings = selector.vectorstore.embeddings
    query_embeddings = embeddings.embed_documents(queries)
    
    # 2. 批量检索
    results = []
    for query, embedding in zip(queries, query_embeddings):
        # 使用预计算的 embedding 进行检索
        docs = selector.vectorstore.similarity_search_by_vector(
            embedding,
            k=selector.k
        )
        examples = [doc.metadata for doc in docs]
        results.append(examples)
    
    return results

# 使用示例
queries = [
    "如何实现用户登录？",
    "如何实现文件上传？",
    "如何实现数据分页？",
]

selector = SemanticSimilarityExampleSelector.from_examples(
    examples=examples,
    embeddings=OpenAIEmbeddings(),
    vectorstore_cls=Chroma,
    k=3
)

# 批量处理
import time

start = time.time()
results = batch_select_examples(queries, selector)
batch_time = time.time() - start

# 对比：逐个处理
start = time.time()
individual_results = [
    selector.select_examples({"input": q})
    for q in queries
]
individual_time = time.time() - start

print(f"批量处理时间: {batch_time:.3f}s")
print(f"逐个处理时间: {individual_time:.3f}s")
print(f"性能提升: {(individual_time - batch_time) / individual_time * 100:.1f}%")
```

```python
import asyncio

async def async_batch_select_examples(
```

## 技巧 4：混合策略高级应用

### 技巧名称

**混合策略高级应用**：结合多种选择策略，实现更智能的示例选择

### 原理

单一的选择策略往往无法满足复杂场景的需求。混合策略通过组合多种策略，可以实现：
1. **语义相似度 + 长度限制**：既相关又不超限
2. **语义相似度 + 多样性（MMR）**：既相关又多样
3. **业务规则 + 语义相似度**：既符合规则又相关
4. **动态策略切换**：根据场景自动选择策略

### 代码示例

**场景：智能客服系统的混合策略**

```python
from langchain_core.example_selectors import BaseExampleSelector
from typing import Dict, List, Callable
import time

class SmartExampleSelector(BaseExampleSelector):
    """智能示例选择器：根据场景自动选择策略"""
    
    def __init__(
        self,
        semantic_selector,
        mmr_selector,
        length_selector,
        business_selector
    ):
        self.semantic_selector = semantic_selector
        self.mmr_selector = mmr_selector
        self.length_selector = length_selector
        self.business_selector = business_selector
    
    def add_example(self, example: Dict[str, str]) -> None:
        """添加示例到所有选择器"""
        self.semantic_selector.add_example(example)
        self.mmr_selector.add_example(example)
        self.length_selector.add_example(example)
        self.business_selector.add_example(example)
    
    def _detect_scenario(self, input_variables: Dict[str, str]) -> str:
        """检测场景类型"""
        text = " ".join(input_variables.values()).lower()
        
        # 场景检测逻辑
        if "紧急" in text or "urgent" in text:
            return "urgent"
        elif len(text) > 200:
            return "long_query"
        elif "多个" in text or "several" in text:
            return "diverse"
        else:
            return "normal"
    
    def select_examples(self, input_variables: Dict[str, str]) -> List[Dict]:
        """根据场景选择策略"""
        scenario = self._detect_scenario(input_variables)
        
        if scenario == "urgent":
            # 紧急场景：使用业务规则选择器（快速）
            return self.business_selector.select_examples(input_variables)
        
        elif scenario == "long_query":
            # 长查询：使用长度限制选择器
            return self.length_selector.select_examples(input_variables)
        
        elif scenario == "diverse":
            # 需要多样性：使用 MMR 选择器
            return self.mmr_selector.select_examples(input_variables)
        
        else:
            # 正常场景：使用语义相似度选择器
            return self.semantic_selector.select_examples(input_variables)

# 使用示例
smart_selector = SmartExampleSelector(
    semantic_selector=semantic_selector,
    mmr_selector=mmr_selector,
    length_selector=length_selector,
    business_selector=business_selector
)

# 测试不同场景
test_cases = [
    "紧急：系统崩溃了！",
    "我想了解多个不同的支付方式，包括信用卡、支付宝、微信支付等",
    "如何实现用户登录？",
]

for query in test_cases:
    selected = smart_selector.select_examples({"input": query})
    print(f"查询: {query}")
    print(f"选择的示例: {[ex['input'] for ex in selected]}\n")
```

### 适用场景

1. **复杂业务场景**：需要根据不同情况选择不同策略
2. **多租户系统**：不同租户需要不同的选择策略
3. **A/B 测试**：需要对比不同策略的效果
4. **自适应系统**：根据反馈自动调整策略

### 技巧名称

**生产环境监控与调优**：通过监控和分析，持续优化 ExampleSelector 的性能和效果

### 原理

在生产环境中，ExampleSelector 的性能和效果会随着时间和数据变化而变化。通过监控和调优，可以：
1. **发现性能瓶颈**：识别慢查询和资源消耗
2. **评估选择质量**：分析示例相关性和效果
3. **优化配置参数**：根据数据调整 k 值、max_length 等
4. **持续改进**：基于反馈不断优化

```python
import time
import json
from typing import Dict, List
from collections import defaultdict

class MonitoredExampleSelector:
    """带监控的示例选择器"""
    
    def __init__(self, selector, log_file: str = "selector_metrics.json"):
        self.selector = selector
        self.log_file = log_file
        self.metrics = defaultdict(list)
    
    def select_examples(self, input_variables: Dict[str, str]) -> List[Dict]:
        """选择示例并记录指标"""
        start_time = time.time()
        
        # 选择示例
        examples = self.selector.select_examples(input_variables)
        
        # 记录指标
        elapsed_time = time.time() - start_time
        self.metrics["response_times"].append(elapsed_time)
        self.metrics["example_counts"].append(len(examples))
        self.metrics["queries"].append(input_variables.get("input", ""))
        
        # 定期保存指标
        if len(self.metrics["response_times"]) % 100 == 0:
            self.save_metrics()
        
        return examples
    
    def save_metrics(self):
        """保存指标到文件"""
        with open(self.log_file, 'w') as f:
            json.dump({
                "avg_response_time": sum(self.metrics["response_times"]) / len(self.metrics["response_times"]),
                "avg_example_count": sum(self.metrics["example_counts"]) / len(self.metrics["example_counts"]),
                "total_queries": len(self.metrics["queries"]),
                "slow_queries": [
                    q for q, t in zip(self.metrics["queries"], self.metrics["response_times"])
                    if t > 0.5
                ]
            }, f, indent=2)
    
    def analyze_performance(self):
        """分析性能"""
        if not self.metrics["response_times"]:
            return "No data"
        
        response_times = self.metrics["response_times"]
        example_counts = self.metrics["example_counts"]
        
        return {
            "avg_response_time": sum(response_times) / len(response_times),
            "max_response_time": max(response_times),
            "min_response_time": min(response_times),
            "avg_example_count": sum(example_counts) / len(example_counts),
            "total_queries": len(response_times),
        }

# 使用示例
monitored_selector = MonitoredExampleSelector(selector)

# 处理查询
for query in queries:
    examples = monitored_selector.select_examples({"input": query})

# 分析性能
performance = monitored_selector.analyze_performance()
print(json.dumps(performance, indent=2))
```

```python
class AutoTuningSelector:
    """自动调优的示例选择器"""
    
    def __init__(self, selector, target_response_time: float = 0.2):
        self.selector = selector
        self.target_response_time = target_response_time
        self.current_k = selector.k
        self.response_times = []
    
    def select_examples(self, input_variables: Dict[str, str]) -> List[Dict]:
        """选择示例并自动调优"""
        start_time = time.time()
        examples = self.selector.select_examples(input_variables)
        elapsed_time = time.time() - start_time
        
        # 记录响应时间
        self.response_times.append(elapsed_time)
        
        # 每 100 次查询后调优
        if len(self.response_times) >= 100:
            self.tune()
            self.response_times = []
        
        return examples
    
    def tune(self):
        """自动调优 k 值"""
        avg_time = sum(self.response_times) / len(self.response_times)
        
        if avg_time > self.target_response_time:
            # 响应时间过长，减少 k 值
            new_k = max(1, self.current_k - 1)
            print(f"响应时间过长 ({avg_time:.3f}s)，减少 k 值: {self.current_k} -> {new_k}")
            self.current_k = new_k
            self.selector.k = new_k
        
        elif avg_time < self.target_response_time * 0.5:
            # 响应时间很短，可以增加 k 值
            new_k = self.current_k + 1
            print(f"响应时间较短 ({avg_time:.3f}s)，增加 k 值: {self.current_k} -> {new_k}")
            self.current_k = new_k
            self.selector.k = new_k

# 使用示例
auto_tuning_selector = AutoTuningSelector(selector, target_response_time=0.2)

# 处理大量查询，系统会自动调优
for query in large_query_list:
    examples = auto_tuning_selector.select_examples({"input": query})
```
