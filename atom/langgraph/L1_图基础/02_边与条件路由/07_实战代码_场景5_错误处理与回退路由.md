## 完整代码

```python
"""
LangGraph 实战代码 - 场景5：错误处理与回退路由
演示：构建健壮的 API 调用系统
"""

from langgraph.graph import StateGraph, START, END
from typing import Literal
from typing_extensions import TypedDict
import random


# ===== 1. 定义状态 =====
class APIState(TypedDict):
    """API 调用状态"""
    request_url: str        # 请求 URL
    request_data: dict      # 请求数据
    response: dict          # 响应数据
    error: str              # 错误信息
    retry_count: int        # 重试次数
    fallback_used: bool     # 是否使用降级
    status: str             # 状态：success/error/fallback
    log: list[str]          # 操作日志


# ===== 2. 定义节点函数 =====
def call_primary_api(state: APIState) -> dict:
    """调用主 API"""
    print(f"[主 API] 调用 API: {state['request_url']}")
    print(f"[主 API] 重试次数: {state.get('retry_count', 0)}")

    # 模拟 API 调用（70% 成功率）
    success = random.random() > 0.3

    if success:
        print("[主 API] 调用成功")
        return {
            "response": {"status": "success", "data": "主 API 返回的数据"},
            "error": "",
            "status": "success",
            "log": state.get("log", []) + ["主 API 调用成功"]
        }
    else:
        print("[主 API] 调用失败")
        return {
            "response": {},
            "error": "主 API 调用失败：连接超时",
            "status": "error",
            "retry_count": state.get("retry_count", 0) + 1,
            "log": state.get("log", []) + ["主 API 调用失败"]
        }


def call_fallback_api(state: APIState) -> dict:
    """调用降级 API"""
    print("[降级 API] 主 API 失败，使用降级 API...")

    # 模拟降级 API 调用（100% 成功率）
    print("[降级 API] 调用成功")

    return {
        "response": {"status": "success", "data": "降级 API 返回的数据"},
        "error": "",
        "status": "fallback",
        "fallback_used": True,
        "log": state.get("log", []) + ["降级 API 调用成功"]
    }


def handle_error(state: APIState) -> dict:
    """处理错误"""
    print("[错误处理] 处理错误...")
    print(f"[错误处理] 错误信息: {state['error']}")

    return {
        "status": "error",
        "log": state.get("log", []) + ["错误已记录"]
    }


# ===== 3. 定义路由函数 =====
def route_after_api_call(state: APIState) -> Literal["retry", "fallback", "error", END]:
    """API 调用后的路由决策"""
    status = state["status"]
    retry_count = state.get("retry_count", 0)

    print(f"[路由决策] 状态: {status}, 重试次数: {retry_count}")

    # 成功，结束
    if status == "success":
        print("[路由决策] → END (成功)")
        return END

    # 失败，检查重试次数
    if status == "error":
        # 重试次数未达上限，重试
        if retry_count < 3:
            print("[路由决策] → retry (重试)")
            return "retry"
        # 重试次数达上限，降级
        else:
            print("[路由决策] → fallback (降级)")
            return "fallback"

    # 降级成功，结束
    if status == "fallback":
        print("[路由决策] → END (降级成功)")
        return END

    # 其他情况，错误处理
    print("[路由决策] → error (错误处理)")
    return "error"


# ===== 4. 构建图 =====
def create_error_handling_graph():
    """创建错误处理图"""
    # 创建状态图
    graph = StateGraph(APIState)

    # 添加节点
    graph.add_node("primary_api", call_primary_api)
    graph.add_node("fallback_api", call_fallback_api)
    graph.add_node("error_handler", handle_error)

    # 添加边
    graph.add_edge(START, "primary_api")

    # 添加条件边（错误处理路由）
    graph.add_conditional_edges(
        "primary_api",
        route_after_api_call,
        {
            "retry": "primary_api",      # 重试：循环回自己
            "fallback": "fallback_api",  # 降级：使用降级 API
            "error": "error_handler",    # 错误：错误处理
            END: END                     # 成功：结束
        }
    )

    # 降级 API 后结束
    graph.add_edge("fallback_api", END)

    # 错误处理后结束
    graph.add_edge("error_handler", END)

    # 编译图
    app = graph.compile()

    return app


# ===== 5. 测试不同场景 =====
def test_api_call():
    """测试 API 调用"""
    print("=" * 60)
    print("测试场景：API 调用（自动重试与降级）")
    print("=" * 60)
    print()

    app = create_error_handling_graph()

    initial_state = {
        "request_url": "https://api.example.com/data",
        "request_data": {"query": "test"},
        "response": {},
        "error": "",
        "retry_count": 0,
        "fallback_used": False,
        "status": "",
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("最终结果：")
    print(f"  状态: {result['status']}")
    print(f"  重试次数: {result['retry_count']}")
    print(f"  是否降级: {result['fallback_used']}")
    print(f"  响应: {result['response']}")
    print()
    print("操作日志：")
    for i, log_entry in enumerate(result['log'], 1):
        print(f"  {i}. {log_entry}")
    print()


# ===== 6. 主函数 =====
def main():
    """主函数"""
    print("=" * 60)
    print("LangGraph 实战 - 场景5：错误处理与回退路由")
    print("=" * 60)
    print()

    # 测试多次以观察不同结果
    for i in range(3):
        print(f"\n{'=' * 60}")
        print(f"第 {i+1} 次测试")
        print(f"{'=' * 60}\n")
        test_api_call()


if __name__ == "__main__":
    main()
```

## 代码详解

### 1. 循环路由机制

```python
graph.add_conditional_edges(
    "primary_api",
    route_after_api_call,
    {
        "retry": "primary_api",      # 循环回自己
        "fallback": "fallback_api",
        "error": "error_handler",
        END: END
    }
)
```

**关键点**：
- **循环路由**：`"retry": "primary_api"` 形成循环
- **退出条件**：retry_count >= 3 时退出循环
- **降级策略**：重试失败后使用降级 API

---

### 2. 路由函数设计

```python
def route_after_api_call(state: APIState) -> Literal["retry", "fallback", "error", END]:
    """API 调用后的路由决策"""
    status = state["status"]
    retry_count = state.get("retry_count", 0)

    # 成功，结束
    if status == "success":
        return END

    # 失败，检查重试次数
    if status == "error":
        if retry_count < 3:
            return "retry"  # 重试
        else:
            return "fallback"  # 降级

    # 降级成功，结束
    if status == "fallback":
        return END

    return "error"
```

**关键点**：
- **退出条件**：retry_count < 3
- **降级触发**：retry_count >= 3
- **状态驱动**：基于 status 和 retry_count

---

### 3. 执行流程图

```
START
  ↓
primary_api (主 API)
  ↓
route_after_api_call
  ↓
┌─────┼─────┼─────┐
↓     ↓     ↓     ↓
retry fallback error END
(重试) (降级) (错误) (成功)
↓       ↓     ↓
primary_api fallback_api error_handler
(循环)    ↓     ↓
         END   END
```

**执行路径**：
- **成功**：primary_api → END
- **重试**：primary_api → retry → primary_api → ... → END
- **降级**：primary_api → retry × 3 → fallback → fallback_api → END

## 总结

**错误处理与回退路由的特点**：
1. **自动重试**：失败后自动重试
2. **退出条件**：防止无限循环
3. **降级策略**：重试失败后使用备用方案
4. **错误记录**：记录错误日志

**适用场景**：
- API 调用重试
- 数据库查询重试
- LLM 调用重试
- 网络请求重试
- 任何需要容错的操作

**关键代码**：
