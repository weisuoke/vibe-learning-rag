## 场景描述

### 业务需求

构建一个分布式配置管理系统，需要：
1. **多节点配置更新**：不同节点可以更新不同的配置项
2. **配置合并**：自动合并来自多个节点的配置更新
3. **优先级处理**：后更新的配置覆盖先前的配置
4. **嵌套字典支持**：支持多层嵌套的配置结构

### 技术挑战

- 如何合并来自多个节点的字典更新？
- 如何处理配置的优先级？
- 如何避免配置冲突？
- 如何支持部分配置更新？

### 解决方案

使用 LangGraph 的 `operator.or_` Reducer：
- **字典合并**：自动合并多个字典
- **键覆盖**：相同键的值会被后来的值覆盖
- **部分更新**：只需返回需要更新的配置项

## 核心代码实现

### 1. 状态定义 (config_manager/state.py)

```python
"""
状态定义模块
使用 operator.or_ Reducer 管理配置字典
"""
from typing import Annotated, TypedDict
import operator


class ConfigState(TypedDict):
    """
    配置管理状态

    Attributes:
        config: 配置字典，使用 operator.or_ Reducer
                - 新配置会与旧配置合并
                - 相同键的值会被覆盖
        metadata: 元数据字典，也使用 operator.or_
    """
    config: Annotated[dict, operator.or_]
    metadata: Annotated[dict, operator.or_]


class NestedConfigState(TypedDict):
    """
    支持嵌套配置的状态

    注意：operator.or_ 只合并顶层键
    嵌套字典需要自定义 Reducer
    """
    config: Annotated[dict, operator.or_]
    version: str  # 无 Reducer，直接替换
```

```python
    覆盖配置节点
```

### 3. 图构建 (config_manager/graph.py)

```python
"""
图构建模块
定义配置管理的工作流
"""
from langgraph.graph import StateGraph, START, END
from config_manager.state import ConfigState
from config_manager.nodes import (
    database_config_node,
    cache_config_node,
    api_config_node,
    merge_env_config_node
)


def create_config_graph():
    """
    创建配置管理图

    工作流：
    START -> database -> cache -> api -> env -> END

    所有节点的配置会自动合并
    """
    builder = StateGraph(ConfigState)

    # 添加节点
    builder.add_node("database", database_config_node)
    builder.add_node("cache", cache_config_node)
    builder.add_node("api", api_config_node)
    builder.add_node("env", merge_env_config_node)

    # 添加边（顺序执行）
    builder.add_edge(START, "database")
    builder.add_edge("database", "cache")
    builder.add_edge("cache", "api")
    builder.add_edge("api", "env")
    builder.add_edge("env", END)

    # 编译
    graph = builder.compile()

    return graph


def create_parallel_config_graph():
    """
    创建并行配置管理图

    工作流：
    START -> [database, cache, api] -> env -> END

    database, cache, api 并行执行
    """
    builder = StateGraph(ConfigState)

    # 添加节点
    builder.add_node("database", database_config_node)
    builder.add_node("cache", cache_config_node)
    builder.add_node("api", api_config_node)
    builder.add_node("env", merge_env_config_node)

    # 并行执行
    builder.add_edge(START, "database")
    builder.add_edge(START, "cache")
    builder.add_edge(START, "api")

    # 汇聚到 env
    builder.add_edge("database", "env")
    builder.add_edge("cache", "env")
    builder.add_edge("api", "env")

    builder.add_edge("env", END)

    # 编译
    graph = builder.compile()

    return graph
```

```python
"""
配置优先级处理示例
演示后更新的配置覆盖先前的配置
"""
from dotenv import load_dotenv
import operator

load_dotenv()


def demonstrate_priority():
    """演示配置优先级"""
    # 初始配置
    config1 = {
        "database": {
            "host": "localhost",
            "port": 5432
        },
        "app_name": "MyApp"
    }

    # 第一次更新
    config2 = {
        "database": {
            "port": 3306  # 覆盖端口
        },
        "cache": {
            "type": "redis"
        }
    }

    # 第二次更新
    config3 = {
        "database": {
            "host": "remote-db.example.com"  # 覆盖主机
        }
    }

    # 使用 operator.or_ 合并
    merged = config1
    merged = operator.or_(merged, config2)
    merged = operator.or_(merged, config3)

    print("=" * 60)
    print("配置合并结果")
    print("=" * 60)
    print(f"database.host: {merged['database']['host']}")  # remote-db.example.com
    print(f"database.port: {merged['database']['port']}")  # 3306
    print(f"app_name: {merged['app_name']}")              # MyApp
    print(f"cache.type: {merged['cache']['type']}")       # redis


if __name__ == "__main__":
    demonstrate_priority()
```

### 示例 3: 嵌套配置处理 (examples/nested_config.py)

```python
"""
嵌套配置处理示例
演示 operator.or_ 对嵌套字典的处理
"""
from dotenv import load_dotenv
import operator
from typing import Annotated, TypedDict
from langgraph.graph import StateGraph, START, END

load_dotenv()


def deep_merge_dicts(dict1: dict, dict2: dict) -> dict:
    """
    深度合并字典

    递归合并嵌套字典
    """
    result = dict1.copy()

    for key, value in dict2.items():
        if key in result and isinstance(result[key], dict) and isinstance(value, dict):
            result[key] = deep_merge_dicts(result[key], value)
        else:
            result[key] = value

    return result


class DeepConfigState(TypedDict):
    """使用自定义深度合并 Reducer"""
    config: Annotated[dict, deep_merge_dicts]


def node1(state: DeepConfigState) -> dict:
    return {
        "config": {
            "database": {
                "host": "localhost",
                "port": 5432,
                "credentials": {
                    "user": "admin"
                }
            }
        }
    }


def node2(state: DeepConfigState) -> dict:
    return {
        "config": {
            "database": {
                "credentials": {
                    "password": "secret"  # 添加密码，保留 user
                }
            }
        }
    }


def main():
    # 创建图
    builder = StateGraph(DeepConfigState)
    builder.add_node("node1", node1)
    builder.add_node("node2", node2)
    builder.add_edge(START, "node1")
    builder.add_edge("node1", "node2")
    builder.add_edge("node2", END)
    graph = builder.compile()

    # 执行
    result = graph.invoke({"config": {}})

    print("=" * 60)
    print("深度合并结果")
    print("=" * 60)
    print(f"database.host: {result['config']['database']['host']}")
    print(f"database.port: {result['config']['database']['port']}")
    print(f"database.credentials.user: {result['config']['database']['credentials']['user']}")
    print(f"database.credentials.password: {result['config']['database']['credentials']['password']}")


if __name__ == "__main__":
    main()
```

## 常见问题与解决方案

### 问题 1: 嵌套字典被覆盖

**问题描述**：
`operator.or_` 只合并顶层键，嵌套字典会被整体替换。

**解决方案**：
```python
def deep_merge(dict1: dict, dict2: dict) -> dict:
    """深度合并字典"""
    result = dict1.copy()
    for key, value in dict2.items():
        if key in result and isinstance(result[key], dict) and isinstance(value, dict):
            result[key] = deep_merge(result[key], value)
        else:
            result[key] = value
    return result

class State(TypedDict):
    config: Annotated[dict, deep_merge]
```

### 问题 2: 配置冲突检测

**问题描述**：
多个节点可能更新相同的配置项，导致冲突。

**解决方案**：
```python
def safe_merge(dict1: dict, dict2: dict) -> dict:
    """安全合并，检测冲突"""
    result = dict1.copy()
    conflicts = []

    for key, value in dict2.items():
        if key in result and result[key] != value:
            conflicts.append(f"Conflict on key '{key}': {result[key]} vs {value}")
        result[key] = value

    if conflicts:
        print(f"警告：检测到配置冲突：{conflicts}")

    return result
```

### 问题 3: 配置验证

**问题描述**：
合并后的配置可能不符合预期格式。

**解决方案**：
```python
from pydantic import BaseModel, ValidationError

class DatabaseConfig(BaseModel):
    host: str
    port: int
    name: str

def validate_config_node(state: ConfigState) -> dict:
    """验证配置节点"""
    config = state["config"]

    try:
        # 验证数据库配置
        if "database" in config:
            DatabaseConfig(**config["database"])
    except ValidationError as e:
        print(f"配置验证失败：{e}")

    return {"config": {}}
```

## 性能优化建议

### 1. 避免大字典合并

```python
# 不推荐：合并大字典
def bad_node(state):
    return {"config": huge_dict}  # 每次都复制整个字典

# 推荐：只返回需要更新的部分
def good_node(state):
    return {"config": {"key": "value"}}  # 只更新必要的键
```

### 2. 使用浅拷贝

```python
# operator.or_ 内部使用浅拷贝
# 对于大型嵌套结构，考虑使用引用
def efficient_merge(dict1: dict, dict2: dict) -> dict:
    # 只复制顶层
    return {**dict1, **dict2}
```

### 3. 批量更新

```python
# 不推荐：多次小更新
def node1(state):
    return {"config": {"key1": "value1"}}

def node2(state):
    return {"config": {"key2": "value2"}}

# 推荐：批量更新
def batch_node(state):
    return {
        "config": {
            "key1": "value1",
            "key2": "value2"
        }
    }
```

### 4. 缓存配置

```python
from functools import lru_cache

@lru_cache(maxsize=1)
def get_static_config():
```

## 总结

本示例展示了如何使用 `operator.or_` Reducer 构建配置管理系统：

1. **字典合并**：自动合并多个节点的配置更新
2. **优先级处理**：后更新的配置覆盖先前的配置
3. **部分更新**：只需返回需要更新的配置项
4. **深度合并**：自定义 Reducer 支持嵌套字典合并

**关键要点**：
- `operator.or_` 只合并顶层键
- 相同键的值会被后来的值覆盖
- 适用于配置管理、参数合并等场景
- 需要深度合并时使用自定义 Reducer
