## 场景 1：自动化守护进程安装

### 原理

通过 Onboarding Wizard 自动安装守护进程，包括配置创建、服务安装和启动验证。

### 实现

```typescript
// scripts/install-daemon.ts
/**
 * 自动化守护进程安装脚本
 * 支持 macOS (launchd) 和 Linux (systemd)
 */

import { spawn } from 'child_process';
import * as os from 'os';
import * as fs from 'fs';
import * as path from 'path';

interface DaemonInstallOptions {
  profile?: string;
  port?: number;
  token?: string;
  autoStart?: boolean;
}

/**
 * 安装守护进程
 */
async function installDaemon(options: DaemonInstallOptions = {}): Promise<void> {
  const {
    profile = 'default',
    port = 18789,
    token,
    autoStart = true
  } = options;

  console.log('🚀 Installing OpenClaw Gateway daemon...');
  console.log(`   Profile: ${profile}`);
  console.log(`   Port: ${port}`);
  console.log(`   Platform: ${os.platform()}`);
  console.log('');

  // 构建安装命令
  const args = ['onboard', '--install-daemon'];

  if (profile !== 'default') {
    args.push('--profile', profile);
  }

  // 设置环境变量
  const env = { ...process.env };
  if (token) {
    env.OPENCLAW_GATEWAY_TOKEN = token;
  }
  if (port !== 18789) {
    env.OPENCLAW_GATEWAY_PORT = port.toString();
  }

  // 执行安装
  return new Promise((resolve, reject) => {
    const install = spawn('openclaw', args, {
      stdio: 'inherit',
      shell: true,
      env
    });

    install.on('error', (error) => {
      console.error('❌ Installation failed:', error);
      reject(error);
    });

    install.on('exit', async (code) => {
      if (code === 0) {
        console.log('✅ Daemon installed successfully');

        if (autoStart) {
          console.log('🔄 Starting daemon...');
          await startDaemon(profile);
        }

        // 验证安装
        await verifyDaemon(profile);
        resolve();
      } else {
        reject(new Error(`Installation failed with code ${code}`));
      }
    });
  });
}

/**
 * 启动守护进程
 */
async function startDaemon(profile: string = 'default'): Promise<void> {
  const args = ['gateway', 'start'];
  if (profile !== 'default') {
    args.push('--profile', profile);
  }

  return new Promise((resolve, reject) => {
    const start = spawn('openclaw', args, {
      stdio: 'inherit',
      shell: true
    });

    start.on('exit', (code) => {
      if (code === 0) {
        console.log('✅ Daemon started');
        resolve();
      } else {
        reject(new Error(`Start failed with code ${code}`));
      }
    });
  });
}

/**
 * 验证守护进程安装
 */
async function verifyDaemon(profile: string = 'default'): Promise<void> {
  console.log('🔍 Verifying daemon installation...');

  const args = ['gateway', 'status'];
  if (profile !== 'default') {
    args.push('--profile', profile);
  }

  return new Promise((resolve, reject) => {
    const status = spawn('openclaw', args, {
      stdio: 'inherit',
      shell: true
    });

    status.on('exit', (code) => {
      if (code === 0) {
        console.log('✅ Daemon is running');
        resolve();
      } else {
        console.log('⚠️  Daemon verification failed');
        reject(new Error('Verification failed'));
      }
    });
  });
}

// 使用示例
if (require.main === module) {
  const token = process.env.OPENCLAW_GATEWAY_TOKEN || 'your-token-here';

  installDaemon({
    profile: 'default',
    port: 18789,
    token,
    autoStart: true
  }).catch((error) => {
    console.error('Installation failed:', error);
    process.exit(1);
  });
}

export { installDaemon, startDaemon, verifyDaemon };
```

## 场景 2：多实例守护进程部署

### 原理

在同一主机上部署多个 Gateway 实例，使用不同的 profile 和端口进行隔离。

### 实现

```typescript
// scripts/deploy-multiple-daemons.ts
/**
 * 多实例守护进程部署脚本
 * 部署 production、staging、development 三个实例
 */

import { spawn } from 'child_process';

interface InstanceConfig {
  profile: string;
  port: number;
  description: string;
}

const INSTANCES: InstanceConfig[] = [
  { profile: 'production', port: 18789, description: 'Production Gateway' },
  { profile: 'staging', port: 18790, description: 'Staging Gateway' },
  { profile: 'development', port: 18791, description: 'Development Gateway' }
];

/**
 * 部署单个实例
 */
async function deployInstance(config: InstanceConfig): Promise<void> {
  console.log(`\n📦 Deploying ${config.description}...`);
  console.log(`   Profile: ${config.profile}`);
  console.log(`   Port: ${config.port}`);

  const args = [
    'onboard',
    '--install-daemon',
    '--profile', config.profile
  ];

  const env = {
    ...process.env,
    OPENCLAW_GATEWAY_PORT: config.port.toString(),
    OPENCLAW_GATEWAY_TOKEN: process.env[`OPENCLAW_${config.profile.toUpperCase()}_TOKEN`] || 'default-token'
  };

  return new Promise((resolve, reject) => {
    const install = spawn('openclaw', args, {
      stdio: 'inherit',
      shell: true,
      env
    });

    install.on('exit', (code) => {
      if (code === 0) {
        console.log(`✅ ${config.description} deployed`);
        resolve();
      } else {
        reject(new Error(`Deployment failed for ${config.profile}`));
      }
    });
  });
}

/**
 * 部署所有实例
 */
async function deployAllInstances(): Promise<void> {
  console.log('🚀 Deploying multiple Gateway instances...\n');

  for (const instance of INSTANCES) {
    try {
      await deployInstance(instance);
    } catch (error) {
      console.error(`❌ Failed to deploy ${instance.profile}:`, error);
      throw error;
    }
  }

  console.log('\n✅ All instances deployed successfully');
  console.log('\n📊 Instance Summary:');
  INSTANCES.forEach(instance => {
    console.log(`   - ${instance.description}: port ${instance.port}`);
  });

  // 验证所有实例
  console.log('\n🔍 Verifying all instances...');
  await verifyAllInstances();
}

/**
 * 验证所有实例
 */
async function verifyAllInstances(): Promise<void> {
  for (const instance of INSTANCES) {
    const args = ['gateway', 'status', '--profile', instance.profile];

    await new Promise((resolve) => {
      const status = spawn('openclaw', args, {
        stdio: 'pipe',
        shell: true
      });

      status.on('exit', (code) => {
        if (code === 0) {
          console.log(`   ✅ ${instance.profile}: running`);
        } else {
          console.log(`   ❌ ${instance.profile}: not running`);
        }
        resolve(null);
      });
    });
  }
}

// 使用示例
if (require.main === module) {
  deployAllInstances().catch((error) => {
    console.error('Deployment failed:', error);
    process.exit(1);
  });
}

export { deployAllInstances, deployInstance };
```

## 场景 3：守护进程生命周期管理

### 原理

提供统一的守护进程生命周期管理接口，支持启动、停止、重启、状态查询等操作。

### 实现

```typescript
// scripts/daemon-manager.ts
/**
 * 守护进程生命周期管理器
 * 提供统一的管理接口
 */

import { spawn, exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

interface DaemonStatus {
  profile: string;
  running: boolean;
  pid?: number;
  port?: number;
  uptime?: string;
}

class DaemonManager {
  private profile: string;

  constructor(profile: string = 'default') {
    this.profile = profile;
  }

  /**
   * 启动守护进程
   */
  async start(): Promise<void> {
    console.log(`🚀 Starting daemon (${this.profile})...`);

    const args = ['gateway', 'start'];
    if (this.profile !== 'default') {
      args.push('--profile', this.profile);
    }

    return this.executeCommand(args);
  }

  /**
   * 停止守护进程
   */
  async stop(): Promise<void> {
    console.log(`🛑 Stopping daemon (${this.profile})...`);

    const args = ['gateway', 'stop'];
    if (this.profile !== 'default') {
      args.push('--profile', this.profile);
    }

    return this.executeCommand(args);
  }

  /**
   * 重启守护进程
   */
  async restart(): Promise<void> {
    console.log(`🔄 Restarting daemon (${this.profile})...`);

    const args = ['gateway', 'restart'];
    if (this.profile !== 'default') {
      args.push('--profile', this.profile);
    }

    return this.executeCommand(args);
  }

  /**
   * 获取守护进程状态
   */
  async status(): Promise<DaemonStatus> {
    const args = ['gateway', 'status', '--json'];
    if (this.profile !== 'default') {
      args.push('--profile', this.profile);
    }

    try {
      const { stdout } = await execAsync(`openclaw ${args.join(' ')}`);
      const status = JSON.parse(stdout);

      return {
        profile: this.profile,
        running: status.service?.running || false,
        pid: status.service?.pid,
        port: status.gateway?.port,
        uptime: status.service?.uptime
      };
    } catch (error) {
      return {
        profile: this.profile,
        running: false
      };
    }
  }

  /**
   * 卸载守护进程
   */
  async uninstall(): Promise<void> {
    console.log(`🗑️  Uninstalling daemon (${this.profile})...`);

    const args = ['gateway', 'uninstall'];
    if (this.profile !== 'default') {
      args.push('--profile', this.profile);
    }

    return this.executeCommand(args);
  }

  /**
   * 执行命令
   */
  private executeCommand(args: string[]): Promise<void> {
    return new Promise((resolve, reject) => {
      const process = spawn('openclaw', args, {
        stdio: 'inherit',
        shell: true
      });

      process.on('exit', (code) => {
        if (code === 0) {
          resolve();
        } else {
          reject(new Error(`Command failed with code ${code}`));
        }
      });
    });
  }
}

// 使用示例
if (require.main === module) {
  const manager = new DaemonManager('default');

  (async () => {
    // 启动
    await manager.start();

    // 查看状态
    const status = await manager.status();
    console.log('Status:', status);

    // 重启
    // await manager.restart();

    // 停止
    // await manager.stop();
  })().catch(console.error);
}

export { DaemonManager, DaemonStatus };
```

loginctl enable-linger $USER

## 总结

后台守护进程安装是生产部署的核心：

- **自动化安装**：`openclaw onboard --install-daemon`
- **多实例部署**：使用 profile 隔离
- **生命周期管理**：统一的启动/停止/重启接口
- **状态监控**：定期检查和日志监控

掌握这些技巧可以确保 Gateway 在生产环境稳定运行。
