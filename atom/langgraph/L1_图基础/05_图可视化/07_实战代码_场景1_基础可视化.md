# 图可视化 - 实战代码: 场景1 基础可视化

## 场景描述

在开发 LangGraph 工作流时,需要快速查看图的结构,验证节点和边的连接关系是否正确。

## 完整代码

```python
"""
场景1: 基础可视化
演示: 创建简单的 LangGraph 工作流并进行基础可视化
"""

from langgraph.graph import StateGraph, START, END
from typing import TypedDict
from IPython.display import Image, display

# ===== 1. 定义状态 =====
class State(TypedDict):
    """工作流状态定义"""
    messages: list[str]
    step: int

# ===== 2. 定义节点函数 =====
def retrieve(state: State) -> State:
    """检索节点"""
    print(f"[Retrieve] 当前状态: {state}")
    return {
        "messages": state["messages"] + ["retrieved"],
        "step": state.get("step", 0) + 1
    }

def process(state: State) -> State:
    """处理节点"""
    print(f"[Process] 当前状态: {state}")
    return {
        "messages": state["messages"] + ["processed"],
        "step": state.get("step", 0) + 1
    }

def generate(state: State) -> State:
    """生成节点"""
    print(f"[Generate] 当前状态: {state}")
    return {
        "messages": state["messages"] + ["generated"],
        "step": state.get("step", 0) + 1
    }

# ===== 3. 创建工作流 =====
print("=== 创建工作流 ===")
workflow = StateGraph(State)

# 添加节点
workflow.add_node("retrieve", retrieve)
workflow.add_node("process", process)
workflow.add_node("generate", generate)

# 添加边
workflow.add_edge(START, "retrieve")
workflow.add_edge("retrieve", "process")
workflow.add_edge("process", "generate")
workflow.add_edge("generate", END)

# 编译图
app = workflow.compile()
print("工作流创建完成")

# ===== 4. 方法1: 生成 Mermaid 文本 =====
print("\n=== 方法1: 生成 Mermaid 文本 ===")
mermaid_text = app.get_graph().draw_mermaid()
print(mermaid_text)

# ===== 5. 方法2: 生成 PNG 图像 =====
print("\n=== 方法2: 生成 PNG 图像 ===")
try:
    png_data = app.get_graph().draw_mermaid_png()
    print(f"PNG 生成成功,大小: {len(png_data)} 字节")
    
    # 保存到文件
    with open("workflow_basic.png", "wb") as f:
        f.write(png_data)
    print("PNG 已保存到: workflow_basic.png")
except Exception as e:
    print(f"PNG 生成失败: {e}")
    print("使用 Mermaid 文本作为后备方案")

# ===== 6. 方法3: Jupyter 自动显示 =====
print("\n=== 方法3: Jupyter 自动显示 ===")
print("在 Jupyter 中运行以下代码:")
print("app  # 自动显示图表")

# ===== 7. 方法4: 手动显示 =====
print("\n=== 方法4: 手动显示 ===")
print("在 Jupyter 中运行以下代码:")
print("from IPython.display import Image, display")
print("display(Image(app.get_graph().draw_mermaid_png()))")

# ===== 8. 验证图结构 =====
print("\n=== 验证图结构 ===")
graph = app.get_graph()
print(f"节点数量: {len(graph.nodes)}")
print(f"边数量: {len(graph.edges)}")

print("\n节点列表:")
for node in graph.nodes:
    print(f"  - {node.id}")

print("\n边列表:")
for edge in graph.edges:
    edge_type = "条件边" if edge.conditional else "普通边"
    print(f"  - {edge.source} -> {edge.target} [{edge_type}]")

# ===== 9. 执行工作流 =====
print("\n=== 执行工作流 ===")
initial_state = {"messages": [], "step": 0}
result = app.invoke(initial_state)
print(f"最终结果: {result}")

# ===== 10. 保存 Mermaid 文本到文件 =====
print("\n=== 保存 Mermaid 文本到文件 ===")
mermaid_text = app.get_graph().draw_mermaid()
with open("workflow_basic.mmd", "w") as f:
    f.write(mermaid_text)
print("Mermaid 文本已保存到: workflow_basic.mmd")

print("\n=== 基础可视化完成 ===")
```

## 关键点说明

### 1. 四种可视化方法

```python
# 方法1: Mermaid 文本 (最快,适合开发)
mermaid_text = app.get_graph().draw_mermaid()

# 方法2: PNG 图像 (适合文档)
png_data = app.get_graph().draw_mermaid_png()

# 方法3: Jupyter 自动显示 (最简洁)
app  # 在 Jupyter 中

# 方法4: Jupyter 手动显示 (最灵活)
display(Image(app.get_graph().draw_mermaid_png()))
```

### 2. 图结构验证

```python
graph = app.get_graph()
print(f"节点数: {len(graph.nodes)}")
print(f"边数: {len(graph.edges)}")
```

### 3. 文件保存

```python
# 保存 PNG
with open("graph.png", "wb") as f:
    f.write(png_data)

# 保存 Mermaid 文本
with open("graph.mmd", "w") as f:
    f.write(mermaid_text)
```

2. **调试阶段**: 检查节点和边的连接

## 常见问题

### Q1: PNG 生成失败怎么办?

**A**: 使用 try-except 包裹,后备使用 Mermaid 文本:
```python
try:
    png_data = app.get_graph().draw_mermaid_png()
except Exception as e:
    print(f"PNG 生成失败: {e}")
    mermaid_text = app.get_graph().draw_mermaid()
    print(mermaid_text)
```

### Q2: 如何在非 Jupyter 环境中查看图表?

**A**: 保存 Mermaid 文本,使用在线工具渲染:
```python
mermaid_text = app.get_graph().draw_mermaid()
with open("graph.mmd", "w") as f:
    f.write(mermaid_text)
# 访问 https://mermaid.live/ 粘贴文本
```

### Q3: 如何验证图结构是否正确?

**A**: 检查节点数和边数:
```python
graph = app.get_graph()
assert len(graph.nodes) == 5  # START + 3 nodes + END
assert len(graph.edges) == 4  # 4 edges
```
