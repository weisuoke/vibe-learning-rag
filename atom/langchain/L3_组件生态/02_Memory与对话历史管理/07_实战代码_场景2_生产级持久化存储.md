## 场景概述

生产环境中，内存存储（InMemoryChatMessageHistory）无法满足需求：
- **数据丢失风险**：服务重启后对话历史丢失
- **扩展性问题**：无法跨多个服务实例共享
- **性能瓶颈**：大量对话历史占用内存

本场景演示如何使用持久化存储（Redis、PostgreSQL、SQLite）实现生产级对话历史管理。

**核心要点**：
- Redis：高性能、支持TTL、适合会话管理
- PostgreSQL：持久化、支持复杂查询、适合长期存储
- SQLite：轻量级、无需额外服务、适合小型应用

## 实战示例1：Redis 持久化存储

### 核心特性

- **高性能**：内存存储，读写速度快
- **TTL 支持**：自动清理过期会话
- **分布式**：支持多实例共享

### 完整代码

```python
"""
Redis 持久化存储示例
演示如何使用 Redis 存储对话历史，支持 TTL 和连接池
"""

from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_redis import RedisChatMessageHistory
import redis
from dotenv import load_dotenv

load_dotenv()

# ============================================================
# 1. Redis 连接池配置
# ============================================================

# 创建 Redis 连接池（生产环境推荐）
redis_pool = redis.ConnectionPool(
    host='localhost',
    port=6379,
    db=0,
    max_connections=10,  # 最大连接数
    decode_responses=True  # 自动解码为字符串
)

# ============================================================
# 2. Session History 工厂函数
# ============================================================

def get_redis_session_history(session_id: str) -> RedisChatMessageHistory:
    """
    获取 Redis 存储的对话历史

    参数:
        session_id: 会话ID

    返回:
        RedisChatMessageHistory 实例
    """
    return RedisChatMessageHistory(
        session_id=session_id,
        url="redis://localhost:6379/0",
        ttl=3600  # 1小时过期（生产环境推荐设置）
    )

# ============================================================
# 3. 创建对话链
# ============================================================

# 初始化 LLM
llm = ChatOpenAI(
    model="gpt-4o-mini",
    temperature=0.7
)

# 包装为带历史的对话链
conversation = RunnableWithMessageHistory(
    llm,
    get_redis_session_history,
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 4. 测试对话
# ============================================================

def test_redis_storage():
    """测试 Redis 持久化存储"""

    session_id = "user_123_session_1"
    config = {"configurable": {"session_id": session_id}}

    print("=== Redis 持久化存储测试 ===\n")

    # 第一轮对话
    print("👤 用户: 我叫张三，是一名软件工程师")
    response1 = conversation.invoke(
        {"input": "我叫张三，是一名软件工程师"},
        config=config
    )
    print(f"🤖 AI: {response1.content}\n")

    # 第二轮对话（测试记忆）
    print("👤 用户: 我的职业是什么？")
    response2 = conversation.invoke(
        {"input": "我的职业是什么？"},
        config=config
    )
    print(f"🤖 AI: {response2.content}\n")

    # 验证 Redis 中的数据
    print("=== Redis 数据验证 ===")
    history = get_redis_session_history(session_id)
    messages = history.messages
    print(f"✅ 存储的消息数量: {len(messages)}")
    print(f"✅ TTL 设置: 3600秒（1小时）")

    # 查看 Redis 中的键
    r = redis.Redis(connection_pool=redis_pool)
    keys = r.keys(f"*{session_id}*")
    print(f"✅ Redis 键: {keys}")

if __name__ == "__main__":
    test_redis_storage()
```

1. **连接池配置**：使用 `redis.ConnectionPool` 提高性能
2. **TTL 设置**：自动清理过期会话，避免内存泄漏
3. **键命名规范**：使用 `langchain:chat_history:{session_id}` 格式

## 实战示例2：PostgreSQL 持久化存储

### 核心特性

- **持久化**：数据永久保存，不会丢失
- **复杂查询**：支持 SQL 查询和分析
- **事务支持**：保证数据一致性

### 完整代码

```python
"""
PostgreSQL 持久化存储示例
演示如何使用 PostgreSQL 存储对话历史，支持复杂查询
"""

from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_postgres import PostgresChatMessageHistory
from sqlalchemy import create_engine
from sqlalchemy.pool import QueuePool
from dotenv import load_dotenv
import os

load_dotenv()

# ============================================================
# 1. PostgreSQL 连接配置
# ============================================================

# 数据库连接字符串
DATABASE_URL = os.getenv(
    "DATABASE_URL",
    "postgresql://user:password@localhost:5432/langchain_db"
)

# 创建连接引擎（带连接池）
engine = create_engine(
    DATABASE_URL,
    poolclass=QueuePool,
    pool_size=10,  # 连接池大小
    max_overflow=20,  # 最大溢出连接数
    pool_pre_ping=True  # 连接前检查有效性
)

# ============================================================
# 2. 初始化数据库表
# ============================================================

def init_database():
    """初始化数据库表结构"""
    from langchain_postgres import PostgresChatMessageHistory

    # 创建表（如果不存在）
    PostgresChatMessageHistory.create_tables(engine)
    print("✅ 数据库表初始化完成")

# ============================================================
# 3. Session History 工厂函数
# ============================================================

def get_postgres_session_history(session_id: str) -> PostgresChatMessageHistory:
    """
    获取 PostgreSQL 存储的对话历史

    参数:
        session_id: 会话ID

    返回:
        PostgresChatMessageHistory 实例
    """
    return PostgresChatMessageHistory(
        session_id=session_id,
        connection_string=DATABASE_URL,
        table_name="chat_history"  # 自定义表名
    )

# ============================================================
# 4. 创建对话链
# ============================================================

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

conversation = RunnableWithMessageHistory(
    llm,
    get_postgres_session_history,
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 5. 测试对话
# ============================================================

def test_postgres_storage():
    """测试 PostgreSQL 持久化存储"""

    # 初始化数据库
    init_database()

    session_id = "user_456_session_1"
    config = {"configurable": {"session_id": session_id}}

    print("\n=== PostgreSQL 持久化存储测试 ===\n")

    # 第一轮对话
    print("👤 用户: 我最喜欢的编程语言是 Python")
    response1 = conversation.invoke(
        {"input": "我最喜欢的编程语言是 Python"},
        config=config
    )
    print(f"🤖 AI: {response1.content}\n")

    # 第二轮对话
    print("👤 用户: 我喜欢什么编程语言？")
    response2 = conversation.invoke(
        {"input": "我喜欢什么编程语言？"},
        config=config
    )
    print(f"🤖 AI: {response2.content}\n")

    # 验证数据库中的数据
    print("=== PostgreSQL 数据验证 ===")
    history = get_postgres_session_history(session_id)
    messages = history.messages
    print(f"✅ 存储的消息数量: {len(messages)}")
    print(f"✅ 表名: chat_history")
    print(f"✅ 会话ID: {session_id}")

    # 查询数据库统计
    with engine.connect() as conn:
        result = conn.execute(
            "SELECT COUNT(*) FROM chat_history WHERE session_id = %s",
            (session_id,)
        )
        count = result.scalar()
        print(f"✅ 数据库记录数: {count}")

if __name__ == "__main__":
    test_postgres_storage()
```

## 性能对比测试

### 测试代码

```python
"""
性能对比测试
对比 Redis、PostgreSQL、SQLite 的读写性能
"""

import time
from typing import Callable

def benchmark_storage(
    get_history_func: Callable,
    session_id: str,
    num_messages: int = 100
):
    """
    性能基准测试

    参数:
        get_history_func: 获取历史的函数
        session_id: 会话ID
        num_messages: 测试消息数量
    """
    history = get_history_func(session_id)

    # 写入测试
    start_time = time.time()
    for i in range(num_messages):
        history.add_user_message(f"测试消息 {i}")
        history.add_ai_message(f"回复 {i}")
    write_time = time.time() - start_time

    # 读取测试
    start_time = time.time()
    messages = history.messages
    read_time = time.time() - start_time

    return {
        "write_time": write_time,
        "read_time": read_time,
        "total_messages": len(messages)
    }

# 运行测试
print("=== 性能对比测试 ===\n")

# Redis 测试
redis_result = benchmark_storage(
    get_redis_session_history,
    "benchmark_redis",
    100
)
print(f"Redis:")
print(f"  写入时间: {redis_result['write_time']:.3f}秒")
print(f"  读取时间: {redis_result['read_time']:.3f}秒")
print(f"  消息数量: {redis_result['total_messages']}\n")

# PostgreSQL 测试
postgres_result = benchmark_storage(
    get_postgres_session_history,
    "benchmark_postgres",
    100
)
print(f"PostgreSQL:")
print(f"  写入时间: {postgres_result['write_time']:.3f}秒")
print(f"  读取时间: {postgres_result['read_time']:.3f}秒")
print(f"  消息数量: {postgres_result['total_messages']}\n")

# SQLite 测试
sqlite_result = benchmark_storage(
    get_sqlite_session_history,
    "benchmark_sqlite",
    100
)
print(f"SQLite:")
print(f"  写入时间: {sqlite_result['write_time']:.3f}秒")
print(f"  读取时间: {sqlite_result['read_time']:.3f}秒")
print(f"  消息数量: {sqlite_result['total_messages']}\n")
```

## 生产环境最佳实践

### 1. 存储方案选择

```python
"""
根据场景选择存储方案
"""

def choose_storage_backend(scenario: str):
    """
    根据场景选择存储方案

    场景:
        - "high_concurrency": 高并发场景 → Redis
        - "long_term": 长期存储场景 → PostgreSQL
        - "small_app": 小型应用 → SQLite
    """
    if scenario == "high_concurrency":
        return get_redis_session_history
    elif scenario == "long_term":
        return get_postgres_session_history
    elif scenario == "small_app":
        return get_sqlite_session_history
    else:
        raise ValueError(f"未知场景: {scenario}")
```

### 2. 连接池配置

```python
生产环境连接池配置

redis_pool = redis.ConnectionPool(
```

### 3. 错误处理

```python
"""
生产环境错误处理
"""

def get_session_history_with_fallback(session_id: str):
    """
    带降级的 Session History 获取

    优先使用 Redis，失败时降级到内存存储
    """
    try:
        return get_redis_session_history(session_id)
    except Exception as e:
        print(f"⚠️ Redis 连接失败，降级到内存存储: {e}")
        from langchain_core.chat_history import InMemoryChatMessageHistory
        return InMemoryChatMessageHistory()
```

## 常见问题

### Q1: 如何清理过期会话？

**Redis**：使用 TTL 自动清理
```python
RedisChatMessageHistory(
    session_id=session_id,
    url="redis://localhost:6379/0",
    ttl=3600  # 1小时后自动删除
)
```

**PostgreSQL**：定期执行清理任务
```python
# 删除7天前的会话
DELETE FROM chat_history
WHERE created_at < NOW() - INTERVAL '7 days';
```

### Q2: 如何处理大量历史消息？

使用分页查询：
```python
def get_recent_messages(session_id: str, limit: int = 10):
    """只获取最近的 N 条消息"""
    history = get_postgres_session_history(session_id)
    messages = history.messages
    return messages[-limit:]  # 返回最后 N 条
```
