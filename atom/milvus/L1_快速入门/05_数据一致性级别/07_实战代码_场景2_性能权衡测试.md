# 性能权衡测试完整代码
# pip install pymilvus numpy matplotlib psutil

from pymilvus import connections, Collection, CollectionSchema, FieldSchema, DataType, ConsistencyLevel, utility
import numpy as np
import time
import psutil
import matplotlib.pyplot as plt
from concurrent.futures import ThreadPoolExecutor, as_completed
from datetime import datetime

# 连接Milvus
connections.connect(alias="default", host="localhost", port="19530")

# 创建测试集合
def create_test_collection(name, consistency_level):
    if utility.has_collection(name):
        utility.drop_collection(name)

    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
        FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=128),
        FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=500)
    ]
    schema = CollectionSchema(fields=fields)
    collection = Collection(name=name, schema=schema, consistency_level=consistency_level)

    index_params = {"index_type": "IVF_FLAT", "metric_type": "L2", "params": {"nlist": 128}}
    collection.create_index(field_name="vector", index_params=index_params)
    collection.load()

    # 插入初始数据
    entities = [{"vector": np.random.rand(128).tolist(), "text": f"doc_{i}"} for i in range(10000)]
    collection.insert(entities)
    collection.flush()

    return collection

# 1. 延迟基准测试
def benchmark_latency(collection, consistency_level, iterations=1000):
    print(f"\n{'='*60}")
    print(f"Latency Benchmark: {consistency_level}")
    print(f"{'='*60}")

    latencies = []
    for i in range(iterations):
        query_vector = np.random.rand(128).tolist()
        start = time.time()
        results = collection.search(
            data=[query_vector],
            anns_field="vector",
            param={"metric_type": "L2", "params": {"nprobe": 10}},
            limit=10
        )
        latency = (time.time() - start) * 1000  # ms
        latencies.append(latency)

        if (i + 1) % 100 == 0:
            print(f"Progress: {i+1}/{iterations}")

    latencies = np.array(latencies)
    print(f"\nResults:")
    print(f"  Mean: {np.mean(latencies):.2f}ms")
    print(f"  Median (P50): {np.percentile(latencies, 50):.2f}ms")
    print(f"  P95: {np.percentile(latencies, 95):.2f}ms")
    print(f"  P99: {np.percentile(latencies, 99):.2f}ms")
    print(f"  Min: {np.min(latencies):.2f}ms")
    print(f"  Max: {np.max(latencies):.2f}ms")

    return {
        "mean": np.mean(latencies),
        "p50": np.percentile(latencies, 50),
        "p95": np.percentile(latencies, 95),
        "p99": np.percentile(latencies, 99),
        "min": np.min(latencies),
        "max": np.max(latencies),
        "data": latencies
    }

# 2. 吞吐量测试
def benchmark_throughput(collection, consistency_level, duration_seconds=30):
    print(f"\n{'='*60}")
    print(f"Throughput Benchmark: {consistency_level}")
    print(f"Duration: {duration_seconds}s")
    print(f"{'='*60}")

    start_time = time.time()
    query_count = 0

    while time.time() - start_time < duration_seconds:
        query_vector = np.random.rand(128).tolist()
        results = collection.search(
            data=[query_vector],
            anns_field="vector",
            param={"metric_type": "L2", "params": {"nprobe": 10}},
            limit=10
        )
        query_count += 1

        if query_count % 100 == 0:
            elapsed = time.time() - start_time
            current_qps = query_count / elapsed
            print(f"Progress: {elapsed:.1f}s, Current QPS: {current_qps:.1f}")

    total_time = time.time() - start_time
    qps = query_count / total_time

    print(f"\nResults:")
    print(f"  Total queries: {query_count}")
    print(f"  Total time: {total_time:.2f}s")
    print(f"  QPS: {qps:.1f}")

    return {"qps": qps, "total_queries": query_count, "total_time": total_time}

# 3. 并发性能测试
def benchmark_concurrent(collection, consistency_level, num_threads=10, queries_per_thread=100):
    print(f"\n{'='*60}")
    print(f"Concurrent Benchmark: {consistency_level}")
    print(f"Threads: {num_threads}, Queries/thread: {queries_per_thread}")
    print(f"{'='*60}")

    def worker(thread_id):
        latencies = []
        for i in range(queries_per_thread):
            query_vector = np.random.rand(128).tolist()
            start = time.time()
            results = collection.search(
                data=[query_vector],
                anns_field="vector",
                param={"metric_type": "L2", "params": {"nprobe": 10}},
                limit=10
            )
            latency = (time.time() - start) * 1000
            latencies.append(latency)
        return latencies

    start_time = time.time()
    all_latencies = []

    with ThreadPoolExecutor(max_workers=num_threads) as executor:
        futures = [executor.submit(worker, i) for i in range(num_threads)]
        for future in as_completed(futures):
            all_latencies.extend(future.result())

    total_time = time.time() - start_time
    total_queries = num_threads * queries_per_thread
    qps = total_queries / total_time

    all_latencies = np.array(all_latencies)
    print(f"\nResults:")
    print(f"  Total queries: {total_queries}")
    print(f"  Total time: {total_time:.2f}s")
    print(f"  QPS: {qps:.1f}")
    print(f"  Mean latency: {np.mean(all_latencies):.2f}ms")
    print(f"  P95 latency: {np.percentile(all_latencies, 95):.2f}ms")
    print(f"  P99 latency: {np.percentile(all_latencies, 99):.2f}ms")

    return {
        "qps": qps,
        "mean_latency": np.mean(all_latencies),
        "p95_latency": np.percentile(all_latencies, 95),
        "p99_latency": np.percentile(all_latencies, 99)
    }

# 4. 资源消耗监控
def monitor_resources(collection, consistency_level, duration_seconds=10):
    print(f"\n{'='*60}")
    print(f"Resource Monitoring: {consistency_level}")
    print(f"{'='*60}")

    cpu_usage = []
    memory_usage = []

    start_time = time.time()
    while time.time() - start_time < duration_seconds:
        # 执行查询
        query_vector = np.random.rand(128).tolist()
        collection.search(
            data=[query_vector],
            anns_field="vector",
            param={"metric_type": "L2", "params": {"nprobe": 10}},
            limit=10
        )

        # 记录资源使用
        cpu_usage.append(psutil.cpu_percent(interval=0.1))
        memory_usage.append(psutil.virtual_memory().percent)

    print(f"\nResults:")
    print(f"  Avg CPU: {np.mean(cpu_usage):.1f}%")
    print(f"  Max CPU: {np.max(cpu_usage):.1f}%")
    print(f"  Avg Memory: {np.mean(memory_usage):.1f}%")
    print(f"  Max Memory: {np.max(memory_usage):.1f}%")

    return {
        "avg_cpu": np.mean(cpu_usage),
        "max_cpu": np.max(cpu_usage),
        "avg_memory": np.mean(memory_usage),
        "max_memory": np.max(memory_usage)
    }

# 5. 可视化结果
def visualize_results(results):
    fig, axes = plt.subplots(2, 2, figsize=(15, 10))

    # 延迟对比
    ax1 = axes[0, 0]
    levels = list(results.keys())
    metrics = ['mean', 'p95', 'p99']
    x = np.arange(len(levels))
    width = 0.25

    for i, metric in enumerate(metrics):
        values = [results[level]['latency'][metric] for level in levels]
        ax1.bar(x + i*width, values, width, label=metric.upper())

    ax1.set_xlabel('Consistency Level')
    ax1.set_ylabel('Latency (ms)')
    ax1.set_title('Latency Comparison')
    ax1.set_xticks(x + width)
    ax1.set_xticklabels(levels)
    ax1.legend()
    ax1.grid(True, alpha=0.3)

    # 吞吐量对比
    ax2 = axes[0, 1]
    qps_values = [results[level]['throughput']['qps'] for level in levels]
    ax2.bar(levels, qps_values, color=['red', 'orange', 'green'])
    ax2.set_xlabel('Consistency Level')
    ax2.set_ylabel('QPS')
    ax2.set_title('Throughput Comparison')
    ax2.grid(True, alpha=0.3)

    # 并发性能对比
    ax3 = axes[1, 0]
    concurrent_qps = [results[level]['concurrent']['qps'] for level in levels]
    ax3.bar(levels, concurrent_qps, color=['red', 'orange', 'green'])
    ax3.set_xlabel('Consistency Level')
    ax3.set_ylabel('Concurrent QPS')
    ax3.set_title('Concurrent Performance')
    ax3.grid(True, alpha=0.3)

    # 资源消耗对比
    ax4 = axes[1, 1]
    cpu_values = [results[level]['resources']['avg_cpu'] for level in levels]
    memory_values = [results[level]['resources']['avg_memory'] for level in levels]

    x = np.arange(len(levels))
    width = 0.35
    ax4.bar(x - width/2, cpu_values, width, label='CPU %')
    ax4.bar(x + width/2, memory_values, width, label='Memory %')
    ax4.set_xlabel('Consistency Level')
    ax4.set_ylabel('Usage %')
    ax4.set_title('Resource Usage')
    ax4.set_xticks(x)
    ax4.set_xticklabels(levels)
    ax4.legend()
    ax4.grid(True, alpha=0.3)

    plt.tight_layout()
    plt.savefig('consistency_benchmark_results.png', dpi=300)
    print(f"\n✅ Results saved to: consistency_benchmark_results.png")

# 主测试函数
def main():
    print("="*60)
    print("Milvus Consistency Level Performance Benchmark")
    print("="*60)

    results = {}

    # 测试三种一致性级别
    for level_name, level_value in [
        ("Strong", "Strong"),
        ("Bounded", "Bounded"),
        ("Eventually", "Eventually")
    ]:
        print(f"\n{'#'*60}")
        print(f"# Testing {level_name} Consistency")
        print(f"{'#'*60}")

        # 创建集合
        collection = create_test_collection(f"perf_test_{level_name.lower()}", level_value)

        # 运行测试
        results[level_name] = {
            "latency": benchmark_latency(collection, level_name, iterations=1000),
            "throughput": benchmark_throughput(collection, level_name, duration_seconds=30),
            "concurrent": benchmark_concurrent(collection, level_name, num_threads=10, queries_per_thread=100),
            "resources": monitor_resources(collection, level_name, duration_seconds=10)
        }

        # 清理
        utility.drop_collection(f"perf_test_{level_name.lower()}")

    # 打印总结
    print(f"\n{'='*60}")
    print("Performance Summary")
    print(f"{'='*60}")

    print(f"\n{'Metric':<20} {'Strong':<15} {'Bounded':<15} {'Eventually':<15}")
    print("-" * 65)

    # 延迟
    print(f"{'Mean Latency (ms)':<20} {results['Strong']['latency']['mean']:<15.2f} {results['Bounded']['latency']['mean']:<15.2f} {results['Eventually']['latency']['mean']:<15.2f}")
    print(f"{'P95 Latency (ms)':<20} {results['Strong']['latency']['p95']:<15.2f} {results['Bounded']['latency']['p95']:<15.2f} {results['Eventually']['latency']['p95']:<15.2f}")
    print(f"{'P99 Latency (ms)':<20} {results['Strong']['latency']['p99']:<15.2f} {results['Bounded']['latency']['p99']:<15.2f} {results['Eventually']['latency']['p99']:<15.2f}")

    # 吞吐量
    print(f"{'QPS':<20} {results['Strong']['throughput']['qps']:<15.1f} {results['Bounded']['throughput']['qps']:<15.1f} {results['Eventually']['throughput']['qps']:<15.1f}")
    print(f"{'Concurrent QPS':<20} {results['Strong']['concurrent']['qps']:<15.1f} {results['Bounded']['concurrent']['qps']:<15.1f} {results['Eventually']['concurrent']['qps']:<15.1f}")

    # 资源
    print(f"{'Avg CPU %':<20} {results['Strong']['resources']['avg_cpu']:<15.1f} {results['Bounded']['resources']['avg_cpu']:<15.1f} {results['Eventually']['resources']['avg_cpu']:<15.1f}")
    print(f"{'Avg Memory %':<20} {results['Strong']['resources']['avg_memory']:<15.1f} {results['Bounded']['resources']['avg_memory']:<15.1f} {results['Eventually']['resources']['avg_memory']:<15.1f}")

    # 性能提升百分比
    print(f"\n{'='*60}")
    print("Performance Improvement vs Strong")
    print(f"{'='*60}")

    bounded_latency_improvement = (results['Strong']['latency']['mean'] - results['Bounded']['latency']['mean']) / results['Strong']['latency']['mean'] * 100
    eventually_latency_improvement = (results['Strong']['latency']['mean'] - results['Eventually']['latency']['mean']) / results['Strong']['latency']['mean'] * 100

    bounded_qps_improvement = (results['Bounded']['throughput']['qps'] - results['Strong']['throughput']['qps']) / results['Strong']['throughput']['qps'] * 100
    eventually_qps_improvement = (results['Eventually']['throughput']['qps'] - results['Strong']['throughput']['qps']) / results['Strong']['throughput']['qps'] * 100

    print(f"Bounded:")
    print(f"  Latency reduction: {bounded_latency_improvement:.1f}%")
    print(f"  QPS improvement: {bounded_qps_improvement:.1f}%")

    print(f"\nEventually:")
    print(f"  Latency reduction: {eventually_latency_improvement:.1f}%")
    print(f"  QPS improvement: {eventually_qps_improvement:.1f}%")

    # 可视化
    visualize_results(results)

if __name__ == "__main__":
    main()
