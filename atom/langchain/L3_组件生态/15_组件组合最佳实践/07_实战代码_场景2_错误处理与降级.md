# 实战代码 场景2：错误处理与降级

## 场景描述

构建一个具有完善错误处理的 LangChain 管道，展示 Fallback、Retry 和异常分类的实际应用。

## 完整可运行代码

```python
"""
错误处理与降级实战
演示：Fallback 降级、Retry 重试、异常分类、自定义错误处理
"""

import os
import time
from dotenv import load_dotenv
load_dotenv()

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser, JsonOutputParser
from langchain_core.runnables import RunnableLambda, RunnablePassthrough
from langchain_core.callbacks import BaseCallbackHandler

# ===== 1. 基础 Fallback：模型降级 =====
print("=== 1. 基础 Fallback：模型降级 ===")

# 主模型 + 备选模型
primary_model = ChatOpenAI(model="gpt-4o", temperature=0)
fallback_model = ChatOpenAI(model="gpt-4o-mini", temperature=0)

# 模型级降级
model_with_fallback = primary_model.with_fallbacks(
    [fallback_model],
)

prompt = ChatPromptTemplate.from_template("用一句话解释：{topic}")
chain = prompt | model_with_fallback | StrOutputParser()

result = chain.invoke({"topic": "量子计算"})
print(f"结果: {result}")

# ===== 2. Retry：自动重试 =====
print("\n=== 2. Retry：自动重试 ===")

# 模拟不稳定的函数
call_count = 0

def unstable_function(input_text: str) -> str:
    """模拟一个不稳定的函数，前2次调用会失败"""
    global call_count
    call_count += 1
    if call_count <= 2:
        raise ConnectionError(f"第{call_count}次调用失败：网络超时")
    return f"成功处理: {input_text}"

# 添加重试
unstable_runnable = RunnableLambda(unstable_function).with_retry(
    retry_if_exception_type=(ConnectionError,),
    stop_after_attempt=3,
    wait_exponential_jitter=True,
)

call_count = 0  # 重置计数器
try:
    result = unstable_runnable.invoke("测试数据")
    print(f"结果: {result}")
    print(f"总共调用了 {call_count} 次")
except Exception as e:
    print(f"最终失败: {e}")

# ===== 3. Retry + Fallback 组合 =====
print("\n=== 3. Retry + Fallback 组合 ===")

# 先重试，重试失败后降级
model = (
    ChatOpenAI(model="gpt-4o")
    .with_retry(
        stop_after_attempt=2,
        wait_exponential_jitter=True,
    )
    .with_fallbacks([
        ChatOpenAI(model="gpt-4o-mini").with_retry(
            stop_after_attempt=2,
        ),
    ])
)

chain = prompt | model | StrOutputParser()
result = chain.invoke({"topic": "深度学习"})
print(f"结果: {result}")

# ===== 4. 链级 Fallback =====
print("\n=== 4. 链级 Fallback ===")

# 主链
primary_chain = prompt | ChatOpenAI(model="gpt-4o") | StrOutputParser()

# 备选链（更简单的 prompt）
simple_prompt = ChatPromptTemplate.from_template("简单解释{topic}")
fallback_chain = simple_prompt | ChatOpenAI(model="gpt-4o-mini") | StrOutputParser()

# 最后的兜底
def emergency_response(inputs):
    topic = inputs.get("topic", "未知主题") if isinstance(inputs, dict) else str(inputs)
    return f"抱歉，暂时无法回答关于「{topic}」的问题，请稍后重试。"

# 组合：主链 → 备选链 → 兜底
chain_with_fallback = primary_chain.with_fallbacks(
    [fallback_chain, RunnableLambda(emergency_response)]
)

result = chain_with_fallback.invoke({"topic": "机器学习"})
print(f"结果: {result}")

# ===== 5. 输出解析错误处理 =====
print("\n=== 5. 输出解析错误处理 ===")

from pydantic import BaseModel, Field

class TopicSummary(BaseModel):
    title: str = Field(description="主题标题")
    summary: str = Field(description="一句话摘要")
    keywords: list[str] = Field(description="关键词列表")

json_parser = JsonOutputParser(pydantic_object=TopicSummary)

json_prompt = ChatPromptTemplate.from_template(
    "用 JSON 格式总结以下主题：{topic}\n\n{format_instructions}"
).partial(format_instructions=json_parser.get_format_instructions())

# JSON 解析链 + 重试（模型可能输出格式不对）
json_chain = (
    json_prompt
    | ChatOpenAI(model="gpt-4o-mini", temperature=0)
    | json_parser
)

# 解析失败时的降级：返回纯文本
text_chain = prompt | ChatOpenAI(model="gpt-4o-mini") | StrOutputParser()

safe_chain = json_chain.with_fallbacks([text_chain])

result = safe_chain.invoke({"topic": "人工智能"})
print(f"结果类型: {type(result)}")
print(f"结果: {result}")

# ===== 6. 批量处理的错误处理 =====
print("\n=== 6. 批量处理的错误处理 ===")

topics = [
    {"topic": "Python"},
    {"topic": "JavaScript"},
    {"topic": "Rust"},
]

# return_exceptions=True 让失败的返回异常对象而不是抛出
results = chain_with_fallback.batch(topics, return_exceptions=True)

for topic, result in zip(topics, results):
    if isinstance(result, Exception):
        print(f"  {topic['topic']}: 失败 - {result}")
    else:
        print(f"  {topic['topic']}: {result[:80]}...")

# ===== 7. 自定义错误处理回调 =====
print("\n=== 7. 自定义错误处理回调 ===")

class ErrorTrackingCallback(BaseCallbackHandler):
    def __init__(self):
        self.errors = []
        self.retries = 0

    def on_chain_error(self, error, **kwargs):
        self.errors.append(str(error))
        print(f"  链错误: {str(error)[:100]}")

    def on_llm_error(self, error, **kwargs):
        self.errors.append(str(error))
        print(f"  LLM错误: {str(error)[:100]}")

    def on_retry(self, retry_state, **kwargs):
        self.retries += 1
        print(f"  重试第 {self.retries} 次")

error_tracker = ErrorTrackingCallback()
result = chain_with_fallback.invoke(
    {"topic": "区块链"},
    config={"callbacks": [error_tracker]},
)
print(f"结果: {result[:100]}...")
print(f"错误数: {len(error_tracker.errors)}, 重试数: {error_tracker.retries}")

print("\n错误处理与降级演示完成")
```

## 错误处理决策树

```
异常发生
├── 是否是临时性错误？（网络超时、限流）
│   ├── 是 → Retry（指数退避，2-3次）
│   │   └── 重试仍失败 → Fallback（备选模型/链）
│   └── 否 → 是否是可恢复错误？
│       ├── 是（输出格式错误）→ Retry（换 prompt 重试）
│       └── 否（认证错误、输入错误）→ 直接抛出
└── 所有方案都失败 → 兜底响应（hardcoded response）
```

## 关键 API 速查

| 方法 | 用途 | 典型参数 |
|------|------|----------|
| `.with_fallbacks([...])` | 降级到备选方案 | `fallbacks`: Runnable 列表 |
| `.with_retry(...)` | 自动重试 | `stop_after_attempt`, `retry_if_exception_type` |
| `.batch(..., return_exceptions=True)` | 批量处理不中断 | 失败项返回 Exception 对象 |
| `RunnableLambda(fn)` | 包装兜底函数 | 任意 callable |

## 最佳实践总结

1. **分层降级**：Retry → Fallback（备选模型）→ Fallback（备选链）→ 兜底响应
2. **异常分类**：临时性错误重试，永久性错误直接抛出，格式错误换策略
3. **指数退避**：`wait_exponential_jitter=True` 避免重试风暴
4. **批量容错**：`return_exceptions=True` 让单个失败不影响整体
5. **可观测性**：用 Callback 追踪错误和重试，便于监控和调优

## RAG 场景应用

在 RAG 管道中，错误处理尤为关键：

- **Embedding 服务不可用**：降级到本地 Embedding 模型
- **向量库查询超时**：重试 + 降级到关键词检索
- **LLM 生成失败**：降级到更小的模型，或返回检索到的原文
- **输出解析失败**：重试（调整 prompt）→ 降级到纯文本输出
