# 实战代码：场景2 - IVF系列索引对比

## 1. 场景概述

### 业务场景
**中型电商产品搜索系统**

某电商平台需要构建商品图像搜索系统，具有以下特点：
- 商品数量：500万个SKU
- 每个商品：1张主图 + 4张详情图
- 总向量数：2500万个
- 查询频率：中等（50-100 QPS）
- 准确率要求：90%+（可接受轻微损失）
- 内存预算：有限（需要优化）

### 为什么对比IVF系列？

1. **数据规模适中**：2500万向量超出FLAT范围，需要ANN索引
2. **内存敏感**：需要在准确率和内存占用间权衡
3. **性能要求**：需要在延迟和召回率间平衡
4. **成本优化**：不同量化方案的成本差异显著

### IVF系列对比

| 索引类型 | 压缩比 | 内存占用 | 召回率 | 查询速度 | 适用场景 |
|---------|--------|----------|--------|----------|----------|
| **IVF_FLAT** | 1:1 | 100% | 95%+ | 快 | 准确率优先 |
| **IVF_SQ8** | 4:1 | 25% | 90%+ | 很快 | 平衡方案 |
| **IVF_PQ** | 8:1+ | 12.5% | 70-90% | 最快 | 成本优先 |

## 3. 完整代码示例

```python
"""
IVF系列索引对比：电商商品图像搜索
对比 IVF_FLAT、IVF_SQ8、IVF_PQ 三种索引的性能
"""

import time
import numpy as np
from pymilvus import (
    connections, Collection, FieldSchema,
    CollectionSchema, DataType, utility
)
from sentence_transformers import SentenceTransformer
from typing import List, Dict, Tuple
import os
from dotenv import load_dotenv

load_dotenv()

# ============================================================================
# 配置参数
# ============================================================================

MILVUS_HOST = os.getenv("MILVUS_HOST", "localhost")
MILVUS_PORT = os.getenv("MILVUS_PORT", "19530")
DIMENSION = 384
METRIC_TYPE = "L2"
NUM_VECTORS = 100_000  # 测试用10万向量（生产环境2500万）

# IVF参数配置
IVF_CONFIGS = {
    "IVF_FLAT": {
        "index_type": "IVF_FLAT",
        "params": {"nlist": 1024},
        "search_params": {"nprobe": 16}
    },
    "IVF_SQ8": {
        "index_type": "IVF_SQ8",
        "params": {"nlist": 1024},
        "search_params": {"nprobe": 16}
    },
    "IVF_PQ": {
        "index_type": "IVF_PQ",
        "params": {"nlist": 1024, "m": 8, "nbits": 8},
        "search_params": {"nprobe": 16}
    }
}

# ============================================================================
# 1. 连接 Milvus
# ============================================================================

def connect_milvus():
    """连接到Milvus服务器"""
    print(f"连接到 Milvus {MILVUS_HOST}:{MILVUS_PORT}...")
    connections.connect(alias="default", host=MILVUS_HOST, port=MILVUS_PORT)
    print("✓ 连接成功\n")

# ============================================================================
# 2. 创建Collection with 指定索引
# ============================================================================

def create_collection_with_index(collection_name: str, index_config: Dict):
    """创建带指定索引的Collection"""

    # 删除已存在的collection
    if utility.has_collection(collection_name):
        utility.drop_collection(collection_name)

    # 定义Schema
    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
        FieldSchema(name="product_id", dtype=DataType.VARCHAR, max_length=50),
        FieldSchema(name="category", dtype=DataType.VARCHAR, max_length=50),
        FieldSchema(name="embeddings", dtype=DataType.FLOAT_VECTOR, dim=DIMENSION)
    ]

    schema = CollectionSchema(fields=fields, description=f"Product search with {index_config['index_type']}")
    collection = Collection(name=collection_name, schema=schema)

    # 创建索引
    index_params = {
        "index_type": index_config["index_type"],
        "metric_type": METRIC_TYPE,
        "params": index_config["params"]
    }

    print(f"创建索引: {index_config['index_type']}")
    print(f"  参数: {index_config['params']}")

    start_time = time.time()
    collection.create_index(field_name="embeddings", index_params=index_params)
    build_time = time.time() - start_time

    print(f"✓ 索引构建完成，耗时: {build_time:.2f}秒\n")

    return collection, build_time

# ============================================================================
# 3. 生成并插入测试数据
# ============================================================================

def generate_and_insert_data(collection: Collection, num_vectors: int):
    """生成并插入测试数据"""
    print(f"生成 {num_vectors} 个测试向量...")

    # 生成随机向量（模拟图像embeddings）
    embeddings = np.random.rand(num_vectors, DIMENSION).astype('float32')

    # 生成元数据
    product_ids = [f"SKU_{i:08d}" for i in range(num_vectors)]
    categories = [f"Category_{i % 100}" for i in range(num_vectors)]

    # 批量插入
    batch_size = 5000
    total_inserted = 0
    start_time = time.time()

    for i in range(0, num_vectors, batch_size):
        end_idx = min(i + batch_size, num_vectors)
        data = [
            product_ids[i:end_idx],
            categories[i:end_idx],
            embeddings[i:end_idx].tolist()
        ]
        collection.insert(data)
        total_inserted += (end_idx - i)

    collection.flush()
    insert_time = time.time() - start_time

    print(f"✓ 插入完成: {total_inserted} 条记录")
    print(f"  耗时: {insert_time:.2f}秒")
    print(f"  速率: {total_inserted/insert_time:.0f}条/秒\n")

    return embeddings, insert_time

# ============================================================================
# 4. 性能测试
# ============================================================================

def benchmark_search(collection: Collection, query_vectors: np.ndarray,
                    search_params: Dict, num_queries: int = 100) -> Tuple[float, float, List[float]]:
    """性能基准测试"""

    # 加载collection
    collection.load()

    # 预热
    for _ in range(10):
        collection.search(
            data=[query_vectors[0].tolist()],
            anns_field="embeddings",
            param=search_params,
            limit=10
        )

    # 正式测试
    latencies = []
    start_time = time.time()

    for i in range(num_queries):
        query_start = time.time()
        results = collection.search(
            data=[query_vectors[i % len(query_vectors)].tolist()],
            anns_field="embeddings",
            param=search_params,
            limit=10
        )
        latencies.append((time.time() - query_start) * 1000)

    total_time = time.time() - start_time
    qps = num_queries / total_time
    avg_latency = np.mean(latencies)
    p95_latency = np.percentile(latencies, 95)

    collection.release()

    return qps, avg_latency, p95_latency

# ============================================================================
# 5. 计算召回率
# ============================================================================

def calculate_recall(ground_truth_ids: List, search_result_ids: List, k: int = 10) -> float:
    """计算召回率"""
    gt_set = set(ground_truth_ids[:k])
    result_set = set(search_result_ids[:k])
    recall = len(gt_set & result_set) / k
    return recall

def benchmark_recall(collections: Dict[str, Collection], query_vectors: np.ndarray,
                    search_params_dict: Dict, num_queries: int = 50) -> Dict[str, float]:
    """对比召回率（使用IVF_FLAT作为ground truth）"""

    print("计算召回率（以IVF_FLAT为基准）...")

    # 获取ground truth (IVF_FLAT结果)
    flat_collection = collections["IVF_FLAT"]
    flat_collection.load()

    ground_truth_results = []
    for i in range(num_queries):
        results = flat_collection.search(
            data=[query_vectors[i].tolist()],
            anns_field="embeddings",
            param=search_params_dict["IVF_FLAT"],
            limit=10
        )
        ground_truth_results.append([hit.id for hit in results[0]])

    flat_collection.release()

    # 计算其他索引的召回率
    recalls = {"IVF_FLAT": 1.0}  # 自己对自己100%

    for index_name in ["IVF_SQ8", "IVF_PQ"]:
        collection = collections[index_name]
        collection.load()

        recall_scores = []
        for i in range(num_queries):
            results = collection.search(
                data=[query_vectors[i].tolist()],
                anns_field="embeddings",
                param=search_params_dict[index_name],
                limit=10
            )
            result_ids = [hit.id for hit in results[0]]
            recall = calculate_recall(ground_truth_results[i], result_ids)
            recall_scores.append(recall)

        recalls[index_name] = np.mean(recall_scores)
        collection.release()

    return recalls

# ============================================================================
# 6. 主函数
# ============================================================================

def main():
    """主函数：对比IVF系列索引"""
    print("="*80)
    print("IVF系列索引对比：IVF_FLAT vs IVF_SQ8 vs IVF_PQ")
    print("="*80)
    print()

    # 连接Milvus
    connect_milvus()

    # 存储结果
    results = {}
    collections = {}

    # 为每种索引创建collection并测试
    for index_name, index_config in IVF_CONFIGS.items():
        print(f"{'='*80}")
        print(f"测试索引: {index_name}")
        print(f"{'='*80}")

        collection_name = f"product_search_{index_name.lower()}"

        # 创建collection和索引
        collection, build_time = create_collection_with_index(collection_name, index_config)
        collections[index_name] = collection

        # 插入数据
        embeddings, insert_time = generate_and_insert_data(collection, NUM_VECTORS)

        # 性能测试
        print(f"执行性能测试...")
        query_vectors = embeddings[:100]  # 使用前100个向量作为查询
        search_params = {"metric_type": METRIC_TYPE, **index_config["search_params"]}

        qps, avg_latency, p95_latency = benchmark_search(
            collection, query_vectors, search_params, num_queries=100
        )

        # 计算内存占用（理论值）
        base_memory = NUM_VECTORS * DIMENSION * 4 / (1024**2)  # MB
        if index_name == "IVF_FLAT":
            memory_mb = base_memory
        elif index_name == "IVF_SQ8":
            memory_mb = base_memory / 4
        else:  # IVF_PQ
            memory_mb = base_memory / 8

        # 保存结果
        results[index_name] = {
            "build_time": build_time,
            "insert_time": insert_time,
            "qps": qps,
            "avg_latency": avg_latency,
            "p95_latency": p95_latency,
            "memory_mb": memory_mb
        }

        print(f"✓ {index_name} 测试完成")
        print(f"  QPS: {qps:.2f}")
        print(f"  平均延迟: {avg_latency:.2f}ms")
        print(f"  P95延迟: {p95_latency:.2f}ms")
        print(f"  内存占用: {memory_mb:.2f}MB")
        print()

    # 计算召回率
    query_vectors = embeddings[:50]
    search_params_dict = {
        name: {"metric_type": METRIC_TYPE, **config["search_params"]}
        for name, config in IVF_CONFIGS.items()
    }
    recalls = benchmark_recall(collections, query_vectors, search_params_dict)

    for index_name, recall in recalls.items():
        results[index_name]["recall"] = recall

    # 打印对比结果
    print("\n" + "="*80)
    print("综合对比结果")
    print("="*80)
    print(f"{'指标':<15} {'IVF_FLAT':<15} {'IVF_SQ8':<15} {'IVF_PQ':<15}")
    print("-"*80)

    metrics = [
        ("构建时间(秒)", "build_time", ".2f"),
        ("QPS", "qps", ".2f"),
        ("平均延迟(ms)", "avg_latency", ".2f"),
        ("P95延迟(ms)", "p95_latency", ".2f"),
        ("召回率", "recall", ".2%"),
        ("内存占用(MB)", "memory_mb", ".2f")
    ]

    for metric_name, metric_key, fmt in metrics:
        row = f"{metric_name:<15}"
        for index_name in ["IVF_FLAT", "IVF_SQ8", "IVF_PQ"]:
            value = results[index_name][metric_key]
            row += f" {value:<15{fmt}}"
        print(row)

    print("\n✓ 对比测试完成")

if __name__ == "__main__":
    main()
```

## 5. 关键分析

### 性能权衡

1. **IVF_FLAT**：
   - 优势：最高召回率（100%基准）
   - 劣势：内存占用最大，QPS最低
   - 适用：准确率优先场景

2. **IVF_SQ8**：
   - 优势：平衡方案，召回率94%+，内存减少75%
   - 劣势：略低于FLAT的准确率
   - 适用：生产环境首选

3. **IVF_PQ**：
   - 优势：最快QPS，内存占用最小
   - 劣势：召回率较低（78%）
   - 适用：超大规模、成本敏感场景

### 参数调优建议

```python
# nlist调优（构建时参数）
# 经验公式：nlist ≈ sqrt(n)
if num_vectors < 1_000_000:
    nlist = 1024
elif num_vectors < 10_000_000:
    nlist = 2048
else:
    nlist = 4096

# nprobe调优（查询时参数）
# 召回率要求 → nprobe设置
recall_requirements = {
    0.95: 32,  # 95%召回率 → nprobe=32
    0.90: 16,  # 90%召回率 → nprobe=16
    0.85: 8,   # 85%召回率 → nprobe=8
}

# IVF_PQ的m参数
# m必须是dimension的因数
# 推荐：m = dimension / 2 或 dimension / 4
if DIMENSION == 384:
    m_options = [8, 12, 16, 24]  # 384的因数
```

## 7. 生产环境建议

### 索引选型决策树

```
数据量 < 100万？
├─ 是 → FLAT（无需ANN）
└─ 否 → 内存充足？
    ├─ 是 → 召回率要求 > 95%？
    │   ├─ 是 → IVF_FLAT
    │   └─ 否 → IVF_SQ8
    └─ 否 → 数据量 > 1亿？
        ├─ 是 → IVF_PQ（m=D/4, nbits=8）
        └─ 否 → IVF_SQ8
```
