## 场景概述

本场景演示如何使用 Milvus 2.6 的 Embedding Functions 特性，构建一个最简单的 RAG 文档问答系统。

**核心特性**:
- ✅ Data-in, Data-out 模式：直接插入原始文本
- ✅ 自动向量化：Milvus 自动调用 OpenAI API
- ✅ 语义检索：基于向量相似度搜索
- ✅ 简单高效：3 步完成 RAG 流程

**应用场景**:
- 文档知识库问答
- 技术文档检索
- 客服知识库
- 内容推荐系统

## 完整代码示例

### 步骤 1: 导入库和配置

```python
"""
场景1: 基础数据插入与检索
使用 Milvus 2.6 Embedding Functions 实现文档问答
"""

import os
from dotenv import load_dotenv
from pymilvus import (
    MilvusClient,
    DataType,
    Function,
    FunctionType
)
from openai import OpenAI

# 加载环境变量
load_dotenv()

# 配置
MILVUS_URI = "http://localhost:19530"
COLLECTION_NAME = "doc_qa_collection"
OPENAI_API_KEY = os.getenv("OPENAI_API_KEY")
OPENAI_BASE_URL = os.getenv("OPENAI_BASE_URL", "https://api.openai.com/v1")

# 初始化客户端
milvus_client = MilvusClient(uri=MILVUS_URI)
openai_client = OpenAI(api_key=OPENAI_API_KEY, base_url=OPENAI_BASE_URL)
```

### 步骤 2: 创建 Collection（配置 Embedding Function）

```python
def create_collection():
    """
    创建 Collection 并配置 Embedding Function

    关键点:
    1. 使用 Embedding Function 自动向量化
    2. 配置 OpenAI text-embedding-3-small 模型
    3. 输入字段: document (原始文本)
    4. 输出字段: dense (向量)
    """

    # 如果 Collection 已存在，先删除
    if milvus_client.has_collection(COLLECTION_NAME):
        milvus_client.drop_collection(COLLECTION_NAME)
        print(f"已删除旧 Collection: {COLLECTION_NAME}")

    # 创建 Schema
    schema = milvus_client.create_schema()

    # 添加字段
    schema.add_field(
        field_name="id",
        datatype=DataType.INT64,
        is_primary=True,
        auto_id=True  # 自动生成主键
    )

    schema.add_field(
        field_name="document",
        datatype=DataType.VARCHAR,
        max_length=5000  # 文档最大长度
    )

    schema.add_field(
        field_name="dense",
        datatype=DataType.FLOAT_VECTOR,
        dim=1536  # OpenAI text-embedding-3-small 的维度
    )

    # 配置 Embedding Function（核心特性）
    schema.add_function(
        Function(
            name="openai_embedding",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["document"],  # 输入: 原始文本
            output_field_names=["dense"],    # 输出: 向量
            params={
                "provider": "openai",
                "model_name": "text-embedding-3-small",
                "api_key": OPENAI_API_KEY,
                "base_url": OPENAI_BASE_URL
            }
        )
    )

    # 创建索引参数
    index_params = milvus_client.prepare_index_params()
    index_params.add_index(
        field_name="dense",
        index_type="AUTOINDEX",  # 自动选择最优索引
        metric_type="COSINE"     # 余弦相似度（推荐用于文本）
    )

    # 创建 Collection
    milvus_client.create_collection(
        collection_name=COLLECTION_NAME,
        schema=schema,
        index_params=index_params
    )

    print(f"✅ Collection 创建成功: {COLLECTION_NAME}")
    print(f"   - 自动向量化: OpenAI text-embedding-3-small")
    print(f"   - 向量维度: 1536")
    print(f"   - 度量类型: COSINE")
```

### 步骤 4: 语义检索

```python
def search_documents(query_text, top_k=3):
    """
    语义检索文档

    参数:
        query_text: 查询文本（原始文本，不需要手动向量化）
        top_k: 返回前 K 个最相似的文档

    关键点:
    1. 直接传入原始查询文本
    2. Milvus 自动调用 OpenAI API 生成查询向量
    3. 基于 COSINE 相似度搜索
    """

    print(f"\n🔍 查询: {query_text}")
    print(f"   - Top-{top_k} 最相似文档")

    # 执行搜索（Milvus 自动向量化查询文本）
    results = milvus_client.search(
        collection_name=COLLECTION_NAME,
        data=[query_text],  # 直接传入原始文本
        anns_field="dense",
        limit=top_k,
        output_fields=["document"]  # 返回文档内容
    )

    # 打印结果
    print("\n📄 检索结果:")
    for i, hit in enumerate(results[0], 1):
        print(f"\n{i}. 相似度: {hit['distance']:.4f}")
        print(f"   文档: {hit['entity']['document'][:100]}...")

    return results[0]
```

### 步骤 5: RAG 问答（检索 + 生成）

```python
def rag_question_answering(question):
    """
    RAG 问答流程

    流程:
    1. 语义检索: 从 Milvus 检索相关文档
    2. 上下文构建: 提取文档内容作为上下文
    3. LLM 生成: 基于上下文生成答案

    参数:
        question: 用户问题

    返回:
        answer: LLM 生成的答案
    """

    print(f"\n💬 问题: {question}")

    # 步骤 1: 检索相关文档（Top-3）
    search_results = search_documents(question, top_k=3)

    # 步骤 2: 提取文档内容作为上下文
    contexts = [hit['entity']['document'] for hit in search_results]
    context_text = "\n\n".join(contexts)

    print(f"\n📚 检索到 {len(contexts)} 条相关文档")

    # 步骤 3: 构建 Prompt
    prompt = f"""基于以下上下文回答问题。如果上下文中没有相关信息，请说"我不知道"。

上下文:
{context_text}

问题: {question}

答案:"""

    # 步骤 4: 调用 LLM 生成答案
    print("\n🤖 正在生成答案...")
    response = openai_client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "system", "content": "你是一个专业的技术助手，基于提供的上下文回答问题。"},
            {"role": "user", "content": prompt}
        ],
        temperature=0.3,  # 降低随机性，提高准确性
        max_tokens=500
    )

    answer = response.choices[0].message.content

    print(f"\n✅ 答案:\n{answer}")

    return answer
```

### 步骤 6: 主函数

```python
def main():
    """
    主函数: 完整的 RAG 文档问答流程
    """

    print("=" * 60)
    print("场景1: 基础数据插入与检索")
    print("使用 Milvus 2.6 Embedding Functions")
    print("=" * 60)

    # 1. 创建 Collection
    print("\n【步骤 1】创建 Collection")
    create_collection()

    # 2. 插入文档
    print("\n【步骤 2】插入文档")
    insert_documents()

    # 3. 语义检索测试
    print("\n【步骤 3】语义检索测试")
    search_documents("What is Milvus?", top_k=3)

    # 4. RAG 问答测试
    print("\n【步骤 4】RAG 问答测试")

    questions = [
        "What is the Data-in, Data-out pattern in Milvus 2.6?",
        "How does Milvus improve search performance?",
        "What embedding providers does Milvus support?"
    ]

    for question in questions:
        rag_question_answering(question)
        print("\n" + "-" * 60)

    print("\n✅ 所有测试完成！")

if __name__ == "__main__":
    main()
```

## 代码解析

### 核心特性 1: Embedding Function 配置

```python
schema.add_function(
    Function(
        name="openai_embedding",
        function_type=FunctionType.TEXTEMBEDDING,
        input_field_names=["document"],  # 输入字段
        output_field_names=["dense"],    # 输出字段
        params={
            "provider": "openai",
            "model_name": "text-embedding-3-small",
            "api_key": OPENAI_API_KEY
        }
    )
)
```

**关键点**:
- `input_field_names`: 指定原始文本字段
- `output_field_names`: 指定向量字段
- `provider`: 嵌入模型提供商（OpenAI）
- `model_name`: 具体模型（text-embedding-3-small）

### 核心特性 2: 直接插入原始文本

```python
documents = [
    {"document": "Milvus is an open-source vector database..."}
]

milvus_client.insert(
    collection_name=COLLECTION_NAME,
    data=documents  # 直接插入原始文本
)
```

**传统方式 vs Milvus 2.6**:

| 步骤 | 传统方式 | Milvus 2.6 |
|------|----------|------------|
| 1 | 准备文本 | 准备文本 |
| 2 | 调用嵌入 API | ~~调用嵌入 API~~ |
| 3 | 获取向量 | ~~获取向量~~ |
| 4 | 插入向量 | 插入文本（自动向量化） |
| 5 | 查询时重复 2-3 | 查询文本（自动向量化） |

**简化**: 5 步 → 3 步

### 核心特性 3: 直接查询原始文本

```python
results = milvus_client.search(
```

**关键点**:

- Milvus 自动向量化查询文本
- 保证插入和查询使用同一模型

## 最佳实践

### 1. 批量插入

```python
# ❌ 逐条插入（慢）
for doc in documents:
    milvus_client.insert(COLLECTION_NAME, [doc])

# ✅ 批量插入（快 10 倍）

```
