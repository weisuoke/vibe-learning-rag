## 场景 1: 异步文档加载基础

### 使用 aload()

```python
from langchain_community.document_loaders import WebBaseLoader
import asyncio

async def load_single_url(url: str) -> List[Document]:
    """异步加载单个 URL"""
    loader = WebBaseLoader(url)
    documents = await loader.aload()
    return documents

# 使用
async def main():
    url = "https://example.com"
    documents = await load_single_url(url)
    print(f"加载了 {len(documents)} 个文档")

# 运行
asyncio.run(main())
```

### 使用 alazy_load()

```python
from langchain_community.document_loaders import PyPDFLoader
import asyncio

async def stream_load_pdf(file_path: str):
    """流式异步加载 PDF"""
    loader = PyPDFLoader(file_path)

    async for doc in loader.alazy_load():
        print(f"加载页面: {doc.metadata.get('page', 'unknown')}")
        # 实时处理每一页
        yield doc

# 使用
async def main():
    async for doc in stream_load_pdf("document.pdf"):
        print(f"内容预览: {doc.page_content[:100]}")

asyncio.run(main())
```

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from langchain_community.document_loaders import WebBaseLoader
from typing import List

app = FastAPI(title="Document Loader API")

class LoadRequest(BaseModel):
    url: str

class LoadResponse(BaseModel):
    count: int
    preview: str

@app.post("/load", response_model=LoadResponse)
async def load_document(request: LoadRequest):
    """异步加载文档"""
    try:
        loader = WebBaseLoader(request.url)
        documents = await loader.aload()

        return LoadResponse(
            count=len(documents),
            preview=documents[0].page_content[:200] if documents else ""
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

### 批量加载 API

```python
from fastapi import FastAPI
from pydantic import BaseModel
from typing import List
import asyncio

app = FastAPI()

class BatchLoadRequest(BaseModel):
    urls: List[str]

class BatchLoadResponse(BaseModel):
    total_documents: int
    results: List[dict]

@app.post("/batch-load", response_model=BatchLoadResponse)
async def batch_load_documents(request: BatchLoadRequest):
    """批量异步加载文档"""
    async def load_url(url: str) -> dict:
        try:
            loader = WebBaseLoader(url)
            documents = await loader.aload()
            return {
                "url": url,
                "status": "success",
                "count": len(documents)
            }
        except Exception as e:
            return {
                "url": url,
                "status": "error",
                "error": str(e)
            }

    # 并发加载所有 URL
    results = await asyncio.gather(*[load_url(url) for url in request.urls])

    total_documents = sum(r.get("count", 0) for r in results)

    return BatchLoadResponse(
        total_documents=total_documents,
        results=results
    )
```

## 场景 3: 并发控制策略

### 使用 Semaphore 限制并发

```python
import asyncio
from langchain_community.document_loaders import WebBaseLoader

async def load_with_semaphore(
    urls: List[str],
    max_concurrent: int = 5
) -> List[Document]:
    """使用信号量控制并发数"""
    semaphore = asyncio.Semaphore(max_concurrent)

    async def load_url(url: str) -> List[Document]:
        async with semaphore:
            try:
                loader = WebBaseLoader(url)
                return await loader.aload()
            except Exception as e:
                print(f"加载 {url} 失败: {e}")
                return []

    # 并发加载，但最多同时 5 个
    results = await asyncio.gather(*[load_url(url) for url in urls])

    # 合并所有文档
    all_documents = []
    for docs in results:
        all_documents.extend(docs)

    return all_documents

# 使用
async def main():
    urls = [f"https://example.com/page{i}" for i in range(1, 21)]
    documents = await load_with_semaphore(urls, max_concurrent=5)
    print(f"加载了 {len(documents)} 个文档")

asyncio.run(main())
```

### 使用 aiohttp 自定义并发

```python
import aiohttp
import asyncio
from langchain_core.documents import Document

async def fetch_with_aiohttp(
    urls: List[str],
    max_concurrent: int = 10
) -> List[Document]:
    """使用 aiohttp 实现高性能并发加载"""
    semaphore = asyncio.Semaphore(max_concurrent)

    async def fetch_url(session: aiohttp.ClientSession, url: str) -> Optional[Document]:
        async with semaphore:
            try:
                async with session.get(url, timeout=10) as response:
                    html = await response.text()
                    return Document(
                        page_content=html,
                        metadata={"source": url, "status": response.status}
                    )
            except Exception as e:
                print(f"抓取 {url} 失败: {e}")
                return None

    async with aiohttp.ClientSession() as session:
        tasks = [fetch_url(session, url) for url in urls]
        results = await asyncio.gather(*tasks)

    # 过滤失败的请求
    return [doc for doc in results if doc is not None]

# 使用
async def main():
    urls = [f"https://example.com/page{i}" for i in range(1, 51)]
    documents = await fetch_with_aiohttp(urls, max_concurrent=10)
    print(f"成功抓取 {len(documents)}/{len(urls)} 个网页")

asyncio.run(main())
```

### 分批异步处理

```python
import asyncio
from typing import List, Iterator

async def batch_process_async(
    urls: List[str],
    batch_size: int = 10,
    max_concurrent: int = 5
) -> Iterator[List[Document]]:
    """分批异步处理大量 URL"""
    for i in range(0, len(urls), batch_size):
        batch_urls = urls[i:i + batch_size]
        print(f"处理批次 {i // batch_size + 1}: {len(batch_urls)} 个 URL")

        # 异步加载这一批
        documents = await load_with_semaphore(batch_urls, max_concurrent)

        yield documents

# 使用
async def main():
    urls = [f"https://example.com/page{i}" for i in range(1, 101)]

    async for batch_docs in batch_process_async(urls, batch_size=10):
        print(f"批次完成: {len(batch_docs)} 个文档")
        # 处理这一批文档（例如：向量化、存储）

asyncio.run(main())
```

### 后台任务处理

```python
from fastapi import FastAPI, BackgroundTasks
from pydantic import BaseModel
import asyncio

app = FastAPI()

class BackgroundLoadRequest(BaseModel):
    urls: List[str]
    task_id: str

# 任务状态存储
task_status = {}

async def background_load_task(urls: List[str], task_id: str):
    """后台异步加载任务"""
    task_status[task_id] = {"status": "processing", "progress": 0}

    try:
        documents = await load_with_semaphore(urls, max_concurrent=5)

        task_status[task_id] = {
            "status": "completed",
            "progress": 100,
            "count": len(documents)
        }
    except Exception as e:
        task_status[task_id] = {
            "status": "failed",
            "error": str(e)
        }

@app.post("/background-load")
async def start_background_load(
    request: BackgroundLoadRequest,
    background_tasks: BackgroundTasks
):
    """启动后台加载任务"""
    background_tasks.add_task(
        background_load_task,
        request.urls,
        request.task_id
    )

    return {"task_id": request.task_id, "status": "started"}

@app.get("/task-status/{task_id}")
async def get_task_status(task_id: str):
    """查询任务状态"""
    if task_id not in task_status:
        raise HTTPException(status_code=404, detail="Task not found")

    return task_status[task_id]
```

## 场景 5: 流式响应

### 流式文档加载

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
import asyncio
import json

app = FastAPI()

async def stream_load_documents(urls: List[str]):
    """流式加载文档并返回"""
    for url in urls:
        try:
            loader = WebBaseLoader(url)
            documents = await loader.aload()

            # 流式返回每个文档
            for doc in documents:
                yield json.dumps({
                    "url": url,
                    "content": doc.page_content[:200],
                    "metadata": doc.metadata
                }) + "\n"

        except Exception as e:
            yield json.dumps({
                "url": url,
                "error": str(e)
            }) + "\n"

@app.post("/stream-load")
async def stream_load(urls: List[str]):
    """流式加载 API"""
    return StreamingResponse(
        stream_load_documents(urls),
        media_type="application/x-ndjson"
    )
```

## 场景 6: 生产级异步 RAG 系统

### 完整的异步 RAG API

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_community.vectorstores import Chroma
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.chains import RetrievalQA
from langchain_community.document_loaders import WebBaseLoader
import asyncio
from typing import List

app = FastAPI(title="Async RAG API")

# 全局向量存储
vectorstore = None
qa_chain = None

class IndexRequest(BaseModel):
    urls: List[str]

class QueryRequest(BaseModel):
    query: str
    k: int = 3

class IndexResponse(BaseModel):
    status: str
    documents_count: int
    chunks_count: int

class QueryResponse(BaseModel):
    query: str
    answer: str
    sources: List[str]

@app.on_event("startup")
async def startup_event():
    """初始化向量存储和 QA 链"""
    global vectorstore, qa_chain

    embeddings = OpenAIEmbeddings()
    vectorstore = Chroma(
        embedding_function=embeddings,
        collection_name="async_rag"
    )

    llm = ChatOpenAI(model="gpt-4", temperature=0)
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        chain_type="stuff",
        retriever=vectorstore.as_retriever(),
        return_source_documents=True
    )

@app.post("/index", response_model=IndexResponse)
async def index_documents(request: IndexRequest):
    """异步索引文档"""
    try:
        # 1. 异步加载文档
        documents = await load_with_semaphore(request.urls, max_concurrent=5)

        # 2. 文本分块
        text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=1000,
            chunk_overlap=200
        )
        chunks = text_splitter.split_documents(documents)

        # 3. 添加到向量存储
        vectorstore.add_documents(chunks)

        return IndexResponse(
            status="success",
            documents_count=len(documents),
            chunks_count=len(chunks)
        )

    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/query", response_model=QueryResponse)
async def query_documents(request: QueryRequest):
    """异步查询文档"""
    try:
        # 异步执行查询
        result = await asyncio.to_thread(
            qa_chain.invoke,
            {"query": request.query}
        )

        sources = [
            doc.metadata.get("source", "unknown")
            for doc in result["source_documents"]
        ]

        return QueryResponse(
            query=request.query,
            answer=result["result"],
            sources=sources
        )

    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

# 运行: uvicorn main:app --reload
```

## 场景 7: 性能优化

### 连接池管理

```python
import aiohttp
from typing import Optional

class AsyncDocumentLoader:
    """带连接池的异步文档加载器"""

    def __init__(self, max_connections: int = 100):
        self.connector = aiohttp.TCPConnector(limit=max_connections)
        self.session: Optional[aiohttp.ClientSession] = None

    async def __aenter__(self):
        self.session = aiohttp.ClientSession(connector=self.connector)
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.session.close()

    async def load_url(self, url: str) -> Optional[Document]:
        """加载单个 URL"""
        try:
            async with self.session.get(url, timeout=10) as response:
                html = await response.text()
                return Document(
                    page_content=html,
                    metadata={"source": url}
                )
        except Exception as e:
            print(f"加载 {url} 失败: {e}")
            return None

    async def load_urls(self, urls: List[str]) -> List[Document]:
        """批量加载 URL"""
        tasks = [self.load_url(url) for url in urls]
        results = await asyncio.gather(*tasks)
        return [doc for doc in results if doc is not None]

# 使用
async def main():
    urls = [f"https://example.com/page{i}" for i in range(1, 101)]

    async with AsyncDocumentLoader(max_connections=50) as loader:
        documents = await loader.load_urls(urls)
        print(f"加载了 {len(documents)} 个文档")

asyncio.run(main())
```

### 缓存策略

```python
from functools import lru_cache
import hashlib
import json

class CachedAsyncLoader:
    """带缓存的异步加载器"""

    def __init__(self):
        self.cache = {}

    def _get_cache_key(self, url: str) -> str:
        """生成缓存键"""
        return hashlib.md5(url.encode()).hexdigest()

    async def load_url(self, url: str, use_cache: bool = True) -> Optional[Document]:
        """加载 URL（带缓存）"""
        cache_key = self._get_cache_key(url)

        # 检查缓存
        if use_cache and cache_key in self.cache:
            print(f"缓存命中: {url}")
            return self.cache[cache_key]

        # 加载文档
        loader = WebBaseLoader(url)
        documents = await loader.aload()

        if documents:
            doc = documents[0]
            self.cache[cache_key] = doc
            return doc

        return None

# 使用
async def main():
    loader = CachedAsyncLoader()

    # 第一次加载
    doc1 = await loader.load_url("https://example.com")

    # 第二次加载（使用缓存）
    doc2 = await loader.load_url("https://example.com")

asyncio.run(main())
```

## 场景 8: 错误处理与重试

### 自动重试机制

```python
import asyncio
from typing import Optional

async def load_with_retry(
    url: str,
    max_retries: int = 3,
    backoff_factor: float = 2.0
) -> Optional[Document]:
    """带重试的异步加载"""
    for attempt in range(max_retries):
        try:
            loader = WebBaseLoader(url)
            documents = await loader.aload()
            return documents[0] if documents else None

        except Exception as e:
            if attempt < max_retries - 1:
                wait_time = backoff_factor ** attempt
                print(f"重试 {url} ({attempt + 1}/{max_retries})，等待 {wait_time}秒")
                await asyncio.sleep(wait_time)
            else:
                print(f"加载 {url} 失败，已达最大重试次数: {e}")
                return None

# 使用
async def main():
    url = "https://example.com"
    doc = await load_with_retry(url, max_retries=3)

asyncio.run(main())
```

## 总结

### 异步策略对比

| 策略 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| aload() | 简单易用 | 功能有限 | 单个文档 |
| alazy_load() | 内存友好 | 实现复杂 | 大文件流式处理 |
| asyncio.gather() | 并发高效 | 资源占用高 | 批量加载 |
| Semaphore | 可控并发 | 需要手动管理 | 生产环境 |
| aiohttp | 性能最高 | 实现复杂 | 高并发场景 |

### 最佳实践

1. **并发控制**:
   - 使用 Semaphore 限制并发数
   - 根据服务器性能调整并发数
   - 避免过度并发导致资源耗尽

2. **错误处理**:
   - 添加重试机制
   - 使用指数退避策略
   - 记录失败请求

3. **性能优化**:
   - 使用连接池
   - 实现缓存策略
   - 分批处理大量任务

4. **生产部署**:
   - 使用 FastAPI 构建 API
   - 实现后台任务处理
   - 添加监控和日志
