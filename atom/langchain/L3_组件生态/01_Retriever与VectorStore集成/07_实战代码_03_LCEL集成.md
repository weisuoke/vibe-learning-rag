## 示例1:基础RAG管道

```python
"""
基础RAG管道示例
演示:Retriever + Prompt + LLM的标准组合
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.runnables import RunnablePassthrough
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv

load_dotenv()

# 1. 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
texts = [
    "LangChain is a framework for developing applications powered by LLMs.",
    "LCEL is LangChain Expression Language for building chains.",
    "RAG is Retrieval-Augmented Generation technique.",
    "Vector stores are used to store and retrieve embeddings.",
]
vectorstore.add_texts(texts)

# 2. 创建Retriever
retriever = vectorstore.as_retriever(search_kwargs={"k": 2})

# 3. 定义Prompt
prompt = ChatPromptTemplate.from_template("""
根据以下上下文回答问题:

上下文:
{context}

问题: {question}

回答:
""")

# 4. 格式化函数
def format_docs(docs):
    return "\n\n".join([doc.page_content for doc in docs])

# 5. 构建RAG管道
rag_chain = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | prompt
    | ChatOpenAI(model="gpt-4")
    | StrOutputParser()
)

# 6. 使用管道
question = "What is LangChain?"
answer = rag_chain.invoke(question)
print(f"问题: {question}")
print(f"回答: {answer}")
```

```python
"""
多路检索管道示例
演示:并行使用多个Retriever
"""

from langchain_core.vectorstores import InMemoryVectorStore

vectorstore1 = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore1.add_texts([
    "LangChain is a framework",
    "LCEL is a language"
])

vectorstore2 = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore2.add_texts([
    "RAG is a technique",
    "Vector stores store embeddings"
])

# 创建两个Retriever
retriever1 = vectorstore1.as_retriever(search_kwargs={"k": 1})
retriever2 = vectorstore2.as_retriever(search_kwargs={"k": 1})

# 格式化函数
def format_docs(docs):
    return "\n".join([doc.page_content for doc in docs])

# 构建多路检索管道

根据以下两个来源的上下文回答问题:

来源1:
{context1}

来源2:
{context2}

问题: {question}

回答:
""")

rag_chain = (
    {
        "context1": retriever1 | format_docs,
        "context2": retriever2 | format_docs,
        "question": RunnablePassthrough()
    }
    | prompt
    | ChatOpenAI()
    | StrOutputParser()
)

# 使用
answer = rag_chain.invoke("Tell me about LangChain and RAG")
print(answer)
```

## 示例3:带历史记录的对话RAG

```python
"""
对话RAG示例
演示:集成对话历史
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.runnables import RunnablePassthrough
from langchain_core.output_parsers import StrOutputParser
from langchain_core.messages import HumanMessage, AIMessage
from dotenv import load_dotenv

load_dotenv()

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore.add_texts([
    "LangChain is a framework",
    "LCEL is a language",
    "RAG is a technique"
])

retriever = vectorstore.as_retriever()

# 定义Prompt(包含历史记录)
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是一个有帮助的助手。根据上下文回答问题。"),
    MessagesPlaceholder(variable_name="history"),
    ("human", "上下文:\n{context}\n\n问题: {question}")
])

def format_docs(docs):
    return "\n".join([doc.page_content for doc in docs])

# 构建管道
rag_chain = (
    {
        "context": retriever | format_docs,
        "question": RunnablePassthrough(),
        "history": lambda x: x.get("history", [])
    }
    | prompt
    | ChatOpenAI()
    | StrOutputParser()
)

# 使用(带历史记录)
history = []

# 第一轮对话
question1 = "What is LangChain?"
answer1 = rag_chain.invoke({"question": question1, "history": history})
print(f"Q1: {question1}")
print(f"A1: {answer1}\n")

# 更新历史
history.extend([
    HumanMessage(content=question1),
    AIMessage(content=answer1)
])

# 第二轮对话
question2 = "What about LCEL?"
answer2 = rag_chain.invoke({"question": question2, "history": history})
print(f"Q2: {question2}")
print(f"A2: {answer2}")
```

## 示例6:条件路由RAG

```python
"""
条件路由RAG示例
演示:根据问题类型选择不同的处理路径
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.runnables import RunnableBranch, RunnablePassthrough
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv

load_dotenv()

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore.add_texts([
    "LangChain is a framework",
    "LCEL is a language"
])

retriever = vectorstore.as_retriever()

def format_docs(docs):
    return "\n".join([doc.page_content for doc in docs])

# 定义不同的Prompt
rag_prompt = ChatPromptTemplate.from_template("""
上下文: {context}
问题: {question}
回答:
""")

direct_prompt = ChatPromptTemplate.from_template("""
问题: {question}
回答:
""")

# 判断是否需要检索
def needs_retrieval(x):
    question = x if isinstance(x, str) else x.get("question", "")
    keywords = ["what", "how", "explain", "tell me"]
    return any(kw in question.lower() for kw in keywords)

# 构建条件路由
rag_branch = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | rag_prompt
    | ChatOpenAI()
    | StrOutputParser()
)

direct_branch = (
    {"question": RunnablePassthrough()}
    | direct_prompt
    | ChatOpenAI()
    | StrOutputParser()
)

chain = RunnableBranch(
    (needs_retrieval, rag_branch),
    direct_branch
)

# 测试
print("需要检索的问题:")
answer1 = chain.invoke("What is LangChain?")
print(f"A: {answer1}\n")

print("不需要检索的问题:")
answer2 = chain.invoke("Hello!")
print(f"A: {answer2}")
```

## 示例7:ReRank集成

```python
"""
ReRank集成示例
演示:检索后重排序
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.runnables import RunnablePassthrough, RunnableLambda
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv

load_dotenv()

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore.add_texts([
    "LangChain is a framework for LLM applications",
    "LCEL is LangChain Expression Language",
    "RAG is Retrieval-Augmented Generation",
    "Vector stores store embeddings"
])

retriever = vectorstore.as_retriever(search_kwargs={"k": 4})

# 简单的ReRank函数(实际应用中使用专业的ReRank模型)
def rerank_docs(docs_and_query):
    docs = docs_and_query["docs"]
    query = docs_and_query["query"]
    # 这里简化处理,实际应该使用ReRank模型
    # 按文档长度排序(示例)
    sorted_docs = sorted(docs, key=lambda d: len(d.page_content), reverse=True)
    return sorted_docs[:2]  # 只保留top-2

def format_docs(docs):
    return "\n".join([doc.page_content for doc in docs])

# 构建管道
prompt = ChatPromptTemplate.from_template("""
上下文: {context}
问题: {question}
回答:
""")

rag_chain = (
    {
        "docs": retriever,
        "query": RunnablePassthrough()
    }
    | RunnableLambda(rerank_docs)
    | RunnableLambda(lambda docs: {
        "context": format_docs(docs),
        "question": RunnablePassthrough()
    })
    | prompt
    | ChatOpenAI()
    | StrOutputParser()
)

# 使用
answer = rag_chain.invoke("What is LangChain?")
print(answer)
```
