def step2_node(state: TaskState) -> dict:

class ProductionMonitor:
    """
    生产环境监控器

    实际应用：
    - 性能监控
    - 错误追踪
    - 资源使用监控
    """

    def __init__(self):
        self.metrics = {
            "total_executions": 0,
            "successful_executions": 0,
            "failed_executions": 0,
            "total_time": 0.0
        }

    def record_execution(self, success: bool, duration: float):
        """记录执行"""
        self.metrics["total_executions"] += 1
        if success:
            self.metrics["successful_executions"] += 1
        else:
            self.metrics["failed_executions"] += 1
        self.metrics["total_time"] += duration

    def report(self):
        """生成报告"""
        print("生产环境监控报告:")
        print(f"  总执行次数: {self.metrics['total_executions']}")
        print(f"  成功次数: {self.metrics['successful_executions']}")
        print(f"  失败次数: {self.metrics['failed_executions']}")
        print(f"  成功率: {(self.metrics['successful_executions']/self.metrics['total_executions']*100):.1f}%")
        print(f"  平均执行时间: {(self.metrics['total_time']/self.metrics['total_executions']):.3f}秒")

class RetryHandler:
    """
    重试处理器

    实际应用：
    - 网络错误重试
    - 临时故障恢复
    - 提高系统可靠性
    """

    def __init__(self, max_retries=3):
        self.max_retries = max_retries

    def execute_with_retry(self, app, inputs, config, durability="async"):
        """带重试的执行"""
        for attempt in range(self.max_retries):
            try:
                print(f"尝试 {attempt + 1}/{self.max_retries}...")
                result = app.invoke(inputs, config=config, durability=durability)
                print(f"  成功！")
                return result
            except Exception as e:
                print(f"  失败: {e}")
                if attempt < self.max_retries - 1:
                    wait_time = 2 ** attempt  # 指数退避
                    print(f"  等待 {wait_time}秒后重试...")
                    time.sleep(wait_time)
                else:
                    print(f"  达到最大重试次数，放弃")
                    raise

        },
        {"configurable": {"thread_id": "retry-001"}},

    )

class BenchmarkRunner:
    """
    性能基准测试

    实际应用：
    - 性能评估
    - 配置优化
    - 容量规划
    """

    def __init__(self):
        self.results = []

    def run_benchmark(self, app, durability, num_runs=10):
        """运行基准测试"""
        print(f"基准测试: durability={durability}, runs={num_runs}")

        times = []
        for i in range(num_runs):
            config = {"configurable": {"thread_id": f"bench-{durability}-{i}"}}
            inputs = {
                "task_id": f"bench-task-{i}",
                "step": 0,
                "data": {},
                "status": "pending"
            }

            start = time.time()
            app.invoke(inputs, config=config, durability=durability)
            times.append(time.time() - start)

        avg_time = sum(times) / len(times)
        min_time = min(times)
        max_time = max(times)

        print(f"  平均: {avg_time:.3f}秒")
        print(f"  最小: {min_time:.3f}秒")
        print(f"  最大: {max_time:.3f}秒")
        print()

        return {
            "durability": durability,
            "avg": avg_time,
            "min": min_time,
            "max": max_time
        }

## 关键知识点

### 1. Durability 模式配置

```python
# sync: 同步持久化
result = app.invoke(inputs, config=config, durability="sync")

# async: 异步持久化（默认）
result = app.invoke(inputs, config=config, durability="async")

# exit: 退出时持久化
result = app.invoke(inputs, config=config, durability="exit")
```

### 2. 性能权衡

```
可靠性 ←→ 性能

sync    ████████████ ██
async   ████████     ████████
exit    ████         ████████████

可靠性: sync > async > exit
性能:   exit > async > sync
```

### 3. 错误处理

```python
# 实现重试机制
for attempt in range(max_retries):
    try:
        result = app.invoke(inputs, config=config)
        break
    except Exception as e:
        if attempt < max_retries - 1:
            time.sleep(2 ** attempt)
        else:
            raise
```
