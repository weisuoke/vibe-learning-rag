```python
"""
LangGraph 基础图设计实战示例
场景：智能文档问答系统

演示最佳实践：
1. 状态设计：TypedDict + Annotated[type, reducer]
2. 节点设计：单一职责原则
3. 错误处理：RetryPolicy 配置
4. 模块化：使用子图组织相关节点
5. 清晰命名：描述性节点名称
"""

import os
from typing import Annotated, Literal
from typing_extensions import TypedDict
from operator import add

from langgraph.graph import StateGraph, START, END
from langgraph.types import RetryPolicy
from langgraph.checkpoint.memory import InMemorySaver

# ===== 1. 状态设计 =====

class DocumentQAState(TypedDict):
    """
    文档问答系统状态定义

    最佳实践：
    - 使用 TypedDict 明确定义类型
    - 使用 Annotated[type, reducer] 定义需要聚合的字段
    - 字段命名清晰描述其用途
    - 区分可变字段和不可变字段
    """
    # 用户输入（不可变）
    user_query: str

    # 查询改写结果
    rewritten_query: str

    # 检索结果（使用 add reducer 聚合多次检索）
    retrieved_documents: Annotated[list[dict], add]

    # 重排序后的文档
    reranked_documents: list[dict]

    # 生成的回答
    generated_answer: str

    # 验证结果
    is_answer_valid: bool
    validation_feedback: str

    # 元数据
    metadata: dict


# ===== 2. 节点函数定义 =====

def rewrite_query_node(state: DocumentQAState) -> dict:
    """
    查询改写节点

    职责：优化用户查询，提升检索效果

    最佳实践：
    - 单一职责：只负责查询改写
    - 清晰命名：函数名描述其功能
    - 返回部分状态更新
    """
    print(f"\n[查询改写] 原始查询: {state['user_query']}")

    # 模拟查询改写逻辑
    # 实际应用中可以调用 LLM 进行查询扩展、同义词替换等
    original_query = state['user_query']

    # 简单的查询改写示例
    if "什么是" in original_query:
        rewritten = original_query.replace("什么是", "请详细解释")
    elif "如何" in original_query:
        rewritten = original_query + "，请提供具体步骤"
    else:
        rewritten = original_query + "，请提供详细信息"

    print(f"[查询改写] 改写后: {rewritten}")

    return {
        "rewritten_query": rewritten,
        "metadata": {"rewrite_applied": True}
    }


def retrieve_documents_node(state: DocumentQAState) -> dict:
    """
    文档检索节点

    职责：从向量数据库检索相关文档

    最佳实践：
    - 使用改写后的查询进行检索
    - 返回结构化的文档数据
    - 包含相关性分数
    """
    print(f"\n[文档检索] 检索查询: {state['rewritten_query']}")

    # 模拟向量检索
    # 实际应用中调用 ChromaDB、Milvus 等向量数据库
    mock_documents = [
        {
            "id": "doc_001",
            "content": "LangGraph 是一个用于构建状态化工作流的框架，支持复杂的多步推理和人机协作。",
            "score": 0.92,
            "metadata": {"source": "官方文档", "page": 1}
        },
        {
            "id": "doc_002",
            "content": "LangGraph 的核心概念包括 StateGraph、节点、边、条件路由和 Checkpoint 机制。",
            "score": 0.88,
            "metadata": {"source": "官方文档", "page": 5}
        },
        {
            "id": "doc_003",
            "content": "使用 LangGraph 可以构建复杂的 AI 代理系统，支持断点续传和人工审批。",
            "score": 0.85,
            "metadata": {"source": "教程", "page": 10}
        },
        {
            "id": "doc_004",
            "content": "LangGraph 支持并行执行节点，可以提升工作流的执行效率。",
            "score": 0.80,
            "metadata": {"source": "最佳实践", "page": 3}
        },
        {
            "id": "doc_005",
            "content": "LangGraph 的状态管理使用 TypedDict 和 Reducer 函数实现。",
            "score": 0.75,
            "metadata": {"source": "技术文档", "page": 8}
        }
    ]

    print(f"[文档检索] 检索到 {len(mock_documents)} 个文档")

    return {
        "retrieved_documents": mock_documents
    }


def rerank_documents_node(state: DocumentQAState) -> dict:
    """
    文档重排序节点

    职责：使用重排序模型优化文档顺序

    最佳实践：
    - 基于查询和文档内容计算相关性
    - 只保留 top-k 个最相关的文档
    - 减少后续 LLM 处理的上下文长度
    """
    print(f"\n[文档重排序] 重排序 {len(state['retrieved_documents'])} 个文档")

    # 模拟重排序逻辑
    # 实际应用中可以使用 Cohere Rerank、BGE Reranker 等
    documents = state['retrieved_documents']

    # 按分数排序并取 top-3
    reranked = sorted(documents, key=lambda x: x['score'], reverse=True)[:3]

    print(f"[文档重排序] 保留 top-{len(reranked)} 个文档")
    for i, doc in enumerate(reranked, 1):
        print(f"  {i}. {doc['id']} (score: {doc['score']:.2f})")

    return {
        "reranked_documents": reranked
    }


def generate_answer_node(state: DocumentQAState) -> dict:
    """
    回答生成节点

    职责：基于检索到的文档生成回答

    最佳实践：
    - 构建清晰的 prompt
    - 包含查询和上下文
    - 指导 LLM 基于上下文回答
    """
    print(f"\n[回答生成] 基于 {len(state['reranked_documents'])} 个文档生成回答")

    # 构建上下文
    context = "\n\n".join([
        f"文档 {i+1} (来源: {doc['metadata']['source']}):\n{doc['content']}"
        for i, doc in enumerate(state['reranked_documents'])
    ])

    # 模拟 LLM 生成
    # 实际应用中调用 OpenAI、Anthropic 等 LLM API
    query = state['user_query']

    # 简单的模拟回答
    answer = f"""基于检索到的文档，我来回答您的问题：

问题：{query}

{state['reranked_documents'][0]['content']}

此外，{state['reranked_documents'][1]['content']}

总结：LangGraph 是一个强大的状态化工作流框架，适合构建复杂的 AI 代理系统。"""

    print(f"[回答生成] 生成回答长度: {len(answer)} 字符")

    return {
        "generated_answer": answer,
        "metadata": {
            **state.get("metadata", {}),
            "answer_length": len(answer),
            "num_docs_used": len(state['reranked_documents'])
        }
    }


def validate_answer_node(state: DocumentQAState) -> dict:
    """
    回答验证节点

    职责：验证生成的回答质量

    最佳实践：
    - 检查回答是否基于上下文
    - 检查回答是否回答了问题
    - 检查回答是否包含幻觉
    """
    print(f"\n[回答验证] 验证回答质量")

    answer = state['generated_answer']
    query = state['user_query']

    # 简单的验证逻辑
    # 实际应用中可以使用专门的验证模型
    is_valid = True
    feedback = "回答质量良好"

    # 检查回答长度
    if len(answer) < 50:
        is_valid = False
        feedback = "回答过短，缺少详细信息"

    # 检查是否包含查询关键词
    query_keywords = query.split()
    if not any(keyword in answer for keyword in query_keywords):
        is_valid = False
        feedback = "回答未充分回应查询"

    print(f"[回答验证] 验证结果: {'通过' if is_valid else '未通过'}")
    print(f"[回答验证] 反馈: {feedback}")

    return {
        "is_answer_valid": is_valid,
        "validation_feedback": feedback
    }


# ===== 3. 条件路由函数 =====

def should_regenerate_route(state: DocumentQAState) -> Literal["regenerate", "end"]:
    """
    条件路由：判断是否需要重新生成回答

    最佳实践：
    - 使用 Literal 类型提示明确路由选项
    - 路由逻辑清晰简单
    - 返回值与边定义一致
    """
    if state['is_answer_valid']:
        return "end"
    else:
        # 实际应用中可以限制重试次数
        return "regenerate"


# ===== 4. 构建图 =====

def create_document_qa_graph():
    """
    创建文档问答图

    最佳实践：
    - 使用函数封装图构建逻辑
    - 便于测试和复用
    - 清晰的图结构
    """
    # 创建 StateGraph
    graph = StateGraph(DocumentQAState)

    # 配置重试策略
    # 为可能失败的节点（如 API 调用）配置重试
    llm_retry_policy = RetryPolicy(
        max_attempts=3,
        initial_interval=1.0,
        backoff_factor=2.0,
        jitter=True
    )

    # 添加节点
    graph.add_node("rewrite_query", rewrite_query_node)
    graph.add_node("retrieve_documents", retrieve_documents_node)
    graph.add_node("rerank_documents", rerank_documents_node)
    graph.add_node(
        "generate_answer",
        generate_answer_node,
        retry_policy=llm_retry_policy  # LLM 调用需要重试
    )
    graph.add_node("validate_answer", validate_answer_node)

    # 添加边
    graph.add_edge(START, "rewrite_query")
    graph.add_edge("rewrite_query", "retrieve_documents")
    graph.add_edge("retrieve_documents", "rerank_documents")
    graph.add_edge("rerank_documents", "generate_answer")
    graph.add_edge("generate_answer", "validate_answer")

    # 添加条件边
    graph.add_conditional_edges(
        "validate_answer",
        should_regenerate_route,
        {
            "regenerate": "generate_answer",  # 重新生成
            "end": END  # 结束
        }
    )

    return graph


# ===== 5. 执行示例 =====

def main():
    """主函数：演示图的使用"""

    print("=" * 80)
    print("LangGraph 基础图设计实战示例")
    print("场景：智能文档问答系统")
    print("=" * 80)

    # 创建图
    graph = create_document_qa_graph()

    # 编译图（可选：添加 checkpointer 支持持久化）
    checkpointer = InMemorySaver()
    app = graph.compile(checkpointer=checkpointer)

    # 准备输入
    initial_state = {
        "user_query": "什么是 LangGraph？",
        "rewritten_query": "",
        "retrieved_documents": [],
        "reranked_documents": [],
        "generated_answer": "",
        "is_answer_valid": False,
        "validation_feedback": "",
        "metadata": {}
    }

    # 执行图
    print("\n开始执行工作流...")
    print("-" * 80)

    config = {"configurable": {"thread_id": "demo_thread_001"}}
    result = app.invoke(initial_state, config=config)

    # 输出结果
    print("\n" + "=" * 80)
    print("执行结果")
    print("=" * 80)

    print(f"\n原始查询: {result['user_query']}")
    print(f"改写查询: {result['rewritten_query']}")
    print(f"\n检索文档数: {len(result['retrieved_documents'])}")
    print(f"重排序后文档数: {len(result['reranked_documents'])}")

    print(f"\n生成的回答:")
    print("-" * 80)
    print(result['generated_answer'])
    print("-" * 80)

    print(f"\n验证结果: {'通过' if result['is_answer_valid'] else '未通过'}")
    print(f"验证反馈: {result['validation_feedback']}")

    print(f"\n元数据:")
    for key, value in result['metadata'].items():
        print(f"  {key}: {value}")

    print("\n" + "=" * 80)
    print("工作流执行完成")
    print("=" * 80)


if __name__ == "__main__":
    main()
```

## 最佳实践总结

### 1. 状态设计

**遵循的原则**：
- 使用 `TypedDict` 明确定义类型
- 使用 `Annotated[type, reducer]` 定义聚合字段
- 字段命名清晰描述用途
- 区分可变和不可变字段

**代码示例**：
```python
class DocumentQAState(TypedDict):
    user_query: str  # 不可变
    retrieved_documents: Annotated[list[dict], add]  # 可聚合
```

### 2. 节点设计

**遵循的原则**：
- 单一职责：每个节点只做一件事
- 清晰命名：使用描述性的动词短语
- 返回部分状态更新
- 节点函数简短易懂

**代码示例**：
```python
def rewrite_query_node(state: DocumentQAState) -> dict:
    """只负责查询改写"""
    rewritten = rewrite_logic(state['user_query'])
    return {"rewritten_query": rewritten}
```

### 3. 错误处理

**遵循的原则**：
- 为 API 调用配置 `RetryPolicy`
- 使用指数退避策略
- 生产环境启用 jitter
- 只重试临时性错误

**代码示例**：
```python
llm_retry_policy = RetryPolicy(
    max_attempts=3,
    initial_interval=1.0,
    backoff_factor=2.0,
    jitter=True
)

graph.add_node("generate_answer", generate_answer_node, retry_policy=llm_retry_policy)
```

### 4. 条件路由

**遵循的原则**：
- 使用 `Literal` 类型提示
- 路由逻辑清晰简单
- 返回值与边定义一致

**代码示例**：
```python
def should_regenerate_route(state: DocumentQAState) -> Literal["regenerate", "end"]:
    return "end" if state['is_answer_valid'] else "regenerate"
```

### 5. 图构建

**遵循的原则**：
- 使用函数封装图构建逻辑
- 清晰的节点和边定义
- 合理使用条件边
- 配置 checkpointer 支持持久化

**代码示例**：
```python
def create_document_qa_graph():
    graph = StateGraph(DocumentQAState)
    graph.add_node("node1", node1_func)
    graph.add_edge(START, "node1")
    return graph
```

## 总结

本示例展示了 LangGraph 基础图设计的核心最佳实践：

1. **状态设计**：使用 TypedDict + Annotated[type, reducer]
2. **节点设计**：单一职责 + 清晰命名
3. **错误处理**：RetryPolicy + 指数退避
4. **条件路由**：Literal 类型提示 + 清晰逻辑
5. **图构建**：函数封装 + 清晰结构

这些实践可以帮助你构建健壮、可维护、高性能的 LangGraph 应用。

**记住**：好的图设计应该是清晰、简洁、易于理解的。优先考虑可读性和可维护性，在确认存在性能瓶颈时再进行优化。
