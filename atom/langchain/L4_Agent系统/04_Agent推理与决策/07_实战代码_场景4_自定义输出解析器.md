```python
load_dotenv()

# ===== 1. 自定义输出解析器 =====
class ChineseReActOutputParser(AgentOutputParser):
    """
    自定义中文 ReAct 输出解析器

    支持的格式：
    - 中文关键词：思考、动作、动作输入、最终答案
    - 英文关键词：Thought、Action、Action Input、Final Answer
    - 混合格式：支持中英文混用

    特点：
    - 更宽松的格式匹配（支持空格、换行变化）
    - 详细的错误提示（中文）
    - 自修正机制（send_to_llm=True）
    """

    def parse(self, text: str) -> Union[AgentAction, AgentFinish]:
        """
        解析 LLM 输出文本，返回 AgentAction 或 AgentFinish

        Args:
            text: LLM 的原始输出文本

        Returns:
            AgentAction: 需要调用工具
            AgentFinish: 任务完成，返回最终答案

        Raises:
            OutputParserException: 解析失败时抛出，包含错误提示
        """

        # 第一步：检查是否是最终答案（优先级最高）
        # 支持中英文关键词
        final_answer_patterns = [
            r"最终答案[:：]\s*(.+)",
            r"Final\s+Answer[:：]\s*(.+)",
        ]

        for pattern in final_answer_patterns:
            match = re.search(pattern, text, re.IGNORECASE | re.DOTALL)
            if match:
                answer = match.group(1).strip()
                return AgentFinish(
                    return_values={"output": answer},
                    log=text
                )

        # 第二步：解析 Action 和 Action Input
        # 支持中英文关键词，支持多种格式变体
        action_patterns = [
            # 中文格式：动作: xxx\n动作输入: xxx
            (r"动作[:：]\s*(.+?)\s*\n\s*动作输入[:：]\s*(.+)", "中文格式"),
            # 英文格式：Action: xxx\nAction Input: xxx
            (r"Action\s*\d*\s*[:：]\s*(.+?)\s*\n\s*Action\s*Input\s*\d*\s*[:：]\s*(.+)", "英文格式"),
            # 混合格式：Action: xxx\n动作输入: xxx
            (r"(?:Action|动作)\s*\d*\s*[:：]\s*(.+?)\s*\n\s*(?:Action\s*Input|动作输入)\s*\d*\s*[:：]\s*(.+)", "混合格式"),
        ]

        for pattern, format_name in action_patterns:
            match = re.search(pattern, text, re.IGNORECASE | re.DOTALL)
            if match:
                tool = match.group(1).strip()
                tool_input = match.group(2).strip()

                # 移除可能的引号
                tool_input = tool_input.strip('"').strip("'")

                return AgentAction(
                    tool=tool,
                    tool_input=tool_input,
                    log=text
                )

        # 第三步：解析失败，生成详细的错误提示
        # 检查是否包含 Thought/思考（说明 LLM 在推理，但格式不对）
        has_thought = bool(re.search(r"(?:Thought|思考)\s*[:：]", text, re.IGNORECASE))

        if has_thought:
            # LLM 在推理，但没有正确输出 Action
            error_message = (
                f"解析失败：找到了推理过程，但缺少动作指令。\n\n"
                f"你的输出：\n{text}\n\n"
                f"正确格式示例：\n"
                f"思考：我需要搜索相关信息\n"
                f"动作：search_tool\n"
                f"动作输入：搜索关键词\n\n"
                f"或者如果已经有答案：\n"
                f"思考：我现在知道答案了\n"
                f"最终答案：这是最终答案"
            )
        else:
            # LLM 完全没有按格式输出
            error_message = (
                f"解析失败：输出格式不正确。\n\n"
                f"你的输出：\n{text}\n\n"
                f"请严格按照以下格式输出：\n"
                f"思考：[你的推理过程]\n"
                f"动作：[工具名称]\n"
                f"动作输入：[工具的输入参数]\n\n"
                f"或者：\n"
                f"思考：[你的推理过程]\n"
                f"最终答案：[最终答案]"
            )

        # 抛出异常，触发自修正机制
        raise OutputParserException(
            error_message,
            observation=error_message,  # 这段文字会被发送给 LLM
            llm_output=text,
            send_to_llm=True  # 关键！让 LLM 看到错误并重新生成
        )

    def parse_with_prompt(self, completion: str, prompt: str) -> Union[AgentAction, AgentFinish]:
        """
        带 prompt 上下文的解析（可选实现）

        在某些场景下，解析器需要知道原始 prompt 才能正确解析输出。
        例如：根据 prompt 中的工具列表验证 LLM 选择的工具是否存在。
        """
        # 先尝试正常解析
        try:
            return self.parse(completion)
        except OutputParserException as e:
            # 解析失败时，可以利用 prompt 信息提供更好的错误提示
            # 例如：从 prompt 中提取可用工具列表
            tools_match = re.search(r"可用工具[:：]\s*(.+?)(?:\n\n|\Z)", prompt, re.DOTALL)
            if tools_match:
                available_tools = tools_match.group(1).strip()
                enhanced_error = (
                    f"{e.observation}\n\n"
                    f"可用的工具列表：\n{available_tools}"
                )
                raise OutputParserException(
                    enhanced_error,
                    observation=enhanced_error,
                    llm_output=completion,
                    send_to_llm=True
                )
            else:
                # 没有额外信息，直接抛出原始异常
                raise


# ===== 2. 定义工具 =====
@tool
def search_tool(query: str) -> str:
    """搜索知识库。输入应该是搜索关键词。"""
    kb = {
        "python": "Python 是一种高级编程语言，由 Guido van Rossum 于 1991 年创建。",
        "langchain": "LangChain 是用于构建 LLM 应用的开源框架，支持 Agent、RAG 等模式。",
        "react": "ReAct 是 Reasoning + Acting 的缩写，通过 Thought→Action→Observation 循环解决问题。",
    }
    query_lower = query.lower()
    for key, value in kb.items():
        if key in query_lower:
            return value
    return f"未找到 '{query}' 的相关信息"


@tool
def calculator_tool(expression: str) -> str:
    """计算数学表达式。输入应该是有效的数学表达式。"""
    try:
        import math
        result = eval(expression, {"__builtins__": {}}, {
            "sqrt": math.sqrt, "pow": pow, "abs": abs
        })
        return f"计算结果：{result}"
    except Exception as e:
        return f"计算错误：{e}"


tools = [search_tool, calculator_tool]


# ===== 3. 构建中文 ReAct Prompt =====
chinese_prompt_template = """你是一个智能助手，可以使用工具来回答问题。

可用工具：
{tools}

请严格按照以下格式回答：

问题：用户的问题
思考：分析问题，决定下一步做什么
动作：选择一个工具，必须是 [{tool_names}] 中的一个
动作输入：工具的输入参数
观察：工具的返回结果
... (可以重复 思考/动作/动作输入/观察 多次)
思考：我现在知道最终答案了
最终答案：对原始问题的最终答案

开始！

问题：{input}
思考：{agent_scratchpad}"""

prompt = PromptTemplate.from_template(chinese_prompt_template)


# ===== 4. 手动构建 Agent（使用自定义解析器）=====
# 注意：不能用 create_react_agent，因为它使用内置解析器
# 我们需要手动构建 LCEL 链

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

# 绑定 stop sequence
llm_with_stop = llm.bind(stop=["\n观察", "\nObservation"])

# 格式化 scratchpad
def format_scratchpad(intermediate_steps):
    """将中间步骤格式化为字符串"""
    thoughts = ""
    for action, observation in intermediate_steps:
        thoughts += action.log
        thoughts += f"\n观察：{observation}\n思考："
    return thoughts

# 构建 LCEL 链
agent = (
    RunnablePassthrough.assign(
        agent_scratchpad=lambda x: format_scratchpad(x.get("intermediate_steps", []))
    )
    | prompt
    | llm_with_stop
    | ChineseReActOutputParser()  # 使用自定义解析器
)

# 包装为 RunnableAgent
runnable_agent = RunnableAgent(
    runnable=agent,
    input_keys_arg=["input"],
    return_keys_arg=["output"],
)


# ===== 5. 创建 AgentExecutor =====
agent_executor = AgentExecutor(
    agent=runnable_agent,
    tools=tools,
    verbose=True,
    max_iterations=5,
    handle_parsing_errors=True,  # 启用自修正机制
    return_intermediate_steps=True,
)
```

## 关键知识点解析

### 1. 自定义解析器的核心结构

```python
class CustomOutputParser(AgentOutputParser):
    """自定义解析器必须继承 AgentOutputParser"""

    def parse(self, text: str) -> Union[AgentAction, AgentFinish]:
        """
        核心方法：解析 LLM 输出

        返回值只有两种可能：
        - AgentAction：需要调用工具，循环继续
        - AgentFinish：任务完成，循环结束
        """
        # 1. 检查是否是最终答案（优先级最高）
        if "Final Answer:" in text:
            return AgentFinish(...)

        # 2. 解析 Action 和 Action Input
        if match_action_pattern(text):
            return AgentAction(...)

        # 3. 解析失败，抛出异常触发自修正
        raise OutputParserException(
            "错误提示",
            send_to_llm=True  # 关键！
        )
```

### 2. 自修正机制的工作流程

```
第 1 轮：
LLM 输出：我觉得应该搜索一下  ← 格式不对
    ↓
Parser：解析失败！
    ↓
抛出 OutputParserException(send_to_llm=True)
    ↓
AgentExecutor 捕获异常，将错误提示作为 Observation
    ↓
Scratchpad 更新：
  Thought: 我觉得应该搜索一下
  Observation: 解析失败：输出格式不正确。请严格按照以下格式...
    ↓
第 2 轮：
LLM 看到错误提示，重新生成：
  思考：我需要搜索 Python
  动作：search_tool
  动作输入：Python  ← 格式正确了！
    ↓
Parser：解析成功！返回 AgentAction
```

### 3. parse() vs parse_with_prompt()

```python
# parse() - 基础解析
def parse(self, text: str):
    """只看 LLM 输出，不看 prompt"""
    return parse_logic(text)

# parse_with_prompt() - 上下文感知解析
def parse_with_prompt(self, completion: str, prompt: str):
    """
    同时看 LLM 输出和原始 prompt

    使用场景：
    - 验证 LLM 选择的工具是否在 prompt 的工具列表中
    - 根据 prompt 中的格式要求调整解析逻辑
    - 提供更详细的错误提示（包含 prompt 中的信息）
    """
    try:
        return self.parse(completion)
    except OutputParserException:
        # 从 prompt 中提取额外信息，增强错误提示
        tools = extract_tools_from_prompt(prompt)
        raise OutputParserException(
            f"工具选择错误。可用工具：{tools}"
        )
```

### 4. 与内置解析器的对比

| 维度 | 内置 ReActSingleInput | 自定义 ChineseReAct |
|------|----------------------|---------------------|
| 格式支持 | 仅英文 `Action:` | 中英文混用 |
| 格式宽松度 | 严格（空格敏感） | 宽松（正则灵活匹配） |
| 错误提示 | 英文通用提示 | 中文详细提示 |
| 扩展性 | 不可修改 | 完全可定制 |
| 适用场景 | 标准英文 ReAct | 中文、特殊格式 |

## 常见问题与解决方案

### Q1：自定义解析器不生效？

```python
# ❌ 错误：使用 create_react_agent（它会强制使用内置解析器）
agent = create_react_agent(llm, tools, prompt)

# ✅ 正确：手动构建 LCEL 链
agent = (
    RunnablePassthrough.assign(agent_scratchpad=...)
    | prompt
    | llm_with_stop
    | CustomOutputParser()  # 使用自定义解析器
)
```

### Q2：如何调试解析器？

```python
parser = ChineseReActOutputParser()
```

```python
agent_executor = AgentExecutor(
    agent=agent,
    tools=tools,
    handle_parsing_errors=True,  # 必须设为 True
```

## 小结

| 要点 | 内容 |
|------|------|
| 核心方法 | `parse()` 和 `parse_with_prompt()` |
| 返回类型 | `AgentAction` 或 `AgentFinish` |
| 自修正机制 | `raise OutputParserException(send_to_llm=True)` |
| 使用场景 | 多语言、特殊格式、自定义决策逻辑 |
| 关键区别 | 内置解析器不可修改，自定义解析器完全可控 |
