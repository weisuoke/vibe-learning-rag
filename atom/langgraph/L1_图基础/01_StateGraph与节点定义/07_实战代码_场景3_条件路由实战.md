## 完整实战代码

```python
"""
LangGraph 条件路由实战示例
演示：内容审核工作流 - 多分支路由 + 循环执行

场景说明：
一个智能内容审核系统，根据内容质量和安全性进行多轮审核和修改
- 初始检查：判断内容是否需要审核
- 安全审核：检查内容安全性
- 质量审核：检查内容质量
- 自动修复：尝试自动修复问题
- 人工审核：复杂问题转人工处理
- 循环执行：直到通过或达到最大重试次数

来源：
- 条件路由模式：Context7 官方文档
- 循环执行设计：Twitter 最佳实践
- 多分支路由：源码分析 + 社区案例
"""

import operator
from typing import Annotated, Literal, TypedDict
from langgraph.graph import StateGraph, START, END


# ===== 1. 定义状态 Schema =====
class ContentState(TypedDict):
    """内容审核状态

    来源：Context7 官方文档 - State 定义规范
    """
    content: str                                    # 待审核内容
    safety_score: float                             # 安全评分 (0-1)
    quality_score: float                            # 质量评分 (0-1)
    issues: Annotated[list[str], operator.add]      # 问题列表（累积）
    retry_count: int                                # 重试次数
    status: str                                     # 当前状态
    history: Annotated[list[str], operator.add]     # 审核历史（累积）


# ===== 2. 定义节点函数 =====

def initial_check(state: ContentState) -> dict:
    """初始检查节点

    判断内容是否需要审核
    """
    content = state["content"]

    print(f"\n[初始检查] 内容长度: {len(content)} 字符")

    # 简单规则：空内容或过短内容直接拒绝
    if not content or len(content) < 10:
        return {
            "status": "rejected",
            "issues": ["内容过短或为空"],
            "history": ["初始检查: 拒绝"]
        }

    # 正常内容进入审核流程
    return {
        "status": "pending",
        "history": ["初始检查: 通过"]
    }


def safety_check(state: ContentState) -> dict:
    """安全审核节点

    检查内容安全性
    来源：实际业务场景抽象
    """
    content = state["content"]

    print(f"[安全审核] 检查内容安全性...")

    # 模拟安全检查（实际应调用安全审核 API）
    unsafe_keywords = ["暴力", "色情", "赌博", "违法"]
    found_issues = [kw for kw in unsafe_keywords if kw in content]

    if found_issues:
        safety_score = 0.3
        return {
            "safety_score": safety_score,
            "issues": [f"发现敏感词: {', '.join(found_issues)}"],
            "status": "unsafe",
            "history": [f"安全审核: 不通过 (评分: {safety_score})"]
        }
    else:
        safety_score = 0.95
        return {
            "safety_score": safety_score,
            "status": "safe",
            "history": [f"安全审核: 通过 (评分: {safety_score})"]
        }


def quality_check(state: ContentState) -> dict:
    """质量审核节点

    检查内容质量
    """
    content = state["content"]

    print(f"[质量审核] 检查内容质量...")

    # 模拟质量检查
    quality_score = min(1.0, len(content) / 100)  # 简单规则：长度越长质量越高

    if quality_score < 0.5:
        return {
            "quality_score": quality_score,
            "issues": ["内容质量不足"],
            "status": "low_quality",
            "history": [f"质量审核: 不通过 (评分: {quality_score:.2f})"]
        }
    else:
        return {
            "quality_score": quality_score,
            "status": "quality_ok",
            "history": [f"质量审核: 通过 (评分: {quality_score:.2f})"]
        }


def auto_fix(state: ContentState) -> dict:
    """自动修复节点

    尝试自动修复内容问题
    """
    content = state["content"]
    retry_count = state.get("retry_count", 0)

    print(f"[自动修复] 第 {retry_count + 1} 次尝试修复...")

    # 模拟自动修复（实际应调用 LLM 进行内容改写）
    fixed_content = content + " [已优化]"

    return {
        "content": fixed_content,
        "retry_count": retry_count + 1,
        "status": "fixed",
        "history": [f"自动修复: 第 {retry_count + 1} 次尝试"]
    }


def human_review(state: ContentState) -> dict:
    """人工审核节点

    复杂问题转人工处理
    """
    print(f"[人工审核] 转人工处理...")

    return {
        "status": "human_review",
        "history": ["转人工审核"]
    }


def approve(state: ContentState) -> dict:
    """批准节点

    内容通过审核
    """
    print(f"[批准] 内容审核通过！")

    return {
        "status": "approved",
        "history": ["最终结果: 批准"]
    }


def reject(state: ContentState) -> dict:
    """拒绝节点

    内容未通过审核
    """
    print(f"[拒绝] 内容审核不通过")

    return {
        "status": "rejected",
        "history": ["最终结果: 拒绝"]
    }


# ===== 3. 定义路由函数 =====

def route_after_initial(state: ContentState) -> Literal["safety_check", "reject"]:
    """初始检查后的路由

    来源：Context7 官方文档 - 条件路由示例
    """
    if state["status"] == "rejected":
        return "reject"
    else:
        return "safety_check"


def route_after_safety(state: ContentState) -> Literal["quality_check", "human_review"]:
    """安全审核后的路由

    多分支路由示例
    """
    if state["status"] == "unsafe":
        # 安全问题直接转人工
        return "human_review"
    else:
        # 安全通过，继续质量审核
        return "quality_check"


def route_after_quality(state: ContentState) -> Literal["approve", "auto_fix"]:
    """质量审核后的路由"""
    if state["status"] == "quality_ok":
        return "approve"
    else:
        return "auto_fix"


def route_after_fix(state: ContentState) -> Literal["safety_check", "human_review", END]:
    """自动修复后的路由

    循环执行示例：修复后重新审核，或达到最大重试次数
    来源：Twitter 最佳实践 - 循环执行设计
    """
    retry_count = state.get("retry_count", 0)
    max_retries = 3

    if retry_count >= max_retries:
        # 达到最大重试次数，转人工
        print(f"  → 已达到最大重试次数 ({max_retries})，转人工审核")
        return "human_review"
    else:
        # 重新进入审核流程
        print(f"  → 重新进入审核流程 (第 {retry_count} 次)")
        return "safety_check"


def route_after_human(state: ContentState) -> Literal[END]:
    """人工审核后的路由

    人工审核是终点
    """
    return END


# ===== 4. 构建图 =====

def create_content_review_graph() -> StateGraph:
    """创建内容审核图

    来源：Context7 官方文档 - StateGraph 构建模式
    """
    # 创建 StateGraph
    builder = StateGraph(ContentState)

    # 添加节点
    builder.add_node("initial_check", initial_check)
    builder.add_node("safety_check", safety_check)
    builder.add_node("quality_check", quality_check)
    builder.add_node("auto_fix", auto_fix)
    builder.add_node("human_review", human_review)
    builder.add_node("approve", approve)
    builder.add_node("reject", reject)

    # 添加边
    builder.add_edge(START, "initial_check")

    # 添加条件边（核心：多分支路由）
    builder.add_conditional_edges("initial_check", route_after_initial)
    builder.add_conditional_edges("safety_check", route_after_safety)
    builder.add_conditional_edges("quality_check", route_after_quality)
    builder.add_conditional_edges("auto_fix", route_after_fix)  # 循环执行
    builder.add_conditional_edges("human_review", route_after_human)

    # 终点边
    builder.add_edge("approve", END)
    builder.add_edge("reject", END)

    return builder


# ===== 5. 测试示例 =====

def test_content_review():
    """测试内容审核工作流"""

    # 编译图
    builder = create_content_review_graph()
    graph = builder.compile()

    print("=" * 60)
    print("LangGraph 条件路由实战 - 内容审核工作流")
    print("=" * 60)

    # 测试用例 1：正常内容
    print("\n【测试用例 1】正常内容")
    print("-" * 60)
    result1 = graph.invoke({
        "content": "这是一篇关于人工智能技术发展的文章，内容详实，观点客观。人工智能正在改变我们的生活方式。",
        "safety_score": 0.0,
        "quality_score": 0.0,
        "issues": [],
        "retry_count": 0,
        "status": "init",
        "history": []
    })

    print("\n最终状态:", result1["status"])
    print("审核历史:")
    for h in result1["history"]:
        print(f"  - {h}")

    # 测试用例 2：质量不足的内容（触发循环）
    print("\n\n【测试用例 2】质量不足的内容（触发自动修复循环）")
    print("-" * 60)
    result2 = graph.invoke({
        "content": "短文本",
        "safety_score": 0.0,
        "quality_score": 0.0,
        "issues": [],
        "retry_count": 0,
        "status": "init",
        "history": []
    })

    print("\n最终状态:", result2["status"])
    print("重试次数:", result2["retry_count"])
    print("审核历史:")
    for h in result2["history"]:
        print(f"  - {h}")

    # 测试用例 3：不安全内容（转人工）
    print("\n\n【测试用例 3】不安全内容（转人工审核）")
    print("-" * 60)
    result3 = graph.invoke({
        "content": "这是一篇包含暴力内容的文章，不适合发布。",
        "safety_score": 0.0,
        "quality_score": 0.0,
        "issues": [],
        "retry_count": 0,
        "status": "init",
        "history": []
    })

    print("\n最终状态:", result3["status"])
    print("发现问题:", result3["issues"])
    print("审核历史:")
    for h in result3["history"]:
        print(f"  - {h}")

    print("\n" + "=" * 60)
    print("测试完成！")
    print("=" * 60)


if __name__ == "__main__":
    test_content_review()
```

## 核心技术点解析

### 1. add_conditional_edges 使用

**基础语法**：
```python
builder.add_conditional_edges(
    source_node,      # 源节点
    routing_function  # 路由函数
)
```

**来源**：Context7 官方文档

**关键点**：
- 路由函数接受 `state` 作为参数
- 返回目标节点名称（字符串）或 `END`
- 使用 `Literal` 类型提示所有可能的路由目标

### 2. 路由函数设计模式

**模式 1：二分支路由**
```python
def route_binary(state: State) -> Literal["path_a", "path_b"]:
    if condition:
        return "path_a"
    else:
        return "path_b"
```

**模式 2：多分支路由**
```python
def route_multi(state: State) -> Literal["path_a", "path_b", "path_c", END]:
    if condition_1:
        return "path_a"
    elif condition_2:
        return "path_b"
    elif condition_3:
        return "path_c"
    else:
        return END
```

**模式 3：循环路由**
```python
def route_loop(state: State) -> Literal["retry_node", "exit_node"]:
    if state["retry_count"] < max_retries:
        return "retry_node"  # 循环回去
    else:
        return "exit_node"   # 退出循环
```

**来源**：Twitter 最佳实践 + 源码分析

### 3. 循环执行设计

**关键要素**：
1. **计数器**：在 State 中维护 `retry_count`
2. **终止条件**：设置 `max_retries` 避免无限循环
3. **循环边**：`add_edge("fix_node", "check_node")` 形成循环
4. **退出路由**：条件路由判断何时退出循环

**来源**：Context7 官方示例 + Twitter 最佳实践

### 4. 多分支流程设计

**工作流模式**：
```
START → initial_check
         ↓
    [条件路由 1]
    ↙         ↘
safety_check  reject → END
    ↓
[条件路由 2]
↙         ↘
quality   human_review → END
    ↓
[条件路由 3]
↙         ↘
approve   auto_fix
  ↓          ↓
 END    [循环回 safety_check]
```

**来源**：实际业务场景抽象

## 实战经验总结

### 最佳实践

1. **明确路由逻辑**
   - 使用 `Literal` 类型提示所有可能的路由目标
   - 路由函数逻辑清晰，避免复杂嵌套
   - 每个分支都有明确的业务含义

2. **循环控制**
   - 必须设置最大重试次数
   - 在 State 中维护循环计数器
   - 提供明确的退出条件

3. **状态管理**
   - 使用 `Annotated[list, operator.add]` 累积历史记录
   - 状态字段命名清晰（如 `status`, `retry_count`）
   - 每个节点只更新必要的状态字段

4. **调试友好**
   - 添加 `print` 输出关键节点执行信息
   - 记录完整的审核历史
   - 状态字段包含足够的调试信息

**来源**：Twitter 最佳实践 + Reddit 社区经验

### 常见陷阱

1. **无限循环**
   - ❌ 错误：没有设置最大重试次数
   - ✅ 正确：设置 `max_retries` 并在路由函数中检查

2. **路由函数返回类型错误**
   - ❌ 错误：返回不在 `Literal` 中的节点名称
   - ✅ 正确：确保所有返回值都在类型提示中

3. **状态更新冲突**
   - ❌ 错误：多个节点同时修改同一状态字段
   - ✅ 正确：明确每个节点的状态更新职责

**来源**：Reddit 社区常见问题
