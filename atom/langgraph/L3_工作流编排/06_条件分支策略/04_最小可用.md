# 条件分支策略 - 最小可用

> 掌握以下 20% 的核心知识，就能解决 80% 的条件分支场景

掌握以下内容，就能在 LangGraph 中自如地使用条件分支：

---

## 4.1 基础条件边（最常用 - 必须掌握）

条件边是一切条件分支的基础。一句话：**路由函数接收状态，返回字符串，决定下一步去哪。**

### 核心 API

```python
graph.add_conditional_edges(
    source,        # 从哪个节点出发
    path,          # 路由函数：接收 state，返回目标
    path_map=None, # 可选：返回值 → 节点名映射
)
```

### 完整可运行示例

```python
"""基础条件边：根据意图路由到不同处理节点"""
from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    query: str
    intent: str
    result: str


def classify(state: State) -> dict:
    """意图分类（简化版）"""
    query = state["query"]
    if "天气" in query:
        return {"intent": "weather"}
    if "新闻" in query:
        return {"intent": "news"}
    return {"intent": "chat"}


# ✅ 路由函数：返回字符串，决定下一步
def route(state: State) -> Literal["weather", "news", "chat"]:
    return state["intent"]


def weather_node(state: State) -> dict:
    return {"result": f"天气查询: {state['query']}"}


def news_node(state: State) -> dict:
    return {"result": f"新闻搜索: {state['query']}"}


def chat_node(state: State) -> dict:
    return {"result": f"闲聊回复: {state['query']}"}


builder = StateGraph(State)
builder.add_node("classify", classify)
builder.add_node("weather", weather_node)
builder.add_node("news", news_node)
builder.add_node("chat", chat_node)

builder.add_edge(START, "classify")

# 条件边：根据意图路由
builder.add_conditional_edges("classify", route, {
    "weather": "weather",
    "news": "news",
    "chat": "chat",
})

builder.add_edge("weather", END)
builder.add_edge("news", END)
builder.add_edge("chat", END)

graph = builder.compile()

# 测试
r = graph.invoke({"query": "今天天气怎么样", "intent": "", "result": ""})
print(r["result"])  # 天气查询: 今天天气怎么样
```

### 要点速记

- 路由函数返回值必须和 path_map 的 key 对应（或直接返回节点名）
- 返回 `"__end__"` 或 `END` 可以直接结束图执行
- Literal 类型注解帮助图可视化，不影响运行逻辑

## 4.2 path_map 映射（解耦路由逻辑与节点名）

path_map 让路由函数的返回值和实际节点名解耦。好处：**路由函数返回语义化的字符串，path_map 负责映射到真实节点。**

### 为什么需要 path_map？

```python
# 不用 path_map：路由函数返回值 = 节点名（耦合）
def route(state) -> Literal["process_node_v2", "error_handler_node"]:
    if state["ok"]:
        return "process_node_v2"    # 节点改名了，这里也要改
    return "error_handler_node"

# ✅ 用 path_map：路由函数返回语义值（解耦）
def route(state) -> Literal["success", "error"]:
    if state["ok"]:
        return "success"            # 语义清晰，不关心节点叫什么
    return "error"

graph.add_conditional_edges("check", route, {
    "success": "process_node_v2",   # 映射关系在这里维护
    "error": "error_handler_node",
})
```

### 三种 path_map 形式

```python
# 形式1：字典映射（最常用）
graph.add_conditional_edges("node", route, {
    "option_a": "node_a",
    "option_b": "node_b",
})

# 形式2：不传 path_map（返回值直接当节点名）
graph.add_conditional_edges("node", route)
# 此时 route 必须返回真实的节点名

# 形式3：列表形式（仅声明可能的目标节点）
graph.add_conditional_edges("node", route, ["node_a", "node_b"])
# 等价于 {"node_a": "node_a", "node_b": "node_b"}
```

## 4.3 循环守卫（防止无限循环 - 必须掌握）

只要图中有循环（条件边指回之前的节点），就必须加循环守卫。

### 最简循环守卫

```python
from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    value: str
    steps: int
    max_steps: int


def process(state: State) -> dict:
    """处理节点"""
    return {
        "value": state["value"] + ".",
        "steps": state.get("steps", 0) + 1,
    }


# ✅ 循环守卫：检查步数
def should_continue(state: State) -> Literal["continue", "stop"]:
    if state.get("steps", 0) >= state.get("max_steps", 10):
        return "stop"       # 到达上限，强制停止
    if len(state["value"]) > 20:
        return "stop"       # 业务条件满足，正常停止
    return "continue"


builder = StateGraph(State)
builder.add_node("process", process)

builder.add_edge(START, "process")
builder.add_conditional_edges("process", should_continue, {
    "continue": "process",  # 循环回自身
    "stop": END,
})

graph = builder.compile()

result = graph.invoke({"value": "start", "steps": 0, "max_steps": 5})
print(f"结果: {result['value']}")   # start.....
print(f"步数: {result['steps']}")   # 5
```

### 要点速记

- 循环 = 条件边指向之前的节点
- 守卫 = 在条件边中检查计数器
- **永远不要写没有退出条件的循环**
- 两个退出条件：业务条件满足 OR 步数到达上限

## 4.4 Send 并行（Map-Reduce 模式）

Send 解决的问题：**运行时才知道要并行处理多少个任务。**

### 最简 Send 示例

```python
"""Send 并行：动态创建 N 个并行任务"""
import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Send


class MainState(TypedDict):
    items: list[str]
    results: Annotated[list[str], operator.add]


class WorkerState(TypedDict):
    item: str


def worker(state: WorkerState) -> dict:
    """处理单个任务"""
    return {"results": [f"处理完成: {state['item']}"]}


# ✅ 路由函数返回 Send 列表
def fan_out(state: MainState) -> list[Send]:
    return [Send("worker", {"item": i}) for i in state["items"]]


builder = StateGraph(MainState)
builder.add_node("worker", worker)

builder.add_conditional_edges(START, fan_out)
builder.add_edge("worker", END)

graph = builder.compile()

result = graph.invoke({"items": ["文档A", "文档B", "文档C"], "results": []})
print(result["results"])
# ['处理完成: 文档A', '处理完成: 文档B', '处理完成: 文档C']
```

### 要点速记

- `Send(node_name, state_dict)` — 目标节点 + 传递的状态
- 路由函数返回 `list[Send]`，每个 Send 创建一个并行分支
- 并行结果用 `Annotated[list, operator.add]` 合并
- Send 通过 `add_conditional_edges` 触发，不是 `add_edge`

## 4.5 Command 动态路由（高级 - 按需掌握）

Command 让节点同时完成两件事：**更新状态 + 决定下一步去哪。**

### 最简 Command 示例

```python
"""Command 动态路由：节点内部控制流程"""
from typing import Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Command


class State(TypedDict):
    score: int
    result: str


# ✅ 节点返回 Command，同时更新状态和路由
def evaluate(state: State) -> Command[Literal["approve", "reject"]]:
    if state["score"] >= 60:
        return Command(
            update={"result": "通过"},
            goto="approve",
        )
    return Command(
        update={"result": "不通过"},
        goto="reject",
    )


def approve(state: State) -> dict:
    print(f"审批通过: {state['result']}")
    return {}


def reject(state: State) -> dict:
    print(f"审批拒绝: {state['result']}")
    return {}


builder = StateGraph(State)
builder.add_node("evaluate", evaluate)
builder.add_node("approve", approve)
builder.add_node("reject", reject)

builder.add_edge(START, "evaluate")
# Command 自带路由，不需要 add_conditional_edges
builder.add_edge("approve", END)
builder.add_edge("reject", END)

graph = builder.compile()

graph.invoke({"score": 85, "result": ""})  # 审批通过: 通过
graph.invoke({"score": 40, "result": ""})  # 审批拒绝: 不通过
```

### Command vs 条件边：怎么选？

| 场景 | 推荐 |
|------|------|
| 简单 if/else 分支 | 条件边（更直观） |
| 路由和业务逻辑紧耦合 | Command（更内聚） |
| 需要同时更新状态和路由 | Command |
| 子图到父图通信 | Command（唯一选择） |

## 总结：最小可用知识清单

```
✅ 条件边：add_conditional_edges(source, route_fn, path_map)
✅ path_map：解耦路由返回值和节点名
✅ 循环守卫：状态计数器 + 条件边检查上限
✅ Send：Send(node, state) 实现动态并行
✅ Command：Command(update=..., goto=...) 节点内路由
```
