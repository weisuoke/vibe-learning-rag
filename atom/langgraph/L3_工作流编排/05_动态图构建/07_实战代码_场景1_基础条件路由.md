# 实战场景1：基础条件路由 - 多路分支与回退

> 构建一个文档处理工作流：根据文件类型自动分流到不同处理节点，未知类型走回退逻辑，最终汇总所有处理结果。

---

## 场景描述

想象你在构建一个 RAG 系统的文档预处理管道。用户上传的文档格式五花八门——PDF、Word、图片、甚至一些奇怪的格式。你需要：

1. **分类节点**：根据文件扩展名判断文档类型
2. **专用处理节点**：PDF、Word、Image 各有专门的处理逻辑
3. **回退节点**：未知格式走兜底处理
4. **汇总节点**：收集所有处理结果，输出最终报告

### 双重类比

**前端类比：** 就像一个文件上传组件，根据 MIME type 分发到不同的预览器——PDF 用 PDF.js，图片用 `<img>`，Word 用在线预览，其他格式显示"不支持预览"。

**日常生活类比：** 就像医院的分诊台——来了一个病人，护士根据症状分到内科、外科、眼科，搞不清楚的送到全科。最后所有科室的诊断结果汇总到病历里。

### 工作流图示

```
                    ┌──→ [PDF处理]    ──┐
                    │                    │
[START] → [分类] ──┼──→ [Word处理]   ──┼──→ [汇总] → [END]
                    │                    │
                    ├──→ [Image处理]  ──┤
                    │                    │
                    └──→ [回退处理]   ──┘
```

## 完整可运行代码

```python
"""
实战场景1：基础条件路由 - 文档处理工作流
演示：add_conditional_edges + path_map 实现多路分支与回退

运行要求：pip install langgraph
无需外部 API，完全本地运行
"""
from typing import TypedDict, Annotated
import operator
from langgraph.graph import StateGraph, START, END


# ===== 1. 定义状态 =====
class DocState(TypedDict):
    """文档处理工作流的状态

    Attributes:
        filename: 输入的文件名（如 "report.pdf"）
        doc_type: 分类结果（pdf/word/image/unknown）
        content: 模拟的文档内容
        process_log: 处理日志（使用 operator.add 作为 reducer，支持多节点追加）
        result: 最终处理结果
    """
    filename: str
    doc_type: str
    content: str
    process_log: Annotated[list[str], operator.add]
    result: str


# ===== 2. 分类节点 =====
def classify_document(state: DocState) -> dict:
    """根据文件扩展名判断文档类型

    这是工作流的入口节点，负责解析文件名并确定类型。
    在真实 RAG 系统中，这里可能会用 magic bytes 或 MIME type 检测。
    """
    filename = state["filename"]
    extension = filename.rsplit(".", 1)[-1].lower() if "." in filename else ""

    # 扩展名 → 类型映射
    type_map = {
        "pdf": "pdf",
        "doc": "word",
        "docx": "word",
        "png": "image",
        "jpg": "image",
        "jpeg": "image",
        "gif": "image",
    }

    doc_type = type_map.get(extension, "unknown")
    print(f"[分类] 文件: {filename} → 类型: {doc_type}")

    return {
        "doc_type": doc_type,
        "process_log": [f"分类完成: {filename} → {doc_type}"],
    }


# ===== 3. 路由函数 =====
def route_by_doc_type(state: DocState) -> str:
    """根据文档类型决定下一个处理节点

    这是条件边的核心——路由函数。
    它接收当前状态，返回一个字符串标识（不是节点名！）。
    这个标识会通过 path_map 映射到实际的节点名。

    返回值说明：
    - "pdf"     → 通过 path_map 映射到 "process_pdf" 节点
    - "word"    → 通过 path_map 映射到 "process_word" 节点
    - "image"   → 通过 path_map 映射到 "process_image" 节点
    - "unknown" → 通过 path_map 映射到 "fallback" 节点
    """
    doc_type = state["doc_type"]
    print(f"[路由] 类型 '{doc_type}' → 分发到对应处理节点")
    return doc_type


# ===== 4. 四个处理节点 =====
def process_pdf(state: DocState) -> dict:
    """PDF 处理节点

    在真实 RAG 系统中，这里会用 pypdf/pdfplumber 提取文本，
    处理表格、图片、页眉页脚等。
    """
    filename = state["filename"]
    print(f"[PDF处理] 正在处理 {filename}")
    print(f"  - 提取文本内容")
    print(f"  - 解析表格结构")
    print(f"  - 提取元数据（作者、页数等）")

    content = f"[PDF内容] 从 {filename} 提取了 42 页文本，包含 3 个表格"
    return {
        "content": content,
        "process_log": [f"PDF处理完成: {filename}"],
    }


def process_word(state: DocState) -> dict:
    """Word 处理节点

    在真实 RAG 系统中，这里会用 python-docx 提取文本，
    保留标题层级、列表结构等。
    """
    filename = state["filename"]
    print(f"[Word处理] 正在处理 {filename}")
    print(f"  - 提取段落文本")
    print(f"  - 保留标题层级")
    print(f"  - 提取嵌入图片描述")

    content = f"[Word内容] 从 {filename} 提取了 15 个段落，3 级标题结构"
    return {
        "content": content,
        "process_log": [f"Word处理完成: {filename}"],
    }


def process_image(state: DocState) -> dict:
    """图片处理节点

    在真实 RAG 系统中，这里会用 OCR（如 Tesseract）提取文字，
    或用多模态模型（如 GPT-4V）生成图片描述。
    """
    filename = state["filename"]
    print(f"[Image处理] 正在处理 {filename}")
    print(f"  - OCR 文字识别")
    print(f"  - 图片描述生成")
    print(f"  - 提取 EXIF 元数据")

    content = f"[Image内容] 从 {filename} 识别了 128 个字符，生成了图片描述"
    return {
        "content": content,
        "process_log": [f"Image处理完成: {filename}"],
    }


# ===== 5. 回退节点 =====
def fallback_handler(state: DocState) -> dict:
    """回退处理节点 - 处理未知格式

    当文件类型无法识别时，走这个兜底逻辑。
    可以尝试通用文本提取，或标记为需要人工处理。
    """
    filename = state["filename"]
    print(f"[回退处理] 未知格式: {filename}")
    print(f"  - 尝试纯文本读取")
    print(f"  - 标记为需要人工审核")

    content = f"[回退] {filename} 格式未知，已尝试纯文本提取，建议人工审核"
    return {
        "content": content,
        "process_log": [f"回退处理: {filename}（未知格式）"],
    }


# ===== 6. 汇总节点 =====
def summarize_results(state: DocState) -> dict:
    """汇总节点 - 收集处理结果，生成最终报告"""
    filename = state["filename"]
    doc_type = state["doc_type"]
    content = state["content"]
    log = state["process_log"]

    print(f"\n[汇总] 生成处理报告")
    print(f"  文件: {filename}")
    print(f"  类型: {doc_type}")
    print(f"  处理步骤: {len(log)} 步")
    print(f"  内容预览: {content[:60]}...")

    result = (
        f"处理报告 | 文件: {filename} | 类型: {doc_type} | "
        f"步骤数: {len(log)} | 状态: 成功"
    )
    return {
        "result": result,
        "process_log": [f"汇总完成，生成最终报告"],
    }


# ===== 7. 构建工作流图 =====
def build_document_workflow():
    """构建文档处理工作流

    关键步骤：
    1. 创建 StateGraph
    2. 添加所有节点
    3. 用 add_conditional_edges 连接分类节点到处理节点
    4. 用 add_edge 连接处理节点到汇总节点
    5. 编译
    """
    graph = StateGraph(DocState)

    # --- 添加节点 ---
    graph.add_node("classify", classify_document)
    graph.add_node("process_pdf", process_pdf)
    graph.add_node("process_word", process_word)
    graph.add_node("process_image", process_image)
    graph.add_node("fallback", fallback_handler)
    graph.add_node("summarize", summarize_results)

    # --- 入口边 ---
    graph.add_edge(START, "classify")

    # --- 条件边：分类 → 处理（核心！）---
    graph.add_conditional_edges(
        source="classify",          # 从分类节点出发
        path=route_by_doc_type,     # 路由函数决定去哪
        path_map={                  # 路由函数返回值 → 实际节点名
            "pdf": "process_pdf",       # 返回 "pdf"   → 去 process_pdf
            "word": "process_word",     # 返回 "word"  → 去 process_word
            "image": "process_image",   # 返回 "image" → 去 process_image
            "unknown": "fallback",      # 返回 "unknown" → 去 fallback
        },
    )

    # --- 所有处理节点 → 汇总 ---
    graph.add_edge("process_pdf", "summarize")
    graph.add_edge("process_word", "summarize")
    graph.add_edge("process_image", "summarize")
    graph.add_edge("fallback", "summarize")

    # --- 汇总 → 结束 ---
    graph.add_edge("summarize", END)

    return graph.compile()


# ===== 8. 运行测试 =====
if __name__ == "__main__":
    app = build_document_workflow()

    # 测试用例：覆盖所有分支
    test_files = [
        "quarterly_report.pdf",     # PDF 分支
        "meeting_notes.docx",       # Word 分支
        "architecture_diagram.png", # Image 分支
        "data_dump.xyz",            # Unknown → 回退分支
    ]

    for filename in test_files:
        print("\n" + "=" * 60)
        print(f"测试文件: {filename}")
        print("=" * 60)

        result = app.invoke({
            "filename": filename,
            "doc_type": "",
            "content": "",
            "process_log": [],
            "result": "",
        })

        print(f"\n最终结果: {result['result']}")
        print(f"完整日志: {result['process_log']}")
```

## 代码逐行解析

### 1. 状态定义（State）

```python
class DocState(TypedDict):
    filename: str                                        # 输入文件名
    doc_type: str                                        # 分类结果
    content: str                                         # 处理后的内容
    process_log: Annotated[list[str], operator.add]      # 处理日志（关键！）
    result: str                                          # 最终结果
```

`process_log` 使用了 `Annotated[list[str], operator.add]` 作为 reducer。这意味着每个节点返回的 `process_log` 列表会被**追加**到现有列表中，而不是覆盖。这是 LangGraph 处理多节点写入同一字段的标准模式。

**前端类比：** 就像 Redux 中的 reducer——`operator.add` 相当于 `(state, action) => [...state, ...action.payload]`。

### 2. 路由函数

```python
def route_by_doc_type(state: DocState) -> str:
    return state["doc_type"]  # 返回 "pdf" / "word" / "image" / "unknown"
```

路由函数非常简单——它只需要读取状态中的 `doc_type` 字段，返回一个字符串。这个字符串**不是节点名**，而是一个标识符，会通过 `path_map` 映射到实际节点名。

为什么要这样设计？因为路由函数的返回值可能是业务语义（如 "pdf"），而节点名可能是技术命名（如 "process_pdf"）。`path_map` 解耦了这两者。

### 3. path_map 映射

```python
graph.add_conditional_edges(
    source="classify",
    path=route_by_doc_type,
    path_map={
        "pdf": "process_pdf",
        "word": "process_word",
        "image": "process_image",
        "unknown": "fallback",
    },
)
```

`path_map` 是一个字典，将路由函数的返回值映射到实际的节点名。

**如果不用 path_map 呢？** 路由函数的返回值必须直接是节点名：

```python
# 不用 path_map 的写法（路由函数直接返回节点名）
def route_by_doc_type(state: DocState) -> str:
    if state["doc_type"] == "pdf":
        return "process_pdf"      # 必须是节点名
    elif state["doc_type"] == "word":
        return "process_word"
    elif state["doc_type"] == "image":
        return "process_image"
    return "fallback"

graph.add_conditional_edges("classify", route_by_doc_type)  # 不需要 path_map
```

两种写法效果相同，但 `path_map` 版本更清晰——路由逻辑和节点映射分离，便于维护。

> [来源: LangGraph 源码 graph/state.py - add_conditional_edges 方法签名]

### 4. Reducer 的作用

```python
process_log: Annotated[list[str], operator.add]
```

如果没有 `operator.add` reducer，后面的节点写入 `process_log` 会覆盖前面的值。有了 reducer，每个节点的日志会被追加：

```
分类节点写入: ["分类完成: report.pdf → pdf"]
PDF节点写入:  ["PDF处理完成: report.pdf"]
汇总节点写入: ["汇总完成，生成最终报告"]

最终 process_log = ["分类完成...", "PDF处理完成...", "汇总完成..."]  # 三者合并
```

### 5. 边的连接模式

```python
# 入口 → 分类（静态边，固定路径）
graph.add_edge(START, "classify")

# 分类 → 处理（条件边，动态路径）
graph.add_conditional_edges("classify", route_by_doc_type, path_map={...})

# 处理 → 汇总（静态边，固定路径）
graph.add_edge("process_pdf", "summarize")
graph.add_edge("process_word", "summarize")
graph.add_edge("process_image", "summarize")
graph.add_edge("fallback", "summarize")

# 汇总 → 结束（静态边，固定路径）
graph.add_edge("summarize", END)
```

注意这个模式：**条件边只在需要动态决策的地方使用**（分类 → 处理），其他地方用静态边就够了。不要过度使用条件边——简单的固定路径用 `add_edge` 更清晰。

## 扩展思考

### 1. 如何添加新的文档类型？

比如要支持 Excel 文件（.xlsx），只需三步：

```python
# 步骤1：在分类节点的映射中添加新类型
type_map = {
    "pdf": "pdf",
    "doc": "word",
    "docx": "word",
    "xlsx": "excel",    # 新增
    "xls": "excel",     # 新增
    # ...
}

# 步骤2：添加处理节点
def process_excel(state: DocState) -> dict:
    filename = state["filename"]
    print(f"[Excel处理] 正在处理 {filename}")
    content = f"[Excel内容] 从 {filename} 提取了 5 个工作表"
    return {
        "content": content,
        "process_log": [f"Excel处理完成: {filename}"],
    }

graph.add_node("process_excel", process_excel)

# 步骤3：在 path_map 中添加映射
path_map={
    "pdf": "process_pdf",
    "word": "process_word",
    "image": "process_image",
    "excel": "process_excel",   # 新增
    "unknown": "fallback",
}

# 步骤4：添加到汇总的边
graph.add_edge("process_excel", "summarize")
```

这就是条件边 + path_map 的优势——**添加新分支不需要修改路由函数**，只需要扩展映射表。

### 2. 如何添加错误处理？

在真实系统中，处理节点可能会失败。可以用 try-except + 状态字段实现：

```python
class DocState(TypedDict):
    filename: str
    doc_type: str
    content: str
    process_log: Annotated[list[str], operator.add]
    result: str
    error: str  # 新增：错误信息


def process_pdf(state: DocState) -> dict:
    try:
        filename = state["filename"]
        # ... 实际处理逻辑 ...
        return {
            "content": f"处理成功: {filename}",
            "process_log": [f"PDF处理完成: {filename}"],
            "error": "",
        }
    except Exception as e:
        return {
            "content": "",
            "process_log": [f"PDF处理失败: {e}"],
            "error": str(e),
        }


# 在汇总节点中检查错误
def summarize_results(state: DocState) -> dict:
    if state.get("error"):
        result = f"处理失败: {state['error']}"
    else:
        result = f"处理成功: {state['filename']}"
    return {"result": result, "process_log": ["汇总完成"]}
```

### 3. 如何支持批量文件处理？

如果要同时处理多个文件，可以结合 Send API（参见 `03_核心概念_2_Send_API与Map_Reduce.md`）：

```python
from langgraph.types import Send

def dispatch_files(state: BatchState):
    """将多个文件分发到独立的处理流程"""
    return [
        Send("classify", {"filename": f, "process_log": []})
        for f in state["filenames"]
    ]
```

这样每个文件会独立走完整个 分类 → 处理 → 汇总 流程，最后结果通过 reducer 合并。

## 关键收获

| 概念 | 本场景中的体现 |
|------|---------------|
| `add_conditional_edges` | 分类节点根据文档类型动态路由 |
| `path` 路由函数 | `route_by_doc_type` 读取状态返回类型标识 |
| `path_map` 映射 | 将 "pdf"/"word" 等标识映射到实际节点名 |
| 回退分支 | "unknown" 映射到 fallback 节点 |
| `Annotated` reducer | `process_log` 使用 `operator.add` 追加日志 |
| Fan-in 汇聚 | 4 个处理节点都通过静态边汇聚到 summarize |
