# 实战代码 - 场景1：基础状态转换函数

## 场景说明

本场景演示 LangGraph 节点函数的标准写法，涵盖状态转换函数的五种核心模式：

1. 基本节点函数签名——接收完整 State，返回部分更新 dict
2. 有/无 Reducer 字段的不同返回策略
3. 异步节点函数的写法
4. 带 `config` 参数的节点函数
5. 通过 `stream` 逐步观察每个节点的状态变化

这是状态转换函数的最基础用法——理解"读取 → 处理 → 返回部分更新"的三步法，是编写所有 LangGraph 节点的前提。

## 完整代码

```python
"""
LangGraph 状态转换函数 - 基础示例
演示：节点函数的标准写法与返回模式

核心知识点：
- 基本节点函数签名 (State) -> dict
- 无 Reducer 字段：返回值直接覆盖
- 有 Reducer 字段：返回值增量追加
- 部分状态返回：只返回变化的字段
- 异步节点函数
- 带 config 的节点函数
- stream 逐步观察状态变化

运行环境：Python 3.13+, langgraph
安装依赖：uv add langgraph
"""

import asyncio
from typing import Annotated
from typing_extensions import TypedDict
from operator import add
from langgraph.graph import StateGraph, START, END
from langchain_core.runnables import RunnableConfig


# ============================================================
# 1. 定义状态 Schema —— 有/无 Reducer 的混合状态
# ============================================================

class WorkflowState(TypedDict):
    """工作流状态定义

    无 Reducer 字段（LastValue Channel）：
    - query, result, status → 每次返回直接覆盖旧值

    有 Reducer 字段（BinaryOperatorAggregate Channel）：
    - logs → 使用 operator.add，每次返回的列表会追加到已有列表
    - step_count → 使用 operator.add，每次返回的数值会累加
    """
    # 无 Reducer：直接覆盖
    query: str
    result: str
    status: str

    # 有 Reducer：增量追加
    logs: Annotated[list[str], add]
    step_count: Annotated[int, add]


# ============================================================
# 2. 基本节点函数 —— 三步法：读取 → 处理 → 返回部分更新
# ============================================================

def preprocess_node(state: WorkflowState) -> dict:
    """预处理节点：清洗输入查询

    三步法演示：
    1. 读取：从 state 中取出 query
    2. 处理：去除空格、转小写
    3. 返回：只返回变化的字段（query, status, logs, step_count）
       - result 没变，所以不返回 → 保持原值
    """
    # 第一步：读取需要的字段
    query = state["query"]

    # 第二步：执行转换逻辑
    cleaned = query.strip().lower()

    # 第三步：只返回变化的字段
    return {
        "query": cleaned,              # 无 Reducer → 覆盖为清洗后的值
        "status": "preprocessing",     # 无 Reducer → 覆盖
        "logs": [f"预处理: '{cleaned}'"],  # 有 Reducer → 追加到列表
        "step_count": 1,               # 有 Reducer → 累加 1
        # 注意：result 没有出现 → 保持原值不变
    }


def process_node(state: WorkflowState) -> dict:
    """处理节点：执行核心业务逻辑

    演示：读取上一步清洗后的 query，生成处理结果
    """
    query = state["query"]
    # 模拟核心处理逻辑（实际场景中可能是 LLM 调用、向量检索等）
    result = f"处理结果: {query.upper()}"
    return {
        "result": result,
        "status": "processing",
        "logs": [f"处理: {result}"],
        "step_count": 1,
    }


def postprocess_node(state: WorkflowState) -> dict:
    """后处理节点：格式化输出

    演示：读取 result 字段，做最终格式化
    """
    result = state["result"]
    formatted = f"【最终结果】{result}"
    return {
        "result": formatted,
        "status": "completed",
        "logs": ["后处理完成"],
        "step_count": 1,
    }


# ============================================================
# 3. 构建图 —— 线性三步管道
# ============================================================

builder = StateGraph(WorkflowState)
builder.add_node("preprocess", preprocess_node)
builder.add_node("process", process_node)
builder.add_node("postprocess", postprocess_node)

builder.add_edge(START, "preprocess")
builder.add_edge("preprocess", "process")
builder.add_edge("process", "postprocess")
builder.add_edge("postprocess", END)

app = builder.compile()


# ============================================================
# 4. 执行并观察最终结果
# ============================================================

print("=" * 60)
print("=== 示例1：基础状态转换函数 ===")
print("=" * 60)

initial_state = {
    "query": "  Hello LangGraph  ",
    "result": "",
    "status": "init",
    "logs": ["开始执行"],
    "step_count": 0,
}

result = app.invoke(initial_state)

print(f"\n查询: {result['query']}")
print(f"结果: {result['result']}")
print(f"状态: {result['status']}")
print(f"总步数: {result['step_count']}")
print(f"\n执行日志:")
for log in result["logs"]:
    print(f"  - {log}")


# ============================================================
# 5. 使用 stream 逐步观察每个节点的状态变化
# ============================================================

print()
print("=" * 60)
print("=== 示例2：stream 逐步观察状态变化 ===")
print("=" * 60)

initial_state_2 = {
    "query": "  RAG Pipeline  ",
    "result": "",
    "status": "init",
    "logs": ["流式执行开始"],
    "step_count": 0,
}

for step in app.stream(initial_state_2):
    # step 是一个 dict，key 是节点名，value 是该节点返回的部分更新
    for node_name, node_output in step.items():
        print(f"\n--- 节点: {node_name} ---")
        print(f"  返回内容: {node_output}")
        # 注意：stream 返回的是节点的原始返回值（部分更新）
        # 不是合并后的完整状态


# ============================================================
# 6. 异步节点函数
# ============================================================

class AsyncState(TypedDict):
    """异步示例的状态"""
    input_text: str
    processed: str
    logs: Annotated[list[str], add]


async def async_clean_node(state: AsyncState) -> dict:
    """异步清洗节点

    异步节点的签名和同步完全一样，只是加了 async/await。
    LangGraph 会自动识别并在异步事件循环中执行。
    适用场景：网络请求、数据库查询、文件 I/O 等。
    """
    # 模拟异步 I/O 操作（实际场景中可能是 await httpx.get(...)）
    await asyncio.sleep(0.01)
    text = state["input_text"].strip()
    return {
        "processed": text.lower(),
        "logs": [f"异步清洗: '{text}' → '{text.lower()}'"],
    }


async def async_enrich_node(state: AsyncState) -> dict:
    """异步增强节点"""
    await asyncio.sleep(0.01)
    enriched = f"[enriched] {state['processed']}"
    return {
        "processed": enriched,
        "logs": [f"异步增强: '{enriched}'"],
    }


async def run_async_example():
    """运行异步图示例"""
    async_builder = StateGraph(AsyncState)
    async_builder.add_node("clean", async_clean_node)
    async_builder.add_node("enrich", async_enrich_node)
    async_builder.add_edge(START, "clean")
    async_builder.add_edge("clean", "enrich")
    async_builder.add_edge("enrich", END)

    async_app = async_builder.compile()

    result = await async_app.ainvoke({
        "input_text": "  Async LangGraph  ",
        "processed": "",
        "logs": [],
    })

    print(f"\n异步结果: {result['processed']}")
    print(f"异步日志: {result['logs']}")


print()
print("=" * 60)
print("=== 示例3：异步节点函数 ===")
print("=" * 60)

asyncio.run(run_async_example())


# ============================================================
# 7. 带 config 的节点函数
# ============================================================

class ConfigState(TypedDict):
    """带配置的状态"""
    query: str
    result: str
    model_used: str
    logs: Annotated[list[str], add]


def config_node(state: ConfigState, config: RunnableConfig) -> dict:
    """带 config 的节点函数

    config 是可选的第二参数，框架通过检查函数签名自动注入。
    常用于传递运行时配置：API key、模型名称、用户 ID 等。

    关键规则：
    - config 必须是第二个参数（state 在前，config 在后）
    - 通过 config.get("configurable", {}) 获取自定义配置
    - 在 graph.invoke(input, config={...}) 时传入
    """
    # 从 config 中获取运行时参数
    configurable = config.get("configurable", {})
    model_name = configurable.get("model", "default-model")
    temperature = configurable.get("temperature", 0.7)

    query = state["query"]
    # 模拟使用配置参数处理
    result = f"用 {model_name}(temp={temperature}) 处理: {query}"

    return {
        "result": result,
        "model_used": model_name,
        "logs": [f"使用模型: {model_name}, 温度: {temperature}"],
    }


print()
print("=" * 60)
print("=== 示例4：带 config 的节点函数 ===")
print("=" * 60)

config_builder = StateGraph(ConfigState)
config_builder.add_node("process", config_node)
config_builder.add_edge(START, "process")
config_builder.add_edge("process", END)

config_app = config_builder.compile()

# 通过 config 传递运行时参数
result = config_app.invoke(
    {"query": "什么是 RAG？", "result": "", "model_used": "", "logs": []},
    config={"configurable": {"model": "gpt-4o", "temperature": 0.3}},
)

print(f"\n结果: {result['result']}")
print(f"使用模型: {result['model_used']}")
print(f"日志: {result['logs']}")

# 不传 config → 使用默认值
result_default = config_app.invoke(
    {"query": "默认配置测试", "result": "", "model_used": "", "logs": []},
)

print(f"\n默认结果: {result_default['result']}")
print(f"默认模型: {result_default['model_used']}")


# ============================================================
# 8. 空返回与条件跳过
# ============================================================

class GuardState(TypedDict):
    """守卫节点示例的状态"""
    input_text: str
    result: str
    skipped: bool


def guard_node(state: GuardState) -> dict:
    """守卫节点：条件跳过

    返回空字典 {} 表示不更新任何字段。
    这在幂等检查、条件跳过等场景中非常有用。
    """
    text = state.get("input_text") or ""
    if not text.strip():
        # 输入为空，跳过处理，不更新任何字段
        return {}
    return {
        "result": f"已处理: {text.strip()}",
        "skipped": False,
    }


print()
print("=" * 60)
print("=== 示例5：空返回与条件跳过 ===")
print("=" * 60)

guard_builder = StateGraph(GuardState)
guard_builder.add_node("guard", guard_node)
guard_builder.add_edge(START, "guard")
guard_builder.add_edge("guard", END)

guard_app = guard_builder.compile()

# 正常输入
r1 = guard_app.invoke({"input_text": "hello", "result": "旧结果", "skipped": True})
print(f"\n正常输入 → result: {r1['result']}, skipped: {r1['skipped']}")

# 空输入 → 返回 {}，状态不变
r2 = guard_app.invoke({"input_text": "", "result": "旧结果", "skipped": True})
print(f"空输入   → result: {r2['result']}, skipped: {r2['skipped']}")
```

## 关键知识点解析

### 1. 节点函数签名速查表

| 签名 | 适用场景 | 示例 |
|------|----------|------|
| `def f(state) -> dict` | 简单数据处理、转换 | 文本清洗、格式化 |
| `def f(state, config) -> dict` | 需要运行时配置 | LLM 调用、API 请求 |
| `async def f(state) -> dict` | I/O 密集型操作 | 网络请求、数据库查询 |
| `async def f(state, config) -> dict` | 异步 + 运行时配置 | 异步 LLM 调用 |

### 2. 有/无 Reducer 字段的返回规则

| 字段类型 | 返回行为 | 示例 |
|----------|----------|------|
| 无 Reducer（`query: str`） | 返回值直接覆盖旧值 | `return {"query": "new"}` → query 变为 "new" |
| 有 Reducer（`logs: Annotated[list, add]`） | 返回值通过 Reducer 合并 | `return {"logs": ["new"]}` → logs 追加 "new" |
| 未返回的字段 | 保持原值不变 | 不返回 query → query 不变 |
| 返回空字典 `{}` | 所有字段保持原值 | 条件跳过场景 |

### 3. 三步法口诀

```
读取 → 处理 → 返回部分更新
```

每个节点函数都遵循这个模式：
1. **读取**：从 `state` 中取出需要的字段
2. **处理**：执行业务逻辑（LLM 调用、检索、计算等）
3. **返回**：只返回变化的字段，未变化的字段不要返回

### 4. stream vs invoke 的区别

```python
# invoke → 返回最终的完整状态
result = app.invoke(initial_state)
# result = {"query": "...", "result": "...", "status": "...", "logs": [...], "step_count": 3}

# stream → 逐步返回每个节点的原始返回值（部分更新）
for step in app.stream(initial_state):
    # step = {"preprocess": {"query": "...", "status": "...", ...}}
    # step = {"process": {"result": "...", "status": "...", ...}}
    # step = {"postprocess": {"result": "...", "status": "...", ...}}
    pass
```

- `invoke`：一次性执行完毕，返回合并后的完整状态
- `stream`：逐步返回每个节点的原始输出，适合调试和监控

### 5. 常见陷阱

```python
# ❌ 陷阱1：对有 Reducer 的字段手动拼接旧值
def bad_node(state: WorkflowState) -> dict:
    return {"logs": state["logs"] + ["new"]}  # 会导致重复！

# ✅ 正确：只返回增量
def good_node(state: WorkflowState) -> dict:
    return {"logs": ["new"]}  # Reducer 自动追加


# ❌ 陷阱2：返回未变化的字段
def bad_node_2(state: WorkflowState) -> dict:
    return {
        "query": state["query"],  # 多余！可能触发 LastValue 限制
        "result": "new result",
    }

# ✅ 正确：只返回变化的字段
def good_node_2(state: WorkflowState) -> dict:
    return {"result": "new result"}


# ❌ 陷阱3：config 参数位置错误
def bad_node_3(config: RunnableConfig, state: WorkflowState) -> dict: ...  # 报错！

# ✅ 正确：state 在前，config 在后
def good_node_3(state: WorkflowState, config: RunnableConfig) -> dict: ...
```

## 在 RAG 系统中的应用

基础状态转换函数在 RAG 系统中的典型用法：

- **预处理节点**：清洗用户查询（去除空格、标准化格式、提取关键词）
- **检索节点**：读取 query，调用向量数据库，返回 `{"documents": [...]}`
- **生成节点**：读取 query + documents，调用 LLM，返回 `{"answer": "..."}`
- **后处理节点**：格式化输出、添加引用来源、过滤敏感内容
- **日志追踪**：通过 `Annotated[list, add]` 的 logs 字段记录每步操作

```python
# RAG 场景的典型节点函数
class RAGState(TypedDict):
    query: str                              # 覆盖模式
    documents: list[str]                    # 覆盖模式
    answer: str                             # 覆盖模式
    logs: Annotated[list[str], add]         # 追加模式

def retrieve(state: RAGState) -> dict:
    docs = vector_search(state["query"])
    return {"documents": docs, "logs": [f"检索到 {len(docs)} 篇文档"]}

def generate(state: RAGState, config: RunnableConfig) -> dict:
    model = config.get("configurable", {}).get("model", "gpt-4o")
    answer = call_llm(state["query"], state["documents"], model=model)
    return {"answer": answer, "logs": [f"使用 {model} 生成回答"]}
```
