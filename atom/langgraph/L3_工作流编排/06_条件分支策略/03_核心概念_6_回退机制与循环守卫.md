# 核心概念6：回退机制与循环守卫

## 一句话定义

**回退机制让工作流在遇到错误时优雅降级而非崩溃，循环守卫通过计数器防止条件分支陷入无限循环——两者结合，确保 Agent 工作流在不确定环境中安全、可控地运行。**

---

## 为什么需要回退机制？

### Agent 工作流中的不确定性

传统软件的 if-else 分支是确定性的：输入确定，输出就确定。但 Agent 工作流完全不同：

```python
# 传统软件：确定性分支
if user_age >= 18:
    return "成年"    # 100% 确定的结果
else:
    return "未成年"  # 100% 确定的结果

# Agent 工作流：充满不确定性
def agent_step(state):
    result = llm.invoke(state["query"])     # ❌ LLM 可能返回垃圾内容
    tool_result = tool.run(result)          # ❌ 工具可能超时
    score = evaluate(tool_result)           # ❌ 评估可能不准确
    if score > 0.8:
        return "done"                       # 这个分支可能永远走不到！
    return "retry"                          # 无限重试的风险！
```

### 四种典型风险

| 风险 | 场景 | 后果 |
|------|------|------|
| LLM 调用失败 | API 限流、网络超时、模型过载 | 节点抛异常，图执行中断 |
| 工具执行超时 | 外部 API 无响应、数据库查询卡死 | 节点永远不返回 |
| 质量不达标 | LLM 输出不符合格式、检索结果不相关 | 条件边反复路由到重试，形成死循环 |
| 逻辑死锁 | 两个节点互相路由、状态永远不满足退出条件 | 无限循环，资源耗尽 |

**核心问题：没有回退机制的条件分支，就像没有刹车的汽车——能跑，但迟早出事。**

---

## 循环守卫（Loop Guard）

### 问题：条件分支 + 循环 = 潜在的无限循环

```python
# ⚠️ 危险：没有循环守卫的重试逻辑
def should_retry(state) -> Literal["retry", "done"]:
    if state["quality_score"] < 0.8:
        return "retry"    # 如果 LLM 一直输出低质量内容呢？
    return "done"

builder.add_conditional_edges("evaluate", should_retry, {
    "retry": "generate",   # 回到生成节点
    "done": "output",
})
# 最坏情况：generate → evaluate → generate → evaluate → ... 永远不停
```

### 解决方案：max_steps 计数器模式

核心思路：**在状态中维护一个计数器，每次循环 +1，条件边检查计数器决定继续还是停止。**

```python
from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    query: str
    answer: str
    quality_score: float
    retry_count: int          # 循环计数器
    max_retries: int          # 最大重试次数


def generate(state: State) -> dict:
    """生成回答（模拟 LLM 调用）"""
    # 实际项目中这里调用 LLM
    attempt = state.get("retry_count", 0) + 1
    print(f"  [生成] 第 {attempt} 次尝试...")

    # 模拟：第 3 次才生成高质量内容
    if attempt >= 3:
        return {"answer": "高质量回答", "quality_score": 0.9}
    return {"answer": "低质量回答", "quality_score": 0.5}


def evaluate(state: State) -> dict:
    """评估回答质量"""
    score = state["quality_score"]
    print(f"  [评估] 质量分数: {score}")
    return {"retry_count": state.get("retry_count", 0) + 1}


# ✅ 循环守卫：检查计数器
def should_continue(state: State) -> Literal["retry", "halt"]:
    """条件边：决定继续重试还是停止"""
    retry_count = state.get("retry_count", 0)
    max_retries = state.get("max_retries", 3)

    # 守卫条件1：质量达标 → 正常结束
    if state.get("quality_score", 0) >= 0.8:
        print(f"  [路由] 质量达标，结束")
        return "halt"

    # 守卫条件2：超过最大重试次数 → 强制停止
    if retry_count >= max_retries:
        print(f"  [路由] 已重试 {retry_count} 次，强制停止")
        return "halt"

    print(f"  [路由] 第 {retry_count} 次重试（最多 {max_retries} 次）")
    return "retry"


# 构建图
builder = StateGraph(State)
builder.add_node("generate", generate)
builder.add_node("evaluate", evaluate)

builder.add_edge(START, "generate")
builder.add_edge("generate", "evaluate")

# 条件边 + 循环守卫
builder.add_conditional_edges("evaluate", should_continue, {
    "retry": "generate",    # 回到生成节点
    "halt": END,            # 结束
})

graph = builder.compile()

# 测试
result = graph.invoke({
    "query": "什么是 RAG？",
    "answer": "",
    "quality_score": 0.0,
    "retry_count": 0,
    "max_retries": 5,
})
print(f"\n最终回答: {result['answer']}")
print(f"重试次数: {result['retry_count']}")
```

**运行输出：**

```
  [生成] 第 1 次尝试...
  [评估] 质量分数: 0.5
  [路由] 第 1 次重试（最多 5 次）
  [生成] 第 2 次尝试...
  [评估] 质量分数: 0.5
  [路由] 第 2 次重试（最多 5 次）
  [生成] 第 3 次尝试...
  [评估] 质量分数: 0.9
  [路由] 质量达标，结束

最终回答: 高质量回答
重试次数: 3
```

### 循环守卫的关键要素

```
┌─────────────────────────────────────────────┐
│              循环守卫三要素                     │
├─────────────────────────────────────────────┤
│  1. 计数器字段：retry_count（状态中维护）       │
│  2. 上限配置：max_retries（可配置）             │
│  3. 条件检查：在路由函数中判断是否超限           │
└─────────────────────────────────────────────┘
```

## 错误路由模式

### 核心思路：捕获异常 → 写入状态 → 条件边路由

传统做法是让异常直接抛出，图执行中断。更好的做法是：**在节点内部捕获异常，将错误信息写入状态，让条件边根据错误类型决定下一步。**

```python
# ❌ 传统做法：异常直接抛出
def process_node(state):
    result = call_llm(state["query"])  # 如果超时，整个图崩溃
    return {"result": result}

# ✅ 错误路由：捕获异常，写入状态
def process_node(state):
    try:
        result = call_llm(state["query"])
        return {"result": result, "status": "success"}
    except TimeoutError:
        return {"status": "timeout", "error_count": state.get("error_count", 0) + 1}
    except ValueError as e:
        return {"status": "invalid", "error_msg": str(e)}
```

### 完整错误路由示例

```python
from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END
import random


class State(TypedDict):
    query: str
    result: str
    status: str              # success / timeout / invalid / unknown
    error_count: int
    error_msg: str


def process_node(state: State) -> dict:
    """模拟一个可能失败的处理节点"""
    try:
        # 模拟不同类型的错误
        roll = random.random()
        if roll < 0.3:
            raise TimeoutError("LLM 响应超时")
        if roll < 0.5:
            raise ValueError("输出格式不合法")

        return {
            "result": f"处理成功: {state['query']}",
            "status": "success",
        }
    except TimeoutError as e:
        print(f"  [错误] 超时: {e}")
        return {
            "status": "timeout",
            "error_msg": str(e),
            "error_count": state.get("error_count", 0) + 1,
        }
    except ValueError as e:
        print(f"  [错误] 无效: {e}")
        return {
            "status": "invalid",
            "error_msg": str(e),
            "error_count": state.get("error_count", 0) + 1,
        }


# ✅ 条件边：根据错误类型路由到不同处理路径
def route_by_status(state: State) -> Literal["success", "retry", "fallback"]:
    """根据状态中的 status 字段决定下一步"""
    status = state["status"]

    if status == "success":
        return "success"

    # 超时错误：可以重试（但有次数限制）
    if status == "timeout" and state.get("error_count", 0) < 3:
        print(f"  [路由] 超时，第 {state['error_count']} 次重试")
        return "retry"

    # 其他错误或重试次数用完：降级处理
    print(f"  [路由] 降级处理（状态: {status}，错误次数: {state.get('error_count', 0)}）")
    return "fallback"


def success_handler(state: State) -> dict:
    print(f"  [成功] {state['result']}")
    return {}


def fallback_handler(state: State) -> dict:
    """降级处理：返回兜底结果"""
    print(f"  [降级] 使用缓存结果，原因: {state.get('error_msg', '未知')}")
    return {"result": f"[降级结果] 针对 '{state['query']}' 的缓存回答"}


# 构建图
builder = StateGraph(State)
builder.add_node("process", process_node)
builder.add_node("success_handler", success_handler)
builder.add_node("fallback_handler", fallback_handler)

builder.add_edge(START, "process")

# 错误路由：不同状态 → 不同路径
builder.add_conditional_edges("process", route_by_status, {
    "success": "success_handler",
    "retry": "process",            # 回到自身重试
    "fallback": "fallback_handler",
})

builder.add_edge("success_handler", END)
builder.add_edge("fallback_handler", END)

graph = builder.compile()
```

### 错误路由流程图

```
                    ┌──────────┐
                    │ process  │
                    └────┬─────┘
                         │
                    ┌────▼─────┐
                    │ 检查状态  │
                    └────┬─────┘
                         │
            ┌────────────┼────────────┐
            │            │            │
       success       timeout       invalid
            │            │            │
            │       重试次数<3?        │
            │        ╱     ╲          │
            │      是       否        │
            │       │        │        │
            ▼       ▼        ▼        ▼
        ┌──────┐ ┌──────┐ ┌──────────┐
        │ 成功 │ │ 重试 │ │ 降级处理  │
        └──────┘ └──────┘ └──────────┘
```

## 降级策略（Fallback Strategy）

### 什么是降级？

降级就是：**主方案失败时，自动切换到备用方案，保证系统仍然能返回结果，而不是直接报错。**

### 前端类比：图片加载失败

```html
<!-- 前端的降级策略 -->
<img
  src="high-res.jpg"
  onerror="this.src='low-res.jpg'"
  alt="如果图片都加载不了，至少显示这段文字"
/>
```

### 四种常见降级路径

```
┌─────────────────────────────────────────────────────────┐
│                    降级策略层级                            │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  Level 1: 重试（同一方案再试一次）                         │
│     主模型超时 → 等一下再调用主模型                         │
│                                                         │
│  Level 2: 降级（切换到备用方案）                           │
│     主模型失败 → 切换到备用模型                             │
│     GPT-4 失败 → 切换到 GPT-3.5                          │
│                                                         │
│  Level 3: 缓存（使用历史结果）                             │
│     API 全部失败 → 返回缓存的相似问题答案                   │
│                                                         │
│  Level 4: 人工兜底（转交人类处理）                         │
│     所有自动方案失败 → 通知人工介入                         │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

### 降级路由实现

```python
from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    query: str
    result: str
    model_tier: int          # 1=主模型, 2=备用, 3=缓存
    error_count: int
    fallback_reason: str


def primary_model(state: State) -> dict:
    """主模型（模拟 GPT-4 失败）"""
    return {"status": "error", "error_count": 1, "fallback_reason": "GPT-4 超时", "model_tier": 1}


def backup_model(state: State) -> dict:
    """备用模型（模拟 GPT-3.5 也失败）"""
    return {"status": "error", "error_count": 2, "fallback_reason": "GPT-3.5 也超时", "model_tier": 2}


def cache_fallback(state: State) -> dict:
    """缓存降级：返回预设答案"""
    return {"result": f"[缓存回答] 关于 '{state['query']}' 的预设答案", "model_tier": 3}


def route_fallback(state: State) -> Literal["backup", "cache", "done"]:
    """降级路由：逐级降级"""
    if state.get("status") == "success":
        return "done"
    tier = state.get("model_tier", 0)
    if tier <= 1:
        return "backup"     # 主模型失败 → 备用模型
    return "cache"           # 备用也失败 → 缓存兜底


builder = StateGraph(State)
builder.add_node("primary", primary_model)
builder.add_node("backup", backup_model)
builder.add_node("cache", cache_fallback)

builder.add_edge(START, "primary")
builder.add_conditional_edges("primary", route_fallback, {
    "backup": "backup", "cache": "cache", "done": END,
})
builder.add_conditional_edges("backup", route_fallback, {
    "cache": "cache", "done": END, "backup": "cache",
})
builder.add_edge("cache", END)

graph = builder.compile()

result = graph.invoke({"query": "什么是向量数据库？", "result": "", "model_tier": 0, "error_count": 0, "fallback_reason": ""})
print(f"最终结果: {result['result']}")
# 最终结果: [缓存回答] 关于 '什么是向量数据库？' 的预设答案
```

## 指数退避与状态恢复

### 指数退避：避免频繁重试

重试时不要立即重试，而是逐渐增加等待时间：

```python
import random

# 指数退避公式
wait_time = min(2 ** retry_count, max_wait)

# 实际等待时间：
# 第 1 次重试：2^1 = 2 秒
# 第 2 次重试：2^2 = 4 秒
# 第 3 次重试：2^3 = 8 秒
# 封顶：min(2^n, 30) = 最多 30 秒

# 加入随机抖动（避免多个客户端同时重试）
wait_time = min(2 ** retry_count, 30) + random.uniform(0, 1)
```

### 状态恢复：重试时保留已完成的工作

```python
def process_with_checkpoint(state) -> dict:
    """带检查点的处理：跳过已完成的步骤"""
    completed = state.get("completed_steps", [])

    if "step1" not in completed:
        step1_result = expensive_step1(state)
    else:
        step1_result = state["step1_result"]  # 直接用缓存

    try:
        step2_result = expensive_step2(step1_result)
        return {"completed_steps": ["step1", "step2"], "status": "success"}
    except Exception:
        return {
            "step1_result": step1_result,       # 保存 step1 结果
            "completed_steps": ["step1"],        # 记录已完成步骤
            "status": "error",
        }
```

**关键思路：把中间结果存入状态，重试时跳过已完成的步骤。**

## 最佳实践

### 1. 总是设置 max_steps

```python
# ✅ 每个循环都要有退出条件
def should_continue(state) -> Literal["continue", "stop"]:
    if state["steps"] >= state.get("max_steps", 10):
        return "stop"    # 硬性上限
    if state["quality"] >= 0.8:
        return "stop"    # 质量达标
    return "continue"
```

### 2. 记录每次重试的原因

```python
# ✅ 记录为什么重试
except Exception as e:
    return {
        "error_log": [f"第 {state['retry_count'] + 1} 次失败: {type(e).__name__}: {e}"],
        "retry_count": state.get("retry_count", 0) + 1,
    }
```

### 3. 降级时通知用户

```python
# ✅ 告诉用户发生了降级
return {
    "result": cached_answer,
    "metadata": {"is_fallback": True, "confidence": "low"},
}
```

### 4. 避免静默失败

```python
# ✅ 明确告知失败，而不是假装成功
if state.get("all_attempts_failed"):
    return {"result": "抱歉，所有处理方案均已失败", "suggestion": "请稍后重试"}
```

## 回退机制速查表

| 模式 | 适用场景 | 实现方式 |
|------|----------|----------|
| 循环守卫 | 防止无限循环 | 状态中维护 `retry_count`，条件边检查上限 |
| 错误路由 | 不同错误不同处理 | try-except 捕获，`status` 字段路由 |
| 指数退避 | 避免频繁重试 | `wait = min(2^n, max)` |
| 模型降级 | 主模型不可用 | 条件边路由到备用模型节点 |
| 缓存兜底 | 所有模型不可用 | 返回预计算的缓存结果 |
| 人工兜底 | 自动方案全部失败 | 路由到人工处理节点 |
| 状态恢复 | 重试时保留进度 | 检查点字段 + 跳过已完成步骤 |

## 与 RAG 开发的关系

回退机制在 RAG 系统中尤为重要：

1. **检索阶段**：向量数据库超时 → 降级到关键词搜索 → 降级到缓存结果
2. **生成阶段**：GPT-4 超时 → 降级到 GPT-3.5 → 降级到模板回答
3. **质量评估**：Self-RAG 中评估不达标 → 重新检索 → 重新生成（需要循环守卫）
4. **多步推理**：Agent 工具调用失败 → 重试 → 换工具 → 直接回答

```python
# RAG 中的典型回退场景
def route_rag_result(state) -> Literal["generate", "re_retrieve", "fallback"]:
    """RAG 检索质量路由"""
    if state["retrieval_score"] > 0.8:
        return "generate"           # 检索质量好，直接生成
    if state["retry_count"] < 2:
        return "re_retrieve"        # 质量不够，重新检索（换策略）
    return "fallback"               # 多次检索都不行，降级处理
```

## 总结

回退机制和循环守卫是条件分支策略中的"安全网"：

- **循环守卫**：给每个循环装上"刹车"，通过计数器防止无限循环
- **错误路由**：把异常变成状态，让条件边智能处理不同错误
- **降级策略**：主方案失败时逐级降级，保证系统始终有响应
- **状态恢复**：重试时保留已完成的工作，避免重复计算

**记住一个原则：在 Agent 工作流中，"优雅降级"永远比"直接崩溃"好。**
