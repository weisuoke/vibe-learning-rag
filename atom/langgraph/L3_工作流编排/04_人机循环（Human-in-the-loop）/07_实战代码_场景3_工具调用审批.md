## 场景说明

**业务场景**：AI Agent 在执行敏感操作前，需要人工确认

典型例子：
- Agent 要发送邮件 → 先让人确认收件人和内容
- Agent 要执行数据库操作 → 先让人确认 SQL 语句
- Agent 要调用外部 API → 先让人确认参数

这是 Agent 安全性的核心保障。我们用 `@tool` 装饰器定义工具，用 `interrupt()` 在执行前暂停。

## 子场景 1：基础工具审批（approve / reject）

### 思路

最简单的工具审批：展示工具名称和参数，等待人工确认。

```
[START] → [plan_tool_call] → [approve_tool] → approve → [execute_tool] → [END]
                                             → reject  → [skip_tool]   → [END]
```

### 完整代码

```python
"""
子场景 1：基础工具审批
- 使用 @tool 装饰器定义工具
- 在执行工具前用 interrupt() 展示调用信息
- 人工确认后才真正执行
"""

from typing import Any, Literal, Optional, TypedDict
from langchain_core.tools import tool
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


# ========== 1. 定义工具 ==========

@tool
def send_email(to: str, subject: str, body: str) -> str:
    """发送邮件给指定收件人"""
    # 实际项目中这里会调用邮件 API
    print(f"  [邮件已发送] 收件人: {to}")
    print(f"  主题: {subject}")
    print(f"  内容: {body[:50]}...")
    return f"邮件已发送给 {to}"


@tool
def delete_record(table: str, record_id: str) -> str:
    """从数据库中删除记录"""
    # 实际项目中这里会执行 SQL
    print(f"  [记录已删除] 表: {table}, ID: {record_id}")
    return f"已删除 {table} 中的记录 {record_id}"


# 工具注册表：名称 → 工具对象
TOOL_REGISTRY: dict[str, Any] = {
    "send_email": send_email,
    "delete_record": delete_record,
}


# ========== 2. 定义状态 ==========

class ToolApprovalState(TypedDict):
    """工具审批状态"""
    tool_name: str                    # 要调用的工具名称
    tool_args: dict[str, Any]         # 工具参数
    status: str                       # 流程状态
    result: Optional[str]             # 工具执行结果
    rejection_reason: Optional[str]   # 拒绝原因


# ========== 3. 定义节点 ==========

def plan_tool_call(state: ToolApprovalState):
    """规划工具调用（模拟 Agent 决策）"""
    print(f"\n[规划] Agent 决定调用工具: {state['tool_name']}")
    print(f"  参数: {state['tool_args']}")
    return {"status": "pending_approval"}


def approve_tool(state: ToolApprovalState) -> Command[Literal["execute_tool", "skip_tool"]]:
    """工具审批节点：展示工具信息，等待人工确认"""

    decision = interrupt({
        "type": "tool_approval",
        "tool_name": state["tool_name"],
        "tool_args": state["tool_args"],
        "tool_description": TOOL_REGISTRY.get(state["tool_name"], lambda: None).__doc__ or "",
        "instruction": "请确认是否执行此工具调用。输入 'approve' 或 'reject'",
    })

    if isinstance(decision, dict):
        action = decision.get("action", "reject")
        reason = decision.get("reason", "")
    else:
        action = str(decision).lower()
        reason = ""

    if action == "approve":
        return Command(goto="execute_tool", update={"status": "approved"})
    else:
        return Command(
            goto="skip_tool",
            update={"status": "rejected", "rejection_reason": reason or "人工拒绝"},
        )


def execute_tool(state: ToolApprovalState):
    """执行工具"""
    tool_fn = TOOL_REGISTRY.get(state["tool_name"])
    if tool_fn is None:
        return {"result": f"错误：未找到工具 {state['tool_name']}", "status": "error"}

    print(f"\n[执行] 调用工具 {state['tool_name']}")
    # 使用 .invoke() 调用 LangChain 工具
    result = tool_fn.invoke(state["tool_args"])
    return {"result": result, "status": "completed"}


def skip_tool(state: ToolApprovalState):
    """跳过工具执行"""
    print(f"\n[跳过] 工具 {state['tool_name']} 被拒绝执行")
    print(f"  原因: {state.get('rejection_reason', '未知')}")
    return {"result": None, "status": "skipped"}


# ========== 4. 构建图 ==========

builder = StateGraph(ToolApprovalState)
builder.add_node("plan", plan_tool_call)
builder.add_node("approve", approve_tool)
builder.add_node("execute_tool", execute_tool)
builder.add_node("skip_tool", skip_tool)

builder.add_edge(START, "plan")
builder.add_edge("plan", "approve")
builder.add_edge("execute_tool", END)
builder.add_edge("skip_tool", END)

graph = builder.compile(checkpointer=MemorySaver())


# ===== 演示 A：批准发送邮件 =====

print("=" * 60)
print("演示 A：批准发送邮件")
print("=" * 60)

config_a = {"configurable": {"thread_id": "tool-approval-001"}}

graph.invoke(
    {
        "tool_name": "send_email",
        "tool_args": {
            "to": "client@example.com",
            "subject": "项目进度报告",
            "body": "尊敬的客户，以下是本周项目进度...",
        },
        "status": "draft",
        "result": None,
        "rejection_reason": None,
    },
    config_a,
)

# 查看待审批信息
snapshot = graph.get_state(config_a)
for task in snapshot.tasks:
    for intr in task.interrupts:
        info = intr.value
        print(f"\n待审批工具: {info['tool_name']}")
        print(f"参数: {info['tool_args']}")

# 批准执行
result_a = graph.invoke(Command(resume={"action": "approve"}), config_a)
print(f"\n状态: {result_a['status']}")
print(f"结果: {result_a['result']}")


# ===== 演示 B：拒绝删除操作 =====

print("\n" + "=" * 60)
print("演示 B：拒绝删除操作")
print("=" * 60)

config_b = {"configurable": {"thread_id": "tool-approval-002"}}

graph.invoke(
    {
        "tool_name": "delete_record",
        "tool_args": {"table": "users", "record_id": "USR-001"},
        "status": "draft",
        "result": None,
        "rejection_reason": None,
    },
    config_b,
)

# 拒绝执行
result_b = graph.invoke(
    Command(resume={"action": "reject", "reason": "不能删除用户数据，需要先备份"}),
    config_b,
)
print(f"\n状态: {result_b['status']}")
print(f"拒绝原因: {result_b['rejection_reason']}")
```

### 关键点

1. `@tool` 装饰器定义的工具通过 `.invoke()` 调用，与 LangChain 生态无缝集成
2. `interrupt()` 传递的信息包含工具名、参数、描述，客户端可以渲染成可视化审批界面
3. 审批和执行分成两个独立节点，职责清晰
4. 拒绝时记录原因，方便后续审计

## 子场景 2：参数编辑审批（审批人可修改工具参数）

### 思路

审批人不仅可以 approve/reject，还可以修改工具参数后再执行。

比如：Agent 要发邮件给 A，审批人改成发给 B；Agent 要删除记录，审批人改成归档。

```
[START] → [plan] → [approve] → approve        → [execute_tool] → [END]
                              → edit + approve → [execute_tool] → [END]  (参数被修改)
                              → reject         → [skip_tool]   → [END]
```

### 完整代码

```python
"""
子场景 2：参数编辑审批
- 审批人可以修改工具参数后批准执行
- 修改后的参数通过 Command(update=...) 写入状态
- 执行节点使用修改后的参数
"""

from typing import Any, Literal, Optional, TypedDict
from langchain_core.tools import tool
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


# ========== 工具定义 ==========

@tool
def transfer_money(from_account: str, to_account: str, amount: float) -> str:
    """执行转账操作"""
    print(f"  [转账完成] {from_account} → {to_account}: ¥{amount:.2f}")
    return f"已从 {from_account} 转账 ¥{amount:.2f} 到 {to_account}"


@tool
def send_notification(channel: str, message: str) -> str:
    """发送通知消息"""
    print(f"  [通知已发送] 渠道: {channel}")
    print(f"  内容: {message}")
    return f"通知已通过 {channel} 发送"


TOOL_REGISTRY: dict[str, Any] = {
    "transfer_money": transfer_money,
    "send_notification": send_notification,
}


# ========== 状态定义 ==========

class EditableToolState(TypedDict):
    """可编辑工具审批状态"""
    tool_name: str
    tool_args: dict[str, Any]         # 原始参数
    edited_args: dict[str, Any]       # 编辑后的参数（可能与原始相同）
    status: str
    result: Optional[str]
    was_edited: bool                  # 参数是否被修改
    edit_summary: Optional[str]       # 修改摘要


def plan_node(state: EditableToolState):
    """规划工具调用"""
    print(f"\n[规划] 工具: {state['tool_name']}, 参数: {state['tool_args']}")
    return {"status": "pending", "edited_args": state["tool_args"], "was_edited": False}


def approve_node(state: EditableToolState) -> Command[Literal["execute_tool", "skip_tool"]]:
    """审批节点：支持 approve / edit / reject"""

    decision = interrupt({
        "type": "tool_edit_approval",
        "tool_name": state["tool_name"],
        "tool_args": state["tool_args"],
        "instruction": (
            "请审批此工具调用，支持以下操作：\n"
            "  1. {'action': 'approve'} - 直接批准\n"
            "  2. {'action': 'edit', 'edited_args': {...}} - 修改参数后批准\n"
            "  3. {'action': 'reject', 'reason': '...'} - 拒绝"
        ),
    })

    if not isinstance(decision, dict):
        decision = {"action": str(decision)}

    action = decision.get("action", "reject")

    if action == "approve":
        return Command(
            goto="execute_tool",
            update={"status": "approved", "edited_args": state["tool_args"], "was_edited": False},
        )
    elif action == "edit":
        edited_args = decision.get("edited_args", state["tool_args"])
        # 生成修改摘要
        changes = []
        for key in set(list(state["tool_args"].keys()) + list(edited_args.keys())):
            old_val = state["tool_args"].get(key)
            new_val = edited_args.get(key)
            if old_val != new_val:
                changes.append(f"{key}: {old_val} → {new_val}")
        summary = "; ".join(changes) if changes else "无变化"

        return Command(
            goto="execute_tool",
            update={
                "status": "approved_edited",
                "edited_args": edited_args,
                "was_edited": True,
                "edit_summary": summary,
            },
        )
    else:
        return Command(
            goto="skip_tool",
            update={"status": "rejected"},
        )


def execute_tool_node(state: EditableToolState):
    """执行工具（使用编辑后的参数）"""
    tool_fn = TOOL_REGISTRY.get(state["tool_name"])
    if tool_fn is None:
        return {"result": f"错误：未找到工具 {state['tool_name']}", "status": "error"}

    # 关键：使用 edited_args 而不是 tool_args
    args_to_use = state["edited_args"]

    print(f"\n[执行] 工具: {state['tool_name']}")
    if state["was_edited"]:
        print(f"  参数已修改: {state['edit_summary']}")
    print(f"  使用参数: {args_to_use}")

    result = tool_fn.invoke(args_to_use)
    return {"result": result, "status": "completed"}


def skip_tool_node(state: EditableToolState):
    """跳过"""
    print(f"\n[跳过] 工具 {state['tool_name']} 被拒绝")
    return {"status": "skipped"}


# 构建图
builder = StateGraph(EditableToolState)
builder.add_node("plan", plan_node)
builder.add_node("approve", approve_node)
builder.add_node("execute_tool", execute_tool_node)
builder.add_node("skip_tool", skip_tool_node)

builder.add_edge(START, "plan")
builder.add_edge("plan", "approve")
builder.add_edge("execute_tool", END)
builder.add_edge("skip_tool", END)

graph = builder.compile(checkpointer=MemorySaver())


# ===== 演示 A：直接批准转账 =====

print("=" * 60)
print("演示 A：直接批准转账")
print("=" * 60)

config_a = {"configurable": {"thread_id": "edit-tool-001"}}

graph.invoke(
    {
        "tool_name": "transfer_money",
        "tool_args": {"from_account": "ACC-001", "to_account": "ACC-002", "amount": 1000.0},
        "edited_args": {},
        "status": "draft",
        "result": None,
        "was_edited": False,
        "edit_summary": None,
    },
    config_a,
)

result_a = graph.invoke(Command(resume={"action": "approve"}), config_a)
print(f"\n状态: {result_a['status']}")
print(f"结果: {result_a['result']}")
print(f"参数被修改: {result_a['was_edited']}")


# ===== 演示 B：修改转账金额后批准 =====

print("\n" + "=" * 60)
print("演示 B：修改转账金额后批准")
print("=" * 60)

config_b = {"configurable": {"thread_id": "edit-tool-002"}}

graph.invoke(
    {
        "tool_name": "transfer_money",
        "tool_args": {"from_account": "ACC-001", "to_account": "ACC-003", "amount": 50000.0},
        "edited_args": {},
        "status": "draft",
        "result": None,
        "was_edited": False,
        "edit_summary": None,
    },
    config_b,
)

# 审批人认为金额过大，调低到 10000
result_b = graph.invoke(
    Command(resume={
        "action": "edit",
        "edited_args": {"from_account": "ACC-001", "to_account": "ACC-003", "amount": 10000.0},
    }),
    config_b,
)
print(f"\n状态: {result_b['status']}")
print(f"结果: {result_b['result']}")
print(f"参数被修改: {result_b['was_edited']}")
print(f"修改摘要: {result_b['edit_summary']}")


# ===== 演示 C：修改通知渠道后批准 =====

print("\n" + "=" * 60)
print("演示 C：修改通知渠道后批准")
print("=" * 60)

config_c = {"configurable": {"thread_id": "edit-tool-003"}}

graph.invoke(
    {
        "tool_name": "send_notification",
        "tool_args": {"channel": "sms", "message": "系统维护通知：今晚 22:00 停机维护"},
        "edited_args": {},
        "status": "draft",
        "result": None,
        "was_edited": False,
        "edit_summary": None,
    },
    config_c,
)

# 审批人改为通过邮件发送
result_c = graph.invoke(
    Command(resume={
        "action": "edit",
        "edited_args": {"channel": "email", "message": "系统维护通知：今晚 22:00 停机维护，预计 2 小时"},
    }),
    config_c,
)
print(f"\n状态: {result_c['status']}")
print(f"结果: {result_c['result']}")
print(f"修改摘要: {result_c['edit_summary']}")
```

### 关键点

1. 状态中区分 `tool_args`（原始参数）和 `edited_args`（编辑后参数），方便审计
2. 审批节点自动生成修改摘要（`edit_summary`），记录哪些参数被改了
3. 执行节点始终使用 `edited_args`，无论是否被修改过
4. `@tool` 装饰器的工具通过 `.invoke(dict)` 调用，参数以字典形式传入

## 子场景 3：多工具批量审批

### 思路

Agent 可能一次规划多个工具调用，审批人可以一次性审批所有工具。

每个工具可以独立 approve/reject/edit，最后只执行被批准的工具。

```
[START] → [plan_batch] → [batch_approve] → [execute_batch] → [END]
```

### 完整代码

```python
"""
子场景 3：多工具批量审批
- Agent 一次规划多个工具调用
- 审批人对每个工具独立做出决策
- 只执行被批准的工具
"""

from typing import Any, Optional, TypedDict
from langchain_core.tools import tool
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


# ========== 工具定义 ==========

@tool
def create_user(username: str, role: str) -> str:
    """创建新用户"""
    print(f"  [用户已创建] {username} (角色: {role})")
    return f"用户 {username} 已创建，角色: {role}"


@tool
def grant_permission(username: str, permission: str) -> str:
    """授予用户权限"""
    print(f"  [权限已授予] {username} → {permission}")
    return f"已授予 {username} 权限: {permission}"


@tool
def send_welcome_email(username: str, email: str) -> str:
    """发送欢迎邮件"""
    print(f"  [欢迎邮件已发送] {email}")
    return f"欢迎邮件已发送给 {email}"


TOOL_REGISTRY: dict[str, Any] = {
    "create_user": create_user,
    "grant_permission": grant_permission,
    "send_welcome_email": send_welcome_email,
}


# ========== 状态定义 ==========

class ToolCall(TypedDict):
    """单个工具调用"""
    tool_name: str
    tool_args: dict[str, Any]


class BatchToolState(TypedDict):
    """批量工具审批状态"""
    tool_calls: list[ToolCall]        # 待执行的工具调用列表
    decisions: list[dict[str, Any]]   # 每个工具的审批决策
    results: list[dict[str, Any]]     # 执行结果
    status: str


# ========== 节点定义 ==========

def plan_batch(state: BatchToolState):
    """规划批量工具调用"""
    print(f"\n[批量规划] 共 {len(state['tool_calls'])} 个工具调用:")
    for i, tc in enumerate(state["tool_calls"]):
        print(f"  {i+1}. {tc['tool_name']}({tc['tool_args']})")
    return {"status": "pending_batch_approval", "decisions": [], "results": []}


def batch_approve(state: BatchToolState):
    """批量审批节点：一次性展示所有工具调用，等待逐个审批"""

    # 构建审批信息
    approval_items = []
    for i, tc in enumerate(state["tool_calls"]):
        approval_items.append({
            "index": i,
            "tool_name": tc["tool_name"],
            "tool_args": tc["tool_args"],
        })

    decisions = interrupt({
        "type": "batch_tool_approval",
        "tool_calls": approval_items,
        "instruction": (
            "请对每个工具调用做出决策。返回格式：\n"
            "[\n"
            "  {'action': 'approve'},           # 批准第1个\n"
            "  {'action': 'reject'},            # 拒绝第2个\n"
            "  {'action': 'edit', 'edited_args': {...}}  # 修改第3个\n"
            "]"
        ),
    })

    # decisions 应该是一个列表，每个元素对应一个工具调用的决策
    if not isinstance(decisions, list):
        # 如果返回单个值，应用到所有工具
        decisions = [{"action": str(decisions)}] * len(state["tool_calls"])

    return {"decisions": decisions, "status": "batch_approved"}


def execute_batch(state: BatchToolState):
    """批量执行：只执行被批准的工具"""
    results = []

    for i, tc in enumerate(state["tool_calls"]):
        decision = state["decisions"][i] if i < len(state["decisions"]) else {"action": "reject"}
        action = decision.get("action", "reject")

        if action in ("approve", "edit"):
            # 确定使用的参数
            args = decision.get("edited_args", tc["tool_args"]) if action == "edit" else tc["tool_args"]

            tool_fn = TOOL_REGISTRY.get(tc["tool_name"])
            if tool_fn:
                print(f"\n[执行 {i+1}/{len(state['tool_calls'])}] {tc['tool_name']}")
                if action == "edit":
                    print(f"  (参数已修改)")
                result = tool_fn.invoke(args)
                results.append({
                    "tool_name": tc["tool_name"],
                    "status": "executed",
                    "result": result,
                    "was_edited": action == "edit",
                })
            else:
                results.append({
                    "tool_name": tc["tool_name"],
                    "status": "error",
                    "result": f"工具 {tc['tool_name']} 未找到",
                })
        else:
            print(f"\n[跳过 {i+1}/{len(state['tool_calls'])}] {tc['tool_name']} (被拒绝)")
            results.append({
                "tool_name": tc["tool_name"],
                "status": "skipped",
                "result": None,
            })

    return {"results": results, "status": "completed"}


# 构建图
builder = StateGraph(BatchToolState)
builder.add_node("plan", plan_batch)
builder.add_node("approve", batch_approve)
builder.add_node("execute", execute_batch)

builder.add_edge(START, "plan")
builder.add_edge("plan", "approve")
builder.add_edge("approve", "execute")
builder.add_edge("execute", END)

graph = builder.compile(checkpointer=MemorySaver())
```

### 关键点

1. 批量审批用一个 `interrupt()` 传递所有工具调用信息，resume 值是一个列表
2. 列表中每个元素对应一个工具的决策，支持独立的 approve/reject/edit
3. 执行节点遍历决策列表，只执行被批准的工具
4. 这种模式适合 Agent 一次规划多步操作的场景（如入职流程、部署流程）

## 三个场景对比

| 特性 | 场景 1：基础审批 | 场景 2：参数编辑 | 场景 3：批量审批 |
|------|-----------------|-----------------|-----------------|
| 工具数量 | 单个 | 单个 | 多个 |
| 审批操作 | approve/reject | approve/edit/reject | 每个工具独立决策 |
| 参数修改 | 不支持 | 支持 | 支持 |
| interrupt 次数 | 1 次 | 1 次 | 1 次 |
| resume 格式 | `{"action": "..."}` | `{"action": "edit", "edited_args": {...}}` | `[{...}, {...}, ...]` |
| 适用场景 | 简单确认 | 需要调整参数 | Agent 多步规划 |

## 核心模式总结

工具调用审批的通用模式：

```python
# 1. 定义工具
@tool
def my_tool(param: str) -> str:
    """工具描述"""
    return f"执行结果: {param}"

# 2. 审批节点：展示工具信息，等待决策
def approve_node(state):
    decision = interrupt({
        "tool_name": state["tool_name"],
        "tool_args": state["tool_args"],
    })

    action = decision.get("action")
    if action == "approve":
        return Command(goto="execute")
    elif action == "edit":
        edited = decision.get("edited_args")
        return Command(goto="execute", update={"tool_args": edited})
    else:
        return Command(goto="skip")

# 3. 执行节点：调用工具
def execute_node(state):
    tool_fn = REGISTRY[state["tool_name"]]
    result = tool_fn.invoke(state["tool_args"])
    return {"result": result}
```

这个模式的核心思想：**Agent 负责规划，人类负责把关**。Agent 决定"做什么"，人类确认"是否做"和"怎么做"。
