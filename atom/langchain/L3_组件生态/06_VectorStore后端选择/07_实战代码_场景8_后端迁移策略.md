```python
"""
向量存储后端迁移框架
提供通用的迁移策略和工具
"""

from langchain_core.vectorstores import VectorStore
from langchain_core.documents import Document
from langchain_openai import OpenAIEmbeddings
from typing import List, Dict, Any, Optional
from dotenv import load_dotenv
import time
from tqdm import tqdm

# 加载环境变量
load_dotenv()


class VectorStoreMigrator:
    """向量存储迁移器"""
    
    def __init__(self, source: VectorStore, target: VectorStore):
        """
        初始化迁移器
        
        Args:
            source: 源向量存储
            target: 目标向量存储
        """
        self.source = source
        self.target = target
        self.migration_log = []
    
    def extract_documents(self, batch_size: int = 100) -> List[Document]:
        """
        从源向量存储提取文档
        
        Args:
            batch_size: 批次大小
            
        Returns:
            文档列表
        """
        print("\n[步骤1] 从源向量存储提取文档...")
        
        # 注意：不同的向量存储有不同的提取方法
        # 这里提供通用的提取策略
        
        documents = []
        
        # 方法1：如果源支持 get_by_ids
        try:
            # 尝试获取所有文档
            # 注意：这需要知道所有文档的 ID
            # 实际应用中可能需要其他方法
            pass
        except NotImplementedError:
            print("  源向量存储不支持 get_by_ids，使用其他方法...")
        
        # 方法2：通过检索获取文档
        # 这是一个通用但不完美的方法
        print("  使用检索方法提取文档...")
        
        # 使用多个查询来覆盖更多文档
        queries = [
            "文档", "内容", "数据", "信息", "知识",
            "document", "content", "data", "information", "knowledge"
        ]
        
        seen_ids = set()
        for query in queries:
            try:
                results = self.source.similarity_search(query, k=batch_size)
                for doc in results:
                    if doc.id and doc.id not in seen_ids:
                        documents.append(doc)
                        seen_ids.add(doc.id)
            except Exception as e:
                print(f"  查询 '{query}' 失败: {e}")
        
        print(f"✓ 提取了 {len(documents)} 个文档")
        return documents
    
    def migrate_documents(
        self,
        documents: List[Document],
        batch_size: int = 100,
        verify: bool = True
    ) -> Dict[str, Any]:
        """
        迁移文档到目标向量存储
        
        Args:
            documents: 要迁移的文档列表
            batch_size: 批次大小
            verify: 是否验证迁移结果
            
        Returns:
            迁移统计信息
        """
        print(f"\n[步骤2] 迁移 {len(documents)} 个文档到目标向量存储...")
        
        total_docs = len(documents)
        migrated_count = 0
        failed_count = 0
        
        # 分批迁移
        for i in tqdm(range(0, total_docs, batch_size), desc="迁移进度"):
            batch = documents[i:i + batch_size]
            
            try:
                # 添加到目标向量存储
                start_time = time.time()
                ids = self.target.add_documents(batch)
                elapsed = time.time() - start_time
                
                migrated_count += len(ids)
                
                self.migration_log.append({
                    "batch": i // batch_size + 1,
                    "docs_count": len(batch),
                    "success": True,
                    "time": elapsed
                })
                
            except Exception as e:
                failed_count += len(batch)
                print(f"\n  批次 {i // batch_size + 1} 失败: {e}")
                
                self.migration_log.append({
                    "batch": i // batch_size + 1,
                    "docs_count": len(batch),
                    "success": False,
                    "error": str(e)
                })
        
        print(f"\n✓ 迁移完成:")
        print(f"  成功: {migrated_count} 个文档")
        print(f"  失败: {failed_count} 个文档")
        
        # 验证迁移结果
        if verify:
            self.verify_migration(documents)
        
        return {
            "total": total_docs,
            "migrated": migrated_count,
            "failed": failed_count,
            "success_rate": migrated_count / total_docs if total_docs > 0 else 0
        }
    
    def verify_migration(self, original_documents: List[Document]) -> bool:
        """
        验证迁移结果
        
        Args:
            original_documents: 原始文档列表
            
        Returns:
            验证是否通过
        """
        print("\n[步骤3] 验证迁移结果...")
        
        # 抽样验证
        sample_size = min(10, len(original_documents))
        sample_docs = original_documents[:sample_size]
        
        verified_count = 0
        
        for doc in sample_docs:
            try:
                # 在目标向量存储中检索
                results = self.target.similarity_search(
                    doc.page_content[:100],  # 使用前100个字符
                    k=1
                )
                
                if results and results[0].page_content == doc.page_content:
                    verified_count += 1
                    
            except Exception as e:
                print(f"  验证失败: {e}")
        
        success_rate = verified_count / sample_size
        print(f"✓ 验证完成: {verified_count}/{sample_size} ({success_rate*100:.1f}%)")
        
        return success_rate > 0.8  # 80% 通过率
    
    def generate_report(self) -> str:
        """生成迁移报告"""
        report = "\n" + "=" * 60 + "\n"
        report += "迁移报告\n"
        report += "=" * 60 + "\n"
        
        total_batches = len(self.migration_log)
        successful_batches = sum(1 for log in self.migration_log if log["success"])
        
        report += f"\n总批次数: {total_batches}\n"
        report += f"成功批次: {successful_batches}\n"
        report += f"失败批次: {total_batches - successful_batches}\n"
        
        if successful_batches > 0:
            avg_time = sum(
                log["time"] for log in self.migration_log if log["success"]
            ) / successful_batches
            report += f"平均批次时间: {avg_time:.2f}秒\n"
        
        return report


def migrate_inmemory_to_chroma():
    """从 InMemory 迁移到 Chroma"""
    print("\n" + "=" * 60)
    print("场景1：从 InMemory 迁移到 Chroma")
    print("=" * 60)
    
    from langchain_core.vectorstores import InMemoryVectorStore
    from langchain_chroma import Chroma
    
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    
    # 1. 创建源向量存储（InMemory）
    print("\n[准备] 创建源向量存储...")
    source = InMemoryVectorStore(embeddings)
    
    # 添加测试数据
    documents = [
        Document(
            id=str(i),
            page_content=f"测试文档 {i}：这是一段测试内容。" * 5,
            metadata={"id": i, "category": f"cat_{i % 3}"}
        )
        for i in range(100)
    ]
    source.add_documents(documents)
    print(f"✓ 源向量存储包含 {len(documents)} 个文档")
    
    # 2. 创建目标向量存储（Chroma）
    print("\n[准备] 创建目标向量存储...")
    target = Chroma(
        collection_name="migrated_collection",
        embedding_function=embeddings
    )
    print("✓ 目标向量存储已创建")
    
    # 3. 执行迁移
    migrator = VectorStoreMigrator(source, target)
    
    # 直接迁移（因为我们有完整的文档列表）
    stats = migrator.migrate_documents(documents, batch_size=20)
    
    # 4. 生成报告
    print(migrator.generate_report())
    
    # 5. 清理
    target.delete_collection()


def migrate_chroma_to_faiss():
    """从 Chroma 迁移到 FAISS"""
    print("\n" + "=" * 60)
    print("场景2：从 Chroma 迁移到 FAISS")
    print("=" * 60)
    
    from langchain_chroma import Chroma
    from langchain_community.vectorstores import FAISS
    
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    
    # 1. 创建源向量存储（Chroma）
    print("\n[准备] 创建源向量存储...")
    source = Chroma(
        collection_name="source_collection",
        embedding_function=embeddings
    )
    
    # 添加测试数据
    documents = [
        Document(
            page_content=f"文档 {i}：内容较长以模拟真实场景。" * 10,
            metadata={"id": i}
        )
        for i in range(50)
    ]
    source.add_documents(documents)
    print(f"✓ 源向量存储包含 {len(documents)} 个文档")
    
    # 2. 创建目标向量存储（FAISS）
    print("\n[准备] 创建目标向量存储...")
    # FAISS 需要从文档创建
    target = FAISS.from_documents([], embeddings)
    print("✓ 目标向量存储已创建")
    
    # 3. 执行迁移
    migrator = VectorStoreMigrator(source, target)
    stats = migrator.migrate_documents(documents, batch_size=10)
    
    # 4. 保存 FAISS 索引
    print("\n[保存] 保存 FAISS 索引...")
    target.save_local("./migrated_faiss_index")
    print("✓ FAISS 索引已保存")
    
    # 5. 生成报告
    print(migrator.generate_report())
    
    # 6. 清理
    source.delete_collection()


def migrate_with_zero_downtime():
    """零停机迁移策略"""
    print("\n" + "=" * 60)
    print("场景3：零停机迁移策略")
    print("=" * 60)
    
    from langchain_chroma import Chroma
    from langchain_community.vectorstores import FAISS
    
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    
    # 1. 准备源和目标
    print("\n[阶段1] 准备源和目标向量存储...")
    source = Chroma(
        collection_name="production_db",
        embedding_function=embeddings
    )
    
    # 添加数据
    documents = [
        Document(page_content=f"生产文档 {i}")
        for i in range(30)
    ]
    source.add_documents(documents)
    
    target = FAISS.from_documents([], embeddings)
    print("✓ 准备完成")
    
    # 2. 双写阶段（新数据同时写入两个存储）
    print("\n[阶段2] 双写阶段...")
    print("  新数据将同时写入源和目标")
    
    new_documents = [
        Document(page_content=f"新文档 {i}")
        for i in range(10)
    ]
    
    # 同时写入
    source.add_documents(new_documents)
    target.add_documents(new_documents)
    print("✓ 双写完成")
    
    # 3. 后台迁移历史数据
    print("\n[阶段3] 后台迁移历史数据...")
    migrator = VectorStoreMigrator(source, target)
    stats = migrator.migrate_documents(documents, batch_size=10)
    
    # 4. 验证阶段
    print("\n[阶段4] 验证数据一致性...")
    is_valid = migrator.verify_migration(documents + new_documents)
    
    if is_valid:
        print("✓ 验证通过，可以切换到新存储")
    else:
        print("✗ 验证失败，需要检查问题")
    
    # 5. 切换阶段
    print("\n[阶段5] 切换到新存储...")
    print("  1. 停止写入源存储")
    print("  2. 将读流量切换到目标存储")
    print("  3. 监控新存储性能")
    print("✓ 切换完成")
    
    # 6. 清理
    source.delete_collection()


def migrate_with_data_transformation():
    """带数据转换的迁移"""
    print("\n" + "=" * 60)
    print("场景4：带数据转换的迁移")
    print("=" * 60)
    
    from langchain_core.vectorstores import InMemoryVectorStore
    from langchain_chroma import Chroma
    
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    
    # 1. 创建源向量存储
    source = InMemoryVectorStore(embeddings)
    
    old_documents = [
        Document(
            page_content=f"旧格式文档 {i}",
            metadata={"old_field": i}
        )
        for i in range(20)
    ]
    source.add_documents(old_documents)
    
    # 2. 数据转换函数
    def transform_document(doc: Document) -> Document:
        """转换文档格式"""
        return Document(
            page_content=doc.page_content,
            metadata={
                "new_field": doc.metadata.get("old_field"),
                "migrated_at": "2024-01-01",
                "version": "v2"
            }
        )
    
    # 3. 转换并迁移
    print("\n[迁移] 转换并迁移文档...")
    transformed_docs = [transform_document(doc) for doc in old_documents]
    
    target = Chroma(
        collection_name="transformed_collection",
        embedding_function=embeddings
    )
    target.add_documents(transformed_docs)
    
    print(f"✓ 迁移了 {len(transformed_docs)} 个文档")
    print("  示例转换:")
    print(f"    原始: {old_documents[0].metadata}")
    print(f"    转换后: {transformed_docs[0].metadata}")
    
    # 4. 清理
    target.delete_collection()
```

## 迁移策略总结

### 1. 迁移前准备

**评估阶段**：
- 评估数据规模
- 评估迁移时间
- 评估停机时间
- 评估风险

**准备工作**：
- 备份源数据
- 测试目标存储
- 准备回滚方案
- 通知相关人员

### 2. 迁移方案选择

**方案1：停机迁移**
- 适合：小规模数据（< 10K）
- 优点：简单直接
- 缺点：需要停机

**方案2：双写迁移**
- 适合：中等规模数据（10K - 100K）
- 优点：零停机
- 缺点：需要双写逻辑

**方案3：分批迁移**
- 适合：大规模数据（> 100K）
- 优点：可控风险
- 缺点：时间较长

### 3. 迁移步骤

**步骤1：准备阶段**

**步骤2：迁移阶段**

## 常见问题

### Q1: 如何最小化停机时间？

使用双写策略：
1. 新数据同时写入源和目标
2. 后台迁移历史数据
3. 验证后切换读流量
4. 停止写入源存储

### Q2: 如何验证迁移结果？

1. 抽样验证：随机抽取文档验证
2. 全量验证：对比所有文档（小规模）
3. 功能验证：测试所有功能
4. 性能验证：对比性能指标

### Q3: 迁移失败如何回滚？

1. 保留源向量存储
2. 切换流量回源存储
3. 分析失败原因
4. 修复后重新迁移

### Q4: 如何处理大规模数据迁移？

1. 分批迁移（每批 1000-10000）
2. 并行迁移（多线程/多进程）
3. 增量迁移（先迁移旧数据）
4. 监控资源使用

### Q5: 不同后端的数据格式不兼容怎么办？

1. 编写数据转换函数
2. 在迁移过程中转换
3. 验证转换结果
4. 记录转换规则
