## 第二层：TypeScript 深度集成

### 1. 作为库使用

**基础集成**：
```typescript
// src/openclaw-client.ts
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export class OpenClawClient {
  private gatewayUrl: string;

  constructor(gatewayUrl = 'http://localhost:18789') {
    this.gatewayUrl = gatewayUrl;
  }

  async sendMessage(to: string, message: string): Promise<void> {
    const { stdout } = await execAsync(
      `openclaw message send --to "${to}" --message "${message}" --json`
    );
    return JSON.parse(stdout);
  }

  async agentQuery(message: string, thinking = 'medium'): Promise<string> {
    const { stdout } = await execAsync(
      `openclaw agent --message "${message}" --thinking ${thinking} --json`
    );
    const result = JSON.parse(stdout);
    return result.response;
  }

  async getChannelStatus(): Promise<any[]> {
    const { stdout } = await execAsync('openclaw channels status --json');
    return JSON.parse(stdout).channels;
  }
}
```

**使用示例**：
```typescript
// src/app.ts
import { OpenClawClient } from './openclaw-client';

const client = new OpenClawClient();

async function main() {
  // 发送消息
  await client.sendMessage('@all', '系统上线通知');

  // Agent 查询
  const response = await client.agentQuery('总结今天的任务');
  console.log(response);

  // 检查通道状态
  const channels = await client.getChannelStatus();
  console.log('在线通道:', channels.filter(c => c.status === 'online'));
}

main();
```

### 2. 事件驱动架构

**监听 Gateway 事件**：
```typescript
// src/gateway-monitor.ts
import { EventEmitter } from 'events';
import { exec } from 'child_process';

export class GatewayMonitor extends EventEmitter {
  private interval: NodeJS.Timeout;

  start(intervalMs = 5000) {
    this.interval = setInterval(async () => {
      try {
        const { stdout } = await execAsync('openclaw gateway health --json');
        const health = JSON.parse(stdout);

        if (health.status === 'unhealthy') {
          this.emit('unhealthy', health);
        } else {
          this.emit('healthy', health);
        }
      } catch (error) {
        this.emit('error', error);
      }
    }, intervalMs);
  }

  stop() {
    clearInterval(this.interval);
  }
}

// 使用
const monitor = new GatewayMonitor();

monitor.on('unhealthy', (health) => {
  console.error('Gateway 不健康:', health);
  // 自动重启
  exec('openclaw gateway restart');
});

monitor.on('healthy', (health) => {
  console.log('Gateway 健康:', health);
});

monitor.start();
```

### 3. 错误处理与重试

**健壮的客户端**：
```typescript
// src/robust-client.ts
export class RobustOpenClawClient {
  private maxRetries = 3;
  private retryDelay = 1000;

  async sendMessageWithRetry(
    to: string,
    message: string
  ): Promise<void> {
    for (let i = 0; i < this.maxRetries; i++) {
      try {
        await this.sendMessage(to, message);
        return;
      } catch (error) {
        if (i === this.maxRetries - 1) throw error;

        console.log(`重试 ${i + 1}/${this.maxRetries}...`);
        await new Promise(resolve =>
          setTimeout(resolve, this.retryDelay * (i + 1))
        );
      }
    }
  }

  private async sendMessage(to: string, message: string): Promise<void> {
    const { stdout, stderr } = await execAsync(
      `openclaw message send --to "${to}" --message "${message}" --json`
    );

    if (stderr) {
      throw new Error(`发送失败: ${stderr}`);
    }

    const result = JSON.parse(stdout);
    if (result.status !== 'success') {
      throw new Error(`发送失败: ${result.error}`);
    }
  }
}
```

  async sendBatchMessages(
    recipients: string[],
    message: string

### 2. 缓存策略

**配置缓存**：
```typescript
// src/config-cache.ts
export class ConfigCache {
  private cache = new Map<string, { value: any; expiry: number }>();
  private ttl = 60000; // 1 分钟

  async get(key: string): Promise<any> {
    const cached = this.cache.get(key);

    if (cached && cached.expiry > Date.now()) {
      return cached.value;
    }

    // 从 CLI 读取
    const { stdout } = await execAsync(`openclaw config get ${key}`);
    const value = stdout.trim();

    this.cache.set(key, {
      value,
      expiry: Date.now() + this.ttl,
    });

    return value;
  }

  invalidate(key: string) {
    this.cache.delete(key);
  }
}
```

### 3. 连接池管理

**Gateway 连接池**：
```typescript
// src/gateway-pool.ts
export class GatewayPool {
  private connections: Map<string, GatewayConnection> = new Map();

  async getConnection(profile = 'default'): Promise<GatewayConnection> {
    if (!this.connections.has(profile)) {
      const conn = new GatewayConnection(profile);
      await conn.connect();
      this.connections.set(profile, conn);
    }

    return this.connections.get(profile)!;
  }

  async closeAll() {
    for (const conn of this.connections.values()) {
      await conn.close();
    }
    this.connections.clear();
  }
}

class GatewayConnection {
  constructor(private profile: string) {}

  async connect() {
    // 启动 Gateway
    exec(`openclaw --profile ${this.profile} gateway &`);
    await new Promise(resolve => setTimeout(resolve, 3000));
  }

  async close() {
    // 停止 Gateway
    await execAsync(`openclaw --profile ${this.profile} gateway stop`);
  }
}
```

// src/monitoring.ts
export class ProductionMonitor {
  private metrics = {
    gatewayUptime: 0,
    messagesSent: 0,
    messagesFailedCount: 0,
    channelsOnline: 0,
  };

  async collectMetrics() {
    // Gateway 状态
    const gatewayStatus = await this.getGatewayStatus();
    this.metrics.gatewayUptime = gatewayStatus.uptime;

    // 通道状态
    const channels = await this.getChannelStatus();
    this.metrics.channelsOnline = channels.filter(
      c => c.status === 'online'
    ).length;

    return this.metrics;
  }

  async sendToMonitoring(service: 'datadog' | 'prometheus') {
    const metrics = await this.collectMetrics();

    if (service === 'prometheus') {
      // 发送到 Prometheus
      await this.sendToPrometheus(metrics);
    } else {
      // 发送到 Datadog
      await this.sendToDatadog(metrics);
    }
  }
}

### 2. 与队列系统集成

**RabbitMQ 集成**：
```typescript
// src/queue-integration.ts
import amqp from 'amqplib';
import { OpenClawClient } from './openclaw-client';

export class QueueProcessor {
  private client = new OpenClawClient();

  async start() {
    const connection = await amqp.connect('amqp://localhost');
    const channel = await connection.createChannel();

    await channel.assertQueue('openclaw-messages');

    channel.consume('openclaw-messages', async (msg) => {
      if (msg) {
        const { to, message } = JSON.parse(msg.content.toString());

        try {
          await this.client.sendMessage(to, message);
          channel.ack(msg);
        } catch (error) {
          console.error('处理失败:', error);
          channel.nack(msg);
        }
      }
    });
  }
}
```
