## 最小实现

### 1. 最简单的 Loader

```python
from langchain_core.document_loaders import BaseLoader
from langchain_core.documents import Document
from typing import Iterator

class MinimalLoader(BaseLoader):
    """最小可用的 DocumentLoader"""

    def __init__(self, file_path: str):
        self.file_path = file_path

    def lazy_load(self) -> Iterator[Document]:
        """只需要实现这一个方法"""
        with open(self.file_path, "r", encoding="utf-8") as f:
            content = f.read()
            yield Document(
                page_content=content,
                metadata={"source": self.file_path}
            )

# 使用
loader = MinimalLoader("example.txt")
docs = loader.load()
print(f"加载了 {len(docs)} 个文档")
print(f"内容: {docs[0].page_content[:100]}...")
```

**核心要点**:
- 只需继承 `BaseLoader`
- 只需实现 `lazy_load()` 方法
- 返回 `Document` 对象
- 自动获得 `load()`, `aload()`, `alazy_load()` 方法

## 完整示例

### 场景: 加载文本文件并分割

```python
from langchain_core.document_loaders import BaseLoader
from langchain_core.documents import Document
from langchain.text_splitter import RecursiveCharacterTextSplitter
from typing import Iterator

class SimpleTextLoader(BaseLoader):
    """简单的文本加载器"""

    def __init__(self, file_path: str):
        self.file_path = file_path

    def lazy_load(self) -> Iterator[Document]:
        """懒加载文本文件"""
        with open(self.file_path, "r", encoding="utf-8") as f:
            content = f.read()
            yield Document(
                page_content=content,
                metadata={
                    "source": self.file_path,
                    "file_type": "text"
                }
            )

# 使用示例
if __name__ == "__main__":
    # 1. 加载文档
    loader = SimpleTextLoader("example.txt")
    docs = loader.load()
    print(f"✓ 加载了 {len(docs)} 个文档")

    # 2. 分割文档
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=500,
        chunk_overlap=50
    )
    chunks = splitter.split_documents(docs)
    print(f"✓ 分割为 {len(chunks)} 个块")

    # 3. 查看结果
    for i, chunk in enumerate(chunks[:3]):
        print(f"\n块 {i+1}:")
        print(f"  内容: {chunk.page_content[:100]}...")
        print(f"  元数据: {chunk.metadata}")
```

**输出示例**:
```
✓ 加载了 1 个文档
✓ 分割为 5 个块

块 1:
  内容: This is the first paragraph of the document...
  元数据: {'source': 'example.txt', 'file_type': 'text'}

块 2:
  内容: This is the second paragraph...
  元数据: {'source': 'example.txt', 'file_type': 'text'}
```

## 进阶示例

### 场景: 加载目录中的所有文件

```python
from pathlib import Path
from langchain_core.document_loaders import BaseLoader
from langchain_core.documents import Document
from typing import Iterator

class DirectoryTextLoader(BaseLoader):
    """加载目录中的所有文本文件"""

    def __init__(self, directory: str):
        self.directory = Path(directory)

    def lazy_load(self) -> Iterator[Document]:
        """懒加载目录中的所有 .txt 文件"""
        for file_path in self.directory.glob("*.txt"):
            with open(file_path, "r", encoding="utf-8") as f:
                content = f.read()
                yield Document(
                    page_content=content,
                    metadata={
                        "source": str(file_path),
                        "filename": file_path.name
                    }
                )

# 使用示例
if __name__ == "__main__":
    # 加载目录
    loader = DirectoryTextLoader("docs/")

    # 方式 1: 一次性加载
    docs = loader.load()
    print(f"✓ 加载了 {len(docs)} 个文档")

    # 方式 2: 流式加载
    for i, doc in enumerate(loader.lazy_load()):
        print(f"✓ 加载文件 {i+1}: {doc.metadata['filename']}")
```

## RAG 最小示例

### 完整的 RAG 管道

```python
from langchain_core.document_loaders import BaseLoader
from langchain_core.documents import Document
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_community.vectorstores import Chroma
from langchain.chains import RetrievalQA
from typing import Iterator
import os

# 设置 API Key
os.environ["OPENAI_API_KEY"] = "your-api-key"

class SimpleLoader(BaseLoader):
    """简单的文档加载器"""

    def __init__(self, file_path: str):
        self.file_path = file_path

    def lazy_load(self) -> Iterator[Document]:
        with open(self.file_path, "r", encoding="utf-8") as f:
            content = f.read()
            yield Document(
                page_content=content,
                metadata={"source": self.file_path}
            )

# RAG 管道
if __name__ == "__main__":
    # 1. 加载文档
    loader = SimpleLoader("knowledge.txt")
    docs = loader.load()
    print(f"✓ 加载了 {len(docs)} 个文档")

    # 2. 分割文档
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=500,
        chunk_overlap=50
    )
    chunks = splitter.split_documents(docs)
    print(f"✓ 分割为 {len(chunks)} 个块")

    # 3. 向量化
    embeddings = OpenAIEmbeddings()
    vector_store = Chroma.from_documents(chunks, embeddings)
    print(f"✓ 向量化完成")

    # 4. 构建 QA 链
    llm = ChatOpenAI(model="gpt-3.5-turbo")
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        retriever=vector_store.as_retriever()
    )
    print(f"✓ QA 链构建完成")

    # 5. 查询
    query = "What is the main topic?"
    result = qa_chain.invoke({"query": query})
    print(f"\n问题: {query}")
    print(f"回答: {result['result']}")
```

## 错误处理版本

### 带错误处理的 Loader

```python
from langchain_core.document_loaders import BaseLoader
from langchain_core.documents import Document
from typing import Iterator
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class RobustLoader(BaseLoader):
    """带错误处理的加载器"""

    def __init__(self, file_path: str, encoding: str = "utf-8"):
        self.file_path = file_path
        self.encoding = encoding

    def lazy_load(self) -> Iterator[Document]:
        """带错误处理的懒加载"""
        try:
            # 检查文件是否存在
            if not os.path.exists(self.file_path):
                logger.error(f"文件不存在: {self.file_path}")
                return

            # 读取文件
            with open(self.file_path, "r", encoding=self.encoding) as f:
                content = f.read()

            # 检查内容是否为空
            if not content.strip():
                logger.warning(f"文件为空: {self.file_path}")
                return

            # 返回文档
            yield Document(
                page_content=content,
                metadata={
                    "source": self.file_path,
                    "encoding": self.encoding
                }
            )

            logger.info(f"✓ 成功加载: {self.file_path}")

        except UnicodeDecodeError as e:
            logger.error(f"编码错误: {self.file_path}, {e}")
        except Exception as e:
            logger.error(f"加载失败: {self.file_path}, {e}")

# 使用示例
if __name__ == "__main__":
    # 测试正常文件
    loader1 = RobustLoader("example.txt")
    docs1 = loader1.load()

    # 测试不存在的文件
    loader2 = RobustLoader("nonexistent.txt")
    docs2 = loader2.load()

    # 测试编码错误
    loader3 = RobustLoader("gbk_file.txt", encoding="utf-8")
    docs3 = loader3.load()
```

## 性能优化版本

### 带缓存的 Loader

```python
from langchain_core.document_loaders import BaseLoader
from langchain_core.documents import Document
from typing import Iterator
import hashlib
import pickle
from pathlib import Path

class CachedLoader(BaseLoader):
    """带缓存的加载器"""

    def __init__(self, file_path: str, cache_dir: str = ".cache"):
        self.file_path = file_path
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(exist_ok=True)

    def _get_cache_path(self) -> Path:
        """生成缓存文件路径"""
        file_hash = hashlib.md5(self.file_path.encode()).hexdigest()
        return self.cache_dir / f"{file_hash}.pkl"

    def _is_cache_valid(self) -> bool:
        """检查缓存是否有效"""
        cache_path = self._get_cache_path()
        if not cache_path.exists():
            return False

        # 检查文件是否被修改
        file_mtime = Path(self.file_path).stat().st_mtime
        cache_mtime = cache_path.stat().st_mtime
        return cache_mtime > file_mtime

    def lazy_load(self) -> Iterator[Document]:
        """带缓存的懒加载"""
        cache_path = self._get_cache_path()

        # 尝试从缓存加载
        if self._is_cache_valid():
            print(f"✓ 从缓存加载: {self.file_path}")
            with open(cache_path, "rb") as f:
                docs = pickle.load(f)
                for doc in docs:
                    yield doc
            return

        # 从文件加载
        print(f"✓ 从文件加载: {self.file_path}")
        with open(self.file_path, "r", encoding="utf-8") as f:
            content = f.read()
            doc = Document(
                page_content=content,
                metadata={"source": self.file_path}
            )

        # 保存到缓存
        with open(cache_path, "wb") as f:
            pickle.dump([doc], f)

        yield doc

# 使用示例
if __name__ == "__main__":
    import time

    # 第一次加载 - 从文件
    start = time.time()
    loader = CachedLoader("large_file.txt")
    docs = loader.load()
    print(f"第一次加载耗时: {time.time() - start:.2f}秒")

    # 第二次加载 - 从缓存
    start = time.time()
    loader = CachedLoader("large_file.txt")
    docs = loader.load()
    print(f"第二次加载耗时: {time.time() - start:.2f}秒")
```

```python
import unittest
from pathlib import Path
from langchain_core.documents import Document

class TestMinimalLoader(unittest.TestCase):
    """测试最小加载器"""

    def setUp(self):
        """创建测试文件"""
        self.test_file = Path("test.txt")
        self.test_file.write_text("Hello, world!", encoding="utf-8")

    def tearDown(self):
        """删除测试文件"""
        if self.test_file.exists():
            self.test_file.unlink()

    def test_load(self):
        """测试加载功能"""
        loader = MinimalLoader(str(self.test_file))
        docs = loader.load()

        self.assertEqual(len(docs), 1)
        self.assertEqual(docs[0].page_content, "Hello, world!")
        self.assertEqual(docs[0].metadata["source"], str(self.test_file))

        """测试懒加载功能"""
        loader = MinimalLoader(str(self.test_file))
        docs = list(loader.lazy_load())

        self.assertEqual(len(docs), 1)
        self.assertIsInstance(docs[0], Document)

if __name__ == "__main__":
    unittest.main()
```

## 常见模式

### 模式 1: 单文件加载

```python
class SingleFileLoader(BaseLoader):

        with open(self.file_path) as f:
```

### 模式 2: 多文件加载

```python
class MultiFileLoader(BaseLoader):
    def lazy_load(self):
        for file_path in self.file_paths:
            with open(file_path) as f:
                yield Document(page_content=f.read())
```

### 模式 3: 分页加载

```python
class PagedLoader(BaseLoader):
    def lazy_load(self):
        for page_num, page_content in enumerate(self.pages):
            yield Document(
                page_content=page_content,
                metadata={"page": page_num}
            )
```
