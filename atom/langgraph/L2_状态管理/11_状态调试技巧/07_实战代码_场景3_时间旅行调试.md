# 实战代码 - 场景3：时间旅行调试

## 场景说明

本场景演示 LangGraph 最强大的调试能力——时间旅行。你将学会：

1. 通过 `checkpoint_id` 回溯到任意历史状态
2. 检查历史状态的完整内容
3. 修改历史状态后创建分叉（Fork）
4. 从分叉点继续执行，得到不同的结果
5. 对比原始路径和分叉路径的差异

核心思路：发现问题 → 回到出错前 → 修正状态 → 重新执行。就像游戏里的"读档"——存档点就是 checkpoint，读档后你可以做出不同的选择。

[来源: reference/context7_langgraph_01.md | LangGraph 官方文档]
[来源: reference/source_状态调试_01.md | 源码分析]

## 完整代码

```python
"""
LangGraph 状态调试 - 时间旅行调试
演示：历史回溯、状态分叉、从历史点重新执行

核心知识点：
- checkpoint_id 是时间旅行的"坐标"
- get_state_history() 遍历历史找到目标状态
- update_state() 修改历史状态创建分叉
- invoke(None, config) 从指定状态继续执行

运行环境：Python 3.13+, langgraph
安装依赖：uv add langgraph
"""

from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.memory import InMemorySaver
import operator


# ============================================================
# 1. 定义状态 —— 模拟一个简单的 RAG 处理管道
# ============================================================

class State(TypedDict):
    query: str                                    # 用户查询
    steps: Annotated[list[str], operator.add]     # 执行步骤日志（追加模式）
    answer: str                                   # 当前回答


# ============================================================
# 2. 定义节点函数
# ============================================================

def step_a(state: State) -> dict:
    """步骤A：查询预处理"""
    return {
        "steps": ["step_a: 预处理完成"],
        "answer": f"A处理了: {state['query']}",
    }


def step_b(state: State) -> dict:
    """步骤B：检索增强"""
    return {
        "steps": ["step_b: 检索增强完成"],
        "answer": f"B加工了: {state['answer']}",
    }


def step_c(state: State) -> dict:
    """步骤C：生成回答（模拟一个"错误"的处理）"""
    # 故意搞砸，用于演示时间旅行调试
    return {
        "steps": ["step_c: 生成出错!"],
        "answer": f"C搞砸了: {state['answer']}",
    }


# ============================================================
# 3. 构建图 —— 线性管道 A → B → C
# ============================================================

builder = StateGraph(State)
builder.add_node("step_a", step_a)
builder.add_node("step_b", step_b)
builder.add_node("step_c", step_c)

builder.add_edge(START, "step_a")
builder.add_edge("step_a", "step_b")
builder.add_edge("step_b", "step_c")
builder.add_edge("step_c", END)

checkpointer = InMemorySaver()
graph = builder.compile(checkpointer=checkpointer)

config = {"configurable": {"thread_id": "time-travel-1"}}


# ============================================================
# 4. 第一次执行 —— 发现 step_c 搞砸了
# ============================================================

print("=" * 65)
print("  阶段 1：首次执行，发现问题")
print("=" * 65)

result = graph.invoke(
    {"query": "RAG原理", "steps": []},
    config,
)

print(f"\n执行步骤:")
for s in result["steps"]:
    print(f"  {s}")
print(f"\n最终结果: {result['answer']}")
print(f"\n诊断: step_c 搞砸了！需要回到 step_b 之后修正")


# ============================================================
# 5. 遍历历史，找到 step_b 完成后的状态
# ============================================================

print()
print("=" * 65)
print("  阶段 2：遍历历史，定位目标状态")
print("=" * 65)

print(f"\n完整执行历史:")
print(f"{'序号':<4} {'source':<8} {'step':<6} {'next':<25} {'说明'}")
print("-" * 70)

target_config = None

for i, snapshot in enumerate(graph.get_state_history(config)):
    step = snapshot.metadata.get("step", -1)
    source = snapshot.metadata.get("source", "")
    next_nodes = snapshot.next
    writes = snapshot.metadata.get("writes", {})
    node_name = list(writes.keys())[0] if writes else "initial"

    # 标记目标状态
    marker = ""
    if snapshot.next == ("step_c",):
        target_config = snapshot.config
        marker = " <<< 目标状态"

    print(f"{i:<4} {source:<8} {step:<6} {str(next_nodes):<25} {node_name}{marker}")

if target_config:
    cp_id = target_config["configurable"]["checkpoint_id"]
    print(f"\n找到目标状态！")
    print(f"  checkpoint_id: {cp_id}")
    print(f"  含义: step_b 已完成，step_c 尚未执行")


# ============================================================
# 6. 检查历史状态的内容
# ============================================================

print()
print("=" * 65)
print("  阶段 3：检查历史状态")
print("=" * 65)

if target_config:
    past_snapshot = graph.get_state(target_config)
    print(f"\n回到过去的状态:")
    print(f"  query:  {past_snapshot.values['query']}")
    print(f"  answer: {past_snapshot.values['answer']}")
    print(f"  steps:  {past_snapshot.values['steps']}")
    print(f"  下一步将执行: {past_snapshot.next}")
    print(f"\n此时 step_c 还没执行，answer 还是正确的！")


# ============================================================
# 7. 修改历史状态，创建分叉（Fork）
# ============================================================

print()
print("=" * 65)
print("  阶段 4：修改历史状态，创建分叉")
print("=" * 65)

if target_config:
    # 用 update_state 修改历史状态
    # 模拟：我们手动修正 answer，假装 step_b 给出了更好的结果
    fork_config = graph.update_state(
        target_config,
        values={
            "answer": "B正确加工了: RAG原理（含检索增强上下文）",
            "steps": ["[Fork] 手动修正了 step_b 的输出"],
        },
        as_node="step_b",  # 以 step_b 的身份写入
    )

    fork_cp_id = fork_config["configurable"]["checkpoint_id"]
    print(f"\n分叉创建成功！")
    print(f"  新 checkpoint_id: {fork_cp_id}")

    # 验证分叉后的状态
    fork_snapshot = graph.get_state(fork_config)
    print(f"  修改后的 answer: {fork_snapshot.values['answer']}")
    print(f"  下一步将执行: {fork_snapshot.next}")
    print(f"  metadata.source: {fork_snapshot.metadata.get('source')}")


# ============================================================
# 8. 从分叉点继续执行
# ============================================================

print()
print("=" * 65)
print("  阶段 5：从分叉点继续执行")
print("=" * 65)

if target_config:
    # invoke(None) 表示不提供新输入，从 checkpoint 恢复执行
    fork_result = graph.invoke(None, config=fork_config)

    print(f"\n分叉执行的步骤:")
    for s in fork_result["steps"]:
        print(f"  {s}")
    print(f"\n分叉后的结果: {fork_result['answer']}")


# ============================================================
# 9. 对比原始路径 vs 分叉路径
# ============================================================

print()
print("=" * 65)
print("  阶段 6：对比原始路径 vs 分叉路径")
print("=" * 65)

print(f"\n{'指标':<12} {'原始路径':<40} {'分叉路径'}")
print("-" * 90)
print(f"{'最终回答':<12} {result['answer']:<40} {fork_result['answer']}")
print(f"{'步骤数':<12} {len(result['steps']):<40} {len(fork_result['steps'])}")
print(f"{'是否出错':<12} {'是（C搞砸了）':<40} {'否（C正常处理）'}")

print(f"\n原始路径步骤:")
for s in result["steps"]:
    print(f"  {s}")
print(f"\n分叉路径步骤:")
for s in fork_result["steps"]:
    print(f"  {s}")


# ============================================================
# 10. 查看分叉后的完整历史树
# ============================================================

print()
print("=" * 65)
print("  阶段 7：查看分叉后的完整历史")
print("=" * 65)

full_history = list(graph.get_state_history(config))

print(f"\n分叉后的完整历史（{len(full_history)} 个快照）:")
print(f"{'序号':<4} {'source':<8} {'step':<6} {'next':<25} {'说明'}")
print("-" * 70)

for i, snap in enumerate(full_history):
    source = snap.metadata.get("source", "?")
    step = snap.metadata.get("step", "?")
    next_nodes = snap.next
    writes = snap.metadata.get("writes", {})
    node_name = list(writes.keys())[0] if writes else "initial"

    marker = ""
    if source == "update":
        marker = " ← Fork 分叉点"
    elif i == 0:
        marker = " ← 最新（分叉路径终点）"

    print(f"{i:<4} {source:<8} {step:<6} {str(next_nodes):<25} {node_name}{marker}")


# ============================================================
# 11. 高级：从同一历史点多次分叉
# ============================================================

print()
print("=" * 65)
print("  阶段 8：高级 —— 从同一历史点多次分叉")
print("=" * 65)

if target_config:
    # 分叉 B：模拟另一种修正方案
    fork_b_config = graph.update_state(
        target_config,
        values={
            "answer": "B方案二: RAG原理（精简版上下文）",
            "steps": ["[Fork-B] 使用精简上下文"],
        },
        as_node="step_b",
    )
    fork_b_result = graph.invoke(None, config=fork_b_config)

    # 分叉 C：模拟第三种修正方案
    fork_c_config = graph.update_state(
        target_config,
        values={
            "answer": "B方案三: RAG原理（多文档融合上下文）",
            "steps": ["[Fork-C] 使用多文档融合"],
        },
        as_node="step_b",
    )
    fork_c_result = graph.invoke(None, config=fork_c_config)

    print(f"\n多分叉对比:")
    print(f"{'版本':<10} {'最终回答'}")
    print("-" * 60)
    print(f"{'原始':<10} {result['answer']}")
    print(f"{'Fork-A':<10} {fork_result['answer']}")
    print(f"{'Fork-B':<10} {fork_b_result['answer']}")
    print(f"{'Fork-C':<10} {fork_c_result['answer']}")

    print(f"\n所有分叉都基于同一个历史状态（step_b 完成后），")
    print(f"只是修改了不同的 answer，然后各自独立执行 step_c。")
    print(f"这就是时间旅行的核心价值——快速对比不同方案。")


# ============================================================
# 12. 调试工作流总结
# ============================================================

print()
print("=" * 65)
print("  时间旅行调试工作流总结")
print("=" * 65)

print("""
完整流程：

  1. 执行图，发现结果不对
     └─ result = graph.invoke(input, config)

  2. 遍历历史，定位问题步骤
     └─ for snap in graph.get_state_history(config):
     └─     if snap.next == ("问题节点",): target = snap

  3. 检查历史状态
     └─ past = graph.get_state(target.config)
     └─ print(past.values)  # 看看出错前的状态

  4. 修改状态，创建分叉
     └─ fork = graph.update_state(
     └─     target.config,
     └─     values={"key": "修正值"},
     └─     as_node="节点名",
     └─ )

  5. 从分叉点继续执行
     └─ new_result = graph.invoke(None, config=fork)

  6. 对比原始 vs 分叉结果
     └─ 确认修正是否有效

关键 API:
  - get_state_history(config)  → 遍历历史
  - get_state(config)          → 检查特定状态
  - update_state(config, ...)  → 修改状态（创建分叉）
  - invoke(None, config)       → 从 checkpoint 恢复执行
""")
```

## 关键知识点解析

### 1. 时间旅行的三个核心 API

```python
# API 1: 遍历历史 → 找到目标快照
for snapshot in graph.get_state_history(config):
    if snapshot.next == ("目标节点",):
        target = snapshot
        break

# API 2: 修改状态 → 创建分叉
fork_config = graph.update_state(
    target.config,                    # 基于哪个历史状态
    values={"key": "new_value"},      # 要修改的字段
    as_node="node_name",              # 以哪个节点的身份写入
)

# API 3: 恢复执行 → 从分叉点继续
result = graph.invoke(None, config=fork_config)
```

### 2. invoke(None) 的含义

| 调用方式 | 含义 |
|----------|------|
| `graph.invoke({"query": "..."}, config)` | 提供新输入，从头执行 |
| `graph.invoke(None, config_with_checkpoint_id)` | 不提供输入，从 checkpoint 恢复执行 |

第一个参数为 `None` 时，LangGraph 会从 config 中的 `checkpoint_id` 加载历史状态，然后从 `next` 指定的节点继续执行。

### 3. as_node 参数的作用

`update_state()` 的 `as_node` 参数告诉 LangGraph "这次修改是以哪个节点的身份写入的"。这决定了分叉后从哪个节点继续执行：

```python
# 以 step_b 身份写入 → 下一步执行 step_c
graph.update_state(config, values={...}, as_node="step_b")

# 以 step_a 身份写入 → 下一步执行 step_b
graph.update_state(config, values={...}, as_node="step_a")
```

### 4. 定位目标状态的技巧

```python
# 方法 A：通过 next 字段定位（推荐）
# "下一步要执行 step_c" = "step_b 刚完成"
for snap in graph.get_state_history(config):
    if snap.next == ("step_c",):
        target = snap
        break

# 方法 B：通过 writes 字段定位
# "step_b 写入了数据" = "step_b 刚完成"
for snap in graph.get_state_history(config):
    if "step_b" in snap.metadata.get("writes", {}):
        target = snap
        break

# 方法 C：通过 step 编号定位
for snap in graph.get_state_history(config):
    if snap.metadata.get("step") == 3:
        target = snap
        break
```

## 在 RAG 系统中的应用

时间旅行调试在 RAG 系统中的典型场景：

| 场景 | 操作 | 说明 |
|------|------|------|
| 检索质量调试 | 回到检索步骤，替换文档后重新生成 | 发现回答不好时，手动提供高质量文档看效果 |
| Prompt 调优 | 回到生成步骤前，修改 prompt 参数 | 对比不同 prompt 模板的生成效果 |
| 错误复现 | 通过 checkpoint_id 精确回到出错状态 | 复现并分析间歇性错误 |
| A/B 测试 | 从同一检索结果多次分叉 | 对比不同 temperature、top_k 的效果 |
| 人工审核 | 审核员查看中间结果，不满意时修正 | 人机协作的质量控制 |
