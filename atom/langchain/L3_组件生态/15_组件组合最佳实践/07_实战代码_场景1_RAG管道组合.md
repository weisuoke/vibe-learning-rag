# 实战代码 场景1：RAG 管道组合

## 场景描述
构建一个完整的 RAG（检索增强生成）管道，展示如何选择和组合 LangChain 组件。

## 完整可运行代码

```python
"""
RAG 管道组合实战
演示：完整的文档问答系统，包含文档加载、分块、向量化、检索、生成
"""

import os
from dotenv import load_dotenv
load_dotenv()

from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough, RunnableParallel
from langchain_core.documents import Document
from langchain_chroma import Chroma
from langchain_text_splitters import RecursiveCharacterTextSplitter

# ===== 1. 准备文档数据 =====
print("=== 1. 准备文档数据 ===")

# 模拟文档（实际项目中从文件加载）
documents = [
    Document(
        page_content="RAG（检索增强生成）是一种将信息检索与文本生成相结合的技术。"
        "它通过从外部知识库中检索相关文档，将其作为上下文提供给大语言模型，"
        "从而生成更准确、更有依据的回答。RAG 的核心优势在于减少幻觉、"
        "提供最新信息、支持领域特定知识。",
        metadata={"source": "rag_intro.md", "chapter": "概述"},
    ),
    Document(
        page_content="RAG 系统的核心流程包括：1) 文档加载与解析，将各种格式的文档"
        "转换为文本；2) 文本分块，将长文档切分为适合检索的片段；"
        "3) 向量化，使用 Embedding 模型将文本转换为向量；"
        "4) 向量存储，将向量存入向量数据库；5) 检索，根据用户查询"
        "找到最相关的文档片段；6) 生成，将检索到的上下文注入 prompt，"
        "由 LLM 生成最终回答。",
        metadata={"source": "rag_flow.md", "chapter": "流程"},
    ),
    Document(
        page_content="Embedding 模型是 RAG 系统的关键组件。常用的 Embedding 模型"
        "包括 OpenAI 的 text-embedding-3-small（性价比高）和 "
        "text-embedding-3-large（精度更高）。选择 Embedding 模型时"
        "需要考虑：维度大小、检索精度、推理速度和成本。"
        "对于大多数应用，text-embedding-3-small 已经足够。",
        metadata={"source": "embedding.md", "chapter": "Embedding"},
    ),
    Document(
        page_content="文本分块策略直接影响检索质量。RecursiveCharacterTextSplitter "
        "是最常用的分块器，它按照段落、句子、单词的优先级递归分割。"
        "推荐的 chunk_size 为 500-1000 字符，chunk_overlap 为 50-200 字符。"
        "过小的 chunk 会丢失上下文，过大的 chunk 会引入噪音。",
        metadata={"source": "chunking.md", "chapter": "分块"},
    ),
]

print(f"加载了 {len(documents)} 个文档")

# ===== 2. 文本分块 =====
print("\n=== 2. 文本分块 ===")

text_splitter = RecursiveCharacterTextSplitter(
    chunk_size=200,
    chunk_overlap=50,
    separators=["\n\n", "\n", "。", "；", " ", ""],
)

splits = text_splitter.split_documents(documents)
print(f"分块后得到 {len(splits)} 个片段")

# ===== 3. 向量化 + 存储 =====
print("\n=== 3. 向量化 + 存储 ===")

embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
vectorstore = Chroma.from_documents(
    documents=splits,
    embedding=embeddings,
    collection_name="rag_demo",
)
print("向量存储创建完成")

# ===== 4. 构建检索器 =====
print("\n=== 4. 构建检索器 ===")

retriever = vectorstore.as_retriever(
    search_type="similarity",
    search_kwargs={"k": 3},
)

# ===== 5. 构建 RAG 链 =====
print("\n=== 5. 构建 RAG 链 ===")

# 格式化检索到的文档
def format_docs(docs):
    return "\n\n".join(
        f"[来源: {doc.metadata.get('source', '未知')}]\n{doc.page_content}"
        for doc in docs
    )

# RAG prompt
rag_prompt = ChatPromptTemplate.from_messages([
    ("system",
     "你是一个专业的技术助手。根据以下参考资料回答用户的问题。\n"
     "如果参考资料中没有相关信息，请明确说明。\n\n"
     "参考资料：\n{context}"),
    ("human", "{question}"),
])

# 模型
model = ChatOpenAI(model="gpt-4o-mini", temperature=0)

# 组合 RAG 链
# 关键组合模式：RunnableParallel 同时传递 question 和 context
rag_chain = (
    RunnableParallel(
        question=RunnablePassthrough(),
        context=retriever | format_docs,
    )
    | rag_prompt
    | model
    | StrOutputParser()
)

print("RAG 链构建完成")
print("\n管道结构：")
rag_chain.get_graph().print_ascii()

# ===== 6. 测试 RAG 链 =====
print("\n=== 6. 测试 RAG 链 ===")

questions = [
    "什么是 RAG？",
    "RAG 系统的核心流程有哪些？",
    "如何选择 Embedding 模型？",
    "文本分块的推荐参数是什么？",
]

for question in questions:
    print(f"\n❓ 问题: {question}")
    answer = rag_chain.invoke(question)
    print(f"💡 回答: {answer[:200]}...")

# ===== 7. 流式输出 =====
print("\n\n=== 7. 流式输出 ===")
print("❓ 问题: RAG 的核心优势是什么？")
print("💡 回答: ", end="")
for chunk in rag_chain.stream("RAG 的核心优势是什么？"):
    print(chunk, end="", flush=True)
print()

# ===== 8. 批量处理 =====
print("\n=== 8. 批量处理 ===")
results = rag_chain.batch(
    ["什么是RAG？", "Embedding模型怎么选？"],
    config={"max_concurrency": 2},
)
for i, result in enumerate(results):
    print(f"结果{i+1}: {result[:100]}...")

# 清理
vectorstore.delete_collection()
print("\n✅ RAG 管道演示完成")
```

## 关键组合模式解析

### 模式1：RunnableParallel 传递 question + context
```python
RunnableParallel(
    question=RunnablePassthrough(),  # 透传原始输入
    context=retriever | format_docs,  # 检索 + 格式化
)
```
这是 RAG 链最核心的组合模式：同时传递用户问题和检索到的上下文。

### 模式2：Retriever | format_docs 串行
```python
retriever | format_docs
```
检索器返回 Document 列表，format_docs 将其格式化为字符串。

### 模式3：完整管道串行
```python
parallel_step | prompt | model | parser
```
标准的 LCEL 四步管道。

## 扩展：添加来源引用

```python
# 带来源引用的 RAG 链
from langchain_core.runnables import RunnableLambda

def add_sources(inputs):
    """在回答后附加来源信息"""
    docs = inputs["docs"]
    answer = inputs["answer"]
    sources = set(doc.metadata.get("source", "未知") for doc in docs)
    return f"{answer}\n\n📚 参考来源: {', '.join(sources)}"

rag_chain_with_sources = (
    RunnableParallel(
        question=RunnablePassthrough(),
        docs=retriever,
    )
    | RunnablePassthrough.assign(
        context=lambda x: format_docs(x["docs"]),
    )
    | RunnablePassthrough.assign(
        answer=rag_prompt | model | StrOutputParser(),
    )
    | RunnableLambda(add_sources)
)
```
