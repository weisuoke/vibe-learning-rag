## 4.1 使用LangChain预构建Loader

**核心概念：** LangChain提供100+预构建的文档加载器，覆盖常见格式

```python
"""
最小可用示例：使用LangChain加载常见文档
"""

from langchain_community.document_loaders import (
    PyPDFLoader,           # PDF文档
    TextLoader,            # 纯文本
    Docx2txtLoader,        # Word文档
    UnstructuredHTMLLoader # HTML网页
)

# ===== 1. 加载PDF =====
pdf_loader = PyPDFLoader("report.pdf")
pdf_docs = pdf_loader.load()

# ===== 2. 加载文本文件 =====
txt_loader = TextLoader("notes.txt", encoding="utf-8")
txt_docs = txt_loader.load()

# ===== 3. 加载Word文档 =====
docx_loader = Docx2txtLoader("proposal.docx")
docx_docs = docx_loader.load()

# ===== 4. 加载HTML =====
html_loader = UnstructuredHTMLLoader("article.html")
html_docs = html_loader.load()

# 所有Loader返回统一的Document列表
print(f"PDF文档数: {len(pdf_docs)}")
print(f"第一页内容: {pdf_docs[0].page_content[:100]}...")
print(f"元数据: {pdf_docs[0].metadata}")
```

**在RAG中的应用：**
- 快速构建文档加载管道
- 不需要手动处理格式细节
- 统一的接口便于后续处理

## 4.2 理解Document对象结构

**核心概念：** Document是LangChain的标准文档表示，包含文本和元数据

```python
from langchain.schema import Document

# Document对象结构
doc = Document(
    page_content="这是文档的文本内容...",  # 文本内容
    metadata={                              # 元数据
        "source": "report.pdf",             # 来源文件
        "page": 1,                          # 页码
        "title": "技术报告",                # 标题
        "author": "张三",                   # 作者
        "created_date": "2025-01-15"        # 创建日期
    }
)

# 访问内容
print(doc.page_content)  # 文本内容
print(doc.metadata)      # 元数据字典

# 在RAG中使用
from langchain_text_splitters import RecursiveCharacterTextSplitter

splitter = RecursiveCharacterTextSplitter(chunk_size=500)
chunks = splitter.split_documents([doc])  # 接收Document列表

# 每个chunk保留原始元数据
for chunk in chunks:
    print(f"来源: {chunk.metadata['source']}")
    print(f"内容: {chunk.page_content[:50]}...")
```

**在RAG中的应用：**
- 统一的数据结构贯穿整个RAG流程
- 元数据用于检索过滤和结果展示
- 分块后保留原始文档信息

## 4.3 批量加载目录下的文档

**核心概念：** 使用DirectoryLoader批量加载整个目录的文档

```python
from langchain_community.document_loaders import DirectoryLoader, TextLoader

# ===== 1. 加载目录下所有txt文件 =====
loader = DirectoryLoader(
    "docs/",                    # 目录路径
    glob="**/*.txt",            # 文件模式（递归查找所有txt）
    loader_cls=TextLoader,      # 使用的Loader类
    loader_kwargs={"encoding": "utf-8"}  # Loader参数
)

docs = loader.load()
print(f"加载了 {len(docs)} 个文档")

# ===== 2. 加载多种格式 =====
from langchain_community.document_loaders import (
    DirectoryLoader,
    PyPDFLoader,
    Docx2txtLoader
)

# 加载PDF
pdf_loader = DirectoryLoader("docs/", glob="**/*.pdf", loader_cls=PyPDFLoader)
pdf_docs = pdf_loader.load()

# 加载DOCX
docx_loader = DirectoryLoader("docs/", glob="**/*.docx", loader_cls=Docx2txtLoader)
docx_docs = docx_loader.load()

# 合并所有文档
all_docs = pdf_docs + docx_docs
print(f"总共加载 {len(all_docs)} 个文档")

# ===== 3. RAG应用：构建知识库 =====
from langchain.vectorstores import Chroma
from langchain.embeddings import OpenAIEmbeddings

# 加载所有文档
loader = DirectoryLoader("knowledge_base/", glob="**/*.pdf", loader_cls=PyPDFLoader)
documents = loader.load()

# 分块
from langchain_text_splitters import RecursiveCharacterTextSplitter
splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=200)
chunks = splitter.split_documents(documents)

# 向量化并存储
embeddings = OpenAIEmbeddings()
vectorstore = Chroma.from_documents(chunks, embeddings)

print(f"知识库构建完成，包含 {len(chunks)} 个文本块")
```

**在RAG中的应用：**
- 快速构建知识库
- 支持增量更新（新增文档）
- 自动处理目录结构

## 4.4 提取和保留元数据

**核心概念：** 元数据对RAG检索质量至关重要，需要正确提取和保留

```python
from langchain_community.document_loaders import PyPDFLoader
from datetime import datetime
import os

# ===== 1. 自动提取元数据 =====
loader = PyPDFLoader("report.pdf")
docs = loader.load()

# 查看自动提取的元数据
print(docs[0].metadata)
# {
#   'source': 'report.pdf',
#   'page': 0,
#   'total_pages': 10
# }

# ===== 2. 添加自定义元数据 =====
for doc in docs:
    # 添加文件信息
    doc.metadata["file_size"] = os.path.getsize("report.pdf")
    doc.metadata["loaded_at"] = datetime.now().isoformat()

    # 添加业务元数据
    doc.metadata["department"] = "技术部"
    doc.metadata["category"] = "技术报告"
    doc.metadata["year"] = 2025

# ===== 3. 在检索时使用元数据 =====
from langchain.vectorstores import Chroma
from langchain.embeddings import OpenAIEmbeddings

# 构建向量库
vectorstore = Chroma.from_documents(docs, OpenAIEmbeddings())

# 基于元数据过滤检索
results = vectorstore.similarity_search(
    "RAG技术",
    k=5,
    filter={"year": 2025, "department": "技术部"}  # 只检索2025年技术部的文档
)

# ===== 4. 在生成时引用元数据 =====
from langchain.chains import RetrievalQA
from langchain.llms import OpenAI

qa_chain = RetrievalQA.from_chain_type(
    llm=OpenAI(),
    retriever=vectorstore.as_retriever(),
    return_source_documents=True  # 返回源文档
)

result = qa_chain({"query": "RAG的核心流程是什么？"})

# 显示答案和来源
print(f"答案: {result['result']}")
print("\n来源:")
for doc in result['source_documents']:
    print(f"- {doc.metadata['source']} (第{doc.metadata['page']}页)")
```

**在RAG中的应用：**
- 按时间、部门、类别过滤检索结果
- 提供答案的可追溯性
- 支持多租户场景（按用户/组织过滤）

## 4.5 实现基本的错误处理

**核心概念：** 文档加载可能失败（文件损坏、编码错误），需要容错处理

```python
from langchain_community.document_loaders import PyPDFLoader, TextLoader
from typing import List
from langchain.schema import Document

def load_documents_safely(file_paths: List[str]) -> List[Document]:
    """
    安全加载多个文档，失败的文档不影响其他文档
    """
    all_docs = []
    failed_files = []

    for file_path in file_paths:
        try:
            # 根据文件扩展名选择Loader
            if file_path.endswith('.pdf'):
                loader = PyPDFLoader(file_path)
            elif file_path.endswith('.txt'):
                loader = TextLoader(file_path, encoding='utf-8')
            else:
                print(f"⚠️  不支持的文件格式: {file_path}")
                continue

            # 加载文档
            docs = loader.load()
            all_docs.extend(docs)
            print(f"✅ 成功加载: {file_path} ({len(docs)} 页)")

        except UnicodeDecodeError:
            # 编码错误，尝试其他编码
            try:
                loader = TextLoader(file_path, encoding='gbk')
                docs = loader.load()
                all_docs.extend(docs)
                print(f"✅ 使用GBK编码加载: {file_path}")
            except Exception as e:
                failed_files.append((file_path, str(e)))
                print(f"❌ 加载失败: {file_path} - {e}")

        except Exception as e:
            failed_files.append((file_path, str(e)))
            print(f"❌ 加载失败: {file_path} - {e}")

    # 汇总结果
    print(f"\n📊 加载完成:")
    print(f"  成功: {len(all_docs)} 个文档")
    print(f"  失败: {len(failed_files)} 个文件")

    if failed_files:
        print("\n失败文件列表:")
        for file_path, error in failed_files:
            print(f"  - {file_path}: {error}")

    return all_docs

# ===== 使用示例 =====
file_paths = [
    "docs/report1.pdf",
    "docs/report2.pdf",
    "docs/notes.txt",
    "docs/corrupted.pdf",  # 假设这个文件损坏
]

documents = load_documents_safely(file_paths)

# 继续RAG流程
if documents:
    print(f"\n继续处理 {len(documents)} 个成功加载的文档...")
    # 分块、向量化、存储...
else:
    print("\n⚠️  没有成功加载任何文档")
```

**在RAG中的应用：**
- 批量加载时不会因个别文件失败而中断
- 记录失败文件便于后续处理
- 提高系统的健壮性
