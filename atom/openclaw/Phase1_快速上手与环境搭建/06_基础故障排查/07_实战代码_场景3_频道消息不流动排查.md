# OpenClaw 基础故障排查 - 实战场景 3：频道消息不流动排查

## 一、场景概述

**频道消息不流动是指通道显示已连接但消息无法正常传递的问题，通过配对验证、提及门控检查和消息路由诊断，可以快速恢复消息流。**

**来源**: `reference/source_channels_troubleshooting.md`

## 二、问题分析

### 2.1 四大主要原因

| 原因类型 | 症状 | 发生频率 | 修复难度 |
|---------|------|---------|---------|
| 配对待定 | DM 消息无回复 | 50% | 简单 |
| 提及门控 | 群组消息无回复 | 30% | 简单 |
| 路由配置错误 | 特定消息无回复 | 15% | 中等 |
| Agent 未启动 | 所有消息无回复 | 5% | 简单 |

## 三、TypeScript 诊断实现

### 3.1 完整诊断类

```typescript
// channel-flow-diagnostic.ts
import { spawn } from 'child_process';

interface ChannelFlowIssue {
  type: 'pairing' | 'mention_gating' | 'routing' | 'agent' | 'unknown';
  channel: string;
  severity: 'high' | 'medium' | 'low';
  message: string;
  solution: string;
  autoFixable: boolean;
  details?: any;
}

interface DiagnosticResult {
  channelsOk: string[];
  channelsWithIssues: string[];
  issues: ChannelFlowIssue[];
  recommendations: string[];
}

class ChannelFlowDiagnostic {
  private channels: string[] = ['telegram', 'whatsapp', 'discord', 'slack'];

  async diagnose(): Promise<DiagnosticResult> {
    const channelsOk: string[] = [];
    const channelsWithIssues: string[] = [];
    const issues: ChannelFlowIssue[] = [];

    console.log('=== Channel Message Flow Diagnostic ===\n');

    // 1. 检查所有通道状态
    for (const channel of this.channels) {
      console.log(`Checking ${channel}...`);
      
      const channelIssues = await this.checkChannel(channel);
      
      if (channelIssues.length === 0) {
        channelsOk.push(channel);
        console.log(`✓ ${channel} is healthy`);
      } else {
        channelsWithIssues.push(channel);
        issues.push(...channelIssues);
        console.log(`✗ ${channel} has ${channelIssues.length} issue(s)`);
      }
    }

    // 2. 生成建议
    const recommendations = this.generateRecommendations(issues);

    return {
      channelsOk,
      channelsWithIssues,
      issues,
      recommendations
    };
  }

  private async checkChannel(channel: string): Promise<ChannelFlowIssue[]> {
    const issues: ChannelFlowIssue[] = [];

    // 1. 检查通道连接状态
    const connected = await this.isChannelConnected(channel);
    if (!connected) {
      issues.push({
        type: 'agent',
        channel,
        severity: 'high',
        message: `${channel} is not connected`,
        solution: `Run: openclaw channels reconnect ${channel}`,
        autoFixable: true
      });
      return issues; // 如果未连接，其他检查无意义
    }

    // 2. 检查配对状态
    const pairingIssue = await this.checkPairing(channel);
    if (pairingIssue) {
      issues.push(pairingIssue);
    }

    // 3. 检查提及门控
    const mentionIssue = await this.checkMentionGating(channel);
    if (mentionIssue) {
      issues.push(mentionIssue);
    }

    // 4. 检查 Agent 状态
    const agentIssue = await this.checkAgent(channel);
    if (agentIssue) {
      issues.push(agentIssue);
    }

    // 5. 检查消息路由
    const routingIssue = await this.checkRouting(channel);
    if (routingIssue) {
      issues.push(routingIssue);
    }

    return issues;
  }

  private async isChannelConnected(channel: string): Promise<boolean> {
    try {
      const output = await this.runCommand('openclaw', ['channels', 'status', '--probe']);
      return output.includes(`${channel}: ok`) || output.includes(`${channel}: connected`);
    } catch {
      return false;
    }
  }

  private async checkPairing(channel: string): Promise<ChannelFlowIssue | null> {
    try {
      const output = await this.runCommand('openclaw', ['pairing', 'list', channel]);
      
      // 检查是否有待处理的配对请求
      const pendingMatch = output.match(/Pending.*:\s*\n(.*)/s);
      if (pendingMatch && pendingMatch[1].trim().length > 0) {
        const pendingUsers = pendingMatch[1].trim().split('\n').map(line => line.trim());
        
        return {
          type: 'pairing',
          channel,
          severity: 'high',
          message: `${pendingUsers.length} pairing request(s) pending on ${channel}`,
          solution: `Run: openclaw pairing approve ${channel} <sender>`,
          autoFixable: true,
          details: { pendingUsers }
        };
      }
      
      return null;
    } catch {
      return null;
    }
  }

  private async checkMentionGating(channel: string): Promise<ChannelFlowIssue | null> {
    try {
      // 检查日志中是否有提及门控的消息
      const logs = await this.runCommand('openclaw', ['logs', '--tail', '100']);
      
      if (logs.includes(`drop guild message`) && logs.includes(`mention required`) && logs.includes(channel)) {
        return {
          type: 'mention_gating',
          channel,
          severity: 'medium',
          message: `Mention gating is blocking messages on ${channel}`,
          solution: `Mention bot in group (@BotName) or disable: openclaw config set channels.${channel}.requireMention false`,
          autoFixable: false
        };
      }
      
      return null;
    } catch {
      return null;
    }
  }

  private async checkAgent(channel: string): Promise<ChannelFlowIssue | null> {
    try {
      const output = await this.runCommand('openclaw', ['status', '--all']);
      
      // 检查是否有 Agent 配置
      if (output.includes(`no agent configured for channel`) && output.includes(channel)) {
        return {
          type: 'agent',
          channel,
          severity: 'high',
          message: `No agent configured for ${channel}`,
          solution: `Configure agent routing for ${channel}`,
          autoFixable: false
        };
      }
      
      return null;
    } catch {
      return null;
    }
  }

  private async checkRouting(channel: string): Promise<ChannelFlowIssue | null> {
    try {
      const logs = await this.runCommand('openclaw', ['logs', '--tail', '100']);
      
      if (logs.includes(`no route found`) && logs.includes(channel)) {
        return {
          type: 'routing',
          channel,
          severity: 'medium',
          message: `Message routing failed for ${channel}`,
          solution: `Check routing configuration`,
          autoFixable: false
        };
      }
      
      return null;
    } catch {
      return null;
    }
  }

  private generateRecommendations(issues: ChannelFlowIssue[]): string[] {
    const recommendations: string[] = [];

    // 按严重程度和类型分组
    const highPriorityIssues = issues.filter(i => i.severity === 'high');
    const pairingIssues = issues.filter(i => i.type === 'pairing');
    const mentionIssues = issues.filter(i => i.type === 'mention_gating');

    if (highPriorityIssues.length > 0) {
      recommendations.push('High priority issues found:');
      highPriorityIssues.forEach(issue => {
        recommendations.push(`  - ${issue.message}`);
        recommendations.push(`    Solution: ${issue.solution}`);
      });
    }

    if (pairingIssues.length > 0) {
      recommendations.push('');
      recommendations.push('Quick fix for pairing issues:');
      pairingIssues.forEach(issue => {
        if (issue.details?.pendingUsers) {
          issue.details.pendingUsers.forEach((user: string) => {
            recommendations.push(`  openclaw pairing approve ${issue.channel} ${user}`);
          });
        }
      });
    }

    if (mentionIssues.length > 0) {
      recommendations.push('');
      recommendations.push('For mention gating issues:');
      recommendations.push('  Option 1: Mention bot in group messages (@BotName)');
      recommendations.push('  Option 2: Disable mention gating (not recommended for public groups)');
    }

    if (recommendations.length === 0) {
      recommendations.push('All channels are healthy');
    }

    return recommendations;
  }

  async autoFix(issues: ChannelFlowIssue[]): Promise<void> {
    console.log('\n=== Attempting Auto-Fix ===\n');

    for (const issue of issues) {
      if (!issue.autoFixable) {
        console.log(`⚠ Cannot auto-fix: ${issue.message}`);
        console.log(`  Manual action required: ${issue.solution}`);
        continue;
      }

      console.log(`Fixing: ${issue.message}...`);

      try {
        switch (issue.type) {
          case 'pairing':
            await this.fixPairing(issue);
            break;
          case 'agent':
            await this.fixAgent(issue);
            break;
        }
        console.log('✓ Fixed');
      } catch (error) {
        console.log(`✗ Failed: ${error.message}`);
      }
    }
  }

  private async fixPairing(issue: ChannelFlowIssue): Promise<void> {
    if (issue.details?.pendingUsers) {
      for (const user of issue.details.pendingUsers) {
        await this.runCommand('openclaw', ['pairing', 'approve', issue.channel, user]);
      }
    }
  }

  private async fixAgent(issue: ChannelFlowIssue): Promise<void> {
    if (issue.message.includes('not connected')) {
      await this.runCommand('openclaw', ['channels', 'reconnect', issue.channel]);
    }
  }

  private runCommand(cmd: string, args: string[]): Promise<string> {
    return new Promise((resolve, reject) => {
      const proc = spawn(cmd, args);
      let output = '';

      proc.stdout.on('data', (data) => {
        output += data.toString();
      });

      proc.stderr.on('data', (data) => {
        output += data.toString();
      });

      proc.on('close', (code) => {
        if (code === 0) {
          resolve(output);
        } else {
          reject(new Error(`Command failed with code ${code}`));
        }
      });
    });
  }
}

// 使用示例
async function main() {
  const diagnostic = new ChannelFlowDiagnostic();
  
  // 运行诊断
  const result = await diagnostic.diagnose();
  
  // 输出结果
  console.log('\n=== Diagnostic Result ===');
  console.log(`Healthy channels: ${result.channelsOk.join(', ') || 'None'}`);
  console.log(`Channels with issues: ${result.channelsWithIssues.join(', ') || 'None'}`);
  
  if (result.issues.length > 0) {
    console.log('\nIssues Found:');
    result.issues.forEach(issue => {
      console.log(`  [${issue.severity.toUpperCase()}] ${issue.channel}: ${issue.message}`);
    });
    
    console.log('\nRecommendations:');
    result.recommendations.forEach(rec => {
      console.log(`  ${rec}`);
    });
    
    // 尝试自动修复
    const autoFixableIssues = result.issues.filter(i => i.autoFixable);
    if (autoFixableIssues.length > 0) {
      await diagnostic.autoFix(autoFixableIssues);
    }
  } else {
    console.log('✓ All channels are healthy');
  }
}

main();
```

## 四、Bash 排查脚本

### 4.1 完整排查脚本

```bash
#!/bin/bash
# channel-flow-troubleshoot.sh

echo "=== OpenClaw Channel Message Flow Troubleshooting ==="
echo "Start time: $(date)"
echo ""

# 颜色定义
GREEN='\033[0;32m'
RED='\033[0;31m'
YELLOW='\033[1;33m'
NC='\033[0m'

# 通道列表
CHANNELS=("telegram" "whatsapp" "discord" "slack")
ISSUES_FOUND=0

# 检查每个通道
for CHANNEL in "${CHANNELS[@]}"; do
  echo "=== Checking $CHANNEL ==="
  
  # 1. 检查连接状态
  echo "[1/4] Checking connection..."
  STATUS=$(openclaw channels status --probe 2>&1 | grep "$CHANNEL")
  
  if echo "$STATUS" | grep -q "connected\|ok"; then
    echo -e "${GREEN}✓${NC} $CHANNEL is connected"
  else
    echo -e "${RED}✗${NC} $CHANNEL is not connected"
    echo "  Run: openclaw channels reconnect $CHANNEL"
    ISSUES_FOUND=$((ISSUES_FOUND + 1))
    continue
  fi
  
  # 2. 检查配对状态
  echo "[2/4] Checking pairing..."
  PAIRING=$(openclaw pairing list "$CHANNEL" 2>&1)
  
  if echo "$PAIRING" | grep -q "Pending"; then
    echo -e "${YELLOW}⚠${NC} Pairing requests pending"
    echo "$PAIRING" | grep -A 10 "Pending"
    echo "  Run: openclaw pairing approve $CHANNEL <sender>"
    ISSUES_FOUND=$((ISSUES_FOUND + 1))
  else
    echo -e "${GREEN}✓${NC} No pending pairing requests"
  fi
  
  # 3. 检查提及门控
  echo "[3/4] Checking mention gating..."
  LOGS=$(openclaw logs --tail 50 2>&1)
  
  if echo "$LOGS" | grep -q "drop guild message.*mention required.*$CHANNEL"; then
    echo -e "${YELLOW}⚠${NC} Mention gating is blocking messages"
    echo "  Option 1: Mention bot in group (@BotName)"
    echo "  Option 2: Disable mention gating:"
    echo "    openclaw config set channels.$CHANNEL.requireMention false"
    ISSUES_FOUND=$((ISSUES_FOUND + 1))
  else
    echo -e "${GREEN}✓${NC} No mention gating issues"
  fi
  
  # 4. 检查 Agent 状态
  echo "[4/4] Checking agent..."
  AGENT_STATUS=$(openclaw status --all 2>&1)
  
  if echo "$AGENT_STATUS" | grep -q "no agent configured.*$CHANNEL"; then
    echo -e "${RED}✗${NC} No agent configured"
    echo "  Configure agent routing for $CHANNEL"
    ISSUES_FOUND=$((ISSUES_FOUND + 1))
  else
    echo -e "${GREEN}✓${NC} Agent is configured"
  fi
  
  echo ""
done

# 总结
echo "=== Summary ==="
echo "Issues found: $ISSUES_FOUND"

if [ $ISSUES_FOUND -eq 0 ]; then
  echo -e "${GREEN}✓ All channels are healthy${NC}"
  exit 0
else
  echo -e "${RED}✗ Issues need to be resolved${NC}"
  echo ""
  echo "Quick fix steps:"
  echo "  1. Approve pending pairing requests"
  echo "  2. Mention bot in group messages or disable mention gating"
  echo "  3. Check agent configuration"
  echo "  4. Verify message routing"
  exit 1
fi
```

### 4.2 批量配对批准脚本

```bash
#!/bin/bash
# approve-all-pairing.sh

echo "=== Approve All Pending Pairing Requests ==="
echo ""

CHANNELS=("telegram" "whatsapp" "discord" "slack")

for CHANNEL in "${CHANNELS[@]}"; do
  echo "Checking $CHANNEL..."
  
  # 获取待处理的配对请求
  PENDING=$(openclaw pairing list "$CHANNEL" 2>&1 | grep -A 100 "Pending" | grep "^  -" | sed 's/^  - //' | awk '{print $1}')
  
  if [ -z "$PENDING" ]; then
    echo "  No pending requests"
    continue
  fi
  
  # 批准所有待处理的请求
  echo "$PENDING" | while read -r SENDER; do
    echo "  Approving: $SENDER"
    openclaw pairing approve "$CHANNEL" "$SENDER"
  done
  
  echo ""
done

echo "✓ All pending pairing requests approved"
