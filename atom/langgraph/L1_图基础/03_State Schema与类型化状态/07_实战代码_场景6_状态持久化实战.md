# 实战代码 - 场景6：状态持久化实战

> 完整可运行的状态持久化与 Checkpointer 示例

## 场景描述

在 LangGraph 中，状态持久化允许保存和恢复图的执行状态。本场景演示如何使用 Checkpointer 实现状态持久化、多轮对话和状态回溯。

**应用场景**：
- 多轮对话系统
- 长时间运行的工作流
- 状态恢复与回溯
- 分布式任务处理

## 核心知识点

1. **MemorySaver**：内存持久化
2. **PostgresSaver**：数据库持久化
3. **Thread ID**：会话隔离
4. **Checkpoint**：状态快照
5. **State 恢复**：从 checkpoint 恢复

## 完整代码：MemorySaver

```python
from typing import TypedDict, Annotated
import operator
from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.memory import MemorySaver

# 1. 定义状态 Schema
class State(TypedDict):
    messages: Annotated[list, operator.add]
    user_id: str
    turn_count: int

# 2. 定义节点
def process_message(state: State) -> dict:
    """处理消息"""
    turn = state.get('turn_count', 0) + 1
    user_msg = state['messages'][-1] if state['messages'] else ""

    print(f"\n=== Turn {turn} ===")
    print(f"User: {user_msg}")

    response = f"Response to: {user_msg}"
    print(f"Assistant: {response}")

    return {
        "messages": [response],
        "turn_count": turn
    }

# 3. 构建图
graph = StateGraph(State)
graph.add_node("process", process_message)
graph.add_edge(START, "process")
graph.add_edge("process", END)

# 4. 使用 MemorySaver 编译
checkpointer = MemorySaver()
app = graph.compile(checkpointer=checkpointer)

# 5. 多轮对话
if __name__ == "__main__":
    # 配置：使用 thread_id 隔离会话
    config = {"configurable": {"thread_id": "user_123"}}

    # 第一轮对话
    result1 = app.invoke({
        "messages": ["Hello"],
        "user_id": "user_123",
        "turn_count": 0
    }, config=config)

    print(f"\nState after turn 1: {result1}")

    # 第二轮对话（状态自动恢复）
    result2 = app.invoke({
        "messages": ["How are you?"],
        "user_id": "user_123"
    }, config=config)

    print(f"\nState after turn 2: {result2}")

    # 第三轮对话
    result3 = app.invoke({
        "messages": ["Tell me a joke"],
        "user_id": "user_123"
    }, config=config)

    print(f"\nState after turn 3: {result3}")
    print(f"\nTotal messages: {len(result3['messages'])}")
    print(f"Total turns: {result3['turn_count']}")
```

## 完整代码：PostgresSaver

```python
from typing import TypedDict, Annotated
import operator
from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.postgres import PostgresSaver
from psycopg_pool import ConnectionPool

# 1. 定义状态 Schema
class State(TypedDict):
    messages: Annotated[list, operator.add]
    user_id: str
    session_id: str

# 2. 定义节点
def chat_node(state: State) -> dict:
    """聊天节点"""
    user_msg = state['messages'][-1] if state['messages'] else ""
    response = f"Echo: {user_msg}"

    return {"messages": [response]}

# 3. 构建图
graph = StateGraph(State)
graph.add_node("chat", chat_node)
graph.add_edge(START, "chat")
graph.add_edge("chat", END)

# 4. 使用 PostgresSaver 编译
# 创建连接池
pool = ConnectionPool(
    "postgresql://user:password@localhost:5432/langgraph",
    min_size=1,
    max_size=10
)

# 创建 checkpointer
checkpointer = PostgresSaver(pool)

# 编译图
app = graph.compile(checkpointer=checkpointer)

# 5. 使用
if __name__ == "__main__":
    config = {"configurable": {"thread_id": "session_456"}}

    # 第一轮
    result1 = app.invoke({
        "messages": ["Hello"],
        "user_id": "user_456",
        "session_id": "session_456"
    }, config=config)

    # 第二轮（状态从数据库恢复）
    result2 = app.invoke({
        "messages": ["How are you?"],
        "user_id": "user_456",
        "session_id": "session_456"
    }, config=config)

    print(f"Messages: {result2['messages']}")
```

## 状态回溯

```python
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END

# 1. 定义状态
class State(TypedDict):
    count: int
    history: Annotated[list, operator.add]

# 2. 定义节点
def increment(state: State) -> dict:
    new_count = state['count'] + 1
    return {
        "count": new_count,
        "history": [f"Incremented to {new_count}"]
    }

# 3. 构建图
graph = StateGraph(State)
graph.add_node("increment", increment)
graph.add_edge(START, "increment")
graph.add_edge("increment", END)

checkpointer = MemorySaver()
app = graph.compile(checkpointer=checkpointer)

# 4. 执行多次
config = {"configurable": {"thread_id": "counter_1"}}

for i in range(5):
    result = app.invoke({"count": 0, "history": []}, config=config)
    print(f"Step {i+1}: count = {result['count']}")

# 5. 获取历史 checkpoints
checkpoints = list(app.get_state_history(config))
print(f"\nTotal checkpoints: {len(checkpoints)}")

# 6. 回溯到特定 checkpoint
if len(checkpoints) >= 3:
    # 获取第3个 checkpoint
    checkpoint_3 = checkpoints[2]
    print(f"\nCheckpoint 3 state: {checkpoint_3.values}")

    # 从 checkpoint 3 继续执行
    config_with_checkpoint = {
        "configurable": {
            "thread_id": "counter_1",
            "checkpoint_id": checkpoint_3.config["configurable"]["checkpoint_id"]
        }
    }

    result = app.invoke({"count": 0, "history": []}, config_with_checkpoint)
    print(f"After resuming from checkpoint 3: {result}")
```

## 多用户状态隔离

```python
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END

# 1. 定义状态
class UserState(TypedDict):
    user_id: str
    messages: Annotated[list, operator.add]
    preferences: dict

# 2. 定义节点
def user_node(state: UserState) -> dict:
    user_id = state['user_id']
    msg = state['messages'][-1] if state['messages'] else ""

    response = f"[User {user_id}] Response to: {msg}"

    return {"messages": [response]}

# 3. 构建图
graph = StateGraph(UserState)
graph.add_node("process", user_node)
graph.add_edge(START, "process")
graph.add_edge("process", END)

checkpointer = MemorySaver()
app = graph.compile(checkpointer=checkpointer)

# 4. 多用户并发使用
if __name__ == "__main__":
    # 用户 A
    config_a = {"configurable": {"thread_id": "user_a"}}
    result_a = app.invoke({
        "user_id": "user_a",
        "messages": ["Hello from A"],
        "preferences": {"lang": "en"}
    }, config=config_a)

    # 用户 B
    config_b = {"configurable": {"thread_id": "user_b"}}
    result_b = app.invoke({
        "user_id": "user_b",
        "messages": ["Hello from B"],
        "preferences": {"lang": "zh"}
    }, config=config_b)

    # 用户 A 第二轮
    result_a2 = app.invoke({
        "user_id": "user_a",
        "messages": ["How are you?"]
    }, config=config_a)

    print(f"User A messages: {result_a2['messages']}")
    print(f"User B messages: {result_b['messages']}")
```

## 状态清理

```python
from langgraph.checkpoint.memory import MemorySaver
import time

# 1. 定义状态
class State(TypedDict):
    data: str
    timestamp: float

# 2. 定义节点
def process(state: State) -> dict:
    return {
        "data": f"Processed: {state['data']}",
        "timestamp": time.time()
    }

# 3. 构建图
graph = StateGraph(State)
graph.add_node("process", process)
graph.add_edge(START, "process")
graph.add_edge("process", END)

checkpointer = MemorySaver()
app = graph.compile(checkpointer=checkpointer)

# 4. 清理旧状态
def cleanup_old_states(app, max_age_seconds=3600):
    """清理超过指定时间的状态"""
    current_time = time.time()

    # 获取所有 thread_ids（需要自己维护）
    thread_ids = ["thread_1", "thread_2", "thread_3"]

    for thread_id in thread_ids:
        config = {"configurable": {"thread_id": thread_id}}

        try:
            # 获取最新状态
            state = app.get_state(config)

            if state and state.values:
                timestamp = state.values.get('timestamp', 0)

                # 如果状态过旧，删除
                if current_time - timestamp > max_age_seconds:
                    print(f"Cleaning up old state for {thread_id}")
                    # 注意：MemorySaver 不支持删除，需要使用支持删除的 checkpointer
        except Exception as e:
            print(f"Error cleaning up {thread_id}: {e}")

# 5. 定期清理
if __name__ == "__main__":
    # 创建一些状态
    for i in range(3):
        config = {"configurable": {"thread_id": f"thread_{i}"}}
        app.invoke({"data": f"data_{i}", "timestamp": time.time()}, config=config)

    # 清理
    cleanup_old_states(app, max_age_seconds=3600)
```

## 性能优化

### 1. 批量保存

```python
# 使用批量操作减少数据库访问
def batch_save_states(app, states, configs):
    """批量保存状态"""
    for state, config in zip(states, configs):
        app.invoke(state, config=config)
```

### 2. 异步持久化

```python
import asyncio
from langgraph.checkpoint.postgres import AsyncPostgresSaver

async def async_save():
    """异步保存状态"""
    checkpointer = AsyncPostgresSaver(pool)
    app = graph.compile(checkpointer=checkpointer)

    result = await app.ainvoke(state, config=config)
    return result
```

### 3. 压缩状态

```python
import json
import gzip

def compress_state(state: dict) -> bytes:
    """压缩状态"""
    json_str = json.dumps(state)
    return gzip.compress(json_str.encode())

def decompress_state(data: bytes) -> dict:
    """解压状态"""
    json_str = gzip.decompress(data).decode()
    return json.loads(json_str)
```

## 最佳实践

1. **选择合适的 Checkpointer**：
   - 开发环境：MemorySaver
   - 生产环境：PostgresSaver 或 RedisSaver

2. **使用 Thread ID 隔离**：
   - 每个用户/会话使用唯一的 thread_id
   - 避免状态泄露

3. **定期清理**：
   - 删除过期状态
   - 归档历史数据

4. **监控状态大小**：
   - 避免状态过大
   - 使用 reducer 控制增长

5. **错误处理**：
   - 处理持久化失败
   - 实现重试机制

## 常见陷阱

### 陷阱1：忘记配置 thread_id

```python
# 错误：没有 thread_id
result = app.invoke(state)  # 每次都是新会话

# 正确：使用 thread_id
config = {"configurable": {"thread_id": "user_123"}}
result = app.invoke(state, config=config)
```

### 陷阱2：状态过大

```python
# 错误：状态无限增长
class State(TypedDict):
    messages: Annotated[list, operator.add]  # 无限追加

# 正确：限制状态大小
def trim_messages(a: list, b: list) -> list:
    all_msgs = a + b
    return all_msgs[-100:]  # 只保留最近100条

class State(TypedDict):
    messages: Annotated[list, trim_messages]
```

### 陷阱3：不处理持久化失败

```python
# 错误：不处理异常
result = app.invoke(state, config=config)

# 正确：处理异常
try:
    result = app.invoke(state, config=config)
except Exception as e:
    print(f"Persistence failed: {e}")
    # 重试或降级处理
```
