```python
# 创建测试集合
def create_test_collection(collection_name, index_type):
    if client.has_collection(collection_name):
        client.drop_collection(collection_name)

    schema = client.create_schema(auto_id=True)
    schema.add_field("id", DataType.INT64, is_primary=True)
    schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
    schema.add_field("age", DataType.INT32)
    schema.add_field("status", DataType.VARCHAR, max_length=20)
    schema.add_field("score", DataType.DOUBLE)

    client.create_collection(collection_name=collection_name, schema=schema)

    # 插入数据
    statuses = ["active", "pending", "deleted"]
    data = []
    for i in range(100000):

            "vector": vector,
            "age": random.randint(20, 60),
            "status": random.choice(statuses),
            "score": random.uniform(0, 100)

    # 创建索引
    index_params = client.prepare_index_params()
    index_params.add_index("vector", "AUTOINDEX", "COSINE")
    index_params.add_index("age", index_type)
    index_params.add_index("status", "BITMAP")
    index_params.add_index("score", index_type)

    client.create_index(collection_name, index_params)

    return collection_name

# 创建不同索引类型的集合
collections = {

    "stl_sort": create_test_collection("test_stl_sort", "STL_SORT"),
```

## 场景2：基数分析与索引选择

### 基数计算

```python
def analyze_cardinality(collection_name, field_name):
    """分析字段基数"""
    # 查询所有数据
    results = client.query(
        collection_name=collection_name,
        filter="id > 0",
        output_fields=[field_name],
        limit=100000
    )

    # 计算基数
    values = [r[field_name] for r in results]
    unique_values = len(set(values))
    total_values = len(values)
    cardinality = unique_values / total_values

    return {
        "unique_values": unique_values,
        "total_values": total_values,
        "cardinality": cardinality
    }

# 分析字段基数
print("\n=== 字段基数分析 ===")
for field in ["age", "status", "score"]:
    stats = analyze_cardinality(collections["inverted"], field)
    print(f"\n{field}:")
    print(f"  唯一值: {stats['unique_values']}")
    print(f"  总数: {stats['total_values']}")
    print(f"  基数: {stats['cardinality']:.4f}")

    # 推荐索引类型
    if stats['cardinality'] < 0.01:
        print(f"  推荐: BITMAP")
    else:
        print(f"  推荐: INVERTED")
```

### 索引选择策略

```python
def recommend_index_type(cardinality, data_type):
    """根据基数和数据类型推荐索引"""
    if cardinality < 0.01:  # 低基数（< 1%）
        return "BITMAP"
    elif data_type in ["INT", "DOUBLE", "FLOAT"]:
        return "INVERTED"  # 数值类型
    elif data_type == "VARCHAR":
        return "INVERTED"  # 字符串类型
    else:
        return "AUTOINDEX"  # 不确定时使用自动索引

# 示例
print("\n=== 索引选择示例 ===")
fields = [
    ("age", 0.04, "INT"),
    ("status", 0.003, "VARCHAR"),
    ("score", 0.80, "DOUBLE"),
    ("city", 0.005, "VARCHAR")
]

for field_name, cardinality, data_type in fields:
    recommended = recommend_index_type(cardinality, data_type)
    print(f"{field_name:10s} (基数={cardinality:.3f}): {recommended}")
```

## 场景5：查询优化

### 条件顺序优化

```python
def test_filter_order(collection_name, query_vector):
    """测试过滤条件顺序对性能的影响"""

    # 不好的顺序（选择性低的在前）
    filter_bad = "age >= 20 AND age <= 50 AND status == 'active'"
    # age范围保留80%数据（选择性低）
    # status保留33%数据（选择性高）

    # 好的顺序（选择性高的在前）
    filter_good = "status == 'active' AND age >= 20 AND age <= 50"

    # 测试
    start = time.time()
    client.search(collection_name, [query_vector], filter=filter_bad, limit=10)
    time_bad = (time.time() - start) * 1000

    start = time.time()
    client.search(collection_name, [query_vector], filter=filter_good, limit=10)
    time_good = (time.time() - start) * 1000

    return time_bad, time_good

print("\n=== 条件顺序优化 ===")
time_bad, time_good = test_filter_order(collections["inverted"], query_vector)
print(f"选择性低在前: {time_bad:.2f} ms")
print(f"选择性高在前: {time_good:.2f} ms")
print(f"性能提升: {time_bad / time_good:.2f}x")
```

```python
def test_in_vs_or(collection_name, query_vector):
```

```python
from pymilvus import MilvusClient, DataType
from sentence_transformers import SentenceTransformer
import random
import time

def main():
    # 初始化
    client = MilvusClient("http://localhost:19530")
    model = SentenceTransformer('all-MiniLM-L6-v2')

    # 创建测试集合
    collection_name = "index_optimization_demo"
    if client.has_collection(collection_name):
        client.drop_collection(collection_name)

    schema = client.create_schema(auto_id=True)
    schema.add_field("id", DataType.INT64, is_primary=True)
    schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
    schema.add_field("age", DataType.INT32)
    schema.add_field("status", DataType.VARCHAR, max_length=20)
    schema.add_field("score", DataType.DOUBLE)

    client.create_collection(collection_name=collection_name, schema=schema)

    # 插入数据
    statuses = ["active", "pending", "deleted"]
    data = []
    for i in range(10000):
        text = f"User {i}"
        vector = model.encode(text).tolist()
        data.append({
            "vector": vector,
            "age": random.randint(20, 60),
            "status": random.choice(statuses),
            "score": random.uniform(0, 100)

    # 创建索引
    index_params = client.prepare_index_params()
    index_params.add_index("vector", "AUTOINDEX", "COSINE")
    index_params.add_index("age", "INVERTED")
    index_params.add_index("status", "BITMAP")  # 低基数使用BITMAP
    index_params.add_index("score", "INVERTED")

    client.create_index(collection_name, index_params)

    query_vector = model.encode("Test query").tolist()

    # 测试1：基础查询
    print("=== 基础查询 ===")
    start = time.time()
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="age > 30 AND status == 'active'",
        limit=10,
        output_fields=["age", "status", "score"]
    )
    elapsed = (time.time() - start) * 1000
    print(f"查询延迟: {elapsed:.2f} ms")

        collection_name=collection_name,

    print(f"查询延迟: {elapsed:.2f} ms")

    # 测试3：IN 查询

        filter="status IN ['active', 'pending']",
```

```python
# 1. 分析基数
基数 = DISTINCT(field) / COUNT(*)
```

## 类比理解

### 前端开发类比

**INVERTED = Map 查找**：
```javascript
// Map 查找 O(1)
const ageMap = new Map();
ageMap.set(25, [1, 5, 10]);
const result = ageMap.get(25);

// Milvus INVERTED
index_params.add_index("age", "INVERTED")
```

**BITMAP = Set 查找**：
```javascript
// Set 查找 O(1)
const activeUsers = new Set([1, 3, 5, 7]);
const isActive = activeUsers.has(5);

// Milvus BITMAP
index_params.add_index("status", "BITMAP")
```

### 日常生活类比

**INVERTED = 电话簿**：
```
电话簿：按姓名排序，快速查找
Milvus INVERTED：按值排序，快速查找
```

**BITMAP = 座位图**：
```
电影院座位图：已售/可售（只有2种状态）
Milvus BITMAP：适用于低基数字段
```
