# 实战代码 - 场景5：RAG 工作流中的边使用

## 场景说明

本示例展示如何在 RAG（检索增强生成）工作流中使用不同类型的边：
- **普通边**：固定的流程步骤（START -> retrieve）
- **条件边**：根据文档质量评估动态路由（grade_documents -> generate/web_search）
- **循环边**：查询改写后重新检索（transform_query -> retrieve）

这是一个简化的 Self-RAG 工作流，展示了边的类型如何支持复杂的决策逻辑。

## 完整代码

```python
"""
LangGraph RAG 工作流 - 边的类型实战
演示：检索 -> 评估 -> 生成的完整流程
"""

from typing import TypedDict, Literal, List
from langgraph.graph import StateGraph, START, END

# ===== 1. 定义状态 =====
class RAGState(TypedDict):
    """RAG 工作流的状态定义"""
    question: str                    # 用户问题
    documents: List[str]             # 检索到的文档
    generation: str                  # 生成的答案
    web_search_needed: bool          # 是否需要网络搜索
    retry_count: int                 # 重试次数


# ===== 2. 定义节点函数 =====

def retrieve(state: RAGState) -> dict:
    """
    检索节点：根据问题检索相关文档
    """
    question = state["question"]
    print(f"\n[检索] 正在检索问题: {question}")

    # 模拟文档检索（实际应用中会调用向量数据库）
    mock_docs = {
        "什么是 LangGraph": [
            "LangGraph 是一个用于构建有状态、多参与者应用的框架",
            "LangGraph 基于 LangChain，提供了图结构的工作流编排能力"
        ],
        "Python 的优势": [
            "Python 是一种简单易学的编程语言",
            "Python 在数据科学和 AI 领域应用广泛"
        ]
    }

    # 根据问题匹配文档
    docs = []
    for key, values in mock_docs.items():
        if any(word in question for word in key.split()):
            docs = values
            break

    if not docs:
        docs = ["未找到相关文档"]

    print(f"[检索] 找到 {len(docs)} 个文档")
    return {"documents": docs, "retry_count": state.get("retry_count", 0)}


def grade_documents(state: RAGState) -> dict:
    """
    文档评估节点：评估检索到的文档质量
    """
    documents = state["documents"]
    question = state["question"]

    print(f"\n[评估] 正在评估 {len(documents)} 个文档的相关性")

    # 模拟文档质量评估（实际应用中会使用 LLM 评估）
    # 简单规则：如果文档包含问题关键词，认为相关
    relevant = False
    for doc in documents:
        if any(word in doc for word in question.split() if len(word) > 2):
            relevant = True
            break

    # 如果文档不相关且重试次数少于2次，需要网络搜索
    web_search_needed = not relevant and state.get("retry_count", 0) < 2

    print(f"[评估] 文档相关性: {'相关' if relevant else '不相关'}")
    print(f"[评估] 需要网络搜索: {web_search_needed}")

    return {"web_search_needed": web_search_needed}


def web_search(state: RAGState) -> dict:
    """
    网络搜索节点：当本地文档不足时进行网络搜索
    """
    question = state["question"]
    print(f"\n[网络搜索] 正在搜索: {question}")

    # 模拟网络搜索结果
    web_docs = [
        f"网络搜索结果：关于 '{question}' 的最新信息",
        "网络搜索结果：相关的补充资料"
    ]

    # 合并原有文档和网络搜索结果
    all_docs = state["documents"] + web_docs
    print(f"[网络搜索] 添加了 {len(web_docs)} 个网络文档")

    return {"documents": all_docs}


def transform_query(state: RAGState) -> dict:
    """
    查询改写节点：改写查询以提高检索质量
    """
    question = state["question"]
    retry_count = state.get("retry_count", 0)

    print(f"\n[查询改写] 正在改写查询 (第 {retry_count + 1} 次)")

    # 模拟查询改写（实际应用中会使用 LLM）
    transformed = f"{question} 详细解释"
    print(f"[查询改写] 原查询: {question}")
    print(f"[查询改写] 新查询: {transformed}")

    return {
        "question": transformed,
        "retry_count": retry_count + 1
    }


def generate(state: RAGState) -> dict:
    """
    生成节点：基于文档生成答案
    """
    question = state["question"]
    documents = state["documents"]

    print(f"\n[生成] 正在生成答案")
    print(f"[生成] 基于 {len(documents)} 个文档")

    # 模拟答案生成（实际应用中会调用 LLM）
    context = "\n".join(f"- {doc}" for doc in documents[:3])
    answer = f"""基于检索到的文档，回答如下：

问题：{question}

参考文档：
{context}

答案：根据以上文档，这是一个综合性的回答。"""

    print(f"[生成] 答案长度: {len(answer)} 字符")

    return {"generation": answer}


# ===== 3. 定义路由函数 =====

def decide_to_generate(state: RAGState) -> Literal["web_search", "generate"]:
    """
    决策函数：根据文档评估结果决定下一步

    返回值：
    - "web_search": 文档不相关，需要网络搜索
    - "generate": 文档相关，直接生成答案
    """
    if state["web_search_needed"]:
        return "web_search"
    return "generate"


def decide_after_search(state: RAGState) -> Literal["transform_query", "generate"]:
    """
    决策函数：网络搜索后决定是改写查询还是生成答案

    返回值：
    - "transform_query": 重试次数未超限，改写查询重新检索
    - "generate": 重试次数已达上限，直接生成答案
    """
    retry_count = state.get("retry_count", 0)

    # 如果重试次数超过1次，直接生成答案
    if retry_count >= 1:
        return "generate"
    return "transform_query"


# ===== 4. 构建工作流图 =====

def create_rag_workflow():
    """
    创建 RAG 工作流图
    """
    # 创建状态图
    workflow = StateGraph(RAGState)

    # 添加节点
    workflow.add_node("retrieve", retrieve)
    workflow.add_node("grade_documents", grade_documents)
    workflow.add_node("web_search", web_search)
    workflow.add_node("transform_query", transform_query)
    workflow.add_node("generate", generate)

    # ===== 添加边 =====

    # 1. 普通边：START -> retrieve（固定入口）
    workflow.add_edge(START, "retrieve")

    # 2. 普通边：retrieve -> grade_documents（固定流程）
    workflow.add_edge("retrieve", "grade_documents")

    # 3. 条件边：grade_documents -> web_search/generate（根据评估结果路由）
    workflow.add_conditional_edges(
        "grade_documents",
        decide_to_generate,
        {
            "web_search": "web_search",
            "generate": "generate"
        }
    )

    # 4. 条件边：web_search -> transform_query/generate（根据重试次数路由）
    workflow.add_conditional_edges(
        "web_search",
        decide_after_search,
        {
            "transform_query": "transform_query",
            "generate": "generate"
        }
    )

    # 5. 循环边：transform_query -> retrieve（查询改写后重新检索）
    workflow.add_edge("transform_query", "retrieve")

    # 6. 普通边：generate -> END（固定出口）
    workflow.add_edge("generate", END)

    # 编译图
    return workflow.compile()


# ===== 5. 测试工作流 =====

def test_rag_workflow():
    """
    测试 RAG 工作流的不同路径
    """
    graph = create_rag_workflow()

    print("=" * 60)
    print("测试场景 1：文档相关，直接生成答案")
    print("=" * 60)

    result1 = graph.invoke({
        "question": "什么是 LangGraph",
        "documents": [],
        "generation": "",
        "web_search_needed": False,
        "retry_count": 0
    })

    print(f"\n最终答案：\n{result1['generation']}")

    print("\n" + "=" * 60)
    print("测试场景 2：文档不相关，触发网络搜索")
    print("=" * 60)

    result2 = graph.invoke({
        "question": "量子计算的最新进展",
        "documents": [],
        "generation": "",
        "web_search_needed": False,
        "retry_count": 0
    })

    print(f"\n最终答案：\n{result2['generation']}")

    print("\n" + "=" * 60)
    print("测试场景 3：多次重试后生成答案")
    print("=" * 60)

    result3 = graph.invoke({
        "question": "深度学习的未来趋势",
        "documents": [],
        "generation": "",
        "web_search_needed": False,

    })

    print(f"\n最终答案：\n{result3['generation']}")


# ===== 6. 可视化工作流 =====

def visualize_workflow():
    """
    打印工作流的结构信息
    """
    graph = create_rag_workflow()

    print("\n" + "=" * 60)
    print("RAG 工作流结构")
    print("=" * 60)

    print("\n节点列表：")
    print("- retrieve: 检索文档")

    print("- transform_query: 查询改写")
    print("- generate: 生成答案")

    print("\n边的类型：")
    print("1. 普通边 (Normal Edge):")
    print("   - START -> retrieve")
    print("   - retrieve -> grade_documents")
    print("   - transform_query -> retrieve (循环边)")
    print("   - generate -> END")

    print("\n2. 条件边 (Conditional Edge):")
    print("   - grade_documents -> [web_search | generate]")
    print("     条件：文档是否相关")
    print("   - web_search -> [transform_query | generate]")
    print("     条件：重试次数是否超限")

    print("\n工作流路径：")
    print("路径1（文档相关）：")
    print("  START -> retrieve -> grade_documents -> generate -> END")

    print("\n路径2（需要网络搜索）：")
    print("  START -> retrieve -> grade_documents -> web_search -> generate -> END")

    print("\n路径3（需要查询改写）：")
    print("  START -> retrieve -> grade_documents -> web_search")
    print("  -> transform_query -> retrieve -> ... -> generate -> END")


# ===== 7. 主函数 =====

if __name__ == "__main__":
    print("LangGraph RAG 工作流 - 边的类型实战\n")

    # 可视化工作流结构
    visualize_workflow()

    # 测试工作流

```

## 代码说明

### 1. 边的类型使用

**普通边（Normal Edge）**：
```python
# 固定流程，无条件执行
workflow.add_edge(START, "retrieve")
workflow.add_edge("retrieve", "grade_documents")
workflow.add_edge("generate", END)
```

**条件边（Conditional Edge）**：
```python
# 根据路由函数返回值动态决定下一个节点
workflow.add_conditional_edges(
    "grade_documents",
    decide_to_generate,  # 路由函数
    {
        "web_search": "web_search",
        "generate": "generate"
    }
)
```

**循环边（Loop Edge）**：
```python
# 形成循环，支持迭代优化
workflow.add_edge("transform_query", "retrieve")
```

### 2. 路由函数设计

路由函数必须：
- 接收状态作为参数
- 返回 `Literal` 类型的节点名称
- 返回值必须在 `path_map` 中定义

```python
def decide_to_generate(state: RAGState) -> Literal["web_search", "generate"]:
    if state["web_search_needed"]:
        return "web_search"
    return "generate"
```

### 3. 工作流路径

本示例支持三种执行路径：

1. **直接生成路径**（文档相关）：
   ```
   retrieve -> grade_documents -> generate
   ```

2. **网络搜索路径**（文档不相关）：
   ```
   retrieve -> grade_documents -> web_search -> generate
   ```

3. **查询改写路径**（需要优化查询）：
   ```
   retrieve -> grade_documents -> web_search -> transform_query -> retrieve -> ...
   ```

### 4. 状态管理

状态在节点间传递，每个节点返回部分状态更新：

```python
def retrieve(state: RAGState) -> dict:
    # 只返回需要更新的字段
    return {"documents": docs, "retry_count": 0}
```

## 边的选择建议

| 场景 | 推荐边类型 | 原因 |
|------|-----------|------|
| 固定流程 | 普通边 | 简单直接，性能最好 |
| 二分支决策 | 条件边 | 清晰的决策逻辑 |
| 多分支路由 | 条件边 | 支持复杂的路由规则 |
| 迭代优化 | 循环边 | 支持重试和改进 |
| 并行执行 | Send + 条件边 | 动态并行处理 |

## 常见问题

### Q1: 条件边的路由函数可以是异步的吗？

A: 不可以。路由函数必须是同步函数，因为它只负责决策，不执行实际操作。

### Q2: 如何避免无限循环？

A: 在状态中维护计数器，在路由函数中检查：

```python
def decide_after_search(state: RAGState) -> Literal["transform_query", "generate"]:
    if state.get("retry_count", 0) >= 2:
        return "generate"  # 强制退出
    return "transform_query"
```

### Q3: 可以在运行时动态修改边吗？

A: 不可以。边在编译时确定，运行时不可修改。但可以通过条件边实现动态路由。

## 总结

本示例展示了 LangGraph 中三种边的类型在 RAG 工作流中的应用：

1. **普通边**：构建固定的流程骨架
2. **条件边**：实现智能的决策路由
3. **循环边**：支持迭代优化和重试

通过合理组合这些边类型，可以构建出灵活、健壮的 RAG 应用。
