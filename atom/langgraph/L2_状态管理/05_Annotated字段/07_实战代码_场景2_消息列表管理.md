# 实战代码 - 场景2：消息列表管理

## 场景概述

本场景演示如何使用 `add_messages` 函数管理对话历史，这是 LangGraph 中处理消息列表的标准方式。

**核心目标**：
- 掌握 add_messages 的工作原理
- 理解按 ID 更新消息机制
- 处理 Token 限制问题
- 实现对话历史管理

**适用场景**：
- 聊天机器人
- 对话式 AI Agent
- 多轮对话系统
- 上下文管理

[来源: reference/context7_langgraph_01.md | LangGraph 官方文档]

## 原理讲解

### 1. add_messages 函数机制

```python
from langgraph.graph.message import add_messages

# 函数签名
def add_messages(
    old_messages: list[BaseMessage],
    new_messages: list[BaseMessage]
) -> list[BaseMessage]:
    """
    合并消息列表，按 ID 更新现有消息
    """
```

**核心特性**：
1. **按 ID 更新**：如果新消息的 ID 与旧消息相同，则覆盖
2. **追加新消息**：如果 ID 不存在，则追加到列表末尾
3. **保持顺序**：维护消息的时间顺序

[来源: reference/context7_langgraph_01.md | LangGraph 官方文档]

### 2. 与 operator.add 的区别

| 特性 | operator.add | add_messages |
|------|--------------|--------------|
| **合并方式** | 简单拼接 | 按 ID 智能合并 |
| **重复处理** | 允许重复 | 自动去重 |
| **更新支持** | 不支持 | 支持按 ID 更新 |
| **适用场景** | 普通列表 | 消息列表 |

```python
# operator.add 示例
[msg1, msg2] + [msg3] = [msg1, msg2, msg3]

# add_messages 示例
[msg(id=1), msg(id=2)] + [msg(id=1, updated)] = [msg(id=1, updated), msg(id=2)]
```

[来源: reference/search_annotated_github_01.md]

### 3. 消息 ID 机制

```python
from langchain_core.messages import HumanMessage, AIMessage

# 自动生成 ID
msg1 = HumanMessage(content="Hello")
print(msg1.id)  # 自动生成的 UUID

# 手动指定 ID
msg2 = HumanMessage(content="Hello", id="custom-id-1")
print(msg2.id)  # "custom-id-1"
```

**ID 生成规则**：
- 未指定 ID：自动生成 UUID
- 指定 ID：使用指定的 ID
- ID 相同：视为同一条消息

[来源: reference/context7_langgraph_01.md | LangGraph 官方文档]

## 完整代码示例

### 示例 1：基础消息管理

```python
"""
基础消息列表管理
演示 add_messages 的基本用法
"""

from typing_extensions import TypedDict, Annotated
from langchain_core.messages import HumanMessage, AIMessage, BaseMessage
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages


# 1. 定义状态
class ConversationState(TypedDict):
    """对话状态"""
    messages: Annotated[list[BaseMessage], add_messages]
    user_id: str


# 2. 定义节点
def user_input(state: ConversationState) -> dict:
    """用户输入节点"""
    return {
        "messages": [HumanMessage(content="你好，我是用户")]
    }


def ai_response(state: ConversationState) -> dict:
    """AI 响应节点"""
    # 获取最后一条用户消息
    last_message = state["messages"][-1]

    return {
        "messages": [AIMessage(content=f"你好！我收到了你的消息：{last_message.content}")]
    }


def user_followup(state: ConversationState) -> dict:
    """用户追问节点"""
    return {
        "messages": [HumanMessage(content="能帮我解释一下 LangGraph 吗？")]
    }


def ai_explain(state: ConversationState) -> dict:
    """AI 解释节点"""
    return {
        "messages": [AIMessage(content="LangGraph 是一个用于构建状态化 AI 应用的框架")]
    }


# 3. 构建图
def create_conversation_graph():
    """创建对话图"""
    builder = StateGraph(ConversationState)

    builder.add_node("user_input", user_input)
    builder.add_node("ai_response", ai_response)
    builder.add_node("user_followup", user_followup)
    builder.add_node("ai_explain", ai_explain)

    builder.add_edge(START, "user_input")
    builder.add_edge("user_input", "ai_response")
    builder.add_edge("ai_response", "user_followup")
    builder.add_edge("user_followup", "ai_explain")
    builder.add_edge("ai_explain", END)

    return builder.compile()


# 4. 运行
def main():
    """主函数"""
    graph = create_conversation_graph()

    result = graph.invoke({
        "messages": [],
        "user_id": "user-123"
    })

    print("=" * 60)
    print("对话历史")
    print("=" * 60)
    for i, msg in enumerate(result["messages"], 1):
        role = "用户" if isinstance(msg, HumanMessage) else "AI"
        print(f"{i}. [{role}] {msg.content}")


if __name__ == "__main__":
    main()
```

**运行输出**：

```
============================================================
对话历史
============================================================
1. [用户] 你好，我是用户
2. [AI] 你好！我收到了你的消息：你好，我是用户
3. [用户] 能帮我解释一下 LangGraph 吗？
4. [AI] LangGraph 是一个用于构建状态化 AI 应用的框架
```

### 示例 2：按 ID 更新消息

```python
"""
按 ID 更新消息
演示如何覆盖现有消息
"""

from typing_extensions import TypedDict, Annotated
from langchain_core.messages import HumanMessage, AIMessage, BaseMessage
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages


class State(TypedDict):
    """状态定义"""
    messages: Annotated[list[BaseMessage], add_messages]


def add_initial_message(state: State) -> dict:
    """添加初始消息"""
    return {
        "messages": [
            HumanMessage(content="初始消息", id="msg-1"),
            AIMessage(content="初始响应", id="msg-2")
        ]
    }


def update_first_message(state: State) -> dict:
    """更新第一条消息"""
    return {
        "messages": [
            HumanMessage(content="更新后的消息", id="msg-1")  # 相同 ID，会覆盖
        ]
    }


def add_new_message(state: State) -> dict:
    """添加新消息"""
    return {
        "messages": [
            HumanMessage(content="新消息", id="msg-3")  # 新 ID，会追加
        ]
    }


def create_graph():
    """创建图"""
    builder = StateGraph(State)

    builder.add_node("initial", add_initial_message)
    builder.add_node("update", update_first_message)
    builder.add_node("add_new", add_new_message)

    builder.add_edge(START, "initial")
    builder.add_edge("initial", "update")
    builder.add_edge("update", "add_new")
    builder.add_edge("add_new", END)

    return builder.compile()


def main():
    """主函数"""
    graph = create_graph()

    result = graph.invoke({"messages": []})

    print("=" * 60)
    print("消息更新演示")
    print("=" * 60)
    print(f"\n最终消息数量: {len(result['messages'])}")
    print("\n消息列表:")
    for msg in result["messages"]:
        role = "Human" if isinstance(msg, HumanMessage) else "AI"
        print(f"  [{role}] (ID: {msg.id}) {msg.content}")


if __name__ == "__main__":
    main()
```

**运行输出**：

```
============================================================
消息更新演示
============================================================

最终消息数量: 3

消息列表:
  [Human] (ID: msg-1) 更新后的消息
  [AI] (ID: msg-2) 初始响应
  [Human] (ID: msg-3) 新消息
```

**关键观察**：
- `msg-1` 被更新为"更新后的消息"
- `msg-2` 保持不变
- `msg-3` 作为新消息追加

### 示例 3：Token 限制处理

```python
"""
Token 限制处理
演示如何管理消息列表大小
"""

from typing_extensions import TypedDict, Annotated
from langchain_core.messages import HumanMessage, AIMessage, BaseMessage, trim_messages
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages


class State(TypedDict):
    """状态定义"""
    messages: Annotated[list[BaseMessage], add_messages]


def add_message_batch_1(state: State) -> dict:
    """添加第一批消息"""
    return {
        "messages": [
            HumanMessage(content=f"消息 {i}") for i in range(1, 6)
        ]
    }


def add_message_batch_2(state: State) -> dict:
    """添加第二批消息"""
    return {
        "messages": [
            AIMessage(content=f"响应 {i}") for i in range(1, 6)
        ]
    }


def trim_old_messages(state: State) -> dict:
    """修剪旧消息，保留最近的 5 条"""
    messages = state["messages"]

    # 保留最近的 5 条消息
    if len(messages) > 5:
        trimmed = messages[-5:]
    else:
        trimmed = messages

    print(f"[trim_old_messages] 原始消息数: {len(messages)}, 修剪后: {len(trimmed)}")

    # 返回修剪后的消息列表
    # 注意：这里需要返回完整的列表，而不是增量
    return {"messages": trimmed}


def create_graph():
    """创建图"""
    builder = StateGraph(State)

    builder.add_node("batch1", add_message_batch_1)
    builder.add_node("batch2", add_message_batch_2)
    builder.add_node("trim", trim_old_messages)

    builder.add_edge(START, "batch1")
    builder.add_edge("batch1", "batch2")
    builder.add_edge("batch2", "trim")
    builder.add_edge("trim", END)

    return builder.compile()


def main():
    """主函数"""
    graph = create_graph()

    result = graph.invoke({"messages": []})

    print("\n" + "=" * 60)
    print("Token 限制处理结果")
    print("=" * 60)
    print(f"\n最终消息数量: {len(result['messages'])}")
    print("\n保留的消息:")
    for i, msg in enumerate(result["messages"], 1):
        role = "Human" if isinstance(msg, HumanMessage) else "AI"
        print(f"  {i}. [{role}] {msg.content}")


if __name__ == "__main__":
    main()
```

**运行输出**：

```

============================================================
Token 限制处理结果
============================================================

最终消息数量: 5

保留的消息:
  1. [Human] 消息 4
  2. [Human] 消息 5
  3. [AI] 响应 3
  4. [AI] 响应 4
  5. [AI] 响应 5
```

### 示例 4：自定义消息 Reducer

```python
"""
自定义消息 Reducer
实现更复杂的消息管理逻辑
"""

from typing_extensions import TypedDict, Annotated
from langchain_core.messages import HumanMessage, AIMessage, BaseMessage
from langgraph.graph import StateGraph, START, END


def custom_message_reducer(
    old_messages: list[BaseMessage],
    new_messages: list[BaseMessage]
) -> list[BaseMessage]:
    """
    自定义消息 Reducer
    - 按 ID 更新现有消息
    - 限制最大消息数为 10
    - 保留系统消息
    """
    # 创建消息字典（按 ID 索引）
    message_dict = {msg.id: msg for msg in old_messages if msg.id}

    # 更新或添加新消息
    for msg in new_messages:
        if msg.id:
            message_dict[msg.id] = msg
        else:
            # 生成临时 ID
            import uuid
            msg.id = str(uuid.uuid4())
            message_dict[msg.id] = msg

    # 转换回列表
    result = list(message_dict.values())

    # 限制最大数量（保留最近的 10 条）
    if len(result) > 10:
        result = result[-10:]

    return result


class State(TypedDict):
    """状态定义"""
    messages: Annotated[list[BaseMessage], custom_message_reducer]


def add_messages(state: State) -> dict:
    """添加消息"""
    return {
        "messages": [
            HumanMessage(content=f"消息 {i}", id=f"msg-{i}")
            for i in range(1, 13)
        ]
    }


def create_graph():
    """创建图"""
    builder = StateGraph(State)
    builder.add_node("add", add_messages)
    builder.add_edge(START, "add")
    builder.add_edge("add", END)
    return builder.compile()


def main():
    """主函数"""
    graph = create_graph()

    result = graph.invoke({"messages": []})

    print("=" * 60)
    print("自定义 Reducer 结果")
    print("=" * 60)
    print(f"\n最终消息数量: {len(result['messages'])} (限制为 10)")
    print("\n消息列表:")
    for msg in result["messages"]:
        print(f"  (ID: {msg.id}) {msg.content}")


if __name__ == "__main__":
    main()
```

### 场景 1：聊天机器人

```python
"""
聊天机器人实现
完整的对话管理系统
"""

from typing_extensions import TypedDict, Annotated
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage, BaseMessage
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages


class ChatState(TypedDict):
    """聊天状态"""
    messages: Annotated[list[BaseMessage], add_messages]
    user_name: str
    session_id: str


def initialize_chat(state: ChatState) -> dict:
    """初始化聊天"""
    return {
        "messages": [
            SystemMessage(content=f"你是一个友好的助手，正在与 {state['user_name']} 对话")
        ]
    }


def process_user_message(state: ChatState) -> dict:
    """处理用户消息"""
    # 模拟用户输入
    return {
        "messages": [
            HumanMessage(content="你好！")
        ]
    }


def generate_ai_response(state: ChatState) -> dict:
    """生成 AI 响应"""
    last_message = state["messages"][-1]

    # 简单的响应逻辑
    if "你好" in last_message.content:
        response = f"你好，{state['user_name']}！很高兴见到你。"
    else:
        response = "我明白了，还有什么我可以帮助你的吗？"

    return {
        "messages": [AIMessage(content=response)]
    }


def create_chatbot():
    """创建聊天机器人"""
    builder = StateGraph(ChatState)

    builder.add_node("init", initialize_chat)
    builder.add_node("user", process_user_message)
    builder.add_node("ai", generate_ai_response)

    builder.add_edge(START, "init")
    builder.add_edge("init", "user")
    builder.add_edge("user", "ai")
    builder.add_edge("ai", END)

    return builder.compile()


def main():
    """主函数"""
    chatbot = create_chatbot()

    result = chatbot.invoke({
        "messages": [],
        "user_name": "张三",
        "session_id": "session-123"
    })

    print("=" * 60)
    print("聊天机器人对话")
    print("=" * 60)
    for msg in result["messages"]:
        if isinstance(msg, SystemMessage):
            print(f"[系统] {msg.content}")
        elif isinstance(msg, HumanMessage):
            print(f"[用户] {msg.content}")
        elif isinstance(msg, AIMessage):
            print(f"[AI] {msg.content}")


if __name__ == "__main__":
    main()
```

## 常见陷阱与解决方案

### 陷阱 1：忘记使用 add_messages

```python
# ❌ 错误：使用 operator.add
class State(TypedDict):
    messages: Annotated[list[BaseMessage], operator.add]
    # 问题：无法按 ID 更新，会产生重复消息

# ✅ 正确：使用 add_messages
class State(TypedDict):
    messages: Annotated[list[BaseMessage], add_messages]
```

[来源: reference/context7_langgraph_01.md | LangGraph 官方文档]

---

### 陷阱 2：Token 限制未处理

```python
# ❌ 问题：消息列表无限增长
class State(TypedDict):
    messages: Annotated[list[BaseMessage], add_messages]
    # 问题：可能超过 LLM 的 token 限制

# ✅ 解决方案：定期修剪消息
def trim_messages_reducer(old: list, new: list) -> list:
    from langgraph.graph.message import add_messages
    merged = add_messages(old, new)
    return merged[-10:]  # 只保留最近 10 条

class State(TypedDict):
    messages: Annotated[list[BaseMessage], trim_messages_reducer]
```

## 性能优化建议

### 1. 使用消息 ID

```python
# ✅ 好：使用 ID
msg = HumanMessage(content="Hello", id="msg-1")

# ❌ 不好：不使用 ID（无法更新）
msg = HumanMessage(content="Hello")
```

### 2. 定期清理历史

```python
def should_trim(messages: list) -> bool:
    """判断是否需要修剪"""
    return len(messages) > 50

def trim_strategy(messages: list) -> list:
    """修剪策略：保留最近 30 条"""
    return messages[-30:]
```

### 3. 分离不同类型的消息

```python
class State(TypedDict):
    # 分离用户消息和系统消息
    user_messages: Annotated[list[BaseMessage], add_messages]
    system_logs: Annotated[list[str], operator.add]
```
