## 一、5分钟快速上手

### 1.1 最基础的用法

```python
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

# 1. 创建模板
template = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    ("human", "{input}")
])

# 2. 创建链
model = ChatOpenAI(model="gpt-4o-mini")
chain = template | model

# 3. 使用
response = chain.invoke({"input": "What is Python?"})
print(response.content)
```

**核心要点**:
- `from_messages()` 创建模板
- 元组格式 `(role, content)` 最简洁
- `|` 操作符链接组件
- `invoke()` 执行调用

## 二、三大核心模式

### 模式1：基础问答（80%场景）

```python
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

# 创建问答模板
qa_template = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant. Answer concisely."),
    ("human", "{question}")
])

# 使用
model = ChatOpenAI()
chain = qa_template | model

result = chain.invoke({"question": "What is machine learning?"})
print(result.content)
```

**适用场景**: 简单问答、信息查询、知识问答

---

### 模式2：带对话历史（对话系统）

```python
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_openai import ChatOpenAI

# 创建带历史的模板
chat_template = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    MessagesPlaceholder("history"),  # 对话历史占位符
    ("human", "{input}")
])

model = ChatOpenAI()
chain = chat_template | model

# 第一轮对话
history = []
response1 = chain.invoke({
    "history": history,
    "input": "My name is Alice."
})
print(f"AI: {response1.content}")

# 更新历史
history.extend([
    ("human", "My name is Alice."),
    ("ai", response1.content)
])

# 第二轮对话（AI会记住名字）
response2 = chain.invoke({
    "history": history,
    "input": "What's my name?"
})
print(f"AI: {response2.content}")  # "Your name is Alice."
```

**适用场景**: 聊天机器人、客服系统、多轮对话

**关键点**:
- `MessagesPlaceholder("history")` 插入历史
- 手动维护历史列表
- 元组格式 `("human", "...")` 自动转换

### 模式3：RAG 问答（检索增强）

```python
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

# RAG 模板
rag_template = ChatPromptTemplate.from_messages([
    ("system", "Answer based on the context. If unsure, say so."),
    ("human", "Context: {context}"),
    ("human", "Question: {question}")
])

model = ChatOpenAI()
chain = rag_template | model

# 使用（假设已有检索结果）
context = "Python is a high-level programming language created by Guido van Rossum."
result = chain.invoke({
    "context": context,
    "question": "Who created Python?"
})
print(result.content)  # "Guido van Rossum"
```

**适用场景**: 文档问答、知识库检索、RAG 系统

**关键点**:
- 系统消息定义规则
- 上下文单独传递
- 明确告知 AI 基于上下文回答

## 三、常用技巧速查

### 技巧1：预填充固定变量

```python
from langchain_core.prompts import ChatPromptTemplate

# 创建通用模板
base_template = ChatPromptTemplate.from_messages([
    ("system", "You are a {role} assistant."),
    ("human", "{input}")
])

# 预填充角色
python_assistant = base_template.partial(role="Python programming")
data_assistant = base_template.partial(role="data analysis")

# 使用时只需传入 input
python_assistant.invoke({"input": "How to use list comprehension?"})
data_assistant.invoke({"input": "How to analyze time series?"})
```

**使用场景**: 创建专门的助手、减少重复传参

---

### 技巧2：可选历史记录

```python
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

# optional=True 让历史可选
template = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    MessagesPlaceholder("history", optional=True),  # 可选
    ("human", "{input}")
])

# 可以不传 history
template.invoke({"input": "Hello"})

# 也可以传 history
template.invoke({
    "history": [("human", "Hi"), ("ai", "Hello!")],
    "input": "How are you?"
})
```

**使用场景**: 同时支持单轮和多轮对话

---

### 技巧3：限制历史消息数量

```python
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

# n_messages=5 只保留最后5条
template = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    MessagesPlaceholder("history", n_messages=5),  # 限制数量
    ("human", "{input}")
])

# 即使传入10条历史，也只使用最后5条
long_history = [("human", f"Q{i}"), ("ai", f"A{i}") for i in range(10)]
template.invoke({"history": long_history, "input": "Current question"})
```

**使用场景**: 避免上下文窗口溢出、控制成本

## 四、实战代码模板

### 模板1：简单聊天机器人

```python
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_openai import ChatOpenAI

class SimpleChatbot:
    """简单聊天机器人"""

    def __init__(self):
        self.model = ChatOpenAI(model="gpt-4o-mini")
        self.template = ChatPromptTemplate.from_messages([
            ("system", "You are a friendly chatbot."),
            MessagesPlaceholder("history"),
            ("human", "{input}")
        ])
        self.chain = self.template | self.model
        self.history = []

    def chat(self, user_input: str) -> str:
        """发送消息并获取回复"""
        # 调用模型
        response = self.chain.invoke({
            "history": self.history,
            "input": user_input
        })

        # 更新历史
        self.history.extend([
            ("human", user_input),
            ("ai", response.content)
        ])

        return response.content

# 使用
bot = SimpleChatbot()
print(bot.chat("Hello!"))
print(bot.chat("What's the weather like?"))
```

### 模板2：RAG 问答系统

```python
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

class SimpleRAG:
    """简单 RAG 系统"""

    def __init__(self):
        self.model = ChatOpenAI(model="gpt-4o-mini")
        self.template = ChatPromptTemplate.from_messages([
            ("system", """Answer based on the context.
Rules:
1. Only use information from context
2. If unsure, say "I don't know"
3. Be concise"""),
            ("human", "Context: {context}"),
            ("human", "Question: {question}")
        ])
        self.chain = self.template | self.model

    def query(self, question: str, context: str) -> str:
        """查询 RAG 系统"""
        response = self.chain.invoke({
            "context": context,
            "question": question
        })
        return response.content

# 使用
rag = SimpleRAG()

# 假设已有检索结果
context = "LangChain is a framework for building LLM applications."
answer = rag.query("What is LangChain?", context)
print(answer)
```

### 模板3：对话式 RAG

```python
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_openai import ChatOpenAI

class ConversationalRAG:
    """对话式 RAG 系统"""

    def __init__(self):
        self.model = ChatOpenAI(model="gpt-4o-mini")
        self.template = ChatPromptTemplate.from_messages([
            ("system", "Answer based on context. Use conversation history for context."),
            ("human", "Context: {context}"),
            MessagesPlaceholder("history", optional=True),
            ("human", "{question}")
        ])
        self.chain = self.template | self.model
        self.history = []

    def query(self, question: str, context: str) -> str:
        """查询并保持对话历史"""
        response = self.chain.invoke({
            "context": context,
            "history": self.history,
            "question": question
        })

        # 更新历史
        self.history.extend([
            ("human", question),
            ("ai", response.content)
        ])

        return response.content

    def clear_history(self):
        """清空历史"""
        self.history = []

# 使用
rag = ConversationalRAG()

# 第一轮
answer1 = rag.query(
    "What is Python?",
    "Python is a programming language."
)
print(answer1)

# 第二轮（会记住之前的对话）
answer2 = rag.query(
    "What are its features?",
    "Python is easy to learn and powerful."
)
print(answer2)
```

## 八、错误处理模板

```python
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

def safe_chat(template, model, **kwargs):
    """安全的聊天函数"""
    try:
        # 检查必需变量
        required = set(template.input_variables)
        provided = set(kwargs.keys())

        missing = required - provided
        if missing:
            return f"Error: Missing variables: {missing}"

        # 执行调用
        chain = template | model
        response = chain.invoke(kwargs)
        return response.content

    except Exception as e:
        return f"Error: {str(e)}"

# 使用
template = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    ("human", "{input}")
])

model = ChatOpenAI()
result = safe_chat(template, model, input="Hello")
print(result)
```
