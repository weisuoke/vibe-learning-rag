## 示例1:最简单的集成

### 代码

```python
"""
最简单的Retriever集成示例
演示:创建VectorStore -> 转换为Retriever -> 检索文档
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv

# 加载环境变量
load_dotenv()

# 1. 创建VectorStore
print("步骤1:创建VectorStore...")
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())

# 2. 添加文档
print("步骤2:添加文档...")
texts = [
    "LangChain is a framework for developing applications powered by LLMs.",
    "LCEL is LangChain Expression Language for building chains.",
    "RAG is Retrieval-Augmented Generation.",
    "Vector stores are used to store and retrieve embeddings.",
    "Retrievers provide a unified interface for document retrieval."
]
vectorstore.add_texts(texts)
print(f"已添加 {len(texts)} 个文档")

# 3. 转换为Retriever
print("\n步骤3:转换为Retriever...")
retriever = vectorstore.as_retriever()
print(f"Retriever类型: {type(retriever).__name__}")

# 4. 检索文档
print("\n步骤4:检索文档...")
query = "What is LangChain?"
docs = retriever.invoke(query)

print(f"\n查询: {query}")
print(f"返回文档数: {len(docs)}")
for i, doc in enumerate(docs, 1):
    print(f"\n文档{i}:")
    print(f"  内容: {doc.page_content}")
    print(f"  元数据: {doc.metadata}")
```

## 示例2:配置检索参数

### 代码

```python
"""
配置检索参数示例
演示:调整k值、添加元数据、运行时覆盖参数
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_core.documents import Document
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv

load_dotenv()

# 创建VectorStore
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())

# 添加带元数据的文档
docs = [
    Document(
        page_content="LangChain is a framework for LLM applications",
        metadata={"source": "docs", "category": "framework"}
    ),
    Document(
        page_content="LCEL is LangChain Expression Language",
        metadata={"source": "docs", "category": "language"}
    ),
    Document(
        page_content="RAG is Retrieval-Augmented Generation",
        metadata={"source": "blog", "category": "technique"}
    ),
    Document(
        page_content="Vector stores store embeddings",
        metadata={"source": "docs", "category": "storage"}
    ),
]
vectorstore.add_documents(docs)

# 配置1:默认参数(k=4)
print("配置1:默认参数")
retriever = vectorstore.as_retriever()
results = retriever.invoke("LangChain")
print(f"返回文档数: {len(results)}")

# 配置2:指定k值
print("\n配置2:指定k=2")
retriever = vectorstore.as_retriever(search_kwargs={"k": 2})
results = retriever.invoke("LangChain")
print(f"返回文档数: {len(results)}")
for doc in results:
    print(f"  - {doc.page_content[:50]}...")

# 配置3:运行时覆盖k值
print("\n配置3:运行时覆盖k=3")
results = retriever.invoke("LangChain", k=3)
print(f"返回文档数: {len(results)}")

# 配置4:元数据过滤
print("\n配置4:元数据过滤(source=docs)")
retriever = vectorstore.as_retriever(
    search_kwargs={
        "k": 10,
        "filter": {"source": "docs"}
    }
)
results = retriever.invoke("LangChain")
print(f"返回文档数: {len(results)}")
for doc in results:
    print(f"  - {doc.page_content[:40]}... [source={doc.metadata['source']}]")
```

## 示例3:同步与异步调用

### 代码

```python
"""
同步与异步调用示例
演示:invoke vs ainvoke, batch vs abatch
"""

import asyncio
from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import time

load_dotenv()

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
texts = [
    "LangChain is a framework",
    "LCEL is an expression language",
    "RAG is a technique",
    "Vector stores store embeddings"
]
vectorstore.add_texts(texts)
retriever = vectorstore.as_retriever(search_kwargs={"k": 2})

# 1. 同步调用
print("1. 同步调用 (invoke)")
start = time.time()
docs = retriever.invoke("LangChain")
sync_time = time.time() - start
print(f"   返回文档数: {len(docs)}")
print(f"   耗时: {sync_time:.3f}s")

# 2. 异步调用
print("\n2. 异步调用 (ainvoke)")
async def async_retrieve():
    start = time.time()
    docs = await retriever.ainvoke("LangChain")
    async_time = time.time() - start
    print(f"   返回文档数: {len(docs)}")
    print(f"   耗时: {async_time:.3f}s")

asyncio.run(async_retrieve())

# 3. 批量同步调用
print("\n3. 批量同步调用 (batch)")
queries = ["LangChain", "LCEL", "RAG"]
start = time.time()
batch_results = retriever.batch(queries)
batch_time = time.time() - start
print(f"   查询数: {len(queries)}")
print(f"   总耗时: {batch_time:.3f}s")
print(f"   平均耗时: {batch_time/len(queries):.3f}s")

# 4. 批量异步调用
print("\n4. 批量异步调用 (abatch)")
async def async_batch():
    start = time.time()
    batch_results = await retriever.abatch(queries)
    abatch_time = time.time() - start
    print(f"   查询数: {len(queries)}")
    print(f"   总耗时: {abatch_time:.3f}s")
    print(f"   平均耗时: {abatch_time/len(queries):.3f}s")

asyncio.run(async_batch())

# 5. 并发异步调用
print("\n5. 并发异步调用 (asyncio.gather)")
async def concurrent_retrieve():
    start = time.time()
    tasks = [retriever.ainvoke(q) for q in queries]
    results = await asyncio.gather(*tasks)
    concurrent_time = time.time() - start
    print(f"   查询数: {len(queries)}")
    print(f"   总耗时: {concurrent_time:.3f}s")
    print(f"   平均耗时: {concurrent_time/len(queries):.3f}s")

asyncio.run(concurrent_retrieve())
```

### 运行结果

```
1. 同步调用 (invoke)
   返回文档数: 2
   耗时: 0.156s

2. 异步调用 (ainvoke)
   返回文档数: 2
   耗时: 0.162s

3. 批量同步调用 (batch)
   查询数: 3
   总耗时: 0.478s
   平均耗时: 0.159s

4. 批量异步调用 (abatch)
   查询数: 3
   总耗时: 0.195s
   平均耗时: 0.065s

5. 并发异步调用 (asyncio.gather)
   查询数: 3
   总耗时: 0.188s
   平均耗时: 0.063s
```

**观察**:
- 单次调用:同步和异步速度相近
- 批量调用:异步显著更快(3倍+)
- 并发调用:最快的方式

## 示例4:错误处理

### 代码

```python
"""
错误处理示例
演示:捕获异常、降级策略、重试机制
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import logging

load_dotenv()

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
texts = ["LangChain is a framework", "LCEL is a language"]
vectorstore.add_texts(texts)
retriever = vectorstore.as_retriever()

# 1. 基础错误处理
def safe_retrieve(query: str):
    """带错误处理的检索"""
    try:
        docs = retriever.invoke(query)
        logger.info(f"检索成功: {len(docs)} 个文档")
        return docs
    except Exception as e:
        logger.error(f"检索失败: {e}")
        return []

print("1. 基础错误处理")
docs = safe_retrieve("LangChain")
print(f"   返回文档数: {len(docs)}")

# 2. 降级策略
def retrieve_with_fallback(query: str):
    """带降级策略的检索"""
    try:
        # 尝试主检索器
        docs = retriever.invoke(query)
        if len(docs) > 0:
            logger.info("主检索器成功")
            return docs
    except Exception as e:
        logger.warning(f"主检索器失败: {e}")

    # 降级:返回默认文档
    logger.info("使用降级策略")
    from langchain_core.documents import Document
    return [Document(page_content="默认文档:无法检索到相关内容")]

print("\n2. 降级策略")
docs = retrieve_with_fallback("LangChain")
print(f"   返回文档数: {len(docs)}")

# 3. 重试机制
def retrieve_with_retry(query: str, max_retries: int = 3):
    """带重试机制的检索"""
    import time

    for attempt in range(max_retries):
        try:
            docs = retriever.invoke(query)
            logger.info(f"检索成功 (尝试 {attempt + 1}/{max_retries})")
            return docs
        except Exception as e:
            logger.warning(f"尝试 {attempt + 1}/{max_retries} 失败: {e}")
            if attempt < max_retries - 1:
                wait_time = 2 ** attempt  # 指数退避
                logger.info(f"等待 {wait_time}s 后重试...")
                time.sleep(wait_time)
            else:
                logger.error("所有重试都失败")
                return []

print("\n3. 重试机制")
docs = retrieve_with_retry("LangChain")
print(f"   返回文档数: {len(docs)}")

# 4. 完整的生产级错误处理
class SafeRetriever:
    """生产级安全检索器"""

    def __init__(self, retriever, max_retries=3, fallback_docs=None):
        self.retriever = retriever
        self.max_retries = max_retries
        self.fallback_docs = fallback_docs or []

    def retrieve(self, query: str):
        """带完整错误处理的检索"""
        # 尝试重试
        for attempt in range(self.max_retries):
            try:
                docs = self.retriever.invoke(query)
                if len(docs) > 0:
                    logger.info(f"检索成功: {len(docs)} 个文档")
                    return docs
                else:
                    logger.warning("检索返回空结果")
            except Exception as e:
                logger.error(f"检索失败 (尝试 {attempt + 1}/{self.max_retries}): {e}")
                if attempt < self.max_retries - 1:
                    import time
                    time.sleep(2 ** attempt)

        # 所有重试失败,使用降级策略
        logger.warning("使用降级策略")
        return self.fallback_docs

print("\n4. 生产级错误处理")
from langchain_core.documents import Document
safe_retriever = SafeRetriever(
    retriever,
    fallback_docs=[Document(page_content="降级文档")]
)
docs = safe_retriever.retrieve("LangChain")
print(f"   返回文档数: {len(docs)}")
```

```python
"""
回调和追踪示例
演示:使用回调追踪检索过程
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from langchain_core.callbacks import BaseCallbackHandler
from dotenv import load_dotenv

load_dotenv()

# 自定义回调处理器
class CustomRetrieverCallback(BaseCallbackHandler):
    """自定义检索回调"""

    def on_retriever_start(self, serialized, query, **kwargs):
        print(f"[START] 开始检索: {query}")

    def on_retriever_end(self, documents, **kwargs):
        print(f"[END] 检索完成: {len(documents)} 个文档")
        for i, doc in enumerate(documents, 1):
            print(f"  文档{i}: {doc.page_content[:50]}...")

    def on_retriever_error(self, error, **kwargs):
        print(f"[ERROR] 检索失败: {error}")

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
texts = [
    "LangChain is a framework",
    "LCEL is a language",
    "RAG is a technique"
]
vectorstore.add_texts(texts)
retriever = vectorstore.as_retriever(search_kwargs={"k": 2})

# 使用回调
print("使用自定义回调:")
docs = retriever.invoke(
    "LangChain",
    config={"callbacks": [CustomRetrieverCallback()]}
)

# 使用标准输出回调
print("\n使用标准输出回调:")
from langchain_core.callbacks import StdOutCallbackHandler
docs = retriever.invoke(
    "LCEL",
    config={"callbacks": [StdOutCallbackHandler()]}
)
```

## 完整示例:端到端集成

### 代码

```python
"""
端到端集成示例
演示:从文档加载到检索的完整流程
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_core.documents import Document
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import logging

load_dotenv()
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def main():
    """端到端检索流程"""

    # 1. 准备文档
    logger.info("步骤1: 准备文档")
    documents = [
        Document(
            page_content="LangChain is a framework for developing applications powered by LLMs.",
            metadata={"source": "docs", "page": 1}
        ),
        Document(
            page_content="LCEL is LangChain Expression Language for building chains.",
            metadata={"source": "docs", "page": 2}
        ),
        Document(
            page_content="RAG is Retrieval-Augmented Generation technique.",
            metadata={"source": "blog", "page": 1}
        ),
        Document(
            page_content="Vector stores are used to store and retrieve embeddings.",
            metadata={"source": "docs", "page": 3}
        ),
    ]
    logger.info(f"准备了 {len(documents)} 个文档")

    # 2. 创建VectorStore
    logger.info("步骤2: 创建VectorStore")
    vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
    vectorstore.add_documents(documents)
    logger.info("文档已添加到VectorStore")

    # 3. 创建Retriever
    logger.info("步骤3: 创建Retriever")
    retriever = vectorstore.as_retriever(
        search_kwargs={"k": 3},
        tags=["production", "v1.0"]
    )
    logger.info("Retriever已创建")

    # 4. 执行检索
    logger.info("步骤4: 执行检索")
    queries = [
        "What is LangChain?",
        "Tell me about LCEL",
        "What is RAG?"
    ]

    for query in queries:
        logger.info(f"\n查询: {query}")
        docs = retriever.invoke(query)
        logger.info(f"返回 {len(docs)} 个文档:")
        for i, doc in enumerate(docs, 1):
            logger.info(f"  {i}. {doc.page_content[:60]}...")
            logger.info(f"     元数据: {doc.metadata}")

    # 5. 批量检索
    logger.info("\n步骤5: 批量检索")
    batch_results = retriever.batch(queries)
    logger.info(f"批量检索完成: {len(batch_results)} 个结果")
    for i, (query, docs) in enumerate(zip(queries, batch_results), 1):
        logger.info(f"  查询{i}: {query} -> {len(docs)} 个文档")

if __name__ == "__main__":
    main()
```
