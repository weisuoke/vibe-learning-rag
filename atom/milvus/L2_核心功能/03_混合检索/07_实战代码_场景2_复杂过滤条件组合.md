# 实战代码 - 场景2：复杂过滤条件组合

> **目标**：掌握混合检索中的复杂标量过滤，实现多条件组合查询

---

## 场景描述

构建一个电商产品搜索系统，支持：
- **向量检索**：语义搜索产品
- **标量过滤**：价格范围、品牌、类别、标签等多维度过滤
- **复杂条件**：AND/OR/NOT 组合、JSON 字段、Array 字段

**技术栈**：
- pymilvus 2.5+
- 布尔表达式
- JSON/Array 过滤

## Schema 设计

```python
from pymilvus import (
    MilvusClient,
    DataType,
    Function,
    FunctionType
)

client = MilvusClient(uri="http://localhost:19530", token="root:Milvus")

# 创建 Schema
schema = client.create_schema(auto_id=False)

# 基础字段
schema.add_field(field_name="id", datatype=DataType.INT64, is_primary=True)
schema.add_field(field_name="name", datatype=DataType.VARCHAR, max_length=500)
schema.add_field(field_name="description", datatype=DataType.VARCHAR, 
                 max_length=2000, enable_analyzer=True)

# 标量字段（用于过滤）
schema.add_field(field_name="price", datatype=DataType.FLOAT)
schema.add_field(field_name="brand", datatype=DataType.VARCHAR, max_length=100)
schema.add_field(field_name="category", datatype=DataType.VARCHAR, max_length=100)
schema.add_field(field_name="stock", datatype=DataType.INT64)
schema.add_field(field_name="rating", datatype=DataType.FLOAT)

# JSON 字段（存储复杂属性）
schema.add_field(field_name="attributes", datatype=DataType.JSON)

# Array 字段（存储标签）
schema.add_field(field_name="tags", datatype=DataType.ARRAY, 
                 element_type=DataType.VARCHAR, max_length=50, max_capacity=20)

# 向量字段
schema.add_field(field_name="text_dense", datatype=DataType.FLOAT_VECTOR, dim=384)
schema.add_field(field_name="text_sparse", datatype=DataType.SPARSE_FLOAT_VECTOR)

# BM25 函数
bm25_function = Function(
    name="text_bm25_emb",
    input_field_names=["description"],
    output_field_names=["text_sparse"],
    function_type=FunctionType.BM25,
)
schema.add_function(bm25_function)
```

## 索引配置

```python
index_params = client.prepare_index_params()

# 向量索引
index_params.add_index(field_name="text_dense", index_type="AUTOINDEX", metric_type="IP")
index_params.add_index(field_name="text_sparse", index_type="SPARSE_INVERTED_INDEX",
                       metric_type="BM25")

# 标量索引（提升过滤性能）
index_params.add_index(field_name="price", index_type="STL_SORT")
index_params.add_index(field_name="brand", index_type="INVERTED")
index_params.add_index(field_name="category", index_type="INVERTED")
index_params.add_index(field_name="rating", index_type="STL_SORT")

# 创建 Collection
client.create_collection(
    collection_name="ecommerce_products",
    schema=schema,
    index_params=index_params
)
```

## 场景1：基础标量过滤

### 价格范围过滤

```python
from pymilvus import AnnSearchRequest, RRFRanker

query_text = "专业笔记本电脑"
query_dense = model.encode([query_text])[0].tolist()

# 创建检索请求（带价格过滤）
request_dense = AnnSearchRequest(
    data=[query_dense],
    anns_field="text_dense",
    param={"nprobe": 10},
    expr="price >= 10000 AND price <= 20000",  # 价格范围
    limit=5
)

request_sparse = AnnSearchRequest(
    data=[query_text],
    anns_field="text_sparse",
    expr="price >= 10000 AND price <= 20000",
    limit=5
)

ranker = RRFRanker(k=60)

results = client.hybrid_search(
    collection_name="ecommerce_products",
    reqs=[request_dense, request_sparse],
    ranker=ranker,
    limit=3,
    output_fields=["name", "price", "brand"]
)

print("\n🔍 价格范围过滤 (10000-20000)：")
for hits in results:
    for hit in hits:
        print(f"  {hit['entity']['name']} - ¥{hit['entity']['price']} ({hit['entity']['brand']})")
```

**输出**：
```
🔍 价格范围过滤 (10000-20000)：
  MacBook Pro 16 - ¥19999.0 (Apple)
  ThinkPad X1 Carbon - ¥12999.0 (联想)
```

## 过滤表达式参考

### 比较运算符

```python
# 等于
"price == 1000"

# 不等于
"brand != 'Apple'"

# 大于/小于
"price > 5000"
"rating < 4.5"

# 大于等于/小于等于
"stock >= 100"
"price <= 10000"
```

### 逻辑运算符

```python
# AND
"price > 5000 AND brand == 'Apple'"

# OR
"brand == 'Apple' OR brand == '小米'"

# NOT
"NOT (brand == 'Apple')"

# 组合
"(brand == 'Apple' OR brand == '小米') AND price < 10000"
```

### IN 运算符

```python
# 单值 IN
"brand in ['Apple', '小米', '华为']"

# 数字 IN
"price in [1999, 2999, 3999]"
```

### JSON 字段

```python
# 访问 JSON 字段
"attributes['color'] == '黑色'"

# JSON 布尔值
"attributes['5G'] == true"

# JSON_CONTAINS
"JSON_CONTAINS(attributes, 'color')"

# JSON_CONTAINS_ALL
"JSON_CONTAINS_ALL(attributes, ['color', 'storage'])"
```

### Array 字段

```python
# ARRAY_CONTAINS
"ARRAY_CONTAINS(tags, '旗舰')"

# ARRAY_CONTAINS_ALL
"ARRAY_CONTAINS_ALL(tags, ['旗舰', '5G'])"

# ARRAY_CONTAINS_ANY
"ARRAY_CONTAINS_ANY(tags, ['旗舰', '拍照'])"

# ARRAY_LENGTH
"ARRAY_LENGTH(tags) > 2"
```

## 性能优化建议

### 1. 为常用过滤字段创建索引

```python
# 标量索引类型
index_params.add_index(field_name="price", index_type="STL_SORT")      # 数值范围查询
index_params.add_index(field_name="brand", index_type="INVERTED")      # 字符串精确匹配
index_params.add_index(field_name="category", index_type="INVERTED")   # 字符串精确匹配
```

### 2. 过滤条件顺序优化

```python
# ❌ 低效：先过滤低选择性字段
"category == '手机' AND brand == 'Apple'"

# ✅ 高效：先过滤高选择性字段
"brand == 'Apple' AND category == '手机'"
```

### 3. 使用 IN 代替多个 OR

```python
# ❌ 低效
"brand == 'Apple' OR brand == '小米' OR brand == '华为'"

# ✅ 高效
"brand in ['Apple', '小米', '华为']"
```

### 4. 避免复杂的 NOT 条件

```python
# ❌ 低效
"NOT (brand == 'Apple' OR brand == '小米')"

# ✅ 高效：使用 IN + 反向逻辑
"brand in ['华为', '联想', 'OPPO']"
```

## 完整可运行代码

```python
"""
复杂过滤条件组合示例
功能：混合检索 + 多维度标量过滤
"""

from pymilvus import (
    MilvusClient, DataType, Function, FunctionType,
    AnnSearchRequest, RRFRanker
)
from sentence_transformers import SentenceTransformer

def main():
    # 1. 连接并创建 Collection
    client = MilvusClient(uri="http://localhost:19530", token="root:Milvus")
    
    if client.has_collection("ecommerce_products"):
        client.drop_collection("ecommerce_products")
    
    # 2. 创建 Schema（省略，见上文）
    # 3. 插入数据（省略，见上文）
    
    # 4. 执行复杂过滤查询
    model = SentenceTransformer('sentence-transformers/all-MiniLM-L6-v2')
    query_text = "专业笔记本电脑"
    query_dense = model.encode([query_text])[0].tolist()
    
    # 示例：复杂嵌套条件
    complex_expr = '(brand == "Apple" OR brand == "小米") AND price < 10000 AND rating >= 4.7'
    
    request_dense = AnnSearchRequest(
        data=[query_dense],
        anns_field="text_dense",
        expr=complex_expr,
        limit=5
    )
    
    request_sparse = AnnSearchRequest(
        data=[query_text],
        anns_field="text_sparse",
        expr=complex_expr,
        limit=5
    )
    
    ranker = RRFRanker(k=60)
    
    results = client.hybrid_search(
        collection_name="ecommerce_products",
        reqs=[request_dense, request_sparse],
        ranker=ranker,
        limit=3,
        output_fields=["name", "brand", "price", "rating"]
    )
    
    print(f"\n🔍 查询: {query_text}")
    print(f"过滤条件: {complex_expr}\n")
    
    for hits in results:
        for i, hit in enumerate(hits):
            print(f"Top {i+1}: {hit['entity']['name']}")
            print(f"  品牌: {hit['entity']['brand']}, 价格: ¥{hit['entity']['price']}, 评分: {hit['entity']['rating']}")

if __name__ == "__main__":
    main()
```
