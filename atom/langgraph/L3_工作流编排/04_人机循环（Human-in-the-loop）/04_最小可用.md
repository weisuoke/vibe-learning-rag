## 4.1 三行代码模式：interrupt() + Command(resume=) + checkpointer

### 概念说明

**人机循环的最小实现只需要三个要素**：

| 要素 | 作用 | 代码 |
|------|------|------|
| `interrupt()` | 在节点内暂停执行 | `answer = interrupt("你的问题")` |
| `Command(resume=)` | 恢复执行并传递人类决策 | `graph.invoke(Command(resume="回答"), config)` |
| `checkpointer` | 保存中断时的状态 | `graph = builder.compile(checkpointer=MemorySaver())` |

缺少任何一个，人机循环都无法工作。

### 代码示例

```python
"""
最小人机循环：三行代码模式
"""

from typing import TypedDict
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command  # 要素 1 & 2


class State(TypedDict):
    question: str
    answer: str


def ask_human(state: State):
    # 要素 1：interrupt() —— 暂停并向客户端传递信息
    human_answer = interrupt(f"请回答: {state['question']}")
    return {"answer": human_answer}


# 构建图
builder = StateGraph(State)
builder.add_node("ask", ask_human)
builder.add_edge(START, "ask")
builder.add_edge("ask", END)

# 要素 3：checkpointer —— 保存中断状态
graph = builder.compile(checkpointer=MemorySaver())

# 执行
config = {"configurable": {"thread_id": "demo-1"}}

# 第一次调用：遇到 interrupt，暂停
result = graph.invoke({"question": "你好吗？", "answer": ""}, config)
print("图已暂停，等待人类输入...")

# 要素 2：Command(resume=) —— 恢复执行
result = graph.invoke(Command(resume="我很好！"), config)
print(f"结果: {result['answer']}")  # "我很好！"
```

## 4.2 基本流程：创建图 → compile → invoke → 检查中断 → 恢复

### 概念说明

人机循环的完整生命周期分为 5 步：

```
步骤 1: 创建图（定义节点和边）
    ↓
步骤 2: compile(checkpointer=MemorySaver())
    ↓
步骤 3: graph.invoke(input, config) → 遇到 interrupt 暂停
    ↓
步骤 4: graph.get_state(config) → 检查中断信息
    ↓
步骤 5: graph.invoke(Command(resume=value), config) → 恢复执行
```

### 代码示例

```python
"""
完整生命周期演示：5 步流程
"""

from typing import TypedDict, Optional
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command


# ========== 步骤 1：创建图 ==========

class ReviewState(TypedDict):
    content: str
    feedback: Optional[str]
    status: str


def review_node(state: ReviewState):
    """审核节点：展示内容并等待反馈"""
    feedback = interrupt({
        "type": "review_request",
        "content": state["content"],
        "instruction": "请审核以下内容，输入反馈意见或 'ok' 表示通过",
    })

    if feedback == "ok":
        return {"feedback": "通过", "status": "approved"}
    else:
        return {"feedback": feedback, "status": "needs_revision"}


builder = StateGraph(ReviewState)
builder.add_node("review", review_node)
builder.add_edge(START, "review")
builder.add_edge("review", END)


# ========== 步骤 2：编译（必须提供 checkpointer） ==========

checkpointer = MemorySaver()
graph = builder.compile(checkpointer=checkpointer)


# ========== 步骤 3：首次调用（触发中断） ==========

config = {"configurable": {"thread_id": "review-001"}}
result = graph.invoke(
    {"content": "LangGraph 是一个强大的框架", "feedback": None, "status": "pending"},
    config,
)
print("图已暂停")


# ========== 步骤 4：检查中断状态 ==========

snapshot = graph.get_state(config)

# 查看下一步要执行的节点
print(f"等待节点: {snapshot.next}")
# 输出: ('review',)

# 查看中断详情
for task in snapshot.tasks:
    for intr in task.interrupts:
        print(f"中断信息: {intr.value}")
        # 输出: {"type": "review_request", "content": "LangGraph 是...", ...}


# ========== 步骤 5：恢复执行 ==========

result = graph.invoke(Command(resume="ok"), config)
print(f"状态: {result['status']}")    # "approved"
print(f"反馈: {result['feedback']}")  # "通过"
```

## 4.3 thread_id：每个对话/任务一个唯一 ID

### 概念说明

`thread_id` 是人机循环的"会话标识"，它决定了：
- 中断状态保存在哪里
- 恢复时从哪个状态继续

**核心规则**：
- 同一个交互流程必须使用相同的 `thread_id`
- 不同的交互流程使用不同的 `thread_id`
- `thread_id` 可以是任意字符串（UUID、用户ID、订单号等）

### 代码示例

```python
"""
thread_id 使用演示：多用户并发
"""

import uuid
from typing import TypedDict
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command


class State(TypedDict):
    user: str
    answer: str


def ask_node(state: State):
    answer = interrupt(f"{state['user']}，你喜欢什么颜色？")
    return {"answer": answer}


builder = StateGraph(State)
builder.add_node("ask", ask_node)
builder.add_edge(START, "ask")
builder.add_edge("ask", END)

graph = builder.compile(checkpointer=MemorySaver())


# ===== 多用户并发：每个用户一个 thread_id =====

# 用户 A 的会话
config_a = {"configurable": {"thread_id": f"user-alice-{uuid.uuid4().hex[:8]}"}}
graph.invoke({"user": "Alice", "answer": ""}, config_a)

# 用户 B 的会话（完全独立）
config_b = {"configurable": {"thread_id": f"user-bob-{uuid.uuid4().hex[:8]}"}}
graph.invoke({"user": "Bob", "answer": ""}, config_b)

# 分别恢复（互不影响）
result_a = graph.invoke(Command(resume="蓝色"), config_a)
result_b = graph.invoke(Command(resume="红色"), config_b)

print(f"Alice: {result_a['answer']}")  # "蓝色"
print(f"Bob: {result_b['answer']}")    # "红色"


# ===== 常见错误：thread_id 不匹配 =====

# ❌ 用不同的 thread_id 恢复 → 找不到中断状态
# wrong_config = {"configurable": {"thread_id": "wrong-id"}}
# graph.invoke(Command(resume="绿色"), wrong_config)  # 报错！

# ✅ 用相同的 thread_id 恢复
# graph.invoke(Command(resume="绿色"), config_a)  # 正确
```

## 4.4 状态检查：graph.get_state(config) 查看中断信息

### 概念说明

`graph.get_state(config)` 返回一个 `StateSnapshot` 对象，包含当前图的完整状态：

```python
# [来源: sourcecode/langgraph/libs/langgraph/langgraph/types.py:268]
class StateSnapshot(NamedTuple):
    values: dict          # 当前状态值
    next: tuple[str, ...]  # 下一步要执行的节点
    tasks: tuple           # 任务列表（包含中断信息）
    # ... 其他字段
```

**关键字段**：
- `snapshot.next` — 如果非空，说明图还没执行完（有中断）
- `snapshot.tasks[i].interrupts` — 具体的中断信息列表
- `snapshot.values` — 当前的状态数据

### 代码示例

```python
"""
状态检查：获取中断详情
"""

from typing import TypedDict
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command


class State(TypedDict):
    data: str
    result: str


def process_node(state: State):
    decision = interrupt({
        "question": "如何处理这条数据？",
        "data_preview": state["data"][:100],
        "options": ["approve", "reject", "edit"],
    })
    return {"result": f"决策: {decision}"}


builder = StateGraph(State)
builder.add_node("process", process_node)
builder.add_edge(START, "process")
builder.add_edge("process", END)

graph = builder.compile(checkpointer=MemorySaver())
config = {"configurable": {"thread_id": "check-demo"}}

# 触发中断
graph.invoke({"data": "重要数据内容...", "result": ""}, config)


# ===== 检查中断状态 =====

snapshot = graph.get_state(config)

# 1. 检查是否有待处理的中断
has_interrupt = len(snapshot.next) > 0
print(f"是否有中断: {has_interrupt}")  # True

# 2. 查看等待执行的节点
print(f"等待节点: {snapshot.next}")  # ('process',)

# 3. 查看中断详情
for task in snapshot.tasks:
    print(f"任务节点: {task.name}")
    for intr in task.interrupts:
        print(f"  中断 ID: {intr.id}")
        print(f"  中断值: {intr.value}")
        # 输出:
        #   中断 ID: a1b2c3d4...
        #   中断值: {"question": "如何处理这条数据？", ...}

# 4. 查看当前状态值
print(f"当前状态: {snapshot.values}")

# 5. 根据中断信息做决策，然后恢复
result = graph.invoke(Command(resume="approve"), config)
print(f"结果: {result['result']}")  # "决策: approve"
```

## 4.5 条件路由：Command(goto="node_name") 根据人类决策路由

### 概念说明

`Command` 不仅可以传递 resume 值，还可以通过 `goto` 参数控制图的执行路径：

```python
# [来源: sourcecode/langgraph/libs/langgraph/langgraph/types.py:368]
Command(
    resume=value,           # 恢复值
    goto="target_node",     # 跳转到指定节点
    update={"key": "val"},  # 同时更新状态
)
```

这让人类可以在中断时决定"接下来走哪条路"。

### 代码示例

```python
"""
条件路由：根据人类决策选择不同的执行路径
"""

from typing import TypedDict, Literal, Optional
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command


class OrderState(TypedDict):
    order_id: str
    amount: float
    status: str
    reason: Optional[str]


def review_order(state: OrderState) -> Command[Literal["fulfill", "refund", "escalate"]]:
    """审核订单：人类决定处理方式"""

    decision = interrupt({
        "type": "order_review",
        "order_id": state["order_id"],
        "amount": state["amount"],
        "options": {
            "approve": "批准订单，进入发货流程",
            "refund": "退款处理",
            "escalate": "升级到主管审批",
        },
    })

    # 根据人类决策路由到不同节点
    if decision == "approve":
        return Command(goto="fulfill", update={"status": "approved"})
    elif decision == "refund":
        return Command(goto="refund", update={"status": "refunding"})
    else:
        return Command(goto="escalate", update={"status": "escalated"})


def fulfill_order(state: OrderState):
    print(f"订单 {state['order_id']} 已发货")
    return {"status": "fulfilled"}


def refund_order(state: OrderState):
    print(f"订单 {state['order_id']} 已退款 ¥{state['amount']}")
    return {"status": "refunded"}


def escalate_order(state: OrderState):
    print(f"订单 {state['order_id']} 已升级审批")
    return {"status": "escalated_pending"}


# 构建图
builder = StateGraph(OrderState)
builder.add_node("review", review_order)
builder.add_node("fulfill", fulfill_order)
builder.add_node("refund", refund_order)
builder.add_node("escalate", escalate_order)

builder.add_edge(START, "review")
builder.add_edge("fulfill", END)
builder.add_edge("refund", END)
builder.add_edge("escalate", END)

graph = builder.compile(checkpointer=MemorySaver())
config = {"configurable": {"thread_id": "order-001"}}

# 第一次调用：触发审核中断
graph.invoke(
    {"order_id": "ORD-2024-001", "amount": 299.0, "status": "pending", "reason": None},
    config,
)

# 人类决策：批准订单
result = graph.invoke(Command(resume="approve"), config)
print(f"最终状态: {result['status']}")  # "fulfilled"
```

**运行输出**：
```
订单 ORD-2024-001 已发货
最终状态: fulfilled
```

## 总结：这些知识足以

掌握以上 5 个核心知识点，你就能：

1. **实现基础人机交互**
   - 在任意节点暂停执行，等待人类输入
   - 收集用户反馈、审批决策、表单数据

2. **构建审批工作流**
   - 根据人类决策路由到不同处理分支
   - 实现 approve/reject/escalate 等多路径流程

3. **管理多用户并发**
   - 每个用户/任务使用独立的 thread_id
   - 多个中断互不干扰

4. **调试和监控中断状态**
   - 使用 get_state() 查看当前中断信息
   - 了解图的执行进度和等待状态

5. **处理多步交互**
   - 单节点内多个 interrupt 按顺序匹配
   - 实现多步表单、多轮确认等场景

### 条件路由

```python
def node(state) -> Command[Literal["a", "b"]]:
    decision = interrupt("选择路径")
    if decision == "a":
        return Command(goto="a")

```

### 常用恢复方式

```python
# 同步恢复
graph.invoke(Command(resume=value), config)

# 异步恢复

# 流式恢复
for chunk in graph.stream(Command(resume=value), config):
    print(chunk)
```
