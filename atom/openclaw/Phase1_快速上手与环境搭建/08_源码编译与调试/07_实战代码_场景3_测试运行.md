### 1.2 单元测试配置

**文件**：`vitest.unit.config.ts`

```typescript
import { defineConfig } from 'vitest/config'

export default defineConfig({
  test: {
    name: 'unit',
    include: ['src/**/*.test.ts'],
    exclude: [
      '**/node_modules/**',
      '**/dist/**',
      '**/*.e2e.test.ts',
      '**/*.live.test.ts'
    ],
    pool: 'threads',
    poolOptions: {
      threads: {
        singleThread: false,
        isolate: true
      }
    },
    coverage: {
      provider: 'v8',
      reporter: ['text', 'json', 'html'],
      include: ['src/**/*.ts'],
      exclude: [
        '**/*.test.ts',
        '**/*.d.ts',
        '**/types/**'
      ]
    }
  }
})
```

**关键配置**：
- `pool: 'threads'`：使用线程池（快速）
- `isolate: true`：每个测试文件隔离
- `coverage.provider: 'v8'`：使用 V8 覆盖率

### 1.3 E2E 测试配置

**文件**：`vitest.e2e.config.ts`

```typescript
import { defineConfig } from 'vitest/config'

export default defineConfig({
  test: {
    name: 'e2e',
    include: ['**/*.e2e.test.ts'],
    pool: 'vmForks',
    poolOptions: {
      vmForks: {
        singleFork: false,
        isolate: true
      }

    testTimeout: 30000,
    hookTimeout: 30000
  }

```

**关键配置**：
- `pool: 'vmForks'`：完全隔离（避免状态污染）
- `testTimeout: 30000`：E2E 测试超时 30 秒

## 步骤 3：运行 E2E 测试

### 3.1 E2E 测试示例

**文件**：`src/gateway/routes.e2e.test.ts`

```typescript
import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { startGateway, stopGateway } from '../gateway'

describe('Gateway E2E Tests', () => {
  let gatewayUrl: string

  beforeAll(async () => {
    // 启动 Gateway
    const gateway = await startGateway({ port: 18790 })
    gatewayUrl = `http://localhost:18790`
  })

  afterAll(async () => {
    // 停止 Gateway
    await stopGateway()
  })

  it('should return health status', async () => {
    const response = await fetch(`${gatewayUrl}/health`)
    const data = await response.json()

    expect(response.status).toBe(200)
    expect(data.status).toBe('ok')
  })

  it('should process message', async () => {
    const response = await fetch(`${gatewayUrl}/api/message`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ text: 'Hello OpenClaw' })
    })

    const data = await response.json()

    expect(response.status).toBe(200)
    expect(data.response).toBeDefined()
  })
})
```

## 步骤 5：测试编写实战

### 5.1 基础单元测试

**测试文件**：`src/core/config.test.ts`

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest'
import { loadConfig, saveConfig, validateConfig } from './config'
import fs from 'fs'

describe('Config Management', () => {
  const testConfigPath = '/tmp/test-config.json'

  beforeEach(() => {
    // 测试前准备
    if (fs.existsSync(testConfigPath)) {
      fs.unlinkSync(testConfigPath)
    }
  })

  afterEach(() => {
    // 测试后清理
    if (fs.existsSync(testConfigPath)) {
      fs.unlinkSync(testConfigPath)
    }
  })

  it('should load default config', () => {
    const config = loadConfig()

    expect(config).toBeDefined()
    expect(config.provider).toBe('anthropic')
    expect(config.model).toBe('claude-opus-4')
  })

  it('should save and load config', () => {
    const config = {
      provider: 'openai',
      model: 'gpt-4',
      apiKey: 'test-key'
    }

    saveConfig(testConfigPath, config)
    const loaded = loadConfig(testConfigPath)

    expect(loaded).toEqual(config)
  })

  it('should validate config schema', () => {
    const validConfig = {
      provider: 'anthropic',
      model: 'claude-opus-4'
    }

    expect(() => validateConfig(validConfig)).not.toThrow()

    const invalidConfig = {
      provider: 'invalid'
    }

    expect(() => validateConfig(invalidConfig)).toThrow()
  })
})
```

### 5.2 异步测试

**测试文件**：`src/core/session.test.ts`

```typescript
import { describe, it, expect } from 'vitest'
import { createSession, loadSession, deleteSession } from './session'

describe('Session Management', () => {
  it('should create and load session', async () => {
    // 创建 session
    const session = await createSession('user_123')

    expect(session.id).toBeDefined()
    expect(session.userId).toBe('user_123')
    expect(session.messages).toEqual([])

    // 加载 session
    const loaded = await loadSession(session.id)

    expect(loaded).toEqual(session)

    // 清理
    await deleteSession(session.id)
  })

  it('should handle concurrent sessions', async () => {
    // 并发创建多个 session
    const sessions = await Promise.all([
      createSession('user_1'),
      createSession('user_2'),
      createSession('user_3')
    ])

    expect(sessions).toHaveLength(3)
    expect(new Set(sessions.map(s => s.id)).size).toBe(3)

    // 清理
    await Promise.all(sessions.map(s => deleteSession(s.id)))
  })
})
```

```typescript
import { describe, it, expect, vi } from 'vitest'
import { processMessage } from './agent'
import * as anthropic from '@anthropic-ai/sdk'

// Mock Anthropic SDK
vi.mock('@anthropic-ai/sdk', () => ({
  default: vi.fn(() => ({
    messages: {
      create: vi.fn().mockResolvedValue({
        content: [{ text: 'Mocked response' }]
      })
    }
  }))
}))

describe('Agent Processing', () => {
  it('should process message with LLM', async () => {
    const response = await processMessage('Hello')

    expect(response).toBe('Mocked response')
  })

  it('should handle LLM errors', async () => {
    // Mock 错误
    const mockCreate = vi.fn().mockRejectedValue(new Error('API Error'))
    vi.mocked(anthropic.default).mockReturnValue({
      messages: { create: mockCreate }
    } as any)

    await expect(processMessage('Hello')).rejects.toThrow('API Error')
  })
})
```

## 故障排查

### 问题 1：测试超时

**症状**：
```
Error: Test timed out in 5000ms
```

**解决方案**：
```typescript

    testTimeout: 30000
  }
```

---

### 问题 2：测试失败但本地通过

**症状**：
```
CI: ✗ test failed
Local: ✓ test passed
```

**原因**：
- 环境变量不同
- 时区差异
- 文件系统差异

**解决方案**：
```typescript
// 使用固定时区
process.env.TZ = 'UTC'

// 使用临时目录
import os from 'os'
const tmpDir = os.tmpdir()

// Mock 时间
vi.setSystemTime(new Date('2024-01-01'))
```

## 核心洞察

### 1. 测试配置很重要

**表面**：一个配置文件就够了

**实际**：
- 不同测试类型需要不同配置
- Pool 模式影响隔离和性能
- 超时设置需要根据测试类型调整

---

### 2. 覆盖率不是越高越好

**表面**：100% 覆盖率最好

**实际**：
- 80-90% 覆盖率通常足够
- 关注关键路径覆盖
- 避免为了覆盖率而写无意义测试

---

### 3. Mock 要谨慎使用

**表面**：Mock 所有外部依赖

**实际**：
- 过度 Mock 导致测试脆弱
- 集成测试比单元测试更有价值
- Mock 应该模拟真实行为
