# 实战代码 - 场景1：基础边使用

> 本文档展示 LangGraph 中普通边和序列边的基础用法，通过完整可运行的示例帮助你快速掌握边的使用

---

## 场景说明

本示例将构建一个简单的文档处理工作流，演示：

1. **普通边（Normal Edge）**：使用 `add_edge()` 连接节点
2. **序列边（Sequence Edge）**：使用 `add_sequence()` 简化线性流程
3. **实际应用**：文档处理、数据转换、结果输出

**工作流场景**：
- 加载文档 → 文本清洗 → 分词处理 → 统计分析 → 输出结果

---

## 完整代码示例

```python
"""
LangGraph 基础边使用实战
演示：普通边和序列边在文档处理工作流中的应用
"""

from typing import TypedDict, List
from langgraph.graph import StateGraph, START, END

# ===== 1. 定义状态结构 =====
class DocumentState(TypedDict):
    """文档处理状态"""
    raw_text: str              # 原始文本
    cleaned_text: str          # 清洗后的文本
    tokens: List[str]          # 分词结果
    word_count: int            # 词数统计
    char_count: int            # 字符数统计
    processing_steps: List[str]  # 处理步骤记录


# ===== 2. 定义节点函数 =====

def load_document(state: DocumentState) -> dict:
    """
    节点1：加载文档
    模拟从文件或数据库加载文档内容
    """
    print("\n[节点1] 加载文档...")

    # 模拟加载的文档内容
    raw_text = """
    Hello World! This is a sample document.
    It contains multiple sentences.
    Some have   extra   spaces.
    """

    print(f"  ✓ 加载完成，文本长度: {len(raw_text)} 字符")

    return {
        "raw_text": raw_text,
        "processing_steps": ["文档加载"]
    }


def clean_text(state: DocumentState) -> dict:
    """
    节点2：文本清洗
    去除多余空格、换行符等
    """
    print("\n[节点2] 清洗文本...")

    raw_text = state["raw_text"]

    # 清洗操作
    cleaned = raw_text.strip()                    # 去除首尾空白
    cleaned = " ".join(cleaned.split())           # 规范化空格
    cleaned = cleaned.replace("\n", " ")          # 去除换行

    print(f"  ✓ 清洗完成")
    print(f"    原始: {len(raw_text)} 字符")
    print(f"    清洗后: {len(cleaned)} 字符")

    steps = state.get("processing_steps", [])
    steps.append("文本清洗")

    return {
        "cleaned_text": cleaned,
        "processing_steps": steps
    }


def tokenize(state: DocumentState) -> dict:
    """
    节点3：分词处理
    将文本分割成单词列表
    """
    print("\n[节点3] 分词处理...")

    cleaned_text = state["cleaned_text"]

    # 简单分词（按空格分割）
    tokens = cleaned_text.split()

    print(f"  ✓ 分词完成，共 {len(tokens)} 个词")
    print(f"    前5个词: {tokens[:5]}")

    steps = state.get("processing_steps", [])
    steps.append("分词处理")

    return {
        "tokens": tokens,
        "processing_steps": steps
    }


def analyze_statistics(state: DocumentState) -> dict:
    """
    节点4：统计分析
    计算词数、字符数等统计信息
    """
    print("\n[节点4] 统计分析...")

    tokens = state["tokens"]
    cleaned_text = state["cleaned_text"]

    word_count = len(tokens)
    char_count = len(cleaned_text)

    print(f"  ✓ 统计完成")
    print(f"    词数: {word_count}")
    print(f"    字符数: {char_count}")

    steps = state.get("processing_steps", [])
    steps.append("统计分析")

    return {
        "word_count": word_count,
        "char_count": char_count,
        "processing_steps": steps
    }


def output_result(state: DocumentState) -> dict:
    """
    节点5：输出结果
    打印最终处理结果
    """
    print("\n[节点5] 输出结果...")
    print("=" * 60)
    print("文档处理完成！")
    print("=" * 60)
    print(f"原始文本: {state['raw_text'][:50]}...")
    print(f"清洗后文本: {state['cleaned_text'][:50]}...")
    print(f"词数: {state['word_count']}")
    print(f"字符数: {state['char_count']}")
    print(f"处理步骤: {' → '.join(state['processing_steps'])}")
    print("=" * 60)

    return {}


# ===== 3. 示例1：使用普通边构建工作流 =====

def example1_normal_edges():
    """
    示例1：使用普通边（add_edge）构建工作流
    特点：显式定义每一条边，清晰明确
    """
    print("\n" + "=" * 60)
    print("示例1：使用普通边构建工作流")
    print("=" * 60)

    # 创建状态图
    builder = StateGraph(DocumentState)

    # 添加节点
    builder.add_node("load", load_document)
    builder.add_node("clean", clean_text)
    builder.add_node("tokenize", tokenize)
    builder.add_node("analyze", analyze_statistics)
    builder.add_node("output", output_result)

    # 使用普通边连接节点（显式定义每一条边）
    builder.add_edge(START, "load")        # 起点 → 加载文档
    builder.add_edge("load", "clean")      # 加载文档 → 清洗文本
    builder.add_edge("clean", "tokenize")  # 清洗文本 → 分词处理
    builder.add_edge("tokenize", "analyze")  # 分词处理 → 统计分析
    builder.add_edge("analyze", "output")  # 统计分析 → 输出结果
    builder.add_edge("output", END)        # 输出结果 → 终点

    # 编译图
    graph = builder.compile()

    # 执行工作流
    print("\n开始执行工作流...")
    result = graph.invoke({})

    print("\n✓ 工作流执行完成")
    return result


# ===== 4. 示例2：使用序列边简化工作流 =====

def example2_sequence_edges():
    """
    示例2：使用序列边（add_sequence）简化工作流
    特点：自动添加节点和边，代码更简洁
    """
    print("\n" + "=" * 60)
    print("示例2：使用序列边简化工作流")
    print("=" * 60)

    # 创建状态图
    builder = StateGraph(DocumentState)

    # 使用序列边：自动添加节点和边
    # 等价于：add_node + add_edge 的组合
    builder.add_sequence([
        load_document,      # 节点1
        clean_text,         # 节点2
        tokenize,           # 节点3
        analyze_statistics, # 节点4
        output_result       # 节点5
    ])

    # 编译图
    graph = builder.compile()

    # 执行工作流
    print("\n开始执行工作流...")
    result = graph.invoke({})

    print("\n✓ 工作流执行完成")
    return result


# ===== 5. 示例3：混合使用普通边和序列边 =====

def preprocess(state: DocumentState) -> dict:
    """预处理节点：加载 + 清洗"""
    print("\n[预处理] 加载并清洗文档...")

    # 加载
    raw_text = "Hello World! This is a test.  "
    # 清洗
    cleaned = raw_text.strip().replace("  ", " ")

    print(f"  ✓ 预处理完成")

    return {
        "raw_text": raw_text,
        "cleaned_text": cleaned,
        "processing_steps": ["预处理"]
    }


def postprocess(state: DocumentState) -> dict:
    """后处理节点：格式化输出"""
    print("\n[后处理] 格式化输出...")

    print(f"  ✓ 最终结果: {state['word_count']} 词, {state['char_count']} 字符")

    steps = state.get("processing_steps", [])
    steps.append("后处理")

    return {"processing_steps": steps}


def example3_mixed_edges():
    """
    示例3：混合使用普通边和序列边
    特点：灵活组合，适应复杂场景
    """
    print("\n" + "=" * 60)
    print("示例3：混合使用普通边和序列边")
    print("=" * 60)

    # 创建状态图
    builder = StateGraph(DocumentState)

    # 添加预处理节点（使用普通边）
    builder.add_node("preprocess", preprocess)
    builder.add_edge(START, "preprocess")

    # 添加核心处理流程（使用序列边）
    builder.add_node("tokenize", tokenize)
    builder.add_node("analyze", analyze_statistics)
    builder.add_edge("preprocess", "tokenize")
    builder.add_edge("tokenize", "analyze")

    # 添加后处理节点（使用普通边）
    builder.add_node("postprocess", postprocess)
    builder.add_edge("analyze", "postprocess")
    builder.add_edge("postprocess", END)

    # 编译图
    graph = builder.compile()

    # 执行工作流
    print("\n开始执行工作流...")
    result = graph.invoke({})

    print("\n✓ 工作流执行完成")
    return result


# ===== 6. 主函数 =====

def main():
    """主函数：运行所有示例"""
    print("\n" + "=" * 60)
    print("LangGraph 基础边使用实战")
    print("=" * 60)

    # 示例1：普通边
    print("\n\n")
    example1_normal_edges()

    # 示例2：序列边
    print("\n\n")
    example2_sequence_edges()

    # 示例3：混合使用
    print("\n\n")
    example3_mixed_edges()

    print("\n\n" + "=" * 60)
    print("所有示例执行完成！")
    print("=" * 60)


if __name__ == "__main__":
    main()
```

## 关键要点总结

### 1. 普通边（Normal Edge）

**API**：`builder.add_edge(source, target)`

**特点**：
- 显式定义每一条边
- 清晰明确，易于理解
- 适合复杂的非线性流程

**使用场景**：
- 需要精确控制流程走向
- 有分支或合并的复杂图结构
- 需要连接到 START 或 END 节点

**示例**：
```python
builder.add_edge(START, "node1")
builder.add_edge("node1", "node2")
builder.add_edge("node2", END)
```

### 2. 序列边（Sequence Edge）

**API**：`builder.add_sequence([node1, node2, node3, ...])`

**特点**：
- 自动添加节点和边
- 代码简洁，减少重复
- 适合线性流程

**使用场景**：
- 简单的顺序执行流程
- 没有分支的线性处理
- 快速原型开发

**等价转换**：
```python
# 使用序列边
builder.add_sequence([func1, func2, func3])

# 等价于
builder.add_node("func1", func1)
builder.add_node("func2", func2)
builder.add_node("func3", func3)
builder.add_edge(START, "func1")
builder.add_edge("func1", "func2")
builder.add_edge("func2", "func3")
builder.add_edge("func3", END)
```

### 3. 选择建议

| 场景 | 推荐方式 | 原因 |
|------|----------|------|
| 简单线性流程 | 序列边 | 代码简洁，易于维护 |
| 复杂分支流程 | 普通边 | 灵活控制，清晰明确 |
| 需要连接 START/END | 普通边 | 序列边会自动处理 |
| 快速原型开发 | 序列边 | 快速搭建，减少代码 |
| 生产环境 | 混合使用 | 根据实际需求灵活选择 |

## 常见问题

### Q1：什么时候用普通边，什么时候用序列边？

**A**：
- **序列边**：适合简单的线性流程，代码更简洁
- **普通边**：适合需要精确控制的复杂流程，更灵活

### Q2：序列边可以和普通边混用吗？

**A**：可以！示例3展示了混合使用的方式。

### Q3：序列边会自动连接 START 和 END 吗？

**A**：是的，`add_sequence()` 会自动：
- 将第一个节点连接到 START
- 将最后一个节点连接到 END
- 自动添加中间的所有边

### Q4：如何查看图的结构？

**A**：使用 `graph.get_graph().draw_mermaid()` 可以生成 Mermaid 图：

```python
graph = builder.compile()
print(graph.get_graph().draw_mermaid())
```
