## 策略1:MaxBatch优化

### 1.1 Provider批量限制

不同Provider的MaxBatch差异显著:

| Provider | MaxBatch | 推荐批量大小 | 备注 |
|----------|----------|-------------|------|
| OpenAI | 128 | 64-128 | 可配置batchFactor |
| VoyageAI | 128 | 64-128 | 最快 |
| Cohere | 96 | 48-96 | 平衡性能 |
| VertexAI | 128 | 64-128 | GCP优化 |
| Zilliz | 64 | 32-64 | 集成方案 |
| SiliconFlow | 32 | 16-32 | 亚洲优化 |
| DashScope | 6-25 | 6 | v3模型限制 |
| TEI | 32 | 16-32 | 自托管 |
| Bedrock | 1 | 1 | 无批量支持 |

**来源**: `reference/source_architecture.md:84-235`

### 1.2 实现代码

```python
"""
MaxBatch优化批量处理
根据Provider限制动态调整批量大小
"""

import os
from typing import List, Dict, Tuple
from dataclasses import dataclass
import time
from pymilvus import MilvusClient, Function, FunctionType, CollectionSchema, FieldSchema, DataType

@dataclass
class BatchConfig:
    provider: str
    max_batch: int
    optimal_batch: int
    model: str
    dim: int

class BatchProcessor:
    """批量处理优化器"""
    
    def __init__(self):
        self.configs = {
            "openai": BatchConfig("openai", 128, 64, "text-embedding-3-small", 1536),
            "voyageai": BatchConfig("voyageai", 128, 64, "voyage-3-large", 1024),
            "cohere": BatchConfig("cohere", 96, 48, "embed-english-v3.0", 1024),
            "dashscope": BatchConfig("dashscope", 6, 6, "text-embedding-v3", 1024),
        }
    
    def chunk_texts(self, texts: List[str], provider: str) -> List[List[str]]:
        """将文本分块为最优批量大小"""
        config = self.configs.get(provider)
        if not config:
            raise ValueError(f"Unknown provider: {provider}")
        
        batch_size = config.optimal_batch
        chunks = []
        
        for i in range(0, len(texts), batch_size):
            chunk = texts[i:i + batch_size]
            chunks.append(chunk)
        
        return chunks
    
    def create_collection(self, client: MilvusClient, provider: str, collection_name: str):
        """创建优化的Collection"""
        config = self.configs[provider]
        
        # 定义Embedding Function
        params = {
            "provider": config.provider,
            "model_name": config.model,
            "dim": config.dim
        }
        
        # 添加API密钥
        if provider == "openai":
            params["api_key"] = os.getenv("OPENAI_API_KEY")
        elif provider == "voyageai":
            params["api_key"] = os.getenv("VOYAGEAI_API_KEY")
        elif provider == "cohere":
            params["api_key"] = os.getenv("COHERE_API_KEY")
        elif provider == "dashscope":
            params["api_key"] = os.getenv("DASHSCOPE_API_KEY")
        
        embedding_func = Function(
            name=f"{provider}_ef",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["text"],
            output_field_names=["vector"],
            params=params
        )
        
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=65535),
            FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=config.dim)
        ]
        
        schema = CollectionSchema(
            fields=fields,
            functions=[embedding_func],
            description=f"Batch-optimized collection using {provider}"
        )
        
        if client.has_collection(collection_name):
            client.drop_collection(collection_name)
        
        client.create_collection(collection_name, schema=schema)
        print(f"✅ 创建Collection: {collection_name} (MaxBatch: {config.max_batch})")
    
    def batch_insert(self, client: MilvusClient, collection_name: str, 
                     texts: List[str], provider: str) -> Dict[str, float]:
        """批量插入并返回性能指标"""
        chunks = self.chunk_texts(texts, provider)
        config = self.configs[provider]
        
        total_time = 0
        total_texts = len(texts)
        
        print(f"\n📊 批量插入: {total_texts}条文本, 分{len(chunks)}批")
        
        for i, chunk in enumerate(chunks):
            start = time.time()
            
            data = [{"text": text} for text in chunk]
            client.insert(collection_name, data)
            
            elapsed = time.time() - start
            total_time += elapsed
            
            print(f"  批次 {i+1}/{len(chunks)}: {len(chunk)}条, 耗时 {elapsed:.2f}s")
        
        avg_time_per_text = total_time / total_texts
        throughput = total_texts / total_time
        
        metrics = {
            "total_time": total_time,
            "avg_time_per_text": avg_time_per_text,
            "throughput": throughput,
            "batch_count": len(chunks),
            "batch_size": config.optimal_batch
        }
        
        print(f"\n📈 性能指标:")
        print(f"  总耗时: {total_time:.2f}s")
        print(f"  平均每条: {avg_time_per_text*1000:.2f}ms")
        print(f"  吞吐量: {throughput:.2f} texts/s")
        
        return metrics

# 使用示例
def demo_batch_optimization():
    """演示批量优化"""
    client = MilvusClient(uri="http://localhost:19530")
    processor = BatchProcessor()
    
    # 生成测试数据
    texts = [f"Document {i}: This is a test document for batch processing." for i in range(200)]
    
    # 测试不同Provider
    providers = ["voyageai", "cohere", "dashscope"]
    
    for provider in providers:
        print(f"\n{'='*60}")
        print(f"测试Provider: {provider}")
        print(f"{'='*60}")
        
        collection_name = f"batch_{provider}"
        
        # 创建Collection
        processor.create_collection(client, provider, collection_name)
        
        # 批量插入
        metrics = processor.batch_insert(client, collection_name, texts, provider)

if __name__ == "__main__":
    demo_batch_optimization()
```

## 策略2:并行处理

### 2.1 设计思路

利用Python的并发能力并行处理多个批次:
- **多线程**: 适合I/O密集型任务(API调用)
- **多进程**: 适合CPU密集型任务(本地embedding)
- **异步IO**: 适合高并发场景

**来源**: `reference/source_architecture.md:58-318`

### 2.2 实现代码

```python
"""
并行批量处理
使用线程池并行调用embedding API
"""

import concurrent.futures
from typing import List, Dict
import threading

class ParallelBatchProcessor:
    """并行批量处理器"""
    
    def __init__(self, max_workers: int = 4):
        self.max_workers = max_workers
        self.lock = threading.Lock()
        self.progress = {"completed": 0, "total": 0}
    
    def process_batch(self, client: MilvusClient, collection_name: str, 
                     batch: List[str], batch_id: int) -> Dict:
        """处理单个批次"""
        start = time.time()
        
        data = [{"text": text} for text in batch]
        result = client.insert(collection_name, data)
        
        elapsed = time.time() - start
        
        # 更新进度
        with self.lock:
            self.progress["completed"] += len(batch)
            progress_pct = (self.progress["completed"] / self.progress["total"]) * 100
            print(f"  批次 {batch_id}: {len(batch)}条, 耗时 {elapsed:.2f}s, 进度 {progress_pct:.1f}%")
        
        return {
            "batch_id": batch_id,
            "count": len(batch),
            "time": elapsed,
            "insert_count": result["insert_count"]
        }
    
    def parallel_insert(self, client: MilvusClient, collection_name: str,
                       texts: List[str], batch_size: int = 64) -> Dict:
        """并行批量插入"""
        # 分批
        batches = [texts[i:i + batch_size] for i in range(0, len(texts), batch_size)]
        
        self.progress = {"completed": 0, "total": len(texts)}
        
        print(f"\n🚀 并行批量插入: {len(texts)}条文本, {len(batches)}批, {self.max_workers}线程")
        
        start = time.time()
        results = []
        
        # 使用线程池并行处理
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [
                executor.submit(self.process_batch, client, collection_name, batch, i)
                for i, batch in enumerate(batches)
            ]
            
            for future in concurrent.futures.as_completed(futures):
                try:
                    result = future.result()
                    results.append(result)
                except Exception as e:
                    print(f"❌ 批次处理失败: {e}")
        
        total_time = time.time() - start
        throughput = len(texts) / total_time
        
        print(f"\n📈 并行处理性能:")
        print(f"  总耗时: {total_time:.2f}s")
        print(f"  吞吐量: {throughput:.2f} texts/s")
        print(f"  加速比: {len(batches) / total_time:.2f}x")
        
        return {
            "total_time": total_time,
            "throughput": throughput,
            "batch_results": results
        }

# 使用示例
def demo_parallel_processing():
    """演示并行处理"""
    client = MilvusClient(uri="http://localhost:19530")
    processor = ParallelBatchProcessor(max_workers=4)
    
    # 创建Collection
    collection_name = "parallel_docs"
    # ... (使用之前的create_collection方法)
    
    # 生成测试数据
    texts = [f"Document {i}: Parallel processing test." for i in range(500)]
    
    # 并行插入
    metrics = processor.parallel_insert(client, collection_name, texts, batch_size=64)

if __name__ == "__main__":
    demo_parallel_processing()
```

## 策略3:内存优化

### 3.1 流式处理

对于超大数据集,使用流式处理避免内存溢出:

```python
"""
流式批量处理
逐批读取和处理,避免内存溢出
"""

from typing import Iterator, List
import json

class StreamBatchProcessor:
    """流式批量处理器"""
    
    def __init__(self, batch_size: int = 64):
        self.batch_size = batch_size
    
    def read_jsonl_stream(self, file_path: str) -> Iterator[str]:
        """流式读取JSONL文件"""
        with open(file_path, 'r', encoding='utf-8') as f:
            for line in f:
                data = json.loads(line)
                yield data.get("text", "")
    
    def batch_generator(self, stream: Iterator[str]) -> Iterator[List[str]]:
        """生成批次"""
        batch = []
        
        for text in stream:
            batch.append(text)
            
            if len(batch) >= self.batch_size:
                yield batch
                batch = []
        
        # 处理最后一批
        if batch:
            yield batch
    
    def stream_insert(self, client: MilvusClient, collection_name: str, 
                     file_path: str) -> Dict:
        """流式插入"""
        print(f"\n💧 流式处理: {file_path}")
        
        start = time.time()
        total_count = 0
        batch_count = 0
        
        # 创建流式读取器
        stream = self.read_jsonl_stream(file_path)
        batches = self.batch_generator(stream)
        
        # 逐批处理
        for batch in batches:
            data = [{"text": text} for text in batch]
            client.insert(collection_name, data)
            
            total_count += len(batch)
            batch_count += 1
            
            if batch_count % 10 == 0:
                print(f"  已处理: {total_count}条, {batch_count}批")
        
        total_time = time.time() - start
        throughput = total_count / total_time
        
        print(f"\n📈 流式处理性能:")
        print(f"  总条数: {total_count}")
        print(f"  总耗时: {total_time:.2f}s")
        print(f"  吞吐量: {throughput:.2f} texts/s")
        
        return {
            "total_count": total_count,
            "total_time": total_time,
            "throughput": throughput
        }

# 使用示例
def demo_stream_processing():
    """演示流式处理"""
    client = MilvusClient(uri="http://localhost:19530")
    processor = StreamBatchProcessor(batch_size=64)
    
    # 假设有大文件
    file_path = "large_dataset.jsonl"
    
    # 流式插入
    metrics = processor.stream_insert(client, "stream_docs", file_path)

if __name__ == "__main__":
    demo_stream_processing()
```

## 策略4:性能监控

### 4.1 实时监控

```python
"""
批量处理性能监控
实时跟踪吞吐量、延迟和错误率
"""

from dataclasses import dataclass, field
from typing import List
import time

@dataclass
class PerformanceMetrics:
    """性能指标"""
    start_time: float = field(default_factory=time.time)
    total_texts: int = 0
    successful_texts: int = 0
    failed_texts: int = 0
    batch_times: List[float] = field(default_factory=list)
    
    def add_batch(self, count: int, elapsed: float, success: bool = True):
        """添加批次结果"""
        self.total_texts += count
        if success:
            self.successful_texts += count
        else:
            self.failed_texts += count
        self.batch_times.append(elapsed)
    
    def get_summary(self) -> Dict:
        """获取性能摘要"""
        total_time = time.time() - self.start_time
        
        return {
            "total_time": total_time,
            "total_texts": self.total_texts,
            "successful_texts": self.successful_texts,
            "failed_texts": self.failed_texts,
            "success_rate": self.successful_texts / self.total_texts if self.total_texts > 0 else 0,
            "throughput": self.successful_texts / total_time if total_time > 0 else 0,
            "avg_batch_time": sum(self.batch_times) / len(self.batch_times) if self.batch_times else 0,
            "min_batch_time": min(self.batch_times) if self.batch_times else 0,
            "max_batch_time": max(self.batch_times) if self.batch_times else 0
        }
    
    def print_summary(self):
        """打印性能摘要"""
        summary = self.get_summary()
        
        print(f"\n{'='*60}")
        print(f"性能摘要")
        print(f"{'='*60}")
        print(f"总耗时: {summary['total_time']:.2f}s")
        print(f"总条数: {summary['total_texts']}")
        print(f"成功: {summary['successful_texts']}")
        print(f"失败: {summary['failed_texts']}")
        print(f"成功率: {summary['success_rate']*100:.2f}%")
        print(f"吞吐量: {summary['throughput']:.2f} texts/s")
        print(f"平均批次时间: {summary['avg_batch_time']:.2f}s")
        print(f"最快批次: {summary['min_batch_time']:.2f}s")
        print(f"最慢批次: {summary['max_batch_time']:.2f}s")

class MonitoredBatchProcessor:
    """带监控的批量处理器"""
    
    def __init__(self, batch_size: int = 64):
        self.batch_size = batch_size
        self.metrics = PerformanceMetrics()
    
    def monitored_insert(self, client: MilvusClient, collection_name: str,
                        texts: List[str]) -> PerformanceMetrics:
        """带监控的批量插入"""
        batches = [texts[i:i + self.batch_size] for i in range(0, len(texts), self.batch_size)]
        
        print(f"\n📊 开始批量插入: {len(texts)}条文本, {len(batches)}批")
        
        for i, batch in enumerate(batches):
            start = time.time()
            
            try:
                data = [{"text": text} for text in batch]
                client.insert(collection_name, data)
                
                elapsed = time.time() - start
                self.metrics.add_batch(len(batch), elapsed, success=True)
                
                print(f"  ✅ 批次 {i+1}/{len(batches)}: {len(batch)}条, {elapsed:.2f}s")
                
            except Exception as e:
                elapsed = time.time() - start
                self.metrics.add_batch(len(batch), elapsed, success=False)
                
                print(f"  ❌ 批次 {i+1}/{len(batches)} 失败: {e}")
        
        # 打印摘要
        self.metrics.print_summary()
        
        return self.metrics

# 使用示例
def demo_monitored_processing():
    """演示监控处理"""
    client = MilvusClient(uri="http://localhost:19530")
    processor = MonitoredBatchProcessor(batch_size=64)
    
    # 生成测试数据
    texts = [f"Document {i}: Monitored processing test." for i in range(300)]
    
    # 带监控的插入
    metrics = processor.monitored_insert(client, "monitored_docs", texts)

if __name__ == "__main__":
    demo_monitored_processing()
```

## 生产部署建议

### 5.1 批量大小选择

**推荐策略**:
1. **小数据集(<1000条)**: 使用MaxBatch的50%
2. **中数据集(1000-10000条)**: 使用MaxBatch的75%
3. **大数据集(>10000条)**: 使用MaxBatch的100%

### 5.2 并发控制

**推荐配置**:
- **API Provider**: 2-4个线程(避免rate limit)
- **自托管TEI**: 8-16个线程(充分利用CPU)
- **混合场景**: 动态调整

### 5.3 错误处理

**重试策略**:
```python
def retry_batch(func, max_retries=3, backoff=2):
    """批次重试装饰器"""
    for attempt in range(max_retries):
        try:
            return func()
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            time.sleep(backoff ** attempt)
```

### 5.4 内存管理

**最佳实践**:
- 使用流式处理处理大文件
- 及时释放已处理的批次
- 监控内存使用率
- 设置合理的批量大小上限
