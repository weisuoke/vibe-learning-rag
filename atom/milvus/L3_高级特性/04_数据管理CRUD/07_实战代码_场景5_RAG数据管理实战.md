## 完整代码

```python
"""
RAG 数据管理完整实战
演示：文档导入 → 元数据管理 → 检索 → 更新 → 清理
"""

from pymilvus import MilvusClient
import numpy as np
import time
from typing import List, Dict
import hashlib

# ===== 1. 初始化 RAG 系统 =====
print("=== 1. 初始化 RAG 系统 ===")

client = MilvusClient(uri="http://localhost:19530")
collection_name = "rag_knowledge_base"

# 删除旧 Collection
if client.has_collection(collection_name):
    client.drop_collection(collection_name)

# 创建 Collection
client.create_collection(
    collection_name=collection_name,
    dimension=384,  # 使用 sentence-transformers 的维度
    metric_type="COSINE",
    auto_id=False,
)
print(f"✓ 创建知识库: {collection_name}")

# ===== 2. 文档批量导入 =====
print("\n=== 2. 文档批量导入 ===")

def generate_doc_id(content: str) -> int:
    """基于内容生成唯一 ID"""
    return int(hashlib.md5(content.encode()).hexdigest()[:8], 16)

def generate_embedding(text: str) -> List[float]:
    """模拟 Embedding 生成（实际应使用真实模型）"""
    rng = np.random.default_rng(seed=hash(text) % 2**32)
    return rng.random(384).tolist()

# 模拟文档数据
documents = [
    {
        "title": "Milvus 快速入门",
        "content": "Milvus 是一个开源向量数据库，支持高性能向量检索...",
        "category": "tutorial",
        "tags": ["milvus", "vector-db", "tutorial"],
        "author": "Alice",
        "created_at": "2026-01-01",
    },
    {
        "title": "RAG 系统设计",
        "content": "RAG（Retrieval-Augmented Generation）结合了检索和生成...",
        "category": "architecture",
        "tags": ["rag", "llm", "architecture"],
        "author": "Bob",
        "created_at": "2026-01-15",
    },
    {
        "title": "向量检索优化",
        "content": "向量检索性能优化包括索引选择、参数调优、批量处理...",
        "category": "optimization",
        "tags": ["optimization", "performance", "vector-search"],
        "author": "Charlie",
        "created_at": "2026-02-01",
    },
]

# 准备插入数据
insert_data = []
for doc in documents:
    doc_id = generate_doc_id(doc["content"])
    embedding = generate_embedding(doc["content"])

    insert_data.append({
        "id": doc_id,
        "vector": embedding,
        "title": doc["title"],
        "content": doc["content"],
        "category": doc["category"],
        "tags": doc["tags"],
        "author": doc["author"],
        "created_at": doc["created_at"],
        "updated_at": int(time.time()),
        "version": 1,
    })

# 批量插入
result = client.insert(collection_name, insert_data)
print(f"✓ 导入文档: {result['insert_count']} 篇")

# ===== 3. 语义检索 =====
print("\n=== 3. 语义检索 ===")

# 用户查询
user_query = "如何优化向量检索性能？"
query_embedding = generate_embedding(user_query)

# 执行检索
search_results = client.search(
    collection_name=collection_name,
    data=[query_embedding],
    limit=3,
    output_fields=["title", "content", "category", "author"]
)

print(f"查询: {user_query}")
print(f"✓ 检索到 {len(search_results[0])} 篇相关文档:")
for i, hit in enumerate(search_results[0], 1):
    print(f"\n{i}. {hit['entity']['title']}")
    print(f"   作者: {hit['entity']['author']}")
    print(f"   类别: {hit['entity']['category']}")
    print(f"   相似度: {1 - hit['distance']:.4f}")
    print(f"   内容: {hit['entity']['content'][:50]}...")

# ===== 4. 混合检索（语义 + 元数据过滤）=====
print("\n=== 4. 混合检索 ===")

# 只检索 tutorial 类别的文档
filtered_results = client.search(
    collection_name=collection_name,
    data=[query_embedding],
    filter='category == "tutorial"',
    limit=3,
    output_fields=["title", "category"]
)

print(f"✓ 过滤条件: category == 'tutorial'")
print(f"✓ 检索到 {len(filtered_results[0])} 篇文档:")
for hit in filtered_results[0]:
    print(f"  - {hit['entity']['title']} ({hit['entity']['category']})")

# ===== 5. 文档更新 =====
print("\n=== 5. 文档更新 ===")

# 更新第一篇文档
doc_to_update = insert_data[0]
updated_content = doc_to_update["content"] + " 更新：添加了新的示例代码。"

# 生成新的 Embedding
new_embedding = generate_embedding(updated_content)

# Upsert 更新
upsert_data = [{
    "id": doc_to_update["id"],
    "vector": new_embedding,
    "title": doc_to_update["title"],
    "content": updated_content,
    "category": doc_to_update["category"],
    "tags": doc_to_update["tags"],
    "author": doc_to_update["author"],
    "created_at": doc_to_update["created_at"],
    "updated_at": int(time.time()),
    "version": doc_to_update["version"] + 1,
}]

result = client.upsert(collection_name, upsert_data)
print(f"✓ 更新文档: {result['upsert_count']} 篇")

# 验证更新
updated_doc = client.query(
    collection_name,
    ids=[doc_to_update["id"]],
    output_fields=["title", "version", "content"]
)[0]
print(f"  文档: {updated_doc['title']}")
print(f"  版本: {updated_doc['version']}")
print(f"  内容: {updated_doc['content'][:60]}...")

# ===== 6. 增量导入 =====
print("\n=== 6. 增量导入 ===")

# 新文档
new_documents = [
    {
        "title": "Milvus 性能调优",
        "content": "Milvus 性能调优包括硬件配置、索引优化、查询优化...",
        "category": "optimization",
        "tags": ["milvus", "performance", "tuning"],
        "author": "David",
        "created_at": "2026-02-20",
    }
]

# 准备数据
new_data = []
for doc in new_documents:
    doc_id = generate_doc_id(doc["content"])
    embedding = generate_embedding(doc["content"])

    new_data.append({
        "id": doc_id,
        "vector": embedding,
        "title": doc["title"],
        "content": doc["content"],
        "category": doc["category"],
        "tags": doc["tags"],
        "author": doc["author"],
        "created_at": doc["created_at"],
        "updated_at": int(time.time()),
        "version": 1,
    })

# 插入新文档
result = client.insert(collection_name, new_data)
print(f"✓ 增量导入: {result['insert_count']} 篇新文档")

# ===== 7. 数据统计 =====
print("\n=== 7. 数据统计 ===")

# 按类别统计
categories = ["tutorial", "architecture", "optimization"]
for category in categories:
    docs = client.query(
        collection_name,
        filter=f'category == "{category}"',
        output_fields=["id"],
        limit=1000
    )
    print(f"  {category}: {len(docs)} 篇")

# ===== 8. 数据清理 =====
print("\n=== 8. 数据清理 ===")

# 删除旧文档（模拟：删除 2026-01-01 之前的文档）
delete_result = client.delete(
    collection_name,
    filter='created_at < "2026-01-02"'
)
print(f"✓ 清理旧文档: {delete_result['delete_count']} 篇")

# 执行 Compaction
client.compact(collection_name)
print("✓ 执行 Compaction 优化存储")

# ===== 9. 清理资源 =====
print("\n=== 9. 清理资源 ===")
client.drop_collection(collection_name)
print(f"✓ 删除知识库: {collection_name}")

print("\n=== 完成 ===")
```

## 关键知识点

### 1. 文档 ID 生成策略

**基于内容哈希**：
```python
def generate_doc_id(content: str) -> int:
    return int(hashlib.md5(content.encode()).hexdigest()[:8], 16)
```

**优势**：
- 相同内容生成相同 ID
- 自动去重
- 支持幂等性

## RAG 最佳实践

### 1. 文档分块策略

```python
def chunk_document(content: str, chunk_size: int = 500) -> List[str]:
    """将长文档分块"""
    words = content.split()
    chunks = []
    for i in range(0, len(words), chunk_size):
        chunk = " ".join(words[i:i + chunk_size])
        chunks.append(chunk)
    return chunks
```

### 2. 元数据设计

```python
metadata = {
```

```
1. 检查文档是否存在（基于 ID）
2. 存在 → Upsert 更新
3. 不存在 → Insert 插入
4. 定期 Compaction
```
