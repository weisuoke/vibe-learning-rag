## 一、分布式部署概览

### 1.1 为什么需要分布式部署？

**单机部署的局限：**
- 存储容量受限于单机磁盘
- 计算能力受限于单机 CPU/内存
- 无法实现高可用
- 无法水平扩展

**分布式部署的优势：**
- ✅ 无限扩展：存储和计算可独立扩展
- ✅ 高可用：多副本保证服务不中断
- ✅ 负载均衡：请求分散到多个节点
- ✅ 故障隔离：单节点故障不影响整体服务

### 1.2 Milvus 分布式架构


## 二、Kubernetes 部署实战

### 2.1 使用 Helm 部署 Milvus 集群

**安装 Helm Chart：**

```bash
# 1. 添加 Milvus Helm 仓库
helm repo add milvus https://zilliztech.github.io/milvus-helm/
helm repo update

# 2. 创建命名空间
kubectl create namespace milvus

# 3. 安装 Milvus 集群
helm install milvus-cluster milvus/milvus \
  --namespace milvus \
  --set cluster.enabled=true \
  --set minio.mode=distributed \
  --set etcd.replicaCount=3
```

### 2.3 Python 部署脚本

**自动化部署脚本：**

```python
import subprocess
import time
import yaml

def deploy_milvus_cluster(namespace="milvus", release_name="milvus-cluster"):
    """部署 Milvus 分布式集群"""

    print(f"Deploying Milvus cluster: {release_name}")
    print("=" * 60)

    # 1. 添加 Helm 仓库
    print("\n[Step 1] Adding Milvus Helm repository...")
    subprocess.run([
        "helm", "repo", "add", "milvus",
        "https://zilliztech.github.io/milvus-helm/"
    ], check=True)

    subprocess.run(["helm", "repo", "update"], check=True)
    print("✅ Helm repository added")

    # 2. 创建命名空间
    print(f"\n[Step 2] Creating namespace: {namespace}...")
    subprocess.run([
        "kubectl", "create", "namespace", namespace
    ], check=False)  # 忽略错误（命名空间可能已存在）
    print(f"✅ Namespace {namespace} ready")

    # 3. 部署 Milvus
    print(f"\n[Step 3] Deploying Milvus cluster...")

    helm_values = {
        "cluster.enabled": "true",
        "proxy.replicas": "3",
        "dataNode.replicas": "3",
        "queryNode.replicas": "3",
        "indexNode.replicas": "2",
        "streamingNode.enabled": "true",
        "streamingNode.replicas": "2",
        "minio.mode": "distributed",
        "minio.replicas": "4",
        "etcd.replicaCount": "3"
    }

    helm_cmd = [
        "helm", "install", release_name, "milvus/milvus",
        "--namespace", namespace
    ]

    for key, value in helm_values.items():
        helm_cmd.extend(["--set", f"{key}={value}"])

    subprocess.run(helm_cmd, check=True)
    print("✅ Milvus cluster deployed")

    # 4. 等待 Pod 就绪
    print(f"\n[Step 4] Waiting for pods to be ready...")
    wait_for_pods(namespace)

    # 5. 获取服务地址
    print(f"\n[Step 5] Getting service endpoint...")
    service_endpoint = get_service_endpoint(namespace, release_name)

    print("\n" + "=" * 60)
    print("Milvus Cluster Deployed Successfully!")
    print("=" * 60)
    print(f"Namespace: {namespace}")
    print(f"Release: {release_name}")
    print(f"Endpoint: {service_endpoint}")
    print("\nConnect using:")
    print(f"  from pymilvus import connections")
    print(f"  connections.connect(host='{service_endpoint}', port='19530')")

def wait_for_pods(namespace, timeout=600):
    """等待所有 Pod 就绪"""

    start_time = time.time()

    while time.time() - start_time < timeout:
        result = subprocess.run([
            "kubectl", "get", "pods",
            "-n", namespace,
            "-o", "jsonpath={.items[*].status.phase}"
        ], capture_output=True, text=True)

        phases = result.stdout.split()

        if all(phase == "Running" for phase in phases):
            print("✅ All pods are running")
            return

        print(f"  Waiting... ({len([p for p in phases if p == 'Running'])}/{len(phases)} pods running)")
        time.sleep(10)

    raise TimeoutError("Pods did not become ready in time")

def get_service_endpoint(namespace, release_name):
    """获取服务端点"""

    result = subprocess.run([
        "kubectl", "get", "svc",
        f"{release_name}-milvus",
        "-n", namespace,
        "-o", "jsonpath={.status.loadBalancer.ingress[0].ip}"
    ], capture_output=True, text=True)

    endpoint = result.stdout.strip()

    if not endpoint:
        # 如果没有 LoadBalancer IP，使用 ClusterIP
        result = subprocess.run([
            "kubectl", "get", "svc",
            f"{release_name}-milvus",
            "-n", namespace,
            "-o", "jsonpath={.spec.clusterIP}"
        ], capture_output=True, text=True)
        endpoint = result.stdout.strip()

    return endpoint

# 运行部署
if __name__ == "__main__":
    deploy_milvus_cluster()
```

## 三、集群管理与监控

### 3.1 集群状态检查

**Python 脚本：**

```python
import subprocess
import json

def check_cluster_status(namespace="milvus"):
    """检查 Milvus 集群状态"""

    print("Milvus Cluster Status")
    print("=" * 60)

    # 1. 检查 Pod 状态
    print("\n[Pods Status]")
    result = subprocess.run([
        "kubectl", "get", "pods",
        "-n", namespace,
        "-o", "json"
    ], capture_output=True, text=True)

    pods = json.loads(result.stdout)

    pod_stats = {
        "Running": 0,
        "Pending": 0,
        "Failed": 0,
        "Unknown": 0
    }

    for pod in pods["items"]:
        phase = pod["status"]["phase"]
        pod_stats[phase] = pod_stats.get(phase, 0) + 1

        name = pod["metadata"]["name"]
        print(f"  {name}: {phase}")

    print(f"\nSummary:")
    for status, count in pod_stats.items():
        print(f"  {status}: {count}")

    # 2. 检查服务状态
    print(f"\n[Services]")
    result = subprocess.run([
        "kubectl", "get", "svc",
        "-n", namespace
    ], capture_output=True, text=True)

    print(result.stdout)

    # 3. 检查存储
    print(f"\n[Persistent Volumes]")
    result = subprocess.run([
        "kubectl", "get", "pvc",
        "-n", namespace
    ], capture_output=True, text=True)

    print(result.stdout)

# 运行检查
check_cluster_status()
```

## 四、RAG 应用场景

### 4.1 场景 1：大规模 RAG 系统部署

**问题：** RAG 系统需要处理数十亿向量

**分布式部署的作用：**
- 多个 Query Node 并行查询
- 多个 Data Node 并行写入
- 存储和计算独立扩展

**Python 代码示例：**

```python
from pymilvus import connections, Collection, utility
import numpy as np

# 连接到分布式集群
connections.connect(
    alias="default",
    host="milvus-cluster.milvus.svc.cluster.local",
    port="19530"
)

def deploy_large_scale_rag():
    """大规模 RAG 系统部署"""

    # 1. 创建 Collection
    from pymilvus import FieldSchema, CollectionSchema, DataType

    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
        FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=65535),
        FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=768)
    ]

    schema = CollectionSchema(fields, description="Large-scale RAG knowledge base")
    collection = Collection("large_rag_kb", schema)

    print("✅ Collection created on distributed cluster")

    # 2. 创建索引
    index_params = {
        "index_type": "IVF_FLAT",
        "metric_type": "L2",
        "params": {"nlist": 2048}
    }

    collection.create_index("embedding", index_params)
    print("✅ Index created (distributed across Index Nodes)")

    # 3. 批量插入数据（分布式写入）
    batch_size = 10000
    total_docs = 1000000

    for i in range(0, total_docs, batch_size):
        documents = [
            {
                "text": f"Document {j}",
                "embedding": np.random.rand(768).tolist()
            }
            for j in range(i, min(i + batch_size, total_docs))
        ]

        texts = [doc["text"] for doc in documents]
        embeddings = [doc["embedding"] for doc in documents]

        collection.insert([texts, embeddings])

        if (i + batch_size) % 100000 == 0:
            print(f"  Inserted {i + batch_size} documents")

    print(f"✅ Inserted {total_docs} documents (distributed across Data Nodes)")

    # 4. 加载 Collection（分布式加载）
    collection.load()
    print("✅ Collection loaded (distributed across Query Nodes)")

    # 5. 查询测试（分布式查询）
    query_vector = np.random.rand(768).tolist()

    results = collection.search(
        data=[query_vector],
        anns_field="embedding",
        param={"metric_type": "L2", "params": {"nprobe": 10}},
        limit=10
    )

    print(f"✅ Query completed (distributed across {utility.get_query_segment_info('large_rag_kb')} segments)")

# 运行
deploy_large_scale_rag()
```

### 4.2 场景 2：高可用 RAG 系统

**问题：** RAG 系统需要 99.9% 可用性

**分布式部署的作用：**
- 多副本保证服务不中断
- 自动故障转移
- 负载均衡

**Python 代码示例：**

```python
def test_high_availability():
    """测试高可用性"""

    # 连接到分布式集群（通过 LoadBalancer）
    connections.connect(
        host="milvus-lb.example.com",
        port="19530"
    )

    collection = Collection("rag_knowledge_base")

    # 模拟持续查询
    success_count = 0
    failure_count = 0

    for i in range(1000):
        try:
            query_vector = np.random.rand(768).tolist()

            results = collection.search(
                data=[query_vector],
                anns_field="embedding",
                param={"metric_type": "L2", "params": {"nprobe": 10}},
                limit=10
            )

            success_count += 1

        except Exception as e:
            failure_count += 1
            print(f"Query {i} failed: {e}")

    availability = (success_count / (success_count + failure_count)) * 100

    print(f"\nHigh Availability Test Results:")
    print(f"  Success: {success_count}")
    print(f"  Failure: {failure_count}")
    print(f"  Availability: {availability:.2f}%")

    if availability >= 99.9:
        print("✅ High availability achieved!")
    else:
        print("⚠️  Availability below target")

# 运行测试
test_high_availability()
```
