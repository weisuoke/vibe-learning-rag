# 场景3：多节点差异化重试

> **实战代码 - 场景3** | 预计阅读：8分钟
> 不同节点使用不同的重试策略

---

## 场景描述

在复杂的 RAG 工作流中，不同节点有不同的特性：
- **数据库查询**：快速失败，少量重试
- **LLM 调用**：慢速操作，多次重试，长间隔
- **文档解析**：CPU 密集，不应重试（编程错误）
- **API 调用**：网络依赖，中等重试

**一刀切的重试策略不合适**，需要针对每个节点的特性定制策略。

---

## 完整代码

```python
"""
场景3：多节点差异化重试
演示：不同节点使用不同的重试策略
"""

import os
import time
import sqlite3
from typing import TypedDict, List
from openai import OpenAI, RateLimitError, APITimeoutError

from langgraph.graph import StateGraph, START, END
from langgraph.types import RetryPolicy

# ===== 1. 定义状态 =====
class State(TypedDict):
    query: str
    user_id: int
    user_history: List[str]
    documents: List[str]
    answer: str

# ===== 2. 数据库节点（快速失败） =====
# 创建内存数据库
conn = sqlite3.connect(":memory:", check_same_thread=False)
cursor = conn.cursor()
cursor.execute("""
    CREATE TABLE user_history (
        user_id INTEGER,
        query TEXT,
        timestamp INTEGER
    )
""")
cursor.execute("INSERT INTO user_history VALUES (1, '什么是 Python?', 1234567890)")
cursor.execute("INSERT INTO user_history VALUES (1, '什么是 LangChain?', 1234567891)")
conn.commit()

def query_user_history(state: State) -> dict:
    """
    查询用户历史（数据库操作）

    特点：
    - 操作快速（< 100ms）
    - 失败通常是锁冲突（sqlite3.OperationalError）
    - 应该快速重试，但不要太多次
    """
    print(f"\n[数据库] 查询用户 {state['user_id']} 的历史...")

    try:
        cursor.execute(
            "SELECT query FROM user_history WHERE user_id = ? ORDER BY timestamp DESC LIMIT 5",
            (state['user_id'],)
        )
        history = [row[0] for row in cursor.fetchall()]

        print(f"[数据库] ✅ 查询成功，找到 {len(history)} 条历史")
        return {"user_history": history}

    except sqlite3.OperationalError as e:
        print(f"[数据库] ❌ 数据库锁定: {e}")
        raise  # 让 RetryPolicy 处理
    except Exception as e:
        print(f"[数据库] ❌ 查询失败: {e}")
        raise

# 数据库重试策略：快速重试，少量尝试
db_retry_policy = RetryPolicy(
    max_attempts=3,           # 最多 3 次尝试
    initial_interval=0.1,     # 首次重试等待 0.1 秒
    backoff_factor=2.0,       # 0.1s → 0.2s → 0.4s
    max_interval=1.0,         # 最长等待 1 秒
    retry_on=sqlite3.OperationalError  # 只重试锁冲突
)

print("数据库重试策略：")
print(f"  - 最多尝试: {db_retry_policy.max_attempts} 次")
print(f"  - 初始间隔: {db_retry_policy.initial_interval}s")
print(f"  - 重试异常: sqlite3.OperationalError")

# ===== 3. 检索节点（中等重试） =====
class MockVectorDB:
    """模拟向量数据库"""
    def __init__(self):
        self.call_count = 0

    def search(self, query: str) -> List[str]:
        self.call_count += 1
        # 模拟偶尔的网络错误
        import random
        if random.random() < 0.2:  # 20% 失败率
            raise ConnectionError("向量数据库连接超时")

        time.sleep(0.05)  # 模拟网络延迟
        return [
            "LangGraph 是一个用于构建有状态的多步骤 LLM 应用的框架。",
            "它支持循环、条件分支和状态管理。",
            "适合构建复杂的 Agent 系统。"
        ]

vector_db = MockVectorDB()

def retrieve_documents(state: State) -> dict:
    """
    检索相关文档（向量数据库）

    特点：
    - 网络依赖，可能超时
    - 失败通常是瞬态的
    - 应该中等重试
    """
    print(f"\n[检索] 检索相关文档...")
    print(f"  查询: {state['query']}")

    try:
        docs = vector_db.search(state['query'])
        print(f"[检索] ✅ 检索成功，找到 {len(docs)} 个文档")
        return {"documents": docs}

    except ConnectionError as e:
        print(f"[检索] ❌ 连接失败: {e}")
        raise

# 检索重试策略：中等重试
retrieval_retry_policy = RetryPolicy(
    max_attempts=4,           # 最多 4 次尝试
    initial_interval=0.5,     # 首次重试等待 0.5 秒
    backoff_factor=2.0,       # 0.5s → 1s → 2s → 4s
    max_interval=10.0,        # 最长等待 10 秒
    retry_on=ConnectionError  # 只重试连接错误
)

print("\n检索重试策略：")
print(f"  - 最多尝试: {retrieval_retry_policy.max_attempts} 次")
print(f"  - 初始间隔: {retrieval_retry_policy.initial_interval}s")
print(f"  - 重试异常: ConnectionError")

# ===== 4. LLM 节点（慢速重试） =====
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

def generate_answer(state: State) -> dict:
    """
    生成答案（LLM 调用）

    特点：
    - 操作慢（几秒到几十秒）
    - 可能遇到速率限制（429）
    - 可能超时
    - 应该多次重试，长间隔
    """
    print(f"\n[LLM] 生成答案...")
    print(f"  查询: {state['query']}")
    print(f"  上下文: {len(state['documents'])} 个文档")

    # 构建上下文
    context = "\n".join([f"- {doc}" for doc in state['documents']])
    history = "\n".join([f"- {q}" for q in state.get('user_history', [])])

    try:
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "你是一个有帮助的助手。"},
                {"role": "user", "content": f"用户历史查询：\n{history}\n\n相关文档：\n{context}\n\n当前问题：{state['query']}"}
            ],
            temperature=0.7,
            max_tokens=500,
            timeout=30.0
        )

        answer = response.choices[0].message.content
        print(f"[LLM] ✅ 生成成功 ({len(answer)} 字符)")
        return {"answer": answer}

    except RateLimitError as e:
        print(f"[LLM] ❌ 速率限制: {e}")
        raise
    except APITimeoutError as e:
        print(f"[LLM] ❌ 请求超时: {e}")
        raise
    except Exception as e:
        print(f"[LLM] ❌ 生成失败: {e}")
        raise

# LLM 重试策略：慢速重试，多次尝试
def should_retry_llm_error(exc: Exception) -> bool:
    """判断 LLM 错误是否应该重试"""
    if isinstance(exc, (RateLimitError, APITimeoutError, ConnectionError)):
        return True
    return False

llm_retry_policy = RetryPolicy(
    max_attempts=5,           # 最多 5 次尝试
    initial_interval=2.0,     # 首次重试等待 2 秒
    backoff_factor=2.0,       # 2s → 4s → 8s → 16s → 32s
    max_interval=60.0,        # 最长等待 60 秒
    jitter=True,              # 启用随机抖动
    retry_on=should_retry_llm_error
)

print("\nLLM 重试策略：")
print(f"  - 最多尝试: {llm_retry_policy.max_attempts} 次")
print(f"  - 初始间隔: {llm_retry_policy.initial_interval}s")
print(f"  - 最大间隔: {llm_retry_policy.max_interval}s")
print(f"  - 随机抖动: {'启用' if llm_retry_policy.jitter else '禁用'}")

# ===== 5. 文档解析节点（不重试） =====
def parse_documents(state: State) -> dict:
    """
    解析文档（纯计算）

    特点：
    - CPU 密集型操作
    - 失败通常是编程错误（如格式错误）
    - 不应该重试（重试也不会成功）
    """
    print(f"\n[解析] 解析文档...")

    try:
        # 模拟解析逻辑
        parsed_docs = [doc.upper() for doc in state['documents']]
        print(f"[解析] ✅ 解析成功")
        return {"documents": parsed_docs}

    except Exception as e:
        print(f"[解析] ❌ 解析失败: {e}")
        raise

# 解析节点：不配置重试策略（失败直接抛出）
print("\n解析节点：不配置重试策略（编程错误不应重试）")

# ===== 6. 构建工作流 =====
builder = StateGraph(State)

# 添加节点（每个节点不同的重试策略）
builder.add_node("query_history", query_user_history, retry_policy=db_retry_policy)
builder.add_node("retrieve", retrieve_documents, retry_policy=retrieval_retry_policy)
builder.add_node("parse", parse_documents)  # 不重试
builder.add_node("generate", generate_answer, retry_policy=llm_retry_policy)

# 添加边
builder.add_edge(START, "query_history")
builder.add_edge("query_history", "retrieve")
builder.add_edge("retrieve", "parse")
builder.add_edge("parse", "generate")
builder.add_edge("generate", END)

graph = builder.compile()

print("\n✅ 工作流构建完成\n")

# ===== 7. 测试 =====
print("="*60)
print("测试：完整 RAG 流程")
print("="*60)

initial_state = {
    "query": "什么是 LangGraph？",
    "user_id": 1,
    "user_history": [],
    "documents": [],
    "answer": ""
}

try:
    result = graph.invoke(initial_state)
    print(f"\n✅ 执行成功")
    print(f"  用户历史: {len(result['user_history'])} 条")
    print(f"  检索文档: {len(result['documents'])} 个")
    print(f"  答案: {result['answer'][:100]}...")
    print(f"\n调用统计:")
    print(f"  向量数据库调用: {vector_db.call_count} 次")
except Exception as e:
    print(f"\n❌ 执行失败: {type(e).__name__}: {e}")

# ===== 8. 重试策略对比 =====
print("\n" + "="*60)
print("重试策略对比")
print("="*60)

comparison = """
┌──────────────┬──────────┬──────────┬──────────┬──────────┐
│ 节点类型     │ 最多尝试 │ 初始间隔 │ 最大间隔 │ 重试异常 │
├──────────────┼──────────┼──────────┼──────────┼──────────┤
│ 数据库查询   │ 3 次     │ 0.1s     │ 1s       │ 锁冲突   │
│ 向量检索     │ 4 次     │ 0.5s     │ 10s      │ 连接错误 │
│ LLM 调用     │ 5 次     │ 2.0s     │ 60s      │ 429/超时 │
│ 文档解析     │ 不重试   │ -        │ -        │ -        │
└──────────────┴──────────┴──────────┴──────────┴──────────┘

设计原则：
1. 快速操作（数据库）→ 快速重试，少量尝试
2. 中速操作（检索）→ 中等重试
3. 慢速操作（LLM）→ 慢速重试，多次尝试
4. 纯计算（解析）→ 不重试（编程错误）

退避序列对比：
- 数据库：0.1s → 0.2s → 0.4s（总计 0.7s）
- 检索：0.5s → 1s → 2s → 4s（总计 7.5s）
- LLM：2s → 4s → 8s → 16s → 32s（总计 62s）
"""

print(comparison)

# ===== 9. 实际应用建议 =====
print("\n" + "="*60)
print("实际应用建议")
print("="*60)

recommendations = """
1. **根据操作特性选择策略**
   - 快速操作（< 100ms）：快速重试，少量尝试
   - 中速操作（100ms-1s）：中等重试
   - 慢速操作（> 1s）：慢速重试，多次尝试
   - 纯计算：不重试

2. **根据失败类型选择策略**
   - 瞬态错误（网络、429）：应该重试
   - 编程错误（ValueError）：不应该重试
   - 资源冲突（锁）：快速重试

3. **考虑用户体验**
   - 总重试时间不要超过用户耐心（通常 < 30s）
   - 关键路径节点（如 LLM）可以多重试
   - 非关键节点（如日志）可以少重试或不重试

4. **监控与调优**
   - 记录每个节点的重试次数和成功率
   - 根据实际数据调整重试参数
   - 设置告警阈值（如重试率 > 20%）

5. **成本考虑**
   - LLM 调用成本高，但重试是必要的
   - 数据库查询成本低，可以多重试
   - 平衡成本和可靠性
"""

print(recommendations)

print("\n" + "="*60)
print("✅ 场景3 完成")
print("="*60)
```

## 关键要点

### 1. 差异化策略设计

```python
# 快速操作 → 快速重试
db_retry = RetryPolicy(max_attempts=3, initial_interval=0.1)

# 中速操作 → 中等重试
retrieval_retry = RetryPolicy(max_attempts=4, initial_interval=0.5)

# 慢速操作 → 慢速重试
llm_retry = RetryPolicy(max_attempts=5, initial_interval=2.0)

# 纯计算 → 不重试
# 不传 retry_policy
```

### 2. 异常类型匹配

```python
# 只重试特定异常
RetryPolicy(retry_on=sqlite3.OperationalError)

# 重试多种异常
RetryPolicy(retry_on=(RateLimitError, APITimeoutError))

# 自定义判断函数
def should_retry(exc: Exception) -> bool:
    return isinstance(exc, (RateLimitError, APITimeoutError))

RetryPolicy(retry_on=should_retry)
```

### 3. 退避序列对比

```
数据库（快）：0.1s → 0.2s → 0.4s（总计 0.7s）
检索（中）：0.5s → 1s → 2s → 4s（总计 7.5s）
LLM（慢）：2s → 4s → 8s → 16s → 32s（总计 62s）
```

### 4. 节点级配置

```python
builder.add_node("node1", func1, retry_policy=policy1)
builder.add_node("node2", func2, retry_policy=policy2)
builder.add_node("node3", func3)  # 不重试
```
