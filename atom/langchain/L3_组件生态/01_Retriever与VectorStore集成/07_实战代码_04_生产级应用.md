## 示例1:完整的生产级Retriever

```python
"""
生产级Retriever封装
包含:错误处理、重试机制、监控、降级策略
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from langchain_core.documents import Document
from dotenv import load_dotenv
import logging
import time
from typing import List, Optional
from functools import wraps

load_dotenv()

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class ProductionRetriever:
    """生产级检索器封装"""

    def __init__(
        self,
        vectorstore,
        search_type: str = "similarity",
        search_kwargs: dict = None,
        max_retries: int = 3,
        timeout: float = 30.0,
        fallback_docs: List[Document] = None
    ):
        self.vectorstore = vectorstore
        self.search_type = search_type
        self.search_kwargs = search_kwargs or {"k": 4}
        self.max_retries = max_retries
        self.timeout = timeout
        self.fallback_docs = fallback_docs or []

        # 创建retriever
        self.retriever = vectorstore.as_retriever(
            search_type=search_type,
            search_kwargs=self.search_kwargs,
            tags=["production", "v1.0"]
        )

        # 统计信息
        self.stats = {
            "total_queries": 0,
            "successful_queries": 0,
            "failed_queries": 0,
            "fallback_used": 0,
            "total_latency": 0.0
        }

    def retrieve(self, query: str, **kwargs) -> List[Document]:
        """
        生产级检索方法

        Args:
            query: 查询字符串
            **kwargs: 额外参数

        Returns:
            文档列表
        """
        self.stats["total_queries"] += 1
        start_time = time.time()

        try:
            # 尝试检索(带重试)
            docs = self._retrieve_with_retry(query, **kwargs)

            # 验证结果
            if not docs:
                logger.warning(f"检索返回空结果: {query}")
                docs = self.fallback_docs
                self.stats["fallback_used"] += 1

            # 记录成功
            self.stats["successful_queries"] += 1
            latency = time.time() - start_time
            self.stats["total_latency"] += latency

            logger.info(
                f"检索成功: query='{query[:50]}...', "
                f"docs={len(docs)}, latency={latency:.3f}s"
            )

            return docs

        except Exception as e:
            # 记录失败
            self.stats["failed_queries"] += 1
            logger.error(f"检索失败: {e}")

            # 使用降级策略
            logger.info("使用降级策略")
            self.stats["fallback_used"] += 1
            return self.fallback_docs

    def _retrieve_with_retry(
        self,
        query: str,
        **kwargs
    ) -> List[Document]:
        """带重试的检索"""
        last_error = None

        for attempt in range(self.max_retries):
            try:
                # 执行检索
                docs = self.retriever.invoke(query, **kwargs)
                return docs

            except Exception as e:
                last_error = e
                logger.warning(
                    f"检索失败 (尝试 {attempt + 1}/{self.max_retries}): {e}"
                )

                if attempt < self.max_retries - 1:
                    # 指数退避
                    wait_time = 2 ** attempt
                    logger.info(f"等待 {wait_time}s 后重试...")
                    time.sleep(wait_time)

        # 所有重试都失败
        raise last_error

    def get_stats(self) -> dict:
        """获取统计信息"""
        stats = self.stats.copy()
        if stats["successful_queries"] > 0:
            stats["avg_latency"] = (
                stats["total_latency"] / stats["successful_queries"]
            )
        else:
            stats["avg_latency"] = 0.0

        stats["success_rate"] = (
            stats["successful_queries"] / stats["total_queries"]
            if stats["total_queries"] > 0
            else 0.0
        )

        return stats

# 使用示例
if __name__ == "__main__":
    # 准备数据
    vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
    vectorstore.add_texts([
        "LangChain is a framework",
        "LCEL is a language",
        "RAG is a technique"
    ])

    # 创建生产级retriever
    retriever = ProductionRetriever(
        vectorstore,
        search_type="similarity",
        search_kwargs={"k": 2},
        max_retries=3,
        fallback_docs=[Document(page_content="默认文档")]
    )

    # 测试检索
    queries = [
        "What is LangChain?",
        "What is LCEL?",
        "What is RAG?"
    ]

    for query in queries:
        docs = retriever.retrieve(query)
        print(f"\n查询: {query}")
        print(f"返回文档数: {len(docs)}")

    # 打印统计信息
    print("\n统计信息:")
    stats = retriever.get_stats()
    for key, value in stats.items():
        print(f"  {key}: {value}")
```

## 示例2:性能监控与追踪

```python
"""
性能监控示例
使用LangSmith追踪和自定义监控
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from langchain_core.callbacks import BaseCallbackHandler
from dotenv import load_dotenv
import time
from collections import defaultdict

load_dotenv()

class PerformanceMonitor(BaseCallbackHandler):
    """性能监控回调"""

    def __init__(self):
        self.metrics = defaultdict(list)
        self.current_run = {}

    def on_retriever_start(self, serialized, query, **kwargs):
        """记录开始时间"""
        run_id = kwargs.get("run_id")
        self.current_run[run_id] = {
            "query": query,
            "start_time": time.time()
        }

    def on_retriever_end(self, documents, **kwargs):
        """记录结束时间和结果"""
        run_id = kwargs.get("run_id")
        if run_id in self.current_run:
            run_data = self.current_run[run_id]
            latency = time.time() - run_data["start_time"]

            # 记录指标
            self.metrics["latency"].append(latency)
            self.metrics["doc_count"].append(len(documents))
            self.metrics["query_length"].append(len(run_data["query"]))

            # 清理
            del self.current_run[run_id]

    def on_retriever_error(self, error, **kwargs):
        """记录错误"""
        self.metrics["errors"].append(str(error))

    def get_summary(self) -> dict:
        """获取性能摘要"""
        if not self.metrics["latency"]:
            return {}

        return {
            "total_queries": len(self.metrics["latency"]),
            "avg_latency": sum(self.metrics["latency"]) / len(self.metrics["latency"]),
            "min_latency": min(self.metrics["latency"]),
            "max_latency": max(self.metrics["latency"]),
            "avg_doc_count": sum(self.metrics["doc_count"]) / len(self.metrics["doc_count"]),
            "error_count": len(self.metrics["errors"])
        }

# 使用示例
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore.add_texts([
    "LangChain is a framework",
    "LCEL is a language",
    "RAG is a technique"
])

retriever = vectorstore.as_retriever()
monitor = PerformanceMonitor()

# 执行检索(带监控)
queries = ["What is LangChain?", "What is LCEL?", "What is RAG?"]
for query in queries:
    docs = retriever.invoke(
        query,
        config={"callbacks": [monitor]}
    )

# 打印性能摘要
print("性能摘要:")
summary = monitor.get_summary()
for key, value in summary.items():
    if isinstance(value, float):
        print(f"  {key}: {value:.4f}")
    else:
        print(f"  {key}: {value}")
```

## 示例3:缓存机制

```python
"""
缓存机制示例
减少重复查询的开销
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from langchain_core.documents import Document
from dotenv import load_dotenv
from functools import lru_cache
import hashlib
import json
from typing import List

load_dotenv()

class CachedRetriever:
    """带缓存的检索器"""

    def __init__(self, vectorstore, cache_size: int = 100):
        self.retriever = vectorstore.as_retriever()
        self.cache_size = cache_size
        self.cache_hits = 0
        self.cache_misses = 0

    def _get_cache_key(self, query: str, **kwargs) -> str:
        """生成缓存键"""
        cache_data = {"query": query, **kwargs}
        cache_str = json.dumps(cache_data, sort_keys=True)
        return hashlib.md5(cache_str.encode()).hexdigest()

    @lru_cache(maxsize=100)
    def _cached_retrieve(self, cache_key: str, query: str) -> tuple:
        """缓存的检索方法"""
        docs = self.retriever.invoke(query)
        # 转换为可缓存的格式
        return tuple((doc.page_content, doc.metadata) for doc in docs)

    def retrieve(self, query: str, **kwargs) -> List[Document]:
        """检索(带缓存)"""
        cache_key = self._get_cache_key(query, **kwargs)

        # 尝试从缓存获取
        try:
            cached_data = self._cached_retrieve(cache_key, query)
            self.cache_hits += 1
            print(f"缓存命中: {query[:50]}...")

            # 转换回Document对象
            return [
                Document(page_content=content, metadata=metadata)
                for content, metadata in cached_data
            ]
        except:
            self.cache_misses += 1
            print(f"缓存未命中: {query[:50]}...")
            docs = self.retriever.invoke(query, **kwargs)
            return docs

    def get_cache_stats(self) -> dict:
        """获取缓存统计"""
        total = self.cache_hits + self.cache_misses
        hit_rate = self.cache_hits / total if total > 0 else 0.0
        return {
            "cache_hits": self.cache_hits,
            "cache_misses": self.cache_misses,
            "hit_rate": hit_rate
        }

# 使用示例
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore.add_texts([
    "LangChain is a framework",
    "LCEL is a language"
])

retriever = CachedRetriever(vectorstore, cache_size=100)

# 测试缓存
queries = ["What is LangChain?", "What is LangChain?", "What is LCEL?"]
for query in queries:
    docs = retriever.retrieve(query)
    print(f"  返回文档数: {len(docs)}\n")

# 打印缓存统计
print("缓存统计:")
stats = retriever.get_cache_stats()
for key, value in stats.items():
    print(f"  {key}: {value}")
```

## 示例4:批量处理优化

```python
"""
批量处理优化示例
提高吞吐量
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import asyncio
import time

load_dotenv()

class BatchRetriever:
    """批量检索器"""

    def __init__(self, vectorstore, batch_size: int = 10):
        self.retriever = vectorstore.as_retriever()
        self.batch_size = batch_size

    def batch_retrieve(self, queries: list) -> list:
        """同步批量检索"""
        start = time.time()
        results = self.retriever.batch(queries)
        latency = time.time() - start

        print(f"同步批量检索:")
        print(f"  查询数: {len(queries)}")
        print(f"  总耗时: {latency:.3f}s")
        print(f"  平均耗时: {latency/len(queries):.3f}s")

        return results

    async def async_batch_retrieve(self, queries: list) -> list:
        """异步批量检索"""
        start = time.time()
        results = await self.retriever.abatch(queries)
        latency = time.time() - start

        print(f"\n异步批量检索:")
        print(f"  查询数: {len(queries)}")
        print(f"  总耗时: {latency:.3f}s")
        print(f"  平均耗时: {latency/len(queries):.3f}s")

        return results

    async def concurrent_retrieve(self, queries: list) -> list:
        """并发检索"""
        start = time.time()
        tasks = [self.retriever.ainvoke(q) for q in queries]
        results = await asyncio.gather(*tasks)
        latency = time.time() - start

        print(f"\n并发检索:")
        print(f"  查询数: {len(queries)}")
        print(f"  总耗时: {latency:.3f}s")
        print(f"  平均耗时: {latency/len(queries):.3f}s")

        return results

# 使用示例
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore.add_texts([
    "LangChain is a framework",
    "LCEL is a language",
    "RAG is a technique"
])

retriever = BatchRetriever(vectorstore)

queries = [
    "What is LangChain?",
    "What is LCEL?",
    "What is RAG?",
    "Tell me about LangChain",
    "Explain LCEL"
]

# 同步批量
results = retriever.batch_retrieve(queries)

# 异步批量
async def test_async():
    results = await retriever.async_batch_retrieve(queries)
    results = await retriever.concurrent_retrieve(queries)

asyncio.run(test_async())
```

## 示例5:健康检查

```python
"""
健康检查示例
监控系统状态
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import time

load_dotenv()

class HealthChecker:
    """健康检查器"""

    def __init__(self, vectorstore):
        self.retriever = vectorstore.as_retriever()
        self.test_query = "health check"

    def check_health(self) -> dict:
        """执行健康检查"""
        health = {
            "status": "unknown",
            "latency": None,
            "error": None
        }

        try:
            # 执行测试查询
            start = time.time()
            docs = self.retriever.invoke(self.test_query)
            latency = time.time() - start

            # 检查结果
            if docs is not None:
                health["status"] = "healthy"
                health["latency"] = latency
            else:
                health["status"] = "degraded"
                health["error"] = "No documents returned"

        except Exception as e:
            health["status"] = "unhealthy"
            health["error"] = str(e)

        return health

# 使用示例
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
vectorstore.add_texts(["test document"])

checker = HealthChecker(vectorstore)
health = checker.check_health()

print("健康检查结果:")
for key, value in health.items():
    print(f"  {key}: {value}")
```
