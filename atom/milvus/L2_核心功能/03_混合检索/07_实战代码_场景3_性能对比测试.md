# 实战代码 - 场景3：性能对比测试

> **目标**：对比纯向量检索 vs 混合检索的性能差异，掌握性能测试方法

---

## 场景描述

通过系统性的性能测试，对比以下检索方式：
- **纯稠密向量检索**：语义搜索
- **纯稀疏向量检索**：BM25 关键词搜索
- **混合检索（RRF）**：稠密 + 稀疏融合
- **混合检索（Weighted）**：加权融合

**测试维度**：
- 检索延迟（Latency）
- 召回率（Recall）
- 精确度（Precision）
- 资源消耗（CPU/Memory）

```python
                       metric_type="BM25")

# 创建 Collection
client.create_collection(
    collection_name="performance_test",
```

## 性能测试框架

### 1. 测试工具类

```python
import time
import numpy as np
from typing import List, Dict, Callable

class PerformanceTester:
    """性能测试工具类"""
    
    def __init__(self, client, collection_name, model):
        self.client = client
        self.collection_name = collection_name
        self.model = model
    
    def measure_latency(self, search_func: Callable, num_runs: int = 10) -> Dict:
        """测量检索延迟"""
        latencies = []
        
        for _ in range(num_runs):
            start_time = time.time()
            search_func()
            end_time = time.time()
            latencies.append((end_time - start_time) * 1000)  # 转换为毫秒
        
        return {
            "mean": np.mean(latencies),
            "std": np.std(latencies),
            "min": np.min(latencies),
            "max": np.max(latencies),
            "p50": np.percentile(latencies, 50),
            "p95": np.percentile(latencies, 95),
            "p99": np.percentile(latencies, 99)
        }
    
    def calculate_recall(self, retrieved_ids: List[int], 
                        ground_truth_ids: List[int]) -> float:
        """计算召回率"""
        retrieved_set = set(retrieved_ids)
        ground_truth_set = set(ground_truth_ids)
        
        if len(ground_truth_set) == 0:
            return 0.0
        
        intersection = retrieved_set.intersection(ground_truth_set)
        return len(intersection) / len(ground_truth_set)
    
    def calculate_precision(self, retrieved_ids: List[int], 
                           ground_truth_ids: List[int]) -> float:
        """计算精确度"""
        retrieved_set = set(retrieved_ids)
        ground_truth_set = set(ground_truth_ids)
        
        if len(retrieved_set) == 0:
            return 0.0
        
        intersection = retrieved_set.intersection(ground_truth_set)
        return len(intersection) / len(retrieved_set)

# 创建测试器
tester = PerformanceTester(client, "performance_test", model)
```

```python
def evaluate_search_quality(query_text: str, ground_truth_category: str):
    """评估检索质量"""
    query_dense = model.encode([query_text])[0].tolist()
    
    # 获取 ground truth（同类别的所有产品）
    ground_truth_ids = [p["id"] for p in test_data if p["category"] == ground_truth_category]
    
    # 测试各种检索方法
    results = {}
    
    # 1. 纯稠密向量
    res_dense = client.search(
        collection_name="performance_test",
        data=[query_dense],
        anns_field="text_dense",
        limit=20,
        output_fields=["id"]
    )
    dense_ids = [hit["id"] for hits in res_dense for hit in hits]
    
    # 2. 纯稀疏向量
    res_sparse = client.search(
        collection_name="performance_test",
        data=[query_text],
        anns_field="text_sparse",
        limit=20,
        output_fields=["id"]
    )
    sparse_ids = [hit["id"] for hits in res_sparse for hit in hits]
    
    # 3. 混合检索（RRF）
    request_dense = AnnSearchRequest(data=[query_dense], anns_field="text_dense", limit=20)
    request_sparse = AnnSearchRequest(data=[query_text], anns_field="text_sparse", limit=20)
    ranker = RRFRanker(k=60)
    
    res_hybrid = client.hybrid_search(
        collection_name="performance_test",
        reqs=[request_dense, request_sparse],
        ranker=ranker,
        limit=20,
        output_fields=["id"]
    )
    hybrid_ids = [hit["id"] for hits in res_hybrid for hit in hits]
    
    # 计算指标
    results["纯稠密向量"] = {
        "recall": tester.calculate_recall(dense_ids, ground_truth_ids),
        "precision": tester.calculate_precision(dense_ids, ground_truth_ids)
    }
    
    results["纯稀疏向量"] = {
        "recall": tester.calculate_recall(sparse_ids, ground_truth_ids),
        "precision": tester.calculate_precision(sparse_ids, ground_truth_ids)
    }
    
    results["混合检索(RRF)"] = {
        "recall": tester.calculate_recall(hybrid_ids, ground_truth_ids),
        "precision": tester.calculate_precision(hybrid_ids, ground_truth_ids)
    }
    
    return results

# 测试多个查询
test_queries = [
    ("Apple 旗舰手机", "手机"),
    ("轻薄笔记本电脑", "笔记本"),
    ("降噪耳机", "耳机")
]

quality_results = []
for query, category in test_queries:
    print(f"\n🔍 查询: {query} (类别: {category})")
    results = evaluate_search_quality(query, category)
    
    for method, metrics in results.items():
        print(f"  {method}:")
        print(f"    召回率: {metrics['recall']:.2%}")
        print(f"    精确度: {metrics['precision']:.2%}")
        
        quality_results.append({
            "查询": query,
            "方法": method,
            "召回率": metrics['recall'],
            "精确度": metrics['precision']
        })

# 汇总质量指标
quality_df = pd.DataFrame(quality_results)
avg_quality = quality_df.groupby("方法")[["召回率", "精确度"]].mean()

print("\n📊 平均质量指标：")
print(avg_quality.to_string())
```

## 资源消耗测试

```python
import psutil
import os

def measure_resource_usage(search_func, num_runs=10):
    """测量资源消耗"""
    process = psutil.Process(os.getpid())
    
    # 初始资源使用
    initial_memory = process.memory_info().rss / 1024 / 1024  # MB
    initial_cpu = process.cpu_percent(interval=0.1)
    
    # 执行检索
    for _ in range(num_runs):
        search_func()
    
    # 最终资源使用
    final_memory = process.memory_info().rss / 1024 / 1024  # MB
    final_cpu = process.cpu_percent(interval=0.1)
    
    return {
        "memory_delta": final_memory - initial_memory,
        "cpu_usage": final_cpu
    }

# 测试资源消耗
print("\n📊 资源消耗测试：")

resource_dense = measure_resource_usage(test_dense_only)
print(f"  纯稠密向量: 内存增量={resource_dense['memory_delta']:.2f}MB, CPU={resource_dense['cpu_usage']:.1f}%")

resource_sparse = measure_resource_usage(test_sparse_only)
print(f"  纯稀疏向量: 内存增量={resource_sparse['memory_delta']:.2f}MB, CPU={resource_sparse['cpu_usage']:.1f}%")
```
