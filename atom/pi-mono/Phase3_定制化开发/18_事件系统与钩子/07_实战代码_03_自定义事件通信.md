        result = cachedData.dependencies;
        break;
      case 'scripts':
        result = cachedData.scripts;
        break;
      case 'metadata':
        result = cachedData.metadata;
        break;
      case 'all':
        result = cachedData;
        break;

        console.log(`  âœ— Unknown query: ${query}`);

          requestId,

  /**

  function analyzeProjectData(data: any): AnalysisResult {

      complexity = 'medium';
    }

    // ç”Ÿæˆå»ºè®®
    const recommendations: string[] = [];

    if (data.files.length > 100) {
      recommendations.push('Consider splitting the project into modules');

    if (totalDependencies > 50) {
      recommendations.push('Review dependencies and remove unused ones');
    }

    if (!data.scripts.test) {
      recommendations.push('Add test scripts to package.json');
    }

    const result: AnalysisResult = {
      totalFiles: data.files.length,
      filesByExtension,
      totalDependencies,
      outdatedDependencies,
      complexity,
      recommendations
    };

    console.log(`  âœ“ Analysis complete`);
    console.log(`    Total files: ${result.totalFiles}`);
    console.log(`    Complexity: ${result.complexity}`);
    console.log(`    Recommendations: ${result.recommendations.length}`);

    return result;
