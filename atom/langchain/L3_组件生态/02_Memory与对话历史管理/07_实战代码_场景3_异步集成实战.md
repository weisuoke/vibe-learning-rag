## 场景概述

- **阻塞问题**：同步数据库操作阻塞主线程
- **并发限制**：无法充分利用异步I/O
- **响应延迟**：用户等待时间过长

本场景演示如何使用异步操作提升对话历史管理的性能。

**核心要点**：
- 异步 RunnableWithMessageHistory
- PostgreSQL 异步连接（asyncpg）
- Redis 异步操作（aioredis）
- 并发性能测试

## 实战示例1：异步 RunnableWithMessageHistory

### 核心特性

- **非阻塞**：异步操作不阻塞主线程
- **高并发**：支持多个并发请求
- **性能提升**：显著降低响应延迟

### 完整代码

```python
"""
异步 RunnableWithMessageHistory 示例
演示如何使用异步操作提升性能
"""

import asyncio
from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from dotenv import load_dotenv

load_dotenv()

# ============================================================
# 1. 异步 Session History 工厂函数
# ============================================================

# 存储字典
store = {}

async def get_async_session_history(session_id: str) -> InMemoryChatMessageHistory:
    """
    异步获取对话历史

    参数:
        session_id: 会话ID

    返回:
        InMemoryChatMessageHistory 实例
    """
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()
    return store[session_id]

# ============================================================
# 2. 创建异步对话链
# ============================================================

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

conversation = RunnableWithMessageHistory(
    llm,
    get_async_session_history,
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 3. 异步对话测试
# ============================================================

async def test_async_conversation():
    """测试异步对话"""

    session_id = "async_user_001"
    config = {"configurable": {"session_id": session_id}}

    print("=== 异步对话测试 ===\n")

    # 第一轮对话（异步）
    print("👤 用户: 我是一名数据科学家")
    response1 = await conversation.ainvoke(
        {"input": "我是一名数据科学家"},
        config=config
    )
    print(f"🤖 AI: {response1.content}\n")

    # 第二轮对话（异步）
    print("👤 用户: 我的职业是什么？")
    response2 = await conversation.ainvoke(
        {"input": "我的职业是什么？"},
        config=config
    )
    print(f"🤖 AI: {response2.content}\n")

    # 验证历史
    history = await get_async_session_history(session_id)
    messages = await history.aget_messages()  # 异步获取消息
    print(f"✅ 存储的消息数量: {len(messages)}")

# 运行异步测试
if __name__ == "__main__":
    asyncio.run(test_async_conversation())
```


### 关键要点

1. **异步方法**：使用 `ainvoke()` 而非 `invoke()`
2. **异步获取**：使用 `aget_messages()` 而非 `messages` 属性
3. **事件循环**：使用 `asyncio.run()` 运行异步函数

## 实战示例2：PostgreSQL 异步集成

### 核心特性

- **asyncpg**：高性能异步 PostgreSQL 驱动
- **连接池**：异步连接池管理
- **批量操作**：异步批量插入和查询

### 完整代码

```python
"""
PostgreSQL 异步集成示例
演示如何使用 asyncpg 实现高性能异步存储
"""

import asyncio
import asyncpg
from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.messages import HumanMessage, AIMessage
from langchain_core.chat_history import BaseChatMessageHistory
from typing import List
from dotenv import load_dotenv
import os

load_dotenv()

# ============================================================
# 1. 自定义异步 PostgreSQL Chat Message History
# ============================================================

class AsyncPostgresChatMessageHistory(BaseChatMessageHistory):
    """
    异步 PostgreSQL 对话历史存储

    使用 asyncpg 实现高性能异步操作
    """

    def __init__(self, session_id: str, pool: asyncpg.Pool):
        self.session_id = session_id
        self.pool = pool
        self._messages = []

    async def aget_messages(self) -> List:
        """异步获取消息"""
        query = """
            SELECT role, content FROM chat_history
            WHERE session_id = $1
            ORDER BY created_at ASC
        """
        rows = await self.pool.fetch(query, self.session_id)

        messages = []
        for row in rows:
            if row['role'] == 'human':
                messages.append(HumanMessage(content=row['content']))
            elif row['role'] == 'ai':
                messages.append(AIMessage(content=row['content']))

        return messages

    async def aadd_message(self, message) -> None:
        """异步添加单条消息"""
        role = 'human' if isinstance(message, HumanMessage) else 'ai'
        query = """
            INSERT INTO chat_history (session_id, role, content)
            VALUES ($1, $2, $3)
        """
        await self.pool.execute(query, self.session_id, role, message.content)

    async def aadd_messages(self, messages: List) -> None:
        """异步批量添加消息"""
        records = [
            (
                self.session_id,
                'human' if isinstance(msg, HumanMessage) else 'ai',
                msg.content
            )
            for msg in messages
        ]

        query = """
            INSERT INTO chat_history (session_id, role, content)
            VALUES ($1, $2, $3)
        """
        await self.pool.executemany(query, records)

    async def aclear(self) -> None:
        """异步清空历史"""
        query = "DELETE FROM chat_history WHERE session_id = $1"
        await self.pool.execute(query, self.session_id)

    # 同步方法（兼容性）
    @property
    def messages(self) -> List:
        """同步获取消息（不推荐）"""
        raise NotImplementedError("请使用 aget_messages() 异步方法")

    def add_message(self, message) -> None:
        """同步添加消息（不推荐）"""
        raise NotImplementedError("请使用 aadd_message() 异步方法")

    def clear(self) -> None:
        """同步清空（不推荐）"""
        raise NotImplementedError("请使用 aclear() 异步方法")

# ============================================================
# 2. 初始化数据库
# ============================================================

async def init_database(pool: asyncpg.Pool):
    """初始化数据库表"""
    query = """
        CREATE TABLE IF NOT EXISTS chat_history (
            id SERIAL PRIMARY KEY,
            session_id VARCHAR(255) NOT NULL,
            role VARCHAR(50) NOT NULL,
            content TEXT NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        );
        CREATE INDEX IF NOT EXISTS idx_session_id ON chat_history(session_id);
    """
    await pool.execute(query)
    print("✅ 数据库表初始化完成")

# ============================================================
# 3. 创建连接池
# ============================================================

async def create_pool():
    """创建异步连接池"""
    DATABASE_URL = os.getenv(
        "DATABASE_URL",
        "postgresql://user:password@localhost:5432/langchain_db"
    )

    pool = await asyncpg.create_pool(
        DATABASE_URL,
        min_size=5,  # 最小连接数
        max_size=20,  # 最大连接数
        command_timeout=60  # 命令超时（秒）
    )

    print("✅ 异步连接池创建完成")
    return pool

# ============================================================
# 4. Session History 工厂函数
# ============================================================

# 全局连接池
_pool = None

async def get_postgres_session_history(session_id: str):
    """获取 PostgreSQL 异步对话历史"""
    global _pool
    if _pool is None:
        _pool = await create_pool()
        await init_database(_pool)

    return AsyncPostgresChatMessageHistory(session_id, _pool)

# ============================================================
# 5. 创建异步对话链
# ============================================================

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

conversation = RunnableWithMessageHistory(
    llm,
    get_postgres_session_history,
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 6. 测试异步对话
# ============================================================

async def test_async_postgres():
    """测试 PostgreSQL 异步集成"""

    session_id = "async_pg_user_001"
    config = {"configurable": {"session_id": session_id}}

    print("\n=== PostgreSQL 异步集成测试 ===\n")

    # 第一轮对话
    print("👤 用户: 我喜欢机器学习")
    response1 = await conversation.ainvoke(
        {"input": "我喜欢机器学习"},
        config=config
    )
    print(f"🤖 AI: {response1.content}\n")

    # 第二轮对话
    print("👤 用户: 我喜欢什么？")
    response2 = await conversation.ainvoke(
        {"input": "我喜欢什么？"},
        config=config
    )
    print(f"🤖 AI: {response2.content}\n")

    # 验证数据库
    history = await get_postgres_session_history(session_id)
    messages = await history.aget_messages()
    print(f"✅ 存储的消息数量: {len(messages)}")

    # 关闭连接池
    global _pool
    if _pool:
        await _pool.close()
        print("✅ 连接池已关闭")

# 运行测试
if __name__ == "__main__":
    asyncio.run(test_async_postgres())
```

```

✅ 数据库表初始化完成

🤖 AI: 机器学习是一个非常有趣的领域！有很多应用场景。

🤖 AI: 你喜欢机器学习。
```

### 关键要点

2. **批量操作**：使用 `executemany()` 批量插入

## 实战示例3：并发性能测试

### 测试目标

对比同步和异步操作的性能差异。

### 完整代码

```python
"""
并发性能测试
对比同步和异步操作的性能
"""

import asyncio
import time
from typing import List

# ============================================================
# 1. 同步性能测试
# ============================================================

def sync_conversation_test(num_requests: int = 10):
    """同步对话测试"""
    from langchain_openai import ChatOpenAI
    from langchain_core.runnables.history import RunnableWithMessageHistory
    from langchain_core.chat_history import InMemoryChatMessageHistory

    store = {}

    def get_session_history(session_id: str):
        if session_id not in store:
            store[session_id] = InMemoryChatMessageHistory()
        return store[session_id]

    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)
    conversation = RunnableWithMessageHistory(
        llm,
        get_session_history,
        input_messages_key="input",
        history_messages_key="history"
    )

    start_time = time.time()

    for i in range(num_requests):
        config = {"configurable": {"session_id": f"sync_user_{i}"}}
        conversation.invoke({"input": f"测试消息 {i}"}, config=config)

    elapsed_time = time.time() - start_time
    return elapsed_time

# ============================================================
# 2. 异步性能测试
# ============================================================

async def async_conversation_test(num_requests: int = 10):
    """异步对话测试"""
    from langchain_openai import ChatOpenAI
    from langchain_core.runnables.history import RunnableWithMessageHistory
    from langchain_core.chat_history import InMemoryChatMessageHistory

    store = {}

    async def get_session_history(session_id: str):
        if session_id not in store:
            store[session_id] = InMemoryChatMessageHistory()
        return store[session_id]

    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)
    conversation = RunnableWithMessageHistory(
        llm,
        get_session_history,
        input_messages_key="input",
        history_messages_key="history"
    )

    start_time = time.time()

    # 并发执行所有请求
    tasks = []
    for i in range(num_requests):
        config = {"configurable": {"session_id": f"async_user_{i}"}}
        task = conversation.ainvoke({"input": f"测试消息 {i}"}, config=config)
        tasks.append(task)

    await asyncio.gather(*tasks)

    elapsed_time = time.time() - start_time
    return elapsed_time

# ============================================================
# 3. 运行性能对比
# ============================================================

async def run_performance_test():
    """运行性能对比测试"""

    num_requests = 10

    print("=== 并发性能测试 ===\n")
    print(f"测试请求数: {num_requests}\n")

    # 同步测试
    print("🔄 运行同步测试...")
    sync_time = sync_conversation_test(num_requests)
    print(f"✅ 同步测试完成: {sync_time:.2f}秒\n")

    # 异步测试
    print("🔄 运行异步测试...")
    async_time = await async_conversation_test(num_requests)
    print(f"✅ 异步测试完成: {async_time:.2f}秒\n")

    # 性能对比
    print("=== 性能对比 ===")
    print(f"同步耗时: {sync_time:.2f}秒")
    print(f"异步耗时: {async_time:.2f}秒")
    print(f"性能提升: {(sync_time / async_time):.2f}x")
    print(f"时间节省: {((sync_time - async_time) / sync_time * 100):.1f}%")

if __name__ == "__main__":
    asyncio.run(run_performance_test())
```

```python
"""
异步错误处理
"""

async def safe_async_invoke(conversation, input_text, config):
    """安全的异步调用"""
    try:
        response = await conversation.ainvoke(
            {"input": input_text},
            config=config
        )
        return response
    except asyncio.TimeoutError:
        print("⚠️ 请求超时")
        return None
    except Exception as e:
        print(f"⚠️ 错误: {e}")
        return None
```

### 4. 并发控制

```python
"""
并发控制
"""

import asyncio
from asyncio import Semaphore

async def controlled_concurrent_requests(
    conversation,
    requests: List,
    max_concurrent: int = 10
):
    """
    控制并发数量

    参数:
        conversation: 对话链
        requests: 请求列表
        max_concurrent: 最大并发数
    """
    semaphore = Semaphore(max_concurrent)

    async def process_request(request):
        async with semaphore:
            return await conversation.ainvoke(
                {"input": request['input']},
                config=request['config']
            )

    tasks = [process_request(req) for req in requests]
    results = await asyncio.gather(*tasks)
    return results
```

## 常见问题

### Q1: 为什么异步操作更快？

**A**: 异步操作的优势：
- **非阻塞I/O**：等待数据库响应时不阻塞主线程
- **并发执行**：多个请求可以同时进行
- **资源利用**：充分利用CPU和网络资源

### Q2: 如何处理异步上下文中的同步调用？

**A**: 使用 `asyncio.to_thread()` 包装同步调用：
```python
import asyncio

# 在异步函数中调用同步函数
result = await asyncio.to_thread(sync_function, arg1, arg2)
```

### Q3: 异步连接池如何管理？

**A**: 最佳实践：
```python
# 应用启动时创建连接池
pool = await create_pool()

# 应用运行时复用连接池
history = AsyncPostgresChatMessageHistory(session_id, pool)

# 应用关闭时清理连接池
await pool.close()
```
