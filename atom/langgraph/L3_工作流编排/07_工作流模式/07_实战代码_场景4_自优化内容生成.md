## 场景描述

构建一个**自优化代码生成器**：生成代码 → 静态检查 → 不合格则带反馈重新生成，直到通过检查或达到最大次数。

模式：**评估者-优化者模式**

```
需求输入 → 代码生成 → 代码检查 ──通过──→ 输出代码
                         │
                    不通过+反馈
                         │
                         └──→ 重新生成（带反馈）
```

---

## 完整代码

```python
"""
自优化代码生成器
模式：评估者-优化者（生成→评估→反馈→改进循环）

生成 Python 函数 → 静态检查（语法、命名、文档） → 不合格则改进
"""
import ast
import re
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


# ===== 1. 定义状态 =====
class CodeGenState(TypedDict):
    requirement: str        # 需求描述
    generated_code: str     # 生成的代码
    check_results: list[dict]  # 检查结果列表
    feedback: str           # 反馈信息
    grade: str              # pass / fail
    attempts: int           # 尝试次数
    max_attempts: int       # 最大尝试次数


# ===== 2. 代码生成器 =====
def code_generator(state: CodeGenState) -> dict:
    """生成或改进代码"""
    requirement = state["requirement"]
    feedback = state.get("feedback", "")
    attempts = state.get("attempts", 0) + 1

    if feedback and attempts > 1:
        # 带反馈改进：模拟根据反馈修改代码
        print(f"[生成器] 第{attempts}次尝试，根据反馈改进...")
        code = _improve_code(state["generated_code"], feedback)
    else:
        # 首次生成：模拟 LLM 生成代码
        print(f"[生成器] 第{attempts}次尝试，首次生成...")
        code = _initial_generate(requirement)

    print(f"[生成器] 生成代码 ({len(code)} 字符)")
    return {"generated_code": code, "attempts": attempts}


def _initial_generate(requirement: str) -> str:
    """模拟首次代码生成（故意有问题，触发优化循环）"""
    # 故意生成有问题的代码：缺少文档字符串、变量命名不规范
    return '''
def calc(x, y):
    r = x + y
    return r
'''


def _improve_code(code: str, feedback: str) -> str:
    """模拟根据反馈改进代码"""
    improved = code

    if "缺少文档字符串" in feedback:
        improved = improved.replace(
            "def calc(x, y):",
            'def calc(x, y):\n    """计算两个数的和"""'
        )

    if "变量命名不规范" in feedback:
        improved = improved.replace("r = ", "result = ")
        improved = improved.replace("return r", "return result")

    if "缺少类型注解" in feedback:
        improved = improved.replace(
            "def calc(x, y):",
            "def calc(x: int, y: int) -> int:"
        )

    return improved


# ===== 3. 代码检查器（评估器）=====
def code_checker(state: CodeGenState) -> dict:
    """检查代码质量"""
    code = state["generated_code"]
    issues = []

    # 检查1：语法检查
    try:
        ast.parse(code)
    except SyntaxError as e:
        issues.append({
            "type": "syntax_error",
            "severity": "critical",
            "message": f"语法错误: {e}",
        })

    # 检查2：文档字符串
    if '"""' not in code and "'''" not in code:
        issues.append({
            "type": "missing_docstring",
            "severity": "warning",
            "message": "缺少文档字符串",
        })

    # 检查3：变量命名（单字母变量）
    single_letter_vars = re.findall(r'\b([a-z])\s*=', code)
    if single_letter_vars:
        issues.append({
            "type": "naming_convention",
            "severity": "warning",
            "message": f"变量命名不规范，发现单字母变量: {single_letter_vars}",
        })

    # 检查4：类型注解
    if "->" not in code:
        issues.append({
            "type": "missing_type_hints",
            "severity": "info",
            "message": "缺少类型注解",
        })

    # 判定结果
    critical_issues = [i for i in issues if i["severity"] == "critical"]
    warning_issues = [i for i in issues if i["severity"] == "warning"]

    if critical_issues:
        grade = "fail"
    elif len(warning_issues) >= 2:
        grade = "fail"
    else:
        grade = "pass"

    feedback = "\n".join(f"- [{i['severity']}] {i['message']}" for i in issues)
    if not feedback:
        feedback = "所有检查通过"

    print(f"[检查器] 发现 {len(issues)} 个问题，评级: {grade}")
    for issue in issues:
        print(f"  [{issue['severity']}] {issue['message']}")

    return {
        "check_results": issues,
        "grade": grade,
        "feedback": feedback,
    }


# ===== 4. 路由函数 =====
def should_continue(state: CodeGenState) -> str:
    """决定是否继续优化"""
    if state["grade"] == "pass":
        print(f"[路由] ✅ 代码通过检查")
        return "accepted"

    max_attempts = state.get("max_attempts", 3)
    if state["attempts"] >= max_attempts:
        print(f"[路由] ⚠️ 达到最大尝试次数 ({max_attempts})")
        return "max_reached"

    print(f"[路由] 🔄 需要改进，继续优化")
    return "retry"


# ===== 5. 构建图 =====
builder = StateGraph(CodeGenState)

builder.add_node("generator", code_generator)
builder.add_node("checker", code_checker)

builder.add_edge(START, "generator")
builder.add_edge("generator", "checker")

builder.add_conditional_edges("checker", should_continue, {
    "accepted": END,
    "retry": "generator",
    "max_reached": END,
})

code_optimizer = builder.compile()


# ===== 6. 测试 =====
if __name__ == "__main__":
    print("=" * 60)
    print("自优化代码生成器")
    print("=" * 60)

    result = code_optimizer.invoke({
        "requirement": "写一个计算两数之和的函数",
        "attempts": 0,
        "max_attempts": 3,
    })

    print(f"\n{'='*60}")
    print(f"最终代码（经过 {result['attempts']} 次迭代）:")
    print("=" * 60)
    print(result["generated_code"])
    print(f"\n最终评级: {result['grade']}")
    print(f"检查反馈: {result['feedback']}")
```

## 关键设计解析

### 1. 确定性评估器

本例用 `ast.parse` 和正则表达式做代码检查，而非 LLM 评估。确定性评估器更可靠：

```python
# ✅ 确定性检查：结果稳定可靠
ast.parse(code)  # 语法检查
re.findall(r'\b([a-z])\s*=', code)  # 命名检查

# ❌ LLM 评估：结果可能不稳定
llm.invoke("评估这段代码的质量")  # 每次可能给不同结果
```

### 2. 分级严重度

```python
# critical → 必须修复（语法错误）
# warning → 应该修复（命名、文档）
# info → 建议修复（类型注解）
```

只有 critical 或多个 warning 才判定为 fail，避免过度严格导致无限循环。

### 3. 最大次数保护

```python
if state["attempts"] >= max_attempts:
    return "max_reached"  # 强制终止，防止无限循环
```

## 学习要点

1. **评估者-优化者模式的完整实现**：生成→检查→反馈→改进循环
2. **确定性评估优于 LLM 评估**：能用规则检查就不用 LLM
3. **终止条件设计**：质量达标 OR 最大次数，双重保护
4. **反馈传递**：检查器的具体反馈驱动生成器的定向改进
