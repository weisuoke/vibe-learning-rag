// src/channel/channel-manager.ts
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export interface Channel {
  id: string;
  type: string;
  status: 'connected' | 'disconnected' | 'degraded';
  lastActivity?: string;
}

export class ChannelManager {
  async list(): Promise<Channel[]> {
    const { stdout } = await execAsync('openclaw channels list --json');
    const result = JSON.parse(stdout);
    return result.channels;
  }

  async getStatus(): Promise<Record<string, string>> {
    const { stdout } = await execAsync('openclaw channels status --json');
    return JSON.parse(stdout);
  }

  async add(type: string, options?: Record<string, string>): Promise<void> {
    let command = `openclaw channels add ${type}`;
    if (options) {
      for (const [key, value] of Object.entries(options)) {
        command += ` --${key} ${value}`;
      }
    }
    await execAsync(command);
  }

  async remove(channelId: string): Promise<void> {
    await execAsync(`openclaw channels remove ${channelId}`);
  }

  async isHealthy(channelId: string): Promise<boolean> {
    const status = await this.getStatus();
    return status[channelId] === 'connected';
  }
}

// src/channel/channel-monitor.ts
import { EventEmitter } from 'events';

export class ChannelMonitor extends EventEmitter {
  private manager = new ChannelManager();
  private interval: NodeJS.Timeout | null = null;
  private previousStatus: Record<string, string> = {};

  start(checkInterval = 30000): void {
    this.interval = setInterval(async () => {
      await this.checkChannels();
    }, checkInterval);
  }

  stop(): void {
    if (this.interval) {
      clearInterval(this.interval);
      this.interval = null;
    }
  }

  private async checkChannels(): Promise<void> {
    try {
      const channels = await this.manager.list();

      for (const channel of channels) {
        const previousStatus = this.previousStatus[channel.id];
        const currentStatus = channel.status;

        if (previousStatus && previousStatus !== currentStatus) {
          this.emit('status-changed', {
            channelId: channel.id,
            from: previousStatus,
            to: currentStatus,
          });

          if (currentStatus === 'disconnected') {
            this.emit('channel-disconnected', channel);
          }
        }

        this.previousStatus[channel.id] = currentStatus;
      }
    } catch (error) {
      this.emit('error', error);
    }
  }
}

// src/channel/channel-router.ts
export class ChannelRouter {
  private manager = new ChannelManager();

  async route(to: string): Promise<string> {
    const channelType = this.detectChannelType(to);
    const channels = await this.manager.list();
    const available = channels.filter(
      c => c.type === channelType && c.status === 'connected'
    );

    if (available.length === 0) {
      throw new Error(`No available ${channelType} channels`);
    }

    return available[0].id;
  }

  private detectChannelType(recipient: string): string {
    if (/^\+?\d{10,15}$/.test(recipient)) return 'whatsapp';
    if (recipient.startsWith('@')) return 'telegram';
    if (recipient.startsWith('#')) return 'slack';
    return 'whatsapp';
  }
}

// src/channel/auto-reconnect.ts
export class AutoReconnect {
  private manager = new ChannelManager();
  private maxRetries = 3;
  private retryDelay = 5000;

  async reconnectChannel(channelId: string): Promise<boolean> {
    for (let attempt = 1; attempt <= this.maxRetries; attempt++) {
      try {
        console.log(`🔄 Reconnecting ${channelId} (attempt ${attempt}/${this.maxRetries})`);

        await this.manager.remove(channelId);
        await this.delay(2000);

        const type = channelId.split('-')[0];
        await this.manager.add(type);

        await this.delay(3000);
        const healthy = await this.manager.isHealthy(channelId);

        if (healthy) {
          console.log(`✅ ${channelId} reconnected successfully`);
          return true;
        }
      } catch (error) {
        console.error(`❌ Reconnect attempt ${attempt} failed:`, error);
      }

      if (attempt < this.maxRetries) {
        await this.delay(this.retryDelay * attempt);
      }
    }

    return false;
  }

  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

import { ChannelMonitor } from './channel';

async function main() {
  const monitor = new ChannelMonitor();

  monitor.on('status-changed', ({ channelId, from, to }) => {
    console.log(`📊 ${channelId}: ${from} → ${to}`);

// examples/auto-reconnect.ts
import { ChannelMonitor, AutoReconnect } from './channel';

async function main() {
  const monitor = new ChannelMonitor();
  const reconnect = new AutoReconnect();

  monitor.on('channel-disconnected', async (channel) => {
    console.log(`🔄 Attempting to reconnect ${channel.id}...`);
    const success = await reconnect.reconnectChannel(channel.id);

    if (success) {
      console.log(`✅ ${channel.id} reconnected`);
    } else {
      console.error(`❌ Failed to reconnect ${channel.id}`);
      // 发送告警
    }
  });

  monitor.start();
}

main().catch(console.error);

class LoadBalancer {
  private currentIndex = 0;

  async getNextChannel(type: string): Promise<string> {
    const manager = new ChannelManager();
    const channels = await manager.list();
    const available = channels.filter(c => c.type === type && c.status === 'connected');

    if (available.length === 0) {
      throw new Error(`No available ${type} channels`);
    }

    const channel = available[this.currentIndex % available.length];
    this.currentIndex++;
    return channel.id;
  }
}
