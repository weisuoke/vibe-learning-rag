> 掌握以下 20% 的核心知识，就能解决 80% 的动态图构建场景

掌握以下内容，就能开始使用 LangGraph 动态图构建：

---

## 4.1 条件边基础（最核心 - 必须掌握）

条件边是动态图的基石。一句话：**根据当前状态，运行时决定下一步去哪个节点。**

### 核心 API

```python
graph.add_conditional_edges(
    source,        # 从哪个节点出发
    path,          # 路由函数：接收 state，返回目标节点名
    path_map=None, # 可选：返回值 → 节点名的映射
)
```

### 完整可运行示例

```python
"""条件边最小示例：根据分数决定通过/不通过"""
from typing import TypedDict
from langgraph.graph import StateGraph, START, END


# 1. 定义状态
class State(TypedDict):
    score: int
    result: str


# 2. 定义节点
def evaluate(state: State) -> dict:
    """评估节点：计算分数"""
    print(f"[评估] 分数: {state['score']}")
    return {}  # 分数已在输入中，不需要更新


def pass_handler(state: State) -> dict:
    return {"result": f"恭喜通过！分数: {state['score']}"}


def fail_handler(state: State) -> dict:
    return {"result": f"未通过，分数: {state['score']}，需要 60 分以上"}


# 3. 路由函数（核心！）
def route_by_score(state: State) -> str:
    """根据分数决定走哪条路"""
    if state["score"] >= 60:
        return "pass"
    return "fail"


# 4. 构建图
builder = StateGraph(State)
builder.add_node("evaluate", evaluate)
builder.add_node("pass_handler", pass_handler)
builder.add_node("fail_handler", fail_handler)

builder.add_edge(START, "evaluate")

# 关键：条件边 + 路由函数 + path_map
builder.add_conditional_edges("evaluate", route_by_score, {
    "pass": "pass_handler",
    "fail": "fail_handler",
})

builder.add_edge("pass_handler", END)
builder.add_edge("fail_handler", END)

graph = builder.compile()

# 5. 测试
result_pass = graph.invoke({"score": 85, "result": ""})
print(result_pass["result"])
# 输出: 恭喜通过！分数: 85

result_fail = graph.invoke({"score": 45, "result": ""})
print(result_fail["result"])
# 输出: 未通过，分数: 45，需要 60 分以上
```

### 要点速记

- **路由函数**接收 `state`，返回字符串（目标节点名或 path_map 的 key）
- **path_map** 可选，用于将语义化返回值映射到实际节点名
- 不用 path_map 时，路由函数的返回值必须和节点名完全一致
- 路由函数也可以返回 `END` 直接结束图执行

```python
# 不用 path_map 的写法（返回值 = 节点名）
def route(state):
    if state["done"]:
        return "__end__"  # END 的字符串值
    return "next_node"

graph.add_conditional_edges("check", route)
```

## 4.2 Send 动态并行（第二重要）

Send 解决的问题：**并行任务数量在编译时未知，需要运行时动态创建。**

### 核心 API

```python
from langgraph.types import Send

# Send(目标节点名, 传递给该节点的状态)
Send("process_item", {"item": "数据A"})
```

### 完整可运行示例

```python
"""Send 最小示例：动态并行处理任务列表"""
import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Send


# 1. 定义状态
class MainState(TypedDict):
    tasks: list[str]                                    # 输入：任务列表
    results: Annotated[list[str], operator.add]          # 输出：结果（用 reducer 合并）


class TaskState(TypedDict):
    task: str                                            # 单个任务的状态


# 2. 定义节点
def process_task(state: TaskState) -> dict:
    """处理单个任务（会被并行调用多次）"""
    task = state["task"]
    result = f"已完成: {task}"
    print(f"  [Worker] {result}")
    return {"results": [result]}


def summarize(state: MainState) -> dict:
    """汇总所有结果"""
    print(f"\n[汇总] 共完成 {len(state['results'])} 个任务")
    return {}


# 3. 路由函数：返回 Send 列表
def fan_out_tasks(state: MainState) -> list[Send]:
    """为每个任务创建一个 Send，实现动态并行"""
    print(f"[分发] 收到 {len(state['tasks'])} 个任务")
    return [Send("process_task", {"task": t}) for t in state["tasks"]]


# 4. 构建图
builder = StateGraph(MainState)
builder.add_node("process_task", process_task)
builder.add_node("summarize", summarize)

# Send 通过条件边触发
builder.add_conditional_edges(START, fan_out_tasks)
builder.add_edge("process_task", "summarize")
builder.add_edge("summarize", END)

graph = builder.compile()

# 5. 测试：任务数量完全由输入决定
result = graph.invoke({
    "tasks": ["写文档", "跑测试", "代码审查"],
    "results": [],
})
print(f"\n所有结果: {result['results']}")
```

**运行输出：**

```
[分发] 收到 3 个任务
  [Worker] 已完成: 写文档
  [Worker] 已完成: 跑测试
  [Worker] 已完成: 代码审查

[汇总] 共完成 3 个任务

所有结果: ['已完成: 写文档', '已完成: 跑测试', '已完成: 代码审查']
```

### 要点速记

- `Send(node, arg)` 两个参数：目标节点名 + 传递的状态
- 在条件边的路由函数中返回 `[Send(...), Send(...)]` 列表
- 每个 Send 携带独立状态，互不干扰
- 并行结果必须用 Reducer 合并（见 4.5）

## 4.3 Command 路由（第三重要）

Command 解决的问题：**在节点内部同时完成状态更新和路由决策。**

### 核心 API

```python
from langgraph.types import Command

# 同时更新状态 + 指定下一个节点
Command(update={"key": "value"}, goto="next_node")
```

### 完整可运行示例

```python
"""Command 最小示例：审批流程"""
import operator
from typing import Annotated, Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Command


# 1. 定义状态
class State(TypedDict):
    amount: int
    messages: Annotated[list[str], operator.add]
    approved: bool


# 2. 定义节点（用 Command 控制路由）
def review(state: State) -> Command[Literal["auto_approve", "manager_review"]]:
    """审核节点：金额小于 1000 自动通过，否则转经理"""
    amount = state["amount"]
    if amount < 1000:
        return Command(
            update={"messages": [f"[审核] 金额 {amount} < 1000，自动通过"]},
            goto="auto_approve",
        )
    return Command(
        update={"messages": [f"[审核] 金额 {amount} >= 1000，需经理审批"]},
        goto="manager_review",
    )


def auto_approve(state: State) -> Command[Literal["__end__"]]:
    return Command(
        update={"approved": True, "messages": ["[系统] 已自动批准"]},
        goto=END,
    )


def manager_review(state: State) -> Command[Literal["__end__"]]:
    # 实际项目中这里会有人工审批逻辑
    return Command(
        update={"approved": True, "messages": ["[经理] 已审批通过"]},
        goto=END,
    )


# 3. 构建图（注意：不需要 add_conditional_edges）
builder = StateGraph(State)
builder.add_node("review", review)
builder.add_node("auto_approve", auto_approve)
builder.add_node("manager_review", manager_review)

builder.add_edge(START, "review")
# Command 自带路由，不需要额外的边定义

graph = builder.compile()

# 4. 测试
print("--- 小额审批 ---")
r1 = graph.invoke({"amount": 500, "messages": [], "approved": False})
for msg in r1["messages"]:
    print(f"  {msg}")
print(f"  结果: {'通过' if r1['approved'] else '拒绝'}")

print("\n--- 大额审批 ---")
r2 = graph.invoke({"amount": 5000, "messages": [], "approved": False})
for msg in r2["messages"]:
    print(f"  {msg}")
print(f"  结果: {'通过' if r2['approved'] else '拒绝'}")
```

**运行输出：**

```
--- 小额审批 ---
  [审核] 金额 500 < 1000，自动通过
  [系统] 已自动批准
  结果: 通过

--- 大额审批 ---
  [审核] 金额 5000 >= 1000，需经理审批
  [经理] 已审批通过
  结果: 通过
```

### 要点速记

- `Command(update=..., goto=...)` 一步完成状态更新 + 路由
- 使用 Command 的节点不需要 `add_conditional_edges`，也不需要 `add_edge`
- 类型注解 `Command[Literal["a", "b"]]` 帮助图可视化，不影响运行
- 不要给返回 Command 的节点添加静态边（两者会同时执行）

### Command vs 条件边：怎么选？

| 场景 | 推荐 |
|------|------|
| 简单 if/else 分支 | 条件边（更直观） |
| 路由逻辑和业务逻辑紧耦合 | Command（更内聚） |
| 需要同时更新状态和路由 | Command |
| 子图到父图通信 | Command（唯一选择） |

## 4.4 图构建三步曲（必须记住）

不管多复杂的图，构建过程都是三步：

```
StateGraph(State) → add_node / add_edge → compile()
```

### 完整可运行示例

```python
"""图构建三步曲：最简完整图"""
from typing import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    message: str


# ===== 第一步：创建图构建器 =====
builder = StateGraph(State)


# ===== 第二步：添加节点和边 =====
def greet(state: State) -> dict:
    return {"message": f"你好！你说的是: {state['message']}"}

def farewell(state: State) -> dict:
    return {"message": state["message"] + " | 再见！"}

# 添加节点
builder.add_node("greet", greet)
builder.add_node("farewell", farewell)

# 添加边（定义执行顺序）
builder.add_edge(START, "greet")
builder.add_edge("greet", "farewell")
builder.add_edge("farewell", END)


# ===== 第三步：编译 =====
graph = builder.compile()
# 编译后图结构冻结，不能再添加节点或边
# 但运行时路径是动态的（如果有条件边）


# 运行
result = graph.invoke({"message": "LangGraph"})
print(result["message"])
# 输出: 你好！你说的是: LangGraph | 再见！
```

### 三步曲速记

```python
# 第一步：创建（传入状态类型）
builder = StateGraph(State)

# 第二步：组装（添加节点和边）
builder.add_node("name", function)       # 添加节点
builder.add_edge(START, "name")          # 静态边
builder.add_conditional_edges(...)       # 条件边

# 第三步：编译（冻结结构，生成可执行图）
graph = builder.compile()

# 运行
result = graph.invoke(initial_state)
```

### 常用方法速查

| 方法 | 作用 | 示例 |
|------|------|------|
| `add_node(name, fn)` | 添加节点 | `builder.add_node("a", func_a)` |
| `add_edge(a, b)` | 静态边：a → b | `builder.add_edge(START, "a")` |
| `add_conditional_edges(src, fn)` | 条件边 | `builder.add_conditional_edges("a", route)` |
| `compile()` | 编译图 | `graph = builder.compile()` |
| `invoke(state)` | 运行图 | `graph.invoke({"key": "val"})` |

## 4.5 状态 Reducer（并行必备）

### 问题：并行节点写同一个字段会怎样？

```python
# 没有 Reducer 时：后写入的覆盖先写入的
# 节点 A 返回 {"results": ["A的结果"]}
# 节点 B 返回 {"results": ["B的结果"]}
# 最终 state["results"] = ["B的结果"]  ← A 的结果丢了！
```

### 解决方案：Annotated + operator.add

```python
import operator
from typing import Annotated
from typing_extensions import TypedDict


class State(TypedDict):
    # 有 Reducer：并行结果自动合并
    results: Annotated[list, operator.add]
    # 节点 A 返回 {"results": ["A的结果"]}
    # 节点 B 返回 {"results": ["B的结果"]}
    # 最终 state["results"] = ["A的结果", "B的结果"]  ← 都保留了！
```

### 完整可运行示例

```python
"""Reducer 示例：并行节点结果合并"""
import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    results: Annotated[list[str], operator.add]  # 关键：operator.add 合并列表


def worker_a(state: State) -> dict:
    print("[Worker A] 执行中...")
    return {"results": ["A 的结果"]}


def worker_b(state: State) -> dict:
    print("[Worker B] 执行中...")
    return {"results": ["B 的结果"]}


def collector(state: State) -> dict:
    print(f"[收集器] 收到 {len(state['results'])} 个结果: {state['results']}")
    return {}


builder = StateGraph(State)
builder.add_node("worker_a", worker_a)
builder.add_node("worker_b", worker_b)
builder.add_node("collector", collector)

# 并行：START 同时指向两个 worker
builder.add_edge(START, "worker_a")
builder.add_edge(START, "worker_b")

# 汇聚：两个 worker 都指向 collector
builder.add_edge("worker_a", "collector")
builder.add_edge("worker_b", "collector")
builder.add_edge("collector", END)

graph = builder.compile()

result = graph.invoke({"results": []})
print(f"\n最终结果: {result['results']}")
```

**运行输出：**

```
[Worker A] 执行中...
[Worker B] 执行中...
[收集器] 收到 2 个结果: ['A 的结果', 'B 的结果']

最终结果: ['A 的结果', 'B 的结果']
```

### 要点速记

- `Annotated[list, operator.add]` 是最常用的 Reducer：列表拼接
- 只要有并行节点写同一个字段，就必须加 Reducer
- 没有 Reducer 的字段，后写入的值会覆盖先写入的
- Reducer 也可以自定义函数，不限于 `operator.add`

```python
# 自定义 Reducer 示例：取最大值
def max_reducer(current: int, new: int) -> int:
    return max(current, new)

class State(TypedDict):
    best_score: Annotated[int, max_reducer]
```
