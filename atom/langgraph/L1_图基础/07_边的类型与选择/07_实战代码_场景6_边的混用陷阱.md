# 实战代码：边的混用陷阱

## 场景说明

本文档展示 LangGraph 中边的类型混用时常见的陷阱和解决方案。

**核心陷阱：**
1. 条件边和普通边混用导致节点重复执行
2. Command 无法覆盖 add_edge 定义的边
3. 条件边路由函数修改状态（违反纯函数原则）
4. 条件边缺少类型注解导致可视化问题

---

## 完整可运行代码

```python
"""
LangGraph 边的混用陷阱实战示例
演示：常见的边类型混用问题和解决方案
"""

from typing import Literal, TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Command


# ===== 状态定义 =====
class State(TypedDict):
    """工作流状态"""
    message: str
    counter: int
    route: str


# ===== 陷阱1：条件边和普通边混用导致节点重复执行 =====
print("=" * 60)
print("陷阱1：条件边和普通边混用导致节点重复执行")
print("=" * 60)

def node_a_wrong(state: State) -> State:
    """节点A - 错误示例"""
    print(f"  [节点A] 执行中... counter={state['counter']}")
    return {"message": "from A", "counter": state["counter"] + 1}

def node_b_wrong(state: State) -> State:
    """节点B - 错误示例"""
    print(f"  [节点B] 执行中... counter={state['counter']}")
    return {"message": "from B", "counter": state["counter"] + 1}

def router_wrong(state: State) -> str:
    """路由函数 - 错误示例"""
    print(f"  [路由] 决策中... message={state['message']}")
    return "node_b"

# 错误示例：同时使用条件边和普通边指向同一节点
print("\n❌ 错误示例：条件边和普通边都指向 node_b")
graph_wrong = StateGraph(State)
graph_wrong.add_node("node_a", node_a_wrong)
graph_wrong.add_node("node_b", node_b_wrong)

# 同时添加条件边和普通边指向 node_b
graph_wrong.add_edge(START, "node_a")
graph_wrong.add_conditional_edges("node_a", router_wrong)  # 条件边 -> node_b
graph_wrong.add_edge("node_a", "node_b")  # 普通边 -> node_b (冲突!)
graph_wrong.add_edge("node_b", END)

app_wrong = graph_wrong.compile()

try:
    result = app_wrong.invoke({"message": "start", "counter": 0})
    print(f"\n结果: counter={result['counter']}")
    print("⚠️  注意：node_b 被执行了两次！counter 增加了 2")
except Exception as e:
    print(f"错误: {e}")

# 正确示例：只使用条件边
print("\n✅ 正确示例：只使用条件边")

def node_a_correct(state: State) -> State:
    """节点A - 正确示例"""
    print(f"  [节点A] 执行中... counter={state['counter']}")
    return {"message": "from A", "counter": state["counter"] + 1}

def node_b_correct(state: State) -> State:
    """节点B - 正确示例"""
    print(f"  [节点B] 执行中... counter={state['counter']}")
    return {"message": "from B", "counter": state["counter"] + 1}

def router_correct(state: State) -> Literal["node_b"]:
    """路由函数 - 正确示例（带类型注解）"""
    print(f"  [路由] 决策中... message={state['message']}")
    return "node_b"

graph_correct = StateGraph(State)
graph_correct.add_node("node_a", node_a_correct)
graph_correct.add_node("node_b", node_b_correct)

graph_correct.add_edge(START, "node_a")
graph_correct.add_conditional_edges("node_a", router_correct)  # 只使用条件边
graph_correct.add_edge("node_b", END)

app_correct = graph_correct.compile()
result = app_correct.invoke({"message": "start", "counter": 0})
print(f"\n结果: counter={result['counter']}")
print("✅ node_b 只执行了一次，counter 正确增加了 1")


# ===== 陷阱2：Command 无法覆盖 add_edge 定义的边 =====
print("\n" + "=" * 60)
print("陷阱2：Command 无法覆盖 add_edge 定义的边")
print("=" * 60)

def node_with_command_wrong(state: State) -> Command:
    """节点返回 Command - 错误示例"""
    print(f"  [节点] 尝试用 Command 跳转到 END")
    # 尝试用 Command 跳过预定义的边
    return Command(goto=END)

def node_next_wrong(state: State) -> State:
    """下一个节点 - 错误示例"""
    print(f"  [下一个节点] 仍然被执行了！")
    return state

print("\n❌ 错误示例：Command 无法覆盖 add_edge")
graph_cmd_wrong = StateGraph(State)
graph_cmd_wrong.add_node("node_cmd", node_with_command_wrong)
graph_cmd_wrong.add_node("node_next", node_next_wrong)

graph_cmd_wrong.add_edge(START, "node_cmd")
graph_cmd_wrong.add_edge("node_cmd", "node_next")  # 预定义的边
graph_cmd_wrong.add_edge("node_next", END)

app_cmd_wrong = graph_cmd_wrong.compile()
result = app_cmd_wrong.invoke({"message": "start", "counter": 0})
print("⚠️  注意：即使返回 Command(goto=END)，node_next 仍然被执行")

# 正确示例：使用条件边配合 Command
print("\n✅ 正确示例：使用条件边配合 Command")

def node_with_command_correct(state: State) -> Command:
    """节点返回 Command - 正确示例"""
    print(f"  [节点] 使用 Command 跳转到 END")
    return Command(goto=END)

def router_for_command(state: State) -> str:
    """路由函数 - 处理 Command"""
    # 这个路由函数不会被调用，因为 Command 会覆盖条件边
    return "node_next"

graph_cmd_correct = StateGraph(State)
graph_cmd_correct.add_node("node_cmd", node_with_command_correct)
graph_cmd_correct.add_node("node_next", node_next_wrong)

graph_cmd_correct.add_edge(START, "node_cmd")
# 使用条件边而非普通边
graph_cmd_correct.add_conditional_edges("node_cmd", router_for_command)
graph_cmd_correct.add_edge("node_next", END)

app_cmd_correct = graph_cmd_correct.compile()
result = app_cmd_correct.invoke({"message": "start", "counter": 0})
print("✅ Command 成功跳转到 END，node_next 未被执行")


# ===== 陷阱3：条件边路由函数修改状态（违反纯函数原则） =====
print("\n" + "=" * 60)
print("陷阱3：条件边路由函数修改状态（违反纯函数原则）")
print("=" * 60)

def router_with_side_effect(state: State) -> str:
    """路由函数 - 错误示例：修改状态"""
    print(f"  [路由] 尝试修改状态...")
    # ❌ 错误：路由函数不应该修改状态
    state["counter"] = 999
    state["route"] = "modified_in_router"
    return "node_b"

def node_check_state(state: State) -> State:
    """检查状态的节点"""
    print(f"  [节点B] counter={state['counter']}, route={state.get('route', 'N/A')}")
    return state

print("\n❌ 错误示例：路由函数修改状态")
graph_side_effect = StateGraph(State)
graph_side_effect.add_node("node_a", lambda s: {"message": "from A", "counter": 1})
graph_side_effect.add_node("node_b", node_check_state)

graph_side_effect.add_edge(START, "node_a")
graph_side_effect.add_conditional_edges("node_a", router_with_side_effect)
graph_side_effect.add_edge("node_b", END)

app_side_effect = graph_side_effect.compile()
result = app_side_effect.invoke({"message": "start", "counter": 0, "route": ""})
print("⚠️  注意：路由函数中的状态修改不会生效")

# 正确示例：路由函数只读取状态，状态修改在节点中完成
print("\n✅ 正确示例：路由函数是纯函数")

def router_pure(state: State) -> str:
    """路由函数 - 正确示例：纯函数"""
    print(f"  [路由] 只读取状态，不修改")
    # ✅ 正确：只读取状态，返回路由决策
    if state["counter"] > 0:
        return "node_b"
    return "node_b"

def node_modify_state(state: State) -> State:
    """节点 - 正确示例：在节点中修改状态"""
    print(f"  [节点B] 在节点中修改状态")
    return {"counter": 999, "route": "modified_in_node"}

graph_pure = StateGraph(State)
graph_pure.add_node("node_a", lambda s: {"message": "from A", "counter": 1})
graph_pure.add_node("node_b", node_modify_state)

graph_pure.add_edge(START, "node_a")
graph_pure.add_conditional_edges("node_a", router_pure)
graph_pure.add_edge("node_b", END)

app_pure = graph_pure.compile()
result = app_pure.invoke({"message": "start", "counter": 0, "route": ""})
print(f"✅ 状态正确修改: counter={result['counter']}, route={result['route']}")


# ===== 陷阱4：条件边缺少类型注解导致可视化问题 =====
print("\n" + "=" * 60)
print("陷阱4：条件边缺少类型注解导致可视化问题")
print("=" * 60)

def router_no_annotation(state: State):
    """路由函数 - 错误示例：缺少返回类型注解"""
    if state["counter"] > 5:
        return "node_high"
    return "node_low"

print("\n❌ 错误示例：缺少类型注解")
print("路由函数返回类型: <无类型注解>")
print("⚠️  可视化工具无法推断可能的路由路径")

# 正确示例：添加类型注解
def router_with_annotation(state: State) -> Literal["node_high", "node_low"]:
    """路由函数 - 正确示例：带类型注解"""
    if state["counter"] > 5:
        return "node_high"
    return "node_low"

print("\n✅ 正确示例：添加类型注解")
print("路由函数返回类型: Literal['node_high', 'node_low']")
print("✅ 可视化工具可以正确显示所有可能的路由路径")

# 或者使用 path_map 显式指定
print("\n✅ 替代方案：使用 path_map")
graph_with_map = StateGraph(State)
graph_with_map.add_node("node_a", lambda s: s)
graph_with_map.add_node("node_high", lambda s: s)
graph_with_map.add_node("node_low", lambda s: s)

graph_with_map.add_edge(START, "node_a")
graph_with_map.add_conditional_edges(
    "node_a",
    router_no_annotation,
    path_map={"node_high": "node_high", "node_low": "node_low"}  # 显式指定路径
)
graph_with_map.add_edge("node_high", END)
graph_with_map.add_edge("node_low", END)

print("✅ 使用 path_map 显式指定路由路径")


# ===== 总结：边的选择最佳实践 =====
print("\n" + "=" * 60)
print("总结：边的选择最佳实践")
print("=" * 60)

print("""
1. 互斥原则
   ❌ 避免：条件边和普通边同时指向同一节点
   ✅ 推荐：每个节点只使用一种边类型

2. 动态路由原则
   ❌ 避免：预定义边 + Command 动态路由
   ✅ 推荐：条件边 + Command 动态路由

3. 纯函数原则
   ❌ 避免：在路由函数中修改状态
   ✅ 推荐：路由函数只读取状态，状态修改在节点中完成

4. 可视化原则
   ❌ 避免：路由函数缺少类型注解
   ✅ 推荐：使用 Literal 类型注解或 path_map

5. 边的选择策略
   - 简单流程 → 普通边 (add_edge)
   - 条件分支 → 条件边 (add_conditional_edges)
   - 并行等待 → 等待边 (add_edge([node1, node2], target))
   - 动态路由 → 条件边 + Command
""")


# ===== 完整正确示例：智能客服路由系统 =====
print("\n" + "=" * 60)
print("完整正确示例：智能客服路由系统")
print("=" * 60)

class CustomerState(TypedDict):
    """客服系统状态"""
    query: str
    category: str
    priority: int
    handled: bool

def classify_query(state: CustomerState) -> CustomerState:
    """分类用户查询"""
    query = state["query"].lower()
    print(f"  [分类] 查询: {query}")

    if "退款" in query or "投诉" in query:
        return {"category": "urgent", "priority": 10}
    elif "账单" in query or "费用" in query:
        return {"category": "billing", "priority": 5}
    else:
        return {"category": "general", "priority": 1}

def route_by_priority(state: CustomerState) -> Literal["urgent_handler", "normal_handler"]:
    """根据优先级路由 - 纯函数，带类型注解"""
    print(f"  [路由] 优先级: {state['priority']}")
    if state["priority"] >= 8:
        return "urgent_handler"
    return "normal_handler"

def urgent_handler(state: CustomerState) -> CustomerState:
    """紧急处理"""
    print(f"  [紧急处理] 处理类别: {state['category']}")
    return {"handled": True}

def normal_handler(state: CustomerState) -> CustomerState:
    """普通处理"""
    print(f"  [普通处理] 处理类别: {state['category']}")
    return {"handled": True}

# 构建图
customer_graph = StateGraph(CustomerState)
customer_graph.add_node("classify", classify_query)
customer_graph.add_node("urgent_handler", urgent_handler)
customer_graph.add_node("normal_handler", normal_handler)

# 使用正确的边类型
customer_graph.add_edge(START, "classify")
customer_graph.add_conditional_edges("classify", route_by_priority)  # 条件边
customer_graph.add_edge("urgent_handler", END)
customer_graph.add_edge("normal_handler", END)

customer_app = customer_graph.compile()

# 测试不同场景
test_cases = [
    {"query": "我要退款！", "category": "", "priority": 0, "handled": False},
    {"query": "查询账单", "category": "", "priority": 0, "handled": False},
    {"query": "一般咨询", "category": "", "priority": 0, "handled": False},
]

for i, test_case in enumerate(test_cases, 1):
    print(f"\n测试用例 {i}: {test_case['query']}")
    result = customer_app.invoke(test_case)
    print(f"结果: category={result['category']}, priority={result['priority']}, handled={result['handled']}")

print("\n" + "=" * 60)
print("✅ 所有示例执行完成")
print("=" * 60)
```

## 关键要点

### 1. 边的混用陷阱

**陷阱1：重复执行**
- 问题：条件边和普通边同时指向同一节点
- 后果：目标节点被执行多次
- 解决：每个节点只使用一种边类型

**陷阱2：Command 失效**
- 问题：预定义的普通边无法被 Command 覆盖
- 后果：动态路由失效
- 解决：使用条件边配合 Command

**陷阱3：状态修改失效**
- 问题：在路由函数中修改状态
- 后果：状态修改不会生效
- 解决：路由函数保持纯函数，状态修改在节点中完成

**陷阱4：可视化错误**
- 问题：路由函数缺少类型注解
- 后果：图可视化显示不完整
- 解决：使用 Literal 类型注解或 path_map

### 2. 最佳实践

1. **互斥原则**：避免多种边类型指向同一节点
2. **纯函数原则**：路由函数只读取状态，不修改状态
3. **类型注解原则**：使用 Literal 提升可维护性
4. **动态优先原则**：需要动态路由时使用条件边 + Command

### 3. 边的选择策略

| 场景 | 推荐边类型 | 原因 |
|------|-----------|------|
| 简单流程 | 普通边 | 确定性路径，性能最优 |
| 条件分支 | 条件边 | 根据状态动态决策 |
| 并行等待 | 等待边 | 同步多个并行任务 |
| 动态路由 | 条件边 + Command | 最大灵活性 |
