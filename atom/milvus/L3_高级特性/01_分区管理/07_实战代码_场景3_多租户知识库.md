## 场景描述

在 SaaS 应用中，我们需要为每个企业客户提供独立的知识库服务。每个租户的数据需要逻辑隔离，同时支持高效的检索。

本场景演示如何使用 Partition Key 实现完整的多租户知识库系统，包括：
- 租户数据隔离
- 租户级别的搜索
- 租户数据管理
- 跨租户分析

## 完整代码示例

```python
"""
Milvus 多租户知识库实战
场景：SaaS 企业知识库系统

功能：
1. 租户注册和初始化
2. 租户文档上传和向量化
3. 租户级别的语义搜索
4. 租户数据统计和管理
5. 跨租户数据分析

依赖：
- pymilvus
- sentence-transformers
- numpy
"""

from pymilvus import (
    connections,
    Collection,
    CollectionSchema,
    FieldSchema,
    DataType,
    utility
)
from sentence_transformers import SentenceTransformer
import numpy as np
from typing import List, Dict, Any
from datetime import datetime


class MultiTenantKnowledgeBase:
    """多租户知识库管理器"""

    def __init__(self, collection_name: str = "multi_tenant_kb"):
        self.collection_name = collection_name
        self.collection = None
        self.embedding_model = None

    def connect_to_milvus(self, host: str = "localhost", port: str = "19530"):
        """连接到 Milvus"""
        print(f"连接到 Milvus: {host}:{port}")
        connections.connect(alias="default", host=host, port=port)
        print("连接成功")

    def initialize_embedding_model(self, model_name: str = "all-MiniLM-L6-v2"):
        """初始化 Embedding 模型"""
        print(f"加载 Embedding 模型: {model_name}")
        self.embedding_model = SentenceTransformer(model_name)
        print("模型加载成功")

    def create_collection(self):
        """创建多租户 Collection"""
        if utility.has_collection(self.collection_name):
            print(f"Collection '{self.collection_name}' 已存在，删除后重新创建")
            utility.drop_collection(self.collection_name)

        # 定义 Schema（使用 tenant_id 作为 Partition Key）
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(
                name="tenant_id",
                dtype=DataType.VARCHAR,
                max_length=64,
                is_partition_key=True  # Partition Key
            ),
            FieldSchema(name="doc_id", dtype=DataType.VARCHAR, max_length=128),
            FieldSchema(name="title", dtype=DataType.VARCHAR, max_length=512),
            FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
            FieldSchema(name="category", dtype=DataType.VARCHAR, max_length=64),
            FieldSchema(name="created_at", dtype=DataType.INT64),
            FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=384)
        ]

        schema = CollectionSchema(
            fields=fields,
            description="多租户知识库 - 使用 Partition Key 实现租户隔离"
        )

        self.collection = Collection(name=self.collection_name, schema=schema)
        print(f"Collection '{self.collection_name}' 创建成功")

    def create_index(self):
        """创建向量索引"""
        print("创建索引...")
        index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "L2",
            "params": {"nlist": 128}
        }
        self.collection.create_index(field_name="embedding", index_params=index_params)
        print("索引创建成功")

    def load_collection(self):
        """加载 Collection"""
        print("加载 Collection...")
        self.collection.load()
        print("Collection 已加载")

    def add_documents(
        self,
        tenant_id: str,
        documents: List[Dict[str, str]]
    ) -> int:
        """
        为租户添加文档

        Args:
            tenant_id: 租户 ID
            documents: 文档列表，每个文档包含 doc_id, title, content, category

        Returns:
            插入的文档数量
        """
        print(f"\n为租户 '{tenant_id}' 添加 {len(documents)} 个文档...")

        # 提取文本并生成 embedding
        texts = [f"{doc['title']} {doc['content']}" for doc in documents]
        embeddings = self.embedding_model.encode(texts).tolist()

        # 准备数据
        data = [
            [tenant_id] * len(documents),  # tenant_id
            [doc['doc_id'] for doc in documents],  # doc_id
            [doc['title'] for doc in documents],  # title
            [doc['content'] for doc in documents],  # content
            [doc['category'] for doc in documents],  # category
            [int(datetime.now().timestamp())] * len(documents),  # created_at
            embeddings  # embedding
        ]

        # 插入数据
        result = self.collection.insert(data)
        print(f"成功插入 {result.insert_count} 个文档")
        return result.insert_count

    def search_documents(
        self,
        tenant_id: str,
        query: str,
        top_k: int = 5,
        category: str = None
    ) -> List[Dict[str, Any]]:
        """
        在租户知识库中搜索

        Args:
            tenant_id: 租户 ID
            query: 查询文本
            top_k: 返回结果数量
            category: 可选的类别过滤

        Returns:
            搜索结果列表
        """
        print(f"\n在租户 '{tenant_id}' 的知识库中搜索: '{query}'")

        # 生成查询向量
        query_embedding = self.embedding_model.encode([query]).tolist()

        # 构建过滤表达式
        expr = f'tenant_id == "{tenant_id}"'
        if category:
            expr += f' && category == "{category}"'

        # 搜索
        search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
        results = self.collection.search(
            data=query_embedding,
            anns_field="embedding",
            param=search_params,
            limit=top_k,
            expr=expr,
            output_fields=["doc_id", "title", "content", "category"]
        )

        # 格式化结果
        formatted_results = []
        for hit in results[0]:
            formatted_results.append({
                "doc_id": hit.entity.get("doc_id"),
                "title": hit.entity.get("title"),
                "content": hit.entity.get("content"),
                "category": hit.entity.get("category"),
                "distance": hit.distance
            })

        print(f"找到 {len(formatted_results)} 个结果")
        return formatted_results

    def get_tenant_stats(self, tenant_id: str) -> Dict[str, Any]:
        """获取租户统计信息"""
        result = self.collection.query(
            expr=f'tenant_id == "{tenant_id}"',
            output_fields=["category"]
        )

        total_docs = len(result)
        categories = {}
        for doc in result:
            cat = doc.get("category", "未分类")
            categories[cat] = categories.get(cat, 0) + 1

        return {
            "tenant_id": tenant_id,
            "total_documents": total_docs,
            "categories": categories
        }

    def delete_tenant_documents(self, tenant_id: str, doc_ids: List[str] = None):
        """删除租户文档"""
        if doc_ids:
            expr = f'tenant_id == "{tenant_id}" && doc_id in {doc_ids}'
            print(f"删除租户 '{tenant_id}' 的指定文档...")
        else:
            expr = f'tenant_id == "{tenant_id}"'
            print(f"删除租户 '{tenant_id}' 的所有文档...")

        self.collection.delete(expr)
        print("删除成功")


def main():
    """主函数 - 演示多租户知识库"""

    # 1. 初始化
    kb = MultiTenantKnowledgeBase()
    kb.connect_to_milvus()
    kb.initialize_embedding_model()
    kb.create_collection()
    kb.create_index()
    kb.load_collection()

    # 2. 租户 A：技术公司
    tenant_a_docs = [
        {
            "doc_id": "tech_001",
            "title": "API 开发指南",
            "content": "本文档介绍了 REST API 的设计原则和最佳实践...",
            "category": "技术文档"
        },
        {
            "doc_id": "tech_002",
            "title": "微服务架构",
            "content": "微服务架构是一种将应用程序构建为一组小型服务的方法...",
            "category": "技术文档"
        },
        {
            "doc_id": "tech_003",
            "title": "性能优化手册",
            "content": "性能优化的关键在于识别瓶颈并针对性地优化...",
            "category": "技术文档"
        }
    ]
    kb.add_documents("tenant_a", tenant_a_docs)

    # 3. 租户 B：教育机构
    tenant_b_docs = [
        {
            "doc_id": "edu_001",
            "title": "Python 编程基础",
            "content": "Python 是一种易学易用的编程语言，适合初学者...",
            "category": "课程资料"
        },
        {
            "doc_id": "edu_002",
            "title": "数据结构与算法",
            "content": "数据结构是计算机存储和组织数据的方式...",
            "category": "课程资料"
        }
    ]
    kb.add_documents("tenant_b", tenant_b_docs)

    # 4. 租户级别搜索
    results_a = kb.search_documents("tenant_a", "如何优化 API 性能", top_k=3)
    print("\n租户 A 搜索结果:")
    for i, result in enumerate(results_a, 1):
        print(f"{i}. {result['title']} (距离: {result['distance']:.4f})")

    results_b = kb.search_documents("tenant_b", "Python 编程入门", top_k=3)
    print("\n租户 B 搜索结果:")
    for i, result in enumerate(results_b, 1):
        print(f"{i}. {result['title']} (距离: {result['distance']:.4f})")

    # 5. 租户统计
    stats_a = kb.get_tenant_stats("tenant_a")
    print(f"\n租户 A 统计: {stats_a}")

    stats_b = kb.get_tenant_stats("tenant_b")
    print(f"租户 B 统计: {stats_b}")

    print("\n演示完成")


if __name__ == "__main__":
    main()
```

## 核心特性

### 1. 租户隔离

使用 Partition Key 实现自动租户隔离：
```python
FieldSchema(
    name="tenant_id",
    dtype=DataType.VARCHAR,
    max_length=64,
    is_partition_key=True
)
```

### 2. 租户级别搜索

搜索时自动过滤租户数据：
```python
expr = f'tenant_id == "{tenant_id}"'
results = collection.search(
    data=query_embedding,
    expr=expr,
    ...
)
```

### 3. 类别过滤

支持在租户内按类别过滤：
```python
expr = f'tenant_id == "{tenant_id}" && category == "{category}"'
```
