# 实战代码 - 场景3：验证错误处理与恢复

## 场景说明

本场景演示如何在 LangGraph 中优雅地处理验证错误。你将看到：

1. 用 `safe_graph_invoke` 封装图调用，捕获所有 `ValidationError`
2. 结构化日志记录每个验证错误的字段、类型和原始输入
3. 降级策略：自动修复常见的输入错误（语言映射、数值截断）
4. 多错误场景下的批量处理与诊断

在生产环境中，用户输入千奇百怪，LLM 输出也可能不符合预期。直接让 `ValidationError` 崩溃整个服务是不可接受的——你需要一个统一的错误处理层。

## 完整代码

```python
"""
LangGraph 状态验证实战 - 场景3：验证错误处理与恢复
演示：捕获验证错误、日志记录、降级策略

运行环境：Python 3.13+, langgraph, pydantic
安装依赖：uv add langgraph pydantic
"""

from pydantic import BaseModel, Field, field_validator, ValidationError
from langgraph.graph import StateGraph, START, END
import logging

# 配置日志：生产环境中验证错误应该被记录而非静默丢弃
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
)
logger = logging.getLogger(__name__)


# ============================================================
# 1. 定义严格验证的状态
# ============================================================

class StrictAgentState(BaseModel):
    """严格验证的 Agent 状态

    每个字段都有明确的约束条件：
    - query: 2~500 字符，不能是纯空白
    - language: 只允许 zh/en/ja/ko 四种语言代码
    - max_results: 1~20 的整数
    """
    query: str = Field(
        min_length=2,
        max_length=500,
        description="用户查询，2~500 字符",
    )
    language: str = Field(
        default="zh",
        pattern=r"^(zh|en|ja|ko)$",
        description="语言代码，仅支持 zh/en/ja/ko",
    )
    max_results: int = Field(
        default=5,
        ge=1,
        le=20,
        description="返回结果数量，1~20",
    )

    @field_validator("query")
    @classmethod
    def clean_query(cls, v: str) -> str:
        """去除首尾空白后再检查是否为空"""
        v = v.strip()
        if not v:
            raise ValueError("查询不能为空白字符")
        return v


# ============================================================
# 2. 输入修复函数
# ============================================================

# PLACEHOLDER_SECTION_FIX

def _try_fix_input(input_data: dict, errors: list) -> dict | None:
    """尝试修复常见的输入错误

    修复策略：
    - language: 将常见语言名称映射为标准代码（chinese → zh）
    - max_results 过小: 钳位到最小值 1
    - max_results 过大: 钳位到最大值 20

    注意：query 相关的错误通常无法自动修复（空查询、超长查询），
    因为我们不应该替用户编造查询内容。

    返回修复后的字典，如果无法修复则返回 None。
    """
    fixed = input_data.copy()
    fixable = False

    for error in errors:
        field = error["loc"][0] if error["loc"] else None

        if field == "language" and error["type"] == "string_pattern_mismatch":
            # 策略：将常见语言名称映射为标准代码
            lang_map = {
                "chinese": "zh", "中文": "zh",
                "english": "en", "英文": "en",
                "japanese": "ja", "日文": "ja",
                "korean": "ko", "韩文": "ko",
            }
            original = fixed.get("language", "").lower()
            if original in lang_map:
                fixed["language"] = lang_map[original]
                logger.info(f"  自动修复 language: '{original}' → '{lang_map[original]}'")
                fixable = True

        elif field == "max_results" and error["type"] == "greater_than_equal":
            # 策略：钳位到最小值
            fixed["max_results"] = 1
            logger.info(f"  自动修复 max_results: {input_data.get('max_results')} → 1")
            fixable = True

        elif field == "max_results" and error["type"] == "less_than_equal":
            # 策略：钳位到最大值
            fixed["max_results"] = 20
            logger.info(f"  自动修复 max_results: {input_data.get('max_results')} → 20")
            fixable = True

    return fixed if fixable else None


# ============================================================
# 3. 安全调用封装
# ============================================================

def safe_graph_invoke(graph, input_data: dict, default_response=None):
    """安全调用图，捕获验证错误并提供降级响应

    处理流程：
    1. 尝试正常调用
    2. 如果 ValidationError → 记录日志 → 尝试修复输入
    3. 修复成功 → 用修复后的输入重试
    4. 修复失败 → 返回默认响应或错误详情

    参数：
        graph: 编译后的 LangGraph 图
        input_data: 用户输入字典
        default_response: 验证失败时的默认返回值

    返回：
        图的正常输出，或降级响应
    """
    try:
        # 第一次尝试：直接调用
        return graph.invoke(input_data)

    except ValidationError as e:
        # ---- 阶段 1：结构化日志记录 ----
        logger.warning(f"输入验证失败: {e.error_count()} 个错误")
        for error in e.errors():
            field_path = ".".join(str(loc) for loc in error["loc"])
            logger.warning(
                f"  字段 '{field_path}': {error['msg']} "
                f"(类型: {error['type']}, 输入: {error.get('input', 'N/A')})"
            )

        # ---- 阶段 2：尝试自动修复 ----
        fixed_input = _try_fix_input(input_data, e.errors())
        if fixed_input:
            logger.info("尝试使用修复后的输入重新调用...")
            try:
                result = graph.invoke(fixed_input)
                logger.info("修复成功，图执行完成")
                return result
            except ValidationError as e2:
                logger.error(
                    f"修复后的输入仍然无效: {e2.error_count()} 个错误"
                )

        # ---- 阶段 3：降级响应 ----
        return default_response or {
            "error": "输入验证失败",
            "error_count": e.error_count(),
            "details": [
                {
                    "field": ".".join(str(loc) for loc in err["loc"]),
                    "message": err["msg"],
                    "type": err["type"],
                }
                for err in e.errors()
            ],
        }


# ============================================================
# 4. 构建 LangGraph 图
# ============================================================

def process_node(state: StrictAgentState):
    """模拟处理节点"""
    return {"query": f"已处理: {state.query} (lang={state.language})"}


builder = StateGraph(StrictAgentState)
builder.add_node("process", process_node)
builder.add_edge(START, "process")
builder.add_edge("process", END)
graph = builder.compile()


# ============================================================
# 5. 测试各种错误场景
# ============================================================

print("=" * 60)
print("=== 场景 A：正常输入 ===")
print("=" * 60)
result = safe_graph_invoke(graph, {"query": "什么是 RAG？"})
print(f"结果: {result}\n")


print("=" * 60)
print("=== 场景 B：空查询（不可修复） ===")
print("=" * 60)
result = safe_graph_invoke(graph, {"query": ""})
print(f"结果: {result}\n")


print("=" * 60)
print("=== 场景 C：无效语言代码（可修复） ===")
print("=" * 60)
result = safe_graph_invoke(graph, {"query": "test query", "language": "chinese"})
print(f"结果: {result}\n")


print("=" * 60)
print("=== 场景 D：max_results 超出范围（可修复） ===")
print("=" * 60)
result = safe_graph_invoke(graph, {"query": "test query", "max_results": 100})
print(f"结果: {result}\n")


print("=" * 60)
print("=== 场景 E：多个错误混合（部分可修复） ===")
print("=" * 60)
result = safe_graph_invoke(
    graph,
    {"query": "", "language": "xyz", "max_results": -5},
)
print(f"结果: {result}\n")


print("=" * 60)
print("=== 场景 F：自定义默认响应 ===")
print("=" * 60)
result = safe_graph_invoke(
    graph,
    {"query": ""},
    default_response={"answer": "抱歉，请输入有效的查询内容", "status": "invalid_input"},
)
print(f"结果: {result}\n")
```

## 关键知识点解析

### 1. 为什么需要 safe_graph_invoke？

LangGraph 的 `graph.invoke()` 在输入阶段触发 Pydantic 验证。如果验证失败，`ValidationError` 会直接抛出，未捕获就会导致服务崩溃。`safe_graph_invoke` 是一个通用的防护层：

```
graph.invoke(input)
    ↓ ValidationError
safe_graph_invoke
    ├── 日志记录（运维可追踪）
    ├── 自动修复（降低用户感知）
    └── 降级响应（服务不中断）
```

### 2. 降级策略的三个层次

| 层次 | 策略 | 适用场景 |
|------|------|---------|
| L1 自动修复 | 映射、钳位、默认值替换 | 语言代码、数值范围等可推断的字段 |
| L2 自定义默认响应 | 返回预设的友好提示 | API 服务需要固定格式的错误响应 |
| L3 结构化错误详情 | 返回字段级别的错误列表 | 前端需要逐字段显示错误提示 |

### 3. 哪些错误可以自动修复？

判断标准：**修复后的值是否仍然符合用户意图**。

- 可修复：`"chinese"` → `"zh"`（明确的映射关系）
- 可修复：`max_results=100` → `20`（钳位到合法范围）
- 不可修复：`query=""`（我们不知道用户想问什么）
- 不可修复：`language="xyz"`（无法推断用户意图）

### 4. 生产环境建议

- 日志级别：验证错误用 `WARNING`，修复成功用 `INFO`，修复失败用 `ERROR`
- 监控指标：统计验证失败率、自动修复成功率，用于发现系统性问题
- API 响应：返回结构化的错误详情（字段 + 消息 + 类型），方便前端展示

## RAG 开发中的应用

在 RAG 系统中，验证错误处理特别重要：

1. **用户查询验证**：空查询、超长查询、注入攻击都需要在入口拦截
2. **LLM 输出验证**：大模型返回的结构化数据可能不符合 schema，需要降级处理
3. **检索参数验证**：`top_k`、`threshold` 等参数超出范围时自动钳位比报错更友好
4. **多轮对话**：对话历史中的脏数据不应该导致整个会话崩溃

```
用户输入 → [safe_invoke 验证] → 检索 → [结果验证] → 生成 → [输出验证] → 响应
              ↓ 失败                      ↓ 失败                ↓ 失败
           友好提示                    降级检索              兜底回复
```
