## 完整代码实现

```python
"""
LangGraph 部分状态更新 - 场景1：基础部分状态返回

演示：
1. 节点函数只返回需要更新的字段
2. 未返回的字段自动保持原值
3. 状态在多个节点间的流转

环境要求：
- Python 3.13+
- langgraph
"""

from typing import TypedDict
from langgraph.graph import StateGraph, START, END
from datetime import datetime
import json


# ===== 1. 定义状态 Schema =====
class TaskState(TypedDict):
    """
    任务处理状态

    字段说明：
    - task_id: 任务ID（初始化后不变）
    - task_name: 任务名称（初始化后不变）
    - status: 当前状态（每个节点可能更新）
    - progress: 进度百分比（每个节点可能更新）
    - logs: 日志列表（每个节点可能追加）
    - result: 处理结果（最后节点设置）
    - error: 错误信息（出错时设置）
    - created_at: 创建时间（初始化后不变）
    - updated_at: 更新时间（每个节点可能更新）
    """
    task_id: str
    task_name: str
    status: str
    progress: int
    logs: list[str]
    result: dict | None
    error: str | None
    created_at: str
    updated_at: str


# ===== 2. 定义节点函数（每个节点只返回部分字段）=====

def node_validate(state: TaskState) -> dict:
    """
    节点1：数据验证

    只更新：status, progress, logs, updated_at
    保持不变：task_id, task_name, result, error, created_at
    """
    print(f"\n[Node 1: Validate] 开始验证任务 {state['task_id']}")

    # 模拟验证逻辑
    is_valid = state['task_name'] and len(state['task_name']) > 0

    if is_valid:
        # 只返回需要更新的字段
        return {
            "status": "validated",
            "progress": 25,
            "logs": state["logs"] + [f"[{datetime.now().isoformat()}] 验证通过"],
            "updated_at": datetime.now().isoformat()
        }
    else:
        return {
            "status": "failed",
            "progress": 0,
            "logs": state["logs"] + [f"[{datetime.now().isoformat()}] 验证失败"],
            "error": "任务名称无效",
            "updated_at": datetime.now().isoformat()
        }


def node_process(state: TaskState) -> dict:
    """
    节点2：数据处理

    只更新：status, progress, logs, updated_at
    保持不变：task_id, task_name, result, error, created_at
    """
    print(f"\n[Node 2: Process] 开始处理任务 {state['task_id']}")

    # 检查前置状态
    if state["status"] == "failed":
        print("  跳过处理（前置验证失败）")
        return {
            "logs": state["logs"] + [f"[{datetime.now().isoformat()}] 跳过处理"],
            "updated_at": datetime.now().isoformat()
        }

    # 模拟处理逻辑
    print(f"  处理任务：{state['task_name']}")

    # 只返回需要更新的字段
    return {
        "status": "processed",
        "progress": 75,
        "logs": state["logs"] + [f"[{datetime.now().isoformat()}] 处理完成"],
        "updated_at": datetime.now().isoformat()
    }


def node_finalize(state: TaskState) -> dict:
    """
    节点3：结果汇总

    只更新：status, progress, result, logs, updated_at
    保持不变：task_id, task_name, error, created_at
    """
    print(f"\n[Node 3: Finalize] 汇总结果 {state['task_id']}")

    # 检查前置状态
    if state["status"] == "failed":
        print("  任务失败，生成失败报告")
        return {
            "status": "completed_with_error",
            "progress": 100,
            "result": {
                "success": False,
                "error": state.get("error", "Unknown error")
            },
            "logs": state["logs"] + [f"[{datetime.now().isoformat()}] 任务失败"],
            "updated_at": datetime.now().isoformat()
        }

    # 生成成功结果
    print("  任务成功，生成成功报告")
    return {
        "status": "completed",
        "progress": 100,
        "result": {
            "success": True,
            "task_id": state["task_id"],
            "task_name": state["task_name"],
            "total_logs": len(state["logs"]) + 1
        },
        "logs": state["logs"] + [f"[{datetime.now().isoformat()}] 任务完成"],
        "updated_at": datetime.now().isoformat()
    }


# ===== 3. 构建图 =====

def build_task_graph():
    """构建任务处理图"""

    # 创建 StateGraph
    builder = StateGraph(TaskState)

    # 添加节点
    builder.add_node("validate", node_validate)
    builder.add_node("process", node_process)
    builder.add_node("finalize", node_finalize)

    # 添加边（定义执行顺序）
    builder.add_edge(START, "validate")
    builder.add_edge("validate", "process")
    builder.add_edge("process", "finalize")
    builder.add_edge("finalize", END)

    # 编译图
    graph = builder.compile()

    return graph


# ===== 4. 执行示例 =====

def run_example_1_success():
    """示例1：成功的任务处理"""
    print("=" * 80)
    print("示例1：成功的任务处理")
    print("=" * 80)

    # 构建图
    graph = build_task_graph()

    # 初始状态
    initial_state = {
        "task_id": "task-001",
        "task_name": "数据导入任务",
        "status": "pending",
        "progress": 0,
        "logs": [],
        "result": None,
        "error": None,
        "created_at": datetime.now().isoformat(),
        "updated_at": datetime.now().isoformat()
    }

    print("\n初始状态：")
    print(json.dumps(initial_state, indent=2, ensure_ascii=False))

    # 执行图
    final_state = graph.invoke(initial_state)

    print("\n" + "=" * 80)
    print("最终状态：")
    print("=" * 80)
    print(json.dumps(final_state, indent=2, ensure_ascii=False))

    # 验证部分状态更新
    print("\n" + "=" * 80)
    print("验证部分状态更新：")
    print("=" * 80)
    print(f"✓ task_id 保持不变: {initial_state['task_id']} -> {final_state['task_id']}")
    print(f"✓ task_name 保持不变: {initial_state['task_name']} -> {final_state['task_name']}")
    print(f"✓ created_at 保持不变: {initial_state['created_at']} -> {final_state['created_at']}")
    print(f"✓ status 被更新: {initial_state['status']} -> {final_state['status']}")
    print(f"✓ progress 被更新: {initial_state['progress']} -> {final_state['progress']}")
    print(f"✓ logs 被追加: {len(initial_state['logs'])} -> {len(final_state['logs'])} 条")
    print(f"✓ result 被设置: {initial_state['result']} -> {final_state['result'] is not None}")
```

## 代码解析

### 1. 状态 Schema 设计

```python
class TaskState(TypedDict):
    task_id: str          # 不变字段
    task_name: str        # 不变字段
    status: str           # 可变字段
    progress: int         # 可变字段
    logs: list[str]       # 可变字段
    result: dict | None   # 可变字段
    error: str | None     # 可变字段
    created_at: str       # 不变字段
    updated_at: str       # 可变字段
```

**设计原则：**
- **不变字段**：初始化后不再修改（如 `task_id`, `task_name`, `created_at`）
- **可变字段**：节点可能更新（如 `status`, `progress`, `logs`）

**[来源: reference/context7_langgraph_01.md - 状态更新语义]**

### 2. 节点函数实现

```python
def node_validate(state: TaskState) -> dict:
    # 只返回需要更新的字段
    return {
        "status": "validated",
        "progress": 25,
        "logs": state["logs"] + ["验证通过"],
        "updated_at": datetime.now().isoformat()
    }
    # task_id, task_name, result, error, created_at 自动保持原值
```

**关键点：**
- ✅ 只返回需要更新的字段
- ✅ 未返回的字段自动保持原值
- ✅ 节点函数是纯函数，无副作用

**[来源: reference/search_部分状态更新_01.md - 最佳实践：节点如纯函数]**

### 3. 状态合并机制

```python
# LangGraph 内部的状态合并逻辑（简化版）
def merge_state(current_state, partial_update):
    new_state = current_state.copy()  # 保留所有原值
    new_state.update(partial_update)   # 只更新返回的字段
    return new_state
```

**执行流程：**

## 核心要点总结

### 1. 部分状态返回的优势

| 优势 | 说明 | 示例 |
|------|------|------|
| **减少代码冗余** | 不需要返回所有字段 | 只返回 `status` 和 `progress` |
| **提高可维护性** | 字段变化局部化 | 修改字段不影响其他节点 |
| **清晰的意图** | 明确哪些字段被更新 | 一眼看出节点的作用 |
| **避免错误** | 不会意外覆盖其他字段 | 不会丢失 `task_id` 等关键信息 |

### 2. 最佳实践

```python
# ✅ 推荐：只返回需要更新的字段
def node_good(state: State):
    return {
        "status": "processed",
        "progress": 50
    }

# ❌ 不推荐：返回所有字段
def node_bad(state: State):
    return {
        "task_id": state["task_id"],      # 多余
        "task_name": state["task_name"],  # 多余
        "status": "processed",
        "progress": 50,
        "logs": state["logs"],            # 多余
        "result": state["result"],        # 多余
        "error": state["error"],          # 多余
        "created_at": state["created_at"],# 多余
        "updated_at": datetime.now().isoformat()
    }
```

### 3. 常见模式

**模式1：状态转换**
```python
def node(state: State):
    return {"status": "new_status"}
```

**模式2：进度更新**
```python
def node(state: State):
    return {
        "progress": state["progress"] + 25,
        "updated_at": datetime.now().isoformat()
    }
```

**模式3：日志追加**
```python
def node(state: State):
    return {
        "logs": state["logs"] + ["新日志"],
        "updated_at": datetime.now().isoformat()
    }
```

**模式4：结果设置**
```python
def node(state: State):
    return {
        "result": {"success": True, "data": "..."},
        "status": "completed"
    }
```
