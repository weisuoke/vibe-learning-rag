## 实战示例1：Thread ID 隔离策略

### 完整代码

```python
"""
Thread ID 隔离策略
演示如何使用 Thread ID 实现多用户隔离
"""

import uuid
from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from dotenv import load_dotenv

load_dotenv()

# ============================================================
# 1. 多用户存储管理
# ============================================================

class MultiUserSessionManager:
    """多用户会话管理器"""

    def __init__(self):
        # 存储结构: {user_id: {session_id: history}}
        self.store = {}

    def get_session_history(self, user_id: str, session_id: str):
        """获取指定用户的会话历史"""
        if user_id not in self.store:
            self.store[user_id] = {}

        if session_id not in self.store[user_id]:
            self.store[user_id][session_id] = InMemoryChatMessageHistory()

        return self.store[user_id][session_id]

    def list_user_sessions(self, user_id: str):
        """列出用户的所有会话"""
        return list(self.store.get(user_id, {}).keys())

    def delete_session(self, user_id: str, session_id: str):
        """删除指定会话"""
        if user_id in self.store and session_id in self.store[user_id]:
            del self.store[user_id][session_id]

# ============================================================
# 2. 创建会话管理器
# ============================================================

manager = MultiUserSessionManager()

def get_session_history_wrapper(session_id: str):
    """
    Session History 包装函数
    从 session_id 中解析 user_id 和 session_id
    """
    # session_id 格式: "user_{user_id}_session_{session_id}"
    parts = session_id.split("_")
    if len(parts) >= 4:
        user_id = parts[1]
        sess_id = parts[3]
    else:
        raise ValueError(f"Invalid session_id format: {session_id}")

    return manager.get_session_history(user_id, sess_id)

# ============================================================
# 3. 创建对话链
# ============================================================

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

conversation = RunnableWithMessageHistory(
    llm,
    get_session_history_wrapper,
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 4. 测试多用户隔离
# ============================================================

def test_multi_user_isolation():
    """测试多用户隔离"""

    print("=== 多用户隔离测试 ===\n")

    # 用户1的对话
    user1_session = "user_alice_session_001"
    config1 = {"configurable": {"session_id": user1_session}}

    print("👤 用户Alice: 我叫Alice，是一名设计师")
    response1 = conversation.invoke(
        {"input": "我叫Alice，是一名设计师"},
        config=config1
    )
    print(f"🤖 AI: {response1.content}\n")

    # 用户2的对话
    user2_session = "user_bob_session_001"
    config2 = {"configurable": {"session_id": user2_session}}

    print("👤 用户Bob: 我叫Bob，是一名工程师")
    response2 = conversation.invoke(
        {"input": "我叫Bob，是一名工程师"},
        config=config2
    )
    print(f"🤖 AI: {response2.content}\n")

    # 验证隔离：用户1询问自己的信息
    print("👤 用户Alice: 我的职业是什么？")
    response3 = conversation.invoke(
        {"input": "我的职业是什么？"},
        config=config1
    )
    print(f"🤖 AI: {response3.content}\n")

    # 验证隔离：用户2询问自己的信息
    print("👤 用户Bob: 我的职业是什么？")
    response4 = conversation.invoke(
        {"input": "我的职业是什么？"},
        config=config2
    )
    print(f"🤖 AI: {response4.content}\n")

    # 验证存储
    print("=== 存储验证 ===")
    alice_sessions = manager.list_user_sessions("alice")
    bob_sessions = manager.list_user_sessions("bob")
    print(f"✅ Alice的会话: {alice_sessions}")
    print(f"✅ Bob的会话: {bob_sessions}")

if __name__ == "__main__":
    test_multi_user_isolation()
```

## 实战示例2：Session 清理机制

### 完整代码

```python
"""
Session 清理机制
演示如何实现 TTL 和定期清理
"""

import time
from datetime import datetime, timedelta
from typing import Dict, Optional

class SessionWithTTL:
    """带 TTL 的会话"""

    def __init__(self, history, ttl_seconds: int = 3600):
        self.history = history
        self.created_at = datetime.now()
        self.last_accessed = datetime.now()
        self.ttl_seconds = ttl_seconds

    def is_expired(self) -> bool:
        """检查是否过期"""
        return datetime.now() - self.last_accessed > timedelta(seconds=self.ttl_seconds)

    def touch(self):
        """更新最后访问时间"""
        self.last_accessed = datetime.now()

class SessionManagerWithCleanup:
    """带清理功能的会话管理器"""

    def __init__(self, ttl_seconds: int = 3600):
        self.store: Dict[str, Dict[str, SessionWithTTL]] = {}
        self.ttl_seconds = ttl_seconds

    def get_session_history(self, user_id: str, session_id: str):
        """获取会话历史（自动清理过期会话）"""
        # 清理过期会话
        self.cleanup_expired_sessions()

        if user_id not in self.store:
            self.store[user_id] = {}

        if session_id not in self.store[user_id]:
            self.store[user_id][session_id] = SessionWithTTL(
                InMemoryChatMessageHistory(),
                self.ttl_seconds
            )

        # 更新访问时间
        session = self.store[user_id][session_id]
        session.touch()

        return session.history

    def cleanup_expired_sessions(self):
        """清理所有过期会话"""
        for user_id in list(self.store.keys()):
            for session_id in list(self.store[user_id].keys()):
                if self.store[user_id][session_id].is_expired():
                    print(f"🗑️ 清理过期会话: {user_id}/{session_id}")
                    del self.store[user_id][session_id]

            # 如果用户没有会话了，删除用户
            if not self.store[user_id]:
                del self.store[user_id]

    def get_stats(self):
        """获取统计信息"""
        total_users = len(self.store)
        total_sessions = sum(len(sessions) for sessions in self.store.values())
        return {
            "total_users": total_users,
            "total_sessions": total_sessions
        }

# ============================================================
# 测试 TTL 清理
# ============================================================

def test_ttl_cleanup():
    """测试 TTL 清理"""

    print("=== TTL 清理测试 ===\n")

    # 创建管理器（TTL = 5秒）
    manager = SessionManagerWithCleanup(ttl_seconds=5)

    # 创建会话
    history1 = manager.get_session_history("user1", "session1")
    history1.add_user_message("测试消息1")

    print(f"✅ 创建会话: user1/session1")
    print(f"📊 统计: {manager.get_stats()}\n")

    # 等待6秒
    print("⏳ 等待6秒...")
    time.sleep(6)

    # 触发清理
    manager.cleanup_expired_sessions()
    print(f"📊 清理后统计: {manager.get_stats()}\n")

if __name__ == "__main__":
    test_ttl_cleanup()
```

## 实战示例3：并发访问控制

### 完整代码

```python
"""
并发访问控制
演示如何处理多个并发请求
"""

import asyncio
from asyncio import Lock
from typing import Dict

class ThreadSafeSessionManager:
    """线程安全的会话管理器"""

    def __init__(self):
        self.store: Dict[str, Dict[str, InMemoryChatMessageHistory]] = {}
        self.locks: Dict[str, Lock] = {}

    async def get_session_history(self, user_id: str, session_id: str):
        """获取会话历史（线程安全）"""
        # 为每个用户创建独立的锁
        if user_id not in self.locks:
            self.locks[user_id] = Lock()

        async with self.locks[user_id]:
            if user_id not in self.store:
                self.store[user_id] = {}

            if session_id not in self.store[user_id]:
                self.store[user_id][session_id] = InMemoryChatMessageHistory()

            return self.store[user_id][session_id]

# ============================================================
# 测试并发访问
# ============================================================

async def test_concurrent_access():
    """测试并发访问"""

    print("=== 并发访问测试 ===\n")

    manager = ThreadSafeSessionManager()

    async def user_conversation(user_id: str, num_messages: int):
        """模拟用户对话"""
        for i in range(num_messages):
            history = await manager.get_session_history(user_id, "session1")
            history.add_user_message(f"用户{user_id}的消息{i}")
            await asyncio.sleep(0.1)  # 模拟处理时间

        print(f"✅ 用户{user_id}完成{num_messages}条消息")

    # 并发执行多个用户的对话
    await asyncio.gather(
        user_conversation("alice", 5),
        user_conversation("bob", 5),
        user_conversation("charlie", 5)
    )

    # 验证结果
    print("\n=== 验证结果 ===")
    for user_id in ["alice", "bob", "charlie"]:
        history = await manager.get_session_history(user_id, "session1")
        print(f"用户{user_id}: {len(history.messages)}条消息")

if __name__ == "__main__":
    asyncio.run(test_concurrent_access())
```

### 2. 权限验证

```python
def verify_access(user_id: str, session_id: str, requesting_user: str) -> bool:
    """验证用户是否有权访问会话"""
    # 从 session_id 中提取 user_id
    parts = session_id.split("_")
    session_owner = parts[1] if len(parts) >= 4 else None

    # 只允许用户访问自己的会话
    return session_owner == requesting_user
```

### 3. 定期清理任务

```python
import asyncio

async def periodic_cleanup(manager, interval_seconds: int = 3600):
    """定期清理任务"""
    while True:
        await asyncio.sleep(interval_seconds)
        manager.cleanup_expired_sessions()
        print(f"✅ 定期清理完成: {datetime.now()}")
```
