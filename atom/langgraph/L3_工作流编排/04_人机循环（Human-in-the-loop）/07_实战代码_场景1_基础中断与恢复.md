## 场景 1.1：基础 invoke 中断与恢复

最简单的人机循环：一个节点、一个中断点，用 `invoke` 同步执行。

```python
"""
场景 1.1：基础 invoke 中断与恢复
- 单节点、单中断点
- 使用 invoke 同步方式
- 演示完整的 暂停 → 恢复 流程
"""

from typing import TypedDict
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


# ========== 1. 定义状态 ==========

class State(TypedDict):
    """最简状态：一个问题，一个回答"""
    question: str
    answer: str


# ========== 2. 定义节点 ==========

def ask_human(state: State):
    """
    这个节点会：
    1. 调用 interrupt() 暂停图执行
    2. 把问题发送给客户端（人类）
    3. 等待人类通过 Command(resume=) 回复
    4. interrupt() 返回人类的回复
    """
    # interrupt() 的参数会发送给客户端
    # 可以是字符串、字典、列表等任何 JSON 可序列化的值
    human_reply = interrupt(f"请回答这个问题: {state['question']}")

    # 只有恢复后才会执行到这里
    return {"answer": human_reply}


# ========== 3. 构建图 ==========

builder = StateGraph(State)
builder.add_node("ask", ask_human)
builder.add_edge(START, "ask")
builder.add_edge("ask", END)

# 关键：必须提供 checkpointer，否则 interrupt 无法工作
graph = builder.compile(checkpointer=MemorySaver())


# ========== 4. 执行：第一次调用（触发中断） ==========

# thread_id 是会话标识，恢复时必须用同一个
config = {"configurable": {"thread_id": "basic-demo-1"}}

print("=" * 50)
print("第一次调用：触发中断")
print("=" * 50)

result = graph.invoke(
    {"question": "你最喜欢的编程语言是什么？", "answer": ""},
    config,
)

# invoke 返回时，图已经暂停
# result 中不会有 answer（因为节点没执行完）
print(f"返回值: {result}")
print("图已暂停，等待人类输入...\n")


# ========== 5. 恢复执行 ==========

print("=" * 50)
print("恢复执行：传递人类回复")
print("=" * 50)

# 使用 Command(resume=) 恢复，传入人类的回答
result = graph.invoke(
    Command(resume="Python！简洁优雅，生态丰富。"),
    config,  # 必须用同一个 config（同一个 thread_id）
)

print(f"最终结果: {result}")
print(f"回答: {result['answer']}")
```

**预期输出**：
```
==================================================
第一次调用：触发中断
==================================================
返回值: {'question': '你最喜欢的编程语言是什么？', 'answer': ''}
图已暂停，等待人类输入...

==================================================
恢复执行：传递人类回复
==================================================
最终结果: {'question': '你最喜欢的编程语言是什么？', 'answer': 'Python！简洁优雅，生态丰富。'}
回答: Python！简洁优雅，生态丰富。
```

## 场景 1.2：stream 模式中断与恢复

用 `stream` 流式方式执行，可以实时观察图的执行过程和中断事件。

```python
"""
场景 1.2：stream 模式中断与恢复
- 使用 stream 流式方式
- 可以观察到 __interrupt__ 事件
- 适合需要实时反馈的场景（如 Web UI）
"""

from typing import TypedDict
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


class ReviewState(TypedDict):
    content: str
    feedback: str
    status: str


def review_node(state: ReviewState):
    """审核节点：展示内容，等待人类反馈"""
    feedback = interrupt({
        "type": "review_request",
        "content": state["content"],
        "instruction": "请审核内容，输入 'ok' 通过或输入修改意见",
    })

    if feedback == "ok":
        return {"feedback": "审核通过", "status": "approved"}
    else:
        return {"feedback": feedback, "status": "needs_revision"}


def notify_node(state: ReviewState):
    """通知节点：发送审核结果"""
    return {"status": f"已通知 - {state['status']}"}


# 构建图
builder = StateGraph(ReviewState)
builder.add_node("review", review_node)
builder.add_node("notify", notify_node)
builder.add_edge(START, "review")
builder.add_edge("review", "notify")
builder.add_edge("notify", END)

graph = builder.compile(checkpointer=MemorySaver())
config = {"configurable": {"thread_id": "stream-demo-1"}}


# ========== stream 第一次调用：观察中断 ==========

print("=" * 50)
print("Stream 第一次调用：观察执行过程")
print("=" * 50)

for chunk in graph.stream(
    {"content": "LangGraph 人机循环教程", "feedback": "", "status": "pending"},
    config,
    stream_mode="updates",  # 使用 updates 模式，每个节点输出一个 chunk
):
    print(f"收到 chunk: {chunk}")

    # 当遇到中断时，会收到 __interrupt__ 键
    if "__interrupt__" in chunk:
        interrupt_info = chunk["__interrupt__"][0]
        print(f"\n>>> 检测到中断！")
        print(f">>> 中断值: {interrupt_info.value}")
        print(f">>> 中断 ID: {interrupt_info.id}")

print("\n图已暂停，等待人类审核...\n")


# ========== stream 恢复执行 ==========

print("=" * 50)
print("Stream 恢复执行")
print("=" * 50)

for chunk in graph.stream(
    Command(resume="ok"),  # 人类审核通过
    config,
    stream_mode="updates",
):
    print(f"收到 chunk: {chunk}")

print("\n流式执行完成！")
```

**预期输出**：
```
==================================================
Stream 第一次调用：观察执行过程
==================================================
收到 chunk: {'__interrupt__': (Interrupt(value={'type': 'review_request', ...}, id='abc123'),)}

>>> 检测到中断！
>>> 中断值: {'type': 'review_request', 'content': 'LangGraph 人机循环教程', ...}
>>> 中断 ID: abc123...

图已暂停，等待人类审核...

==================================================
Stream 恢复执行
==================================================
收到 chunk: {'review': {'feedback': '审核通过', 'status': 'approved'}}
收到 chunk: {'notify': {'status': '已通知 - approved'}}

流式执行完成！
```

### stream_mode 选项说明

```python
# stream_mode="updates" — 每个节点执行后输出一个 chunk（推荐）
# stream_mode="values"  — 每步输出完整的状态快照
# stream_mode="debug"   — 输出详细的调试信息

# 示例：values 模式
for chunk in graph.stream(input_data, config, stream_mode="values"):
    # chunk 是完整的状态字典
    print(f"当前状态: {chunk}")
```

## 场景 1.3：检查中断状态（get_state）

用 `get_state()` 查看图的当前状态，了解中断详情，这在调试和构建 UI 时非常有用。

```python
"""
场景 1.3：检查中断状态
- 使用 get_state() 获取完整的状态快照
- 查看中断信息、等待节点、当前状态值
- 适合调试和构建管理后台
"""

from typing import TypedDict, Optional
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END


class TaskState(TypedDict):
    task_name: str
    priority: str
    assignee: Optional[str]
    result: str


def assign_task(state: TaskState):
    """分配任务：等待管理员指定负责人"""
    assignee = interrupt({
        "type": "task_assignment",
        "task": state["task_name"],
        "priority": state["priority"],
        "message": "请指定任务负责人",
    })
    return {"assignee": assignee}


def execute_task(state: TaskState):
    """执行任务"""
    return {"result": f"{state['assignee']} 已完成任务: {state['task_name']}"}


# 构建图
builder = StateGraph(TaskState)
builder.add_node("assign", assign_task)
builder.add_node("execute", execute_task)
builder.add_edge(START, "assign")
builder.add_edge("assign", "execute")
builder.add_edge("execute", END)

graph = builder.compile(checkpointer=MemorySaver())
config = {"configurable": {"thread_id": "state-demo-1"}}


# ========== 触发中断 ==========

graph.invoke(
    {"task_name": "修复登录Bug", "priority": "高", "assignee": None, "result": ""},
    config,
)
print("任务已创建，等待分配...\n")


# ========== 检查中断状态 ==========

print("=" * 50)
print("检查中断状态")
print("=" * 50)

snapshot = graph.get_state(config)

# 1. 是否有待处理的中断？
has_pending = len(snapshot.next) > 0
print(f"是否有待处理中断: {has_pending}")  # True

# 2. 下一步要执行的节点
print(f"等待执行的节点: {snapshot.next}")  # ('assign',)

# 3. 当前状态值
print(f"当前状态值: {snapshot.values}")
# {'task_name': '修复登录Bug', 'priority': '高', 'assignee': None, 'result': ''}

# 4. 中断详情
print(f"\n--- 中断详情 ---")
for task in snapshot.tasks:
    print(f"任务节点: {task.name}")
    print(f"任务 ID: {task.id}")

    if task.interrupts:
        for intr in task.interrupts:
            print(f"  中断 ID: {intr.id}")
            print(f"  中断值: {intr.value}")
            # 输出:
            #   中断 ID: a1b2c3d4...
            #   中断值: {'type': 'task_assignment', 'task': '修复登录Bug', ...}
    else:
        print("  无中断")

# 5. 元数据
print(f"\n--- 元数据 ---")
print(f"创建时间: {snapshot.created_at}")
print(f"父配置: {snapshot.parent_config is not None}")


# ========== 根据中断信息做决策 ==========

print(f"\n{'=' * 50}")
print("管理员决策：分配给 Alice")
print("=" * 50)

result = graph.invoke(Command(resume="Alice"), config)
print(f"最终结果: {result['result']}")
# "Alice 已完成任务: 修复登录Bug"


# ========== 恢复后再次检查状态 ==========

print(f"\n--- 恢复后状态 ---")
snapshot_after = graph.get_state(config)
print(f"是否有待处理中断: {len(snapshot_after.next) > 0}")  # False
print(f"最终状态: {snapshot_after.values}")
```

**预期输出**：
```
任务已创建，等待分配...

==================================================
检查中断状态
==================================================
是否有待处理中断: True
等待执行的节点: ('assign',)
当前状态值: {'task_name': '修复登录Bug', 'priority': '高', 'assignee': None, 'result': ''}

--- 中断详情 ---
任务节点: assign
任务 ID: abc123-...
  中断 ID: def456-...
  中断值: {'type': 'task_assignment', 'task': '修复登录Bug', 'priority': '高', 'message': '请指定任务负责人'}

--- 元数据 ---
创建时间: 2026-02-28T...
父配置: True

==================================================
```

## 快速参考

### 三种执行方式对比

| 方式 | 适用场景 | 中断时返回 |
|------|----------|-----------|
| `invoke` | 简单同步调用 | 返回当前状态字典 |
| `stream` | 需要实时反馈 | 逐步输出 chunk，含 `__interrupt__` |
| `get_state` | 调试/管理后台 | 返回 `StateSnapshot` 完整快照 |

### 最小代码模板

```python
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver
from langgraph.graph import StateGraph, START, END

# 1. 节点内中断
def my_node(state):
    value = interrupt("提示信息")
    return {"key": value}

# 2. 编译（必须有 checkpointer）
graph = builder.compile(checkpointer=MemorySaver())

# 3. 触发中断
config = {"configurable": {"thread_id": "unique-id"}}
graph.invoke(input_data, config)

# 4. 检查状态（可选）
snapshot = graph.get_state(config)

# 5. 恢复执行
result = graph.invoke(Command(resume="人类回复"), config)
```
