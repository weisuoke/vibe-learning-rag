# 实战代码 - 场景3：多路由决策

> 本文档演示 LangGraph 中的多路由决策，使用条件边实现 switch-case 模式

---

## 场景描述

**场景**：构建一个任务分类系统，根据任务类型将任务路由到不同的处理器。

**特点**：
- 多个目标节点（3个以上）
- 使用条件边实现 switch-case 逻辑
- 支持默认路由（END）
- 适合任务分类、优先级路由等场景

## 完整代码

```python
"""
LangGraph 实战代码 - 场景3：多路由决策
演示：使用条件边实现任务分类系统
"""

from langgraph.graph import StateGraph, START, END
from typing import Literal
from typing_extensions import TypedDict


# ===== 1. 定义状态 =====
class TaskState(TypedDict):
    """任务状态"""
    task_id: str           # 任务ID
    task_type: str         # 任务类型
    priority: str          # 优先级
    content: str           # 任务内容
    result: str            # 处理结果
    handler: str           # 处理器名称
    log: list[str]         # 操作日志


# ===== 2. 定义节点函数 =====
def classify_task(state: TaskState) -> dict:
    """分类任务"""
    print(f"[分类任务] 任务ID: {state['task_id']}")
    print(f"[分类任务] 任务类型: {state['task_type']}")
    print(f"[分类任务] 优先级: {state['priority']}")

    return {
        "log": state.get("log", []) + [f"任务分类完成: {state['task_type']}"]
    }


def handle_urgent(state: TaskState) -> dict:
    """处理紧急任务"""
    print("[紧急处理器] 处理紧急任务...")

    result = f"紧急任务已处理: {state['content']}"

    return {
        "result": result,
        "handler": "urgent_handler",
        "log": state.get("log", []) + ["紧急任务处理完成"]
    }


def handle_normal(state: TaskState) -> dict:
    """处理普通任务"""
    print("[普通处理器] 处理普通任务...")

    result = f"普通任务已处理: {state['content']}"

    return {
        "result": result,
        "handler": "normal_handler",
        "log": state.get("log", []) + ["普通任务处理完成"]
    }


def handle_low(state: TaskState) -> dict:
    """处理低优先级任务"""
    print("[低优先级处理器] 处理低优先级任务...")

    result = f"低优先级任务已处理: {state['content']}"

    return {
        "result": result,
        "handler": "low_handler",
        "log": state.get("log", []) + ["低优先级任务处理完成"]
    }


def handle_batch(state: TaskState) -> dict:
    """处理批量任务"""
    print("[批量处理器] 处理批量任务...")

    result = f"批量任务已处理: {state['content']}"

    return {
        "result": result,
        "handler": "batch_handler",
        "log": state.get("log", []) + ["批量任务处理完成"]
    }


# ===== 3. 定义路由函数 =====
def route_by_priority(state: TaskState) -> Literal["urgent", "normal", "low", "batch", END]:
    """根据优先级和类型路由"""
    priority = state["priority"]
    task_type = state["task_type"]

    print(f"[路由决策] 优先级: {priority}, 类型: {task_type}")

    # 紧急任务
    if priority == "urgent":
        print("[路由决策] → urgent (紧急任务)")
        return "urgent"

    # 批量任务
    elif task_type == "batch":
        print("[路由决策] → batch (批量任务)")
        return "batch"

    # 普通任务
    elif priority == "normal":
        print("[路由决策] → normal (普通任务)")
        return "normal"

    # 低优先级任务
    elif priority == "low":
        print("[路由决策] → low (低优先级任务)")
        return "low"

    # 未知类型，直接结束
    else:
        print("[路由决策] → END (未知类型)")
        return END


# ===== 4. 构建图 =====
def create_multi_route_graph():
    """创建多路由决策图"""
    # 创建状态图
    graph = StateGraph(TaskState)

    # 添加节点
    graph.add_node("classify", classify_task)
    graph.add_node("urgent", handle_urgent)
    graph.add_node("normal", handle_normal)
    graph.add_node("low", handle_low)
    graph.add_node("batch", handle_batch)

    # 添加边
    graph.add_edge(START, "classify")

    # 添加条件边（多路由决策）
    graph.add_conditional_edges(
        "classify",
        route_by_priority,
        {
            "urgent": "urgent",
            "normal": "normal",
            "low": "low",
            "batch": "batch",
            END: END
        }
    )

    # 所有处理器都通向 END
    graph.add_edge("urgent", END)
    graph.add_edge("normal", END)
    graph.add_edge("low", END)
    graph.add_edge("batch", END)

    # 编译图
    app = graph.compile()

    return app


# ===== 5. 测试不同场景 =====
def test_urgent_task():
    """测试紧急任务"""
    print("=" * 60)
    print("测试场景 1：紧急任务")
    print("=" * 60)
    print()

    app = create_multi_route_graph()

    initial_state = {
        "task_id": "T001",
        "task_type": "urgent",
        "priority": "urgent",
        "content": "系统故障需要立即修复",
        "result": "",
        "handler": "",
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("最终结果：")
    print(f"  任务ID: {result['task_id']}")
    print(f"  处理器: {result['handler']}")
    print(f"  结果: {result['result']}")
    print()


def test_batch_task():
    """测试批量任务"""
    print("=" * 60)
    print("测试场景 2：批量任务")
    print("=" * 60)
    print()

    app = create_multi_route_graph()

    initial_state = {
        "task_id": "T002",
        "task_type": "batch",
        "priority": "normal",
        "content": "批量导入用户数据",
        "result": "",
        "handler": "",
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("最终结果：")
    print(f"  任务ID: {result['task_id']}")
    print(f"  处理器: {result['handler']}")
    print(f"  结果: {result['result']}")
    print()


def test_low_priority_task():
    """测试低优先级任务"""
    print("=" * 60)
    print("测试场景 3：低优先级任务")
    print("=" * 60)
    print()

    app = create_multi_route_graph()

    initial_state = {
        "task_id": "T003",
        "task_type": "maintenance",
        "priority": "low",
        "content": "定期系统维护",
        "result": "",
        "handler": "",
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("最终结果：")
    print(f"  任务ID: {result['task_id']}")
    print(f"  处理器: {result['handler']}")
    print(f"  结果: {result['result']}")
    print()


# ===== 6. 主函数 =====
def main():
    """主函数"""
    print("=" * 60)
    print("LangGraph 实战 - 场景3：多路由决策")
    print("=" * 60)
    print()

    # 测试三种场景
    test_urgent_task()
    test_batch_task()
    test_low_priority_task()


if __name__ == "__main__":
    main()
```

## 代码详解

### 1. 路由函数设计

```python
def route_by_priority(state: TaskState) -> Literal["urgent", "normal", "low", "batch", END]:
    """根据优先级和类型路由"""
    priority = state["priority"]
    task_type = state["task_type"]

    # 紧急任务
    if priority == "urgent":
        return "urgent"

    # 批量任务
    elif task_type == "batch":
        return "batch"

    # 普通任务
    elif priority == "normal":
        return "normal"

    # 低优先级任务
    elif priority == "low":
        return "low"

    # 未知类型，直接结束
    else:
        return END
```

**关键点**：
- **多个分支**：支持4个目标节点 + END
- **优先级判断**：按照优先级顺序判断
- **默认路由**：未知类型返回 END
- **类型提示**：使用 `Literal` 明确所有可能的返回值

---

### 2. 执行流程图

```
START
  ↓
classify (分类任务)
  ↓
route_by_priority (路由决策)
  ↓
┌─────┼─────┼─────┼─────┐
↓     ↓     ↓     ↓     ↓
urgent normal low batch END
(紧急) (普通) (低)  (批量) (结束)
↓     ↓     ↓     ↓
└─────┼─────┼─────┘
      ↓
     END
```

**执行路径**：
- **紧急任务**：START → classify → urgent → END
- **批量任务**：START → classify → batch → END
- **普通任务**：START → classify → normal → END
- **低优先级任务**：START → classify → low → END
- **未知类型**：START → classify → END

### 场景 1：工单系统

```python
class TicketState(TypedDict):
    ticket_id: str
    category: str
    severity: str
    status: str

def route_ticket(state: TicketState) -> Literal["critical", "high", "medium", "low", END]:
    """工单路由"""
    severity = state["severity"]

    if severity == "critical":
        return "critical"  # 关键工单
    elif severity == "high":
        return "high"  # 高优先级
    elif severity == "medium":
        return "medium"  # 中等优先级
    elif severity == "low":
        return "low"  # 低优先级
    else:
        return END

# 构建图
graph = StateGraph(TicketState)
graph.add_node("classify", classify_ticket)
graph.add_node("critical", handle_critical)
graph.add_node("high", handle_high)
graph.add_node("medium", handle_medium)
graph.add_node("low", handle_low)

graph.add_edge(START, "classify")
graph.add_conditional_edges("classify", route_ticket)
graph.add_edge("critical", END)
graph.add_edge("high", END)
graph.add_edge("medium", END)
graph.add_edge("low", END)
```

**应用**：客户支持工单系统的优先级路由

### 场景 2：RAG 查询路由

```python
class QueryState(TypedDict):
    query: str
    query_type: str
    result: str

def route_query(state: QueryState) -> Literal["vector_search", "keyword_search", "hybrid_search", "llm_direct", END]:
    """查询路由"""
    query_type = state["query_type"]

    if query_type == "semantic":
```

## 最佳实践

### 1. 使用 Literal 类型提示

```python
# 推荐：明确所有可能的返回值
def route_func(state: State) -> Literal["path_a", "path_b", "path_c", END]:
    ...
```

### 2. 按优先级排序条件

```python
# 推荐：高优先级条件放在前面
def route_func(state: State) -> str:
    if state["priority"] == "critical":  # 最高优先级
        return "critical"
    elif state["priority"] == "high":
        return "high"
    elif state["priority"] == "medium":
        return "medium"
    else:
        return "low"
```

### 3. 提供默认路由

```python
# 推荐：提供默认路由处理未知情况
def route_func(state: State) -> Literal["path_a", "path_b", END]:
    if condition_a:
        return "path_a"
    elif condition_b:
        return "path_b"
    else:
        return END  # 默认路由
```

### 4. 添加日志

```python
def route_func(state: State) -> str:
    value = state["value"]
    print(f"[路由决策] value={value}")

    if value > 0.8:

        return "high"
    elif value > 0.5:
        print("[路由决策] → medium")
        return "medium"
    else:
```

## 总结

**多路由决策的特点**：
1. **多个目标节点**：支持3个以上的目标节点
2. **switch-case 模式**：类似编程语言的 switch-case 语句
3. **优先级判断**：按照优先级顺序判断条件
4. **默认路由**：提供默认路由处理未知情况

**适用场景**：
- 任务分类系统
- 工单优先级路由
- RAG 查询类型路由
- 内容分级处理
- 多渠道消息路由

**关键代码**：
```python
def route_func(state: State) -> Literal["a", "b", "c", "d", END]:
    if condition_a:
        return "a"
    elif condition_b:
        return "b"
    elif condition_c:
        return "c"
    elif condition_d:
        return "d"
    else:
        return END

graph.add_conditional_edges("source", route_func)
```
