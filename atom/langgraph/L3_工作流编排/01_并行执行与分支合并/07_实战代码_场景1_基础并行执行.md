# 实战代码 - 场景1：基础并行执行

> 演示 LangGraph 中最简单的并行执行模式：扇出/扇入（Fan-out/Fan-in）

---

## 场景概述

**目标**：掌握 LangGraph 中最基础的并行执行模式

**核心技术**：
- 静态并行（多条边）
- 扇出/扇入模式
- 状态 Reducer 函数
- 同步点机制

**实际应用**：
- 并行 LLM 调用（笑话、故事、诗歌生成）
- 多数据源并行获取（新闻、天气、股票）
- 并行任务执行与结果聚合

## 完整代码示例

```python
"""
LangGraph 基础并行执行实战
演示：扇出/扇入模式、静态并行、状态 Reducer
"""

import os
import operator
from typing import Annotated, TypedDict
from dotenv import load_dotenv

from langgraph.graph import StateGraph, START, END
from langchain_openai import ChatOpenAI

# 加载环境变量
load_dotenv()

# ===== 场景1：基础扇出扇入模式 =====
print("=" * 60)
print("场景1：基础扇出扇入模式")
print("=" * 60)

# 定义状态（使用 operator.add 作为 reducer）
class BasicState(TypedDict):
    # operator.add reducer 使得列表追加而不是覆盖
    aggregate: Annotated[list, operator.add]

def node_a(state: BasicState):
    """起始节点：初始化数据"""
    print(f'节点 A 执行：当前状态 = {state["aggregate"]}')
    return {"aggregate": ["A"]}

def node_b(state: BasicState):
    """并行节点 B"""
    print(f'节点 B 执行：当前状态 = {state["aggregate"]}')
    return {"aggregate": ["B"]}

def node_c(state: BasicState):
    """并行节点 C"""
    print(f'节点 C 执行：当前状态 = {state["aggregate"]}')
    return {"aggregate": ["C"]}

def node_d(state: BasicState):
    """汇聚节点：等待所有并行节点完成"""
    print(f'节点 D 执行：当前状态 = {state["aggregate"]}')
    return {"aggregate": ["D"]}

# 构建图
builder = StateGraph(BasicState)
builder.add_node("a", node_a)
builder.add_node("b", node_b)
builder.add_node("c", node_c)
builder.add_node("d", node_d)

# 添加边：从 START 到 a
builder.add_edge(START, "a")

# 扇出：从 a 到 b 和 c（两条边 = 并行执行）
builder.add_edge("a", "b")
builder.add_edge("a", "c")

# 扇入：从 b 和 c 到 d（d 等待 b 和 c 都完成）
builder.add_edge("b", "d")
builder.add_edge("c", "d")

# 从 d 到 END
builder.add_edge("d", END)

# 编译图
graph = builder.compile()

# 执行
print("\n执行图...")
result = graph.invoke({"aggregate": []})
print(f"\n最终结果: {result['aggregate']}")
print(f"预期结果: ['A', 'B', 'C', 'D'] (顺序可能不同)")

# ===== 场景2：并行 LLM 调用 =====
print("\n" + "=" * 60)
print("场景2：并行 LLM 调用（笑话、故事、诗歌生成）")
print("=" * 60)

# 初始化 LLM
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

# 定义状态
class LLMState(TypedDict):
    topic: str
    joke: str
    story: str
    poem: str
    combined_output: str

def call_llm_joke(state: LLMState):
    """第一个 LLM 调用：生成笑话"""
    print(f"[LLM 1] 生成笑话，主题: {state['topic']}")
    msg = llm.invoke(f"Write a short joke about {state['topic']}")
    return {"joke": msg.content}

def call_llm_story(state: LLMState):
    """第二个 LLM 调用：生成故事"""
    print(f"[LLM 2] 生成故事，主题: {state['topic']}")
    msg = llm.invoke(f"Write a very short story (2-3 sentences) about {state['topic']}")
    return {"story": msg.content}

def call_llm_poem(state: LLMState):
    """第三个 LLM 调用：生成诗歌"""
    print(f"[LLM 3] 生成诗歌，主题: {state['topic']}")
    msg = llm.invoke(f"Write a short poem (4 lines) about {state['topic']}")
    return {"poem": msg.content}

def aggregator(state: LLMState):
    """聚合器：合并所有 LLM 输出"""
    print("[聚合器] 合并所有结果...")
    combined = f"主题: {state['topic']}\n\n"
    combined += f"【笑话】\n{state['joke']}\n\n"
    combined += f"【故事】\n{state['story']}\n\n"
    combined += f"【诗歌】\n{state['poem']}"
    return {"combined_output": combined}

# 构建图
llm_builder = StateGraph(LLMState)

# 添加节点
llm_builder.add_node("call_llm_joke", call_llm_joke)
llm_builder.add_node("call_llm_story", call_llm_story)
llm_builder.add_node("call_llm_poem", call_llm_poem)
llm_builder.add_node("aggregator", aggregator)

# 扇出：从 START 到三个 LLM 节点（并行执行）
llm_builder.add_edge(START, "call_llm_joke")
llm_builder.add_edge(START, "call_llm_story")
llm_builder.add_edge(START, "call_llm_poem")

# 扇入：从三个 LLM 节点到聚合器
llm_builder.add_edge("call_llm_joke", "aggregator")
llm_builder.add_edge("call_llm_story", "aggregator")
llm_builder.add_edge("call_llm_poem", "aggregator")

# 从聚合器到 END
llm_builder.add_edge("aggregator", END)

# 编译图
llm_graph = llm_builder.compile()

# 执行
print("\n执行图...")
llm_result = llm_graph.invoke({"topic": "cats"})
print("\n" + "=" * 60)
print("最终输出:")
print("=" * 60)
print(llm_result["combined_output"])

# ===== 场景3：多数据源并行获取 =====
print("\n" + "=" * 60)
print("场景3：多数据源并行获取（新闻、天气、股票）")
print("=" * 60)

# 定义状态
class DataState(TypedDict):
    news: str
    weather: str
    stocks: str
    summary: str

def fetch_news(state: DataState):
    """模拟获取新闻数据"""
    print("[新闻] 获取最新新闻...")
    # 实际应用中这里会调用新闻 API
    import time
    time.sleep(0.5)  # 模拟网络延迟
    return {"news": "今日头条：AI 技术取得重大突破"}

def fetch_weather(state: DataState):
    """模拟获取天气数据"""
    print("[天气] 获取天气信息...")
    import time
    time.sleep(0.3)  # 模拟网络延迟
    return {"weather": "今日天气：晴，温度 22°C"}

def fetch_stocks(state: DataState):
    """模拟获取股票数据"""
    print("[股票] 获取股票行情...")
    import time
    time.sleep(0.4)  # 模拟网络延迟
    return {"stocks": "股市行情：上证指数 3200 点"}

def combine_data(state: DataState):
    """合并所有数据"""
    print("[合并] 整合所有数据...")
    summary = f"【每日简报】\n\n"
    summary += f"📰 {state['news']}\n"
    summary += f"🌤️  {state['weather']}\n"
    summary += f"📈 {state['stocks']}"
    return {"summary": summary}

# 构建图
data_builder = StateGraph(DataState)

# 添加节点
data_builder.add_node("fetch_news", fetch_news)
data_builder.add_node("fetch_weather", fetch_weather)
data_builder.add_node("fetch_stocks", fetch_stocks)
data_builder.add_node("combine_data", combine_data)

# 扇出：从 START 到三个数据获取节点（并行执行）
data_builder.add_edge(START, "fetch_news")
data_builder.add_edge(START, "fetch_weather")
data_builder.add_edge(START, "fetch_stocks")

# 扇入：从三个数据获取节点到合并节点
data_builder.add_edge("fetch_news", "combine_data")
data_builder.add_edge("fetch_weather", "combine_data")
data_builder.add_edge("fetch_stocks", "combine_data")

# 从合并节点到 END
data_builder.add_edge("combine_data", END)

# 编译图
data_graph = data_builder.compile()

# 执行
print("\n执行图...")
import time
start_time = time.time()
data_result = data_graph.invoke({})
end_time = time.time()

print("\n" + "=" * 60)
print("最终输出:")
print("=" * 60)
print(data_result["summary"])
print(f"\n执行时间: {end_time - start_time:.2f} 秒")
print("注意：并行执行比串行执行快得多！")

# ===== 场景4：自定义 Reducer 函数 =====
print("\n" + "=" * 60)
print("场景4：自定义 Reducer 函数")
print("=" * 60)

# 自定义 reducer：合并字典
def merge_dicts(left: dict, right: dict) -> dict:
    """自定义 reducer：合并两个字典"""
    return {**left, **right}

# 定义状态
class CustomState(TypedDict):
    # 使用自定义 reducer
    data: Annotated[dict, merge_dicts]

def node_x(state: CustomState):
    """节点 X：添加数据"""
    print(f'节点 X 执行：当前数据 = {state.get("data", {})}')
    return {"data": {"x": 1, "source": "X"}}

def node_y(state: CustomState):
    """节点 Y：添加数据"""
    print(f'节点 Y 执行：当前数据 = {state.get("data", {})}')
    return {"data": {"y": 2, "source": "Y"}}

def node_z(state: CustomState):
    """节点 Z：汇总数据"""
    print(f'节点 Z 执行：当前数据 = {state.get("data", {})}')
    return {"data": {"z": 3, "total": state["data"].get("x", 0) + state["data"].get("y", 0)}}

# 构建图
custom_builder = StateGraph(CustomState)
custom_builder.add_node("x", node_x)
custom_builder.add_node("y", node_y)
custom_builder.add_node("z", node_z)

# 扇出：从 START 到 x 和 y
custom_builder.add_edge(START, "x")
custom_builder.add_edge(START, "y")

# 扇入：从 x 和 y 到 z
custom_builder.add_edge("x", "z")
custom_builder.add_edge("y", "z")

# 从 z 到 END
custom_builder.add_edge("z", END)

# 编译图
custom_graph = custom_builder.compile()

# 执行
print("\n执行图...")
custom_result = custom_graph.invoke({"data": {}})
print(f"\n最终结果: {custom_result['data']}")

# ===== 场景5：可视化图结构 =====
print("\n" + "=" * 60)
print("场景5：可视化图结构")
print("=" * 60)

# 打印图的 Mermaid 表示
print("\n基础扇出扇入图的 Mermaid 表示:")
print(graph.get_graph().draw_mermaid())

print("\n并行 LLM 调用图的 Mermaid 表示:")
print(llm_graph.get_graph().draw_mermaid())

# ===== 总结 =====
print("\n" + "=" * 60)
print("总结")
print("=" * 60)
print("""
基础并行执行的核心要点：

1. **扇出/扇入模式**：
   - 扇出：从一个节点添加多条边到多个节点
   - 扇入：多个节点添加边到同一个节点
   - 并行节点在同一个超步中执行

2. **静态并行**：
   - 使用多条 add_edge 实现
   - 编译时确定并行度
   - 性能开销小

3. **状态 Reducer**：
   - operator.add：列表追加、数值相加
   - 自定义函数：实现复杂合并逻辑
   - 避免状态覆盖问题

4. **同步机制**：
   - 扇入节点等待所有并行节点完成
   - 基于 Bulk Synchronous Parallel 模型
   - 自动处理同步点

5. **实际应用**：
   - 并行 LLM 调用：提高响应速度
   - 多数据源获取：减少总等待时间
   - 任务并行处理：充分利用资源
""")
```

## 核心知识点

### 1. 扇出/扇入模式

**扇出（Fan-out）**：
```python
# 从节点 a 扇出到节点 b 和 c
builder.add_edge("a", "b")
builder.add_edge("a", "c")
```

**扇入（Fan-in）**：
```python
# 从节点 b 和 c 扇入到节点 d
builder.add_edge("b", "d")
builder.add_edge("c", "d")
```

**执行顺序**：
- 节点 a 执行完成后，b 和 c 并行执行
- 节点 d 等待 b 和 c 都完成后再执行

### 2. 状态 Reducer

**内置 Reducer**：
```python
from typing import Annotated
import operator

class State(TypedDict):
    # 列表追加
    items: Annotated[list, operator.add]
    # 数值相加
    count: Annotated[int, operator.add]
```

**自定义 Reducer**：
```python
def merge_dicts(left: dict, right: dict) -> dict:
    return {**left, **right}

class State(TypedDict):
    data: Annotated[dict, merge_dicts]
```

### 3. 同步机制

**Bulk Synchronous Parallel 模型**：
- **超步（Superstep）**：所有并行节点在同一超步中执行
- **同步点（Barrier）**：超步之间有同步点，确保所有节点完成
- **状态合并**：通过 Reducer 函数自动合并状态

**执行流程**：
```
超步 1: 节点 a 执行
        ↓
同步点 1: 等待节点 a 完成
        ↓
超步 2: 节点 b 和 c 并行执行
        ↓
同步点 2: 等待节点 b 和 c 都完成
        ↓
超步 3: 节点 d 执行
```

### 4. 性能优势

**并行 vs 串行**：
```python
# 串行执行：总时间 = 0.5 + 0.3 + 0.4 = 1.2 秒
# 并行执行：总时间 = max(0.5, 0.3, 0.4) = 0.5 秒
```

**适用场景**：
- 多个独立的 LLM 调用
- 多个数据源的并行获取
- 多个独立任务的并行处理

## 实际应用场景

### 场景1：并行 LLM 调用

**问题**：串行调用多个 LLM 导致响应时间过长

**效果**：
- 减少总等待时间
- 提高用户体验
- 充分利用网络带宽

### 场景3：批量任务处理

**问题**：需要对多个文档进行相同的处理

## 常见问题

### Q1：并行节点的执行顺序是否确定？

**答**：不确定。并行节点在同一超步中执行，但具体顺序由运行时决定。

**建议**：不要依赖并行节点的执行顺序，确保它们是独立的。

### Q2：如何控制并行度？

**答**：静态并行的并行度由边的数量决定，无法动态控制。

**建议**：如果需要动态控制并行度，使用 Send API（见场景2）。

### Q3：Reducer 函数何时被调用？

**答**：当多个并行节点更新同一个状态字段时，Reducer 函数自动被调用。

**示例**：
```python
# 节点 b 返回 {"items": ["B"]}
# 节点 c 返回 {"items": ["C"]}
# Reducer 合并：["B"] + ["C"] = ["B", "C"]
```

### Q4：扇入节点如何知道所有并行节点都完成了？

**答**：LangGraph 自动跟踪节点依赖关系，扇入节点会等待所有前驱节点完成。

**机制**：基于 Bulk Synchronous Parallel 模型的同步点。

## 最佳实践

### 1. 确保并行节点独立

```python
# ✅ 好：并行节点独立
def node_b(state):
    return {"result_b": process_b()}

def node_c(state):
    return {"result_c": process_c()}

# ❌ 坏：并行节点有依赖
def node_b(state):
    return {"shared": state["shared"] + 1}  # 竞态条件

def node_c(state):
    return {"shared": state["shared"] + 1}  # 竞态条件
```

### 2. 使用合适的 Reducer

```python
# ✅ 好：使用 operator.add 追加列表
class State(TypedDict):
    results: Annotated[list, operator.add]

# ❌ 坏：没有 reducer，状态会被覆盖
class State(TypedDict):
    results: list  # 最后一个节点的结果会覆盖其他节点
```

### 3. 处理错误

```python
def safe_node(state):
    try:
        result = risky_operation()
        return {"result": result}
    except Exception as e:
        return {"error": str(e)}
```

### 4. 监控性能

```python
import time

def timed_node(state):
    start = time.time()
    result = process()
    duration = time.time() - start
    print(f"节点执行时间: {duration:.2f} 秒")
    return {"result": result}
```
