## 完整代码示例

```python
"""
场景3：配置中断点实现人机交互
演示 LangGraph 的中断机制和人机协作

来源：
- Context7: langgraph_compile_01.md
- 源码: source_编译执行_02.md
"""

import os
import sqlite3
from typing import TypedDict, Literal, Optional
from dotenv import load_dotenv

from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.sqlite import SqliteSaver
from langgraph.types import Command, interrupt

# 加载环境变量
load_dotenv()

# ===== 1. 场景A：静态中断点（interrupt_before/after）=====
print("=== 1. 场景A：静态中断点 ===\n")


class WorkflowState(TypedDict):
    """工作流状态"""
    task_name: str
    data: dict
    status: str


def prepare_node(state: WorkflowState) -> dict:
    """准备节点"""
    print(f"[prepare_node] 执行中...")
    return {
        "data": {"prepared": True},
        "status": "prepared"
    }


def process_node(state: WorkflowState) -> dict:
    """处理节点"""
    print(f"[process_node] 执行中...")
    return {
        "data": {**state["data"], "processed": True},
        "status": "processed"
    }


def finalize_node(state: WorkflowState) -> dict:
    """完成节点"""
    print(f"[finalize_node] 执行中...")
    return {
        "data": {**state["data"], "finalized": True},
        "status": "completed"
    }


# 创建图
workflow_static = StateGraph(WorkflowState)
workflow_static.add_node("prepare", prepare_node)
workflow_static.add_node("process", process_node)
workflow_static.add_node("finalize", finalize_node)

workflow_static.add_edge(START, "prepare")
workflow_static.add_edge("prepare", "process")
workflow_static.add_edge("process", "finalize")
workflow_static.add_edge("finalize", END)

# 配置静态中断点
# 来源: Context7 langgraph_compile_01.md
conn_static = sqlite3.connect("static_interrupt.db", check_same_thread=False)
app_static = workflow_static.compile(
    checkpointer=SqliteSaver(conn_static),
    interrupt_before=["process"],  # 在 process 节点前中断
    interrupt_after=["finalize"]   # 在 finalize 节点后中断
)

print("图编译完成（配置了静态中断点）")
print("  interrupt_before: ['process']")
print("  interrupt_after: ['finalize']\n")

# 准备配置
config_static = {
    "configurable": {
        "thread_id": "static-interrupt-001"
    }
}

# 准备输入
inputs_static = {
    "task_name": "static-interrupt-task",
    "data": {},
    "status": "pending"
}

print("第一次执行（将在 process 节点前中断）...")
result1 = app_static.invoke(inputs_static, config=config_static)
print(f"  状态: {result1['status']}")
print(f"  数据: {result1['data']}\n")

print("恢复执行（将继续到 finalize 节点后中断）...")
result2 = app_static.invoke(None, config=config_static)
print(f"  状态: {result2['status']}")
print(f"  数据: {result2['data']}\n")

print("再次恢复执行（完成所有节点）...")
result3 = app_static.invoke(None, config=config_static)
print(f"  状态: {result3['status']}")
print(f"  数据: {result3['data']}\n")


# ===== 2. 场景B：动态中断（interrupt()函数）=====
print("=== 2. 场景B：动态中断（审批流程）===\n")


class ApprovalState(TypedDict):
    """审批状态"""
    action_details: str
    status: Optional[Literal["pending", "approved", "rejected"]]
    approver: Optional[str]


def approval_node(state: ApprovalState) -> Command[Literal["proceed", "cancel"]]:
    """
    审批节点：等待人工审批

    来源: Context7 langgraph_compile_01.md
    """
    print(f"[approval_node] 等待审批...")
    print(f"  操作详情: {state['action_details']}")

    # 使用 interrupt() 暂停执行，等待人工决策
    decision = interrupt({
        "question": "是否批准此操作？",
        "details": state["action_details"],
        "options": ["approve", "reject"]
    })

    print(f"  收到决策: {decision}")

    # 根据决策路由到不同节点
    return Command(goto="proceed" if decision == "approve" else "cancel")


def proceed_node(state: ApprovalState) -> dict:
    """批准节点"""
    print(f"[proceed_node] 操作已批准")
    return {"status": "approved"}


def cancel_node(state: ApprovalState) -> dict:
    """拒绝节点"""
    print(f"[cancel_node] 操作已拒绝")
    return {"status": "rejected"}


# 创建审批图
workflow_approval = StateGraph(ApprovalState)
workflow_approval.add_node("approval", approval_node)
workflow_approval.add_node("proceed", proceed_node)
workflow_approval.add_node("cancel", cancel_node)

workflow_approval.add_edge(START, "approval")
workflow_approval.add_edge("proceed", END)
workflow_approval.add_edge("cancel", END)

# 编译（必须配置 checkpointer）
conn_approval = sqlite3.connect("approval.db", check_same_thread=False)
app_approval = workflow_approval.compile(checkpointer=SqliteSaver(conn_approval))

print("审批图编译完成\n")

# 执行审批流程
config_approval = {
    "configurable": {
        "thread_id": "approval-001"
    }
}

inputs_approval = {
    "action_details": "转账 $5000 到供应商账户",
    "status": "pending",
    "approver": None
}

print("提交审批请求...")
result_approval = app_approval.invoke(inputs_approval, config=config_approval)

# 检查中断信息
if "__interrupt__" in result_approval:
    print(f"\n中断信息:")
    for interrupt_data in result_approval["__interrupt__"]:
        print(f"  问题: {interrupt_data.value['question']}")
        print(f"  详情: {interrupt_data.value['details']}")
        print(f"  选项: {interrupt_data.value['options']}\n")

print("模拟人工审批（批准）...")
result_approved = app_approval.invoke(
    Command(resume="approve"),
    config=config_approval
)
print(f"  最终状态: {result_approved['status']}\n")


# ===== 3. 场景C：表单验证（循环中断）=====
print("=== 3. 场景C：表单验证（循环中断）===\n")


class FormState(TypedDict):
    """表单状态"""
    age: Optional[int]
    email: Optional[str]


def collect_age_node(state: FormState) -> dict:
    """
    收集年龄：循环验证直到输入有效

    来源: Context7 langgraph_compile_01.md
    """
    print(f"[collect_age_node] 收集年龄...")

    prompt = "请输入您的年龄："

    while True:
        # 中断并等待用户输入
        answer = interrupt(prompt)

        # 验证输入
        if isinstance(answer, int) and answer > 0 and answer < 150:
            print(f"  有效年龄: {answer}")
            return {"age": answer}

        # 输入无效，重新提示
        prompt = f"'{answer}' 不是有效的年龄。请输入 1-150 之间的数字："
        print(f"  {prompt}")


def collect_email_node(state: FormState) -> dict:
    """收集邮箱"""
    print(f"[collect_email_node] 收集邮箱...")

    prompt = "请输入您的邮箱："

    while True:
        answer = interrupt(prompt)

        # 简单验证
        if isinstance(answer, str) and "@" in answer and "." in answer:
            print(f"  有效邮箱: {answer}")
            return {"email": answer}

        prompt = f"'{answer}' 不是有效的邮箱。请输入正确的邮箱地址："
        print(f"  {prompt}")


# 创建表单图
workflow_form = StateGraph(FormState)
workflow_form.add_node("collect_age", collect_age_node)
workflow_form.add_node("collect_email", collect_email_node)

workflow_form.add_edge(START, "collect_age")
workflow_form.add_edge("collect_age", "collect_email")
workflow_form.add_edge("collect_email", END)

# 编译
conn_form = sqlite3.connect("form.db", check_same_thread=False)
app_form = workflow_form.compile(checkpointer=SqliteSaver(conn_form))

print("表单图编译完成\n")

# 执行表单收集
config_form = {
    "configurable": {
        "thread_id": "form-001"
    }
}

inputs_form = {
    "age": None,
    "email": None
}

print("开始收集表单数据...\n")

# 第一次：收集年龄
result_form1 = app_form.invoke(inputs_form, config=config_form)
print(f"中断提示: {result_form1['__interrupt__'][0].value}\n")

# 提供无效年龄
print("提供无效年龄: 'thirty'")
result_form2 = app_form.invoke(Command(resume="thirty"), config=config_form)
print(f"中断提示: {result_form2['__interrupt__'][0].value}\n")

# 提供有效年龄
print("提供有效年龄: 30")
result_form3 = app_form.invoke(Command(resume=30), config=config_form)
print(f"中断提示: {result_form3['__interrupt__'][0].value}\n")

# 提供无效邮箱
print("提供无效邮箱: 'invalid'")
result_form4 = app_form.invoke(Command(resume="invalid"), config=config_form)
print(f"中断提示: {result_form4['__interrupt__'][0].value}\n")

# 提供有效邮箱
print("提供有效邮箱: 'user@example.com'")
result_form5 = app_form.invoke(Command(resume="user@example.com"), config=config_form)
print(f"表单收集完成:")
print(f"  年龄: {result_form5['age']}")
print(f"  邮箱: {result_form5['email']}\n")


# ===== 4. 中断机制对比 =====
print("=== 4. 中断机制对比 ===\n")

comparison = """

│   中断类型   │                     特点                           │

print(comparison)


# ===== 5. 实际应用示例 =====
print("=== 5. 实际应用示例：多级审批流程 ===\n")


class MultiApprovalState(TypedDict):
    """多级审批状态"""
    request: str
    amount: float
    approvals: list[dict]
    status: str


def request_node(state: MultiApprovalState) -> dict:
    """提交请求"""
    print(f"[request_node] 提交请求: {state['request']}")
    return {"status": "pending"}


def manager_approval_node(state: MultiApprovalState) -> Command[Literal["director", "reject"]]:
    """经理审批"""
    print(f"[manager_approval_node] 等待经理审批...")

    decision = interrupt({
        "level": "manager",
        "request": state["request"],
        "amount": state["amount"],
        "question": "经理是否批准？"
    })

    if decision == "approve":
        return Command(
            update={"approvals": state["approvals"] + [{"level": "manager", "decision": "approved"}]},
            goto="director"
        )
    else:
        return Command(
            update={"status": "rejected", "approvals": state["approvals"] + [{"level": "manager", "decision": "rejected"}]},
            goto="reject"
        )


def director_approval_node(state: MultiApprovalState) -> Command[Literal["approve", "reject"]]:
    """总监审批"""
    print(f"[director_approval_node] 等待总监审批...")

    decision = interrupt({
        "level": "director",
        "request": state["request"],
        "amount": state["amount"],
        "previous_approvals": state["approvals"],
        "question": "总监是否批准？"
    })

    if decision == "approve":
        return Command(
            update={"approvals": state["approvals"] + [{"level": "director", "decision": "approved"}]},
            goto="approve"
        )
    else:
        return Command(
            update={"status": "rejected", "approvals": state["approvals"] + [{"level": "director", "decision": "rejected"}]},
            goto="reject"
        )


def approve_node(state: MultiApprovalState) -> dict:
    """批准节点"""
    print(f"[approve_node] 请求已批准")
    return {"status": "approved"}


def reject_node(state: MultiApprovalState) -> dict:
    """拒绝节点"""
    print(f"[reject_node] 请求已拒绝")
    return {"status": "rejected"}


# 创建多级审批图
workflow_multi = StateGraph(MultiApprovalState)
workflow_multi.add_node("request", request_node)
workflow_multi.add_node("manager", manager_approval_node)
workflow_multi.add_node("director", director_approval_node)
workflow_multi.add_node("approve", approve_node)
workflow_multi.add_node("reject", reject_node)

workflow_multi.add_edge(START, "request")
workflow_multi.add_edge("request", "manager")
workflow_multi.add_edge("approve", END)
workflow_multi.add_edge("reject", END)

# 编译
conn_multi = sqlite3.connect("multi_approval.db", check_same_thread=False)
app_multi = workflow_multi.compile(checkpointer=SqliteSaver(conn_multi))

print("多级审批图编译完成\n")

# 执行多级审批
config_multi = {
    "configurable": {
        "thread_id": "multi-approval-001"
    }
}

inputs_multi = {
    "request": "购买新服务器",
    "amount": 50000.0,
    "approvals": [],
    "status": "pending"
}

print("提交审批请求...")
result_m1 = app_multi.invoke(inputs_multi, config=config_multi)
print(f"等待经理审批...\n")

print("经理批准...")
result_m2 = app_multi.invoke(Command(resume="approve"), config=config_multi)
print(f"等待总监审批...\n")

print("总监批准...")
result_m3 = app_multi.invoke(Command(resume="approve"), config=config_multi)
print(f"最终状态: {result_m3['status']}")
print(f"审批记录: {result_m3['approvals']}\n")


# ===== 6. 性能考虑 =====
print("=== 6. 性能考虑 ===\n")

performance_tips = """

print("=== 7. 总结 ===\n")
```

## 关键知识点

### 1. 静态中断配置

```python
app = workflow.compile(
    checkpointer=checkpointer,
    interrupt_before=["node_a"],  # 在节点前中断
    interrupt_after=["node_b"]    # 在节点后中断
)

# 恢复执行
result = app.invoke(None, config=config)
```

### 2. 动态中断使用

```python
def node(state):
    # 中断并等待输入
    answer = interrupt({
        "question": "请输入数据",
        "context": "..."
    })

    # 使用输入继续执行
    return {"result": answer}

# 恢复执行
result = app.invoke(Command(resume="user_input"), config=config)
```

### 3. 中断数据访问

```python
result = app.invoke(inputs, config=config)

# 检查是否有中断
if "__interrupt__" in result:
    for interrupt_data in result["__interrupt__"]:
        print(interrupt_data.value)
```
