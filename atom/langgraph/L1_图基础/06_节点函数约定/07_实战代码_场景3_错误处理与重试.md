## 场景 1：RetryPolicy 基础配置

### 代码示例

```python
"""
场景 1：RetryPolicy 基础配置
演示如何配置基础的重试策略
"""

from langgraph.graph import START, END, StateGraph
from langgraph.types import RetryPolicy
from typing_extensions import TypedDict
import random

# 定义状态
class State(TypedDict):
    result: str
    attempts: int
    error_count: int

# 模拟不稳定的 API 调用
def unreliable_api_call(state: State) -> dict:
    """模拟一个不稳定的 API，有 50% 概率失败"""
    attempts = state.get("attempts", 0) + 1
    print(f"尝试第 {attempts} 次调用...")

    if random.random() < 0.5:  # 50% 失败概率
        raise Exception("API 调用失败：网络超时")

    return {
        "result": "API 调用成功！",
        "attempts": attempts,
        "error_count": state.get("error_count", 0)
    }

# 构建图
builder = StateGraph(State)

# 添加节点并配置 RetryPolicy
builder.add_node(
    "api_call",
    unreliable_api_call,
    retry_policy=RetryPolicy(
        initial_interval=0.5,      # 初始重试间隔 0.5 秒
        backoff_factor=2.0,        # 每次重试间隔翻倍
        max_interval=10.0,         # 最大重试间隔 10 秒
        max_attempts=5,            # 最多重试 5 次
        jitter=True,               # 添加随机抖动，避免雷鸣群效应
        retry_on=Exception         # 对所有异常进行重试
    )
)

builder.add_edge(START, "api_call")
builder.add_edge("api_call", END)

# 编译并运行
graph = builder.compile()

try:
    result = graph.invoke({"result": "", "attempts": 0, "error_count": 0})
    print(f"\n最终结果: {result['result']}")
    print(f"总尝试次数: {result['attempts']}")
except Exception as e:
    print(f"\n所有重试都失败了: {e}")
```

### 关键点

1. **指数退避算法**：重试间隔按指数增长（0.5s → 1s → 2s → 4s → 8s）
2. **抖动机制**：添加随机性，避免多个请求同时重试
3. **最大限制**：限制最大重试次数和最大间隔时间

## 场景 2：自定义重试条件

### 代码示例

```python
"""
场景 2：自定义重试条件
演示如何使用自定义函数判断是否重试
"""

from langgraph.graph import START, END, StateGraph
from langgraph.types import RetryPolicy
from typing_extensions import TypedDict
import random

# 定义状态
class State(TypedDict):
    result: str
    attempts: int
    rate_limit_errors: int
    other_errors: int

# 自定义重试条件：只对速率限制错误重试
def retry_on_rate_limit(error: Exception) -> bool:
    """只对包含 'rate limit' 的错误进行重试"""
    error_msg = str(error).lower()
    return "rate limit" in error_msg or "too many requests" in error_msg

# 模拟 API 调用，可能抛出不同类型的错误
def api_with_different_errors(state: State) -> dict:
    """模拟 API 调用，可能抛出速率限制或其他错误"""
    attempts = state.get("attempts", 0) + 1
    print(f"尝试第 {attempts} 次调用...")

    rand = random.random()
    if rand < 0.3:
        # 30% 概率：速率限制错误（会重试）
        raise Exception("Rate limit exceeded: Too many requests")
    elif rand < 0.5:
        # 20% 概率：其他错误（不会重试）
        raise ValueError("Invalid API key")

    # 50% 概率：成功
    return {
        "result": "API 调用成功！",
        "attempts": attempts,
        "rate_limit_errors": state.get("rate_limit_errors", 0),
        "other_errors": state.get("other_errors", 0)
    }

# 构建图
builder = StateGraph(State)

# 使用自定义重试条件
builder.add_node(
    "api_call",
    api_with_different_errors,
    retry_policy=RetryPolicy(
        initial_interval=1.0,
        backoff_factor=2.0,
        max_interval=10.0,
        max_attempts=3,
        jitter=True,
        retry_on=retry_on_rate_limit  # 使用自定义重试条件
    )
)

builder.add_edge(START, "api_call")
builder.add_edge("api_call", END)

# 编译并运行
graph = builder.compile()

try:
    result = graph.invoke({
        "result": "",
        "attempts": 0,
        "rate_limit_errors": 0,
        "other_errors": 0
    })
    print(f"\n最终结果: {result['result']}")
    print(f"总尝试次数: {result['attempts']}")
except ValueError as e:
    print(f"\n遇到不可重试的错误: {e}")
except Exception as e:
    print(f"\n所有重试都失败了: {e}")
```

### 关键点

1. **自定义判断函数**：`retry_on` 参数接受 `Callable[[Exception], bool]`
2. **选择性重试**：只对特定类型的错误进行重试
3. **快速失败**：对于不可重试的错误，立即抛出

## 场景 3：错误恢复策略

### 代码示例

```python
"""
场景 3：错误恢复策略
演示如何在节点内部捕获错误并实现优雅降级
"""

from langgraph.graph import START, END, StateGraph
from langgraph.types import RetryPolicy
from typing_extensions import TypedDict
import random

# 定义状态
class State(TypedDict):
    result: str
    error: str | None
    fallback_used: bool
    attempts: int

# 主 API 调用（可能失败）
def primary_api_call(state: State) -> dict:
    """主 API 调用，失败时使用降级策略"""
    attempts = state.get("attempts", 0) + 1
    print(f"尝试主 API（第 {attempts} 次）...")

    try:
        # 模拟 API 调用
        if random.random() < 0.7:  # 70% 失败概率
            raise ConnectionError("主 API 连接失败")

        return {
            "result": "主 API 调用成功",
            "error": None,
            "fallback_used": False,
            "attempts": attempts
        }

    except ConnectionError as e:
        # 捕获错误，使用降级策略
        print(f"主 API 失败: {e}，使用降级策略...")

        # 降级策略：使用缓存或默认值
        fallback_result = "使用缓存数据（降级策略）"

        return {
            "result": fallback_result,
            "error": str(e),
            "fallback_used": True,
            "attempts": attempts
        }

# 构建图
builder = StateGraph(State)

# 配置重试策略
builder.add_node(
    "api_call",
    primary_api_call,
    retry_policy=RetryPolicy(
        initial_interval=0.5,
        backoff_factor=2.0,
        max_interval=5.0,
        max_attempts=3,
        jitter=True,
        retry_on=ConnectionError  # 只对连接错误重试
    )
)

builder.add_edge(START, "api_call")
builder.add_edge("api_call", END)

# 编译并运行
graph = builder.compile()

result = graph.invoke({
    "result": "",
    "error": None,
    "fallback_used": False,
    "attempts": 0
})

print(f"\n最终结果: {result['result']}")
print(f"是否使用降级策略: {result['fallback_used']}")
print(f"总尝试次数: {result['attempts']}")
if result['error']:
    print(f"错误信息: {result['error']}")
```

### 关键点

1. **优雅降级**：在节点内部捕获错误，返回降级结果
2. **错误记录**：在状态中记录错误信息
3. **双重保护**：节点内部 try-catch + RetryPolicy

## 场景 4：流式错误处理

### 代码示例

```python
"""
场景 4：流式错误处理
演示如何在流式执行中处理错误
"""

from langgraph.graph import START, END, StateGraph
from langgraph.types import RetryPolicy
from typing_extensions import TypedDict
import asyncio
import random

# 定义状态
class State(TypedDict):
    messages: list[str]
    error: str | None

# 异步节点：流式生成消息
async def streaming_node(state: State) -> dict:
    """流式生成消息，可能在中途失败"""
    messages = state.get("messages", [])

    # 模拟流式生成
    for i in range(5):
        await asyncio.sleep(0.1)

        # 模拟中途失败
        if i == 3 and random.random() < 0.5:
            raise Exception(f"流式生成在第 {i} 步失败")

        messages.append(f"消息 {i}")
        print(f"生成: 消息 {i}")

    return {"messages": messages, "error": None}

# 构建图
builder = StateGraph(State)

builder.add_node(
    "streaming",
    streaming_node,
    retry_policy=RetryPolicy(
        initial_interval=0.5,
        backoff_factor=2.0,
        max_interval=5.0,
        max_attempts=3,
        jitter=False,
        retry_on=Exception
    )
)

builder.add_edge(START, "streaming")
builder.add_edge("streaming", END)

# 编译并运行
graph = builder.compile()

async def run_streaming():
    """运行流式图并处理错误"""
    try:
        # 使用 astream 进行流式执行
        async for event in graph.astream({"messages": [], "error": None}):
            print(f"事件: {event}")

    except Exception as e:
        print(f"\n流式执行失败: {e}")

# 运行
asyncio.run(run_streaming())
```

### 关键点

1. **流式重试**：RetryPolicy 支持流式执行的重试
2. **状态清理**：重试时会清除之前的写入
3. **错误传播**：流式错误会正确传播到上层

## 场景 5：异步取消错误处理

### 代码示例

```python
"""
场景 5：异步取消错误处理
演示如何正确处理 asyncio.CancelledError
"""

from langgraph.graph import START, END, StateGraph
from langgraph.types import RetryPolicy
from typing_extensions import TypedDict
import asyncio
import random

# 定义状态
class State(TypedDict):
    result: str
    cancelled: bool

# 异步节点：可能被取消
async def cancellable_node(state: State) -> dict:
    """可能被取消的异步节点"""
    try:
        print("开始长时间运行的任务...")

        # 模拟长时间运行的任务
        for i in range(10):
            await asyncio.sleep(0.5)
            print(f"进度: {i+1}/10")

            # 模拟随机失败
            if random.random() < 0.1:
                raise Exception("任务执行失败")

        return {
            "result": "任务完成",
            "cancelled": False
        }

    except asyncio.CancelledError:
        # 特殊处理取消错误
        print("任务被取消，正在清理资源...")

        # 清理资源
        await asyncio.sleep(0.1)

        # 重新抛出，确保正确取消
        raise

    except Exception as e:
        # 处理其他错误
        print(f"任务失败: {e}")
        return {
            "result": f"任务失败: {e}",
            "cancelled": False
        }

# 构建图
builder = StateGraph(State)

builder.add_node(
    "task",
    cancellable_node,
    retry_policy=RetryPolicy(
        initial_interval=1.0,
        backoff_factor=2.0,
        max_interval=5.0,
        max_attempts=3,
        jitter=True,
        retry_on=Exception  # 不重试 CancelledError
    )
)

builder.add_edge(START, "task")
builder.add_edge("task", END)

# 编译并运行
graph = builder.compile()

async def run_with_timeout():
    """运行任务，带超时控制"""
    try:
        # 设置 3 秒超时
        result = await asyncio.wait_for(
            graph.ainvoke({"result": "", "cancelled": False}),
            timeout=3.0
        )
        print(f"\n最终结果: {result['result']}")

    except asyncio.TimeoutError:
        print("\n任务超时被取消")

    except Exception as e:
        print(f"\n任务失败: {e}")

# 运行
asyncio.run(run_with_timeout())
```

### 关键点

1. **特殊处理 CancelledError**：单独捕获并重新抛出
2. **资源清理**：在取消时清理资源
3. **不重试取消错误**：CancelledError 不应该被重试

[来源: reference/search_节点函数约定_01.md - 行 78-87, 行 159-174]

## 总结

### RetryPolicy 参数对照表

| 参数 | 类型 | 说明 | 默认值 |
|------|------|------|--------|
| `max_attempts` | int | 最大重试次数 | 3 |
| `initial_interval` | float | 初始重试间隔（秒） | 1.0 |
| `backoff_factor` | float | 退避因子（指数增长） | 2.0 |
| `max_interval` | float | 最大重试间隔（秒） | 10.0 |
| `jitter` | bool | 是否添加随机抖动 | True |
| `retry_on` | Exception \| Callable | 重试条件 | Exception |

[来源: reference/source_节点函数约定_02.md - 行 19-28, reference/context7_langgraph_01.md - 行 219-226]

### 最佳实践

1. **合理配置重试次数**：避免无限重试，通常 3-5 次即可
2. **使用指数退避**：避免对服务器造成压力
3. **添加抖动**：防止雷鸣群效应
4. **选择性重试**：只对可恢复的错误重试
5. **优雅降级**：在节点内部实现降级策略
6. **正确处理取消**：特殊处理 CancelledError
