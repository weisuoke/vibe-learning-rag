# 实战代码 - 场景1：基础编译与执行

> 演示 LangGraph 图的基础编译和执行流程，包括 invoke 和 stream 两种执行模式

---

## 场景概述

本场景演示：
1. 如何编译 StateGraph 成可执行的图
2. 使用 `invoke()` 方法同步执行图
3. 使用 `stream()` 方法流式执行图
4. 理解执行过程中的状态流转
5. 观察节点的执行顺序

**实际应用**：
- 简单的工作流执行
- 快速原型开发
- 调试和测试图结构

---

## 完整代码示例

```python
"""
场景1：基础编译与执行
演示 LangGraph 图的编译和两种基本执行模式

来源：
- Context7: langgraph_examples_01.md
- Context7: langgraph_invoke_01.md
- 源码: source_编译执行_01.md
"""

import os
from typing import TypedDict, Annotated
from operator import add
from dotenv import load_dotenv

from langgraph.graph import StateGraph, START, END

# 加载环境变量
load_dotenv()

# ===== 1. 定义状态 Schema =====
print("=== 1. 定义状态 Schema ===\n")


class WorkflowState(TypedDict):
    """
    工作流状态定义

    类比：
    - 前端：Redux store 的 state 结构
    - 日常：工作台上的文件夹和文档
    """
    input_text: str  # 输入文本
    processed_text: str  # 处理后的文本
    word_count: int  # 单词计数
    steps: Annotated[list[str], add]  # 执行步骤记录（使用 add reducer）


print("状态 Schema 定义完成")
print(f"字段: {list(WorkflowState.__annotations__.keys())}")
print()


# ===== 2. 定义节点函数 =====
print("=== 2. 定义节点函数 ===\n")


def preprocess_node(state: WorkflowState) -> dict:
    """
    预处理节点：清理和标准化输入文本

    类比：
    - 前端：数据清洗组件
    - 日常：文档格式化
    """
    print(f"[preprocess_node] 执行中...")
    print(f"  输入: {state['input_text']}")

    # 清理文本：去除多余空格，转小写
    processed = state['input_text'].strip().lower()

    print(f"  输出: {processed}")

    return {
        "processed_text": processed,
        "steps": ["预处理完成"]
    }


def analyze_node(state: WorkflowState) -> dict:
    """
    分析节点：统计单词数量

    类比：
    - 前端：数据分析组件
    - 日常：文档统计
    """
    print(f"[analyze_node] 执行中...")
    print(f"  输入: {state['processed_text']}")

    # 统计单词数
    words = state['processed_text'].split()
    word_count = len(words)

    print(f"  单词数: {word_count}")

    return {
        "word_count": word_count,
        "steps": ["分析完成"]
    }


def format_node(state: WorkflowState) -> dict:
    """
    格式化节点：生成最终报告

    类比：
    - 前端：渲染组件
    - 日常：生成报告
    """
    print(f"[format_node] 执行中...")

    report = f"""
文本分析报告
============
原始文本: {state['input_text']}
处理后: {state['processed_text']}
单词数: {state['word_count']}
执行步骤: {' -> '.join(state['steps'])}
"""

    print(report)

    return {
        "steps": ["格式化完成"]
    }


print("节点函数定义完成")
print()


# ===== 3. 构建图结构 =====
print("=== 3. 构建图结构 ===\n")

# 创建 StateGraph
workflow = StateGraph(WorkflowState)

# 添加节点
workflow.add_node("preprocess", preprocess_node)
workflow.add_node("analyze", analyze_node)
workflow.add_node("format", format_node)

# 添加边（定义执行顺序）
workflow.add_edge(START, "preprocess")
workflow.add_edge("preprocess", "analyze")
workflow.add_edge("analyze", "format")
workflow.add_edge("format", END)

print("图结构构建完成")
print("节点: preprocess -> analyze -> format")
print()


# ===== 4. 编译图 =====
print("=== 4. 编译图 ===\n")

# 编译成可执行的图
# 来源: Context7 langgraph_examples_01.md
app = workflow.compile()

print("图编译完成")
print(f"类型: {type(app)}")
print(f"可用方法: invoke, stream, ainvoke, astream")
print()


# ===== 5. 执行模式1：invoke（同步执行，获取最终结果）=====
print("=== 5. 执行模式1：invoke ===\n")

# 准备输入
inputs = {
    "input_text": "  Hello World! This is a TEST.  ",
    "processed_text": "",
    "word_count": 0,
    "steps": []
}

print("执行 invoke() 方法...")
print(f"输入: {inputs['input_text']}")
print()

# 执行图
# 来源: Context7 langgraph_invoke_01.md
result = app.invoke(inputs)

print("\n执行完成！")
print(f"最终状态: {result}")
print()


# ===== 6. 执行模式2：stream（流式执行，观察每个节点）=====
print("=== 6. 执行模式2：stream ===\n")

# 准备新的输入
inputs2 = {
    "input_text": "  LangGraph is AWESOME for building AI workflows!  ",
    "processed_text": "",
    "word_count": 0,
    "steps": []
}

print("执行 stream() 方法...")
print(f"输入: {inputs2['input_text']}")
print()

# 流式执行图
# 来源: Context7 langgraph_examples_01.md
print("流式输出（每个节点的执行结果）：")
print("-" * 60)

for output in app.stream(inputs2):
    for key, value in output.items():
        print(f"\n节点 '{key}' 执行完成:")
        print(f"  更新的状态: {value}")
    print("-" * 60)

print("\n流式执行完成！")
print()


# ===== 7. 对比两种执行模式 =====
print("=== 7. 两种执行模式对比 ===\n")

comparison = """
┌─────────────┬──────────────────────────────────────────────────────┐
│   模式      │                     特点                             │
├─────────────┼──────────────────────────────────────────────────────┤
│ invoke()    │ - 同步执行，等待所有节点完成                         │
│             │ - 返回最终状态                                       │
│             │ - 适合：简单工作流，只关心最终结果                   │
│             │ - 类比：前端的 Promise.then()                        │
├─────────────┼──────────────────────────────────────────────────────┤
│ stream()    │ - 流式执行，逐个节点输出                             │
│             │ - 返回每个节点的更新                                 │
│             │ - 适合：调试、监控、实时反馈                         │
│             │ - 类比：前端的 Server-Sent Events                    │
└─────────────┴──────────────────────────────────────────────────────┘
"""

print(comparison)
print()


# ===== 8. 理解编译过程 =====
print("=== 8. 理解编译过程 ===\n")

explanation = """
编译过程（compile）做了什么？
================================

1. 验证图结构
   - 检查所有节点是否连接正确
   - 验证状态 Schema 的一致性
   - 确保没有孤立节点

2. 创建执行引擎
   - 将 StateGraph 转换为 CompiledStateGraph
   - 实现 Runnable 接口（invoke, stream, batch）
   - 准备 Pregel 执行循环

3. 优化执行路径
   - 分析节点依赖关系
   - 确定执行顺序
   - 准备并行执行策略（如果有）

4. 返回可执行对象
   - 类型: CompiledStateGraph
   - 可以被多次调用
   - 线程安全（如果配置了 checkpointer）

来源: 源码 source_编译执行_02.md
"""

print(explanation)
print()


# ===== 9. 执行流程可视化 =====
print("=== 9. 执行流程可视化 ===\n")

flow_diagram = """
执行流程（Pregel 算法）
========================

用户调用 app.invoke(inputs)
         ↓
    [Plan 阶段]
    - 确定要执行的节点
    - 准备输入数据
         ↓
    [Execution 阶段]
    - 执行 preprocess_node
    - 更新状态
         ↓
    [Update 阶段]
    - 应用状态更新
    - 检查是否有下一个节点
         ↓
    [Plan 阶段]
    - 确定下一个节点: analyze
         ↓
    [Execution 阶段]
    - 执行 analyze_node
         ↓
    [Update 阶段]
    - 应用状态更新
         ↓
    [Plan 阶段]
    - 确定下一个节点: format
         ↓
    [Execution 阶段]
    - 执行 format_node
         ↓
    [Update 阶段]
    - 应用状态更新
    - 到达 END 节点
         ↓
    返回最终状态

来源: Context7 langgraph_pregel_01.md
"""

print(flow_diagram)
print()


# ===== 10. 实际应用示例 =====
print("=== 10. 实际应用示例 ===\n")


def create_text_processor():
    """
    创建一个可复用的文本处理器

    实际应用：
    - 文档处理管道
    - 数据清洗工作流
    - 内容分析系统
    """
    workflow = StateGraph(WorkflowState)

    workflow.add_node("preprocess", preprocess_node)
    workflow.add_node("analyze", analyze_node)
    workflow.add_node("format", format_node)

    workflow.add_edge(START, "preprocess")
    workflow.add_edge("preprocess", "analyze")
    workflow.add_edge("analyze", "format")
    workflow.add_edge("format", END)

    return workflow.compile()


# 创建处理器
processor = create_text_processor()

# 批量处理多个文本
texts = [
    "Python is great!",
    "LangGraph makes AI workflows easy.",
    "Compile once, execute many times."
]

print("批量处理文本:")
for i, text in enumerate(texts, 1):
    print(f"\n文本 {i}: {text}")
    result = processor.invoke({
        "input_text": text,
        "processed_text": "",
        "word_count": 0,
        "steps": []
    })
    print(f"  单词数: {result['word_count']}")

print()


# ===== 11. 性能考虑 =====

performance_tips = """
性能优化建议

# ===== 12. 常见错误和解决方案 =====
print("=== 12. 常见错误和解决方案 ===\n")

common_errors = """
常见错误
```

## 关键知识点

### 1. 编译过程

```python
# 编译将 StateGraph 转换为可执行的图
app = workflow.compile()

# 编译后的类型
type(app)  # CompiledStateGraph

# 可用方法
app.invoke()   # 同步执行
app.stream()   # 流式执行
app.ainvoke()  # 异步执行
app.astream()  # 异步流式执行
```

### 2. invoke vs stream

| 特性 | invoke() | stream() |
|------|----------|----------|
| 返回值 | 最终状态 | 每个节点的更新 |
| 执行方式 | 等待所有节点完成 | 逐个节点输出 |
| 适用场景 | 只需要最终结果 | 需要实时反馈、调试 |
| 性能 | 略快（无中间输出） | 略慢（需要序列化输出） |

### 3. Pregel 执行模型

```
每个超步（Superstep）包含三个阶段：

1. Plan（计划）
   - 确定要执行哪些节点
   - 准备输入数据

2. Execution（执行）
   - 并行执行选定的节点
   - 生成状态更新

3. Update（更新）
   - 应用状态更新
   - 检查终止条件
```

## 实际应用场景

### 1. 文档处理管道

```python
# 编译一次
doc_processor = workflow.compile()

for doc in documents:
```

### 2. 实时监控

```python
# 使用 stream 观察执行过程
for output in app.stream(inputs):
    log_to_dashboard(output)
    notify_user(output)
```

### 3. 批量处理

```python
# 异步批量处理
import asyncio

async def process_batch(texts):
    tasks = [app.ainvoke({"input_text": text}) for text in texts]
    return await asyncio.gather(*tasks)
```
