# 实战代码 - 场景3：网页内容提取

## 场景描述

构建一个网页内容提取系统，使用HTMLHeaderTextSplitter智能分割网页内容，保留HTML文档的层级结构，支持从URL直接加载并分块，提取丰富的元数据信息。该场景适用于网页爬虫、技术文档索引、知识库构建等应用。

**应用场景**：
- 技术文档爬取与索引
- 新闻网站内容提取
- 博客文章结构化存储
- 在线教程知识库构建

**核心价值**：
- 保留HTML层级结构（h1, h2, h3等）
- 自动提取标题元数据
- 支持URL直接加载
- 适配网页问答和检索场景

---

## 技术选型

### 为什么选择 HTMLHeaderTextSplitter？

**1. 结构感知能力**
```html
<!-- 普通分块器：丢失层级信息 -->
<h1>主标题</h1>
<p>内容1</p>
<h2>子标题</h2>
<p>内容2</p>
<!-- 可能被随意切断 -->

<!-- HTMLHeaderTextSplitter：保留层级 -->
块1: {content: "内容1", metadata: {h1: "主标题"}}
块2: {content: "内容2", metadata: {h1: "主标题", h2: "子标题"}}
```

**2. 元数据丰富**
- 自动提取标题层级
- 保留文档结构信息
- 便于溯源和导航

**3. 网页优化**
- 直接从URL加载
- 自动处理HTML标签
- 过滤无关内容

**对比其他分块器**：
| 分块器 | 结构感知 | 元数据 | 适用场景 |
|--------|---------|--------|---------|
| HTMLHeaderTextSplitter | ⭐⭐⭐⭐⭐ | ⭐⭐⭐⭐⭐ | 网页内容（推荐） |
| RecursiveCharacterTextSplitter | ⭐⭐⭐ | ⭐⭐ | 通用文本 |
| MarkdownHeaderTextSplitter | ⭐⭐⭐⭐⭐ | ⭐⭐⭐⭐ | Markdown文档 |

## 完整代码实现

```python
"""
网页内容提取系统 - 完整实现
支持：URL加载 + HTML结构保留 + 元数据提取 + 向量检索

依赖安装：
pip install langchain langchain-text-splitters langchain-openai langchain-community chromadb beautifulsoup4 lxml requests python-dotenv
"""

import os
from pathlib import Path
from typing import List, Dict, Any, Optional
from dotenv import load_dotenv
import requests
from bs4 import BeautifulSoup

# LangChain核心组件
from langchain_text_splitters import HTMLHeaderTextSplitter, RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma
from langchain.schema import Document

# 加载环境变量
load_dotenv()


class WebContentExtractor:
    """
    网页内容提取器

    功能：
    1. URL直接加载（支持批量）
    2. HTML结构保留（h1-h6标题）
    3. 元数据提取（标题层级、URL来源）
    4. 二次分块（可选，处理长段落）
    5. 向量化存储与检索

    设计原则：
    - 结构感知：保留HTML层级
    - 元数据丰富：标题、URL、时间戳
    - 灵活配置：可选二次分块
    - 错误处理：网络异常、解析失败
    """

    def __init__(
        self,
        headers_to_split_on: Optional[List[tuple]] = None,
        enable_secondary_split: bool = True,
        chunk_size: int = 1000,
        chunk_overlap: int = 200,
        embedding_model: str = "text-embedding-3-small",
        persist_directory: str = "./web_chroma_db"
    ):
        """
        初始化网页内容提取器

        参数说明：
        - headers_to_split_on: 要分割的HTML标题标签
          * 格式：[("h1", "Header 1"), ("h2", "Header 2")]
          * 默认：h1-h3标题

        - enable_secondary_split: 是否启用二次分块
          * True: 对长段落进行二次分块
          * False: 只按标题分块

        - chunk_size: 二次分块大小（字符数）
        - chunk_overlap: 二次分块重叠（字符数）
        - embedding_model: OpenAI Embedding模型
        - persist_directory: 向量数据库持久化目录
        """
        # 默认标题配置
        if headers_to_split_on is None:
            headers_to_split_on = [
                ("h1", "Header 1"),
                ("h2", "Header 2"),
                ("h3", "Header 3"),
            ]

        self.headers_to_split_on = headers_to_split_on
        self.enable_secondary_split = enable_secondary_split
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.persist_directory = persist_directory

        # 初始化HTML分块器
        self.html_splitter = HTMLHeaderTextSplitter(
            headers_to_split_on=headers_to_split_on
        )

        # 初始化二次分块器（可选）
        if enable_secondary_split:
            self.text_splitter = RecursiveCharacterTextSplitter(
                chunk_size=chunk_size,
                chunk_overlap=chunk_overlap
            )

        # Embedding模型
        self.embeddings = OpenAIEmbeddings(model=embedding_model)

        # 向量存储（延迟初始化）
        self.vectorstore: Optional[Chroma] = None

        print(f"网页内容提取器初始化完成")
        print(f"   - 标题层级: {[h[0] for h in headers_to_split_on]}")
        print(f"   - 二次分块: {'启用' if enable_secondary_split else '禁用'}")
        if enable_secondary_split:
            print(f"   - chunk_size: {chunk_size}")

    def fetch_html(self, url: str, timeout: int = 10) -> str:
        """
        从URL获取HTML内容

        参数：
        - url: 网页URL
        - timeout: 请求超时时间（秒）

        返回：
        - HTML内容字符串
        """
        print(f"\n获取网页: {url}")

        try:
            response = requests.get(url, timeout=timeout)
            response.raise_for_status()
            response.encoding = response.apparent_encoding

            print(f"   ✓ 获取成功: {len(response.text)} 字符")

            return response.text
        except requests.exceptions.RequestException as e:
            print(f"   ✗ 获取失败: {e}")
            raise

    def split_html(
        self,
        html_content: str,
        source_url: Optional[str] = None
    ) -> List[Document]:
        """
        分割HTML内容

        参数：
        - html_content: HTML内容字符串
        - source_url: 来源URL（用于元数据）

        返回：
        - Document对象列表
        """
        print(f"\n分割HTML内容")

        # 使用HTMLHeaderTextSplitter分割
        documents = self.html_splitter.split_text(html_content)

        # 添加来源URL到元数据
        if source_url:
            for doc in documents:
                doc.metadata["source"] = source_url

        print(f"   ✓ 分割完成: {len(documents)} 个块")

        # 统计元数据
        header_counts = {}
        for doc in documents:
            for key in doc.metadata:
                if key.startswith("Header"):
                    header_counts[key] = header_counts.get(key, 0) + 1

        if header_counts:
            print(f"   - 标题分布: {header_counts}")

        return documents

    def split_from_url(self, url: str) -> List[Document]:
        """
        从URL直接分割

        参数：
        - url: 网页URL

        返回：
        - Document对象列表
        """
        # 方法1：使用split_text_from_url（推荐）
        print(f"\n从URL分割: {url}")

        try:
            documents = self.html_splitter.split_text_from_url(url)

            # 添加来源URL到元数据
            for doc in documents:
                doc.metadata["source"] = url

            print(f"   ✓ 分割完成: {len(documents)} 个块")

            return documents
        except Exception as e:
            print(f"   ✗ 分割失败: {e}")
            raise

    def secondary_split(self, documents: List[Document]) -> List[Document]:
        """
        二次分块（处理长段落）

        参数：
        - documents: 初次分块的Document对象列表

        返回：
        - 二次分块后的Document对象列表
        """
        if not self.enable_secondary_split:
            return documents

        print(f"\n二次分块: {len(documents)} 个块")

        # 统计需要二次分块的文档
        long_docs = [doc for doc in documents if len(doc.page_content) > self.chunk_size]

        if not long_docs:
            print(f"   - 无需二次分块")
            return documents

        print(f"   - 需要二次分块: {len(long_docs)} 个块")

        # 二次分块
        all_splits = []
        for doc in documents:
            if len(doc.page_content) > self.chunk_size:
                # 分块并保留元数据
                splits = self.text_splitter.split_documents([doc])
                all_splits.extend(splits)
            else:
                all_splits.append(doc)

        print(f"   ✓ 二次分块完成: {len(all_splits)} 个块")

        return all_splits

    def batch_extract(
        self,
        urls: List[str],
        enable_secondary_split: Optional[bool] = None
    ) -> List[Document]:
        """
        批量提取网页内容

        参数：
        - urls: URL列表
        - enable_secondary_split: 是否启用二次分块（覆盖默认配置）

        返回：
        - Document对象列表
        """
        print(f"\n批量提取: {len(urls)} 个URL")

        all_documents = []

        for i, url in enumerate(urls, 1):
            print(f"\n[{i}/{len(urls)}] 处理: {url}")

            try:
                # 从URL分割
                documents = self.split_from_url(url)

                # 二次分块（可选）
                if enable_secondary_split is None:
                    enable_secondary_split = self.enable_secondary_split

                if enable_secondary_split:
                    documents = self.secondary_split(documents)

                all_documents.extend(documents)

            except Exception as e:
                print(f"   ✗ 处理失败: {e}")
                continue

        print(f"\n批量提取完成: {len(all_documents)} 个块")

        return all_documents

    def create_vectorstore(
        self,
        documents: List[Document],
        collection_name: str = "web_content"
    ) -> Chroma:
        """
        创建向量存储

        参数：
        - documents: Document对象列表
        - collection_name: 集合名称

        返回：
        - Chroma向量存储对象
        """
        print(f"\n创建向量存储: {collection_name}")

        self.vectorstore = Chroma.from_documents(
            documents=documents,
            embedding=self.embeddings,
            collection_name=collection_name,
            persist_directory=self.persist_directory
        )

        print(f"   ✓ 向量存储创建完成")
        print(f"   - 文档数量: {len(documents)}")

        return self.vectorstore

    def load_vectorstore(self, collection_name: str = "web_content") -> Chroma:
        """加载已存在的向量存储"""
        print(f"\n加载向量存储: {collection_name}")

        self.vectorstore = Chroma(
            collection_name=collection_name,
            embedding_function=self.embeddings,
            persist_directory=self.persist_directory
        )

        return self.vectorstore

    def search(
        self,
        query: str,
        k: int = 5,
        filter_by_header: Optional[str] = None
    ) -> List[Dict[str, Any]]:
        """
        搜索网页内容

        参数：
        - query: 查询文本
        - k: 返回结果数量
        - filter_by_header: 按标题过滤（如"Header 1"）

        返回：
        - 搜索结果列表
        """
        if self.vectorstore is None:
            raise ValueError("向量存储未初始化")

        print(f"\n搜索网页内容: {query}")

        # 构建过滤条件
        filter_dict = None
        if filter_by_header:
            filter_dict = {filter_by_header: {"$exists": True}}

        # 相似度搜索
        results = self.vectorstore.similarity_search_with_score(
            query,
            k=k,
            filter=filter_dict
        )

        # 格式化结果
        formatted_results = []
        for i, (doc, score) in enumerate(results, 1):
            formatted_results.append({
                "rank": i,
                "content": doc.page_content,
                "score": score,
                "metadata": doc.metadata
            })

            print(f"\n   [{i}] 相似度: {score:.4f}")
            print(f"       来源: {doc.metadata.get('source', 'unknown')}")

            # 打印标题层级
            headers = {k: v for k, v in doc.metadata.items() if k.startswith("Header")}
            if headers:
                print(f"       标题: {headers}")

            print(f"       内容预览: {doc.page_content[:100]}...")

        return formatted_results
