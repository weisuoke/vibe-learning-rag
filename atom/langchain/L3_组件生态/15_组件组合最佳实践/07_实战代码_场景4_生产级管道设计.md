# 实战代码 场景4：生产级管道设计

## 场景描述

构建一个可配置、可观测、容错的生产级 LangChain 管道，综合运用所有最佳实践。

## 完整可运行代码

```python
"""
生产级管道设计实战
演示：可配置组件 + 错误处理 + 性能优化 + 可观测性 的完整管道
"""

import os
import time
import logging
from typing import Any
from dotenv import load_dotenv
load_dotenv()

from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import (
    RunnablePassthrough,
    RunnableParallel,
    RunnableLambda,
    RunnableBranch,
    ConfigurableField,
)
from langchain_core.callbacks import BaseCallbackHandler
from langchain_core.documents import Document
from langchain_chroma import Chroma

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# ===== 1. 可观测性：自定义回调 =====
print("=== 1. 构建可观测性组件 ===")

class ProductionCallback(BaseCallbackHandler):
    """生产级回调：记录延迟、错误、Token 用量"""

    def __init__(self):
        self.metrics = {
            "total_calls": 0,
            "errors": 0,
            "total_tokens": 0,
            "latencies": [],
        }
        self._start_times = {}

    def on_llm_start(self, serialized, prompts, *, run_id, **kwargs):
        self._start_times[run_id] = time.time()
        self.metrics["total_calls"] += 1

    def on_llm_end(self, response, *, run_id, **kwargs):
        if run_id in self._start_times:
            latency = time.time() - self._start_times[run_id]
            self.metrics["latencies"].append(latency)
        # 记录 token 用量
        if response.llm_output and "token_usage" in response.llm_output:
            usage = response.llm_output["token_usage"]
            self.metrics["total_tokens"] += usage.get("total_tokens", 0)

    def on_llm_error(self, error, *, run_id, **kwargs):
        self.metrics["errors"] += 1
        logger.error(f"LLM 错误: {error}")

    def report(self):
        latencies = self.metrics["latencies"]
        avg_latency = sum(latencies) / len(latencies) if latencies else 0
        print(f"\n📊 性能报告:")
        print(f"  总调用数: {self.metrics['total_calls']}")
        print(f"  错误数: {self.metrics['errors']}")
        print(f"  总 Token: {self.metrics['total_tokens']}")
        print(f"  平均延迟: {avg_latency:.2f}s")
        if latencies:
            print(f"  最大延迟: {max(latencies):.2f}s")
            print(f"  最小延迟: {min(latencies):.2f}s")

# ===== 2. 可配置的模型 =====
print("=== 2. 构建可配置组件 ===")

# 可运行时切换的模型
configurable_model = ChatOpenAI(
    model="gpt-4o-mini",
    temperature=0,
).configurable_fields(
    model_name=ConfigurableField(
        id="model_name",
        name="模型名称",
        description="运行时选择使用的模型",
    ),
    temperature=ConfigurableField(
        id="temperature",
        name="温度",
        description="控制输出的随机性",
    ),
)

# ===== 3. 构建 RAG 管道 =====
print("=== 3. 构建生产级 RAG 管道 ===")

# 准备数据
documents = [
    Document(page_content="LangChain 是一个用于构建 LLM 应用的框架。", metadata={"source": "intro"}),
    Document(page_content="LCEL 是 LangChain 的表达式语言，用于组合组件。", metadata={"source": "lcel"}),
    Document(page_content="RunnableSequence 通过管道操作符连接组件。", metadata={"source": "runnable"}),
]

embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
vectorstore = Chroma.from_documents(documents, embeddings, collection_name="prod_demo")
retriever = vectorstore.as_retriever(search_kwargs={"k": 2})

def format_docs(docs):
    return "\n".join(f"- {doc.page_content}" for doc in docs)

# RAG Prompt
rag_prompt = ChatPromptTemplate.from_messages([
    ("system",
     "你是一个专业助手。根据参考资料回答问题。\n"
     "如果不确定，请说明。\n\n"
     "参考资料：\n{context}"),
    ("human", "{question}"),
])

# 带错误处理的模型
robust_model = (
    configurable_model
    .with_retry(
        stop_after_attempt=3,
        wait_exponential_jitter=True,
    )
    .with_fallbacks([
        ChatOpenAI(model="gpt-4o-mini", temperature=0),
    ])
)

# 完整的生产级 RAG 管道
production_rag_chain = (
    RunnableParallel(
        question=RunnablePassthrough(),
        context=retriever | format_docs,
    )
    | rag_prompt
    | robust_model
    | StrOutputParser()
)

print("管道结构：")
production_rag_chain.get_graph().print_ascii()

# ===== 4. 运行管道 =====
print("\n=== 4. 运行管道 ===")

# 创建回调
metrics = ProductionCallback()

# 使用默认配置
result = production_rag_chain.invoke(
    "什么是 LCEL？",
    config={
        "callbacks": [metrics],
        "run_name": "prod_rag_v1",
        "tags": ["production", "rag"],
        "metadata": {"user_id": "user_001"},
    },
)
print(f"回答: {result}")

# 运行时切换模型
result = production_rag_chain.invoke(
    "LangChain 是什么？",
    config={
        "callbacks": [metrics],
        "configurable": {"model_name": "gpt-4o"},
    },
)
print(f"回答 (gpt-4o): {result}")

# ===== 5. 批量处理 =====
print("\n=== 5. 批量处理 ===")

questions = ["什么是LCEL？", "RunnableSequence是什么？", "LangChain有什么用？"]
results = production_rag_chain.batch(
    questions,
    config={
        "callbacks": [metrics],
        "max_concurrency": 3,
    },
    return_exceptions=True,
)

for q, r in zip(questions, results):
    if isinstance(r, Exception):
        print(f"❌ {q}: {r}")
    else:
        print(f"✅ {q}: {r[:60]}...")

# ===== 6. 流式输出 =====
print("\n=== 6. 流式输出 ===")
print("问题: LCEL 怎么用？")
print("回答: ", end="")
for chunk in production_rag_chain.stream(
    "LCEL 怎么用？",
    config={"callbacks": [metrics]},
):
    print(chunk, end="", flush=True)
print()

# ===== 7. 性能报告 =====
metrics.report()

# 清理
vectorstore.delete_collection()
print("\n✅ 生产级管道演示完成")
```

## 生产级管道设计清单

| 维度 | 实现 | 状态 |
|------|------|------|
| 错误处理 | Retry + Fallback | ✅ |
| 性能优化 | 流式 + 批处理 + 并行 | ✅ |
| 可配置性 | ConfigurableField | ✅ |
| 可观测性 | 自定义 Callback + LangSmith | ✅ |
| 成本控制 | 模型分级 + 缓存 | ✅ |
| 安全性 | 输入验证 + 输出过滤 | ✅ |

## 架构图

```
用户请求
    ↓
[输入验证] → 无效 → 返回错误
    ↓ 有效
[RunnableParallel]
├── question: RunnablePassthrough
└── context: Retriever → format_docs
    ↓
[RAG Prompt]
    ↓
[Model (可配置)]
├── with_retry (3次)
└── with_fallbacks (备选模型)
    ↓
[StrOutputParser]
    ↓
[Callbacks: 延迟/错误/Token 监控]
    ↓
返回结果
```
