## 实战示例1：消息数量限制

### 完整代码

```python
"""
消息数量限制
演示如何限制对话历史的消息数量
"""

from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from dotenv import load_dotenv

load_dotenv()

# ============================================================
# 1. 带消息数量限制的 Session History
# ============================================================

store = {}

def get_session_history_with_limit(session_id: str, max_messages: int = 10):
    """
    获取带消息数量限制的对话历史

    参数:
        session_id: 会话ID
        max_messages: 最大消息数量（默认10条）

    返回:
        InMemoryChatMessageHistory 实例
    """
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()

    history = store[session_id]

    # 限制消息数量（保留最近的 N 条）
    if len(history.messages) > max_messages:
        history.messages = history.messages[-max_messages:]

    return history

# ============================================================
# 2. 创建对话链
# ============================================================

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

conversation = RunnableWithMessageHistory(
    llm,
    lambda session_id: get_session_history_with_limit(session_id, max_messages=6),
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 3. 测试消息数量限制
# ============================================================

def test_message_limit():
    """测试消息数量限制"""

    session_id = "limit_test_001"
    config = {"configurable": {"session_id": session_id}}

    print("=== 消息数量限制测试 ===\n")

    # 发送10条消息
    for i in range(10):
        print(f"👤 用户: 消息 {i+1}")
        response = conversation.invoke(
            {"input": f"这是第 {i+1} 条消息"},
            config=config
        )
        print(f"🤖 AI: {response.content}\n")

        # 检查消息数量
        history = store[session_id]
        print(f"📊 当前消息数量: {len(history.messages)}\n")

if __name__ == "__main__":
    test_message_limit()
```

## 实战示例2：Token 数量限制

### 完整代码

```python
"""
Token 数量限制
演示如何基于 Token 数量限制对话历史
"""

from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.messages.utils import count_tokens_approximately
from dotenv import load_dotenv

load_dotenv()

# ============================================================
# 1. 带 Token 限制的 Session History
# ============================================================

store = {}

def get_session_history_with_token_limit(
    session_id: str,
    max_tokens: int = 1000
):
    """
    获取带 Token 限制的对话历史

    参数:
        session_id: 会话ID
        max_tokens: 最大 Token 数量

    返回:
        InMemoryChatMessageHistory 实例
    """
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()

    history = store[session_id]
    messages = history.messages

    # 计算总 Token 数
    total_tokens = count_tokens_approximately(messages)

    # 如果超过限制，删除最旧的消息
    while total_tokens > max_tokens and len(messages) > 2:
        messages.pop(0)  # 删除最旧的消息
        total_tokens = count_tokens_approximately(messages)

    history.messages = messages
    return history

# ============================================================
# 2. 创建对话链
# ============================================================

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

conversation = RunnableWithMessageHistory(
    llm,
    lambda session_id: get_session_history_with_token_limit(session_id, max_tokens=500),
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 3. 测试 Token 限制
# ============================================================

def test_token_limit():
    """测试 Token 限制"""

    session_id = "token_limit_001"
    config = {"configurable": {"session_id": session_id}}

    print("=== Token 限制测试 ===\n")

    # 发送长消息
    long_messages = [
        "请详细介绍一下 Python 的历史和发展",
        "请解释一下面向对象编程的核心概念",
        "请说明一下函数式编程的特点",
        "请描述一下异步编程的原理"
    ]

    for i, msg in enumerate(long_messages, 1):
        print(f"👤 用户: {msg}")
        response = conversation.invoke({"input": msg}, config=config)
        print(f"🤖 AI: {response.content[:100]}...\n")

        # 检查 Token 数量
        history = store[session_id]
        token_count = count_tokens_approximately(history.messages)
        print(f"📊 当前 Token 数量: {token_count}")
        print(f"📊 消息数量: {len(history.messages)}\n")

if __name__ == "__main__":
    test_token_limit()
```

## 实战示例3：滑动窗口策略

### 完整代码

```python
"""
滑动窗口策略
演示如何实现滑动窗口内存管理
"""

from langchain_openai import ChatOpenAI
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.messages import SystemMessage
from dotenv import load_dotenv

load_dotenv()

# ============================================================
# 1. 滑动窗口 Session History
# ============================================================

store = {}

def get_session_history_with_sliding_window(
    session_id: str,
    window_size: int = 5,
    keep_system_message: bool = True
):
    """
    获取带滑动窗口的对话历史

    参数:
        session_id: 会话ID
        window_size: 窗口大小（保留最近的 N 轮对话）
        keep_system_message: 是否保留系统消息

    返回:
        InMemoryChatMessageHistory 实例
    """
    if session_id not in store:
        history = InMemoryChatMessageHistory()
        # 添加系统消息
        if keep_system_message:
            history.add_message(SystemMessage(
                content="你是一个友好的AI助手，专注于回答用户问题。"
            ))
        store[session_id] = history

    history = store[session_id]
    messages = history.messages

    # 分离系统消息和对话消息
    system_messages = [msg for msg in messages if isinstance(msg, SystemMessage)]
    conversation_messages = [msg for msg in messages if not isinstance(msg, SystemMessage)]

    # 应用滑动窗口（保留最近的 N 轮对话 = 2N 条消息）
    max_conversation_messages = window_size * 2
    if len(conversation_messages) > max_conversation_messages:
        conversation_messages = conversation_messages[-max_conversation_messages:]

    # 重新组合消息
    if keep_system_message:
        history.messages = system_messages + conversation_messages
    else:
        history.messages = conversation_messages

    return history

# ============================================================
# 2. 创建对话链
# ============================================================

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

conversation = RunnableWithMessageHistory(
    llm,
    lambda session_id: get_session_history_with_sliding_window(
        session_id,
        window_size=3,  # 保留最近3轮对话
        keep_system_message=True
    ),
    input_messages_key="input",
    history_messages_key="history"
)

# ============================================================
# 3. 测试滑动窗口
# ============================================================

def test_sliding_window():
    """测试滑动窗口"""

    session_id = "sliding_window_001"
    config = {"configurable": {"session_id": session_id}}

    print("=== 滑动窗口测试 ===\n")

    # 发送多轮对话
    conversations = [
        ("我叫Alice", "你好Alice！"),
        ("我是一名工程师", "很高兴认识你！"),
        ("我喜欢Python", "Python是很棒的语言！"),
        ("我在学习AI", "AI是很有前景的领域！"),
        ("我的名字是什么？", "测试记忆")
    ]

    for i, (user_msg, _) in enumerate(conversations, 1):
        print(f"👤 用户: {user_msg}")
        response = conversation.invoke({"input": user_msg}, config=config)
        print(f"🤖 AI: {response.content}\n")

        # 检查窗口状态
        history = store[session_id]
        conversation_count = (len(history.messages) - 1) // 2  # 减去系统消息
        print(f"📊 当前对话轮数: {conversation_count}")
        print(f"📊 总消息数: {len(history.messages)}\n")

if __name__ == "__main__":
    test_sliding_window()
```

## 生产环境最佳实践

### 1. 混合策略

```python
def get_session_history_hybrid(
    session_id: str,
    max_messages: int = 20,
    max_tokens: int = 2000
):
    """
    混合策略：同时限制消息数量和 Token 数量

    优先使用 Token 限制，消息数量作为兜底
    """
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()

    history = store[session_id]
    messages = history.messages

    # 1. Token 限制
    total_tokens = count_tokens_approximately(messages)
    while total_tokens > max_tokens and len(messages) > 2:
        messages.pop(0)
        total_tokens = count_tokens_approximately(messages)

    # 2. 消息数量限制（兜底）
    if len(messages) > max_messages:
        messages = messages[-max_messages:]

    history.messages = messages
    return history
```

### 2. 动态调整

```python
def get_session_history_dynamic(session_id: str):
    """
    动态调整策略：根据消息长度动态调整窗口大小
    """
    history = store.get(session_id, InMemoryChatMessageHistory())
    messages = history.messages

    # 计算平均消息长度
    if messages:
        avg_length = sum(len(msg.content) for msg in messages) / len(messages)

        # 根据平均长度调整窗口
        if avg_length > 500:
            max_messages = 10  # 长消息，小窗口
        elif avg_length > 200:
            max_messages = 20  # 中等消息
        else:
            max_messages = 30  # 短消息，大窗口

        if len(messages) > max_messages:
            messages = messages[-max_messages:]

    history.messages = messages
    return history
```

### 3. 总结策略

```python
def get_session_history_with_summary(session_id: str):
    """
    总结策略：当消息过多时，总结旧消息
    """
    history = store.get(session_id, InMemoryChatMessageHistory())
    messages = history.messages

    # 如果消息超过20条，总结前10条
    if len(messages) > 20:
        old_messages = messages[:10]
        recent_messages = messages[10:]

        # 生成总结（简化示例）
        summary = f"之前的对话总结：共 {len(old_messages)} 条消息"

        # 用总结替换旧消息
        history.messages = [
            SystemMessage(content=summary)
        ] + recent_messages

    return history
```

## 常见问题

### Q1: 如何选择限制策略？

**A**: 根据场景选择：
- **短对话**：消息数量限制（10-20条）
- **长对话**：Token 限制（1000-2000 tokens）
- **生产环境**：混合策略

### Q2: 如何避免丢失重要信息？

**A**: 使用总结策略：
```python
# 总结旧消息而非直接删除
summary = summarize_old_messages(old_messages)
history.messages = [SystemMessage(content=summary)] + recent_messages
```

### Q3: 如何处理系统消息？

**A**: 始终保留系统消息：
```python
system_messages = [msg for msg in messages if isinstance(msg, SystemMessage)]
conversation_messages = [msg for msg in messages if not isinstance(msg, SystemMessage)]

# 应用限制到对话消息
limited_conversation = apply_limit(conversation_messages)

# 重新组合
history.messages = system_messages + limited_conversation
```

## 总结

本场景演示了三种内存限制策略：

1. **消息数量限制**：保留最近 N 条消息
2. **Token 数量限制**：基于 Token 计数动态删除
3. **滑动窗口策略**：保留最近 N 轮对话

**选择建议**：
- 简单场景 → 消息数量限制
- 长文本场景 → Token 限制
- 生产环境 → 混合策略
