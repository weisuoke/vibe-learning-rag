```python
"""
场景3: 批量数据操作
演示批量插入、BulkInsert、批量搜索的性能优化
"""

import os
import time
import json
import pandas as pd
from dotenv import load_dotenv
from pymilvus import MilvusClient, DataType, Function, FunctionType, utility

load_dotenv()

MILVUS_URI = "http://localhost:19530"
COLLECTION_NAME = "batch_collection"
OPENAI_API_KEY = os.getenv("OPENAI_API_KEY")

milvus_client = MilvusClient(uri=MILVUS_URI)
```

### 2. 创建测试 Collection

```python
def create_collection():
    """创建用于批量操作测试的 Collection"""

    if milvus_client.has_collection(COLLECTION_NAME):
        milvus_client.drop_collection(COLLECTION_NAME)

    schema = milvus_client.create_schema()

    schema.add_field("id", DataType.INT64, is_primary=True, auto_id=True)
    schema.add_field("document", DataType.VARCHAR, max_length=5000)
    schema.add_field("category", DataType.VARCHAR, max_length=100)
    schema.add_field("dense", DataType.FLOAT_VECTOR, dim=1536)

    # 配置 Embedding Function
    schema.add_function(
        Function(
            name="openai_embedding",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["document"],
            output_field_names=["dense"],
            params={
                "provider": "openai",
                "model_name": "text-embedding-3-small",
                "api_key": OPENAI_API_KEY
            }
        )
    )

    index_params = milvus_client.prepare_index_params()
    index_params.add_index(
        field_name="dense",
        index_type="AUTOINDEX",
        metric_type="COSINE"
    )

    milvus_client.create_collection(
        collection_name=COLLECTION_NAME,
        schema=schema,
        index_params=index_params
    )

    print(f"✅ Collection 创建成功: {COLLECTION_NAME}")
```

### 3. 生成测试数据

```python
def generate_test_data(num_docs=1000):
    """
    生成测试文档数据

    参数:
        num_docs: 文档数量

    返回:
        文档列表
    """
    categories = ["tech", "business", "science", "health", "education"]

    documents = []
    for i in range(num_docs):
        category = categories[i % len(categories)]
        doc = {
            "document": f"This is document {i} about {category}. "
                       f"It contains information related to {category} topics. "
                       f"Document ID: {i}",
            "category": category
        }
        documents.append(doc)

    print(f"✅ 生成 {num_docs} 条测试文档")
    return documents
```

### 4. 批量插入性能测试

```python
def test_batch_insert(documents, batch_size=100):
    """
    测试批量插入性能

    参数:
        documents: 文档列表
        batch_size: 批次大小

    返回:
        插入耗时
    """
    print(f"\n【批量插入测试】批次大小: {batch_size}")

    start_time = time.time()

    # 分批插入
    total_inserted = 0
    for i in range(0, len(documents), batch_size):
        batch = documents[i:i + batch_size]
        result = milvus_client.insert(COLLECTION_NAME, batch)
        total_inserted += result['insert_count']

        if (i // batch_size + 1) % 10 == 0:
            print(f"  已插入: {total_inserted}/{len(documents)}")

    elapsed = time.time() - start_time

    print(f"✅ 插入完成")
    print(f"   - 总数: {total_inserted} 条")
    print(f"   - 耗时: {elapsed:.2f}s")
    print(f"   - 速率: {total_inserted/elapsed:.2f} 条/秒")

    return elapsed
```

```python
def test_single_insert(documents, limit=100):
    """
    测试逐条插入性能（仅测试前 limit 条）

    参数:
        documents: 文档列表
        limit: 测试数量限制

    返回:
        插入耗时
    """
    print(f"\n【逐条插入测试】测试数量: {limit}")

    # 创建临时 Collection
    temp_collection = "temp_single_collection"
    if milvus_client.has_collection(temp_collection):
        milvus_client.drop_collection(temp_collection)

    # 复制 schema（简化版）
    schema = milvus_client.create_schema()
    schema.add_field("id", DataType.INT64, is_primary=True, auto_id=True)
    schema.add_field("document", DataType.VARCHAR, max_length=5000)
    schema.add_field("dense", DataType.FLOAT_VECTOR, dim=1536)

    schema.add_function(
        Function(
            name="openai_embedding",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["document"],
            output_field_names=["dense"],
            params={
                "provider": "openai",
                "model_name": "text-embedding-3-small",
                "api_key": OPENAI_API_KEY
            }
        )
    )

    index_params = milvus_client.prepare_index_params()
    index_params.add_index(field_name="dense", index_type="AUTOINDEX", metric_type="COSINE")

    milvus_client.create_collection(temp_collection, schema=schema, index_params=index_params)

    start_time = time.time()

    # 逐条插入
    for i, doc in enumerate(documents[:limit]):
        milvus_client.insert(temp_collection, [doc])
        if (i + 1) % 10 == 0:
            print(f"  已插入: {i + 1}/{limit}")

    elapsed = time.time() - start_time

    print(f"✅ 插入完成")
    print(f"   - 总数: {limit} 条")
    print(f"   - 耗时: {elapsed:.2f}s")
    print(f"   - 速率: {limit/elapsed:.2f} 条/秒")

    # 清理
    milvus_client.drop_collection(temp_collection)

    return elapsed
```

### 6. BulkInsert 文件导入

```python
def test_bulk_insert(documents, file_path="bulk_data.json"):
    """
    测试 BulkInsert 文件导入

    参数:
        documents: 文档列表
        file_path: 导出文件路径

    返回:
        导入耗时
    """
    print(f"\n【BulkInsert 测试】")

    # 步骤 1: 准备数据文件
    print("  步骤 1: 准备数据文件...")
    with open(file_path, 'w') as f:
        for doc in documents:
            json.dump(doc, f)
            f.write('\n')
    print(f"  ✅ 数据文件已生成: {file_path}")

    # 步骤 2: 执行 BulkInsert
    print("  步骤 2: 执行 BulkInsert...")
    start_time = time.time()

    task_id = utility.do_bulk_insert(
        collection_name=COLLECTION_NAME,
        files=[file_path]
    )

    print(f"  任务 ID: {task_id}")

    # 步骤 3: 等待完成
    print("  步骤 3: 等待导入完成...")
    while True:
        state = utility.get_bulk_insert_state(task_id)
        if state.state_name in ["Completed", "Failed"]:
            break
        time.sleep(1)

    elapsed = time.time() - start_time

    if state.state_name == "Completed":
        print(f"✅ BulkInsert 完成")
        print(f"   - 耗时: {elapsed:.2f}s")
        print(f"   - 速率: {len(documents)/elapsed:.2f} 条/秒")
    else:
        print(f"❌ BulkInsert 失败: {state.state_name}")

    # 清理文件
    os.remove(file_path)

    return elapsed
```

```python
def performance_comparison():

    # 逐条插入（仅测试 100 条）

    print(f"批量搜索: {batch_search_time/10:.3f}s/查询")
    print(f"逐个搜索: {single_search_time/10:.3f}s/查询")
    print(f"性能提升: {(single_search_time/batch_search_time):.1f}x")

    print("\n💡 最佳实践:")
    print("- 插入: 使用批量插入，批次大小 500-1000")
    print("- 搜索: 使用批量搜索，一次提交多个查询")
```

## 批次大小优化

### 推荐批次大小

| 数据量 | 推荐批次 | 原因 |
|--------|----------|------|
| < 1万 | 100-500 | 平衡性能和内存 |
| 1万-10万 | 500-1000 | 最佳性能 |
| 10万-100万 | 1000-5000 | 充分利用带宽 |
| > 100万 | BulkInsert | 文件导入最快 |

## 常见问题

### Q1: 批量插入失败怎么办？

```python
def safe_batch_insert(documents, batch_size=500):
    """带错误处理的批量插入"""
    failed_batches = []

    for i in range(0, len(documents), batch_size):
        batch = documents[i:i + batch_size]
        try:
            milvus_client.insert(COLLECTION_NAME, batch)
        except Exception as e:
            print(f"批次 {i//batch_size} 失败: {e}")
            failed_batches.append((i, batch))

    # 重试失败的批次
    for i, batch in failed_batches:
        try:
            milvus_client.insert(COLLECTION_NAME, batch)
            print(f"批次 {i//batch_size} 重试成功")
        except Exception as e:
            print(f"批次 {i//batch_size} 重试失败: {e}")
```

### Q2: 如何监控插入进度？

```python
from tqdm import tqdm

def insert_with_progress(documents, batch_size=500):
    """带进度条的批量插入"""
    total_batches = (len(documents) + batch_size - 1) // batch_size

    with tqdm(total=len(documents), desc="插入进度") as pbar:
        for i in range(0, len(documents), batch_size):
            batch = documents[i:i + batch_size]
            result = milvus_client.insert(COLLECTION_NAME, batch)
            pbar.update(result['insert_count'])
```

### Q3: 内存不足怎么办？

```python
def memory_efficient_insert(documents, batch_size=500):
    """内存高效的批量插入"""
    # 使用生成器避免一次性加载所有数据
    def batch_generator(docs, size):
        for i in range(0, len(docs), size):
            yield docs[i:i + size]

    for batch in batch_generator(documents, batch_size):
        milvus_client.insert(COLLECTION_NAME, batch)
        # 可选: 添加延迟避免过载
        # time.sleep(0.1)
```

## 性能优化技巧

### 1. 并行插入（多线程）

```python
from concurrent.futures import ThreadPoolExecutor

def parallel_insert(documents, batch_size=500, workers=4):
    """并行批量插入"""
    batches = [documents[i:i + batch_size]
               for i in range(0, len(documents), batch_size)]

    with ThreadPoolExecutor(max_workers=workers) as executor:
        futures = [executor.submit(milvus_client.insert, COLLECTION_NAME, batch)
                   for batch in batches]

        for future in futures:
            future.result()
```
