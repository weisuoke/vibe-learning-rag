## 场景描述

生产环境的 RAG 系统需要更高级的容错能力：
- **断路器模式**：防止级联故障，快速失败
- **检查点恢复**：从中断点继续执行，不重复工作
- **完整监控**：记录所有重试、失败和恢复事件
- **降级策略**：多层级 Fallback 链

这是一个综合性的生产级错误恢复系统。

## 完整代码

```python
"""
场景4：生产级错误恢复系统
演示：断路器 + 检查点恢复 + 完整监控
"""

import os
import time
from typing import TypedDict, List, Dict, Optional
from datetime import datetime
from enum import Enum
from openai import OpenAI, RateLimitError, APITimeoutError

from langgraph.graph import StateGraph, START, END
from langgraph.types import RetryPolicy
from langgraph.checkpoint.memory import MemorySaver

# ===== 1. 定义状态 =====
class State(TypedDict):
    query: str
    documents: List[str]
    answer: str
    # 监控字段
    retry_count: int
    fallback_used: str
    execution_time: float
    errors: List[Dict[str, str]]

# ===== 2. 断路器实现 =====
class CircuitState(Enum):
    CLOSED = "closed"      # 正常运行
    OPEN = "open"          # 断路，拒绝请求
    HALF_OPEN = "half_open"  # 半开，测试恢复

class CircuitBreaker:
    """
    断路器模式实现

    状态转换：
    - CLOSED → OPEN: 失败次数超过阈值
    - OPEN → HALF_OPEN: 超时后允许测试
    - HALF_OPEN → CLOSED: 测试成功
    - HALF_OPEN → OPEN: 测试失败
    """
    def __init__(
        self,
        name: str,
        failure_threshold: int = 5,      # 触发断路的失败次数
        timeout: float = 60.0,           # 断路器打开的时长（秒）
        success_threshold: int = 2       # 恢复需要的成功次数
    ):
        self.name = name
        self.failure_threshold = failure_threshold
        self.timeout = timeout
        self.success_threshold = success_threshold

        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.success_count = 0
        self.last_failure_time = None

    def call(self, func, *args, **kwargs):
        """通过断路器调用函数"""
        # 检查是否应该从 OPEN 转到 HALF_OPEN
        if self.state == CircuitState.OPEN:
            if time.time() - self.last_failure_time >= self.timeout:
                print(f"[断路器:{self.name}] 超时结束，进入半开状态")
                self.state = CircuitState.HALF_OPEN
                self.success_count = 0
            else:
                raise Exception(f"断路器 {self.name} 处于打开状态，拒绝请求")

        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result
        except Exception as e:
            self._on_failure()
            raise

    def _on_success(self):
        """成功回调"""
        if self.state == CircuitState.HALF_OPEN:
            self.success_count += 1
            print(f"[断路器:{self.name}] 半开状态成功 ({self.success_count}/{self.success_threshold})")
            if self.success_count >= self.success_threshold:
                print(f"[断路器:{self.name}] 恢复到关闭状态")
                self.state = CircuitState.CLOSED
                self.failure_count = 0
        elif self.state == CircuitState.CLOSED:
            self.failure_count = 0  # 重置失败计数

    def _on_failure(self):
        """失败回调"""
        self.failure_count += 1
        self.last_failure_time = time.time()

        if self.state == CircuitState.HALF_OPEN:
            print(f"[断路器:{self.name}] 半开状态失败，重新打开")
            self.state = CircuitState.OPEN
        elif self.state == CircuitState.CLOSED:
            print(f"[断路器:{self.name}] 失败次数: {self.failure_count}/{self.failure_threshold}")
            if self.failure_count >= self.failure_threshold:
                print(f"[断路器:{self.name}] 失败次数超过阈值，打开断路器")
                self.state = CircuitState.OPEN

    def get_status(self) -> Dict:
        """获取断路器状态"""
        return {
            "name": self.name,
            "state": self.state.value,
            "failure_count": self.failure_count,
            "success_count": self.success_count
        }

# 创建断路器实例
llm_breaker = CircuitBreaker("LLM主模型", failure_threshold=3, timeout=30.0)
retrieval_breaker = CircuitBreaker("向量检索", failure_threshold=5, timeout=60.0)

# ===== 3. 监控系统 =====
class Monitor:
    """监控和日志系统"""
    def __init__(self):
        self.events = []

    def log_event(self, event_type: str, node: str, details: Dict):
        """记录事件"""
        event = {
            "timestamp": datetime.now().isoformat(),
            "type": event_type,
            "node": node,
            "details": details
        }
        self.events.append(event)
        print(f"[监控] {event_type} | {node} | {details}")

    def get_summary(self) -> Dict:
        """获取监控摘要"""
        retry_events = [e for e in self.events if e["type"] == "retry"]
        fallback_events = [e for e in self.events if e["type"] == "fallback"]
        error_events = [e for e in self.events if e["type"] == "error"]

        return {
            "total_events": len(self.events),
            "retry_count": len(retry_events),
            "fallback_count": len(fallback_events),
            "error_count": len(error_events),
            "circuit_breaker_status": {
                "llm": llm_breaker.get_status(),
                "retrieval": retrieval_breaker.get_status()
            }
        }

monitor = Monitor()

# ===== 4. 检索节点（带断路器） =====
class MockVectorDB:
    """模拟向量数据库"""
    def __init__(self, name: str, fail_rate: float = 0.0):
        self.name = name
        self.fail_rate = fail_rate

    def search(self, query: str) -> List[str]:
        import random
        if random.random() < self.fail_rate:
            raise ConnectionError(f"{self.name} 连接失败")

        return [
            f"{self.name} 文档1：LangGraph 是工作流框架",
            f"{self.name} 文档2：支持状态管理和重试",
            f"{self.name} 文档3：适合构建 Agent 系统"
        ]

primary_vector_db = MockVectorDB("Milvus主库", fail_rate=0.4)
backup_vector_db = MockVectorDB("ChromaDB备库", fail_rate=0.1)

def retrieve_with_circuit_breaker(state: State) -> dict:
    """带断路器的检索"""
    print(f"\n[检索] 开始检索...")
    start_time = time.time()

    try:
        # 通过断路器调用主库
        docs = retrieval_breaker.call(primary_vector_db.search, state['query'])
        monitor.log_event("success", "retrieve", {"source": "primary", "docs": len(docs)})
        return {
            "documents": docs,
            "fallback_used": "none",
            "execution_time": time.time() - start_time
        }

    except Exception as e:
        monitor.log_event("error", "retrieve", {"error": str(e), "source": "primary"})

        # 主库失败，尝试备库
        try:
            print(f"[检索] 主库失败，尝试备库...")
            docs = backup_vector_db.search(state['query'])
            monitor.log_event("fallback", "retrieve", {"source": "backup", "docs": len(docs)})
            return {
                "documents": docs,
                "fallback_used": "backup_db",
                "execution_time": time.time() - start_time
            }
        except Exception as e2:
            monitor.log_event("error", "retrieve", {"error": str(e2), "source": "backup"})
            # 返回默认文档
            return {
                "documents": ["默认文档：LangGraph 是一个工作流框架"],
                "fallback_used": "default",
                "execution_time": time.time() - start_time
            }

# 检索重试策略
retrieval_retry = RetryPolicy(
    max_attempts=3,
    initial_interval=0.5,
    retry_on=ConnectionError
)

# ===== 5. LLM 节点（带断路器和多模型 Fallback） =====
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

def generate_with_circuit_breaker(state: State) -> dict:
    """带断路器和多模型 Fallback 的生成"""
    print(f"\n[生成] 开始生成答案...")
    start_time = time.time()

    context = "\n".join([f"- {doc}" for doc in state['documents']])

    # 模型 Fallback 链
    models = [
        ("gpt-4o-mini", "primary"),
        ("gpt-3.5-turbo", "fallback_1"),
    ]

    for model, fallback_level in models:
        try:
            print(f"[生成] 尝试模型: {model}")

            def call_llm():
                response = client.chat.completions.create(
                    model=model,
                    messages=[
                        {"role": "system", "content": "你是一个有帮助的助手。"},
                        {"role": "user", "content": f"基于以下文档回答问题：\n\n{context}\n\n问题：{state['query']}"}
                    ],
                    temperature=0.7,
                    max_tokens=300,
                    timeout=30.0
                )
                return response.choices[0].message.content

            # 通过断路器调用
            answer = llm_breaker.call(call_llm)

            monitor.log_event("success", "generate", {"model": model, "fallback": fallback_level})
            return {
                "answer": answer,
                "fallback_used": fallback_level,
                "execution_time": time.time() - start_time
            }

        except Exception as e:
            monitor.log_event("error", "generate", {"model": model, "error": str(e)})
            print(f"[生成] 模型 {model} 失败: {e}")

            if fallback_level == models[-1][1]:  # 最后一个模型
                # 返回基于检索的默认答案
                default_answer = f"基于检索到的文档，{state['documents'][0]}"
                monitor.log_event("fallback", "generate", {"fallback": "default_answer"})
                return {
                    "answer": default_answer,
                    "fallback_used": "default_answer",
                    "execution_time": time.time() - start_time
                }

# LLM 重试策略
llm_retry = RetryPolicy(
    max_attempts=3,
    initial_interval=1.0,
    retry_on=(RateLimitError, APITimeoutError, ConnectionError)
)

# ===== 6. 构建工作流（带检查点） =====
# 使用内存检查点（生产环境应使用持久化存储）
checkpointer = MemorySaver()

builder = StateGraph(State)

# 添加节点
builder.add_node("retrieve", retrieve_with_circuit_breaker, retry_policy=retrieval_retry)
builder.add_node("generate", generate_with_circuit_breaker, retry_policy=llm_retry)

# 添加边
builder.add_edge(START, "retrieve")
builder.add_edge("retrieve", "generate")
builder.add_edge("generate", END)

# 编译（带检查点）
graph = builder.compile(checkpointer=checkpointer)

print("✅ 生产级错误恢复系统构建完成\n")

# ===== 7. 测试 =====
print("="*60)
print("测试：完整容错流程")
print("="*60)

initial_state = {
    "query": "什么是 LangGraph？",
    "documents": [],
    "answer": "",
    "retry_count": 0,
    "fallback_used": "",
    "execution_time": 0.0,
    "errors": []
}

# 配置检查点
config = {"configurable": {"thread_id": "test-1"}}

try:
    result = graph.invoke(initial_state, config=config)
    print(f"\n✅ 执行成功")
    print(f"  文档数: {len(result['documents'])}")
    print(f"  答案: {result['answer'][:100]}...")
    print(f"  Fallback 使用: {result.get('fallback_used', 'none')}")
    print(f"  执行时间: {result.get('execution_time', 0):.2f}s")
except Exception as e:
    print(f"\n❌ 执行失败: {type(e).__name__}: {e}")

# ===== 8. 监控报告 =====
print("\n" + "="*60)
print("监控报告")
print("="*60)

summary = monitor.get_summary()
print(f"\n总事件数: {summary['total_events']}")
print(f"重试次数: {summary['retry_count']}")
print(f"Fallback 次数: {summary['fallback_count']}")
print(f"错误次数: {summary['error_count']}")

print(f"\n断路器状态:")
for name, status in summary['circuit_breaker_status'].items():
    print(f"  {name}:")
    print(f"    状态: {status['state']}")
    print(f"    失败次数: {status['failure_count']}")
    print(f"    成功次数: {status['success_count']}")

# ===== 9. 检查点恢复演示 =====
print("\n" + "="*60)
print("检查点恢复演示")
print("="*60)

print("\n场景：从检索节点恢复执行")
print("（模拟：检索完成后系统崩溃，从检查点恢复）")

# 获取检查点状态
checkpoint = checkpointer.get(config)
if checkpoint:
    print(f"✅ 找到检查点")
    print(f"  已完成节点: retrieve")
    print(f"  可以从 generate 节点继续执行")
else:
    print("❌ 未找到检查点")

# ===== 10. 实际应用建议 =====
print("\n" + "="*60)
print("实际应用建议")
print("="*60)

recommendations = """
1. **断路器配置**
   - 失败阈值：根据服务 SLA 设置（如 5 次失败）
   - 超时时间：根据恢复时间设置（如 60 秒）
   - 成功阈值：保守设置（如 2-3 次成功）

2. **检查点策略**
   - 使用持久化存储（PostgreSQL、Redis）
   - 在关键节点后保存检查点
   - 定期清理过期检查点

3. **监控指标**
   - 重试率：< 10% 正常，> 20% 需要告警
   - Fallback 率：< 5% 正常，> 10% 需要优化
   - 断路器打开次数：每天 < 3 次
   - 平均执行时间：< 5 秒

4. **告警规则**
   - 断路器打开 → 立即告警（P0）
   - 重试率 > 20% → 告警（P1）
   - Fallback 率 > 10% → 告警（P2）
   - 执行时间 > 10s → 告警（P3）

5. **降级策略**
   - L1: 主模型 + 主数据库
   - L2: 主模型 + 备数据库
   - L3: 备模型 + 备数据库
   - L4: 缓存答案
   - L5: 默认响应

6. **成本优化**
   - 主模型失败后快速降级到便宜模型
   - 使用缓存减少 LLM 调用
   - 断路器打开时直接走缓存

7. **测试策略**
   - 混沌工程：随机注入故障
   - 压力测试：模拟高并发
   - 恢复测试：验证检查点恢复
"""

print(recommendations)

print("\n" + "="*60)
print("✅ 场景4 完成")
print("="*60)
```

## 关键要点

### 1. 断路器三种状态

```
CLOSED（关闭）
  ↓ 失败次数 >= 阈值
OPEN（打开）
  ↓ 超时后
HALF_OPEN（半开）
  ↓ 成功次数 >= 阈值
CLOSED（关闭）
```

### 2. 多层级 Fallback

```python
# 检索 Fallback
主库 → 备库 → 默认文档

# LLM Fallback
GPT-4 → GPT-3.5 → 基于检索的默认答案
```

### 3. 检查点恢复

```python
# 使用检查点
checkpointer = MemorySaver()
graph = builder.compile(checkpointer=checkpointer)

# 带配置执行
config = {"configurable": {"thread_id": "test-1"}}
result = graph.invoke(state, config=config)
```

### 4. 完整监控

```python
monitor.log_event("retry", "node_name", {"attempt": 2})
monitor.log_event("fallback", "node_name", {"source": "backup"})
monitor.log_event("error", "node_name", {"error": str(e)})
```
