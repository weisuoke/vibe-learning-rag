# 07_工作流模式 - 实战代码 场景2：RAG 多步推理管道

> LangGraph L3_工作流编排 | 实战场景 2/6

---

## 场景描述

构建一个**自适应 RAG 管道**：根据查询类型选择不同的检索策略，结合顺序链式 + 路由模式。

```
用户提问 → 查询分类（路由）
              ├→ 事实型 → 直接检索 → 生成回答
              ├→ 分析型 → 查询改写 → 多轮检索 → 综合生成
              └→ 对话型 → 历史上下文 → 检索 → 生成回答
```

---

## 完整代码

```python
"""
自适应 RAG 多步推理管道
模式组合：路由模式 + 顺序链式模式

根据查询类型自动选择最优检索策略：
- 事实型查询：直接检索（快速）
- 分析型查询：查询改写 + 多轮检索（深度）
- 对话型查询：上下文增强检索（连贯）
"""
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


# ===== 1. 定义状态 =====
class RAGState(TypedDict):
    query: str                  # 用户原始查询
    query_type: str             # 查询类型：factual/analytical/conversational
    chat_history: list[str]     # 对话历史
    rewritten_query: str        # 改写后的查询
    retrieved_docs: list[str]   # 检索到的文档
    answer: str                 # 最终回答


# ===== 2. 模拟检索和生成函数 =====
def mock_retrieve(query: str, top_k: int = 3) -> list[str]:
    """模拟向量检索"""
    docs_db = {
        "python": ["Python是一种编程语言", "Python支持多种编程范式", "Python有丰富的库生态"],
        "langgraph": ["LangGraph是状态化工作流框架", "LangGraph基于图结构", "LangGraph支持持久化"],
        "rag": ["RAG结合检索与生成", "RAG提升LLM准确性", "RAG减少幻觉"],
    }
    results = []
    for keyword, docs in docs_db.items():
        if keyword in query.lower():
            results.extend(docs[:top_k])
    return results or [f"关于'{query}'的通用文档"]


def mock_generate(query: str, docs: list[str]) -> str:
    """模拟 LLM 生成回答"""
    context = "\n".join(f"- {d}" for d in docs)
    return f"基于以下资料回答'{query}':\n{context}\n\n综合以上信息，..."


# ===== 3. 查询分类节点（路由器）=====
def classify_query(state: RAGState) -> dict:
    """分类查询类型"""
    query = state["query"].lower()

    # 规则路由（简单高效）
    if any(kw in query for kw in ["是什么", "定义", "多少", "谁"]):
        query_type = "factual"
    elif any(kw in query for kw in ["为什么", "如何", "比较", "分析", "优缺点"]):
        query_type = "analytical"
    elif state.get("chat_history"):
        query_type = "conversational"
    else:
        query_type = "factual"  # 默认

    print(f"[分类] '{state['query']}' → {query_type}")
    return {"query_type": query_type}


def route_query(state: RAGState) -> str:
    """路由函数"""
    return state["query_type"]


# ===== 4. 事实型查询处理链（顺序链式）=====
def factual_retrieve(state: RAGState) -> dict:
    """事实型：直接检索"""
    docs = mock_retrieve(state["query"], top_k=3)
    print(f"[事实型检索] 找到 {len(docs)} 个文档")
    return {"retrieved_docs": docs}


def factual_generate(state: RAGState) -> dict:
    """事实型：直接生成"""
    answer = mock_generate(state["query"], state["retrieved_docs"])
    print(f"[事实型生成] 完成")
    return {"answer": answer}


# ===== 5. 分析型查询处理链（顺序链式 + 多步）=====
def rewrite_query(state: RAGState) -> dict:
    """分析型：查询改写（分解复杂问题）"""
    query = state["query"]
    # 模拟查询改写
    rewritten = f"{query} 详细原理 技术实现 最佳实践"
    print(f"[查询改写] '{query}' → '{rewritten}'")
    return {"rewritten_query": rewritten}


def analytical_retrieve(state: RAGState) -> dict:
    """分析型：深度检索（用改写后的查询）"""
    docs = mock_retrieve(state["rewritten_query"], top_k=5)
    # 模拟多轮检索：追加更多结果
    extra_docs = mock_retrieve(state["query"], top_k=2)
    all_docs = list(set(docs + extra_docs))
    print(f"[分析型检索] 多轮检索找到 {len(all_docs)} 个文档")
    return {"retrieved_docs": all_docs}


def analytical_generate(state: RAGState) -> dict:
    """分析型：综合生成（更详细的回答）"""
    answer = mock_generate(
        state["query"],
        state["retrieved_docs"]
    )
    answer = f"[深度分析]\n{answer}\n\n总结：这是一个需要多角度分析的问题..."
    print(f"[分析型生成] 完成")
    return {"answer": answer}


# ===== 6. 对话型查询处理链（上下文增强）=====
def enrich_with_context(state: RAGState) -> dict:
    """对话型：用历史上下文增强查询"""
    history = state.get("chat_history", [])
    context = " ".join(history[-3:])  # 取最近3轮
    enriched = f"{context} {state['query']}"
    print(f"[上下文增强] 融合 {len(history)} 轮历史")
    return {"rewritten_query": enriched}


def conversational_retrieve(state: RAGState) -> dict:
    """对话型：上下文增强检索"""
    docs = mock_retrieve(state["rewritten_query"], top_k=3)
    print(f"[对话型检索] 找到 {len(docs)} 个文档")
    return {"retrieved_docs": docs}


def conversational_generate(state: RAGState) -> dict:
    """对话型：连贯性生成"""
    answer = mock_generate(state["query"], state["retrieved_docs"])
    answer = f"[接续对话]\n{answer}"
    print(f"[对话型生成] 完成")
    return {"answer": answer}


# ===== 7. 构建图 =====
builder = StateGraph(RAGState)

# 路由节点
builder.add_node("classify", classify_query)

# 事实型处理链
builder.add_node("factual_retrieve", factual_retrieve)
builder.add_node("factual_generate", factual_generate)

# 分析型处理链
builder.add_node("rewrite", rewrite_query)
builder.add_node("analytical_retrieve", analytical_retrieve)
builder.add_node("analytical_generate", analytical_generate)

# 对话型处理链
builder.add_node("enrich_context", enrich_with_context)
builder.add_node("conv_retrieve", conversational_retrieve)
builder.add_node("conv_generate", conversational_generate)

# 边：路由
builder.add_edge(START, "classify")
builder.add_conditional_edges("classify", route_query, {
    "factual": "factual_retrieve",
    "analytical": "rewrite",
    "conversational": "enrich_context",
})

# 边：事实型链
builder.add_edge("factual_retrieve", "factual_generate")
builder.add_edge("factual_generate", END)

# 边：分析型链
builder.add_edge("rewrite", "analytical_retrieve")
builder.add_edge("analytical_retrieve", "analytical_generate")
builder.add_edge("analytical_generate", END)

# 边：对话型链
builder.add_edge("enrich_context", "conv_retrieve")
builder.add_edge("conv_retrieve", "conv_generate")
builder.add_edge("conv_generate", END)

rag_pipeline = builder.compile()


# ===== 8. 测试 =====
if __name__ == "__main__":
    # 测试1：事实型查询
    print("=" * 50)
    print("测试1：事实型查询")
    result = rag_pipeline.invoke({
        "query": "Python是什么？",
        "chat_history": [],
    })
    print(f"回答: {result['answer'][:100]}...")

    # 测试2：分析型查询
    print("\n" + "=" * 50)
    print("测试2：分析型查询")
    result = rag_pipeline.invoke({
        "query": "如何用LangGraph构建RAG系统？",
        "chat_history": [],
    })
    print(f"回答: {result['answer'][:100]}...")

    # 测试3：对话型查询
    print("\n" + "=" * 50)
    print("测试3：对话型查询")
    result = rag_pipeline.invoke({
        "query": "那它支持持久化吗？",
        "chat_history": ["用户问了LangGraph", "助手介绍了基本概念"],
    })
    print(f"回答: {result['answer'][:100]}...")
```

## 学习要点

1. **路由 + 顺序链式组合**：路由做分流，每条路径内部是顺序链
2. **规则路由的实用性**：关键词匹配在 RAG 场景中足够好用
3. **不同查询类型的处理差异**：事实型要快，分析型要深，对话型要连贯
4. **状态设计**：`rewritten_query` 字段被分析型和对话型共用
