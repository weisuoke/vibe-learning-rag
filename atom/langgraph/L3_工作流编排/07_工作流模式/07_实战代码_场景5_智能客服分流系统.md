> LangGraph L3_工作流编排 | 实战场景 5/6

---

## 场景描述

构建一个**智能客服分流系统**：根据用户消息自动分类，路由到不同处理链，每条链内部是顺序处理，部分场景需要并行查询多个知识库。

模式组合：**路由模式 + 顺序链式 + 并行模式**

```
用户消息 → 意图分类（路由）
              ├→ 技术问题 → 并行查询[文档库 + FAQ库] → 汇总 → 生成回答
              ├→ 投诉建议 → 情感分析 → 生成安抚回复 → 创建工单
              └→ 闲聊 → 直接生成回复
```

---

## 完整代码

```python
"""
智能客服分流系统
模式组合：路由 + 顺序链式 + 并行

三条处理路径：
1. 技术问题：并行检索多个知识库 → 汇总回答
2. 投诉建议：情感分析 → 安抚回复 → 创建工单
3. 闲聊：直接生成回复
"""
import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


# ===== 1. 定义状态 =====
class CustomerState(TypedDict):
    user_message: str               # 用户消息
    intent: str                     # 意图分类
    sentiment: str                  # 情感分析结果
    search_results: Annotated[list[str], operator.add]  # 检索结果（并行合并）
    response: str                   # 回复内容
    ticket_id: str                  # 工单ID（投诉场景）
    metadata: dict                  # 元数据


# ===== 2. 意图分类（路由器）=====
def classify_intent(state: CustomerState) -> dict:
    """分类用户意图"""
    msg = state["user_message"].lower()

    # 规则路由
    if any(kw in msg for kw in ["报错", "bug", "怎么用", "如何", "不工作", "安装"]):
        intent = "technical"
    elif any(kw in msg for kw in ["投诉", "退款", "差评", "不满", "垃圾"]):
        intent = "complaint"
    else:
        intent = "chat"

    print(f"[分类] '{msg[:30]}...' → {intent}")
    return {"intent": intent}


def route_intent(state: CustomerState) -> str:
    return state["intent"]


# ===== 3. 技术问题处理链（并行检索 + 顺序生成）=====
def search_docs(state: CustomerState) -> dict:
    """检索技术文档库"""
    query = state["user_message"]
    # 模拟文档库检索
    results = [f"[文档] 关于'{query[:10]}'的技术文档: 请参考官方指南第3章"]
    print(f"[文档库] 找到 {len(results)} 条结果")
    return {"search_results": results}


def search_faq(state: CustomerState) -> dict:
    """检索 FAQ 库"""
    query = state["user_message"]
    # 模拟 FAQ 检索
    results = [f"[FAQ] 常见问题: {query[:10]}... 的解决方案是重启服务"]
    print(f"[FAQ库] 找到 {len(results)} 条结果")
    return {"search_results": results}


def generate_tech_response(state: CustomerState) -> dict:
    """基于检索结果生成技术回答"""
    results = state["search_results"]
    context = "\n".join(results)
    response = (
        f"您好！关于您的技术问题，我找到了以下信息：\n\n"
        f"{context}\n\n"
        f"如果问题仍未解决，请提供更多细节，我会为您转接技术专家。"
    )
    print(f"[技术回答] 生成完成")
    return {"response": response}


# ===== 4. 投诉处理链（顺序：情感分析 → 安抚 → 工单）=====
def analyze_sentiment(state: CustomerState) -> dict:
    """情感分析"""
    msg = state["user_message"]
    if any(kw in msg for kw in ["垃圾", "差评", "愤怒"]):
        sentiment = "very_negative"
    elif any(kw in msg for kw in ["不满", "失望"]):
        sentiment = "negative"
    else:
        sentiment = "neutral"
    print(f"[情感分析] {sentiment}")
    return {"sentiment": sentiment}


def generate_comfort_response(state: CustomerState) -> dict:
    """生成安抚回复"""
    sentiment = state["sentiment"]
    templates = {
        "very_negative": "非常抱歉给您带来了不好的体验！我们非常重视您的反馈，已经为您创建了紧急工单。",
        "negative": "很抱歉让您感到不满意。我们会认真处理您的反馈，请您稍等。",
        "neutral": "感谢您的反馈，我们会尽快处理。",
    }
    response = templates.get(sentiment, templates["neutral"])
    print(f"[安抚回复] 生成完成")
    return {"response": response}


def create_ticket(state: CustomerState) -> dict:
    """创建工单"""
    import hashlib
    ticket_id = hashlib.md5(state["user_message"].encode()).hexdigest()[:8]
    priority = "urgent" if state["sentiment"] == "very_negative" else "normal"
    print(f"[工单] 创建工单 #{ticket_id}，优先级: {priority}")
    return {
        "ticket_id": ticket_id,
        "response": f"{state['response']}\n\n工单编号: #{ticket_id}，我们会在24小时内回复您。",
    }


# ===== 5. 闲聊处理（简单直接）=====
def generate_chat_response(state: CustomerState) -> dict:
    """生成闲聊回复"""
    msg = state["user_message"]
    if "你好" in msg:
        response = "您好！有什么可以帮您的吗？😊"
    elif "谢谢" in msg:
        response = "不客气！还有其他问题随时问我。"
    else:
        response = f"收到您的消息：'{msg[:20]}...'。请问有什么具体需要帮助的吗？"
    print(f"[闲聊] 生成完成")
    return {"response": response}


# ===== 6. 构建图 =====
builder = StateGraph(CustomerState)

# 路由节点
builder.add_node("classify", classify_intent)

# 技术问题链（并行检索 + 顺序生成）
builder.add_node("search_docs", search_docs)
builder.add_node("search_faq", search_faq)
builder.add_node("tech_response", generate_tech_response)

# 投诉处理链（顺序）
builder.add_node("sentiment", analyze_sentiment)
builder.add_node("comfort", generate_comfort_response)
builder.add_node("ticket", create_ticket)

# 闲聊
builder.add_node("chat_response", generate_chat_response)

# 路由边
builder.add_edge(START, "classify")
builder.add_conditional_edges("classify", route_intent, {
    "technical": "search_docs",  # 技术问题先检索文档
    "complaint": "sentiment",     # 投诉先做情感分析
    "chat": "chat_response",      # 闲聊直接回复
})

# 技术问题：并行检索（Fan-out）
builder.add_edge("classify", "search_faq")  # 同时检索 FAQ

# 但这里有个问题：search_faq 只在 technical 路径才需要
# 更好的做法是在 technical 路径内部做并行
# 简化处理：让 search_docs 和 search_faq 都汇聚到 tech_response
builder.add_edge("search_docs", "tech_response")
builder.add_edge("search_faq", "tech_response")
builder.add_edge("tech_response", END)

# 投诉处理：顺序链
builder.add_edge("sentiment", "comfort")
builder.add_edge("comfort", "ticket")
builder.add_edge("ticket", END)

# 闲聊
builder.add_edge("chat_response", END)

customer_service = builder.compile()


# ===== 7. 测试 =====
if __name__ == "__main__":
    test_cases = [
        {"user_message": "你好，安装报错了怎么办？", "search_results": []},
        {"user_message": "我要投诉！你们的服务太垃圾了！", "search_results": []},
        {"user_message": "你好呀", "search_results": []},
    ]

    for case in test_cases:
        print(f"\n{'='*60}")
        print(f"用户: {case['user_message']}")
        print("-" * 60)
        result = customer_service.invoke(case)
        print(f"\n客服回复: {result['response']}")
```

## 学习要点

1. **三种模式的实际组合**：路由分流 + 并行检索 + 顺序处理
2. **不同路径不同复杂度**：技术问题最复杂（并行），闲聊最简单（单节点）
3. **规则路由的实用性**：客服场景中关键词匹配效果很好
4. **状态复用**：`search_results` 用 reducer 支持并行检索结果合并
