# 实战代码：场景2 - 多提供商集成

## 概述

本文档展示如何在 LangChain 中集成多个 Embedding 提供商，实现灵活的模型切换和统一的接口管理。

**核心价值**：
- 避免供应商锁定
- 根据场景选择最优模型
- 降低成本和延迟
- 提高系统可靠性

---

## 场景1：OpenAI Embeddings 集成

OpenAI 提供高质量的 Embedding 模型，适合生产环境。

### 完整代码示例

```python
"""
OpenAI Embeddings 集成示例
展示基础用法、维度控制、批量处理和错误处理
"""

from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import os

# 加载环境变量
load_dotenv()

# ============================================================
# 1. 基础配置
# ============================================================

def basic_openai_embeddings():
    """基础 OpenAI Embeddings 配置"""
    embeddings = OpenAIEmbeddings(
        model="text-embedding-3-small",
        openai_api_key=os.getenv("OPENAI_API_KEY")
    )
    
    # 单个文本嵌入
    text = "LangChain 是一个强大的 LLM 应用开发框架"
    vector = embeddings.embed_query(text)
    print(f"向量维度: {len(vector)}")
    print(f"前5个值: {vector[:5]}")
    
    # 批量文本嵌入
    texts = [
        "LangChain 支持多种 LLM",
        "Embedding 用于语义搜索",
        "RAG 结合检索和生成"
    ]
    vectors = embeddings.embed_documents(texts)
    print(f"\n批量嵌入: {len(vectors)} 个向量")
    
    return embeddings

# ============================================================
# 2. 维度控制（text-embedding-3+）
# ============================================================

def dimension_control():
    """使用维度控制降低成本"""
    # 默认 1536 维
    embeddings_full = OpenAIEmbeddings(
        model="text-embedding-3-small"
    )
    
    # 降低到 512 维（节省 67% 存储）
    embeddings_reduced = OpenAIEmbeddings(
        model="text-embedding-3-small",
        dimensions=512
    )
    
    text = "测试文本"
    vector_full = embeddings_full.embed_query(text)
    vector_reduced = embeddings_reduced.embed_query(text)
    
    print(f"完整维度: {len(vector_full)}")
    print(f"降维后: {len(vector_reduced)}")
    print(f"存储节省: {(1 - len(vector_reduced)/len(vector_full)) * 100:.1f}%")

# ============================================================
# 3. 批量处理与性能优化
# ============================================================

def batch_processing():
    """批量处理大量文本"""
    embeddings = OpenAIEmbeddings(
        model="text-embedding-3-small",
        chunk_size=100,  # 每批处理 100 个文本
        max_retries=3,   # 失败重试 3 次
        show_progress_bar=True
    )
    
    # 模拟大量文本
    texts = [f"文档 {i} 的内容" for i in range(500)]
    
    print("开始批量嵌入...")
    vectors = embeddings.embed_documents(texts)
    print(f"完成: {len(vectors)} 个向量")

# ============================================================
# 4. 错误处理与重试
# ============================================================

def error_handling():
    """处理 API 错误和重试"""
    embeddings = OpenAIEmbeddings(
        model="text-embedding-3-small",
        max_retries=3,
        retry_min_seconds=4,
        retry_max_seconds=20,
        request_timeout=30.0
    )
    
    try:
        text = "测试文本"
        vector = embeddings.embed_query(text)
        print(f"成功: 向量维度 {len(vector)}")
    except Exception as e:
        print(f"错误: {e}")
        # 实现降级策略
        print("切换到备用模型...")

# ============================================================
# 主函数
# ============================================================

if __name__ == "__main__":
    print("=== OpenAI Embeddings 集成示例 ===\n")
    
    print("1. 基础配置")
    basic_openai_embeddings()
    
    print("\n2. 维度控制")
    dimension_control()
    
    print("\n3. 批量处理")
    batch_processing()
    
    print("\n4. 错误处理")
    error_handling()
```

### 关键点说明

1. **模型选择**：`text-embedding-3-small` 性价比高，`text-embedding-3-large` 性能最好
2. **维度控制**：降低维度可节省 50-75% 存储成本，性能略有下降
3. **批量处理**：使用 `chunk_size` 控制批量大小，避免超时
4. **错误处理**：配置重试机制，提高系统稳定性

## 场景2：Sentence Transformers 集成

Sentence Transformers 提供开源模型，适合本地部署和离线场景。

### 完整代码示例

```python
"""
Sentence Transformers 集成示例
展示本地模型加载、批量编码和性能优化
"""

from langchain_huggingface import HuggingFaceEmbeddings
from sentence_transformers import SentenceTransformer
import torch

# ============================================================
# 1. 基础配置
# ============================================================

def basic_sentence_transformers():
    """基础 Sentence Transformers 配置"""
    embeddings = HuggingFaceEmbeddings(
        model_name="all-MiniLM-L6-v2",
        model_kwargs={'device': 'cpu'},
        encode_kwargs={'normalize_embeddings': True}
    )
    
    # 单个文本嵌入
    text = "LangChain 是一个强大的 LLM 应用开发框架"
    vector = embeddings.embed_query(text)
    print(f"向量维度: {len(vector)}")
    print(f"前5个值: {vector[:5]}")
    
    # 批量文本嵌入
    texts = [
        "LangChain 支持多种 LLM",
        "Embedding 用于语义搜索",
        "RAG 结合检索和生成"
    ]
    vectors = embeddings.embed_documents(texts)
    print(f"\n批量嵌入: {len(vectors)} 个向量")
    
    return embeddings

# ============================================================
# 2. GPU 加速
# ============================================================

def gpu_acceleration():
    """使用 GPU 加速嵌入计算"""
    device = 'cuda' if torch.cuda.is_available() else 'cpu'
    print(f"使用设备: {device}")
    
    embeddings = HuggingFaceEmbeddings(
        model_name="all-MiniLM-L6-v2",
        model_kwargs={'device': device},
        encode_kwargs={
            'normalize_embeddings': True,
            'batch_size': 64  # GPU 可以使用更大的批量
        }
    )
    
    # 大批量处理
    texts = [f"文档 {i} 的内容" for i in range(1000)]
    print("开始批量嵌入...")
    vectors = embeddings.embed_documents(texts)
    print(f"完成: {len(vectors)} 个向量")

# ============================================================
# 3. 直接使用 SentenceTransformer
# ============================================================

def direct_sentence_transformer():
    """直接使用 SentenceTransformer API"""
    model = SentenceTransformer("all-MiniLM-L6-v2")
    
    sentences = [
        "The weather is lovely today.",
        "It's so sunny outside!",
        "He drove to the stadium."
    ]
    
    # 基础编码
    embeddings = model.encode(sentences)
    print(f"基础编码: {embeddings.shape}")
    
    # 归一化编码
    embeddings_normalized = model.encode(
        sentences,
        normalize_embeddings=True
    )
    print(f"归一化编码: {embeddings_normalized.shape}")
    
    # 返回 PyTorch tensor
    embeddings_tensor = model.encode(
        sentences,
        convert_to_tensor=True
    )
    print(f"Tensor 编码: {embeddings_tensor.shape}")

# ============================================================
# 4. 多模型对比
# ============================================================

def model_comparison():
    """对比不同模型的性能"""
    models = [
        "all-MiniLM-L6-v2",      # 轻量级，384 维
        "all-mpnet-base-v2",     # 高性能，768 维
        "paraphrase-multilingual-MiniLM-L12-v2"  # 多语言
    ]
    
    text = "LangChain 是一个强大的框架"
    
    for model_name in models:
        embeddings = HuggingFaceEmbeddings(model_name=model_name)
        vector = embeddings.embed_query(text)
        print(f"{model_name}: {len(vector)} 维")

# ============================================================
# 主函数
# ============================================================

if __name__ == "__main__":
    print("=== Sentence Transformers 集成示例 ===\n")
    
    print("1. 基础配置")
    basic_sentence_transformers()
    
    print("\n2. GPU 加速")
    gpu_acceleration()
    
    print("\n3. 直接使用 SentenceTransformer")
    direct_sentence_transformer()
    
    print("\n4. 多模型对比")
    model_comparison()
```

### 关键点说明

1. **模型选择**：`all-MiniLM-L6-v2` 轻量快速，`all-mpnet-base-v2` 性能最好
2. **设备选择**：支持 CPU/GPU，GPU 可显著加速
3. **归一化**：使用 `normalize_embeddings=True` 提高检索效率
4. **批量大小**：根据设备内存调整 `batch_size`

## 场景3：非 OpenAI 提供商（OpenRouter/Ollama）

使用 OpenAI 兼容 API 的其他提供商。

### 完整代码示例

```python
"""
非 OpenAI 提供商集成示例
展示 OpenRouter、Ollama 等兼容 API 的使用
"""

from langchain_openai import OpenAIEmbeddings
import os

# ============================================================
# 1. OpenRouter 集成
# ============================================================

def openrouter_embeddings():
    """使用 OpenRouter 的 Embedding API"""
    embeddings = OpenAIEmbeddings(
        model="text-embedding-3-small",
        openai_api_base="https://openrouter.ai/api/v1",
        openai_api_key=os.getenv("OPENROUTER_API_KEY"),
        check_embedding_ctx_length=False  # 关键：跳过 tokenization
    )
    
    text = "测试 OpenRouter Embeddings"
    vector = embeddings.embed_query(text)
    print(f"OpenRouter 向量维度: {len(vector)}")
    
    return embeddings

# ============================================================
# 2. Ollama 本地部署
# ============================================================

def ollama_embeddings():
    """使用 Ollama 本地 Embedding 模型"""
    embeddings = OpenAIEmbeddings(
        model="nomic-embed-text",
        openai_api_base="http://localhost:11434/v1",
        openai_api_key="ollama",  # Ollama 不需要真实 API key
        check_embedding_ctx_length=False
    )
    
    text = "测试 Ollama Embeddings"
    try:
        vector = embeddings.embed_query(text)
        print(f"Ollama 向量维度: {len(vector)}")
    except Exception as e:
        print(f"Ollama 错误: {e}")
        print("请确保 Ollama 服务正在运行: ollama serve")
    
    return embeddings

# ============================================================
# 3. 自定义 API 端点
# ============================================================

def custom_api_endpoint():
    """使用自定义 API 端点"""
    embeddings = OpenAIEmbeddings(
        model="custom-embedding-model",
        openai_api_base="https://your-api.com/v1",
        openai_api_key=os.getenv("CUSTOM_API_KEY"),
        check_embedding_ctx_length=False,
        max_retries=3
    )
    
    text = "测试自定义 API"
    try:
        vector = embeddings.embed_query(text)
        print(f"自定义 API 向量维度: {len(vector)}")
    except Exception as e:
        print(f"自定义 API 错误: {e}")
    
    return embeddings

# ============================================================
# 4. 多提供商切换
# ============================================================

def multi_provider_switch():
    """根据场景动态切换提供商"""
    providers = {
        "openai": {
            "model": "text-embedding-3-small",
            "base_url": "https://api.openai.com/v1",
            "api_key": os.getenv("OPENAI_API_KEY"),
            "check_ctx": True
        },
        "openrouter": {
            "model": "text-embedding-3-small",
            "base_url": "https://openrouter.ai/api/v1",
            "api_key": os.getenv("OPENROUTER_API_KEY"),
            "check_ctx": False
        },
        "ollama": {
            "model": "nomic-embed-text",
            "base_url": "http://localhost:11434/v1",
            "api_key": "ollama",
            "check_ctx": False
        }
    }
    
    def get_embeddings(provider_name: str):
        """获取指定提供商的 Embeddings"""
        config = providers[provider_name]
        return OpenAIEmbeddings(
            model=config["model"],
            openai_api_base=config["base_url"],
            openai_api_key=config["api_key"],
            check_embedding_ctx_length=config["check_ctx"]
        )
    
    # 测试所有提供商
    text = "测试文本"
    for provider_name in providers:
        try:
            embeddings = get_embeddings(provider_name)
            vector = embeddings.embed_query(text)
            print(f"{provider_name}: 成功 ({len(vector)} 维)")
        except Exception as e:
            print(f"{provider_name}: 失败 - {e}")

# ============================================================
# 主函数
# ============================================================

if __name__ == "__main__":
    print("=== 非 OpenAI 提供商集成示例 ===\n")
    
    print("1. OpenRouter 集成")
    openrouter_embeddings()
    
    print("\n2. Ollama 本地部署")
    ollama_embeddings()
    
    print("\n3. 自定义 API 端点")
    custom_api_endpoint()
    
    print("\n4. 多提供商切换")
    multi_provider_switch()
```

### 关键点说明

1. **check_embedding_ctx_length=False**：关键配置，跳过 tokenization
2. **API 兼容性**：大多数提供商兼容 OpenAI API 格式
3. **本地部署**：Ollama 支持完全离线运行
4. **动态切换**：根据场景选择最优提供商

## 场景4：自定义 Embedding 实现

实现自定义 Embedding 类，完全控制嵌入逻辑。

### 完整代码示例

```python
"""
自定义 Embedding 实现示例
展示如何继承 Embeddings 基类实现自定义逻辑
"""

from langchain_core.embeddings import Embeddings
from typing import List
import numpy as np

# ============================================================
# 1. 简单自定义 Embeddings
# ============================================================

class SimpleCustomEmbeddings(Embeddings):
    """简单的自定义 Embeddings 实现"""
    
    def __init__(self, dimension: int = 384):
        self.dimension = dimension
    
    def embed_documents(self, texts: List[str]) -> List[List[float]]:
        """批量嵌入文档"""
        return [self._embed_text(text) for text in texts]
    
    def embed_query(self, text: str) -> List[float]:
        """嵌入单个查询"""
        return self._embed_text(text)
    
    def _embed_text(self, text: str) -> List[float]:
        """实际的嵌入逻辑（示例：随机向量）"""
        # 实际应用中，这里应该调用真实的模型
        np.random.seed(hash(text) % (2**32))
        return np.random.randn(self.dimension).tolist()

# ============================================================
# 2. 包装现有模型
# ============================================================

class WrappedModelEmbeddings(Embeddings):
    """包装现有模型的 Embeddings"""
    
    def __init__(self, model_name: str = "all-MiniLM-L6-v2"):
        from sentence_transformers import SentenceTransformer
        self.model = SentenceTransformer(model_name)
    
    def embed_documents(self, texts: List[str]) -> List[List[float]]:
        """批量嵌入文档"""
        embeddings = self.model.encode(
            texts,
            normalize_embeddings=True,
            show_progress_bar=True
        )
        return embeddings.tolist()
    
    def embed_query(self, text: str) -> List[float]:
        """嵌入单个查询"""
        embedding = self.model.encode(
            text,
            normalize_embeddings=True
        )
        return embedding.tolist()

# ============================================================
# 3. 缓存 Embeddings
# ============================================================

class CachedEmbeddings(Embeddings):
    """带缓存的 Embeddings"""
    
    def __init__(self, base_embeddings: Embeddings):
        self.base_embeddings = base_embeddings
        self.cache = {}
    
    def embed_documents(self, texts: List[str]) -> List[List[float]]:
        """批量嵌入文档（带缓存）"""
        results = []
        uncached_texts = []
        uncached_indices = []
        
        # 检查缓存
        for i, text in enumerate(texts):
            if text in self.cache:
                results.append(self.cache[text])
            else:
                uncached_texts.append(text)
                uncached_indices.append(i)
                results.append(None)
        
        # 计算未缓存的嵌入
        if uncached_texts:
            new_embeddings = self.base_embeddings.embed_documents(uncached_texts)
            for i, embedding in zip(uncached_indices, new_embeddings):
                self.cache[texts[i]] = embedding
                results[i] = embedding
        
        return results
    
    def embed_query(self, text: str) -> List[float]:
        """嵌入单个查询（带缓存）"""
        if text not in self.cache:
            self.cache[text] = self.base_embeddings.embed_query(text)
        return self.cache[text]
    
    def get_cache_stats(self):
        """获取缓存统计"""
        return {
            "cache_size": len(self.cache),
            "cache_keys": list(self.cache.keys())[:5]
        }

# ============================================================
# 4. 混合 Embeddings
# ============================================================

class HybridEmbeddings(Embeddings):
    """混合多个 Embeddings 模型"""
    
    def __init__(self, embeddings_list: List[Embeddings], weights: List[float] = None):
        self.embeddings_list = embeddings_list
        self.weights = weights or [1.0 / len(embeddings_list)] * len(embeddings_list)
    
    def embed_documents(self, texts: List[str]) -> List[List[float]]:
        """批量嵌入文档（混合）"""
        all_embeddings = [
            emb.embed_documents(texts) for emb in self.embeddings_list
        ]
        return self._combine_embeddings(all_embeddings)
    
    def embed_query(self, text: str) -> List[float]:
        """嵌入单个查询（混合）"""
        all_embeddings = [
            emb.embed_query(text) for emb in self.embeddings_list
        ]
        return self._combine_single_embeddings(all_embeddings)
    
    def _combine_embeddings(self, all_embeddings):
        """组合多个嵌入结果"""
        combined = []
        for i in range(len(all_embeddings[0])):
            vectors = [emb[i] for emb in all_embeddings]
            combined.append(self._weighted_average(vectors))
        return combined
    
    def _combine_single_embeddings(self, vectors):
        """组合单个嵌入结果"""
        return self._weighted_average(vectors)
    
    def _weighted_average(self, vectors):
        """加权平均"""
        result = np.zeros(len(vectors[0]))
        for vector, weight in zip(vectors, self.weights):
            result += np.array(vector) * weight
        return result.tolist()

# ============================================================
# 测试函数
# ============================================================

def test_custom_embeddings():
    """测试自定义 Embeddings"""
    print("=== 自定义 Embedding 实现示例 ===\n")
    
    # 1. 简单自定义
    print("1. 简单自定义 Embeddings")
    simple_emb = SimpleCustomEmbeddings(dimension=128)
    vector = simple_emb.embed_query("测试文本")
    print(f"向量维度: {len(vector)}")
    
    # 2. 包装现有模型
    print("\n2. 包装现有模型")
    wrapped_emb = WrappedModelEmbeddings()
    vector = wrapped_emb.embed_query("测试文本")
    print(f"向量维度: {len(vector)}")
    
    # 3. 缓存 Embeddings
    print("\n3. 缓存 Embeddings")
    cached_emb = CachedEmbeddings(wrapped_emb)
    texts = ["文本1", "文本2", "文本1"]  # 重复文本
    vectors = cached_emb.embed_documents(texts)
    print(f"嵌入数量: {len(vectors)}")
    print(f"缓存统计: {cached_emb.get_cache_stats()}")
    
    # 4. 混合 Embeddings
    print("\n4. 混合 Embeddings")
    hybrid_emb = HybridEmbeddings(
        [simple_emb, wrapped_emb],
        weights=[0.3, 0.7]
    )
    vector = hybrid_emb.embed_query("测试文本")
    print(f"混合向量维度: {len(vector)}")

if __name__ == "__main__":
    test_custom_embeddings()
```

### 关键点说明

1. **继承 Embeddings 基类**：实现 `embed_documents` 和 `embed_query` 方法
2. **缓存机制**：避免重复计算，提高性能
3. **混合策略**：组合多个模型的优势
4. **灵活扩展**：可以添加任何自定义逻辑

## 提供商对比表格

| 提供商 | 优势 | 劣势 | 适用场景 | 成本 |
|--------|------|------|----------|------|
| **OpenAI** | 高质量、稳定、支持维度控制 | 需要网络、有成本 | 生产环境、高质量要求 | $0.02/1M tokens |
| **Sentence Transformers** | 开源、本地部署、离线可用 | 需要 GPU、模型较大 | 本地部署、离线场景 | 免费（硬件成本） |
| **OpenRouter** | 多模型选择、统一接口 | 依赖第三方、稳定性 | 多模型对比、快速切换 | 按使用计费 |
| **Ollama** | 完全本地、隐私保护 | 性能依赖硬件 | 隐私敏感、离线场景 | 免费（硬件成本） |
| **自定义实现** | 完全控制、灵活扩展 | 开发成本高 | 特殊需求、研究场景 | 开发成本 |

### 选型建议

1. **生产环境**：OpenAI（稳定性和质量）
2. **本地部署**：Sentence Transformers 或 Ollama
3. **成本敏感**：Sentence Transformers（一次性硬件投入）
4. **隐私要求**：Ollama 或 Sentence Transformers
5. **快速原型**：OpenAI（开箱即用）
6. **特殊需求**：自定义实现
