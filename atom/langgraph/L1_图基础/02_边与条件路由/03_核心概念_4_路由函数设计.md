# 03_核心概念_4_路由函数设计

## 概念定义

**路由函数（Routing Function）是条件边的核心组件，接受状态作为输入，根据业务逻辑返回目标节点名称，实现动态路由决策。**

## 核心要点

### 什么是路由函数？

路由函数是一个纯函数，接受当前状态作为输入，通过条件判断返回目标节点名称，决定工作流的下一步执行路径。

### 路由函数的三个特点

1. **状态驱动**：基于状态值做决策
2. **返回节点名**：返回目标节点的字符串标识
3. **类型安全**：使用 `Literal` 类型提示明确可能的返回值

---

## 函数签名

### 基本结构

```python
from typing import Literal

def route_func(state: State) -> Literal["node_a", "node_b", END]:
    """路由函数
    
    Args:
        state: 当前状态
        
    Returns:
        目标节点名称
    """
    # 决策逻辑
    if condition:
        return "node_a"
    return "node_b"
```

### 关键要素

1. **输入参数**：`state: State` - 当前状态对象
2. **返回类型**：`Literal[...]` - 明确可能的返回值
3. **返回值**：字符串 - 目标节点名称
4. **决策逻辑**：条件判断 - 基于状态值

## 设计原则

### 原则 1：单一职责

路由函数只负责决策，不执行业务逻辑。

```python
# ✅ 推荐：只做决策
def route_func(state: State) -> Literal["process", "skip"]:
    if state["needs_processing"]:
        return "process"
    return "skip"

# ❌ 不推荐：包含业务逻辑
def route_func(state: State) -> Literal["process", "skip"]:
    # 不应该在路由函数中执行耗时操作
    result = expensive_api_call()
    if result > 0.5:
        return "process"
    return "skip"
```

### 原则 2：逻辑清晰

使用扁平化的条件判断，避免深层嵌套。

```python
# ✅ 推荐：扁平化逻辑
def route_func(state: State) -> Literal["a", "b", "c", "d"]:
    if state["priority"] == "urgent" and state["approved"]:
        return "a"
    if state["priority"] == "urgent":
        return "b"
    if state["approved"]:
        return "c"
    return "d"

# ❌ 不推荐：深层嵌套
def route_func(state: State) -> Literal["a", "b", "c", "d"]:
    if state["priority"] == "urgent":
        if state["approved"]:
            return "a"
        else:
            return "b"
    else:
        if state["approved"]:
            return "c"
        else:
            return "d"
```

### 原则 3：类型安全

使用 `Literal` 类型提示明确所有可能的返回值。

```python
# ✅ 推荐：明确类型
def route_func(state: State) -> Literal["high", "medium", "low", END]:
    score = state["score"]
    if score > 0.8:
        return "high"
    elif score > 0.5:
        return "medium"
    elif score > 0.2:
        return "low"
    return END

# ❌ 不推荐：无类型提示
def route_func(state):
    # 无法知道可能的返回值
    return "high"
```

### 原则 4：边界处理

处理缺失值、异常值和边界情况。

```python
# ✅ 推荐：完整的边界处理
def route_func(state: State) -> Literal["process", "error", END]:
    # 处理缺失值
    score = state.get("score")
    if score is None:
        return "error"
    
    # 处理异常值
    if score < 0 or score > 1:
        return "error"
    
    # 正常路由
    if score > 0.5:
        return "process"
    return END

# ❌ 不推荐：无边界处理
def route_func(state: State) -> Literal["process", END]:
    # 可能抛出 KeyError 或 TypeError
    if state["score"] > 0.5:
        return "process"
    return END
```

## 实战示例

### 示例 1：简单二分支

```python
from langgraph.graph import StateGraph, START, END
from typing import TypedDict, Literal

class State(TypedDict):
    value: int
    result: str

def route_by_value(state: State) -> Literal["positive", "negative"]:
    """根据值的正负路由"""
    if state["value"] >= 0:
        return "positive"
    return "negative"

def process_positive(state: State) -> dict:
    return {"result": f"Positive: {state['value']}"}

def process_negative(state: State) -> dict:
    return {"result": f"Negative: {state['value']}"}

# 构建图
graph = StateGraph(State)
graph.add_node("check", lambda s: s)
graph.add_node("positive", process_positive)
graph.add_node("negative", process_negative)

graph.add_edge(START, "check")
graph.add_conditional_edges(
    "check",
    route_by_value,
    {"positive": "positive", "negative": "negative"}
)
graph.add_edge("positive", END)
graph.add_edge("negative", END)

app = graph.compile()
```

### 示例 2：多分支决策

```python
from typing import Literal

def route_by_priority(state: State) -> Literal["urgent", "high", "normal", "low"]:
    """根据优先级路由"""
    priority = state.get("priority", 0)
    
    if priority >= 90:
        return "urgent"
    elif priority >= 70:
        return "high"
    elif priority >= 40:
        return "normal"
    return "low"

graph.add_conditional_edges(
    "classify",
    route_by_priority,
    {
        "urgent": "handle_urgent",
        "high": "handle_high",
        "normal": "handle_normal",
        "low": "handle_low"
    }
)
```

### 示例 3：带重试的路由

```python
def route_with_retry(state: State) -> Literal["retry", "success", "failed"]:
    """带重试逻辑的路由"""
    retry_count = state.get("retry_count", 0)
    has_error = state.get("error", False)
    
    # 成功，结束
    if not has_error:
        return "success"
    
    # 重试次数未超限，继续重试
    if retry_count < 3:
        print(f"Retry attempt {retry_count + 1}/3")
        return "retry"
    
    # 重试次数超限，失败
    print("Max retries reached")
    return "failed"

graph.add_conditional_edges(
    "process",
    route_with_retry,
    {
        "retry": "process",  # 循环回自己
        "success": "finalize",
        "failed": "error_handler"
    }
)
```

### 示例 4：复杂业务路由

```python
def route_approval(state: State) -> Literal["auto_approve", "manual_review", "reject", END]:
    """审批路由逻辑"""
    score = state.get("score", 0.0)
    amount = state.get("amount", 0)
    user_level = state.get("user_level", "basic")
    
    # 自动拒绝：分数太低
    if score < 0.3:
        print(f"Auto reject: score={score}")
        return "reject"
    
    # 自动批准：高分 + 小额 + 高级用户
    if score > 0.9 and amount < 1000 and user_level in ["premium", "vip"]:
        print(f"Auto approve: score={score}, amount={amount}, level={user_level}")
        return "auto_approve"
    
    # 人工审核：中等分数或大额
    if score >= 0.5 or amount >= 1000:
        print(f"Manual review required: score={score}, amount={amount}")
        return "manual_review"
    
    # 其他情况：拒绝
    print(f"Reject: score={score}, amount={amount}")
    return "reject"

graph.add_conditional_edges(
    "evaluate",
    route_approval,
    {
        "auto_approve": "approve",
        "manual_review": "human_review",
        "reject": "reject",
        END: END
    }
)
```

## 最佳实践

### 1. 添加日志

```python
def route_func(state: State) -> Literal["a", "b"]:
    score = state["score"]
    print(f"[Router] score={score}")
    
    if score > 0.5:
        print("[Router] → path_a")
        return "a"
    
    print("[Router] → path_b")
    return "b"
```

### 2. 使用常量

```python
# 定义路由常量
ROUTE_HIGH = "high_priority"
ROUTE_NORMAL = "normal_priority"
ROUTE_LOW = "low_priority"

def route_func(state: State) -> Literal["high_priority", "normal_priority", "low_priority"]:
    priority = state["priority"]
    
    if priority > 0.8:
        return ROUTE_HIGH
    elif priority > 0.5:
        return ROUTE_NORMAL
    return ROUTE_LOW
```

### 3. 提取决策逻辑

```python
def should_approve(state: State) -> bool:
    """提取决策逻辑"""
    return state["score"] > 0.8 and state["amount"] < 1000

def route_func(state: State) -> Literal["approve", "reject"]:
    if should_approve(state):
        return "approve"
    return "reject"
```

### 4. 使用枚举

```python
from enum import Enum

class Route(str, Enum):
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"

def route_func(state: State) -> Literal["high", "medium", "low"]:
    score = state["score"]
    
    if score > 0.8:
        return Route.HIGH.value
    elif score > 0.5:
        return Route.MEDIUM.value
    return Route.LOW.value
```

## 常见错误

### 错误 1：在路由函数中执行耗时操作

```python
# ❌ 错误
def route_func(state: State) -> Literal["a", "b"]:
    result = expensive_api_call()  # 耗时操作
    if result > 0.5:
        return "a"
    return "b"

# ✅ 正确：在节点中执行
def process_node(state: State) -> dict:
    result = expensive_api_call()
    return {"result": result}

def route_func(state: State) -> Literal["a", "b"]:
    if state["result"] > 0.5:
        return "a"
    return "b"
```

### 错误 2：返回未定义的节点

```python
# ❌ 错误
def route_func(state: State) -> Literal["a", "b"]:
    return "c"  # "c" 不在 Literal 中

# ✅ 正确
def route_func(state: State) -> Literal["a", "b", "c"]:
    return "c"
```

### 错误 3：忘记处理所有分支

```python
# ❌ 错误：可能没有返回值
def route_func(state: State) -> Literal["a", "b"]:
    if state["flag"]:
        return "a"
    # 缺少 else 分支

# ✅ 正确
def route_func(state: State) -> Literal["a", "b"]:
    if state["flag"]:
        return "a"
    return "b"
```

## 性能优化

### 1. 避免重复计算

```python
# ❌ 不推荐
def route_func(state: State) -> Literal["a", "b"]:
    if expensive_calculation(state) > 0.5:
        return "a"
    return "b"

# ✅ 推荐：缓存结果
def route_func(state: State) -> Literal["a", "b"]:
    # 假设结果已在节点中计算并存储
    if state.get("cached_result", 0) > 0.5:
        return "a"
    return "b"
```

### 2. 使用简单条件

```python
# ✅ 快速
def route_func(state: State) -> Literal["a", "b"]:
    return "a" if state["flag"] else "b"

# ❌ 慢
def route_func(state: State) -> Literal["a", "b"]:
    import time
    time.sleep(0.1)  # 模拟复杂计算
    return "a"
```
