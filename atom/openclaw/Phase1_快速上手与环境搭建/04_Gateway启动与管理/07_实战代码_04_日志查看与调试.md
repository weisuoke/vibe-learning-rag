## 场景 1：实时日志监控

### 原理

实时日志监控通过 `openclaw logs --follow` 命令持续监控 Gateway 日志，适合开发调试和问题排查。

### 实现

```typescript
// scripts/monitor-logs.ts
/**
 * 实时日志监控脚本
 * 用途：持续监控 Gateway 日志，支持过滤和高亮
 */

import { spawn } from 'child_process';
import * as readline from 'readline';

interface LogMonitorOptions {
  filter?: string[];
  highlight?: string[];
  level?: 'error' | 'warn' | 'info' | 'debug' | 'trace';
  json?: boolean;
}

/**
 * 实时监控日志
 */
function monitorLogs(options: LogMonitorOptions = {}): void {
  const {
    filter = [],
    highlight = ['error', 'warn', 'failed'],
    level,
    json = false
  } = options;

  console.log('📊 Starting log monitor...');
  console.log(`   Filter: ${filter.length > 0 ? filter.join(', ') : 'none'}`);
  console.log(`   Highlight: ${highlight.join(', ')}`);
  console.log(`   Level: ${level || 'all'}`);
  console.log('');

  // 构建命令
  const args = ['logs', '--follow'];
  if (json) {
    args.push('--json');
  }

  // 启动日志监控
  const logs = spawn('openclaw', args, {
    stdio: ['inherit', 'pipe', 'pipe'],
    shell: true
  });

  // 处理 stdout
  const rl = readline.createInterface({
    input: logs.stdout,
    crlfDelay: Infinity
  });

  rl.on('line', (line) => {
    // 过滤日志
    if (filter.length > 0) {
      const shouldShow = filter.some(f => line.toLowerCase().includes(f.toLowerCase()));
      if (!shouldShow) return;
    }

    // 高亮关键词
    let displayLine = line;
    for (const keyword of highlight) {
      const regex = new RegExp(keyword, 'gi');
      displayLine = displayLine.replace(regex, `\x1b[31m${keyword}\x1b[0m`);
    }

    console.log(displayLine);
  });

  // 处理 stderr
  logs.stderr.on('data', (data) => {
    console.error(`\x1b[31m${data}\x1b[0m`);
  });

  // 处理退出
  logs.on('exit', (code) => {
    console.log(`\n📊 Log monitor stopped (exit code: ${code})`);
    process.exit(code || 0);
  });

  // Ctrl+C 处理
  process.on('SIGINT', () => {
    console.log('\n🛑 Stopping log monitor...');
    logs.kill('SIGINT');
  });
}

// 使用示例
if (require.main === module) {
  monitorLogs({
    filter: ['gateway', 'error'],
    highlight: ['error', 'warn', 'failed', 'refused'],
    level: 'info'
  });
}

export { monitorLogs };
```

## 场景 2：日志解析与分析

### 原理

解析 JSON Lines 格式的日志文件，提取关键信息，生成统计报告。

### 实现

```typescript
// scripts/analyze-logs.ts
/**
 * 日志解析与分析脚本
 * 用途：解析 Gateway 日志，生成统计报告
 */

import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import * as readline from 'readline';

interface LogEntry {
  timestamp: string;
  level: string;
  subsystem: string;
  message: string;
  context?: any;
}

interface LogStats {
  total: number;
  byLevel: Record<string, number>;
  bySubsystem: Record<string, number>;
  errors: LogEntry[];
  warnings: LogEntry[];
}

/**
 * 分析日志文件
 */
async function analyzeLogs(logPath?: string): Promise<LogStats> {
  const defaultPath = path.join(os.homedir(), '.openclaw', 'logs', 'gateway.log');
  const filePath = logPath || defaultPath;

  console.log('📊 Analyzing logs...');
  console.log(`   File: ${filePath}`);
  console.log('');

  if (!fs.existsSync(filePath)) {
    throw new Error(`Log file not found: ${filePath}`);
  }

  const stats: LogStats = {
    total: 0,
    byLevel: {},
    bySubsystem: {},
    errors: [],
    warnings: []
  };

  // 读取日志文件
  const fileStream = fs.createReadStream(filePath);
  const rl = readline.createInterface({
    input: fileStream,
    crlfDelay: Infinity
  });

  for await (const line of rl) {
    if (!line.trim()) continue;

    try {
      const entry: LogEntry = JSON.parse(line);
      stats.total++;

      // 按级别统计
      stats.byLevel[entry.level] = (stats.byLevel[entry.level] || 0) + 1;

      // 按子系统统计
      stats.bySubsystem[entry.subsystem] = (stats.bySubsystem[entry.subsystem] || 0) + 1;

      // 收集错误和警告
      if (entry.level === 'error') {
        stats.errors.push(entry);
      } else if (entry.level === 'warn') {
        stats.warnings.push(entry);
      }
    } catch (error) {
      // 跳过无效的 JSON 行
      continue;
    }
  }

  // 打印统计报告
  printStats(stats);

  return stats;
}

/**
 * 打印统计报告
 */
function printStats(stats: LogStats): void {
  console.log('📊 Log Analysis Report');
  console.log('='.repeat(50));
  console.log('');

  console.log(`Total Entries: ${stats.total}`);
  console.log('');

  console.log('By Level:');
  for (const [level, count] of Object.entries(stats.byLevel)) {
    const percentage = ((count / stats.total) * 100).toFixed(2);
    console.log(`  ${level.padEnd(10)} ${count.toString().padStart(6)} (${percentage}%)`);
  }
  console.log('');

  console.log('By Subsystem:');
  for (const [subsystem, count] of Object.entries(stats.bySubsystem)) {
    const percentage = ((count / stats.total) * 100).toFixed(2);
    console.log(`  ${subsystem.padEnd(15)} ${count.toString().padStart(6)} (${percentage}%)`);
  }
  console.log('');

  if (stats.errors.length > 0) {
    console.log(`Errors (${stats.errors.length}):`);
    stats.errors.slice(0, 5).forEach((entry, i) => {
      console.log(`  ${i + 1}. [${entry.timestamp}] ${entry.message}`);
    });
    if (stats.errors.length > 5) {
      console.log(`  ... and ${stats.errors.length - 5} more`);
    }
    console.log('');
  }

  if (stats.warnings.length > 0) {
    console.log(`Warnings (${stats.warnings.length}):`);
    stats.warnings.slice(0, 5).forEach((entry, i) => {
      console.log(`  ${i + 1}. [${entry.timestamp}] ${entry.message}`);
    });
    if (stats.warnings.length > 5) {
      console.log(`  ... and ${stats.warnings.length - 5} more`);
    }
    console.log('');
  }
}

// 使用示例
if (require.main === module) {
  analyzeLogs().catch((error) => {
    console.error('Analysis failed:', error);
    process.exit(1);
  });
}

export { analyzeLogs, LogStats };
```

## 场景 3：会话转录分析

### 原理

解析会话转录文件（transcript.jsonl），提取工具调用、模型响应等关键信息。

### 实现

```typescript
// scripts/analyze-transcript.ts
/**
 * 会话转录分析脚本
 * 用途：分析 Agent 会话转录，提取工具调用和响应
 */

import * as fs from 'fs';
import * as path from 'path';
import * as readline from 'readline';

interface TranscriptEntry {
  type: string;
  timestamp: string;
  content?: any;
  tool_use?: any;
  tool_result?: any;
}

interface TranscriptStats {
  totalMessages: number;
  toolCalls: number;
  modelResponses: number;
  tools: Record<string, number>;
  timeline: TranscriptEntry[];
}

/**
 * 分析会话转录
 */
async function analyzeTranscript(transcriptPath: string): Promise<TranscriptStats> {
  console.log('📝 Analyzing session transcript...');
  console.log(`   File: ${transcriptPath}`);
  console.log('');

  if (!fs.existsSync(transcriptPath)) {
    throw new Error(`Transcript file not found: ${transcriptPath}`);
  }

  const stats: TranscriptStats = {
    totalMessages: 0,
    toolCalls: 0,
    modelResponses: 0,
    tools: {},
    timeline: []
  };

  // 读取转录文件
  const fileStream = fs.createReadStream(transcriptPath);
  const rl = readline.createInterface({
    input: fileStream,
    crlfDelay: Infinity
  });

  for await (const line of rl) {
    if (!line.trim()) continue;

    try {
      const entry: TranscriptEntry = JSON.parse(line);
      stats.totalMessages++;
      stats.timeline.push(entry);

      // 统计工具调用
      if (entry.type === 'tool_use') {
        stats.toolCalls++;
        const toolName = entry.tool_use?.name || 'unknown';
        stats.tools[toolName] = (stats.tools[toolName] || 0) + 1;
      }

      // 统计模型响应
      if (entry.type === 'text') {
        stats.modelResponses++;
      }
    } catch (error) {
      continue;
    }
  }

  // 打印统计报告
  printTranscriptStats(stats);

  return stats;
}

/**
 * 打印转录统计报告
 */
function printTranscriptStats(stats: TranscriptStats): void {
  console.log('📝 Transcript Analysis Report');
  console.log('='.repeat(50));
  console.log('');

  console.log(`Total Messages: ${stats.totalMessages}`);
  console.log(`Tool Calls: ${stats.toolCalls}`);
  console.log(`Model Responses: ${stats.modelResponses}`);
  console.log('');

  if (Object.keys(stats.tools).length > 0) {
    console.log('Tools Used:');
    const sortedTools = Object.entries(stats.tools).sort((a, b) => b[1] - a[1]);
    sortedTools.forEach(([tool, count]) => {
      console.log(`  ${tool.padEnd(20)} ${count.toString().padStart(4)} calls`);
    });
    console.log('');
  }

  console.log('Timeline (last 10 events):');
  stats.timeline.slice(-10).forEach((entry, i) => {
    const time = new Date(entry.timestamp).toLocaleTimeString();
    console.log(`  ${i + 1}. [${time}] ${entry.type}`);
  });
}

// 使用示例
if (require.main === module) {
  const transcriptPath = process.argv[2];
  if (!transcriptPath) {
    console.error('Usage: ts-node analyze-transcript.ts <transcript-path>');
    process.exit(1);
  }

  analyzeTranscript(transcriptPath).catch((error) => {
    console.error('Analysis failed:', error);
    process.exit(1);
  });
}

export { analyzeTranscript, TranscriptStats };
```

## 最佳实践

### 1. 开发环境日志配置

```json5
{
  logging: {
    level: "debug",
    redactSensitive: true,
    console: {
      enabled: true,
      timestamp: true,
      color: true
    }
  }
}
```

### 2. 生产环境日志配置

```json5
{
  logging: {
    level: "info",
    redactSensitive: true,
    file: {
      enabled: true,
      maxSize: "100MB",
      maxFiles: 10,
      rotation: "daily"
    }
  }
}
```

### 3. 实时监控命令

```bash
# 监控所有日志
openclaw logs --follow

# 监控错误日志
openclaw logs --follow | grep "error"

# 监控特定子系统
openclaw logs --follow | grep "gateway"
```

## 故障排查

### 问题 1：日志文件过大

```bash
# 检查日志大小
du -sh ~/.openclaw/logs/

# 清理旧日志
find ~/.openclaw/logs/ -name "*.log" -mtime +30 -delete

# 配置日志轮转
# 在 openclaw.json 中设置 logging.file.rotation: "daily"
```

### 问题 2：日志级别过低

```bash
# 临时启用详细日志
openclaw gateway --verbose

# 永久配置
openclaw config set logging.level "debug"
```

### 问题 3：敏感信息泄露

```bash
# 启用敏感数据脱敏
openclaw config set logging.redactSensitive true

# 检查日志中的敏感信息
grep -i "token\|password\|secret" ~/.openclaw/logs/gateway.log
```
