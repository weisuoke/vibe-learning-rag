# 实战代码：场景7 - 中间件模式

> 本文档展示如何在 LangGraph 中实现中间件模式，用于处理横切关注点（Cross-cutting Concerns）

---

## 1. 场景描述

### 什么是中间件模式？

**中间件模式**是一种设计模式，通过在主要业务逻辑节点之间插入中间件节点，实现横切关注点的处理，如日志记录、数据验证、性能监控、错误处理等。

**核心特点：**
- **不侵入业务逻辑**：中间件节点独立于业务节点
- **可复用**：同一个中间件可以在多个地方使用
- **可组合**：多个中间件可以链式组合
- **职责单一**：每个中间件只处理一个关注点

### 为什么需要中间件？

在 LangGraph 工作流中，我们经常需要：
- **日志记录**：记录每个节点的输入输出
- **数据验证**：验证状态数据的合法性
- **性能监控**：记录节点执行时间
- **错误处理**：统一处理异常
- **输出格式化**：统一格式化输出

如果在每个业务节点中都实现这些功能，会导致代码重复和维护困难。中间件模式可以优雅地解决这个问题。

---

## 2. 核心挑战

### 挑战1：如何不侵入业务逻辑？

中间件节点应该独立于业务节点，通过图的边连接来组合，而不是修改业务节点的代码。

### 挑战2：如何实现可复用的中间件？

中间件应该是通用的，可以在不同的工作流中复用。这需要中间件设计时考虑通用性和配置化。

### 挑战3：如何组合多个中间件？

多个中间件需要按照正确的顺序执行，并且不能相互干扰。

## 3. 完整代码示例

### 场景：RAG 问答系统的中间件实现

我们将实现一个简单的 RAG 问答系统，使用中间件处理日志、验证和性能监控。

```python
"""
LangGraph 中间件模式实战示例
演示：RAG 问答系统中的中间件应用
"""

from langgraph.graph import StateGraph, START, END
from typing_extensions import TypedDict
import time
from datetime import datetime


# ===== 1. 定义状态 =====
class RAGState(TypedDict):
    query: str              # 用户查询
    documents: list[str]    # 检索到的文档
    answer: str             # 生成的答案
    error: str | None       # 错误信息
    metadata: dict          # 元数据（日志、性能等）


# ===== 2. 中间件节点实现 =====

def logging_middleware(state: RAGState) -> dict:
    """
    日志中间件：记录当前状态

    设计原则：
    - 只读取状态，不修改业务数据
    - 将日志信息存储在 metadata 中
    """
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    log_entry = f"[{timestamp}] State: query={state.get('query', 'N/A')}, " \
                f"docs={len(state.get('documents', []))}, " \
                f"answer={'Yes' if state.get('answer') else 'No'}"

    print(f"📝 {log_entry}")

    # 将日志存储在 metadata 中
    metadata = state.get("metadata", {})
    logs = metadata.get("logs", [])
    logs.append(log_entry)
    metadata["logs"] = logs

    return {"metadata": metadata}


def validation_middleware(state: RAGState) -> dict:
    """
    验证中间件：验证输入数据的合法性

    设计原则：
    - 检查必需字段
    - 验证数据格式
    - 设置错误信息
    """
    errors = []

    # 验证 query
    if not state.get("query") or not state["query"].strip():
        errors.append("Query is empty")

    # 验证 query 长度
    if state.get("query") and len(state["query"]) > 500:
        errors.append("Query is too long (max 500 characters)")

    if errors:
        print(f"❌ Validation failed: {', '.join(errors)}")
        return {"error": "; ".join(errors)}

    print("✅ Validation passed")
    return {}


def performance_middleware(state: RAGState) -> dict:
    """
    性能监控中间件：记录执行时间

    设计原则：
    - 记录时间戳
    - 计算执行时长
    - 存储在 metadata 中
    """
    metadata = state.get("metadata", {})

    # 记录当前时间戳
    current_time = time.time()

    # 如果有上一个时间戳，计算时长
    if "last_timestamp" in metadata:
        duration = current_time - metadata["last_timestamp"]
        print(f"⏱️  Last step took {duration:.3f}s")

        # 记录执行时长
        durations = metadata.get("durations", [])
        durations.append(duration)
        metadata["durations"] = durations

    # 更新时间戳
    metadata["last_timestamp"] = current_time

    return {"metadata": metadata}


# ===== 3. 业务节点实现 =====

def retrieve_documents(state: RAGState) -> dict:
    """检索文档（模拟）"""
    query = state["query"]
    print(f"🔍 Retrieving documents for: {query}")

    # 模拟检索
    time.sleep(0.1)
    documents = [
        f"Document 1 about {query}",
        f"Document 2 about {query}",
        f"Document 3 about {query}"
    ]

    return {"documents": documents}


def generate_answer(state: RAGState) -> dict:
    """生成答案（模拟）"""
    query = state["query"]
    documents = state["documents"]

    print(f"💡 Generating answer based on {len(documents)} documents")

    # 模拟生成
    time.sleep(0.2)
    answer = f"Based on the documents, the answer to '{query}' is: [Generated Answer]"

    return {"answer": answer}


# ===== 4. 构建带中间件的图 =====

def build_rag_graph_with_middleware():
    """构建带中间件的 RAG 图"""
    builder = StateGraph(RAGState)

    # 添加中间件节点
    builder.add_node("log_start", logging_middleware)
    builder.add_node("validate", validation_middleware)
    builder.add_node("perf_after_validate", performance_middleware)

    # 添加业务节点
    builder.add_node("retrieve", retrieve_documents)
    builder.add_node("log_after_retrieve", logging_middleware)
    builder.add_node("perf_after_retrieve", performance_middleware)

    builder.add_node("generate", generate_answer)
    builder.add_node("log_end", logging_middleware)
    builder.add_node("perf_end", performance_middleware)

    # 连接节点（中间件 + 业务逻辑）
    builder.add_edge(START, "log_start")
    builder.add_edge("log_start", "validate")
    builder.add_edge("validate", "perf_after_validate")
    builder.add_edge("perf_after_validate", "retrieve")
    builder.add_edge("retrieve", "log_after_retrieve")
    builder.add_edge("log_after_retrieve", "perf_after_retrieve")
    builder.add_edge("perf_after_retrieve", "generate")
    builder.add_edge("generate", "log_end")
    builder.add_edge("log_end", "perf_end")
    builder.add_edge("perf_end", END)

    return builder.compile()


# ===== 5. 运行示例 =====

if __name__ == "__main__":
    print("=" * 60)
    print("LangGraph 中间件模式示例")
    print("=" * 60)

    # 构建图
    graph = build_rag_graph_with_middleware()

    # 测试用例1：正常查询
    print("\n【测试1：正常查询】")
    result = graph.invoke({
        "query": "What is LangGraph?",
        "documents": [],
        "answer": "",
        "error": None,
        "metadata": {}
    })

    print("\n最终结果：")
    print(f"Query: {result['query']}")
    print(f"Answer: {result['answer']}")
    print(f"Total logs: {len(result['metadata'].get('logs', []))}")
    print(f"Total durations: {result['metadata'].get('durations', [])}")

    # 测试用例2：空查询（验证失败）
    print("\n" + "=" * 60)
    print("【测试2：空查询（验证失败）】")
    result2 = graph.invoke({
        "query": "",
        "documents": [],
        "answer": "",
        "error": None,
        "metadata": {}
    })

    print("\n最终结果：")
    print(f"Error: {result2.get('error', 'None')}")
```

## 5. 关键点解析

### 5.1 中间件节点的设计原则

**原则1：职责单一**
- 每个中间件只处理一个关注点
- 日志中间件只负责日志，验证中间件只负责验证

**原则2：不修改业务数据**
- 中间件应该只读取业务数据，不修改
- 如果需要存储信息，使用 `metadata` 字段

**原则3：返回部分状态**
- 中间件返回 `dict`，只包含需要更新的字段
- 空 `dict` 表示不修改任何状态

### 5.2 中间件的执行顺序

中间件的执行顺序由图的边决定：

```
START → log_start → validate → perf → retrieve → log → perf → generate → log → perf → END
```

**关键点：**
- 日志中间件在每个关键节点后执行
- 验证中间件在业务逻辑前执行
- 性能监控中间件在每个节点后执行

### 5.3 状态传递机制

LangGraph 的状态传递机制：
1. 节点返回 `dict`，表示状态的部分更新
2. LangGraph 自动合并更新到全局状态
3. 下一个节点接收更新后的状态

**示例：**
```python
# 节点1返回
{"documents": ["doc1", "doc2"]}

# 全局状态更新
state["documents"] = ["doc1", "doc2"]

# 节点2接收更新后的状态
state = {"query": "...", "documents": ["doc1", "doc2"], ...}
```

### 5.4 中间件的可复用性

同一个中间件可以在多个地方使用：

```python
# 日志中间件在3个地方使用
builder.add_node("log_start", logging_middleware)
builder.add_node("log_after_retrieve", logging_middleware)
builder.add_node("log_end", logging_middleware)
```

## 6. 进阶技巧

### 6.1 中间件工厂函数

使用工厂函数创建可配置的中间件：

```python
def create_logging_middleware(prefix: str):
    """创建带前缀的日志中间件"""
    def middleware(state: RAGState) -> dict:
        print(f"{prefix} - State: {state.get('query', 'N/A')}")
        return {}
    return middleware

# 使用
builder.add_node("log_retrieve", create_logging_middleware("🔍 RETRIEVE"))
builder.add_node("log_generate", create_logging_middleware("💡 GENERATE"))
```

### 6.2 条件中间件

根据状态决定是否执行中间件：

```python
def conditional_logging_middleware(state: RAGState) -> dict:
    """只在 debug 模式下记录日志"""
    if state.get("metadata", {}).get("debug", False):
        print(f"[DEBUG] State: {state}")
    return {}
```

### 6.3 中间件链式组合

将多个中间件组合成一个节点：

```python
def combined_middleware(state: RAGState) -> dict:
    """组合多个中间件"""
    result = {}

    # 执行日志中间件
    log_result = logging_middleware(state)
    result.update(log_result)

    # 执行性能监控中间件
    perf_result = performance_middleware(state)
    result.update(perf_result)

    return result
```

## 7. 常见问题

### Q1: 中间件是否会影响性能？

**A:** 会有轻微影响，但通常可以忽略。如果性能敏感，可以：
- 使用条件中间件（只在需要时执行）
- 异步执行中间件（如果支持）
- 将多个中间件合并为一个

### Q2: 如何跳过某些中间件？

**A:** 使用条件边或条件中间件：

```python
def conditional_validation(state: RAGState) -> dict:
    if state.get("skip_validation", False):
        return {}
    return validation_middleware(state)
```

### Q3: 中间件可以修改状态吗？

**A:** 可以，但不推荐修改业务数据。推荐做法：
- 只修改 `metadata` 字段
- 如果需要修改业务数据，明确标注为"转换中间件"

### Q4: 如何处理中间件中的错误？

**A:** 在中间件中捕获异常，设置 `error` 字段：

```python
def safe_middleware(state: RAGState) -> dict:
    try:
        # 中间件逻辑
        return {}
    except Exception as e:
        return {"error": f"Middleware error: {str(e)}"}
```

## 8. 最佳实践

### 8.1 中间件设计原则

1. **职责单一**：每个中间件只处理一个关注点
2. **不侵入业务**：不修改业务逻辑
3. **可配置**：使用工厂函数创建可配置的中间件
4. **可复用**：设计通用的中间件，可在多个工作流中使用

### 8.2 中间件命名规范

- 使用描述性名称：`logging_middleware`, `validation_middleware`
- 节点名称包含位置信息：`log_start`, `log_after_retrieve`
- 使用前缀区分类型：`log_*`, `validate_*`, `perf_*`

### 8.3 中间件组织方式

```python
# 推荐：按功能组织中间件
middlewares/
  ├── logging.py      # 日志中间件
  ├── validation.py   # 验证中间件
  ├── performance.py  # 性能监控中间件
  └── error.py        # 错误处理中间件
```
