# 核心概念8：编码与错误处理

> 掌握文档加载中的编码检测和健壮的错误处理策略

---

## 为什么编码与错误处理重要？

文档加载过程中最常见的两类问题：
1. **编码问题**：文件编码不正确导致乱码
2. **加载失败**：文件损坏、格式不支持、权限问题

**在RAG系统中，健壮的错误处理确保系统稳定性和数据质量。**

### 1.2 使用chardet检测编码

```python
"""
使用chardet自动检测文件编码
"""

import chardet

def detect_encoding(file_path: str) -> dict:
    """检测文件编码"""
    with open(file_path, 'rb') as f:
        raw_data = f.read()

    # 检测编码
    result = chardet.detect(raw_data)

    return {
        "encoding": result['encoding'],
        "confidence": result['confidence'],
        "language": result.get('language', 'unknown')
    }

# 使用示例
result = detect_encoding("document.txt")
print(f"检测到编码: {result['encoding']}")
print(f"置信度: {result['confidence']:.2%}")

# 使用检测到的编码读取文件
with open("document.txt", 'r', encoding=result['encoding']) as f:
    content = f.read()
    print(content[:100])
```

### 1.3 多编码尝试策略

```python
"""
多编码尝试策略
按优先级尝试不同编码
"""

from typing import Optional

def read_file_with_fallback(file_path: str) -> tuple[str, str]:
    """
    使用回退策略读取文件
    返回: (content, encoding)
    """
    # 编码优先级列表
    encodings = ['utf-8', 'gbk', 'gb2312', 'big5', 'iso-8859-1', 'ascii']

    for encoding in encodings:
        try:
            with open(file_path, 'r', encoding=encoding) as f:
                content = f.read()
                print(f"✅ 成功使用编码: {encoding}")
                return content, encoding
        except UnicodeDecodeError:
            print(f"❌ {encoding} 解码失败，尝试下一个...")
            continue
        except Exception as e:
            print(f"❌ {encoding} 读取失败: {e}")
            continue

    # 所有编码都失败，使用errors='ignore'
    print("⚠️  所有编码都失败，使用UTF-8并忽略错误")
    with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
        content = f.read()
        return content, 'utf-8 (with errors ignored)'

# 使用示例
content, encoding = read_file_with_fallback("document.txt")
print(f"\n使用编码: {encoding}")
print(f"内容长度: {len(content)} 字符")
```

### 1.4 智能编码检测与读取

```python
"""
结合chardet和回退策略的智能读取
"""

import chardet
from typing import Tuple

class SmartFileReader:
    """智能文件读取器"""

    def __init__(self, confidence_threshold: float = 0.7):
        self.confidence_threshold = confidence_threshold
        self.fallback_encodings = ['utf-8', 'gbk', 'gb2312', 'iso-8859-1']

    def read(self, file_path: str) -> Tuple[str, str]:
        """
        智能读取文件
        返回: (content, encoding)
        """
        # 1. 检测编码
        detected = self._detect_encoding(file_path)

        # 2. 如果置信度高，直接使用检测到的编码
        if detected['confidence'] >= self.confidence_threshold:
            try:
                content = self._read_with_encoding(file_path, detected['encoding'])
                print(f"✅ 使用检测到的编码: {detected['encoding']} (置信度: {detected['confidence']:.2%})")
                return content, detected['encoding']
            except Exception as e:
                print(f"⚠️  检测到的编码失败: {e}")

        # 3. 置信度低或失败，使用回退策略
        print("使用回退策略...")
        return self._read_with_fallback(file_path)

    def _detect_encoding(self, file_path: str) -> dict:
        """检测文件编码"""
        with open(file_path, 'rb') as f:
            raw_data = f.read()
        result = chardet.detect(raw_data)
        return {
            "encoding": result['encoding'],
            "confidence": result['confidence']
        }

    def _read_with_encoding(self, file_path: str, encoding: str) -> str:
        """使用指定编码读取"""
        with open(file_path, 'r', encoding=encoding) as f:
            return f.read()

    def _read_with_fallback(self, file_path: str) -> Tuple[str, str]:
        """回退策略读取"""
        for encoding in self.fallback_encodings:
            try:
                content = self._read_with_encoding(file_path, encoding)
                print(f"✅ 回退成功: {encoding}")
                return content, encoding
            except:
                continue

        # 最后的手段：忽略错误
        print("⚠️  使用UTF-8并忽略错误")
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            return f.read(), 'utf-8 (errors ignored)'

# 使用示例
reader = SmartFileReader(confidence_threshold=0.7)
content, encoding = reader.read("document.txt")
print(f"\n最终使用编码: {encoding}")
```

## 2. 错误处理策略

### 2.1 错误分类

```python
"""
文档加载错误分类
"""

from enum import Enum

class LoadErrorType(Enum):
    """加载错误类型"""
    FILE_NOT_FOUND = "文件不存在"
    PERMISSION_DENIED = "权限不足"
    ENCODING_ERROR = "编码错误"
    FORMAT_ERROR = "格式错误"
    CORRUPTED_FILE = "文件损坏"
    TIMEOUT = "超时"
    NETWORK_ERROR = "网络错误"
    UNKNOWN = "未知错误"

class LoadError(Exception):
    """加载错误基类"""

    def __init__(self, error_type: LoadErrorType, message: str, file_path: str = None):
        self.error_type = error_type
        self.message = message
        self.file_path = file_path
        super().__init__(f"{error_type.value}: {message}")

# 使用示例
try:
    # 模拟加载失败
    raise LoadError(
        LoadErrorType.ENCODING_ERROR,
        "无法使用UTF-8解码",
        "document.txt"
    )
except LoadError as e:
    print(f"错误类型: {e.error_type.value}")
    print(f"错误信息: {e.message}")
    print(f"文件路径: {e.file_path}")
```

### 2.2 健壮的文档加载器

```python
"""
健壮的文档加载器
包含完整的错误处理
"""

from typing import List, Optional
from langchain.schema import Document
import os
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class RobustDocumentLoader:
    """健壮的文档加载器"""

    def __init__(self, max_retries: int = 3, timeout: int = 30):
        self.max_retries = max_retries
        self.timeout = timeout
        self.failed_files = []

    def load_batch(self, file_paths: List[str]) -> tuple[List[Document], List[dict]]:
        """
        批量加载文档
        返回: (成功的文档, 失败的文件信息)
        """
        successful_docs = []

        for file_path in file_paths:
            try:
                doc = self._load_single_file(file_path)
                if doc:
                    successful_docs.append(doc)
                    logger.info(f"✅ 成功加载: {file_path}")
            except Exception as e:
                self._handle_error(file_path, e)

        return successful_docs, self.failed_files

    def _load_single_file(self, file_path: str) -> Optional[Document]:
        """加载单个文件（带重试）"""
        for attempt in range(self.max_retries):
            try:
                return self._load_file(file_path)
            except Exception as e:
                if attempt < self.max_retries - 1:
                    logger.warning(f"尝试 {attempt + 1}/{self.max_retries} 失败: {file_path}")
                    continue
                else:
                    raise

    def _load_file(self, file_path: str) -> Document:
        """实际的文件加载逻辑"""
        # 1. 检查文件是否存在
        if not os.path.exists(file_path):
            raise LoadError(LoadErrorType.FILE_NOT_FOUND, f"文件不存在", file_path)

        # 2. 检查文件权限
        if not os.access(file_path, os.R_OK):
            raise LoadError(LoadErrorType.PERMISSION_DENIED, f"无读取权限", file_path)

        # 3. 检查文件大小
        file_size = os.path.getsize(file_path)
        if file_size == 0:
            raise LoadError(LoadErrorType.CORRUPTED_FILE, f"文件为空", file_path)

        # 4. 智能读取文件
        reader = SmartFileReader()
        try:
            content, encoding = reader.read(file_path)
        except Exception as e:
            raise LoadError(LoadErrorType.ENCODING_ERROR, str(e), file_path)

        # 5. 验证内容质量
        if len(content.strip()) < 10:
            raise LoadError(LoadErrorType.CORRUPTED_FILE, f"内容过短", file_path)

        # 6. 创建Document
        return Document(
            page_content=content,
            metadata={
                "source": file_path,
                "encoding": encoding,
                "file_size": file_size
            }
        )

    def _handle_error(self, file_path: str, error: Exception):
        """处理错误"""
        error_info = {
            "file_path": file_path,
            "error_type": type(error).__name__,
            "error_message": str(error),
            "timestamp": datetime.now().isoformat()
        }

        self.failed_files.append(error_info)
        logger.error(f"❌ 加载失败: {file_path} - {error}")

    def get_failure_report(self) -> str:
        """生成失败报告"""
        if not self.failed_files:
            return "所有文件加载成功"

        report = f"失败文件数: {len(self.failed_files)}\n\n"
        for failure in self.failed_files:
            report += f"文件: {failure['file_path']}\n"
            report += f"错误类型: {failure['error_type']}\n"
            report += f"错误信息: {failure['error_message']}\n"
            report += f"时间: {failure['timestamp']}\n\n"

        return report

# 使用示例
loader = RobustDocumentLoader(max_retries=3)

file_paths = [
    "doc1.txt",
    "doc2.txt",
    "corrupted.txt",  # 假设这个文件损坏
    "missing.txt"     # 假设这个文件不存在
]

docs, failures = loader.load_batch(file_paths)

print(f"\n成功加载: {len(docs)} 个文档")
print(f"失败: {len(failures)} 个文件")
print("\n失败报告:")
print(loader.get_failure_report())
```

### 2.3 降级策略

```python
"""
降级策略：当高级解析器失败时，降级到简单解析器
"""

from typing import List, Callable
from langchain.schema import Document

class FallbackLoader:
    """支持降级的文档加载器"""

    def __init__(self):
        # 解析器优先级列表（从高到低）
        self.parsers = [
            ("高级解析器", self._advanced_parser),
            ("标准解析器", self._standard_parser),
            ("简单解析器", self._simple_parser),
            ("最后手段", self._last_resort_parser)
        ]

    def load(self, file_path: str) -> Document:
        """使用降级策略加载文档"""
        last_error = None

        for parser_name, parser_func in self.parsers:
            try:
                logger.info(f"尝试 {parser_name}...")
                doc = parser_func(file_path)
                logger.info(f"✅ {parser_name} 成功")
                return doc
            except Exception as e:
                logger.warning(f"❌ {parser_name} 失败: {e}")
                last_error = e
                continue

        # 所有解析器都失败
        raise Exception(f"所有解析器都失败: {last_error}")

    def _advanced_parser(self, file_path: str) -> Document:
        """高级解析器（可能失败）"""
        # 模拟高级解析逻辑
        reader = SmartFileReader(confidence_threshold=0.9)
        content, encoding = reader.read(file_path)

        # 高级处理：清洗、格式化等
        content = self._advanced_processing(content)

        return Document(
            page_content=content,
            metadata={"source": file_path, "parser": "advanced", "encoding": encoding}
        )

    def _standard_parser(self, file_path: str) -> Document:
        """标准解析器"""
        reader = SmartFileReader(confidence_threshold=0.7)
        content, encoding = reader.read(file_path)

        return Document(
            page_content=content,
            metadata={"source": file_path, "parser": "standard", "encoding": encoding}
        )

    def _simple_parser(self, file_path: str) -> Document:
        """简单解析器"""
        # 直接使用UTF-8，忽略错误
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            content = f.read()

        return Document(
            page_content=content,
            metadata={"source": file_path, "parser": "simple", "encoding": "utf-8"}
        )

    def _last_resort_parser(self, file_path: str) -> Document:
        """最后手段：读取二进制并尝试解码"""
        with open(file_path, 'rb') as f:
            raw_data = f.read()

        # 尝试多种解码方式
        for encoding in ['utf-8', 'gbk', 'iso-8859-1']:
            try:
                content = raw_data.decode(encoding, errors='ignore')
                if len(content.strip()) > 0:
                    return Document(
                        page_content=content,
                        metadata={
                            "source": file_path,
                            "parser": "last_resort",
                            "encoding": encoding
                        }
                    )
            except:
                continue

        raise Exception("无法解码文件")

    def _advanced_processing(self, content: str) -> str:
        """高级内容处理"""
        # 清理多余空白
        import re
        content = re.sub(r'\n{3,}', '\n\n', content)
        content = re.sub(r' {2,}', ' ', content)
        return content.strip()

# 使用示例
loader = FallbackLoader()
doc = loader.load("document.txt")
print(f"使用的解析器: {doc.metadata['parser']}")
```

## 3. 在RAG中的应用

### 3.1 构建健壮的知识库

```python
"""
构建健壮的RAG知识库
包含完整的错误处理和编码处理
"""

from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain.vectorstores import Chroma
from langchain.embeddings import OpenAIEmbeddings
from datetime import datetime

def build_robust_knowledge_base(
    file_paths: List[str],
    output_dir: str = "./robust_kb"
):
    """构建健壮的知识库"""

    # 1. 加载文档（带错误处理）
    print("=== 1. 加载文档 ===")
    loader = RobustDocumentLoader(max_retries=3)
    documents, failures = loader.load_batch(file_paths)

    print(f"成功加载: {len(documents)} 个文档")
    print(f"失败: {len(failures)} 个文件")

    if not documents:
        print("❌ 没有成功加载任何文档")
        return

    # 2. 分块
    print("\n=== 2. 文本分块 ===")
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=1000,
        chunk_overlap=200
    )
    chunks = splitter.split_documents(documents)
    print(f"分块后得到: {len(chunks)} 个文本块")

    # 3. 向量化并存储
    print("\n=== 3. 向量化并存储 ===")
    try:
        embeddings = OpenAIEmbeddings()
        vectorstore = Chroma.from_documents(
            chunks,
            embeddings,
            persist_directory=output_dir
        )
        print(f"✅ 知识库构建成功")
    except Exception as e:
        print(f"❌ 向量化失败: {e}")
        return

    # 4. 生成报告
    print("\n=== 4. 生成报告 ===")
    report = {
        "timestamp": datetime.now().isoformat(),
        "total_files": len(file_paths),
        "successful_files": len(documents),
        "failed_files": len(failures),
        "total_chunks": len(chunks),
        "failures": failures
    }

    # 保存报告
    import json
    with open(f"{output_dir}/build_report.json", 'w', encoding='utf-8') as f:
        json.dump(report, f, ensure_ascii=False, indent=2)

    print(f"报告已保存到: {output_dir}/build_report.json")

    # 5. 显示失败文件
    if failures:
        print("\n=== 失败文件列表 ===")
        print(loader.get_failure_report())

# 使用示例
file_paths = [
    "docs/report1.txt",
    "docs/report2.txt",
    "docs/report3.txt"
]

build_robust_knowledge_base(file_paths)
```

### 3.2 监控和告警

```python
"""
文档加载监控和告警
"""

class LoadingMonitor:
    """加载监控器"""

    def __init__(self, failure_threshold: float = 0.2):
        self.failure_threshold = failure_threshold
        self.stats = {
            "total": 0,
            "success": 0,
            "failure": 0,
            "encoding_errors": 0,
            "format_errors": 0,
            "other_errors": 0
        }

    def record_success(self):
        """记录成功"""
        self.stats["total"] += 1
        self.stats["success"] += 1

    def record_failure(self, error_type: LoadErrorType):
        """记录失败"""
        self.stats["total"] += 1
        self.stats["failure"] += 1

        if error_type == LoadErrorType.ENCODING_ERROR:
            self.stats["encoding_errors"] += 1
        elif error_type == LoadErrorType.FORMAT_ERROR:
            self.stats["format_errors"] += 1
        else:
            self.stats["other_errors"] += 1

    def get_failure_rate(self) -> float:
        """获取失败率"""
        if self.stats["total"] == 0:
            return 0.0
        return self.stats["failure"] / self.stats["total"]

    def should_alert(self) -> bool:
        """是否应该告警"""
        return self.get_failure_rate() > self.failure_threshold

    def get_report(self) -> str:
        """生成报告"""
        failure_rate = self.get_failure_rate()

        report = f"""
=== 加载监控报告 ===
总文件数: {self.stats['total']}
成功: {self.stats['success']} ({self.stats['success']/self.stats['total']*100:.1f}%)
失败: {self.stats['failure']} ({failure_rate*100:.1f}%)

失败类型分布:
- 编码错误: {self.stats['encoding_errors']}
- 格式错误: {self.stats['format_errors']}
- 其他错误: {self.stats['other_errors']}

告警状态: {'🚨 需要关注' if self.should_alert() else '✅ 正常'}
"""
        return report

# 使用示例
monitor = LoadingMonitor(failure_threshold=0.2)

# 模拟加载过程
for i in range(100):
    if i % 10 == 0:  # 10%失败率
        monitor.record_failure(LoadErrorType.ENCODING_ERROR)
    else:
        monitor.record_success()

print(monitor.get_report())
```

## 总结

**编码与错误处理的核心要点：**

1. **编码检测**: 使用chardet自动检测，配合回退策略
2. **错误分类**: 明确区分不同类型的错误
3. **降级策略**: 高级解析器失败时降级到简单解析器
4. **重试机制**: 对临时性错误进行重试
5. **监控告警**: 跟踪失败率，及时发现问题

**在RAG中的最佳实践:**
- 批量加载时不要因个别文件失败而中断
- 记录详细的错误信息便于后续分析
- 提供失败文件的报告和统计
- 设置失败率阈值进行告警
