# Agent推理与决策 - 实战代码：推理质量评估

> 用 LangSmith 和自定义评估框架量化 Agent 推理质量，对比不同推理策略的效果

---

## 场景说明

**目标：** 构建一个完整的 Agent 推理质量评估系统，对比 ReAct 和 Tool Calling 两种推理策略的效果。

**你将学到：**
1. 如何定义推理质量评估指标（推理步数、工具调用准确率、最终答案正确性）
2. 如何使用 LangSmith 追踪 Agent 执行轨迹
3. 如何实现自定义评估器（Trajectory Evaluator）
4. 如何进行 A/B 测试对比不同推理策略
5. 如何可视化评估结果

**核心概念：**
- **Trajectory Evaluation**：评估 Agent 的推理路径是否合理
- **Tool Call Accuracy**：工具选择和参数生成的准确率
- **Answer Correctness**：最终答案的正确性和完整性
- **Efficiency Metrics**：推理步数、Token 消耗、延迟

```python
Agent 推理质量评估实战
演示：使用 LangSmith 和自定义评估器量化 Agent 推理质量

import json
from typing import List, Dict, Any, Tuple
from dataclasses import dataclass

# 确保 LangSmith 追踪已启用
os.environ["LANGSMITH_TRACING"] = "true"
os.environ["LANGSMITH_PROJECT"] = "agent-reasoning-eval"

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)


# ===== 1. 定义测试工具 =====

@tool
def search_product(query: str) -> str:
    """搜索产品信息数据库。
    输入：产品名称或关键词。
    返回：产品的价格、评分、库存等信息。"""

    products = {
        "macbook pro": "MacBook Pro 14寸：价格 14999 元，评分 4.8/5，库存 50 台",
        "iphone": "iPhone 16 Pro：价格 8999 元，评分 4.7/5，库存 200 台",
        "ipad": "iPad Air：价格 4799 元，评分 4.6/5，库存 150 台",
        "airpods": "AirPods Pro 2：价格 1899 元，评分 4.5/5，库存 300 台",
    }
    for key, value in products.items():
        if key in query.lower():
            return value
    return f"未找到 '{query}' 的产品信息"


@tool
def calculate_discount(expression: str) -> str:
    """计算折扣价格。
    输入：数学表达式，如 '14999 * 0.85' 表示 85 折。
    返回：计算结果。"""

    try:
        result = eval(expression, {"__builtins__": {}})
        return f"计算结果：{result}"
    except Exception as e:
        return f"计算错误：{e}"


@tool
def check_user_budget(user_id: str) -> str:
    """查询用户预算信息。
    输入：用户ID。
    返回：用户的预算范围和偏好。"""

    users = {
        "user_001": "预算：10000-15000 元，偏好：高性能笔记本，用途：编程开发",
        "user_002": "预算：5000-8000 元，偏好：性价比手机，用途：日常使用",
        "user_003": "预算：3000-5000 元，偏好：平板电脑，用途：学习娱乐",
    }
    return users.get(user_id, f"未找到用户 {user_id} 的信息")


tools = [search_product, calculate_discount, check_user_budget]


# ===== 2. 定义评估数据集 =====

@dataclass
class TestCase:
    """测试用例"""
    question: str
    expected_tools: List[str]  # 期望调用的工具列表
    expected_answer_keywords: List[str]  # 期望答案中包含的关键词
    max_steps: int  # 期望的最大推理步数


test_cases = [
    TestCase(
        question="用户 user_001 想买笔记本电脑，预算内有什么推荐？如果打 85 折呢？",
        expected_tools=["check_user_budget", "search_product", "calculate_discount"],
        expected_answer_keywords=["MacBook Pro", "14999", "12749", "预算"],
        max_steps=4,
    ),
    TestCase(
        question="用户 user_002 的预算是多少？",
        expected_tools=["check_user_budget"],
        expected_answer_keywords=["5000", "8000", "手机"],
        max_steps=2,
    ),
    TestCase(
        question="iPhone 16 Pro 打 9 折是多少钱？",
        expected_tools=["search_product", "calculate_discount"],
        expected_answer_keywords=["8999", "8099"],
        max_steps=3,
    ),
]


# ===== 3. 定义评估指标 =====

@dataclass
class EvaluationMetrics:
    """评估指标"""
    # 推理步数
    num_steps: int
    # 工具调用准确率（调用的工具是否符合预期）
    tool_accuracy: float
    # 最终答案正确性（关键词覆盖率）
    answer_correctness: float
    # 是否超过最大步数
    exceeded_max_steps: bool
    # 详细信息
    details: Dict[str, Any]


def evaluate_agent_run(
    test_case: TestCase,
    result: Dict[str, Any],
) -> EvaluationMetrics:
    """评估单次 Agent 运行"""

    intermediate_steps = result.get("intermediate_steps", [])
    output = result.get("output", "")

    # 1. 推理步数
    num_steps = len(intermediate_steps)
    exceeded_max_steps = num_steps > test_case.max_steps

    # 2. 工具调用准确率
    called_tools = [action.tool for action, _ in intermediate_steps]
    expected_tools_set = set(test_case.expected_tools)
    called_tools_set = set(called_tools)

    # 计算准确率：正确调用的工具数 / 期望调用的工具数
    correct_tools = expected_tools_set & called_tools_set
    tool_accuracy = len(correct_tools) / len(expected_tools_set) if expected_tools_set else 1.0

    # 3. 最终答案正确性（关键词覆盖率）
    matched_keywords = [kw for kw in test_case.expected_answer_keywords if kw in output]
    answer_correctness = len(matched_keywords) / len(test_case.expected_answer_keywords)

    # 4. 详细信息
    details = {
        "called_tools": called_tools,
        "expected_tools": test_case.expected_tools,
        "correct_tools": list(correct_tools),
        "missing_tools": list(expected_tools_set - called_tools_set),
        "extra_tools": list(called_tools_set - expected_tools_set),
        "matched_keywords": matched_keywords,
        "missing_keywords": [kw for kw in test_case.expected_answer_keywords if kw not in output],
        "output": output[:200],  # 只保留前200字符
    }

    return EvaluationMetrics(
        num_steps=num_steps,
        tool_accuracy=tool_accuracy,
        answer_correctness=answer_correctness,
        exceeded_max_steps=exceeded_max_steps,
        details=details,
    )


# ===== 4. 创建两种 Agent 进行对比 =====

# Agent 1：ReAct 推理模式
react_template = """Answer the following questions as best you can. You have access to the following tools:

{tools}

Use the following format:

Question: the input question you must answer
Thought: you should always think about what to do
Action: the action to take, should be one of [{tool_names}]
Action Input: the input to the action
Observation: the result of the action
... (this Thought/Action/Action Input/Observation can repeat N times)
Thought: I now know the final answer
Final Answer: the final answer to the original input question

Begin!

Question: {input}
Thought:{agent_scratchpad}"""

react_prompt = PromptTemplate.from_template(react_template)
react_agent = create_react_agent(llm=llm, tools=tools, prompt=react_prompt)
react_executor = AgentExecutor(
    agent=react_agent,
    tools=tools,
    verbose=False,  # 关闭详细输出，避免干扰评估
    max_iterations=6,
    handle_parsing_errors=True,
    return_intermediate_steps=True,
)

# Agent 2：Tool Calling 模式（现代方式）
tool_calling_prompt = PromptTemplate.from_template(
    "You are a helpful assistant. Answer the user's question: {input}"
)
tool_calling_agent = create_tool_calling_agent(llm=llm, tools=tools, prompt=tool_calling_prompt)
tool_calling_executor = AgentExecutor(
    agent=tool_calling_agent,
    tools=tools,
    verbose=False,
    max_iterations=6,
    handle_parsing_errors=True,
    return_intermediate_steps=True,
)


# ===== 5. 运行评估 =====

def run_evaluation(
    agent_name: str,
    executor: AgentExecutor,
    test_cases: List[TestCase],
) -> List[EvaluationMetrics]:
    """运行完整评估"""

    print(f"\n{'=' * 60}")
    print(f"评估 {agent_name}")
    print(f"{'=' * 60}")

    all_metrics = []

    for i, test_case in enumerate(test_cases):
        print(f"\n测试用例 {i + 1}/{len(test_cases)}: {test_case.question[:50]}...")

        # 运行 Agent
        result = executor.invoke({"input": test_case.question})

        # 评估结果
        metrics = evaluate_agent_run(test_case, result)
        all_metrics.append(metrics)

        # 打印评估结果
        print(f"  推理步数: {metrics.num_steps} (期望 ≤ {test_case.max_steps})")
        print(f"  工具准确率: {metrics.tool_accuracy:.1%}")
        print(f"  答案正确性: {metrics.answer_correctness:.1%}")
        print(f"  超过最大步数: {'是' if metrics.exceeded_max_steps else '否'}")

    return all_metrics


# 运行两种 Agent 的评估
react_metrics = run_evaluation("ReAct Agent", react_executor, test_cases)
tool_calling_metrics = run_evaluation("Tool Calling Agent", tool_calling_executor, test_cases)


# ===== 6. 汇总对比分析 =====

def aggregate_metrics(metrics_list: List[EvaluationMetrics]) -> Dict[str, float]:
    """汇总评估指标"""

    return {
        "avg_steps": sum(m.num_steps for m in metrics_list) / len(metrics_list),
        "avg_tool_accuracy": sum(m.tool_accuracy for m in metrics_list) / len(metrics_list),
        "avg_answer_correctness": sum(m.answer_correctness for m in metrics_list) / len(metrics_list),
        "exceeded_rate": sum(m.exceeded_max_steps for m in metrics_list) / len(metrics_list),
    }


react_agg = aggregate_metrics(react_metrics)
tool_calling_agg = aggregate_metrics(tool_calling_metrics)

print(f"\n{'=' * 60}")
print("汇总对比分析")
print(f"{'=' * 60}")

print(f"""
```

## 关键知识点解析

### 1. 评估指标设计

```python
# 好的评估指标应该：
# 1. 可量化（数值化）
# 2. 可对比（不同策略之间）
# 3. 可解释（能说明为什么好或不好）

@dataclass
class EvaluationMetrics:
    # ✅ 可量化：推理步数是整数
    num_steps: int

    # ✅ 可量化：准确率是 0-1 的浮点数
    tool_accuracy: float
    answer_correctness: float

    # ✅ 可对比：布尔值可以统计比例
    exceeded_max_steps: bool

    # ✅ 可解释：详细信息帮助分析原因
    details: Dict[str, Any]
```

### 2. 工具调用准确率计算

```python
# 使用集合运算计算准确率
expected_tools_set = {"check_user_budget", "search_product", "calculate_discount"}
called_tools_set = {"check_user_budget", "search_product"}

# 正确调用的工具（交集）
correct_tools = expected_tools_set & called_tools_set
# → {"check_user_budget", "search_product"}

# 缺失的工具（差集）
missing_tools = expected_tools_set - called_tools_set
# → {"calculate_discount"}

# 多余的工具（差集）
extra_tools = called_tools_set - expected_tools_set
# → set()

# 准确率 = 正确数 / 期望数
tool_accuracy = len(correct_tools) / len(expected_tools_set)
# → 2 / 3 = 0.67
```

### 3. LangSmith 自动追踪

```python
# 只需设置环境变量，LangSmith 会自动追踪所有 LLM 调用
```

### 4. A/B 测试的关键

```python

# - 相同的测试用例
# - 相同的工具集
# - 相同的 LLM 模型
# - 只改变推理策略（ReAct vs Tool Calling）

# ✅ 好的对比（只改变推理策略）
```

## 扩展：使用 LangSmith 的内置评估器

```python
"""
LangSmith 提供了更强大的评估功能，可以直接在平台上运行评估
"""

from langsmith import Client
from langsmith.evaluation import evaluate

client = Client()

# 定义评估数据集
dataset_name = "agent-reasoning-test"
client.create_dataset(dataset_name)

# 添加测试用例到数据集
for test_case in test_cases:
    client.create_example(
        dataset_name=dataset_name,
        inputs={"input": test_case.question},
        outputs={"expected_tools": test_case.expected_tools},
    )

# 定义评估器
def tool_accuracy_evaluator(run, example):
    """评估工具调用准确率"""
    expected_tools = set(example.outputs["expected_tools"])
    # 从 run 中提取实际调用的工具
    called_tools = set()  # 需要从 run.outputs 中解析

    correct = expected_tools & called_tools
    return {
        "key": "tool_accuracy",
        "score": len(correct) / len(expected_tools),
    }

# 运行评估
results = evaluate(
    lambda inputs: react_executor.invoke(inputs),
    data=dataset_name,
    evaluators=[tool_accuracy_evaluator],
    experiment_prefix="react-agent",
)

# 查看结果
print(results)
```

## 小结

推理质量评估的核心要点：

1. **定义清晰的评估指标**：推理步数、工具准确率、答案正确性、效率指标
2. **构建测试数据集**：包含期望的工具调用和答案关键词
3. **A/B 测试对比**：控制变量，只改变推理策略
4. **使用 LangSmith 追踪**：自动记录所有执行轨迹，方便分析
5. **可视化结果**：用图表直观展示对比结果

评估不是一次性的，而是持续的过程——每次优化 Prompt、调整工具、改变策略后，都应该重新评估，确保改进是有效的。
