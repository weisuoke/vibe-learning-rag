# OpenClaw 安装与配置 - 实战代码 Part 1: 基础安装

**目标：** 提供完整、可运行的 TypeScript 自动化安装脚本

本文档包含完整的 OpenClaw 自动化安装代码，可直接在项目中使用。

---

## 完整安装自动化脚本

```typescript
/**
 * OpenClaw 自动化安装脚本
 *
 * 功能：
 * 1. 检测 Node.js 版本
 * 2. 检测包管理器（npm/pnpm）
 * 3. 自动安装 OpenClaw
 * 4. 验证安装结果
 * 5. 生成安装报告
 *
 * 使用方法：
 *   ts-node install-openclaw.ts
 *   或
 *   npx tsx install-openclaw.ts
 */

import { execSync, spawn } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

// ===== 类型定义 =====

interface InstallationResult {
  success: boolean;
  version?: string;
  errors: string[];
  warnings: string[];
  duration: number;
}

interface EnvironmentInfo {
  platform: string;
  nodeVersion: string;
  nodeMajor: number;
  npmVersion?: string;
  pnpmVersion?: string;
  hasNvm: boolean;
  hasAsdf: boolean;
  hasVolta: boolean;
}

// ===== 工具函数 =====

/**
 * 执行命令并返回输出
 */
function execCommand(command: string, options: { silent?: boolean } = {}): string {
  try {
    const output = execSync(command, {
      encoding: 'utf-8',
      stdio: options.silent ? 'pipe' : 'inherit'
    });
    return output.trim();
  } catch (error: any) {
    if (!options.silent) {
      console.error(`命令执行失败: ${command}`);
      console.error(error.message);
    }
    throw error;
  }
}

/**
 * 检查命令是否存在
 */
function commandExists(command: string): boolean {
  try {
    execSync(`command -v ${command}`, { stdio: 'ignore' });
    return true;
  } catch {
    return false;
  }
}

/**
 * 获取版本号
 */
function getVersion(command: string): string | null {
  try {
    const output = execSync(`${command} --version`, {
      encoding: 'utf-8',
      stdio: 'pipe'
    });
    return output.trim();
  } catch {
    return null;
  }
}

// ===== 环境检测 =====

/**
 * 检测运行环境
 */
function detectEnvironment(): EnvironmentInfo {
  console.log('🔍 检测运行环境...\n');

  const platform = os.platform();
  const nodeVersion = process.version;
  const nodeMajor = parseInt(nodeVersion.slice(1).split('.')[0]);

  const npmVersion = getVersion('npm');
  const pnpmVersion = getVersion('pnpm');

  const hasNvm = commandExists('nvm');
  const hasAsdf = commandExists('asdf');
  const hasVolta = commandExists('volta');

  const env: EnvironmentInfo = {
    platform,
    nodeVersion,
    nodeMajor,
    npmVersion: npmVersion || undefined,
    pnpmVersion: pnpmVersion || undefined,
    hasNvm,
    hasAsdf,
    hasVolta
  };

  // 打印环境信息
  console.log('环境信息:');
  console.log(`  平台: ${platform}`);
  console.log(`  Node.js: ${nodeVersion} ${nodeMajor >= 22 ? '✅' : '❌'}`);
  if (npmVersion) console.log(`  npm: ${npmVersion}`);
  if (pnpmVersion) console.log(`  pnpm: ${pnpmVersion}`);
  console.log('');

  console.log('版本管理工具:');
  console.log(`  nvm: ${hasNvm ? '✅' : '❌'}`);
  console.log(`  asdf: ${hasAsdf ? '✅' : '❌'}`);
  console.log(`  volta: ${hasVolta ? '✅' : '❌'}`);
  console.log('');

  return env;
}

/**
 * 验证 Node.js 版本
 */
function validateNodeVersion(env: EnvironmentInfo): boolean {
  if (env.nodeMajor >= 22) {
    console.log('✅ Node.js 版本满足要求\n');
    return true;
  }

  console.error('❌ Node.js 版本过低');
  console.error(`   当前版本: ${env.nodeVersion}`);
  console.error(`   需要版本: 22.0.0+\n`);

  console.log('升级方法:');
  if (env.hasNvm) {
    console.log('  nvm install 22');
    console.log('  nvm use 22');
  } else if (env.hasAsdf) {
    console.log('  asdf install nodejs 22.18.0');
    console.log('  asdf global nodejs 22.18.0');
  } else if (env.hasVolta) {
    console.log('  volta install node@22');
  } else {
    console.log('  访问 https://nodejs.org 下载 Node.js 22+');
  }
  console.log('');

  return false;
}

// ===== 包管理器检测 =====

/**
 * 选择包管理器
 */
function selectPackageManager(env: EnvironmentInfo): 'npm' | 'pnpm' {
  console.log('📦 选择包管理器...\n');

  // 优先使用 pnpm（如果已安装）
  if (env.pnpmVersion) {
    console.log(`✅ 使用 pnpm (${env.pnpmVersion})\n`);
    return 'pnpm';
  }

  // 回退到 npm
  if (env.npmVersion) {
    console.log(`✅ 使用 npm (${env.npmVersion})\n`);
    return 'npm';
  }

  throw new Error('未找到可用的包管理器（npm 或 pnpm）');
}

// ===== OpenClaw 安装 =====

/**
 * 检查 OpenClaw 是否已安装
 */
function isOpenClawInstalled(): { installed: boolean; version?: string } {
  try {
    const version = execCommand('openclaw --version', { silent: true });
    return { installed: true, version };
  } catch {
    return { installed: false };
  }
}

/**
 * 安装 OpenClaw
 */
function installOpenClaw(packageManager: 'npm' | 'pnpm'): InstallationResult {
  const startTime = Date.now();
  const result: InstallationResult = {
    success: false,
    errors: [],
    warnings: [],
    duration: 0
  };

  console.log('📦 开始安装 OpenClaw...\n');

  // 检查是否已安装
  const existing = isOpenClawInstalled();
  if (existing.installed) {
    console.log(`⚠️  OpenClaw 已安装 (版本: ${existing.version})`);
    console.log('   将覆盖安装最新版本\n');
    result.warnings.push(`已存在版本: ${existing.version}`);
  }

  // 执行安装
  try {
    const installCommand = packageManager === 'pnpm'
      ? 'pnpm add -g openclaw@latest'
      : 'npm install -g openclaw@latest';

    console.log(`执行命令: ${installCommand}\n`);
    execCommand(installCommand);

    // 验证安装
    const installed = isOpenClawInstalled();
    if (!installed.installed) {
      throw new Error('安装完成但无法找到 openclaw 命令');
    }

    result.success = true;
    result.version = installed.version;
    result.duration = Date.now() - startTime;

    console.log('\n✅ OpenClaw 安装成功！');
    console.log(`   版本: ${result.version}`);
    console.log(`   耗时: ${(result.duration / 1000).toFixed(2)}s\n`);

  } catch (error: any) {
    result.success = false;
    result.errors.push(error.message);
    result.duration = Date.now() - startTime;

    console.error('\n❌ OpenClaw 安装失败');
    console.error(`   错误: ${error.message}\n`);
  }

  return result;
}

// ===== 安装验证 =====

/**
 * 验证安装
 */
function verifyInstallation(): { passed: boolean; checks: { [key: string]: boolean } } {
  console.log('🔍 验证安装...\n');

  const checks: { [key: string]: boolean } = {};

  // 1. 检查 openclaw 命令
  try {
    execCommand('openclaw --version', { silent: true });
    checks['openclaw 命令'] = true;
    console.log('✅ openclaw 命令可用');
  } catch {
    checks['openclaw 命令'] = false;
    console.log('❌ openclaw 命令不可用');
  }

  // 2. 检查 help 命令
  try {
    execCommand('openclaw --help', { silent: true });
    checks['help 命令'] = true;
    console.log('✅ help 命令可用');
  } catch {
    checks['help 命令'] = false;
    console.log('❌ help 命令不可用');
  }

  // 3. 检查配置目录
  const configDir = path.join(os.homedir(), '.openclaw');
  if (fs.existsSync(configDir)) {
    checks['配置目录'] = true;
    console.log(`✅ 配置目录存在: ${configDir}`);
  } else {
    checks['配置目录'] = false;
    console.log(`⚠️  配置目录不存在: ${configDir}`);
    console.log('   (首次运行 openclaw onboard 时会自动创建)');
  }

  // 4. 检查 PATH
  const openclawPath = commandExists('openclaw');
  checks['PATH 配置'] = openclawPath;
  if (openclawPath) {
    try {
      const which = execCommand('which openclaw', { silent: true });
      console.log(`✅ openclaw 在 PATH 中: ${which}`);
    } catch {
      console.log('✅ openclaw 在 PATH 中');
    }
  } else {
    console.log('❌ openclaw 不在 PATH 中');
  }

  console.log('');

  const passed = Object.values(checks).every(v => v);
  return { passed, checks };
}

// ===== 安装报告 =====

/**
 * 生成安装报告
 */
function generateReport(
  env: EnvironmentInfo,
  installResult: InstallationResult,
  verifyResult: { passed: boolean; checks: { [key: string]: boolean } }
): void {
  console.log('='.repeat(60));
  console.log('OpenClaw 安装报告');
  console.log('='.repeat(60));
  console.log('');

  // 环境信息
  console.log('环境信息:');
  console.log(`  平台: ${env.platform}`);
  console.log(`  Node.js: ${env.nodeVersion}`);
  console.log(`  包管理器: ${env.pnpmVersion ? 'pnpm' : 'npm'}`);
  console.log('');

  // 安装结果
  console.log('安装结果:');
  console.log(`  状态: ${installResult.success ? '✅ 成功' : '❌ 失败'}`);
  if (installResult.version) {
    console.log(`  版本: ${installResult.version}`);
  }
  console.log(`  耗时: ${(installResult.duration / 1000).toFixed(2)}s`);
  console.log('');

  // 警告
  if (installResult.warnings.length > 0) {
    console.log('警告:');
    installResult.warnings.forEach(warning => {
      console.log(`  ⚠️  ${warning}`);
    });
    console.log('');
  }

  // 错误
  if (installResult.errors.length > 0) {
    console.log('错误:');
    installResult.errors.forEach(error => {
      console.log(`  ❌ ${error}`);
    });
    console.log('');
  }

  // 验证结果
  console.log('验证结果:');
  for (const [check, passed] of Object.entries(verifyResult.checks)) {
    console.log(`  ${passed ? '✅' : '❌'} ${check}`);
  }
  console.log('');

  // 下一步
  if (installResult.success && verifyResult.passed) {
    console.log('✅ 安装完成！');
    console.log('');
    console.log('下一步:');
    console.log('  1. 运行 Onboarding Wizard:');
    console.log('     openclaw onboard');
    console.log('');
    console.log('  2. 启动 Gateway:');
    console.log('     openclaw gateway start');
    console.log('');
    console.log('  3. 查看帮助:');
    console.log('     openclaw --help');
  } else {
    console.log('❌ 安装未完成');
    console.log('');
    console.log('故障排查:');
    console.log('  1. 检查 Node.js 版本: node --version');
    console.log('  2. 检查 PATH 配置: echo $PATH');
    console.log('  3. 手动安装: npm install -g openclaw@latest');
    console.log('  4. 查看日志: npm install -g openclaw@latest --verbose');
  }

  console.log('');
  console.log('='.repeat(60));
}

// ===== 主函数 =====

/**
 * 主安装流程
 */
async function main(): Promise<void> {
  console.log('');
  console.log('='.repeat(60));
  console.log('OpenClaw 自动化安装脚本');
  console.log('='.repeat(60));
  console.log('');

  try {
    // 1. 检测环境
    const env = detectEnvironment();

    // 2. 验证 Node.js 版本
    if (!validateNodeVersion(env)) {
      process.exit(1);
    }

    // 3. 选择包管理器
    const packageManager = selectPackageManager(env);

    // 4. 安装 OpenClaw
    const installResult = installOpenClaw(packageManager);

    // 5. 验证安装
    const verifyResult = verifyInstallation();

    // 6. 生成报告
    generateReport(env, installResult, verifyResult);

    // 7. 退出
    process.exit(installResult.success && verifyResult.passed ? 0 : 1);

  } catch (error: any) {
    console.error('');
    console.error('❌ 安装过程中发生错误:');
    console.error(`   ${error.message}`);
    console.error('');
    process.exit(1);
  }
}

// 运行主函数
if (require.main === module) {
  main();
}

// 导出函数供其他模块使用
export {
  detectEnvironment,
  validateNodeVersion,
  selectPackageManager,
  installOpenClaw,
  verifyInstallation,
  generateReport
};
```

```typescript

  }
}

setupOpenClaw();
```

## 故障排查

### 问题 1：权限错误（EACCES）

**错误信息：**
```
Error: EACCES: permission denied, mkdir '/usr/local/lib/node_modules/openclaw'
```

**解决方案：**
```bash
# 方案 1：修改 npm prefix（推荐）
mkdir -p ~/.npm-global
npm config set prefix '~/.npm-global'
echo 'export PATH=~/.npm-global/bin:$PATH' >> ~/.zshrc
source ~/.zshrc

# 方案 2：使用 sudo（不推荐）
sudo npm install -g openclaw@latest

# 方案 3：使用 nvm（最推荐）
nvm install 22
nvm use 22
npm install -g openclaw@latest  # 无需 sudo
```

### 问题 2：命令未找到

**错误信息：**
```
openclaw: command not found
```

**解决方案：**
```bash
# 1. 检查全局 bin 目录
npm bin -g
# 或
pnpm bin -g

# 2. 检查 PATH
echo $PATH | grep "$(npm bin -g)"

# 3. 添加到 PATH
echo 'export PATH="$(npm bin -g):$PATH"' >> ~/.zshrc
source ~/.zshrc
```

### 问题 3：网络超时

**错误信息：**
```
Error: network timeout at: https://registry.npmjs.org/openclaw
```

**解决方案：**
```bash
# 使用国内镜像
npm config set registry https://registry.npmmirror.com
pnpm config set registry https://registry.npmmirror.com

# 或使用代理
npm config set proxy http://proxy.example.com:8080
npm config set https-proxy http://proxy.example.com:8080
```
