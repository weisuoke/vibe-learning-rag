## 场景描述

### 业务需求

构建一个智能聊天机器人，需要：
1. **维护对话历史**：累积用户和 AI 的所有消息
2. **消息更新**：支持修改已发送的消息（如编辑、重新生成）
3. **消息删除**：支持删除特定消息或清空历史
4. **格式转换**：支持 OpenAI 消息格式

### 技术挑战

- 如何高效管理对话历史？
- 如何按 ID 更新或删除消息？
- 如何处理消息的追加和替换？
- 如何支持多种消息格式？

### 解决方案

使用 LangGraph 的 `add_messages` Reducer：
- **按 ID 合并**：相同 ID 的消息会被替换
- **自动追加**：新消息自动追加到列表末尾
- **删除机制**：支持 `RemoveMessage` 删除特定消息
- **格式化**：支持 OpenAI 格式转换

### 1. 状态定义 (chatbot/state.py)

```python
"""
状态定义模块
使用 add_messages Reducer 管理消息列表
"""
from typing import Annotated, TypedDict
from langchain_core.messages import AnyMessage
from langgraph.graph.message import add_messages


class ChatbotState(TypedDict):
    """
    聊天机器人状态

    Attributes:
        messages: 对话历史，使用 add_messages Reducer
                 - 新消息会追加到列表末尾
                 - 相同 ID 的消息会被替换
                 - 支持 RemoveMessage 删除消息
    """
    messages: Annotated[list[AnyMessage], add_messages]


# 使用 OpenAI 格式的状态定义
class ChatbotStateOpenAI(TypedDict):
    """
    支持 OpenAI 格式的聊天机器人状态

    消息会自动转换为 OpenAI 兼容格式：
    - text blocks
    - image_url blocks
    - tool responses as ToolMessage
    """
    messages: Annotated[list[AnyMessage], add_messages(format="langchain-openai")]
```

### 2. 节点函数 (chatbot/nodes.py)

```python
"""
节点函数模块
实现聊天机器人的核心逻辑
"""
from langchain_core.messages import AIMessage, HumanMessage, RemoveMessage
from langchain_openai import ChatOpenAI
from chatbot.state import ChatbotState


# 初始化 LLM
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)


def chatbot_node(state: ChatbotState) -> dict:
    """
    聊天机器人节点

    功能：
    1. 读取对话历史
    2. 调用 LLM 生成回复
    3. 返回新消息（会被 add_messages 追加到历史中）

    Args:
        state: 当前状态，包含 messages 列表

    Returns:
        包含新消息的字典，add_messages 会自动合并
    """
    messages = state["messages"]

    # 调用 LLM
    response = llm.invoke(messages)

    # 返回新消息（会被追加到 messages 列表）
    return {"messages": [response]}


def user_input_node(state: ChatbotState, user_message: str) -> dict:
    """
    用户输入节点

    功能：
    1. 接收用户输入
    2. 创建 HumanMessage
    3. 返回新消息

    Args:
        state: 当前状态
        user_message: 用户输入的文本

    Returns:
        包含用户消息的字典
    """
    return {"messages": [HumanMessage(content=user_message)]}


def regenerate_last_message_node(state: ChatbotState) -> dict:
    """
    重新生成最后一条 AI 消息

    功能：
    1. 删除最后一条 AI 消息
    2. 重新调用 LLM 生成

    Args:
        state: 当前状态

    Returns:
        包含新生成消息的字典
    """
    messages = state["messages"]

    # 找到最后一条 AI 消息
    last_ai_message = None
    for msg in reversed(messages):
        if isinstance(msg, AIMessage):
            last_ai_message = msg
            break

    if last_ai_message is None:
        return {"messages": []}

    # 删除最后一条 AI 消息
    remove_msg = RemoveMessage(id=last_ai_message.id)

    # 重新生成
    response = llm.invoke(messages[:-1])  # 不包含最后一条 AI 消息

    # 返回删除指令和新消息
    return {"messages": [remove_msg, response]}


def clear_history_node(state: ChatbotState) -> dict:
    """
    清空对话历史

    功能：
    使用 REMOVE_ALL_MESSAGES 清空所有历史消息

    Args:
        state: 当前状态

    Returns:
        包含清空指令的字典
    """
    from langgraph.graph.message import REMOVE_ALL_MESSAGES

    return {"messages": [RemoveMessage(id=REMOVE_ALL_MESSAGES)]}


def edit_message_node(state: ChatbotState, message_id: str, new_content: str) -> dict:
    """
    编辑特定消息

    功能：
    通过相同 ID 替换消息内容

    Args:
        state: 当前状态
        message_id: 要编辑的消息 ID
        new_content: 新的消息内容

    Returns:
        包含更新后消息的字典
    """
    messages = state["messages"]

    # 找到要编辑的消息
    target_message = None
    for msg in messages:
        if msg.id == message_id:
            target_message = msg
            break

    if target_message is None:
        return {"messages": []}

    # 创建新消息（相同 ID 会替换旧消息）
    if isinstance(target_message, HumanMessage):
        new_message = HumanMessage(content=new_content, id=message_id)
    elif isinstance(target_message, AIMessage):
        new_message = AIMessage(content=new_content, id=message_id)
    else:
        return {"messages": []}

    return {"messages": [new_message]}
```

### 3. 图构建 (chatbot/graph.py)

```python
"""
图构建模块
定义聊天机器人的工作流
"""
from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.memory import MemorySaver
from chatbot.state import ChatbotState
from chatbot.nodes import chatbot_node


def create_chatbot_graph():
    """
    创建聊天机器人图

    工作流：
    START -> chatbot -> END

    Returns:
        编译后的图对象
    """
    # 创建图
    builder = StateGraph(ChatbotState)

    # 添加节点
    builder.add_node("chatbot", chatbot_node)

    # 添加边
    builder.add_edge(START, "chatbot")
    builder.add_edge("chatbot", END)

    # 编译（带持久化）
    memory = MemorySaver()
    graph = builder.compile(checkpointer=memory)

    return graph
```

### 4. 工具函数 (chatbot/utils.py)

```python
"""
工具函数模块
提供消息格式化、打印等辅助功能
"""
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage


def print_messages(messages: list, title: str = "Messages"):
    """
    打印消息列表

    Args:
        messages: 消息列表
        title: 标题
    """
    print(f"\n{'='*60}")
    print(f"{title}")
    print(f"{'='*60}")

    for i, msg in enumerate(messages, 1):
        role = msg.__class__.__name__.replace("Message", "")
        content = msg.content
        msg_id = getattr(msg, 'id', 'N/A')

        print(f"\n[{i}] {role} (ID: {msg_id[:8]}...)")
        print(f"    {content}")

    print(f"\n{'='*60}\n")


def count_messages_by_type(messages: list) -> dict:
    """
    统计各类型消息数量

    Args:
        messages: 消息列表

    Returns:
        各类型消息的数量字典
    """
    counts = {
        "human": 0,
        "ai": 0,
        "system": 0,
        "other": 0
    }

    for msg in messages:
        if isinstance(msg, HumanMessage):
            counts["human"] += 1
        elif isinstance(msg, AIMessage):
            counts["ai"] += 1
        elif isinstance(msg, SystemMessage):
            counts["system"] += 1
        else:
            counts["other"] += 1

    return counts
```

## 常见问题与解决方案

### 问题 1: 消息 ID 冲突

**问题描述**：
手动指定消息 ID 时，可能与自动生成的 ID 冲突。

**解决方案**：
```python
# 让 add_messages 自动生成 ID
message = HumanMessage(content="Hello")  # ID 会自动生成

# 或者使用有意义的 ID 前缀
message = HumanMessage(content="Hello", id=f"user-{timestamp}-{uuid4()}")
```

### 问题 2: 删除不存在的消息

**问题描述**：
尝试删除不存在的消息 ID 会抛出 `ValueError`。

**解决方案**：
```python
def safe_remove_message(messages: list, message_id: str) -> list:
    """安全删除消息"""
    # 检查消息是否存在
    exists = any(msg.id == message_id for msg in messages)

    if not exists:
        print(f"警告：消息 {message_id} 不存在")
        return messages

    # 删除消息
    return add_messages(messages, [RemoveMessage(id=message_id)])
```

### 问题 3: 消息列表过长

**问题描述**：
对话历史过长会导致 Token 超限。

**解决方案**：
```python
def trim_messages(messages: list, max_messages: int = 20) -> list:
    """保留最近的 N 条消息"""
    if len(messages) <= max_messages:
        return messages

    # 保留最近的消息
    return messages[-max_messages:]

# 在节点中使用
def chatbot_node(state: ChatbotState) -> dict:
    messages = trim_messages(state["messages"], max_messages=20)
    response = llm.invoke(messages)
    return {"messages": [response]}
```

### 问题 4: 消息格式不一致

**问题描述**：
不同来源的消息格式可能不一致。

**解决方案**：
```python
from langchain_core.messages import convert_to_messages

def normalize_messages(messages: list) -> list:
    """标准化消息格式"""
    # convert_to_messages 会自动转换各种格式
    return convert_to_messages(messages)

# 示例
messages = [
    {"role": "user", "content": "Hello"},  # 字典格式
    HumanMessage(content="Hi"),            # LangChain 格式
]
normalized = normalize_messages(messages)
```

## 性能优化建议

### 1. 消息压缩

对于长对话，定期压缩历史消息：

```python
def compress_messages(messages: list, llm) -> list:
    """
    使用 LLM 压缩对话历史
    保留关键信息，减少 Token 消耗
    """
    if len(messages) < 10:
        return messages

    # 保留最近 5 条消息
    recent_messages = messages[-5:]

    # 压缩旧消息
    old_messages = messages[:-5]
    summary_prompt = f"请总结以下对话的关键信息：\n{old_messages}"
    summary = llm.invoke([HumanMessage(content=summary_prompt)])

    # 返回压缩后的消息
    return [SystemMessage(content=f"对话摘要：{summary.content}")] + recent_messages
```

## 总结

本示例展示了如何使用 `add_messages` Reducer 构建聊天机器人：

1. **消息累积**：自动追加新消息到历史
2. **消息更新**：通过相同 ID 替换消息
3. **消息删除**：使用 `RemoveMessage` 删除特定消息
4. **格式转换**：支持 OpenAI 格式

**关键要点**：
- `add_messages` 是 LangGraph 内置的消息列表 Reducer
- 按 ID 合并，相同 ID 会替换而非追加
- 支持 `RemoveMessage` 和 `REMOVE_ALL_MESSAGES`
- 适用于所有需要维护消息历史的场景
