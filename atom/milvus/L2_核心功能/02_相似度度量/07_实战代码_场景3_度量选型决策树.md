## 完整可运行代码

```python
"""
Milvus 度量选型决策树实战
演示：基于数据特征自动推荐最佳相似度度量

基于 2025-2026 最佳实践：
- 匹配 Embedding 模型训练度量
- 检测向量归一化状态
- 考虑性能与准确率权衡

参考来源：
- Zilliz Blog: Similarity Metrics for Vector Search (Mar 11, 2025)
- Milvus AI Quick Reference: Metric Selection
- Milvus Bootcamp CheatSheet
"""

import numpy as np
from typing import List, Dict, Tuple
from dataclasses import dataclass
from enum import Enum

# ===== 1. 定义度量类型和推荐结果 =====
print("=== 度量选型决策树系统 ===\n")

class MetricType(Enum):
    """支持的度量类型"""
    COSINE = "COSINE"
    IP = "IP"
    L2 = "L2"
    HAMMING = "HAMMING"
    JACCARD = "JACCARD"

class VectorType(Enum):
    """向量类型"""
    DENSE_FLOAT = "DENSE_FLOAT"
    BINARY = "BINARY"
    SPARSE = "SPARSE"

@dataclass
class MetricRecommendation:
    """度量推荐结果"""
    primary_metric: MetricType
    alternative_metrics: List[MetricType]
    confidence: float  # 0-1
    reasoning: str
    performance_note: str

    def __str__(self):
        return f"""
【推荐度量】: {self.primary_metric.value}
【备选度量】: {[m.value for m in self.alternative_metrics]}
【置信度】: {self.confidence:.0%}
【推荐理由】: {self.reasoning}
【性能说明】: {self.performance_note}
"""

# ===== 2. 向量特征分析器 =====

class VectorAnalyzer:
    """向量特征分析器"""

    @staticmethod
    def is_normalized(vectors: List[List[float]], tolerance: float = 1e-5) -> bool:
        """
        检测向量是否归一化

        来源：Milvus Bootcamp CheatSheet
        """
        norms = [np.linalg.norm(v) for v in vectors[:min(100, len(vectors))]]
        avg_norm = np.mean(norms)
        std_norm = np.std(norms)

        # 如果平均范数接近1且标准差很小，认为是归一化的
        is_norm = abs(avg_norm - 1.0) < tolerance and std_norm < tolerance

        print(f"  向量范数分析:")
        print(f"    平均范数: {avg_norm:.6f}")
        print(f"    标准差: {std_norm:.6f}")
        print(f"    归一化状态: {'✅ 已归一化' if is_norm else '❌ 未归一化'}")

        return is_norm

    @staticmethod
    def get_vector_type(vectors: List) -> VectorType:
        """检测向量类型"""
        if isinstance(vectors[0], bytes):
            return VectorType.BINARY
        elif isinstance(vectors[0], (list, np.ndarray)):
            return VectorType.DENSE_FLOAT
        else:
            return VectorType.SPARSE

    @staticmethod
    def analyze_magnitude_importance(vectors: List[List[float]]) -> bool:
        """
        分析向量幅度是否重要

        如果向量幅度差异很大，可能幅度有语义意义
        """
        norms = [np.linalg.norm(v) for v in vectors[:min(100, len(vectors))]]
        norm_range = max(norms) - min(norms)
        norm_cv = np.std(norms) / np.mean(norms)  # 变异系数

        # 如果变异系数 > 0.1，认为幅度有意义
        magnitude_matters = norm_cv > 0.1

        print(f"  幅度重要性分析:")
        print(f"    范数范围: [{min(norms):.4f}, {max(norms):.4f}]")
        print(f"    变异系数: {norm_cv:.4f}")
        print(f"    幅度重要性: {'✅ 重要' if magnitude_matters else '❌ 不重要'}")

        return magnitude_matters

# ===== 3. 度量选型决策树 =====

class MetricSelector:
    """
    度量选型决策树

    基于 2025-2026 最佳实践：
    1. 优先匹配 Embedding 模型训练度量
    2. 检测向量归一化状态
    3. 考虑数据类型和应用场景
    4. 权衡性能与准确率

    来源：Zilliz Blog (Mar 11, 2025)
    """

    # 常见 Embedding 模型的推荐度量
    MODEL_METRICS = {
        "text-embedding-3-small": MetricType.COSINE,
        "text-embedding-3-large": MetricType.COSINE,
        "text-embedding-ada-002": MetricType.COSINE,
        "all-MiniLM-L6-v2": MetricType.COSINE,
        "all-mpnet-base-v2": MetricType.COSINE,
        "bge-large-en-v1.5": MetricType.COSINE,
        "bge-large-zh-v1.5": MetricType.COSINE,
        "msmarco-bert-base-dot-v5": MetricType.IP,
        "ResNet50": MetricType.L2,
        "VGG16": MetricType.L2,
    }

    @classmethod
    def select_metric(
        cls,
        vectors: List[List[float]],
        model_name: str = None,
        application_type: str = "general",
        performance_priority: bool = False
    ) -> MetricRecommendation:
        """
        智能度量选型

        参数：
        - vectors: 样本向量
        - model_name: Embedding 模型名称
        - application_type: 应用类型 (general, nlp, image, recommendation)
        - performance_priority: 是否优先考虑性能
        """
        print(f"\n=== 度量选型分析 ===")
        print(f"模型名称: {model_name or '未指定'}")
        print(f"应用类型: {application_type}")
        print(f"性能优先: {performance_priority}")
        print()

        # Step 1: 检查模型训练度量
        if model_name and model_name in cls.MODEL_METRICS:
            model_metric = cls.MODEL_METRICS[model_name]
            print(f"✅ 检测到已知模型，推荐度量: {model_metric.value}")

            return MetricRecommendation(
                primary_metric=model_metric,
                alternative_metrics=[],
                confidence=0.95,
                reasoning=f"模型 {model_name} 使用 {model_metric.value} 训练，匹配训练度量可获得最佳效果",
                performance_note="已针对该度量优化"
            )

        # Step 2: 分析向量特征
        print("📊 分析向量特征:")
        is_normalized = VectorAnalyzer.is_normalized(vectors)
        magnitude_matters = VectorAnalyzer.analyze_magnitude_importance(vectors)
        print()

        # Step 3: 决策树逻辑
        if is_normalized:
            # 归一化向量
            if performance_priority:
                # 性能优先 → IP (最快)
                return MetricRecommendation(
                    primary_metric=MetricType.IP,
                    alternative_metrics=[MetricType.COSINE],
                    confidence=0.90,
                    reasoning="向量已归一化，IP 与 COSINE 等价但性能更快",
                    performance_note="IP 是归一化向量的最快选择"
                )
            else:
                # 准确率优先 → COSINE (最安全)
                return MetricRecommendation(
                    primary_metric=MetricType.COSINE,
                    alternative_metrics=[MetricType.IP],
                    confidence=0.85,
                    reasoning="COSINE 是归一化向量的标准选择，适用于 80% 的场景",
                    performance_note="COSINE 自动处理归一化，最安全的选择"
                )
        else:
            # 未归一化向量
            if magnitude_matters:
                # 幅度有意义 → L2
                return MetricRecommendation(
                    primary_metric=MetricType.L2,
                    alternative_metrics=[MetricType.COSINE],
                    confidence=0.80,
                    reasoning="向量幅度差异大，可能包含语义信息，使用 L2 保留幅度特征",
                    performance_note="L2 适合幅度有意义的场景（如图像特征）"
                )
            else:
                # 幅度不重要 → COSINE
                return MetricRecommendation(
                    primary_metric=MetricType.COSINE,
                    alternative_metrics=[MetricType.L2],
                    confidence=0.75,
                    reasoning="向量未归一化但幅度不重要，COSINE 自动归一化处理",
                    performance_note="COSINE 是通用场景的默认选择"
                )

    @classmethod
    def select_for_binary(cls) -> MetricRecommendation:
        """二值向量度量选型"""
        return MetricRecommendation(
            primary_metric=MetricType.HAMMING,
            alternative_metrics=[MetricType.JACCARD],
            confidence=0.90,
            reasoning="HAMMING 适用于位级差异（图像哈希），JACCARD 适用于集合相似度（标签匹配）",
            performance_note="二值向量计算极快，HAMMING 略快于 JACCARD"
        )

def print_decision_tree():

    print("\n" + "=" * 60)

【二值向量】: HAMMING（位级差异）或 JACCARD（集合相似度）
""")
```

## 核心要点

### 1. 度量选型原则（2025-2026）

**黄金法则**（来源：Zilliz Blog, Mar 11, 2025）：
> 匹配 Embedding 模型训练时使用的度量

**决策优先级**：
1. **模型训练度量** > 2. **向量归一化状态** > 3. **应用场景** > 4. **性能要求**

### 2. 常见模型推荐度量

| 模型 | 推荐度量 | 原因 |
|------|----------|------|
| OpenAI (text-embedding-*) | COSINE | 训练时使用余弦相似度 |
| sentence-transformers | COSINE | 训练时使用余弦相似度 |
| msmarco-bert-base-dot-v5 | IP | 训练时使用点积 |
| ResNet50, VGG16 | L2 | 图像特征，幅度有意义 |
| pHash, dHash | HAMMING | 二值哈希，位级差异 |

### 3. 归一化检测方法

```python
def is_normalized(vectors, tolerance=1e-5):
    norms = [np.linalg.norm(v) for v in vectors]
    avg_norm = np.mean(norms)
    std_norm = np.std(norms)
    return abs(avg_norm - 1.0) < tolerance and std_norm < tolerance
```

**关键指标**：
- 平均范数接近 1.0
- 标准差 < 1e-5

### 4. 性能对比

| 度量 | 归一化向量 | 未归一化向量 | 相对速度 |
|------|-----------|-------------|---------|
| IP | ✅ 最优 | ❌ 不适用 | 1.0x (最快) |
| COSINE | ✅ 推荐 | ✅ 推荐 | 1.1x |
| L2 | ⚠️ 可用 | ✅ 推荐 | 1.2x |

**来源**：Milvus Bootcamp CheatSheet

### 5. 实际应用建议

**NLP 应用**：
- 默认使用 COSINE
- 性能优先且向量归一化 → IP
- 文档长度重要 → L2

**图像应用**：
- 特征向量 → L2 或 COSINE
- 图像哈希 → HAMMING
- 相似度匹配 → COSINE

**推荐系统**：
- 协同过滤 → IP
- 内容推荐 → COSINE
- 用户画像 → COSINE
