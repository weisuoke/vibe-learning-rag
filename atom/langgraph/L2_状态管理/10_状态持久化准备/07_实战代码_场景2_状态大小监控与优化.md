# 实战代码 - 场景2：状态大小监控与优化

## 场景说明

本场景演示如何监控和优化 LangGraph 的状态大小。你将看到：

1. 使用 `sys.getsizeof` 和 `JsonPlusSerializer` 测量 checkpoint 大小
2. 消息累积问题的演示 —— 多轮对话导致状态无限膨胀
3. 消息裁剪策略 —— 只保留最近 N 条消息
4. 大对象外部存储 —— 将大数据移到外部存储，状态中只保留引用
5. 临时数据处理 —— 区分需要持久化的数据和临时数据

**核心问题：** 每次节点执行后，LangGraph 都会保存一个 checkpoint。如果状态中包含不断增长的数据（如消息列表、检索结果），checkpoint 会越来越大，最终导致性能下降甚至内存溢出。

def measure_state_size(label: str, state: dict) -> dict[str, int]:
    """测量状态的各种大小指标

    两种测量方式：
    1. sys.getsizeof —— Python 对象在内存中的大小（浅拷贝，不含嵌套对象）
    2. len(serializer.dumps_typed()) —— 序列化后的字节大小（这才是 checkpoint 的实际大小）

    checkpoint 的实际存储大小 ≈ 序列化后的字节大小
    """
    # 方式 1：Python 内存大小（仅参考，不准确）
    memory_size = sys.getsizeof(state)

    # 方式 2：序列化后的字节大小（准确反映 checkpoint 大小）
    _, data = serializer.dumps_typed(state)
    serialized_size = len(data)

    print(f"  {label}:")
    print(f"    Python 内存大小: {memory_size:,} bytes (sys.getsizeof, 仅浅层)")
    print(f"    序列化后大小:    {serialized_size:,} bytes (checkpoint 实际大小)")

    return {"memory": memory_size, "serialized": serialized_size}


def measure_field_sizes(state: dict) -> None:
    """逐字段测量状态大小，找出最大的字段"""
    print("  逐字段大小分析:")
    field_sizes = []
    for key, value in state.items():
        try:
            _, data = serializer.dumps_typed(value)
            size = len(data)
        except Exception:
            size = 0
        field_sizes.append((key, size))

    # 按大小降序排列
    field_sizes.sort(key=lambda x: x[1], reverse=True)
    total = sum(s for _, s in field_sizes)

    for key, size in field_sizes:
        pct = (size / total * 100) if total > 0 else 0
        bar = "█" * int(pct / 5)  # 简单的柱状图
        print(f"    {key:20s}: {size:6,} bytes ({pct:5.1f}%) {bar}")
    print(f"    {'TOTAL':20s}: {total:6,} bytes")

# 方法 1：自定义 Reducer 实现自动裁剪
MAX_MESSAGES = 10


def trimmed_add(existing: list[str], new: list[str]) -> list[str]:
    """带裁剪的消息累积 Reducer

    工作原理：
    1. 先把新消息追加到现有列表
    2. 如果总数超过 MAX_MESSAGES，只保留最后 MAX_MESSAGES 条
    3. 这样每次 checkpoint 保存的消息数量有上限

    这是最优雅的方案 —— 裁剪逻辑封装在 Reducer 中，节点代码无需关心。
    """
    combined = existing + new
    if len(combined) > MAX_MESSAGES:
        return combined[-MAX_MESSAGES:]
    return combined


class TrimmedChatState(TypedDict):
    """带消息裁剪的对话状态"""
    messages: Annotated[list[str], trimmed_add]  # 自动裁剪的消息列表
    current_query: str
    turn_count: int


def trimmed_chat_turn(state: TrimmedChatState) -> dict:

# 模拟外部存储（实际项目中可以用 Redis、S3、数据库等）
external_store: dict[str, Any] = {}


def store_externally(key: str, data: Any) -> str:
    """将大对象存到外部存储，返回引用 ID"""
    store_id = f"ext_{key}_{len(external_store)}"
    external_store[store_id] = data
    return store_id


def load_from_external(store_id: str) -> Any:
    """从外部存储加载对象"""
    return external_store.get(store_id)


class OptimizedRAGState(TypedDict):
    """优化后的 RAG 状态

    关键设计：
    - documents 字段不存储完整文档内容
    - 而是存储外部存储的引用 ID
    - 需要文档内容时，通过 ID 从外部存储加载
    """
    query: str
    document_refs: list[str]       # 文档引用 ID（而非完整文档）
    document_count: int            # 文档数量（快速访问，无需加载）
    answer: str
    messages: Annotated[list[str], trimmed_add]  # 带裁剪的消息


def optimized_retrieve(state: OptimizedRAGState) -> dict:
    """优化的检索节点：大文档存到外部"""
    query = state["query"]

    # 模拟检索到大文档
    large_docs = [
        {"content": f"关于'{query}'的详细文档 {i}。" + "这是很长的文档内容。" * 200, "score": 0.9 - i * 0.05}
        for i in range(5)
    ]

    # 将大文档存到外部存储，状态中只保留引用
    refs = []
    for i, doc in enumerate(large_docs):
        ref_id = store_externally(f"doc_{query}_{i}", doc)
        refs.append(ref_id)

    return {
        "document_refs": refs,
        "document_count": len(refs),
        "messages": [f"检索完成：{len(refs)} 篇文档已存入外部存储"],
    }


def optimized_generate(state: OptimizedRAGState) -> dict:
    """优化的生成节点：需要时从外部加载文档"""
    refs = state["document_refs"]

    # 从外部存储加载文档（只在需要时加载）
    docs = [load_from_external(ref) for ref in refs]
    context = " | ".join(d["content"][:50] for d in docs if d)

    answer = f"基于 {len(docs)} 篇文档生成的回答：{state['query']}的答案..."
    return {
        "answer": answer,
        "messages": [f"生成完成：基于 {len(docs)} 篇文档"],
    }

# 方法：使用覆盖模式 + 节点清理

class SmartRAGState(TypedDict):
    """区分持久化数据和临时数据的状态设计

    设计原则：
    - 需要持久化的字段：query, answer, messages（跨步骤需要）
    - 临时字段：raw_documents, debug_info（只在当前步骤需要）
    - 临时字段在使用后清空，减少 checkpoint 大小
    """
    # 持久化字段 —— 跨步骤需要
    query: str
    answer: str
    messages: Annotated[list[str], trimmed_add]

    # 临时字段 —— 只在当前步骤需要，用完清空
    # 使用覆盖模式（无 Reducer），每次赋值都是全量替换
    raw_documents: list[dict[str, Any]]  # 检索的原始文档（可能很大）
    debug_info: dict[str, Any]           # 调试信息（不需要持久化）


def smart_retrieve(state: SmartRAGState) -> dict:
    """检索节点：产生临时的大文档数据"""
    query = state["query"]
    docs = [
        {"content": f"文档 {i}: " + "详细内容 " * 100, "score": 0.9 - i * 0.05}
        for i in range(5)
    ]
    return {
        "raw_documents": docs,  # 临时数据，下一步用完就清空
        "debug_info": {"retrieval_time_ms": 150, "index_name": "rag_index"},
        "messages": [f"检索完成：{len(docs)} 篇文档"],
    }


def smart_generate(state: SmartRAGState) -> dict:
    """生成节点：使用临时文档数据，然后清空"""
    docs = state.get("raw_documents", [])
    context = " ".join(d["content"][:30] for d in docs)
    answer = f"基于 {len(docs)} 篇文档的回答：{state['query']}..."

    return {
        "answer": answer,
        # 关键：清空临时字段！
        # 覆盖模式下，赋值空列表/空字典就是清空
        "raw_documents": [],       # 清空大文档
        "debug_info": {},          # 清空调试信息
        "messages": [f"生成完成"],
    }


# 构建图

    print(f"  step={step}, node={node_name:12s}, size={size:6,} bytes, raw_documents={raw_docs_count} 篇")

print(f"\n  注意：generate 步骤后 raw_documents 被清空，checkpoint 变小了！")
print(f"  最终 checkpoint 不包含大文档数据，只保留必要的 query/answer/messages。")

## 关键知识点解析

### 1. 状态大小的两种测量方式

| 方式 | 函数 | 测量内容 | 准确性 |
|------|------|---------|--------|
| Python 内存 | `sys.getsizeof(obj)` | 对象在内存中的浅层大小 | 不准确（不含嵌套对象） |
| 序列化大小 | `len(serializer.dumps_typed(obj)[1])` | 序列化后的字节大小 | 准确（= checkpoint 实际大小） |

推荐使用序列化大小来评估 checkpoint 的存储开销。

### 2. 状态膨胀的三大来源

```
来源 1：消息累积
  messages: Annotated[list, add]  →  每轮对话追加 2 条消息
  20 轮后：40 条消息，~17KB

来源 2：大文档存储
  documents: list[dict]  →  5 篇文档，每篇 ~2KB
  一次检索：~10KB

来源 3：调试/中间数据
  debug_info, raw_embeddings, intermediate_results
  不需要持久化但占用空间
```

### 3. 优化策略总结

| 策略 | 适用场景 | 实现方式 | 效果 |
|------|---------|---------|------|
| 消息裁剪 | 多轮对话 | 自定义 Reducer（trimmed_add） | checkpoint 大小有上限 |
| 外部存储 | 大文档/大数据 | 状态存引用 ID，数据存外部 | 减少 90%+ |
| 临时数据清空 | 中间计算结果 | 节点执行后清空临时字段 | 最终 checkpoint 最小化 |
| checkpoint 裁剪 | 历史 checkpoint 过多 | `checkpointer.prune()` | 减少存储总量 |

### 4. 自定义裁剪 Reducer 模式

```python
MAX_ITEMS = 10

def trimmed_add(existing: list, new: list) -> list:
    """通用的带裁剪累积 Reducer"""
    combined = existing + new
    if len(combined) > MAX_ITEMS:
        return combined[-MAX_ITEMS:]
    return combined

# 使用
class State(TypedDict):
    messages: Annotated[list[str], trimmed_add]
```

这个模式可以复用到任何需要限制列表长度的场景。

### 5. 外部存储模式

```
状态中存储：                    外部存储：
┌──────────────────┐          ┌──────────────────┐
│ document_refs:   │          │ ext_doc_0:       │
│   ["ext_doc_0",  │ ──引用──→│   {content: ..., │
│    "ext_doc_1",  │          │    score: 0.95}  │
│    "ext_doc_2"]  │          │ ext_doc_1:       │
│                  │          │   {content: ..., │
│ checkpoint ~200B │          │    score: 0.88}  │
└──────────────────┘          │ 总计 ~10KB       │
                              └──────────────────┘
```

## 在 RAG 系统中的应用

状态大小优化在 RAG 系统中至关重要：

- **多轮对话 RAG**：对话历史不断增长，必须裁剪消息避免 checkpoint 膨胀
- **文档检索结果**：检索到的文档可能很大，应存到外部存储（Redis/S3），状态中只保留引用
- **Embedding 向量**：高维向量（768/1536 维）占用大量空间，不应存在状态中
- **中间计算结果**：如 rerank 分数、chunk 分割结果等，用完即清空
