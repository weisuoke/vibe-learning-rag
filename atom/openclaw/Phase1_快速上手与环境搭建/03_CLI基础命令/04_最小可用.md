# 04_最小可用

## 目标

通过一个完整的最小可用示例，快速掌握 OpenClaw CLI 的 5 个核心命令，从安装到运行完整的多通道 AI 网关系统。

---

## 最小可用定义

**最小可用 = 3 条命令 + 5 分钟 + 完整体验**

```bash
# 1. 启动网关
openclaw gateway --port 18789

# 2. 发送消息
openclaw message send --to +1234567890 --message "Hello from OpenClaw"

# 3. AI 交互
openclaw agent --message "帮我总结今天的任务"
```

---

## 完整快速开始流程

### 步骤 1: 环境准备

**前置要求**：
- Node.js 22+ (推荐 22.12.0+)
- npm 或 pnpm
- 终端访问权限

**检查环境**：
```bash
# 检查 Node.js 版本
node --version  # 应该 >= 22.0.0

# 检查 npm 版本
npm --version   # 应该 >= 10.0.0
```

---

### 步骤 2: 安装 OpenClaw

**方式 1: npm 全局安装（推荐）**
```bash
# 使用 npm
npm install -g openclaw@latest

# 或使用 pnpm（更快）
pnpm add -g openclaw@latest

# 验证安装
openclaw --version
```

**方式 2: 从源码安装（开发者）**
```bash
# 克隆仓库
git clone https://github.com/openclaw/openclaw.git
cd openclaw

# 安装依赖
pnpm install

# 构建
pnpm build

# 运行
pnpm openclaw --version
```

**预期输出**：
```
openclaw version 0.x.x
```

## 完整 TypeScript 示例

### 示例 1: 自动化脚本

**文件**: `examples/openclaw/cli_basic/01_quick_start.ts`

```typescript
/**
 * OpenClaw CLI 快速开始示例
 *
 * 功能：
 * 1. 检查 Gateway 状态
 * 2. 启动 Gateway（如果未运行）
 * 3. 发送测试消息
 * 4. AI Agent 交互
 *
 * 运行：
 * npx tsx examples/openclaw/cli_basic/01_quick_start.ts
 */

import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

// 颜色输出
const colors = {
  reset: '\x1b[0m',
  green: '\x1b[32m',
  yellow: '\x1b[33m',
  red: '\x1b[31m',
  blue: '\x1b[34m',
};

function log(message: string, color: string = colors.reset) {
  console.log(`${color}${message}${colors.reset}`);
}

// 检查 Gateway 状态
async function checkGatewayStatus(): Promise<boolean> {
  try {
    const { stdout } = await execAsync('openclaw gateway status');
    return stdout.includes('Running');
  } catch (error) {
    return false;
  }
}

// 启动 Gateway
async function startGateway(): Promise<void> {
  log('🚀 启动 Gateway...', colors.blue);

  // 在后台启动 Gateway
  exec('openclaw gateway --port 18789 > /dev/null 2>&1 &');

  // 等待 Gateway 启动
  await new Promise(resolve => setTimeout(resolve, 3000));

  log('✅ Gateway 已启动', colors.green);
}

// 发送测试消息
async function sendTestMessage(): Promise<void> {
  log('📤 发送测试消息...', colors.blue);

  try {
    const { stdout } = await execAsync(
      'openclaw message send --to +1234567890 --message "Hello from OpenClaw CLI"'
    );

    log('✅ 消息已发送', colors.green);
    console.log(stdout);
  } catch (error) {
    log('❌ 消息发送失败', colors.red);
    console.error(error);
  }
}

// AI Agent 交互
async function agentInteraction(): Promise<void> {
  log('🤖 AI Agent 交互...', colors.blue);

  try {
    const { stdout } = await execAsync(
      'openclaw agent --message "帮我总结今天的任务" --thinking medium'
    );

    log('✅ Agent 响应完成', colors.green);
    console.log(stdout);
  } catch (error) {
    log('❌ Agent 交互失败', colors.red);
    console.error(error);
  }
}

// 主函数
async function main() {
  log('=== OpenClaw CLI 快速开始 ===\n', colors.yellow);

  // 1. 检查 Gateway 状态
  log('1️⃣ 检查 Gateway 状态...', colors.blue);
  const isRunning = await checkGatewayStatus();

  if (isRunning) {
    log('✅ Gateway 已运行', colors.green);
  } else {
    log('⚠️  Gateway 未运行，正在启动...', colors.yellow);
    await startGateway();
  }

  console.log('');

  // 2. 发送测试消息
  log('2️⃣ 发送测试消息', colors.blue);
  await sendTestMessage();

  console.log('');

  // 3. AI Agent 交互
  log('3️⃣ AI Agent 交互', colors.blue);
  await agentInteraction();

  console.log('');
  log('=== 快速开始完成 ===', colors.yellow);
}

// 错误处理
main().catch((error) => {
  log('❌ 发生错误:', colors.red);
  console.error(error);
  process.exit(1);
});
```

### 示例 2: 通道管理脚本

**文件**: `examples/openclaw/cli_basic/02_channel_management.ts`

```typescript
/**
 * OpenClaw CLI 通道管理示例
 *
 * 功能：
 * 1. 列出所有通道
 * 2. 检查通道状态
 * 3. 添加新通道
 * 4. 发送多通道消息
 *
 * 运行：
 * npx tsx examples/openclaw/cli_basic/02_channel_management.ts
 */

import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

// 列出所有通道
async function listChannels(): Promise<void> {
  console.log('📋 列出所有通道...\n');

  try {
    const { stdout } = await execAsync('openclaw channels list');
    console.log(stdout);
  } catch (error) {
    console.error('❌ 列出通道失败:', error);
  }
}

// 检查通道状态
async function checkChannelStatus(): Promise<void> {
  console.log('🔍 检查通道状态...\n');

  try {
    const { stdout } = await execAsync('openclaw channels status');
    console.log(stdout);
  } catch (error) {
    console.error('❌ 检查状态失败:', error);
  }
}

// 发送多通道消息
async function sendMultiChannelMessage(): Promise<void> {
  console.log('📤 发送多通道消息...\n');

  const message = '系统测试消息 - ' + new Date().toISOString();

  try {
    const { stdout } = await execAsync(
      `openclaw message send --to @all --message "${message}"`
    );
    console.log(stdout);
  } catch (error) {
    console.error('❌ 发送消息失败:', error);
  }
}

// 主函数
async function main() {
  console.log('=== OpenClaw 通道管理 ===\n');

  await listChannels();
  console.log('');

  await checkChannelStatus();
  console.log('');

  await sendMultiChannelMessage();
  console.log('');

  console.log('=== 通道管理完成 ===');
}

main().catch(console.error);
```

### 示例 3: 配置管理脚本

**文件**: `examples/openclaw/cli_basic/03_config_management.ts`

```typescript
/**
 * OpenClaw CLI 配置管理示例
 *
 * 功能：
 * 1. 读取配置
 * 2. 设置配置
 * 3. 切换模型
 * 4. 验证配置
 *
 * 运行：
 * npx tsx examples/openclaw/cli_basic/03_config_management.ts
 */

import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

// 读取配置
async function getConfig(key: string): Promise<string> {
  try {
    const { stdout } = await execAsync(`openclaw config get ${key}`);
    return stdout.trim();
  } catch (error) {
    return '';
  }
}

// 设置配置
async function setConfig(key: string, value: string): Promise<void> {
  try {
    await execAsync(`openclaw config set ${key} ${value}`);
    console.log(`✅ 设置 ${key} = ${value}`);
  } catch (error) {
    console.error(`❌ 设置 ${key} 失败:`, error);
  }
}

// 显示当前配置
async function showCurrentConfig(): Promise<void> {
  console.log('📋 当前配置:\n');

  const keys = ['provider', 'model', 'workspace'];

  for (const key of keys) {
    const value = await getConfig(key);
    console.log(`  ${key}: ${value || '(未设置)'}`);
  }

  console.log('');
}

// 切换模型
async function switchModel(model: string): Promise<void> {
  console.log(`🔄 切换模型到 ${model}...\n`);

  await setConfig('model', model);

  // 验证切换
  const currentModel = await getConfig('model');
  if (currentModel === model) {
    console.log(`✅ 模型已切换到 ${model}`);
  } else {
    console.log(`❌ 模型切换失败`);
  }
}

// 主函数
async function main() {
  console.log('=== OpenClaw 配置管理 ===\n');

  // 显示当前配置
  await showCurrentConfig();

  // 切换到 claude-opus-4
  await switchModel('claude-opus-4');
  console.log('');

  // 显示更新后的配置
  await showCurrentConfig();

  console.log('=== 配置管理完成 ===');
}

main().catch(console.error);
```

## 常见问题排查

### 问题 1: Gateway 启动失败

**症状**：
```
Error: Port 18789 is already in use
```

**解决方案**：
```bash
# 检查端口占用
lsof -i :18789

# 停止现有 Gateway
openclaw gateway stop

# 或使用其他端口
openclaw gateway --port 18790
```

---

### 问题 2: API Key 未配置

**症状**：
```
Error: ANTHROPIC_API_KEY not found
```

**解决方案**：
```bash
# 方式 1: 环境变量
export ANTHROPIC_API_KEY=sk-ant-api03-...

# 方式 2: 配置文件
openclaw config set provider anthropic
openclaw config set model claude-opus-4

# 方式 3: 重新运行 onboard
openclaw onboard --reset
```

---

### 问题 3: 消息发送失败

**症状**：
```
Error: No channels configured
```

**解决方案**：
```bash
# 添加通道
openclaw channels add telegram

# 或使用 --deliver=false 仅本地测试
openclaw agent --message "测试" --deliver=false
```
