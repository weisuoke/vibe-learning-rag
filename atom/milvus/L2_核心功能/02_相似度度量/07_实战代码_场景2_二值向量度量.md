## 完整可运行代码

```python
"""
Milvus 二值向量度量实战
演示：HAMMING 和 JACCARD 度量在图像哈希场景中的应用

基于 2025-2026 最佳实践：
- 使用 np.packbits 转换二值向量
- 对比 HAMMING 和 JACCARD 的行为差异
- 展示图像去重检测场景

参考来源：
- GitHub Discussion #23585: Binary vector format
- Milvus Binary Vector Docs: https://milvus.io/docs/binary-vector.md
- Milvus BIN_FLAT Index: https://milvus.io/docs/bin-flat.md
"""

import numpy as np
from pymilvus import (
    connections, Collection, CollectionSchema, FieldSchema, DataType,
    utility
)
import time
from typing import List

# ===== 1. 连接 Milvus =====
print("=== 连接 Milvus 2.6 ===")
connections.connect(
    alias="default",
    host="localhost",
    port="19530"
)
print("✅ 已连接到 Milvus")

# ===== 2. 二值向量转换工具 =====
print("\n=== 二值向量转换工具 ===")

def convert_bool_list_to_bytes(bool_list: List[int]) -> bytes:
    """
    将 0/1 列表转换为 bytes 格式（Milvus 要求）

    关键点：
    - 长度必须是 8 的倍数
    - 每 8 个 bit 打包成 1 个 byte
    - 使用位运算进行打包

    来源：GitHub Discussion #23585
    """
    if len(bool_list) % 8 != 0:
        raise ValueError("Length must be multiple of 8")

    byte_array = bytearray(len(bool_list) // 8)
    for i, bit in enumerate(bool_list):
        if bit == 1:
            index = i // 8
            shift = i % 8
            byte_array[index] |= (1 << shift)
    return bytes(byte_array)

def generate_random_binary_vector(dim: int) -> bytes:
    """生成随机二值向量"""
    bool_list = [np.random.randint(0, 2) for _ in range(dim)]
    return convert_bool_list_to_bytes(bool_list)

def bytes_to_bool_list(byte_data: bytes, dim: int) -> List[int]:
    """将 bytes 转换回 0/1 列表（用于可视化）"""
    bool_list = []
    for byte in byte_data:
        for i in range(8):
            bool_list.append((byte >> i) & 1)
    return bool_list[:dim]

# 测试转换
test_bool = [1, 0, 1, 1, 0, 0, 1, 0] * 2  # 16 bits
test_bytes = convert_bool_list_to_bytes(test_bool)
print(f"原始: {test_bool}")
print(f"转换: {test_bytes.hex()}")
print(f"还原: {bytes_to_bool_list(test_bytes, 16)}")
print("✅ 转换工具测试通过")

# ===== 3. 准备测试数据（模拟图像哈希）=====
print("\n=== 准备图像哈希数据 ===")

DIM = 64  # 64-bit hash (常见的 pHash 长度)
NUM_IMAGES = 100

# 生成模拟图像哈希
image_hashes = []
image_ids = []

for i in range(NUM_IMAGES):
    hash_vector = generate_random_binary_vector(DIM)
    image_hashes.append(hash_vector)
    image_ids.append(i)

print(f"✅ 生成 {NUM_IMAGES} 个图像哈希，维度 {DIM}")
print(f"存储大小: {DIM // 8} bytes/vector (vs {DIM * 4} bytes for float32)")
print(f"压缩比: {(DIM * 4) / (DIM // 8):.0f}x")

# ===== 4. 创建两个 Collection（不同度量）=====
print("\n=== 创建 Collections ===")

def create_binary_collection(name: str, metric_type: str) -> Collection:
    """创建二值向量 Collection"""
    # 删除已存在的 collection
    if utility.has_collection(name):
        utility.drop_collection(name)

    # 定义 schema（注意：dim 必须是 8 的倍数）
    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=False),
        FieldSchema(name="binary_vector", dtype=DataType.BINARY_VECTOR, dim=DIM)
    ]
    schema = CollectionSchema(fields=fields, description=f"Binary metric: {metric_type}")

    # 创建 collection
    collection = Collection(name=name, schema=schema)

    # 创建索引（BIN_FLAT：精确搜索）
    index_params = {
        "index_type": "BIN_FLAT",
        "metric_type": metric_type,
        "params": {}  # BIN_FLAT 不需要参数
    }
    collection.create_index(field_name="binary_vector", index_params=index_params)

    return collection

# 创建两个 collections
collections = {
    "HAMMING": create_binary_collection("binary_hamming", "HAMMING"),
    "JACCARD": create_binary_collection("binary_jaccard", "JACCARD")
}
print("✅ 创建 2 个 Collections (HAMMING, JACCARD)")

# ===== 5. 插入数据 =====
print("\n=== 插入数据 ===")
for metric_name, collection in collections.items():
    collection.insert([image_ids, image_hashes])
    collection.flush()
    print(f"✅ {metric_name}: 插入 {NUM_IMAGES} 条数据")

# ===== 6. 加载 Collections =====
print("\n=== 加载 Collections ===")
for metric_name, collection in collections.items():
    collection.load()
    print(f"✅ {metric_name}: 已加载到内存")

# ===== 7. 执行检索对比 =====
print("\n=== 检索对比测试 ===")

# 查询向量（模拟待检索的图像哈希）
query_hash = generate_random_binary_vector(DIM)
query_bool = bytes_to_bool_list(query_hash, DIM)
print(f"查询哈希: {query_bool[:16]}... (前16位)")

TOP_K = 5

results = {}
for metric_name, collection in collections.items():
    start_time = time.time()

    search_results = collection.search(
        data=[query_hash],
        anns_field="binary_vector",
        param={"metric_type": metric_name, "params": {}},
        limit=TOP_K,
        output_fields=["id"]
    )

    latency = (time.time() - start_time) * 1000  # ms

    results[metric_name] = {
        "ids": [hit.id for hit in search_results[0]],
        "distances": [hit.distance for hit in search_results[0]],
        "latency_ms": latency
    }

# ===== 8. 结果分析 =====
print("\n=== 检索结果对比 ===")

for metric_name, result in results.items():
    print(f"【{metric_name}】")
    print(f"  延迟: {result['latency_ms']:.2f} ms")
    print(f"  Top-{TOP_K} IDs: {result['ids']}")
    print(f"  距离/相似度: {[f'{d:.0f}' if metric_name == 'HAMMING' else f'{d:.4f}' for d in result['distances']]}")
    print()

# ===== 9. 度量特性分析 =====
print("=== 度量特性分析 ===")

# 9.1 HAMMING 距离解释
print("1. HAMMING 距离:")
print("   - 计算方式: XOR + popcount（统计不同的位数）")
print("   - 范围: [0, dim]（0 = 完全相同，dim = 完全不同）")
print("   - 适用场景: 图像哈希、指纹识别、快速去重")

# 9.2 JACCARD 距离解释
print("\n2. JACCARD 距离:")
print("   - 计算方式: 1 - (交集 / 并集)")
print("   - 范围: [0, 1]（0 = 完全相同，1 = 完全不同）")
print("   - 适用场景: 集合相似度、标签匹配、推荐系统")

# 9.3 手动验证 HAMMING 距离
print("\n3. 手动验证 HAMMING 距离:")
top1_id = results["HAMMING"]["ids"][0]
top1_hash = image_hashes[top1_id]
top1_bool = bytes_to_bool_list(top1_hash, DIM)

# 计算 XOR
xor_result = [q ^ t for q, t in zip(query_bool, top1_bool)]
hamming_distance = sum(xor_result)

print(f"   查询向量: {query_bool[:16]}...")
print(f"   Top-1 向量: {top1_bool[:16]}...")
print(f"   XOR 结果: {xor_result[:16]}...")
print(f"   手动计算 HAMMING: {hamming_distance}")
print(f"   Milvus 返回: {results['HAMMING']['distances'][0]:.0f}")
print(f"   ✅ 验证{'通过' if abs(hamming_distance - results['HAMMING']['distances'][0]) < 0.1 else '失败'}")

# ===== 10. 实际应用场景 =====
print("\n=== 实际应用场景 ===")

print("【场景1：图像去重检测】")
print("- 使用 pHash 生成 64-bit 哈希")
print("- HAMMING 距离 < 5: 认为是重复图像")
print("- 应用: 相册去重、版权检测")

print("\n【场景2：文本关键词匹配】")
print("- 词汇表编码为二值向量")
print("- JACCARD 距离衡量关键词重叠度")
print("- 应用: 文档分类、标签推荐")

print("\n【场景3：快速初筛】")
print("- 二值向量作为粗筛层")
print("- 候选集再用稠密向量精排")
print("- 应用: 大规模检索系统")

# ===== 11. 性能对比 =====
print("\n=== 性能对比 ===")

print("【存储效率】")
print(f"  Float32 向量 (128-dim): {128 * 4} bytes")
print(f"  Binary 向量 (128-dim): {128 // 8} bytes")
print(f"  压缩比: {(128 * 4) / (128 // 8):.0f}x")

print("\n【计算速度】")
for metric_name in ["HAMMING", "JACCARD"]:
    print(f"  {metric_name}: {results[metric_name]['latency_ms']:.2f} ms")

print("\n【适用场景】")
print("  HAMMING: 图像哈希、指纹识别、快速去重")
print("  JACCARD: 集合相似度、标签匹配、推荐系统")

# ===== 12. 清理资源 =====
print("\n=== 清理资源 ===")
for metric_name, collection in collections.items():
    collection.release()
    utility.drop_collection(collection.name)
    print(f"✅ {metric_name}: 已清理")

connections.disconnect("default")
print("✅ 已断开连接")

print("\n=== 实战完成 ===")
```

## 核心要点

### 1. 二值向量格式要求

**关键约束**（来源：Milvus Binary Vector Docs）：
- 维度必须是 8 的倍数
- 数据必须是 `bytes` 格式
- 每 8 个 bit 打包成 1 个 byte

**转换方法**：
```python
# 方法1：手动位运算（推荐，更灵活）
def convert_bool_list_to_bytes(bool_list):
    byte_array = bytearray(len(bool_list) // 8)
    for i, bit in enumerate(bool_list):
        if bit == 1:
            byte_array[i // 8] |= (1 << (i % 8))
    return bytes(byte_array)

# 方法2：numpy.packbits（来源：GitHub #23585）
import numpy as np
raw_vector = [random.randint(0, 1) for i in range(dim)]
binary_vector = bytes(np.packbits(raw_vector, axis=-1).tolist())
```

### 2. HAMMING vs JACCARD

| 特性 | HAMMING | JACCARD |
|------|---------|---------|
| 计算方式 | XOR + popcount | 1 - (交集/并集) |
| 距离范围 | [0, dim] | [0, 1] |
| 适用场景 | 位级差异 | 集合相似度 |
| 典型应用 | 图像哈希 | 标签匹配 |
| 性能 | 极快 | 稍慢 |

### 3. 实际应用场景

**图像去重检测**（pHash）：
```python
import imagehash
from PIL import Image

# 生成 64-bit pHash
img = Image.open("photo.jpg")
hash_value = imagehash.phash(img, hash_size=8)  # 8x8 = 64 bits

# 转换为 Milvus 格式
binary_vector = hash_value.hash.tobytes()

# HAMMING 距离 < 5: 认为是重复图像
```

**文本关键词匹配**：
```python
# 词汇表编码
vocabulary = ["python", "java", "rust", "go", ...]
text = "I love python and rust"

# 生成二值向量
binary_vector = [1 if word in text else 0 for word in vocabulary]
binary_bytes = convert_bool_list_to_bytes(binary_vector)

# JACCARD 距离衡量关键词重叠度
```

### 4. 性能优势

**存储效率**：
- Float32 (128-dim): 512 bytes
- Binary (128-dim): 16 bytes
- **32x 压缩**

**计算速度**：
- HAMMING: 单次 XOR + popcount（CPU 指令级优化）
- 比 float32 距离计算快 10-100x
