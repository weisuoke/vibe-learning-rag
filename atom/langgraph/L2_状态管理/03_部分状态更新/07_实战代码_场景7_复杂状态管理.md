# 07_实战代码_场景7_复杂状态管理

> 演示如何组合使用多种状态更新策略构建复杂的状态化工作流

---

## 场景概述

**学习目标：**
- 掌握多种状态更新策略的组合使用
- 理解复杂状态管理的最佳实践
- 学会构建生产级的状态化工作流

**适用场景：**
- 多步骤的复杂业务流程
- 需要维护多种类型状态的系统
- 需要灵活控制状态更新的应用

**技术要点：**
- 部分状态返回 + Reducer 函数
- Annotated 字段 + Command 对象
- Overwrite 模式 + Pydantic 模型
- 条件路由 + 状态验证

## 完整实战代码

```python
"""
LangGraph 部分状态更新 - 复杂状态管理实战

演示场景：
1. 多策略组合的任务处理系统
2. 智能对话代理状态管理
3. 生产级 RAG 系统状态管理

来源：
- [来源: reference/source_部分状态更新_01.md] - 源码分析
- [来源: reference/context7_langgraph_01.md] - 官方文档
- [来源: reference/search_部分状态更新_01.md] - 最佳实践
"""

import operator
from typing import Annotated, TypedDict, Sequence, Literal
from pydantic import BaseModel, Field
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langgraph.channels.binop import Overwrite
from langgraph.types import Command
from langchain_core.messages import BaseMessage, HumanMessage, AIMessage, SystemMessage

# ===== 场景1：多策略组合的任务处理系统 =====
print("=" * 60)
print("场景1：多策略组合的任务处理系统")
print("=" * 60)

class TaskMetrics(BaseModel):
    """任务指标（Pydantic 模型）"""
    total_tasks: int = 0
    completed_tasks: int = 0
    failed_tasks: int = 0

class TaskState(TypedDict):
    """任务处理状态 - 组合多种更新策略"""
    # 1. 使用 add_messages Reducer 维护消息历史
    messages: Annotated[Sequence[BaseMessage], add_messages]

    # 2. 使用 operator.add Reducer 累积任务列表
    tasks: Annotated[list[str], operator.add]

    # 3. 使用 operator.add Reducer 累积错误
    errors: Annotated[list[str], operator.add]

    # 4. 普通字段（覆盖策略）
    current_task: str
    status: Literal["idle", "processing", "completed", "failed"]

    # 5. Pydantic 模型（部分更新）
    metrics: TaskMetrics

def initialize_system(state: TaskState) -> dict:
    """初始化系统"""
    print("  [initialize_system] 初始化任务处理系统")
    return {
        "messages": [SystemMessage(content="Task processing system initialized")],
        "status": "idle",
        "metrics": TaskMetrics(total_tasks=0, completed_tasks=0, failed_tasks=0)
    }

def add_task(state: TaskState) -> dict:
    """添加任务"""
    task_id = f"task_{state['metrics'].total_tasks + 1}"
    print(f"  [add_task] 添加任务: {task_id}")

    # 更新多个字段，使用不同策略
    return {
        "tasks": [task_id],  # 使用 Reducer 追加
        "messages": [HumanMessage(content=f"Added {task_id}")],  # 使用 Reducer 追加
        "current_task": task_id,  # 覆盖
        "status": "processing",  # 覆盖
        "metrics": TaskMetrics(total_tasks=state['metrics'].total_tasks + 1)  # Pydantic 部分更新
    }

def process_task(state: TaskState) -> dict:
    """处理任务"""
    task = state["current_task"]
    print(f"  [process_task] 处理任务: {task}")

    # 模拟任务处理
    import random
    success = random.choice([True, True, False])  # 66% 成功率

    if success:
        return {
            "messages": [AIMessage(content=f"Task {task} completed successfully")],
            "metrics": TaskMetrics(
                total_tasks=state['metrics'].total_tasks,
                completed_tasks=state['metrics'].completed_tasks + 1
            )
        }
    else:
        return {
            "messages": [AIMessage(content=f"Task {task} failed")],
            "errors": [f"Error in {task}"],
            "metrics": TaskMetrics(
                total_tasks=state['metrics'].total_tasks,
                failed_tasks=state['metrics'].failed_tasks + 1
            )
        }

def check_status(state: TaskState) -> Command[Literal["add_task", "reset", "end"]]:
    """检查状态并决定下一步"""
    total = state['metrics'].total_tasks
    completed = state['metrics'].completed_tasks
    failed = state['metrics'].failed_tasks

    print(f"  [check_status] 总任务: {total}, 完成: {completed}, 失败: {failed}")

    # 如果失败太多，重置系统
    if failed >= 2:
        print(f"  [check_status] 失败过多，重置系统")
        return Command(
            goto="reset",
            update={
                "status": "failed",
                "messages": [SystemMessage(content="Too many failures, resetting system")]
            }
        )

    # 如果还没处理够 5 个任务，继续添加
    if total < 5:
        return Command(goto="add_task")

    # 否则结束
    return Command(
        goto="end",
        update={"status": "completed"}
    )

def reset_system(state: TaskState) -> dict:
    """重置系统 - 使用 Overwrite 清空累积状态"""
    print(f"  [reset_system] 重置系统")
    return {
        "tasks": Overwrite([]),  # 清空任务列表
        "errors": Overwrite([]),  # 清空错误列表
        "messages": Overwrite([SystemMessage(content="System reset")]),  # 清空消息
        "status": "idle",
        "current_task": "",
        "metrics": TaskMetrics()  # 重置指标
    }

# 构建图
builder = StateGraph(TaskState)
builder.add_node("init", initialize_system)
builder.add_node("add_task", add_task)
builder.add_node("process", process_task)
builder.add_node("check", check_status)
builder.add_node("reset", reset_system)

builder.add_edge(START, "init")
builder.add_edge("init", "add_task")
builder.add_edge("add_task", "process")
builder.add_edge("process", "check")
builder.add_edge("reset", "end")

graph = builder.compile()

# 执行图
print("\n执行流程：")
result = graph.invoke({
    "messages": [],
    "tasks": [],
    "errors": [],
    "current_task": "",
    "status": "idle",
    "metrics": TaskMetrics()
})

print(f"\n最终结果：")
print(f"  状态: {result['status']}")
print(f"  任务列表: {result['tasks']}")
print(f"  错误列表: {result['errors']}")
print(f"  指标: total={result['metrics'].total_tasks}, "
      f"completed={result['metrics'].completed_tasks}, "
      f"failed={result['metrics'].failed_tasks}")
print(f"  消息数: {len(result['messages'])}")

# ===== 场景2：智能对话代理状态管理 =====
print("\n" + "=" * 60)
print("场景2：智能对话代理状态管理")
print("=" * 60)

class ConversationMetadata(BaseModel):
    """对话元数据"""
    user_id: str = ""
    session_id: str = ""
    turn_count: int = 0
    last_intent: str = ""

class AgentState(TypedDict):
    """对话代理状态"""
    messages: Annotated[Sequence[BaseMessage], add_messages]
    context: Annotated[list[str], operator.add]  # 上下文信息累积
    metadata: ConversationMetadata
    should_continue: bool
    max_turns: int

def start_conversation(state: AgentState) -> dict:
    """开始对话"""
    print("  [start_conversation] 开始新对话")
    return {
        "messages": [SystemMessage(content="Hello! How can I help you?")],
        "metadata": ConversationMetadata(
            user_id="user_123",
            session_id="session_456",
            turn_count=0
        ),
        "should_continue": True,
        "max_turns": 5
    }

def process_user_input(state: AgentState) -> dict:
    """处理用户输入"""
    turn = state['metadata'].turn_count + 1
    user_msg = f"User question {turn}"
    print(f"  [process_user_input] 处理: {user_msg}")

    return {
        "messages": [HumanMessage(content=user_msg)],
        "context": [f"Context from turn {turn}"],
        "metadata": ConversationMetadata(
            user_id=state['metadata'].user_id,
            session_id=state['metadata'].session_id,
            turn_count=turn,
            last_intent=f"intent_{turn}"
        )
    }

def generate_response(state: AgentState) -> dict:
    """生成回复"""
    turn = state['metadata'].turn_count
    response = f"AI response to turn {turn}"
    print(f"  [generate_response] 生成: {response}")

    return {
        "messages": [AIMessage(content=response)]
    }

def check_continuation(state: AgentState) -> Command[Literal["process", "summarize", "end"]]:
    """检查是否继续对话"""
    turn = state['metadata'].turn_count
    max_turns = state['max_turns']

    print(f"  [check_continuation] 轮次: {turn}/{max_turns}")

    if turn >= max_turns:
        print(f"  [check_continuation] 达到最大轮次，总结对话")
        return Command(goto="summarize")
    else:
        return Command(goto="process")

def summarize_conversation(state: AgentState) -> dict:
    """总结对话"""
    print(f"  [summarize_conversation] 总结对话")
    summary = f"Conversation summary: {len(state['messages'])} messages, {len(state['context'])} context items"

    return {
        "messages": [SystemMessage(content=summary)],
        "should_continue": False
    }

# 构建图
builder = StateGraph(AgentState)
builder.add_node("start", start_conversation)
builder.add_node("process", process_user_input)
builder.add_node("generate", generate_response)
builder.add_node("check", check_continuation)
builder.add_node("summarize", summarize_conversation)

builder.add_edge(START, "start")
builder.add_edge("start", "process")
builder.add_edge("process", "generate")
builder.add_edge("generate", "check")
builder.add_edge("summarize", END)

graph = builder.compile()

# 执行图
print("\n执行流程：")
result = graph.invoke({
    "messages": [],
    "context": [],
    "metadata": ConversationMetadata(),
    "should_continue": True,
    "max_turns": 3
})

print(f"\n最终结果：")
print(f"  轮次: {result['metadata'].turn_count}")
print(f"  消息数: {len(result['messages'])}")
print(f"  上下文数: {len(result['context'])}")
print(f"  最后意图: {result['metadata'].last_intent}")

# ===== 场景3：生产级 RAG 系统状态管理 =====
print("\n" + "=" * 60)
print("场景3：生产级 RAG 系统状态管理")
print("=" * 60)

class RetrievalMetrics(BaseModel):
    """检索指标"""
    query_count: int = 0
    total_docs_retrieved: int = 0
    avg_relevance_score: float = 0.0

class RAGState(TypedDict):
    """RAG 系统状态"""
    # 对话历史
    messages: Annotated[Sequence[BaseMessage], add_messages]

    # 检索到的文档（累积）
    retrieved_docs: Annotated[list[str], operator.add]

    # 查询历史（累积）
    query_history: Annotated[list[str], operator.add]

    # 当前状态
    current_query: str
    current_answer: str

    # 指标
    metrics: RetrievalMetrics

    # 控制标志
    needs_retrieval: bool
    needs_rerank: bool

def receive_query(state: RAGState) -> dict:
    """接收查询"""
    query = "What is LangGraph?"
    print(f"  [receive_query] 接收查询: {query}")

    return {
        "messages": [HumanMessage(content=query)],
        "query_history": [query],
        "current_query": query,
        "needs_retrieval": True,
        "metrics": RetrievalMetrics(
            query_count=state['metrics'].query_count + 1 if 'metrics' in state else 1
        )
    }

def retrieve_documents(state: RAGState) -> dict:
    """检索文档"""
    query = state['current_query']
    print(f"  [retrieve_documents] 检索文档: {query}")

    # 模拟检索
    docs = [f"Doc {i} about {query}" for i in range(3)]

    return {
        "retrieved_docs": docs,
        "needs_retrieval": False,
        "needs_rerank": True,
        "metrics": RetrievalMetrics(
            query_count=state['metrics'].query_count,
            total_docs_retrieved=state['metrics'].total_docs_retrieved + len(docs)
        )
    }

def rerank_documents(state: RAGState) -> dict:
    """重排序文档"""
    print(f"  [rerank_documents] 重排序文档")

    # 模拟重排序（保留前2个）
    top_docs = state['retrieved_docs'][-2:]

    return {
        "retrieved_docs": Overwrite(top_docs),  # 使用 Overwrite 替换为重排序后的文档
        "needs_rerank": False,
        "metrics": RetrievalMetrics(
            query_count=state['metrics'].query_count,
            total_docs_retrieved=len(top_docs),
            avg_relevance_score=0.85
        )
    }

def generate_answer(state: RAGState) -> dict:
    """生成答案"""
    query = state['current_query']
    docs = state['retrieved_docs']
    print(f"  [generate_answer] 生成答案，使用 {len(docs)} 个文档")

    answer = f"Based on {len(docs)} documents: LangGraph is a framework for building stateful workflows."

    return {
        "messages": [AIMessage(content=answer)],
        "current_answer": answer
    }

def route_next(state: RAGState) -> Literal["retrieve", "rerank", "generate", "end"]:
    """路由到下一步"""
    if state['needs_retrieval']:
        return "retrieve"
    elif state['needs_rerank']:
        return "rerank"
    else:
        return "generate"

# 构建图
builder = StateGraph(RAGState)
builder.add_node("query", receive_query)
builder.add_node("retrieve", retrieve_documents)
builder.add_node("rerank", rerank_documents)
builder.add_node("generate", generate_answer)

builder.add_edge(START, "query")
builder.add_conditional_edges("query", route_next)
builder.add_conditional_edges("retrieve", route_next)
builder.add_conditional_edges("rerank", route_next)
builder.add_edge("generate", END)

graph = builder.compile()

# 执行图
print("\n执行流程：")
result = graph.invoke({
    "messages": [],
    "retrieved_docs": [],
    "query_history": [],
    "current_query": "",
    "current_answer": "",
    "metrics": RetrievalMetrics(),
    "needs_retrieval": False,
    "needs_rerank": False
})

print(f"\n最终结果：")
print(f"  查询: {result['current_query']}")
print(f"  答案: {result['current_answer']}")
print(f"  检索文档数: {len(result['retrieved_docs'])}")
print(f"  查询历史: {result['query_history']}")
print(f"  指标: queries={result['metrics'].query_count}, "
      f"docs={result['metrics'].total_docs_retrieved}, "
      f"relevance={result['metrics'].avg_relevance_score}")

print("\n" + "=" * 60)
print("总结")
print("=" * 60)
print("""
复杂状态管理的关键要点：

1. **多策略组合**：
   - add_messages: 维护对话历史
   - operator.add: 累积列表数据
   - 覆盖策略: 更新单值字段
   - Pydantic 模型: 结构化数据部分更新
   - Overwrite: 显式重置累积状态

2. **状态设计原则**：
   - 明确每个字段的更新策略
   - 使用 Annotated 显式声明 Reducer
   - 使用 Pydantic 模型管理复杂结构
   - 合理使用 Overwrite 重置状态

3. **控制流管理**：
   - 使用 Command 对象同时更新状态和控制流程
   - 使用条件路由实现动态决策
   - 使用状态标志控制执行路径

4. **最佳实践**：
   - 保持状态结构清晰
   - 避免过度使用 Overwrite
   - 使用 Pydantic 模型验证数据
   - 合理拆分节点职责
   - 添加详细的日志和监控

5. **实际应用场景**：
   - 任务处理系统: 多状态协同管理
   - 对话代理: 历史和上下文维护
   - RAG 系统: 检索和生成流程控制
```

## 技术要点总结

### 1. 状态更新策略组合

| 策略 | 使用场景 | 示例 |
|------|----------|------|
| add_messages | 对话历史 | `messages: Annotated[Sequence[BaseMessage], add_messages]` |
| operator.add | 列表累积 | `tasks: Annotated[list[str], operator.add]` |
| 覆盖策略 | 单值更新 | `status: str` |
| Pydantic 模型 | 结构化数据 | `metrics: TaskMetrics` |
| Overwrite | 显式重置 | `Overwrite([])` |

### 2. 状态设计模式

**分层设计：**
- 对话层：messages（历史记录）
- 数据层：retrieved_docs, tasks（累积数据）
- 控制层：status, needs_retrieval（控制标志）
- 指标层：metrics（统计信息）

**更新策略选择：**
- 需要累积 → 使用 Reducer（add_messages, operator.add）
- 需要替换 → 使用覆盖策略
- 需要重置 → 使用 Overwrite
- 需要验证 → 使用 Pydantic 模型

### 3. 控制流管理

**Command 对象：**
```python
return Command(
    goto="next_node",
    update={"field": "value"}
)
```

**条件路由：**
```python
def route(state):
    if condition:
        return "node_a"
    else:
        return "node_b"
```

### 4. 最佳实践

1. **明确更新策略**：每个字段都应该有明确的更新策略
2. **使用类型注解**：使用 Annotated 显式声明 Reducer
3. **结构化数据**：使用 Pydantic 模型管理复杂结构
4. **谨慎使用 Overwrite**：只在必要时使用，避免破坏状态一致性
5. **添加日志**：在关键节点添加日志，便于调试和监控
