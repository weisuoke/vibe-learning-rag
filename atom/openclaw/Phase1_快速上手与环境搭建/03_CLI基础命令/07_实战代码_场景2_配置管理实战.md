```typescript
// src/config/config-manager.ts
import { exec } from 'child_process';
import { promisify } from 'util';
import * as fs from 'fs/promises';
import * as path from 'path';

const execAsync = promisify(exec);

export interface OpenClawConfig {
  provider?: string;
  model?: string;
  workspace?: string;
  gateway?: {
    port?: number;
    verbose?: boolean;
    autoStart?: boolean;
  };
  agent?: {
    thinking?: 'low' | 'medium' | 'high';
    maxTokens?: number;
    temperature?: number;
  };
  channels?: Record<string, any>;
  logging?: {
    level?: 'debug' | 'info' | 'warn' | 'error';
  };
}

export class ConfigManager {
  private configPath: string;
  private profile: string;

  constructor(profile = 'default') {
    this.profile = profile;
    const homeDir = process.env.HOME || '';
    const configDir = profile === 'default' ? '.openclaw' : `.openclaw-${profile}`;
    this.configPath = path.join(homeDir, configDir, 'config', 'settings.json');
  }

  async get(key?: string): Promise<any> {
    if (key) {
      const { stdout } = await execAsync(`openclaw config get ${key}`);
      return stdout.trim();
    }
    const { stdout } = await execAsync('openclaw config get');
    return JSON.parse(stdout);
  }

  async set(key: string, value: any): Promise<void> {
    const valueStr = typeof value === 'string' ? value : JSON.stringify(value);
    await execAsync(`openclaw config set ${key} ${valueStr}`);
  }

  async readConfigFile(): Promise<OpenClawConfig> {
    const content = await fs.readFile(this.configPath, 'utf-8');
    return JSON.parse(content);
  }

  async writeConfigFile(config: OpenClawConfig): Promise<void> {
    await fs.writeFile(this.configPath, JSON.stringify(config, null, 2));
  }

  async validate(): Promise<{ valid: boolean; errors: string[] }> {
    const errors: string[] = [];
    const config = await this.readConfigFile();

    const validProviders = ['anthropic', 'openai', 'google'];
    if (config.provider && !validProviders.includes(config.provider)) {
      errors.push(`Invalid provider: ${config.provider}`);
    }

    if (config.gateway?.port) {
      const port = config.gateway.port;
      if (port < 1024 || port > 65535) {
        errors.push(`Invalid port: ${port}`);
      }
    }

    return { valid: errors.length === 0, errors };
  }
}

// src/config/environment-manager.ts
export class EnvironmentManager {
  private environments = ['development', 'staging', 'production'];

  async setupEnvironment(env: string): Promise<void> {
    const manager = new ConfigManager(env);
    const config = this.getEnvironmentConfig(env);
    await manager.writeConfigFile(config);
    console.log(`✅ Environment ${env} configured`);
  }

  private getEnvironmentConfig(env: string): OpenClawConfig {
    const configs: Record<string, OpenClawConfig> = {
      development: {
        provider: 'anthropic',
        model: 'claude-haiku',
        gateway: { port: 18789, verbose: true },
        agent: { thinking: 'low', maxTokens: 2048 },
        logging: { level: 'debug' },
      },
      staging: {
        provider: 'anthropic',
        model: 'claude-sonnet',
        gateway: { port: 18790, verbose: false },
        agent: { thinking: 'medium', maxTokens: 4096 },
        logging: { level: 'info' },
      },
      production: {
        provider: 'anthropic',
        model: 'claude-opus-4',
        gateway: { port: 18791, verbose: false, autoStart: true },
        agent: { thinking: 'high', maxTokens: 8192 },
        logging: { level: 'warn' },
      },
    };
    return configs[env] || configs.development;
  }

  async switchEnvironment(from: string, to: string): Promise<void> {
    console.log(`🔄 Switching from ${from} to ${to}...`);
    await this.setupEnvironment(to);
    console.log(`✅ Switched to ${to}`);
  }
}

// src/config/config-validator.ts
export class ConfigValidator {
  async validateAll(): Promise<void> {
    const manager = new ConfigManager();
    const { valid, errors } = await manager.validate();

    if (!valid) {
      console.error('❌ Configuration validation failed:');
      errors.forEach(err => console.error(`  - ${err}`));
      throw new Error('Invalid configuration');
    }

    console.log('✅ Configuration is valid');
  }

  async checkRequiredFields(): Promise<void> {
    const manager = new ConfigManager();
    const config = await manager.readConfigFile();

    const required = ['provider', 'model'];
    const missing = required.filter(field => !config[field as keyof OpenClawConfig]);

    if (missing.length > 0) {
      throw new Error(`Missing required fields: ${missing.join(', ')}`);
    }
  }
}
```

## 使用示例

### 示例 1: 多环境配置

```typescript
// examples/multi-environment.ts
import { EnvironmentManager, ConfigManager } from './config';

async function main() {
  const envManager = new EnvironmentManager();

  // 设置所有环境
  await envManager.setupEnvironment('development');
  await envManager.setupEnvironment('staging');
  await envManager.setupEnvironment('production');

  // 切换环境
  await envManager.switchEnvironment('development', 'production');

  // 验证配置
  const manager = new ConfigManager('production');
  const config = await manager.readConfigFile();
  console.log('Production config:', config);
}

main().catch(console.error);
```

### 示例 2: 配置验证

```typescript
// examples/config-validation.ts
import { ConfigValidator } from './config';

async function main() {
  const validator = new ConfigValidator();

  try {
    await validator.checkRequiredFields();
    await validator.validateAll();
    console.log('✅ All validations passed');
  } catch (error) {
    console.error('❌ Validation failed:', error.message);
    process.exit(1);
  }
}

main().catch(console.error);
```

### 示例 3: 配置迁移

```typescript
// examples/config-migration.ts
import { ConfigManager } from './config';

async function migrateConfig() {
  const manager = new ConfigManager();
  const config = await manager.readConfigFile();

  // 迁移逻辑：添加新字段
  if (!config.agent) {
    config.agent = {
      thinking: 'medium',
      maxTokens: 4096,
      temperature: 0.7,
    };
  }

  await manager.writeConfigFile(config);
  console.log('✅ Configuration migrated');
}

migrateConfig().catch(console.error);
```

  const manager = new ConfigManager();

}

async function safeSetConfig(key: string, value: any) {
  const manager = new ConfigManager();

  const validator = new ConfigValidator();
  await validator.validateAll();
}
