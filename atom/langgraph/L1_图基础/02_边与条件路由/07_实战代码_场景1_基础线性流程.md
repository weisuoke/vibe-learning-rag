# 实战代码 - 场景1：基础线性流程

> 本文档演示 LangGraph 中最基础的线性流程，使用普通边连接节点

---

## 场景描述

**场景**：构建一个简单的数据处理管道，按照固定顺序执行：加载数据 → 处理数据 → 保存数据。

**特点**：
- 固定的执行顺序
- 使用普通边（`add_edge`）
- 线性流程，无分支
- 适合简单的 ETL 任务

## 完整代码

```python
"""
LangGraph 实战代码 - 场景1：基础线性流程
演示：使用普通边构建简单的数据处理管道
"""

from langgraph.graph import StateGraph, START, END
from typing_extensions import TypedDict


# ===== 1. 定义状态 =====
class DataState(TypedDict):
    """数据处理状态"""
    raw_data: str          # 原始数据
    processed_data: str    # 处理后的数据
    saved: bool            # 是否已保存
    log: list[str]         # 操作日志


# ===== 2. 定义节点函数 =====
def load_data(state: DataState) -> dict:
    """加载数据节点"""
    print("📥 [加载数据] 开始加载数据...")

    # 模拟从数据源加载数据
    raw_data = "用户输入的原始文本数据"

    print(f"📥 [加载数据] 加载完成: {raw_data}")

    return {
        "raw_data": raw_data,
        "log": state.get("log", []) + ["数据加载完成"]
    }


def process_data(state: DataState) -> dict:
    """处理数据节点"""
    print("⚙️  [处理数据] 开始处理数据...")

    # 模拟数据处理（转换为大写）
    raw_data = state["raw_data"]
    processed_data = raw_data.upper()

    print(f"⚙️  [处理数据] 处理完成: {processed_data}")

    return {
        "processed_data": processed_data,
        "log": state.get("log", []) + ["数据处理完成"]
    }


def save_data(state: DataState) -> dict:
    """保存数据节点"""
    print("💾 [保存数据] 开始保存数据...")

    # 模拟保存数据到数据库
    processed_data = state["processed_data"]
    print(f"💾 [保存数据] 保存到数据库: {processed_data}")

    print("💾 [保存数据] 保存完成")

    return {
        "saved": True,
        "log": state.get("log", []) + ["数据保存完成"]
    }


# ===== 3. 构建图 =====
def create_linear_graph():
    """创建线性流程图"""
    # 创建状态图
    graph = StateGraph(DataState)

    # 添加节点
    graph.add_node("load", load_data)
    graph.add_node("process", process_data)
    graph.add_node("save", save_data)

    # 添加边（线性流程）
    graph.add_edge(START, "load")       # START → load
    graph.add_edge("load", "process")   # load → process
    graph.add_edge("process", "save")   # process → save
    graph.add_edge("save", END)         # save → END

    # 编译图
    app = graph.compile()

    return app


# ===== 4. 执行图 =====
def main():
    """主函数"""
    print("=" * 60)
    print("LangGraph 实战 - 场景1：基础线性流程")
    print("=" * 60)
    print()

    # 创建图
    app = create_linear_graph()

    # 初始状态
    initial_state = {
        "raw_data": "",
        "processed_data": "",
        "saved": False,
        "log": []
    }

    # 执行图
    print("🚀 开始执行线性流程...")
    print()

    result = app.invoke(initial_state)

    # 输出结果
    print()
    print("=" * 60)
    print("✅ 执行完成！")
    print("=" * 60)
    print()
    print("📊 最终状态：")
    print(f"  原始数据: {result['raw_data']}")
    print(f"  处理后数据: {result['processed_data']}")
    print(f"  是否已保存: {result['saved']}")
    print()
    print("📝 操作日志：")
    for i, log_entry in enumerate(result['log'], 1):
        print(f"  {i}. {log_entry}")
    print()


if __name__ == "__main__":
    main()
```

### 2. 节点函数

```python
def load_data(state: DataState) -> dict:
    """加载数据节点"""
```

**关键点**：
- 节点函数接受 `state` 参数
- 返回字典，包含需要更新的状态字段

- 使用 `state.get("log", [])` 安全获取列表

## 执行流程图

```
START
  ↓
load (加载数据)
  ↓
process (处理数据)
  ↓
save (保存数据)
  ↓
END
```

**执行顺序**：
1. START → load：图开始执行，进入 load 节点
2. load → process：load 完成后，自动进入 process 节点
3. process → save：process 完成后，自动进入 save 节点
4. save → END：save 完成后，图执行结束

## 实战应用场景

### 场景 1：文档处理管道

```python
class DocState(TypedDict):
    file_path: str
    content: str
    chunks: list[str]
    embeddings: list[list[float]]

def load_document(state: DocState) -> dict:
    """加载文档"""
    with open(state["file_path"], "r") as f:
        content = f.read()
    return {"content": content}

def chunk_document(state: DocState) -> dict:
    """分块"""
    chunks = [state["content"][i:i+500] for i in range(0, len(state["content"]), 500)]
    return {"chunks": chunks}

def embed_chunks(state: DocState) -> dict:
    """向量化"""
    # 模拟向量化
    embeddings = [[0.1] * 768 for _ in state["chunks"]]
    return {"embeddings": embeddings}

# 构建图
graph = StateGraph(DocState)
graph.add_node("load", load_document)
graph.add_node("chunk", chunk_document)
graph.add_node("embed", embed_chunks)

graph.add_edge(START, "load")
graph.add_edge("load", "chunk")
graph.add_edge("chunk", "embed")
graph.add_edge("embed", END)
```

**应用**：RAG 系统的文档预处理流程

### 场景 2：数据清洗管道

```python
class CleanState(TypedDict):
    raw_data: list[dict]
    validated_data: list[dict]
    cleaned_data: list[dict]
    deduplicated_data: list[dict]

def validate(state: CleanState) -> dict:
    """验证数据"""
    validated = [d for d in state["raw_data"] if d.get("id")]
    return {"validated_data": validated}

def clean(state: CleanState) -> dict:
    """清洗数据"""
    cleaned = [{k: v.strip() if isinstance(v, str) else v
                for k, v in d.items()}
               for d in state["validated_data"]]
    return {"cleaned_data": cleaned}

def deduplicate(state: CleanState) -> dict:
    """去重"""
    seen = set()
    deduplicated = []
    for d in state["cleaned_data"]:
        if d["id"] not in seen:
            seen.add(d["id"])
            deduplicated.append(d)
    return {"deduplicated_data": deduplicated}

# 构建图
graph = StateGraph(CleanState)
graph.add_node("validate", validate)
graph.add_node("clean", clean)
graph.add_node("deduplicate", deduplicate)

graph.add_edge(START, "validate")
graph.add_edge("validate", "clean")
graph.add_edge("clean", "deduplicate")
graph.add_edge("deduplicate", END)
```

**应用**：数据清洗和预处理流程

```python
class APIState(TypedDict):
    request_data: dict
    validated_request: dict
    api_response: dict
    formatted_response: dict

def validate_request(state: APIState) -> dict:
    """验证请求"""
    # 验证逻辑
    return {"validated_request": state["request_data"]}

def call_api(state: APIState) -> dict:
    """调用 API"""
    # API 调用
    response = {"status": "success", "data": "..."}
    return {"api_response": response}

def format_response(state: APIState) -> dict:
    """格式化响应"""
    formatted = {

graph = StateGraph(APIState)
graph.add_node("validate", validate_request)
```

## 最佳实践

### 1. 节点职责单一

```python
# ✅ 推荐：每个节点只做一件事
def load_data(state):
    """只负责加载数据"""
    return {"raw_data": load_from_source()}

def validate_data(state):
    """只负责验证数据"""
    return {"is_valid": validate(state["raw_data"])}

# ❌ 不推荐：一个节点做多件事
def load_and_validate(state):
    """加载和验证混在一起"""
    data = load_from_source()
    is_valid = validate(data)
    return {"raw_data": data, "is_valid": is_valid}
```

---

### 2. 使用日志记录

```python
def process_data(state: DataState) -> dict:
    """处理数据，记录日志"""
    print(f"[process_data] 输入: {state['raw_data']}")

    return {
        "processed_data": processed,
```

### 3. 错误处理

```python
def safe_process(state: DataState) -> dict:
    """安全的数据处理"""
    try:
        processed = state["raw_data"].upper()
        return {
            "processed_data": processed,
            "error": None
        }
    except Exception as e:
        return {
            "processed_data": "",
            "error": str(e)
        }
```

```python
def validate_state(state: DataState) -> dict:

    if not state.get("raw_data"):
```

## 总结

**基础线性流程的特点**：
1. **固定顺序**：节点按照固定顺序执行
2. **使用普通边**：使用 `add_edge` 连接节点
3. **无分支**：没有条件判断，直线执行
4. **简单清晰**：易于理解和维护

**适用场景**：
- ETL 数据处理
- 文档预处理
- API 请求处理
- 数据清洗管道
- 简单的工作流

**关键代码**：
```python
# 添加节点
graph.add_node("node1", func1)
graph.add_node("node2", func2)
graph.add_node("node3", func3)
```
