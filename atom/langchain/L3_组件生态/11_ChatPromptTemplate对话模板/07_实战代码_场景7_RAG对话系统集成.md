## 完整实战：RAG 对话系统

### 代码实现

```python
"""
完整实战：RAG 对话系统
演示如何使用 ChatPromptTemplate 构建完整的 RAG 对话系统
"""

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_community.vectorstores import Chroma
from langchain_core.documents import Document
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough
from typing import Dict, List
from dotenv import load_dotenv
import os

load_dotenv()


class RAGChatSystem:
    """RAG 对话系统"""

    def __init__(
        self,
        collection_name: str = "rag_docs",
        persist_directory: str = "./chroma_db"
    ):
        """初始化 RAG 对话系统"""
        self.collection_name = collection_name
        self.persist_directory = persist_directory
        self.store: Dict[str, InMemoryChatMessageHistory] = {}

        # 初始化组件
        self.embeddings = OpenAIEmbeddings()
        self.llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        self.vectorstore = None
        self.conversation = None

        # 设置提示词模板
        self._setup_prompt()

    def _setup_prompt(self):
        """设置 RAG 提示词模板"""
        self.prompt = ChatPromptTemplate.from_messages([
            ("system", """You are a helpful AI assistant with access to a knowledge base.

Your task:
1. Use the provided context to answer questions accurately
2. Remember the conversation history
3. If the context doesn't contain the answer, say so clearly
4. Be concise and cite relevant information from the context
5. Ask clarifying questions if needed

Context from knowledge base:
{context}"""),
            MessagesPlaceholder(variable_name="history"),
            ("human", "{question}")
        ])

    def _get_session_history(self, session_id: str) -> InMemoryChatMessageHistory:
        """获取会话历史"""
        if session_id not in self.store:
            self.store[session_id] = InMemoryChatMessageHistory()
        return self.store[session_id]

    def load_documents(self, documents: List[Document]):
        """加载文档到向量存储"""
        print(f"Loading {len(documents)} documents into vector store...")

        # 创建或更新向量存储
        self.vectorstore = Chroma.from_documents(
            documents=documents,
            embedding=self.embeddings,
            collection_name=self.collection_name,
            persist_directory=self.persist_directory
        )

        # 创建检索器
        retriever = self.vectorstore.as_retriever(
            search_type="similarity",
            search_kwargs={"k": 3}
        )

        # 构建 RAG 链
        def format_docs(docs):
            """格式化检索到的文档"""
            return "\n\n".join([
                f"[Document {i+1}]\n{doc.page_content}"
                for i, doc in enumerate(docs)
            ])

        # 创建 RAG 链
        rag_chain = (
            {
                "context": retriever | format_docs,
                "question": RunnablePassthrough()
            }
            | self.prompt
            | self.llm
            | StrOutputParser()
        )

        # 包装为带历史的对话链
        self.conversation = RunnableWithMessageHistory(
            rag_chain,
            self._get_session_history,
            input_messages_key="question",
            history_messages_key="history"
        )

        print("✓ Documents loaded and RAG system ready!")

    def chat(self, user_id: str, question: str) -> str:
        """发送问题并获取答案"""
        if self.conversation is None:
            raise ValueError("Please load documents first using load_documents()")

        response = self.conversation.invoke(
            question,
            config={"configurable": {"session_id": user_id}}
        )
        return response

    def get_history(self, user_id: str) -> List:
        """获取用户的对话历史"""
        if user_id in self.store:
            return self.store[user_id].messages
        return []

    def clear_history(self, user_id: str):
        """清空用户的对话历史"""
        if user_id in self.store:
            self.store[user_id].clear()
            print(f"✓ History cleared for user: {user_id}")

    def search_similar(self, query: str, k: int = 3) -> List[Document]:
        """搜索相似文档"""
        if self.vectorstore is None:
            raise ValueError("Please load documents first")
        return self.vectorstore.similarity_search(query, k=k)


def demo_rag_chat_system():
    """演示 RAG 对话系统"""
    print("=== RAG 对话系统演示 ===\n")

    # 1. 创建 RAG 系统
    rag_system = RAGChatSystem()

    # 2. 准备知识库文档
    documents = [
        Document(
            page_content="""LangChain is a framework for developing applications powered by language models.
It was created in October 2022 by Harrison Chase. The framework provides tools for:
- Prompt management and optimization
- Chains for combining multiple LLM calls
- Agents for dynamic decision-making
- Memory for maintaining conversation context
- Retrieval for accessing external data""",
            metadata={"source": "langchain_intro.txt", "topic": "overview"}
        ),
        Document(
            page_content="""ChatPromptTemplate is a core component in LangChain for managing conversation prompts.
It supports:
- Multiple message types (system, human, AI)
- MessagesPlaceholder for dynamic history injection
- Template variables for dynamic content
- Partial variables for pre-filled values
- Integration with LCEL (LangChain Expression Language)""",
            metadata={"source": "prompt_templates.txt", "topic": "prompts"}
        ),
        Document(
            page_content="""RAG (Retrieval-Augmented Generation) combines retrieval and generation:
1. Document Loading: Load documents from various sources
2. Text Splitting: Split documents into chunks
3. Embedding: Convert chunks to vector embeddings
4. Vector Store: Store embeddings in a vector database
5. Retrieval: Find relevant chunks for a query
6. Generation: Generate answers using retrieved context""",
            metadata={"source": "rag_guide.txt", "topic": "rag"}
        ),
        Document(
            page_content="""Vector stores in LangChain support multiple backends:
- Chroma: Lightweight, open-source
- Pinecone: Managed, scalable
- Weaviate: Open-source with GraphQL
- FAISS: Facebook's similarity search library
All support similarity search and metadata filtering.""",
            metadata={"source": "vector_stores.txt", "topic": "storage"}
        )
    ]

    # 3. 加载文档
    rag_system.load_documents(documents)

    # 4. 用户对话
    user_id = "user_001"

    print("\n--- 对话开始 ---\n")

    # 第一个问题
    question1 = "What is LangChain?"
    print(f"User: {question1}")
    answer1 = rag_system.chat(user_id, question1)
    print(f"AI: {answer1}\n")

    # 第二个问题（引用历史）
    question2 = "When was it created?"
    print(f"User: {question2}")
    answer2 = rag_system.chat(user_id, question2)
    print(f"AI: {answer2}\n")

    # 第三个问题（新话题）
    question3 = "What is RAG?"
    print(f"User: {question3}")
    answer3 = rag_system.chat(user_id, question3)
    print(f"AI: {answer3}\n")

    # 第四个问题（结合历史和新上下文）
    question4 = "How does it relate to LangChain?"
    print(f"User: {question4}")
    answer4 = rag_system.chat(user_id, question4)
    print(f"AI: {answer4}\n")

    # 5. 查看对话历史
    print("--- 对话历史 ---")
    history = rag_system.get_history(user_id)
    print(f"Total messages: {len(history)}")
    for i, msg in enumerate(history, 1):
        role = "User" if msg.__class__.__name__ == "HumanMessage" else "AI"
        content = msg.content[:60] + "..." if len(msg.content) > 60 else msg.content
        print(f"  [{i}] {role}: {content}")

    # 6. 测试相似度搜索
    print("\n--- 相似度搜索测试 ---")
    query = "prompt templates"
    similar_docs = rag_system.search_similar(query, k=2)
    print(f"Query: '{query}'")
    print(f"Found {len(similar_docs)} similar documents:")
    for i, doc in enumerate(similar_docs, 1):
        print(f"\n[{i}] Source: {doc.metadata.get('source', 'unknown')}")
        print(f"    Content: {doc.page_content[:100]}...")


if __name__ == "__main__":
    demo_rag_chat_system()
```

## 代码解析

### 关键点1：RAG 提示词设计

```python
self.prompt = ChatPromptTemplate.from_messages([
    ("system", """You are a helpful AI assistant with access to a knowledge base.

Your task:
1. Use the provided context to answer questions accurately
2. Remember the conversation history
3. If the context doesn't contain the answer, say so clearly
4. Be concise and cite relevant information from the context
5. Ask clarifying questions if needed

Context from knowledge base:
{context}"""),
    MessagesPlaceholder(variable_name="history"),
    ("human", "{question}")
])
```

**设计要点**：
- **System 消息**：明确 AI 的角色和任务
- **Context 变量**：动态注入检索到的文档
- **History 占位符**：保持对话连贯性
- **Question 变量**：用户当前问题

### 关键点2：检索链构建

```python
rag_chain = (
    {
        "context": retriever | format_docs,
        "question": RunnablePassthrough()
    }
    | self.prompt
    | self.llm
    | StrOutputParser()
)
```

**执行流程**：
1. `retriever` 检索相关文档
2. `format_docs` 格式化文档为文本
3. `RunnablePassthrough` 传递原始问题
4. `prompt` 格式化为消息
5. `llm` 生成回答
6. `StrOutputParser` 提取文本

### 关键点3：历史管理集成

```python
self.conversation = RunnableWithMessageHistory(
    rag_chain,
    self._get_session_history,
    input_messages_key="question",
    history_messages_key="history"
)
```

**自动处理**：
- 加载历史消息
- 注入到 MessagesPlaceholder
- 保存新的对话轮次
- 支持多用户隔离

## 最佳实践

### 1. 上下文长度控制

```python
# 限制检索文档数量
retriever = self.vectorstore.as_retriever(
    search_type="similarity",
    search_kwargs={"k": 3}  # 只检索前3个最相关文档
)

# 限制历史消息数量
def get_limited_history(session_id: str, max_messages: int = 10):
    history = self.store[session_id]
    if len(history.messages) > max_messages:
        history.messages = history.messages[-max_messages:]
    return history
```

### 2. 文档格式化优化

```python
def format_docs_with_metadata(docs):
    """带元数据的文档格式化"""
    formatted = []
    for i, doc in enumerate(docs, 1):
        source = doc.metadata.get('source', 'unknown')
        formatted.append(
            f"[Document {i} - Source: {source}]\n{doc.page_content}"
        )
    return "\n\n".join(formatted)
```

### 3. 错误处理

```python
def safe_chat(self, user_id: str, question: str) -> str:
    """带错误处理的对话"""
    try:
        return self.chat(user_id, question)
    except Exception as e:
        error_msg = f"Sorry, I encountered an error: {str(e)}"
        print(f"Error: {e}")
        return error_msg
```

## 常见问题

### Q1: 如何处理检索不到相关文档的情况？

**A:** 在提示词中明确指示：
```python
("system", """If the context doesn't contain relevant information,
say: "I don't have enough information in my knowledge base to answer this question."
Do not make up information.""")
```

### Q2: 如何优化检索质量？

**A:** 使用混合检索策略：
```python
# 结合相似度和 MMR（最大边际相关性）
retriever = vectorstore.as_retriever(
    search_type="mmr",
    search_kwargs={
        "k": 5,
        "fetch_k": 20,
        "lambda_mult": 0.5
    }
)
```

### Q3: 如何支持多语言？

**A:** 在提示词中添加语言指示：
```python
("system", """Answer in the same language as the user's question.
If the context is in a different language, translate it appropriately.""")
```
