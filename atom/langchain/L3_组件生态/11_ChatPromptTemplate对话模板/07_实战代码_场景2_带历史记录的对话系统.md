## 实战1：基础 MessagesPlaceholder 使用

### 代码实现

```python
"""
实战1：基础 MessagesPlaceholder 使用
演示如何使用 MessagesPlaceholder 动态注入对话历史
"""

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage
from dotenv import load_dotenv

load_dotenv()

def demo_messages_placeholder():
    """演示 MessagesPlaceholder 的基础用法"""
    print("=== 实战1：基础 MessagesPlaceholder 使用 ===\n")

    # 1. 创建包含 MessagesPlaceholder 的模板
    template = ChatPromptTemplate.from_messages([
        ("system", "You are a helpful assistant. Remember the conversation context."),
        MessagesPlaceholder("history"),
        ("human", "{input}")
    ])

    print("1. 模板创建成功")
    print(f"   输入变量: {template.input_variables}")

    # 2. 第一轮对话（无历史）
    print("\n2. 第一轮对话（无历史）:")
    messages = template.invoke({
        "history": [],
        "input": "My name is Alice."
    })

    print("   格式化后的消息:")
    for msg in messages.messages:
        print(f"   [{msg.__class__.__name__}] {msg.content}")

    # 3. 第二轮对话（有历史）
    print("\n3. 第二轮对话（有历史）:")
    history = [
        HumanMessage(content="My name is Alice."),
        AIMessage(content="Hello Alice! Nice to meet you. How can I help you today?")
    ]

    messages = template.invoke({
        "history": history,
        "input": "What's my name?"
    })

    print("   格式化后的消息:")
    for msg in messages.messages:
        print(f"   [{msg.__class__.__name__}] {msg.content}")

    # 4. 与 LLM 集成
    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    chain = template | llm

    print("\n4. 调用 LLM:")
    response = chain.invoke({
        "history": history,
        "input": "What's my name?"
    })
    print(f"   回答: {response.content}")

if __name__ == "__main__":
    demo_messages_placeholder()
```

### 代码解析

**关键点1：MessagesPlaceholder 定义**
```python
MessagesPlaceholder("history")
```
- 创建一个名为 "history" 的占位符
- 接受消息列表作为输入
- 自动展开消息到模板中

**关键点2：历史消息格式**
```python
history = [
    HumanMessage(content="..."),
    AIMessage(content="...")
]
```
- 使用消息对象构建历史
- 也可以使用元组格式：`[("human", "..."), ("ai", "...")]`
- 自动转换为正确的消息类型

**关键点3：动态注入**
- 第一轮对话：`history=[]`（空列表）
- 后续对话：`history=[...]`（包含历史消息）
- 灵活控制上下文长度

## 实战2：RunnableWithMessageHistory 自动管理

### 代码实现

```python
"""
实战2：RunnableWithMessageHistory 自动管理
演示如何使用 RunnableWithMessageHistory 自动管理对话历史
"""

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_openai import ChatOpenAI
from dotenv import load_dotenv

load_dotenv()

def demo_runnable_with_history():
    """演示 RunnableWithMessageHistory 的使用"""
    print("=== 实战2：RunnableWithMessageHistory 自动管理 ===\n")

    # 1. 创建存储字典（多用户支持）
    store = {}

    def get_session_history(session_id: str) -> InMemoryChatMessageHistory:
        """获取或创建会话历史"""
        if session_id not in store:
            print(f"   创建新会话: {session_id}")
            store[session_id] = InMemoryChatMessageHistory()
        return store[session_id]

    # 2. 创建对话模板
    prompt = ChatPromptTemplate.from_messages([
        ("system", "You are a helpful assistant."),
        MessagesPlaceholder(variable_name="history"),
        ("human", "{input}")
    ])

    # 3. 创建链
    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    chain = prompt | llm

    # 4. 包装为带历史的链
    conversation = RunnableWithMessageHistory(
        chain,
        get_session_history,
        input_messages_key="input",
        history_messages_key="history"
    )

    print("1. 对话系统初始化完成\n")

    # 5. 用户 Alice 的对话
    print("2. 用户 Alice 的对话:")
    response = conversation.invoke(
        {"input": "Hi, I'm Alice"},
        config={"configurable": {"session_id": "alice"}}
    )
    print(f"   Alice: Hi, I'm Alice")
    print(f"   AI: {response.content}\n")

    response = conversation.invoke(
        {"input": "What's my name?"},
        config={"configurable": {"session_id": "alice"}}
    )
    print(f"   Alice: What's my name?")
    print(f"   AI: {response.content}\n")

    # 6. 用户 Bob 的对话
    print("3. 用户 Bob 的对话:")
    response = conversation.invoke(
        {"input": "Hi, I'm Bob"},
        config={"configurable": {"session_id": "bob"}}
    )
    print(f"   Bob: Hi, I'm Bob")
    print(f"   AI: {response.content}\n")

    response = conversation.invoke(
        {"input": "What's my name?"},
        config={"configurable": {"session_id": "bob"}}
    )
    print(f"   Bob: What's my name?")
    print(f"   AI: {response.content}\n")

    # 7. 验证会话隔离
    print("4. 验证会话隔离:")
    print(f"   Alice 的历史消息数: {len(store['alice'].messages)}")
    print(f"   Bob 的历史消息数: {len(store['bob'].messages)}")

if __name__ == "__main__":
    demo_runnable_with_history()
```

## 实战3：完整的聊天机器人实现

### 代码实现

```python
"""
实战3：完整的聊天机器人实现
演示一个生产级的聊天机器人实现
"""

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_openai import ChatOpenAI
from typing import Dict
from dotenv import load_dotenv

load_dotenv()

class ChatBot:
    """聊天机器人类"""

    def __init__(self, model: str = "gpt-4o-mini", temperature: float = 0.7):
        """初始化聊天机器人"""
        self.store: Dict[str, InMemoryChatMessageHistory] = {}
        self.model = model
        self.temperature = temperature
        self._setup_chain()

    def _setup_chain(self):
        """设置对话链"""
        # 创建提示模板
        prompt = ChatPromptTemplate.from_messages([
            ("system", """You are a friendly and helpful AI assistant.

Your characteristics:
- Be conversational and natural
- Remember context from the conversation
- Ask clarifying questions when needed
- Be concise but informative
- Show empathy and understanding"""),
            MessagesPlaceholder(variable_name="history"),
            ("human", "{input}")
        ])

        # 创建 LLM
        llm = ChatOpenAI(model=self.model, temperature=self.temperature)

        # 创建链
        chain = prompt | llm

        # 包装为带历史的链
        self.conversation = RunnableWithMessageHistory(
            chain,
            self._get_session_history,
            input_messages_key="input",
            history_messages_key="history"
        )

    def _get_session_history(self, session_id: str) -> InMemoryChatMessageHistory:
        """获取会话历史"""
        if session_id not in self.store:
            self.store[session_id] = InMemoryChatMessageHistory()
        return self.store[session_id]

    def chat(self, user_id: str, message: str) -> str:
        """发送消息并获取回复"""
        response = self.conversation.invoke(
            {"input": message},
            config={"configurable": {"session_id": user_id}}
        )
        return response.content

    def get_history(self, user_id: str) -> list:
        """获取用户的对话历史"""
        if user_id in self.store:
            return self.store[user_id].messages
        return []

    def clear_history(self, user_id: str):
        """清空用户的对话历史"""
        if user_id in self.store:
            self.store[user_id].clear()

    def get_stats(self) -> dict:
        """获取统计信息"""
        return {
            "total_users": len(self.store),
            "total_messages": sum(len(h.messages) for h in self.store.values())
        }

def demo_chatbot():
    """演示聊天机器人的使用"""
    print("=== 实战3：完整的聊天机器人实现 ===\n")

    # 1. 创建聊天机器人
    bot = ChatBot()
    print("1. 聊天机器人初始化完成\n")

    # 2. 模拟对话
    print("2. 模拟对话场景:\n")

    # 用户1的对话
    user1 = "user_001"
    print(f"[{user1}] Hi, I'm working on a Python project")
    response = bot.chat(user1, "Hi, I'm working on a Python project")
    print(f"[Bot] {response}\n")

    print(f"[{user1}] Can you help me with error handling?")
    response = bot.chat(user1, "Can you help me with error handling?")
    print(f"[Bot] {response}\n")

    print(f"[{user1}] What was I working on?")
    response = bot.chat(user1, "What was I working on?")
    print(f"[Bot] {response}\n")

    # 用户2的对话
    user2 = "user_002"
    print(f"[{user2}] Hello! I need help with cooking")
    response = bot.chat(user2, "Hello! I need help with cooking")
    print(f"[Bot] {response}\n")

    # 3. 查看统计信息
    print("3. 统计信息:")
    stats = bot.get_stats()
    print(f"   总用户数: {stats['total_users']}")
    print(f"   总消息数: {stats['total_messages']}\n")

    # 4. 查看历史
    print("4. 用户1的对话历史:")
    history = bot.get_history(user1)
    for i, msg in enumerate(history, 1):
        role = "User" if msg.__class__.__name__ == "HumanMessage" else "AI"
        print(f"   [{i}] {role}: {msg.content[:50]}...")

if __name__ == "__main__":
    demo_chatbot()
```

"""
完整示例：RAG 对话系统
演示如何将 MessagesPlaceholder 应用到 RAG 系统中
"""

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_openai import ChatOpenAI
from typing import Dict
from dotenv import load_dotenv

load_dotenv()

class RAGChatBot:
    """RAG 对话机器人"""

    def __init__(self):
        self.store: Dict[str, InMemoryChatMessageHistory] = {}
        self._setup_chain()

    def _setup_chain(self):
        """设置 RAG 对话链"""
        prompt = ChatPromptTemplate.from_messages([
            ("system", """You are a helpful assistant that answers questions based on the provided context.

Rules:
1. Use the context to answer questions
2. If the context doesn't contain the answer, say so
3. Remember the conversation history
4. Be concise and accurate"""),
            MessagesPlaceholder(variable_name="history"),
            ("human", """Context:
{context}

Question: {question}""")
        ])

        llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        chain = prompt | llm

        self.conversation = RunnableWithMessageHistory(
            chain,
            self._get_session_history,
            input_messages_key="question",
            history_messages_key="history"

        if session_id not in self.store:
            self.store[session_id] = InMemoryChatMessageHistory()
        return self.store[session_id]

    def ask(self, user_id: str, question: str, context: str) -> str:
        """提问并获取答案"""
        response = self.conversation.invoke(
            {"question": question, "context": context},
            config={"configurable": {"session_id": user_id}}

    history = store[session_id]

### 2. 会话管理

**会话超时清理**
```python
from datetime import datetime, timedelta

class SessionManager:
    def __init__(self, timeout_minutes: int = 30):
        self.store = {}
        self.last_access = {}
        self.timeout = timedelta(minutes=timeout_minutes)

    def get_session(self, session_id: str):
        # 清理过期会话
        self._cleanup_expired()
        # 更新访问时间
        self.last_access[session_id] = datetime.now()
        if session_id not in self.store:
            self.store[session_id] = InMemoryChatMessageHistory()
        return self.store[session_id]

    def _cleanup_expired(self):
        now = datetime.now()
        expired = [
            sid for sid, last_time in self.last_access.items()
            if now - last_time > self.timeout
        ]
        for sid in expired:
            del self.store[sid]
            del self.last_access[sid]
```

## 常见问题

### Q1: MessagesPlaceholder 和普通变量有什么区别？

**A:** 
- **MessagesPlaceholder**: 接受消息列表，自动展开为多个消息
- **普通变量**: 接受单个值，替换到模板中

```python
# MessagesPlaceholder
MessagesPlaceholder("history")  # 接受 [msg1, msg2, ...]

# 普通变量
("human", "{input}")  # 接受单个字符串
```

### Q2: 如何限制历史消息的数量？

**A:** 有两种方法：

**方法1：在获取历史时限制**
```python
def get_session_history(session_id: str):
    history = store[session_id]
    # 只保留最近10条消息
    history.messages = history.messages[-10:]
    return history
```

**方法2：使用 n_messages 参数**
```python
MessagesPlaceholder("history", n_messages=10)
```

### Q3: 如何处理多模态消息（文本+图片）？

**A:** 使用消息对象的 content 列表格式：
```python
from langchain_core.messages import HumanMessage

history = [
    HumanMessage(content=[
        {"type": "text", "text": "What's in this image?"},
        {"type": "image_url", "image_url": {"url": "https://..."}}
    ])
]
```

### Q4: RunnableWithMessageHistory 支持持久化吗？

**A:** 支持，只需替换 InMemoryChatMessageHistory 为持久化存储：
```python
from langchain_community.chat_message_histories import RedisChatMessageHistory

def get_session_history(session_id: str):
    return RedisChatMessageHistory(
        session_id=session_id,
        url="redis://localhost:6379"
    )
```
