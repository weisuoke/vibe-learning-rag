## 1. 创建不同一致性级别的集合

### 1.1 定义Schema

```python
def create_schema():
    """创建集合Schema"""
    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
        FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=128),
        FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=500),
        FieldSchema(name="timestamp", dtype=DataType.INT64)
    ]
    schema = CollectionSchema(fields=fields, description="Consistency level demo")
    return schema
```

### 1.2 创建Strong一致性集合

```python
def create_strong_collection():
    """创建Strong一致性集合"""
    schema = create_schema()

    # 检查集合是否存在
    if utility.has_collection("strong_consistency"):
        utility.drop_collection("strong_consistency")

    # 创建集合（Strong一致性）
    collection = Collection(
        name="strong_consistency",
        schema=schema,
        consistency_level="Strong"  # Strong一致性
    )

    # 创建索引
    index_params = {
        "index_type": "IVF_FLAT",
        "metric_type": "L2",
        "params": {"nlist": 128}
    }
    collection.create_index(field_name="vector", index_params=index_params)

    # 加载集合
    collection.load()

    print(f"✅ Created Strong consistency collection")
    return collection
```

### 1.3 创建Bounded一致性集合

## 2. 测试数据可见性

### 2.1 插入数据并立即搜索

```python
def test_immediate_visibility(collection, consistency_name):
    """测试插入后立即搜索的可见性"""
    print(f"\n{'='*60}")
    print(f"Testing {consistency_name} Consistency - Immediate Visibility")
    print(f"{'='*60}")

    # 生成测试数据
    test_vector = np.random.rand(128).tolist()
    test_text = f"Test document for {consistency_name}"
    test_timestamp = int(time.time() * 1000)

    # 插入数据
    print(f"📝 Inserting data...")
    entities = [{
        "vector": test_vector,
        "text": test_text,
        "timestamp": test_timestamp
    }]
    insert_result = collection.insert(entities)
    print(f"   Inserted ID: {insert_result.primary_keys[0]}")

    # 立即搜索（使用集合默认一致性级别）
    print(f"🔍 Searching immediately...")
    search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
    results = collection.search(
        data=[test_vector],
        anns_field="vector",
        param=search_params,
        limit=1,
        output_fields=["text", "timestamp"]
    )

    # 检查结果
    if len(results[0]) > 0:
        print(f"✅ Found {len(results[0])} result(s) immediately")
        print(f"   Text: {results[0][0].entity.get('text')}")
        print(f"   Distance: {results[0][0].distance:.6f}")
    else:
        print(f"❌ No results found immediately")

    return len(results[0]) > 0
```

### 2.2 延迟搜索测试

def test_delayed_visibility(collection, consistency_name, delay_seconds=2):
    """测试延迟后的可见性"""
    print(f"\n⏱️  Waiting {delay_seconds} seconds...")
    time.sleep(delay_seconds)

        anns_field="vector",
        param=search_params,
        limit=5,
        output_fields=["text", "timestamp"]
    )

    print(f"✅ Found {len(results[0])} result(s) after delay")
    return len(results[0])

## 3. 覆盖一致性级别

### 3.1 在搜索时临时覆盖

```python
def test_override_consistency(collection, original_level, override_level):
    """测试在搜索时覆盖一致性级别"""
    print(f"\n{'='*60}")
    print(f"Testing Override: {original_level} → {override_level}")
    print(f"{'='*60}")

    # 插入数据
    test_vector = np.random.rand(128).tolist()
    test_text = f"Override test: {original_level} to {override_level}"
    entities = [{
        "vector": test_vector,
        "text": test_text,
        "timestamp": int(time.time() * 1000)
    }]
    collection.insert(entities)
    print(f"📝 Inserted data")

    # 使用覆盖的一致性级别搜索
    print(f"🔍 Searching with {override_level} consistency...")
    search_params = {"metric_type": "L2", "params": {"nprobe": 10}}

    # 将字符串转换为ConsistencyLevel枚举
    consistency_map = {
        "Strong": ConsistencyLevel.Strong,
        "Bounded": ConsistencyLevel.Bounded,
        "Eventually": ConsistencyLevel.Eventually,
        "Session": ConsistencyLevel.Session
    }

    results = collection.search(
        data=[test_vector],
        anns_field="vector",
        param=search_params,
        limit=1,
        output_fields=["text"],
        consistency_level=consistency_map[override_level]  # 覆盖一致性级别
    )

    if len(results[0]) > 0:
        print(f"✅ Found result with {override_level} consistency")
        print(f"   Text: {results[0][0].entity.get('text')}")
    else:
        print(f"❌ No result found with {override_level} consistency")

    return len(results[0]) > 0
```

def main():

    print(f"  - Immediate visibility: {'✅ Yes' if eventually_immediate else '❌ No'}")
    print(f"  - Delayed visibility: {eventually_delayed} results")

    print("\n🧹 Cleaning up...")
    utility.drop_collection("strong_consistency")
    utility.drop_collection("bounded_consistency")
    utility.drop_collection("eventually_consistency")
    print("✅ Cleanup complete")

if __name__ == "__main__":

## 6. 关键观察

### 6.1 Strong一致性

```
✅ 立即可见：插入后立即搜索能找到数据
✅ 延迟后可见：延迟后仍然能找到数据
⚠️ 代价：延迟最高，吞吐量最低
```

### 6.2 Bounded一致性

```
✅ 立即可见：大多数情况下立即可见
✅ 延迟后可见：延迟后一定能找到数据
✅ 平衡：性能和一致性的最佳平衡
```

### 6.3 Eventually一致性

```
❌ 立即不可见：插入后立即搜索可能找不到数据
✅ 延迟后可见：延迟几秒后能找到数据
✅ 性能：延迟最低，吞吐量最高
```

import time

def benchmark_consistency(collection, level_name, iterations=100):
    """基准测试不同一致性级别"""
    latencies = []
    test_vector = np.random.rand(128).tolist()

    for _ in range(iterations):
        start = time.time()
        results = collection.search(
            data=[test_vector],
            anns_field="vector",
            param={"metric_type": "L2", "params": {"nprobe": 10}},
            limit=10
        )
        latencies.append(time.time() - start)

    print(f"{level_name} Consistency:")
    print(f"  Mean latency: {np.mean(latencies)*1000:.2f}ms")
    print(f"  P95 latency: {np.percentile(latencies, 95)*1000:.2f}ms")
    print(f"  P99 latency: {np.percentile(latencies, 99)*1000:.2f}ms")
