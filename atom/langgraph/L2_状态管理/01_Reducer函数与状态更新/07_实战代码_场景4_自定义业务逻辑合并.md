## 场景描述

### 业务需求

构建一个任务优先级管理系统，需要：
1. **智能合并**：根据业务规则合并任务列表
2. **优先级处理**：高优先级任务优先保留
3. **去重逻辑**：相同任务只保留一个
4. **条件合并**：根据任务状态决定合并策略

### 技术挑战

- 如何实现复杂的合并逻辑？
- 如何处理任务的优先级？
- 如何避免重复任务？
- 如何支持条件合并？

### 解决方案

使用自定义 Reducer 函数：
- **自定义逻辑**：实现复杂的业务规则
- **灵活控制**：完全控制合并行为
- **类型安全**：支持复杂数据结构

### 1. 数据模型 (task_manager/models.py)

```python
"""
数据模型模块
定义任务相关的数据结构
"""
from dataclasses import dataclass
from enum import Enum
from typing import Optional
from datetime import datetime


class Priority(Enum):
    """任务优先级"""
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    URGENT = 4


class TaskStatus(Enum):
    """任务状态"""
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"
    CANCELLED = "cancelled"


@dataclass
class Task:
    """
    任务数据模型

    Attributes:
        id: 任务 ID
        title: 任务标题
        priority: 优先级
        status: 状态
        created_at: 创建时间
        assignee: 负责人
    """
    id: str
    title: str
    priority: Priority
    status: TaskStatus = TaskStatus.PENDING
    created_at: datetime = None
    assignee: Optional[str] = None

    def __post_init__(self):
        if self.created_at is None:
            self.created_at = datetime.now()

    def __eq__(self, other):
        """根据 ID 判断任务是否相同"""
        if not isinstance(other, Task):
            return False
        return self.id == other.id

    def __hash__(self):
        return hash(self.id)
```

### 2. 自定义 Reducer (task_manager/reducers.py)

```python
"""
自定义 Reducer 模块
实现各种任务合并策略
"""
from typing import List
from task_manager.models import Task, Priority, TaskStatus


def merge_tasks_by_priority(left: List[Task], right: List[Task]) -> List[Task]:
    """
    按优先级合并任务列表

    规则：
    1. 合并两个列表
    2. 去重（相同 ID 的任务只保留一个）
    3. 优先级高的任务优先保留
    4. 优先级相同时，保留最新的任务

    Args:
        left: 左侧任务列表
        right: 右侧任务列表

    Returns:
        合并后的任务列表
    """
    # 使用字典去重，key 为任务 ID
    task_dict = {}

    # 先添加左侧任务
    for task in left:
        task_dict[task.id] = task

    # 添加右侧任务，处理冲突
    for task in right:
        if task.id in task_dict:
            existing_task = task_dict[task.id]

            # 比较优先级
            if task.priority.value > existing_task.priority.value:
                # 新任务优先级更高，替换
                task_dict[task.id] = task
            elif task.priority.value == existing_task.priority.value:
                # 优先级相同，保留最新的
                if task.created_at > existing_task.created_at:
                    task_dict[task.id] = task
        else:
            # 新任务，直接添加
            task_dict[task.id] = task

    # 按优先级排序（高优先级在前）
    result = sorted(
        task_dict.values(),
        key=lambda t: (t.priority.value, t.created_at),
        reverse=True
    )

    return result


def merge_tasks_keep_active(left: List[Task], right: List[Task]) -> List[Task]:
    """
    合并任务，只保留活跃任务

    规则：
    1. 过滤掉已完成和已取消的任务
    2. 去重
    3. 按创建时间排序

    Args:
        left: 左侧任务列表
        right: 右侧任务列表

    Returns:
        合并后的活跃任务列表
    """
    # 合并列表
    all_tasks = left + right

    # 过滤活跃任务
    active_tasks = [
        task for task in all_tasks
        if task.status not in [TaskStatus.COMPLETED, TaskStatus.CANCELLED]
    ]

    # 去重
    task_dict = {task.id: task for task in active_tasks}

    # 按创建时间排序
    result = sorted(
        task_dict.values(),
        key=lambda t: t.created_at
    )

    return result


def merge_tasks_by_assignee(left: List[Task], right: List[Task]) -> List[Task]:
    """
    按负责人分组合并任务

    规则：
    1. 合并任务列表
    2. 按负责人分组
    3. 每个负责人最多保留 5 个任务

    Args:
        left: 左侧任务列表
        right: 右侧任务列表

    Returns:
        合并后的任务列表
    """
    # 合并列表
    all_tasks = left + right

    # 去重
    task_dict = {task.id: task for task in all_tasks}

    # 按负责人分组
    assignee_tasks = {}
    for task in task_dict.values():
        assignee = task.assignee or "unassigned"
        if assignee not in assignee_tasks:
            assignee_tasks[assignee] = []
        assignee_tasks[assignee].append(task)

    # 每个负责人最多保留 5 个任务（按优先级）
    result = []
    for assignee, tasks in assignee_tasks.items():
        # 按优先级排序
        sorted_tasks = sorted(
            tasks,
            key=lambda t: t.priority.value,
            reverse=True
        )
        # 只保留前 5 个
        result.extend(sorted_tasks[:5])

    return result


def merge_tasks_conditional(left: List[Task], right: List[Task]) -> List[Task]:
    """
    条件合并任务

    规则：
    1. 如果右侧有紧急任务，清空左侧所有非紧急任务
    2. 否则，正常合并去重

    Args:
        left: 左侧任务列表
        right: 右侧任务列表

    Returns:
        合并后的任务列表
    """
    # 检查右侧是否有紧急任务
    has_urgent = any(task.priority == Priority.URGENT for task in right)

    if has_urgent:
        # 只保留紧急任务
        urgent_tasks_left = [
            task for task in left
            if task.priority == Priority.URGENT
        ]
        urgent_tasks_right = [
            task for task in right
            if task.priority == Priority.URGENT
        ]

        # 合并去重
        task_dict = {task.id: task for task in urgent_tasks_left + urgent_tasks_right}
        return list(task_dict.values())
    else:
        # 正常合并去重
        task_dict = {task.id: task for task in left + right}
        return list(task_dict.values())
```

### 3. 状态定义 (task_manager/state.py)

```python
"""
状态定义模块
使用自定义 Reducer 管理任务列表
"""
from typing import Annotated, TypedDict, List
from task_manager.models import Task
from task_manager.reducers import (
    merge_tasks_by_priority,
    merge_tasks_keep_active,
    merge_tasks_by_assignee,
    merge_tasks_conditional
)


class TaskStatePriority(TypedDict):
    """按优先级合并的任务状态"""
    tasks: Annotated[List[Task], merge_tasks_by_priority]


class TaskStateActive(TypedDict):
    """只保留活跃任务的状态"""
    tasks: Annotated[List[Task], merge_tasks_keep_active]


class TaskStateAssignee(TypedDict):
    """按负责人分组的任务状态"""
    tasks: Annotated[List[Task], merge_tasks_by_assignee]


class TaskStateConditional(TypedDict):
    """条件合并的任务状态"""
    tasks: Annotated[List[Task], merge_tasks_conditional]
```

### 4. 节点函数 (task_manager/nodes.py)

```python
"""
节点函数模块
实现任务管理的核心逻辑
"""
from task_manager.state import TaskStatePriority
from task_manager.models import Task, Priority, TaskStatus
import random


def create_high_priority_tasks_node(state: TaskStatePriority) -> dict:
    """创建高优先级任务节点"""
    tasks = [
        Task(
            id=f"high-{i}",
            title=f"高优先级任务 {i}",
            priority=Priority.HIGH
        )
        for i in range(3)
    ]
    return {"tasks": tasks}


def create_medium_priority_tasks_node(state: TaskStatePriority) -> dict:
    """创建中优先级任务节点"""
    tasks = [
        Task(
            id=f"medium-{i}",
            title=f"中优先级任务 {i}",
            priority=Priority.MEDIUM
        )
        for i in range(5)
    ]
    return {"tasks": tasks}


def create_urgent_task_node(state: TaskStatePriority) -> dict:
    """创建紧急任务节点"""
    task = Task(
        id="urgent-1",
        title="紧急任务",
        priority=Priority.URGENT
    )
    return {"tasks": [task]}


def update_task_status_node(state: TaskStatePriority) -> dict:
    """更新任务状态节点"""
    tasks = state.get("tasks", [])

    if not tasks:
        return {"tasks": []}

    # 随机选择一个任务更新状态
    task = random.choice(tasks)
    updated_task = Task(
        id=task.id,
        title=task.title,
        priority=task.priority,
        status=TaskStatus.IN_PROGRESS,
        created_at=task.created_at,
        assignee=task.assignee
    )

    return {"tasks": [updated_task]}
```

### 5. 图构建 (task_manager/graph.py)

```python
"""
图构建模块
定义任务管理的工作流
"""
from langgraph.graph import StateGraph, START, END
from task_manager.state import TaskStatePriority
from task_manager.nodes import (
    create_high_priority_tasks_node,
    create_medium_priority_tasks_node,
    create_urgent_task_node
)


def create_task_graph():
    """
    创建任务管理图

    工作流：
    START -> high_priority -> medium_priority -> urgent -> END

    任务会按优先级自动合并
    """
    builder = StateGraph(TaskStatePriority)

    # 添加节点
    builder.add_node("high_priority", create_high_priority_tasks_node)
    builder.add_node("medium_priority", create_medium_priority_tasks_node)
    builder.add_node("urgent", create_urgent_task_node)

    # 添加边
    builder.add_edge(START, "high_priority")
    builder.add_edge("high_priority", "medium_priority")
    builder.add_edge("medium_priority", "urgent")
    builder.add_edge("urgent", END)

    # 编译
    graph = builder.compile()

    return graph
```

去重示例

from task_manager.models import Task, Priority

from datetime import datetime, timedelta

def main():
    # 创建重复任务
    tasks1 = [
        Task(id="task-1", title="任务1", priority=Priority.HIGH),
        Task(id="task-2", title="任务2", priority=Priority.MEDIUM),
    ]

    tasks2 = [
        Task(
            id="task-1",
            title="任务1（更新）",
            priority=Priority.URGENT,
            created_at=datetime.now() + timedelta(seconds=1)

        Task(id="task-3", title="任务3", priority=Priority.LOW),
    ]

    print("=" * 60)
    print("去重前")
    print("=" * 60)
    print(f"列表1: {len(tasks1)} 个任务")
    print(f"列表2: {len(tasks2)} 个任务")
    print(f"总计: {len(tasks1) + len(tasks2)} 个任务")

    # 合并去重
    merged = merge_tasks_by_priority(tasks1, tasks2)

    print("\n" + "=" * 60)
    print("去重后")
    print("=" * 60)
    print(f"合并后: {len(merged)} 个任务")

    for i, task in enumerate(merged, 1):
        print(f"\n[{i}] {task.title}")
        print(f"    ID: {task.id}")

## 常见问题与解决方案

### 问题 1: Reducer 函数签名错误

**问题描述**：
自定义 Reducer 必须接收 2 个参数。

**解决方案**：
```python
# 错误：只有 1 个参数
def bad_reducer(tasks: List[Task]) -> List[Task]:
    return tasks

# 正确：2 个参数
def good_reducer(left: List[Task], right: List[Task]) -> List[Task]:
    return left + right
```

### 问题 2: 类型不匹配

**问题描述**：
Reducer 的输入输出类型必须一致。

**解决方案**：
```python
# 错误：输入是 List[Task]，输出是 Task
def bad_reducer(left: List[Task], right: List[Task]) -> Task:
    return left[0]

# 正确：输入输出都是 List[Task]
def good_reducer(left: List[Task], right: List[Task]) -> List[Task]:
    return left + right
```

### 问题 3: 性能问题

**问题描述**：
复杂的 Reducer 逻辑可能影响性能。

**解决方案**：
```python
# 优化：使用字典去重（O(n)）
def optimized_reducer(left: List[Task], right: List[Task]) -> List[Task]:
    task_dict = {task.id: task for task in left}
    task_dict.update({task.id: task for task in right})
    return list(task_dict.values())

# 避免：使用列表去重（O(n²)）
def slow_reducer(left: List[Task], right: List[Task]) -> List[Task]:
    result = left.copy()
    for task in right:
        if task not in result:
            result.append(task)
    return result
```

## 性能优化建议

### 1. 使用高效数据结构

# 推荐：使用字典
def fast_merge(left: List[Task], right: List[Task]) -> List[Task]:
    task_dict = {task.id: task for task in left + right}
    return list(task_dict.values())

# 避免：嵌套循环

    result = []
    for task in left + right:
        if task not in result:
            result.append(task)
    return result

### 2. 减少不必要的操作

```python
# 推荐：提前返回
def smart_merge(left: List[Task], right: List[Task]) -> List[Task]:
    if not right:
        return left
    if not left:
        return right
    # 执行合并逻辑
    return left + right
```
