```python
"""
LangGraph 性能优化实战：大规模并行任务优化

演示场景：
1. 性能基准测试：对比动态分支 vs 静态并行
2. 批处理优化：减少任务数量
3. 并行度控制：限制资源使用
4. 性能监控：实时监控和分析

技术要点：
- 性能测试框架
- 批处理策略
- 资源限制
- 监控和调优
"""

import os
import time
import asyncio
from typing import List, Literal, TypedDict, Annotated
from typing_extensions import TypedDict
import operator
from dotenv import load_dotenv
from dataclasses import dataclass
from datetime import datetime

from langgraph.graph import StateGraph, START, END
from langgraph.types import Send

# 加载环境变量
load_dotenv()

# ===== 1. 性能监控工具 =====

@dataclass
class PerformanceMetrics:
    """性能指标"""
    task_count: int
    start_time: float
    end_time: float
    duration: float
    memory_usage: float  # MB
    throughput: float  # tasks/second

    def __str__(self):
        return f"""
性能指标:
- 任务数量: {self.task_count}
- 执行时间: {self.duration:.2f}s
- 吞吐量: {self.throughput:.2f} tasks/s
- 内存使用: {self.memory_usage:.2f} MB
"""


class PerformanceMonitor:
    """性能监控器"""

    def __init__(self):
        self.metrics_history = []

    def start_measurement(self):
        """开始测量"""
        return time.time()

    def end_measurement(self, start_time: float, task_count: int) -> PerformanceMetrics:
        """结束测量"""
        end_time = time.time()
        duration = end_time - start_time
        throughput = task_count / duration if duration > 0 else 0

        # 简化的内存使用估算（实际应用中使用 psutil）
        memory_usage = task_count * 0.1  # 假设每个任务 0.1 MB

        metrics = PerformanceMetrics(
            task_count=task_count,
            start_time=start_time,
            end_time=end_time,
            duration=duration,
            memory_usage=memory_usage,
            throughput=throughput
        )

        self.metrics_history.append(metrics)
        return metrics


# ===== 2. 定义状态结构 =====

class TaskState(TypedDict):
    """单个任务的状态"""
    task_id: int
    data: str
    result: str


class OverallState(TypedDict):
    """整体工作流状态"""
    # 输入数据
    input_data: List[str]

    # 任务结果（使用 operator.add 合并）
    results: Annotated[List[str], operator.add]

    # 性能指标
    task_count: int
    start_time: float


# ===== 3. 场景1：动态分支 vs 静态并行对比 =====

def process_task_dynamic(state: TaskState) -> dict:
    """动态分支：处理单个任务"""
    task_id = state["task_id"]
    data = state["data"]

    # 模拟处理（实际应用中可能是 LLM 调用）
    time.sleep(0.01)  # 模拟 10ms 处理时间
    result = f"Task {task_id}: Processed '{data}'"

    return {"results": [result]}


def process_batch_static(state: OverallState) -> dict:
    """静态并行：批量处理任务"""
    input_data = state["input_data"]

    # 批量处理所有任务
    results = []
    for i, data in enumerate(input_data):
        time.sleep(0.01)  # 模拟 10ms 处理时间
        results.append(f"Task {i}: Processed '{data}'")

    return {"results": results}


def create_dynamic_tasks(state: OverallState) -> List[Send]:
    """创建动态并行任务"""
    input_data = state["input_data"]

    return [
        Send(
            "process_task_dynamic",
            {"task_id": i, "data": data, "result": ""}
        )
        for i, data in enumerate(input_data)
    ]


def benchmark_dynamic_vs_static():
    """基准测试：动态分支 vs 静态并行"""
    print("\n" + "="*60)
    print("基准测试：动态分支 vs 静态并行")
    print("="*60)

    monitor = PerformanceMonitor()

    # 测试数据
    test_sizes = [10, 50, 100]

    for size in test_sizes:
        input_data = [f"data_{i}" for i in range(size)]

        print(f"\n测试规模: {size} 个任务")

        # 测试1：动态分支（Send API）
        print("\n1. 动态分支（Send API）")
        builder_dynamic = StateGraph(OverallState)
        builder_dynamic.add_node("process_task_dynamic", process_task_dynamic)
        builder_dynamic.add_edge(START, "generate_tasks")
        builder_dynamic.add_node("generate_tasks", lambda s: s)
        builder_dynamic.add_conditional_edges(
            "generate_tasks",
            create_dynamic_tasks,
            ["process_task_dynamic"]
        )
        builder_dynamic.add_edge("process_task_dynamic", END)
        graph_dynamic = builder_dynamic.compile()

        start_time = monitor.start_measurement()
        result_dynamic = graph_dynamic.invoke({
            "input_data": input_data,
            "results": [],
            "task_count": size,
            "start_time": start_time
        })
        metrics_dynamic = monitor.end_measurement(start_time, size)
        print(metrics_dynamic)

        # 测试2：静态并行（单节点批处理）
        print("\n2. 静态并行（单节点批处理）")
        builder_static = StateGraph(OverallState)
        builder_static.add_node("process_batch_static", process_batch_static)
        builder_static.add_edge(START, "process_batch_static")
        builder_static.add_edge("process_batch_static", END)
        graph_static = builder_static.compile()

        start_time = monitor.start_measurement()
        result_static = graph_static.invoke({
            "input_data": input_data,
            "results": [],
            "task_count": size,
            "start_time": start_time
        })
        metrics_static = monitor.end_measurement(start_time, size)
        print(metrics_static)

        # 对比分析
        speedup = metrics_dynamic.duration / metrics_static.duration
        print(f"\n性能对比:")
        print(f"- 动态分支耗时: {metrics_dynamic.duration:.2f}s")
        print(f"- 静态并行耗时: {metrics_static.duration:.2f}s")
        print(f"- 加速比: {speedup:.2f}x")


# ===== 4. 场景2：批处理优化 =====

class BatchState(TypedDict):
    """批处理状态"""
    batch_id: int
    items: List[str]
    results: Annotated[List[str], operator.add]


def process_batch(state: BatchState) -> dict:
    """处理一个批次"""
    batch_id = state["batch_id"]
    items = state["items"]

    # 批量处理
    results = []
    for item in items:
        time.sleep(0.01)  # 模拟处理
        results.append(f"Batch {batch_id}: Processed '{item}'")

    return {"results": results}


def create_batches(state: OverallState, batch_size: int) -> List[Send]:
    """创建批处理任务"""
    input_data = state["input_data"]

    # 将数据分批
    batches = []
    for i in range(0, len(input_data), batch_size):
        batch = input_data[i:i+batch_size]
        batches.append(batch)

    # 为每个批次创建 Send 对象
    return [
        Send(
            "process_batch",
            {"batch_id": i, "items": batch, "results": []}
        )
        for i, batch in enumerate(batches)
    ]


def benchmark_batch_sizes():
    """基准测试：不同批处理大小"""
    print("\n" + "="*60)
    print("基准测试：批处理优化")
    print("="*60)

    monitor = PerformanceMonitor()

    # 固定总任务数
    total_tasks = 100
    input_data = [f"data_{i}" for i in range(total_tasks)]

    # 测试不同批处理大小
    batch_sizes = [1, 5, 10, 20, 50]

    for batch_size in batch_sizes:
        print(f"\n批处理大小: {batch_size}")

        builder = StateGraph(OverallState)
        builder.add_node("process_batch", process_batch)
        builder.add_edge(START, "generate_batches")
        builder.add_node("generate_batches", lambda s: s)
        builder.add_conditional_edges(
            "generate_batches",
            lambda s: create_batches(s, batch_size),
            ["process_batch"]
        )
        builder.add_edge("process_batch", END)
        graph = builder.compile()

        start_time = monitor.start_measurement()
        result = graph.invoke({
            "input_data": input_data,
            "results": [],
            "task_count": total_tasks,
            "start_time": start_time
        })
        metrics = monitor.end_measurement(start_time, total_tasks)

        batch_count = (total_tasks + batch_size - 1) // batch_size
        print(f"- 批次数量: {batch_count}")
        print(f"- 执行时间: {metrics.duration:.2f}s")
        print(f"- 吞吐量: {metrics.throughput:.2f} tasks/s")


# ===== 5. 场景3：并行度控制 =====

class RateLimitedState(TypedDict):
    """带速率限制的状态"""
    input_data: List[str]
    results: Annotated[List[str], operator.add]
    active_tasks: int
    max_parallel: int


async def process_task_with_limit(state: TaskState) -> dict:
    """带限制的任务处理"""
    task_id = state["task_id"]
    data = state["data"]

    # 模拟异步处理
    await asyncio.sleep(0.01)
    result = f"Task {task_id}: Processed '{data}'"

    return {"results": [result]}


def create_limited_tasks(state: RateLimitedState) -> List[Send]:
    """创建受限的并行任务"""
    input_data = state["input_data"]
    max_parallel = state.get("max_parallel", 10)

    # 限制并行度
    limited_data = input_data[:max_parallel]

    return [
        Send(
            "process_task_with_limit",
            {"task_id": i, "data": data, "result": ""}
        )
        for i, data in enumerate(limited_data)
    ]


def benchmark_parallel_limits():
    """基准测试：并行度限制"""
    print("\n" + "="*60)
    print("基准测试：并行度控制")
    print("="*60)

    monitor = PerformanceMonitor()

    # 固定总任务数
    total_tasks = 100
    input_data = [f"data_{i}" for i in range(total_tasks)]

    # 测试不同并行度限制
    parallel_limits = [5, 10, 20, 50, 100]

    for limit in parallel_limits:
        print(f"\n最大并行度: {limit}")

        # 注意：这里简化了实现，实际应用中需要更复杂的流控机制
        # 这里只是演示概念

        start_time = monitor.start_measurement()

        # 模拟受限并行执行
        batch_size = limit
        batch_count = (total_tasks + batch_size - 1) // batch_size

        for i in range(batch_count):
            batch = input_data[i*batch_size:(i+1)*batch_size]
            # 模拟批次处理
            time.sleep(0.01 * len(batch))

        metrics = monitor.end_measurement(start_time, total_tasks)

        print(f"- 批次数量: {batch_count}")
        print(f"- 执行时间: {metrics.duration:.2f}s")
        print(f"- 吞吐量: {metrics.throughput:.2f} tasks/s")


# ===== 6. 场景4：实时监控和调优 =====

class MonitoredState(TypedDict):
    """带监控的状态"""
    input_data: List[str]
    results: Annotated[List[str], operator.add]
    metrics: dict


def process_with_monitoring(state: TaskState) -> dict:
    """带监控的任务处理"""
    task_id = state["task_id"]
    data = state["data"]

    start_time = time.time()

    # 模拟处理
    time.sleep(0.01)
    result = f"Task {task_id}: Processed '{data}'"

    end_time = time.time()
    duration = end_time - start_time

    # 记录指标
    print(f"  [Task {task_id}] Duration: {duration*1000:.2f}ms")

    return {"results": [result]}


def monitor_execution():
    """监控执行过程"""
    print("\n" + "="*60)
    print("实时监控示例")
    print("="*60)

    # 创建测试数据
    input_data = [f"data_{i}" for i in range(20)]

    print(f"\n开始处理 {len(input_data)} 个任务...")
    print("实时监控输出:")

    builder = StateGraph(OverallState)
    builder.add_node("process_with_monitoring", process_with_monitoring)
    builder.add_edge(START, "generate_tasks")
    builder.add_node("generate_tasks", lambda s: s)
    builder.add_conditional_edges(
        "generate_tasks",
        lambda s: [
            Send("process_with_monitoring", {"task_id": i, "data": data, "result": ""})
            for i, data in enumerate(s["input_data"])
        ],
        ["process_with_monitoring"]
    )
    builder.add_edge("process_with_monitoring", END)
    graph = builder.compile()

    start_time = time.time()
    result = graph.invoke({
        "input_data": input_data,
        "results": [],
        "task_count": len(input_data),
        "start_time": start_time
    })
    end_time = time.time()

    print(f"\n总执行时间: {end_time - start_time:.2f}s")
    print(f"完成任务数: {len(result['results'])}")
```

## 核心优化技术

### 1. 动态分支 vs 静态并行

**动态分支（Send API）**：
- 优点：灵活、运行时确定并行度
- 缺点：任务创建开销、内存压力
- 适用：任务数量不确定、需要动态决策

**静态并行（多条边）**：
- 优点：性能更好、开销更小
- 缺点：不够灵活、编译时确定
- 适用：任务数量固定、简单并行

### 2. 批处理策略

```python
# 将 100 个任务分成 10 个批次
BATCH_SIZE = 10
batches = [tasks[i:i+BATCH_SIZE] for i in range(0, len(tasks), BATCH_SIZE)]

# 为每个批次创建一个 Send 对象
return [Send("process_batch", {"items": batch}) for batch in batches]
```

**关键点**：
- 批处理大小：5-20 个任务/批次
- 权衡：批次越大，开销越小，但灵活性越低
- 根据任务处理时间调整

### 3. 并行度控制

```python
# 限制最大并行任务数
MAX_PARALLEL = 10

def create_limited_tasks(state):
    tasks = state["pending_tasks"]
    # 只创建前 MAX_PARALLEL 个任务
    limited_tasks = tasks[:MAX_PARALLEL]
    return [Send("process", task) for task in limited_tasks]
```

**关键点**：
- 设置合理的最大并行度（10-50）
- 避免资源耗尽（内存、连接）
- 考虑外部限制（API 限流）
