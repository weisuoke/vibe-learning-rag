# 实战代码 - 场景1：RAG文档分块管道

## 场景描述

构建一个完整的RAG文档分块管道，从文档加载到向量检索的端到端流程。该场景模拟真实的RAG应用开发，支持多种文档格式（PDF、DOCX、TXT），使用RecursiveCharacterTextSplitter进行智能分块，并集成向量存储和检索功能。

**应用场景**：
- 企业知识库问答系统
- 技术文档智能检索
- 客服机器人知识管理
- 学术论文分析系统

**核心价值**：
- 统一处理多种文档格式
- 智能分块保证语义完整性
- 参数可调优适应不同场景
- 完整的RAG流程可直接用于生产

---

## 技术选型

### 为什么选择 RecursiveCharacterTextSplitter？

**1. 语义完整性最佳**
- 递归策略：优先按段落→句子→单词→字符分割
- 保留上下文：通过chunk_overlap避免信息丢失
- 适应性强：自动适应不同文档结构

**2. 性能与质量平衡**
- 时间复杂度：O(n*m)，m为分隔符数量（通常≤4）
- 基准测试：512 tokens配置达到69%准确率（2025 Reddit基准）
- 生产验证：OpenSearch、AWS等大厂采用

**3. 参数可控性强**
```python
# 核心参数
chunk_size=1000        # 推荐1000字符（LangChain官方）
chunk_overlap=200      # 20%重叠率（最佳实践）
add_start_index=True   # 便于溯源调试
```

**对比其他分块器**：
| 分块器 | 语义完整性 | 性能 | 适用场景 |
|--------|-----------|------|---------|
| RecursiveCharacterTextSplitter | ⭐⭐⭐⭐⭐ | ⭐⭐⭐⭐ | 通用RAG（推荐） |
| CharacterTextSplitter | ⭐⭐⭐ | ⭐⭐⭐⭐⭐ | 简单文本 |
| TokenTextSplitter | ⭐⭐⭐⭐ | ⭐⭐⭐ | Token精确控制 |
| SemanticChunker | ⭐⭐⭐⭐⭐ | ⭐⭐ | 高质量要求 |

## 完整代码实现

```python
"""
RAG文档分块管道 - 完整实现
支持：PDF、DOCX、TXT多格式 + 向量存储 + 智能检索

依赖安装：
pip install langchain langchain-text-splitters langchain-openai langchain-community chromadb pypdf python-docx python-dotenv
"""

import os
from pathlib import Path
from typing import List, Dict, Any, Optional
from dotenv import load_dotenv

# LangChain核心组件
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import (
    PyPDFLoader,
    Docx2txtLoader,
    TextLoader,
    DirectoryLoader
)
from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma
from langchain.schema import Document

# 加载环境变量
load_dotenv()


class RAGDocumentPipeline:
    """
    RAG文档分块管道

    功能：
    1. 多格式文档加载（PDF、DOCX、TXT）
    2. 智能文本分块（RecursiveCharacterTextSplitter）
    3. 向量化存储（ChromaDB）
    4. 语义检索（相似度搜索）

    设计原则：
    - 单一职责：每个方法只做一件事
    - 可配置性：所有参数可外部配置
    - 可观测性：详细的日志和统计信息
    - 可扩展性：易于添加新的文档格式
    """

    def __init__(
        self,
        chunk_size: int = 1000,
        chunk_overlap: int = 200,
        embedding_model: str = "text-embedding-3-small",
        persist_directory: str = "./chroma_db"
    ):
        """
        初始化RAG管道

        参数说明：
        - chunk_size: 块大小（字符数）
          * 推荐值：1000（LangChain官方）
          * 小文档：500-800
          * 大文档：1500-2000

        - chunk_overlap: 块重叠（字符数）
          * 推荐值：chunk_size的15-20%
          * 最小值：100
          * 最大值：chunk_size的30%

        - embedding_model: OpenAI Embedding模型
          * text-embedding-3-small: 性价比高（推荐）
          * text-embedding-3-large: 质量最高
          * text-embedding-ada-002: 兼容性好

        - persist_directory: 向量数据库持久化目录
        """
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.persist_directory = persist_directory

        # 初始化文本分块器
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            length_function=len,
            add_start_index=True,  # 跟踪原始文档位置
            separators=["\n\n", "\n", " ", ""]  # 默认分隔符优先级
        )

        # 初始化Embedding模型
        self.embeddings = OpenAIEmbeddings(
            model=embedding_model,
            # 可选：使用自定义API端点
            # openai_api_base=os.getenv("OPENAI_BASE_URL")
        )

        # 向量存储（延迟初始化）
        self.vectorstore: Optional[Chroma] = None

        print(f"✅ RAG管道初始化完成")
        print(f"   - chunk_size: {chunk_size}")
        print(f"   - chunk_overlap: {chunk_overlap}")
        print(f"   - embedding_model: {embedding_model}")

    def load_documents(
        self,
        file_path: str,
        file_type: Optional[str] = None
    ) -> List[Document]:
        """
        加载单个文档

        支持格式：
        - PDF: .pdf
        - Word: .docx, .doc
        - Text: .txt, .md

        参数：
        - file_path: 文档路径
        - file_type: 文件类型（可选，自动检测）

        返回：
        - Document对象列表
        """
        file_path = Path(file_path)

        if not file_path.exists():
            raise FileNotFoundError(f"文件不存在: {file_path}")

        # 自动检测文件类型
        if file_type is None:
            file_type = file_path.suffix.lower()

        print(f"\n📄 加载文档: {file_path.name}")

        # 根据文件类型选择加载器
        if file_type == ".pdf":
            loader = PyPDFLoader(str(file_path))
        elif file_type in [".docx", ".doc"]:
            loader = Docx2txtLoader(str(file_path))
        elif file_type in [".txt", ".md"]:
            loader = TextLoader(str(file_path), encoding="utf-8")
        else:
            raise ValueError(f"不支持的文件类型: {file_type}")

        # 加载文档
        documents = loader.load()

        print(f"   ✓ 加载完成: {len(documents)} 页/段")

        return documents

    def load_directory(
        self,
        directory_path: str,
        glob_pattern: str = "**/*.*",
        file_types: Optional[List[str]] = None
    ) -> List[Document]:
        """
        批量加载目录中的文档

        参数：
        - directory_path: 目录路径
        - glob_pattern: 文件匹配模式（默认所有文件）
        - file_types: 文件类型过滤（如[".pdf", ".txt"]）

        返回：
        - Document对象列表
        """
        directory_path = Path(directory_path)

        if not directory_path.exists():
            raise FileNotFoundError(f"目录不存在: {directory_path}")

        print(f"\n📁 批量加载目录: {directory_path}")

        all_documents = []

        # 遍历目录中的文件
        for file_path in directory_path.glob(glob_pattern):
            if not file_path.is_file():
                continue

            # 文件类型过滤
            if file_types and file_path.suffix.lower() not in file_types:
                continue

            try:
                documents = self.load_documents(str(file_path))
                all_documents.extend(documents)
            except Exception as e:
                print(f"   ⚠️  加载失败: {file_path.name} - {e}")

        print(f"\n✅ 批量加载完成: {len(all_documents)} 个文档片段")

        return all_documents

    def split_documents(
        self,
        documents: List[Document]
    ) -> List[Document]:
        """
        分块文档

        参数：
        - documents: Document对象列表

        返回：
        - 分块后的Document对象列表
        """
        print(f"\n✂️  开始分块: {len(documents)} 个文档")

        # 使用RecursiveCharacterTextSplitter分块
        splits = self.text_splitter.split_documents(documents)

        # 统计信息
        total_chars = sum(len(doc.page_content) for doc in documents)
        avg_chunk_size = sum(len(s.page_content) for s in splits) / len(splits) if splits else 0

        print(f"   ✓ 分块完成: {len(splits)} 个块")
        print(f"   - 原始字符数: {total_chars:,}")
        print(f"   - 平均块大小: {avg_chunk_size:.0f} 字符")
        print(f"   - 压缩比: {len(splits)/len(documents):.1f}x")

        return splits

    def create_vectorstore(
        self,
        documents: List[Document],
        collection_name: str = "rag_documents"
    ) -> Chroma:
        """
        创建向量存储

        参数：
        - documents: 分块后的Document对象列表
        - collection_name: 集合名称

        返回：
        - Chroma向量存储对象
        """
        print(f"\n🔢 创建向量存储: {collection_name}")

        # 创建ChromaDB向量存储
        self.vectorstore = Chroma.from_documents(
            documents=documents,
            embedding=self.embeddings,
            collection_name=collection_name,
            persist_directory=self.persist_directory
        )

        print(f"   ✓ 向量存储创建完成")
        print(f"   - 文档数量: {len(documents)}")
        print(f"   - 持久化目录: {self.persist_directory}")

        return self.vectorstore

    def load_vectorstore(
        self,
        collection_name: str = "rag_documents"
    ) -> Chroma:
        """
        加载已存在的向量存储

        参数：
        - collection_name: 集合名称

        返回：
        - Chroma向量存储对象
        """
        print(f"\n📂 加载向量存储: {collection_name}")

        self.vectorstore = Chroma(
            collection_name=collection_name,
            embedding_function=self.embeddings,
            persist_directory=self.persist_directory
        )

        print(f"   ✓ 向量存储加载完成")

        return self.vectorstore

    def search(
        self,
        query: str,
        k: int = 4,
        score_threshold: Optional[float] = None
    ) -> List[Dict[str, Any]]:
        """
        语义检索

        参数：
        - query: 查询文本
        - k: 返回结果数量
        - score_threshold: 相似度阈值（可选）

        返回：
        - 检索结果列表
        """
        if self.vectorstore is None:
            raise ValueError("向量存储未初始化，请先调用create_vectorstore或load_vectorstore")

        print(f"\n🔍 检索查询: {query}")

        # 相似度搜索（带分数）
        if score_threshold is not None:
            results = self.vectorstore.similarity_search_with_relevance_scores(
                query,
                k=k,
                score_threshold=score_threshold
            )
        else:
            results = self.vectorstore.similarity_search_with_score(query, k=k)

        # 格式化结果
        formatted_results = []
        for i, (doc, score) in enumerate(results, 1):
            formatted_results.append({
                "rank": i,
                "content": doc.page_content,
                "score": score,
                "metadata": doc.metadata
            })

            print(f"\n   [{i}] 相似度: {score:.4f}")
            print(f"       内容预览: {doc.page_content[:100]}...")
            if "source" in doc.metadata:
                print(f"       来源: {doc.metadata['source']}")

        return formatted_results

    def run_pipeline(
        self,
        input_path: str,
        collection_name: str = "rag_documents",
        is_directory: bool = False
    ) -> Chroma:
        """
        运行完整的RAG管道

        参数：
        - input_path: 输入路径（文件或目录）
        - collection_name: 集合名称
        - is_directory: 是否为目录

        返回：
        - Chroma向量存储对象
        """
        print("=" * 60)
        print("🚀 启动RAG文档分块管道")
        print("=" * 60)

        # 步骤1：加载文档
        if is_directory:
            documents = self.load_directory(input_path)
        else:
            documents = self.load_documents(input_path)

        # 步骤2：分块文档
        splits = self.split_documents(documents)

        # 步骤3：创建向量存储
        vectorstore = self.create_vectorstore(splits, collection_name)

        print("\n" + "=" * 60)
        print("✅ RAG管道运行完成")
        print("=" * 60)

        return vectorstore
