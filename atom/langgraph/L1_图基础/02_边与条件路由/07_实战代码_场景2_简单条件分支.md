# 实战代码 - 场景2：简单条件分支

> 本文档演示 LangGraph 中的简单条件分支，使用条件边实现 if-else 决策逻辑

---

## 场景描述

**场景**：构建一个内容审核系统，根据内容质量分数决定是发布还是需要人工审核。

**特点**：
- 动态决策（基于状态值）
- 使用条件边（`add_conditional_edges`）
- if-else 分支逻辑
- 适合需要决策的工作流

## 完整代码

```python
"""
LangGraph 实战代码 - 场景2：简单条件分支
演示：使用条件边实现内容审核系统
"""

from langgraph.graph import StateGraph, START, END
from typing import Literal
from typing_extensions import TypedDict


# ===== 1. 定义状态 =====
class ContentState(TypedDict):
    """内容审核状态"""
    content: str           # 内容文本
    quality_score: float   # 质量分数 (0-1)
    status: str            # 状态：pending/approved/rejected/review
    reviewer: str          # 审核人
    log: list[str]         # 操作日志


# ===== 2. 定义节点函数 =====
def analyze_content(state: ContentState) -> dict:
    """分析内容质量"""
    print("🔍 [分析内容] 开始分析内容质量...")

    content = state["content"]

    # 模拟质量评分（基于内容长度和关键词）
    score = 0.0

    # 长度评分
    if len(content) > 100:
        score += 0.3
    elif len(content) > 50:
        score += 0.2
    else:
        score += 0.1

    # 关键词评分
    positive_keywords = ["优质", "专业", "详细", "准确"]
    negative_keywords = ["垃圾", "广告", "spam"]

    for keyword in positive_keywords:
        if keyword in content:
            score += 0.2

    for keyword in negative_keywords:
        if keyword in content:
            score -= 0.3

    # 确保分数在 0-1 范围内
    score = max(0.0, min(1.0, score))

    print(f"🔍 [分析内容] 质量分数: {score:.2f}")

    return {
        "quality_score": score,
        "status": "pending",
        "log": state.get("log", []) + [f"内容分析完成，质量分数: {score:.2f}"]
    }


def approve_content(state: ContentState) -> dict:
    """批准内容"""
    print("✅ [批准内容] 内容质量高，自动批准发布")

    return {
        "status": "approved",
        "log": state.get("log", []) + ["内容已批准发布"]
    }


def review_content(state: ContentState) -> dict:
    """人工审核"""
    print("👤 [人工审核] 内容质量不确定，需要人工审核")

    # 模拟人工审核
    reviewer = "审核员A"

    return {
        "status": "review",
        "reviewer": reviewer,
        "log": state.get("log", []) + [f"内容已提交给 {reviewer} 审核"]
    }


def reject_content(state: ContentState) -> dict:
    """拒绝内容"""
    print("❌ [拒绝内容] 内容质量低，自动拒绝")

    return {
        "status": "rejected",
        "log": state.get("log", []) + ["内容已拒绝"]
    }


# ===== 3. 定义路由函数 =====
def route_by_quality(state: ContentState) -> Literal["approve", "review", "reject"]:
    """根据质量分数路由"""
    score = state["quality_score"]

    print(f"🔀 [路由决策] 质量分数: {score:.2f}")

    # 高质量：自动批准
    if score >= 0.8:
        print("🔀 [路由决策] → approve (高质量)")
        return "approve"

    # 中等质量：人工审核
    elif score >= 0.5:
        print("🔀 [路由决策] → review (中等质量)")
        return "review"

    # 低质量：自动拒绝
    else:
        print("🔀 [路由决策] → reject (低质量)")
        return "reject"


# ===== 4. 构建图 =====
def create_conditional_graph():
    """创建条件分支图"""
    # 创建状态图
    graph = StateGraph(ContentState)

    # 添加节点
    graph.add_node("analyze", analyze_content)
    graph.add_node("approve", approve_content)
    graph.add_node("review", review_content)
    graph.add_node("reject", reject_content)

    # 添加边
    graph.add_edge(START, "analyze")

    # 添加条件边（核心）
    graph.add_conditional_edges(
        "analyze",                # 起始节点
        route_by_quality,         # 路由函数
        {
            "approve": "approve",  # 高质量 → 批准
            "review": "review",    # 中等质量 → 审核
            "reject": "reject"     # 低质量 → 拒绝
        }
    )

    # 所有路径都通向 END
    graph.add_edge("approve", END)
    graph.add_edge("review", END)
    graph.add_edge("reject", END)

    # 编译图
    app = graph.compile()

    return app


# ===== 5. 测试不同场景 =====
def test_high_quality():
    """测试高质量内容"""
    print("=" * 60)
    print("测试场景 1：高质量内容")
    print("=" * 60)
    print()

    app = create_conditional_graph()

    initial_state = {
        "content": "这是一篇优质的专业文章，内容详细且准确，提供了大量有价值的信息。" * 3,
        "quality_score": 0.0,
        "status": "",
        "reviewer": "",
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("📊 最终结果：")
    print(f"  内容: {result['content'][:50]}...")
    print(f"  质量分数: {result['quality_score']:.2f}")
    print(f"  状态: {result['status']}")
    print(f"  审核人: {result['reviewer'] or '无'}")
    print()
    print("📝 操作日志：")
    for i, log_entry in enumerate(result['log'], 1):
        print(f"  {i}. {log_entry}")
    print()


def test_medium_quality():
    """测试中等质量内容"""
    print("=" * 60)
    print("测试场景 2：中等质量内容")
    print("=" * 60)
    print()

    app = create_conditional_graph()

    initial_state = {
        "content": "这是一篇普通的文章，内容还可以，但需要进一步审核。",
        "quality_score": 0.0,
        "status": "",
        "reviewer": "",
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("📊 最终结果：")
    print(f"  内容: {result['content'][:50]}...")
    print(f"  质量分数: {result['quality_score']:.2f}")
    print(f"  状态: {result['status']}")
    print(f"  审核人: {result['reviewer'] or '无'}")
    print()
    print("📝 操作日志：")
    for i, log_entry in enumerate(result['log'], 1):
        print(f"  {i}. {log_entry}")
    print()


def test_low_quality():
    """测试低质量内容"""
    print("=" * 60)
    print("测试场景 3：低质量内容")
    print("=" * 60)
    print()

    app = create_conditional_graph()

    initial_state = {
        "content": "垃圾广告spam",
        "quality_score": 0.0,
        "status": "",
        "reviewer": "",
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("📊 最终结果：")
    print(f"  内容: {result['content'][:50]}...")
    print(f"  质量分数: {result['quality_score']:.2f}")
    print(f"  状态: {result['status']}")
    print(f"  审核人: {result['reviewer'] or '无'}")
    print()
    print("📝 操作日志：")
    for i, log_entry in enumerate(result['log'], 1):
        print(f"  {i}. {log_entry}")
    print()


# ===== 6. 主函数 =====
def main():
    """主函数"""
    print("=" * 60)
    print("LangGraph 实战 - 场景2：简单条件分支")
    print("=" * 60)
    print()

    # 测试三种场景
    test_high_quality()
    test_medium_quality()
    test_low_quality()


if __name__ == "__main__":
    main()
```

## 代码详解

### 1. 路由函数设计

```python
def route_by_quality(state: ContentState) -> Literal["approve", "review", "reject"]:
    """根据质量分数路由"""
    score = state["quality_score"]

    # 高质量：自动批准
    if score >= 0.8:
        return "approve"

    # 中等质量：人工审核
    elif score >= 0.5:
        return "review"

    # 低质量：自动拒绝
    else:
        return "reject"
```

**关键点**：
- **类型提示**：使用 `Literal` 明确可能的返回值
- **清晰的条件**：使用简单的 if-elif-else 逻辑
- **返回节点名称**：返回目标节点的名称字符串
- **只读状态**：路由函数只读取状态，不修改状态

---

### 2. 条件边配置

```python
graph.add_conditional_edges(
    "analyze",                # 起始节点
    route_by_quality,         # 路由函数
    {
        "approve": "approve",  # 路由函数返回 "approve" → 执行 "approve" 节点
        "review": "review",    # 路由函数返回 "review" → 执行 "review" 节点
        "reject": "reject"     # 路由函数返回 "reject" → 执行 "reject" 节点
    }
)
```

**关键点**：
- **起始节点**：`"analyze"` - 条件边从这个节点出发
- **路由函数**：`route_by_quality` - 决定下一个节点
- **path_map**：将路由函数返回值映射到节点名称

---

### 3. 执行流程图

```
START
  ↓
analyze (分析内容)
  ↓
route_by_quality (路由决策)
  ↓
┌─────┼─────┐
↓     ↓     ↓
approve  review  reject
(批准)  (审核)  (拒绝)
↓     ↓     ↓
└─────┼─────┘
      ↓
     END
```

**执行路径**：
- **高质量**：START → analyze → approve → END
- **中等质量**：START → analyze → review → END
- **低质量**：START → analyze → reject → END

### 场景 1：用户注册审核

```python
class UserState(TypedDict):
    username: str
    email: str
    age: int
    risk_score: float
    status: str

def analyze_user(state: UserState) -> dict:
    """分析用户风险"""
    risk_score = 0.0

    # 年龄检查
    if state["age"] < 18:
        risk_score += 0.5

    # 邮箱检查
    if not state["email"].endswith((".com", ".org")):
        risk_score += 0.3

    return {"risk_score": risk_score}

def route_user(state: UserState) -> Literal["auto_approve", "manual_review", "reject"]:
    """路由用户"""
    if state["risk_score"] < 0.3:
        return "auto_approve"
    elif state["risk_score"] < 0.7:
        return "manual_review"
    else:
        return "reject"

# 构建图
graph = StateGraph(UserState)
graph.add_node("analyze", analyze_user)
graph.add_node("auto_approve", lambda s: {"status": "approved"})
graph.add_node("manual_review", lambda s: {"status": "review"})
graph.add_node("reject", lambda s: {"status": "rejected"})

graph.add_edge(START, "analyze")
graph.add_conditional_edges("analyze", route_user)
graph.add_edge("auto_approve", END)
graph.add_edge("manual_review", END)
graph.add_edge("reject", END)
```

**应用**：用户注册系统的自动审核

class OrderState(TypedDict):
    order_id: str
    amount: float
    payment_status: str

def check_payment(state: OrderState) -> dict:
    """检查支付状态"""

## 最佳实践

### 1. 使用类型提示

```python
# ✅ 推荐：使用 Literal 类型提示
def route_func(state: State) -> Literal["path_a", "path_b", "path_c"]:
    """明确的类型提示"""
    if condition_a:
        return "path_a"
    elif condition_b:
        return "path_b"
    else:
        return "path_c"

# ❌ 不推荐：没有类型提示
def route_func(state):
    """没有类型提示"""
    if condition_a:
        return "path_a"
    return "path_b"
```

---

### 2. 清晰的条件逻辑

```python
# ✅ 推荐：清晰的条件逻辑
def route_func(state: State) -> Literal["high", "medium", "low"]:
    """清晰的条件判断"""
    score = state["score"]

    if score > 0.8:
        return "high"
    elif score > 0.5:
        return "medium"
    else:
        return "low"

# ❌ 不推荐：复杂的嵌套逻辑
def route_func(state: State) -> str:
    """复杂的嵌套逻辑"""
    if state["a"]:
        if state["b"]:
            if state["c"]:
                return "path1"
            else:
                return "path2"
        else:
            return "path3"
    else:
        return "path4"
```

### 4. 处理边界情况

```python
def route_func(state: State) -> Literal["path_a", "path_b", "error"]:
    """处理边界情况"""
    score = state.get("score", 0.0)

    # 处理异常值
    if score < 0 or score > 1:
        return "error"

    # 正常路由
    if score > 0.5:
        return "path_a"
    return "path_b"
```

### 5. 使用 path_map 解耦

## 总结

**简单条件分支的特点**：
1. **动态决策**：基于状态值动态选择执行路径
2. **使用条件边**：使用 `add_conditional_edges` 添加条件边
3. **路由函数**：定义路由逻辑，返回目标节点名称
4. **if-else 逻辑**：简单的二选一或多选一决策

**适用场景**：
- 内容审核系统
- 用户注册审核
- 订单处理流程
- RAG 文档相关性判断
- 任何需要决策的工作流

**关键代码**：
```python
# 定义路由函数
def route_func(state: State) -> Literal["path_a", "path_b"]:
    if state["condition"]:
        return "path_a"
    return "path_b"

# 添加条件边
graph.add_conditional_edges(
    "source",
    route_func,
    {
        "path_a": "node_a",
        "path_b": "node_b"
    }
)
```
