## 一、WeightedRanker 核心概念

### 1.1 什么是 WeightedRanker？

**定义**：WeightedRanker 是 Milvus 提供的加权排序器，通过为每个检索请求指定权重来融合多个向量检索结果。

**核心特点**：
- 使用加权求和方式融合结果
- 权重可以是任意正数，不需要归一化
- 适合已知权重的场景
- 可精确控制各向量字段的重要性

**融合公式**：
```
final_score = weight1 * score1 + weight2 * score2 + ... + weightN * scoreN
```

### 1.2 WeightedRanker vs RRFRanker

| 特性 | WeightedRanker | RRFRanker |
|------|----------------|-----------|
| 融合方式 | 加权求和 | 倒数排名融合 |
| 权重控制 | 精确控制 | 自动平衡 |
| 适用场景 | 已知权重 | 不确定权重 |
| 参数调整 | 需要实验 | 参数较少 |

## 三、基础加权混合检索

### 3.1 创建 Collection（多向量字段）

```python
def create_multi_vector_collection(
    client: MilvusClient,
    collection_name: str = "weighted_hybrid_demo"
):
    """创建包含多个向量字段的 Collection"""

    # 检查并删除已存在的 Collection
    if client.has_collection(collection_name):
        client.drop_collection(collection_name)

    # 定义 Schema
    schema = client.create_schema(
        auto_id=True,
        enable_dynamic_field=False,
    )

    # 添加字段
    schema.add_field(field_name="id", datatype=DataType.INT64, is_primary=True)
    schema.add_field(field_name="text", datatype=DataType.VARCHAR, max_length=1000)

    # 添加多个向量字段
    schema.add_field(
        field_name="dense_vector",
        datatype=DataType.FLOAT_VECTOR,
        dim=384  # sentence-transformers/all-MiniLM-L6-v2 维度
    )
    schema.add_field(
        field_name="sparse_vector",
        datatype=DataType.SPARSE_FLOAT_VECTOR
    )

    # 创建索引参数
    index_params = client.prepare_index_params()

    # 为稠密向量创建索引
    index_params.add_index(
        field_name="dense_vector",
        index_type="IVF_FLAT",
        metric_type="IP",  # 内积
        params={"nlist": 128}
    )

    # 为稀疏向量创建索引
    index_params.add_index(
        field_name="sparse_vector",
        index_type="SPARSE_INVERTED_INDEX",
        metric_type="IP",
        params={"drop_ratio_build": 0.2}
    )

    # 创建 Collection
    client.create_collection(
        collection_name=collection_name,
        schema=schema,
        index_params=index_params
    )

    print(f"✓ Collection '{collection_name}' 创建成功")
    return collection_name
```

### 3.3 基础加权混合检索

```python
def weighted_hybrid_search(
    client: MilvusClient,
    collection_name: str,
    query_text: str,
    dense_weight: float = 1.0,
    sparse_weight: float = 1.0,
    limit: int = 5
) -> List[Dict]:
    """执行加权混合检索"""

    # 初始化 Embedding 模型
    model = SentenceTransformer('sentence-transformers/all-MiniLM-L6-v2')

    # 生成查询向量
    query_dense = model.encode(query_text).tolist()

    # 生成模拟稀疏查询向量
    query_sparse = {}
    num_terms = np.random.randint(3, 8)
    indices = np.random.choice(1000, num_terms, replace=False)
    for idx in indices:
        query_sparse[int(idx)] = float(np.random.random())

    # 1. 定义稠密向量检索请求
    dense_req = AnnSearchRequest(
        data=[query_dense],
        anns_field="dense_vector",
        param={"metric_type": "IP", "params": {"nprobe": 10}},
        limit=limit
    )

    # 2. 定义稀疏向量检索请求
    sparse_req = AnnSearchRequest(
        data=[query_sparse],
        anns_field="sparse_vector",
        param={"metric_type": "IP"},
        limit=limit
    )

    # 3. 创建加权排序器
    ranker = WeightedRanker(dense_weight, sparse_weight)

    # 4. 执行混合检索
    results = client.hybrid_search(
        collection_name=collection_name,
        reqs=[dense_req, sparse_req],
        ranker=ranker,
        limit=limit,
        output_fields=["text"]
    )

    return results[0]  # 返回第一个查询的结果
```

## 四、权重调整策略

### 4.1 权重调整原则

**1. 语义理解 vs 精确匹配**
- 稠密向量（dense）：捕获语义关系
- 稀疏向量（sparse）：精确关键词匹配

**2. 权重选择策略**
```python
# 场景1：语义搜索为主（问答系统）
dense_weight = 1.0
sparse_weight = 0.3

# 场景2：精确匹配为主（关键词搜索）
dense_weight = 0.3
sparse_weight = 1.0

# 场景3：平衡模式（通用搜索）
dense_weight = 1.0
sparse_weight = 1.0

# 场景4：强调语义（推荐系统）
dense_weight = 2.0
sparse_weight = 0.5
```

### 4.2 权重实验框架

```python
def weight_experiment(
    client: MilvusClient,
    collection_name: str,
    query_text: str,
    weight_configs: List[Tuple[float, float]],
    limit: int = 5
) -> Dict:
    """
    权重实验框架

    Args:
        weight_configs: [(dense_weight, sparse_weight), ...]

    Returns:
        实验结果字典
    """

    results = {}

    for dense_w, sparse_w in weight_configs:
        config_name = f"dense={dense_w}_sparse={sparse_w}"

        # 执行检索
        search_results = weighted_hybrid_search(
            client=client,
            collection_name=collection_name,
            query_text=query_text,
            dense_weight=dense_w,
            sparse_weight=sparse_w,
            limit=limit
        )

        results[config_name] = {
            "weights": {"dense": dense_w, "sparse": sparse_w},
            "results": search_results
        }

    return results


def print_experiment_results(results: Dict):
    """打印实验结果"""

    print("\n" + "="*80)
    print("权重实验结果对比")
    print("="*80)

    for config_name, data in results.items():
        print(f"\n【配置】{config_name}")
        print(f"  权重: dense={data['weights']['dense']}, sparse={data['weights']['sparse']}")
        print(f"  Top-{len(data['results'])} 结果:")

        for i, hit in enumerate(data['results'], 1):
            print(f"    {i}. [Score: {hit['distance']:.4f}] {hit['entity']['text'][:50]}...")
```

### 5.2 动态权重调整

```python
def adaptive_weight_search(
    client: MilvusClient,
    collection_name: str,
    query_text: str,
    query_type: str = "semantic"  # "semantic", "keyword", "balanced"
) -> List[Dict]:
    """
    根据查询类型动态调整权重

    Args:
        query_type: 查询类型
            - "semantic": 语义搜索（强调理解）
            - "keyword": 关键词搜索（强调精确）
            - "balanced": 平衡搜索
    """

    # 根据查询类型选择权重
    weight_map = {
        "semantic": (2.0, 0.5),   # 强调语义
        "keyword": (0.5, 2.0),    # 强调关键词
        "balanced": (1.0, 1.0),   # 平衡
    }

    dense_weight, sparse_weight = weight_map.get(query_type, (1.0, 1.0))

    print(f"查询类型: {query_type}")
    print(f"权重配置: dense={dense_weight}, sparse={sparse_weight}")

    return weighted_hybrid_search(
        client=client,
        collection_name=collection_name,
        query_text=query_text,
        dense_weight=dense_weight,
        sparse_weight=sparse_weight
    )
```

## 六、权重优化策略

### 6.1 基于评估指标的权重优化

```python
def evaluate_search_quality(
    results: List[Dict],
    ground_truth: List[str]
) -> Dict[str, float]:
    """
    评估检索质量

    Args:
        results: 检索结果
        ground_truth: 真实相关文档列表

    Returns:
        评估指标字典
    """

    # 提取检索到的文档
    retrieved_docs = [hit['entity']['text'] for hit in results]

    # 计算 Precision@K
    relevant_count = sum(1 for doc in retrieved_docs if doc in ground_truth)
    precision = relevant_count / len(retrieved_docs) if retrieved_docs else 0

    # 计算 Recall@K
    recall = relevant_count / len(ground_truth) if ground_truth else 0

    # 计算 F1 Score
    f1 = 2 * (precision * recall) / (precision + recall) if (precision + recall) > 0 else 0

    return {
        "precision": precision,
        "recall": recall,
        "f1": f1,
        "relevant_count": relevant_count
    }


def grid_search_weights(
    client: MilvusClient,
    collection_name: str,
    queries: List[str],
    ground_truths: List[List[str]],
    dense_range: List[float] = [0.5, 1.0, 1.5, 2.0],
    sparse_range: List[float] = [0.5, 1.0, 1.5, 2.0]
) -> Tuple[float, float, float]:
    """
    网格搜索最优权重

    Returns:
        (best_dense_weight, best_sparse_weight, best_f1_score)
    """

    best_f1 = 0
    best_weights = (1.0, 1.0)

    for dense_w in dense_range:
        for sparse_w in sparse_range:
            total_f1 = 0

            # 对每个查询评估
            for query, ground_truth in zip(queries, ground_truths):
                results = weighted_hybrid_search(
                    client=client,
                    collection_name=collection_name,
                    query_text=query,
                    dense_weight=dense_w,
                    sparse_weight=sparse_w
                )

                metrics = evaluate_search_quality(results, ground_truth)
                total_f1 += metrics['f1']

            # 计算平均 F1
            avg_f1 = total_f1 / len(queries)

            if avg_f1 > best_f1:
                best_f1 = avg_f1
                best_weights = (dense_w, sparse_w)

    return best_weights[0], best_weights[1], best_f1
```

### 6.2 权重归一化

```python
def normalize_weights(
    dense_weight: float,
    sparse_weight: float
) -> Tuple[float, float]:
    """
    归一化权重（可选）

    注意：Milvus 的 WeightedRanker 不要求归一化，
    但归一化可以让权重更直观
    """

    total = dense_weight + sparse_weight

    if total == 0:
        return 0.5, 0.5

    return dense_weight / total, sparse_weight / total


# 使用示例
dense_w, sparse_w = normalize_weights(2.0, 1.0)
print(f"归一化权重: dense={dense_w:.2f}, sparse={sparse_w:.2f}")
# 输出: 归一化权重: dense=0.67, sparse=0.33
```

## 七、最佳实践

### 7.1 权重选择建议

| 应用场景 | 推荐权重 | 说明 |
|----------|----------|------|
| 问答系统 | dense=2.0, sparse=0.5 | 强调语义理解 |
| 关键词搜索 | dense=0.5, sparse=2.0 | 强调精确匹配 |
| 文档检索 | dense=1.0, sparse=1.0 | 平衡语义和关键词 |
| 推荐系统 | dense=3.0, sparse=1.0 | 强调语义相似度 |
| 代码搜索 | dense=0.3, sparse=2.0 | 强调精确匹配 |

### 7.2 注意事项

1. **权重不需要归一化**
   - Milvus 会自动处理权重
   - 可以使用任意正数

2. **权重为 0 的情况**
   - 设置为 0 表示忽略该向量字段
   - 至少保留一个非零权重

3. **权重调优建议**
   - 从平衡权重（1.0, 1.0）开始
   - 根据业务需求逐步调整
   - 使用评估指标验证效果

4. **性能考虑**
   - 权重不影响检索性能
   - 仅影响结果融合阶段
