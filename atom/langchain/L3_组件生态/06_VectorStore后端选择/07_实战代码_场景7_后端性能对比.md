## 场景说明

本文档提供完整的向量存储后端性能对比测试框架，帮助你：
- 对比不同后端的性能表现
- 评估各后端在不同数据规模下的表现
- 为生产环境选择合适的后端
- 优化现有系统的性能

**测试维度**：
- 构建时间：索引构建速度
- 检索延迟：查询响应时间
- 吞吐量：每秒查询数（QPS）
- 内存占用：运行时内存消耗
- 准确率：检索结果准确性

### 示例1：性能测试框架

```python
"""
向量存储后端性能对比测试框架
提供完整的性能测试和对比分析
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_chroma import Chroma
from langchain_community.vectorstores import FAISS
from langchain_core.documents import Document
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import time
import numpy as np
import pandas as pd
from typing import List, Dict, Any
import os

# 加载环境变量
load_dotenv()


class VectorStorePerformanceTester:
    """向量存储性能测试器"""
    
    def __init__(self, embeddings):
        self.embeddings = embeddings
        self.results = []
    
    def generate_test_documents(self, size: int) -> List[Document]:
        """生成测试文档"""
        documents = [
            Document(
                page_content=f"测试文档 {i}：这是一段较长的测试内容，用于模拟真实场景。" * 10,
                metadata={"id": i, "category": f"cat_{i % 5}"}
            )
            for i in range(size)
        ]
        return documents
    
    def test_inmemory(self, documents: List[Document]) -> Dict[str, Any]:
        """测试 InMemory 后端"""
        print(f"\n[测试] InMemory - {len(documents)} 个文档")
        
        # 1. 构建索引
        start_time = time.time()
        vector_store = InMemoryVectorStore(self.embeddings)
        vector_store.add_documents(documents)
        build_time = time.time() - start_time
        
        # 2. 检索测试
        query = "测试查询"
        search_times = []
        
        for _ in range(10):
            start_time = time.time()
            vector_store.similarity_search(query, k=10)
            search_times.append(time.time() - start_time)
        
        avg_search_time = np.mean(search_times)
        std_search_time = np.std(search_times)
        
        return {
            "backend": "InMemory",
            "doc_count": len(documents),
            "build_time": build_time,
            "avg_search_time": avg_search_time,
            "std_search_time": std_search_time,
            "qps": 1 / avg_search_time
        }
    
    def test_chroma(self, documents: List[Document]) -> Dict[str, Any]:
        """测试 Chroma 后端"""
        print(f"\n[测试] Chroma - {len(documents)} 个文档")
        
        # 清理旧数据
        collection_name = f"test_{len(documents)}"
        
        # 1. 构建索引
        start_time = time.time()
        vector_store = Chroma(
            collection_name=collection_name,
            embedding_function=self.embeddings
        )
        vector_store.add_documents(documents)
        build_time = time.time() - start_time
        
        # 2. 检索测试
        query = "测试查询"
        search_times = []
        
        for _ in range(10):
            start_time = time.time()
            vector_store.similarity_search(query, k=10)
            search_times.append(time.time() - start_time)
        
        avg_search_time = np.mean(search_times)
        std_search_time = np.std(search_times)
        
        # 清理
        vector_store.delete_collection()
        
        return {
            "backend": "Chroma",
            "doc_count": len(documents),
            "build_time": build_time,
            "avg_search_time": avg_search_time,
            "std_search_time": std_search_time,
            "qps": 1 / avg_search_time
        }
    
    def test_faiss(self, documents: List[Document]) -> Dict[str, Any]:
        """测试 FAISS 后端"""
        print(f"\n[测试] FAISS - {len(documents)} 个文档")
        
        # 1. 构建索引
        start_time = time.time()
        vector_store = FAISS.from_documents(
            documents=documents,
            embedding=self.embeddings
        )
        build_time = time.time() - start_time
        
        # 2. 检索测试
        query = "测试查询"
        search_times = []
        
        for _ in range(10):
            start_time = time.time()
            vector_store.similarity_search(query, k=10)
            search_times.append(time.time() - start_time)
        
        avg_search_time = np.mean(search_times)
        std_search_time = np.std(search_times)
        
        return {
            "backend": "FAISS",
            "doc_count": len(documents),
            "build_time": build_time,
            "avg_search_time": avg_search_time,
            "std_search_time": std_search_time,
            "qps": 1 / avg_search_time
        }
    
    def run_benchmark(self, sizes: List[int]):
        """运行完整的基准测试"""
        print("=" * 60)
        print("向量存储后端性能对比测试")
        print("=" * 60)
        
        for size in sizes:
            print(f"\n{'=' * 60}")
            print(f"数据规模: {size} 个文档")
            print(f"{'=' * 60}")
            
            # 生成测试数据
            documents = self.generate_test_documents(size)
            
            # 测试各个后端
            self.results.append(self.test_inmemory(documents))
            self.results.append(self.test_chroma(documents))
            self.results.append(self.test_faiss(documents))
        
        return self.results
    
    def generate_report(self):
        """生成测试报告"""
        df = pd.DataFrame(self.results)
        
        print("\n" + "=" * 60)
        print("性能测试报告")
        print("=" * 60)
        
        # 按数据规模分组
        for size in df['doc_count'].unique():
            print(f"\n数据规模: {size} 个文档")
            print("-" * 60)
            
            subset = df[df['doc_count'] == size]
            
            print(f"\n{'后端':<15} {'构建时间(s)':<15} {'检索时间(ms)':<15} {'QPS':<10}")
            print("-" * 60)
            
            for _, row in subset.iterrows():
                print(f"{row['backend']:<15} "
                      f"{row['build_time']:<15.2f} "
                      f"{row['avg_search_time']*1000:<15.2f} "
                      f"{row['qps']:<10.1f}")
        
        # 总结
        print("\n" + "=" * 60)
        print("性能排名总结")
        print("=" * 60)
        
        # 构建速度排名
        print("\n构建速度排名（越快越好）:")
        build_ranking = df.groupby('backend')['build_time'].mean().sort_values()
        for i, (backend, time_val) in enumerate(build_ranking.items(), 1):
            print(f"  {i}. {backend}: {time_val:.2f}秒")
        
        # 检索速度排名
        print("\n检索速度排名（越快越好）:")
        search_ranking = df.groupby('backend')['avg_search_time'].mean().sort_values()
        for i, (backend, time_val) in enumerate(search_ranking.items(), 1):
            print(f"  {i}. {backend}: {time_val*1000:.2f}ms")
        
        # QPS 排名
        print("\n吞吐量排名（越高越好）:")
        qps_ranking = df.groupby('backend')['qps'].mean().sort_values(ascending=False)
        for i, (backend, qps) in enumerate(qps_ranking.items(), 1):
            print(f"  {i}. {backend}: {qps:.1f} QPS")
        
        return df


def main():
    """主函数"""
    print("\n" + "=" * 60)
    print("向量存储后端性能对比测试")
    print("=" * 60)
    
    # 初始化
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    tester = VectorStorePerformanceTester(embeddings)
    
    # 运行基准测试
    sizes = [100, 500, 1000]
    tester.run_benchmark(sizes)
    
    # 生成报告
    df = tester.generate_report()
    
    # 保存结果
    df.to_csv("vectorstore_benchmark_results.csv", index=False)
    print("\n结果已保存到: vectorstore_benchmark_results.csv")
    
    print("\n" + "=" * 60)
    print("测试完成！")
    print("=" * 60)


if __name__ == "__main__":
    main()
```

### 示例2：详细性能分析

```python
"""
详细性能分析工具
提供更深入的性能指标分析
"""

import psutil
import tracemalloc
from typing import Dict, Any


class DetailedPerformanceAnalyzer:
    """详细性能分析器"""
    
    def __init__(self, embeddings):
        self.embeddings = embeddings
    
    def analyze_memory_usage(self, vector_store, documents: List[Document]) -> Dict[str, Any]:
        """分析内存使用情况"""
        # 开始内存跟踪
        tracemalloc.start()
        process = psutil.Process()
        
        # 记录初始内存
        mem_before = process.memory_info().rss / 1024 / 1024  # MB
        
        # 添加文档
        vector_store.add_documents(documents)
        
        # 记录最终内存
        mem_after = process.memory_info().rss / 1024 / 1024  # MB
        current, peak = tracemalloc.get_traced_memory()
        tracemalloc.stop()
        
        return {
            "mem_before_mb": mem_before,
            "mem_after_mb": mem_after,
            "mem_increase_mb": mem_after - mem_before,
            "peak_mb": peak / 1024 / 1024
        }
    
    def analyze_search_latency_distribution(self, vector_store, query: str, iterations: int = 100) -> Dict[str, Any]:
        """分析检索延迟分布"""
        latencies = []
        
        for _ in range(iterations):
            start_time = time.time()
            vector_store.similarity_search(query, k=10)
            latencies.append((time.time() - start_time) * 1000)  # ms
        
        latencies = np.array(latencies)
        
        return {
            "mean_ms": np.mean(latencies),
            "median_ms": np.median(latencies),
            "p50_ms": np.percentile(latencies, 50),
            "p95_ms": np.percentile(latencies, 95),
            "p99_ms": np.percentile(latencies, 99),
            "min_ms": np.min(latencies),
            "max_ms": np.max(latencies),
            "std_ms": np.std(latencies)
        }
    
    def analyze_concurrent_performance(self, vector_store, query: str, concurrent_users: int = 10) -> Dict[str, Any]:
        """分析并发性能"""
        import concurrent.futures
        
        def single_search():
            start_time = time.time()
            vector_store.similarity_search(query, k=10)
            return time.time() - start_time
        
        # 并发测试
        start_time = time.time()
        with concurrent.futures.ThreadPoolExecutor(max_workers=concurrent_users) as executor:
            futures = [executor.submit(single_search) for _ in range(concurrent_users * 10)]
            latencies = [f.result() for f in concurrent.futures.as_completed(futures)]
        total_time = time.time() - start_time
        
        return {
            "concurrent_users": concurrent_users,
            "total_requests": len(latencies),
            "total_time_s": total_time,
            "avg_latency_ms": np.mean(latencies) * 1000,
            "throughput_qps": len(latencies) / total_time
        }


def run_detailed_analysis():
    """运行详细性能分析"""
    print("\n" + "=" * 60)
    print("详细性能分析")
    print("=" * 60)
    
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    analyzer = DetailedPerformanceAnalyzer(embeddings)
    
    # 生成测试数据
    documents = [
        Document(page_content=f"文档 {i}：测试内容。" * 10)
        for i in range(1000)
    ]
    
    # 测试 FAISS
    print("\n[分析] FAISS 详细性能")
    vector_store = FAISS.from_documents(documents, embeddings)
    
    # 内存分析
    print("\n1. 内存使用分析:")
    # 注意：这里需要重新创建 vector_store 来测试内存
    mem_stats = analyzer.analyze_memory_usage(
        InMemoryVectorStore(embeddings),
        documents[:100]
    )
    for key, value in mem_stats.items():
        print(f"  {key}: {value:.2f}")
    
    # 延迟分布分析
    print("\n2. 检索延迟分布:")
    latency_stats = analyzer.analyze_search_latency_distribution(
        vector_store,
        "测试查询"
    )
    for key, value in latency_stats.items():
        print(f"  {key}: {value:.2f}")
    
    # 并发性能分析
    print("\n3. 并发性能分析:")
    concurrent_stats = analyzer.analyze_concurrent_performance(
        vector_store,
        "测试查询",
        concurrent_users=5
    )
    for key, value in concurrent_stats.items():
        print(f"  {key}: {value:.2f}" if isinstance(value, float) else f"  {key}: {value}")


if __name__ == "__main__":
    run_detailed_analysis()
```

## 选择建议

### 场景1：快速原型（< 1K 文档）
**推荐**：InMemory
- 零配置
- 快速验证
- 完整功能

### 场景2：本地开发（< 100K 文档）
**推荐**：Chroma
- 持久化
- 易用性高
- 功能完整

### 场景3：高性能需求（静态数据）
**推荐**：FAISS
- 性能极佳
- 本地运行
- 适合静态数据

### 场景4：生产环境（> 100K 文档）
**推荐**：Qdrant 或 Pinecone
- 分布式
- 高可用
- 企业级功能

## 常见问题

### Q1: 如何选择合适的测试规模？

根据实际应用场景：
- 原型验证：100 - 1K
- 中小规模：1K - 100K

### Q2: 性能测试结果不稳定怎么办？

1. 增加测试迭代次数
2. 预热系统（先运行几次）
3. 关闭其他应用
4. 使用固定的测试数据

### Q3: 如何对比生产环境性能？

1. 使用真实数据
2. 模拟真实查询模式
3. 测试并发场景
