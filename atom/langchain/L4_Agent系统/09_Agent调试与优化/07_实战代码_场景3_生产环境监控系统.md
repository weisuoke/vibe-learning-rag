## 场景描述

**目标：** 构建一个生产级 Agent 监控系统，集成 LangSmith 追踪、异常告警、自动评估和运行指标面板。

**核心挑战：**
- 开发环境的调试手段在生产中不适用（不能打印全部日志）
- 需要低开销、高可靠的监控方案
- 异常发生时要自动告警，不能等人发现

**技术栈：**
- LangSmith 追踪（或本地替代方案）
- 自定义回调处理器
- 结构化日志 + 指标收集
- 告警通知

"""
Agent 生产环境监控系统
演示：LangSmith 集成 + 异常告警 + 自动评估 + 运行指标面板

功能：
1. 生产级监控回调（低开销、结构化日志）
2. 异常检测与告警
3. 运行指标聚合与报告
4. LangSmith 追踪集成
"""

import os
import json
import time
import logging
from datetime import datetime, timedelta
from typing import Any, Optional
from collections import defaultdict

# ===== 配置结构化日志（生产环境基础） =====
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(name)s: %(message)s',
    datefmt='%Y-%m-%d %H:%M:%S'

logger = logging.getLogger("agent_monitor")

# 第一部分：生产级监控回调处理器

from langchain_core.callbacks import BaseCallbackHandler


class ProductionMonitorHandler(BaseCallbackHandler):
    """
    生产环境 Agent 监控回调处理器

    设计原则：
    - 低开销：只记录关键指标，不记录完整输入输出
    - 结构化：所有日志使用 JSON 格式，便于日志平台解析
    - 可告警：异常自动触发告警通知
    - 可聚合：指标支持时间窗口聚合
    """

    def __init__(
        self,
        alert_callback=None,          # 告警回调函数
        slow_threshold_seconds: float = 10.0,  # 慢请求阈值
        max_token_budget: int = 10000,         # 单次 Token 预算
        error_rate_threshold: float = 0.1,     # 错误率告警阈值
    ):
        super().__init__()
        self.alert_callback = alert_callback or self._default_alert
        self.slow_threshold = slow_threshold_seconds
        self.max_token_budget = max_token_budget
        self.error_rate_threshold = error_rate_threshold

        # 运行时状态
        self._run_starts = {}       # run_id -> start_time
        self._run_metadata = {}     # run_id -> metadata

        # 聚合指标
        self.metrics = {
            "total_runs": 0,
            "successful_runs": 0,
            "failed_runs": 0,
            "total_tokens": 0,
            "total_cost_usd": 0.0,
            "total_latency_ms": 0,
            "tool_calls": defaultdict(int),
            "error_types": defaultdict(int),
            "hourly_stats": defaultdict(lambda: {
                "runs": 0, "errors": 0, "tokens": 0, "latency_ms": 0
            }),
        }

        # 告警历史（防止重复告警）
        self._alert_cooldown = {}

    # ----- LLM 事件 -----

    def on_llm_start(self, serialized, prompts, *, run_id, **kwargs):
        """LLM 调用开始 - 记录开始时间"""
        self._run_starts[run_id] = time.time()
        self._run_metadata[run_id] = {
            "type": "llm",
            "model": serialized.get("kwargs", {}).get("model_name", "unknown"),
        }

    def on_llm_end(self, response, *, run_id, **kwargs):
        """LLM 调用结束 - 收集指标"""
        duration_ms = self._calc_duration(run_id)

        # 提取 Token 用量
        token_usage = {}
        if hasattr(response, "llm_output") and response.llm_output:
            token_usage = response.llm_output.get("token_usage", {})

        total_tokens = token_usage.get("total_tokens", 0)

        # 更新聚合指标
        self.metrics["total_runs"] += 1
        self.metrics["successful_runs"] += 1
        self.metrics["total_tokens"] += total_tokens
        self.metrics["total_latency_ms"] += duration_ms

        # 计算成本（简化估算）
        cost = self._estimate_cost(
            token_usage,
            self._run_metadata.get(run_id, {}).get("model", "unknown")
        )
        self.metrics["total_cost_usd"] += cost

        # 更新小时统计
        hour_key = datetime.now().strftime("%Y-%m-%d %H:00")
        self.metrics["hourly_stats"][hour_key]["runs"] += 1
        self.metrics["hourly_stats"][hour_key]["tokens"] += total_tokens
        self.metrics["hourly_stats"][hour_key]["latency_ms"] += duration_ms

        # 结构化日志
        log_entry = {
            "event": "llm_complete",
            "run_id": str(run_id),
            "duration_ms": duration_ms,
            "tokens": total_tokens,
            "cost_usd": round(cost, 6),
        }
        logger.info(json.dumps(log_entry))

        # 检查告警条件
        self._check_slow_request(run_id, duration_ms)
        self._check_token_budget(run_id, total_tokens)

    def on_llm_error(self, error, *, run_id, **kwargs):
        """LLM 调用失败 - 记录错误并告警"""
        duration_ms = self._calc_duration(run_id)
        error_type = type(error).__name__

        self.metrics["total_runs"] += 1
        self.metrics["failed_runs"] += 1
        self.metrics["error_types"][error_type] += 1

        hour_key = datetime.now().strftime("%Y-%m-%d %H:00")
        self.metrics["hourly_stats"][hour_key]["errors"] += 1

        log_entry = {
            "event": "llm_error",
            "run_id": str(run_id),
            "error_type": error_type,
            "error_msg": str(error)[:200],  # 截断，避免日志过长
            "duration_ms": duration_ms,
        }
        logger.error(json.dumps(log_entry))

        # 错误直接告警
        self._send_alert(
            level="ERROR",
            title=f"LLM 调用失败: {error_type}",
            details=str(error)[:500],
            run_id=run_id
        )

        # 检查错误率
        self._check_error_rate()

    # ----- 工具事件 -----

    def on_tool_start(self, serialized, input_str, *, run_id, **kwargs):
        """工具调用开始"""
        self._run_starts[run_id] = time.time()
        tool_name = serialized.get("name", "unknown")
        self._run_metadata[run_id] = {"type": "tool", "name": tool_name}
        self.metrics["tool_calls"][tool_name] += 1

    def on_tool_error(self, error, *, run_id, **kwargs):
        """工具调用失败"""
        tool_name = self._run_metadata.get(run_id, {}).get("name", "unknown")
        duration_ms = self._calc_duration(run_id)

        log_entry = {
            "event": "tool_error",
            "tool": tool_name,
            "error": str(error)[:200],
            "duration_ms": duration_ms,
        }
        logger.error(json.dumps(log_entry))

        self._send_alert(
            level="WARNING",
            title=f"工具调用失败: {tool_name}",
            details=str(error)[:500],
            run_id=run_id
        )

    # ----- Agent 事件 -----

    def on_agent_action(self, action, *, run_id, **kwargs):
        """Agent 决定使用工具 - 记录决策"""
        log_entry = {
            "event": "agent_action",
            "tool": action.tool,
            "run_id": str(run_id),
        }
        logger.info(json.dumps(log_entry))

    def on_agent_finish(self, finish, *, run_id, **kwargs):
        """Agent 完成 - 记录最终状态"""
        log_entry = {
            "event": "agent_finish",
            "run_id": str(run_id),
        }
        logger.info(json.dumps(log_entry))

    # ----- 告警逻辑 -----

    def _check_slow_request(self, run_id, duration_ms):
        """检查慢请求"""
        if duration_ms > self.slow_threshold * 1000:
            self._send_alert(
                level="WARNING",
                title=f"慢请求检测: {duration_ms:.0f}ms",
                details=f"阈值: {self.slow_threshold}s, 实际: {duration_ms/1000:.2f}s",
                run_id=run_id
            )

    def _check_token_budget(self, run_id, tokens):
        """检查 Token 预算"""
        if tokens > self.max_token_budget:
            self._send_alert(
                level="WARNING",
                title=f"Token 预算超限: {tokens}",
                details=f"预算: {self.max_token_budget}, 实际: {tokens}",
                run_id=run_id
            )

    def _check_error_rate(self):
        """检查错误率"""
        total = self.metrics["total_runs"]
        if total < 10:  # 样本太少不告警
            return
        error_rate = self.metrics["failed_runs"] / total
        if error_rate > self.error_rate_threshold:
            self._send_alert(
                level="CRITICAL",
                title=f"错误率过高: {error_rate:.1%}",
                details=f"总运行: {total}, 失败: {self.metrics['failed_runs']}",
            )

    def _send_alert(self, level, title, details="", run_id=None):
        """发送告警（带冷却机制，防止告警风暴）"""
        alert_key = f"{level}:{title}"
        now = time.time()

        # 5分钟冷却期
        if alert_key in self._alert_cooldown:
            if now - self._alert_cooldown[alert_key] < 300:
                return

        self._alert_cooldown[alert_key] = now

        alert = {
            "timestamp": datetime.now().isoformat(),
            "level": level,
            "title": title,
            "details": details,
            "run_id": str(run_id) if run_id else None,
        }

        self.alert_callback(alert)

    def _default_alert(self, alert):
        """默认告警处理：打印到日志"""
        level = alert["level"]
        log_func = {
            "WARNING": logger.warning,
            "ERROR": logger.error,
            "CRITICAL": logger.critical,
        }.get(level, logger.warning)

        log_func(f"🚨 告警 [{level}] {alert['title']}: {alert['details']}")

    # ----- 辅助方法 -----

    def _calc_duration(self, run_id):
        """计算运行耗时（毫秒）"""
        start = self._run_starts.pop(run_id, None)
        if start:
            return (time.time() - start) * 1000
        return 0

    def _estimate_cost(self, token_usage, model):
        """估算成本（美元）"""
        pricing = {
            "gpt-4o": {"input": 2.50 / 1_000_000, "output": 10.00 / 1_000_000},
            "gpt-4o-mini": {"input": 0.15 / 1_000_000, "output": 0.60 / 1_000_000},
        }

        rates = pricing.get(model, pricing["gpt-4o-mini"])
        input_tokens = token_usage.get("prompt_tokens", 0)
        output_tokens = token_usage.get("completion_tokens", 0)

        return input_tokens * rates["input"] + output_tokens * rates["output"]

class MetricsDashboard:
    """
    运行指标聚合面板

    从 ProductionMonitorHandler 收集指标，
    生成可读的运行报告。
    """

    def __init__(self, monitor: ProductionMonitorHandler):
        self.monitor = monitor

    def generate_report(self) -> str:
        """生成指标报告"""
        m = self.monitor.metrics
        total = m["total_runs"] or 1  # 避免除零

        report = []
        report.append("=" * 60)
        report.append("📊 Agent 生产环境监控报告")
        report.append(f"⏰ 生成时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        report.append("=" * 60)

        # 概览
        report.append("\n📈 概览指标")
        report.append(f"  总运行次数:   {m['total_runs']}")
        report.append(f"  成功次数:     {m['successful_runs']}")
        report.append(f"  失败次数:     {m['failed_runs']}")
        success_rate = m["successful_runs"] / total * 100
        report.append(f"  成功率:       {success_rate:.1f}%")

        # 性能
        report.append("\n⚡ 性能指标")
        avg_latency = m["total_latency_ms"] / total
        report.append(f"  总 Token 消耗: {m['total_tokens']:,}")
        report.append(f"  总成本:        ${m['total_cost_usd']:.4f}")
        report.append(f"  平均延迟:      {avg_latency:.0f}ms")

        # 工具使用
        if m["tool_calls"]:
            report.append("\n🔧 工具调用统计")
            for tool, count in sorted(
                m["tool_calls"].items(),
                key=lambda x: x[1],
                reverse=True
            ):
                report.append(f"  {tool}: {count}次")

        # 错误分布
        if m["error_types"]:
            report.append("\n❌ 错误类型分布")
            for err, count in sorted(
                m["error_types"].items(),
                key=lambda x: x[1],
                reverse=True
            ):
                report.append(f"  {err}: {count}次")

        # 小时趋势（最近3小时）
        if m["hourly_stats"]:
            report.append("\n📉 小时趋势（最近数据）")
            recent_hours = sorted(m["hourly_stats"].keys())[-3:]
            for hour in recent_hours:
                stats = m["hourly_stats"][hour]
                report.append(
                    f"  {hour} | "
                    f"运行:{stats['runs']} "
                    f"错误:{stats['errors']} "
                    f"Token:{stats['tokens']:,} "
                    f"延迟:{stats['latency_ms']:.0f}ms"
                )

        report.append("\n" + "=" * 60)
        return "\n".join(report)

def setup_langsmith_tracing(
    project_name: str = "agent-production",
    enable: bool = True
):
    """
    配置 LangSmith 追踪（生产环境）

    只需设置环境变量，LangChain 会自动上报追踪数据。
    """
    if enable:
        os.environ["LANGCHAIN_TRACING_V2"] = "true"
        os.environ["LANGCHAIN_PROJECT"] = project_name
        # LANGCHAIN_API_KEY 应通过密钥管理系统注入，不要硬编码

        logger.info(f"LangSmith 追踪已启用, 项目: {project_name}")
    else:
        os.environ.pop("LANGCHAIN_TRACING_V2", None)
        logger.info("LangSmith 追踪已禁用")


def create_monitored_agent(
    model_name: str = "gpt-4o-mini",
    tools: list = None,
    monitor: ProductionMonitorHandler = None,
):
    """
    创建带生产监控的 Agent

    将 ProductionMonitorHandler 与 LangSmith 追踪叠加使用：
    - LangSmith: 提供可视化追踪和回放
    - ProductionMonitorHandler: 提供实时指标和告警
    """
    from langchain_openai import ChatOpenAI
    from langchain_core.messages import HumanMessage, SystemMessage

    # 创建 LLM（带回调）
    callbacks = []
    if monitor:
        callbacks.append(monitor)

    llm = ChatOpenAI(
        model=model_name,
        temperature=0,
        callbacks=callbacks,
        max_retries=3,
    )

    # 如果有工具，绑定工具
    if tools:
        llm = llm.bind_tools(tools)

    return llm

class AgentEvaluator:
    """
    Agent 输出自动评估器

    在生产环境中持续评估 Agent 输出质量，
    发现质量下降时自动告警。
    """

    def __init__(self, monitor: ProductionMonitorHandler):
        self.monitor = monitor
        self.eval_results = []

    def evaluate_output(
        self,
        query: str,
        output: str,
        expected_keywords: list[str] = None,
    ) -> dict:
        """
        评估单次 Agent 输出

        评估维度：
        1. 响应长度（太短可能是空回答，太长可能是废话）
        2. 关键词命中（是否包含期望的关键信息）
        3. 格式完整性（是否有结构化输出）
        """
        result = {
            "timestamp": datetime.now().isoformat(),
            "query": query[:100],
            "scores": {},
            "passed": True,
        }

        # 评估1：响应长度
        output_len = len(output)
        if output_len < 10:
            result["scores"]["length"] = 0.0
            result["passed"] = False
        elif output_len < 50:
            result["scores"]["length"] = 0.5
        else:
            result["scores"]["length"] = 1.0

        # 评估2：关键词命中
        if expected_keywords:
            hits = sum(1 for kw in expected_keywords if kw.lower() in output.lower())
            keyword_score = hits / len(expected_keywords)
            result["scores"]["keyword_match"] = round(keyword_score, 2)
            if keyword_score < 0.3:
                result["passed"] = False

        # 评估3：是否为空或错误输出
        error_patterns = ["i don't know", "无法回答", "error", "exception"]
        has_error = any(p in output.lower() for p in error_patterns)
        result["scores"]["no_error"] = 0.0 if has_error else 1.0

        # 综合分数
        scores = result["scores"].values()
        result["overall_score"] = round(sum(scores) / len(scores), 2) if scores else 0

        self.eval_results.append(result)

        # 质量下降告警
        if not result["passed"]:
            self.monitor._send_alert(
                level="WARNING",
                title=f"Agent 输出质量异常 (分数: {result['overall_score']})",
                details=f"查询: {query[:100]}"
            )

        return result

    def get_quality_summary(self) -> dict:
        """获取质量摘要"""
        if not self.eval_results:
            return {"message": "暂无评估数据"}

        total = len(self.eval_results)
        passed = sum(1 for r in self.eval_results if r["passed"])
        avg_score = sum(r["overall_score"] for r in self.eval_results) / total

        return {
            "total_evaluated": total,
            "passed": passed,
            "failed": total - passed,
            "pass_rate": f"{passed/total*100:.1f}%",
            "avg_score": round(avg_score, 2),
        }

## 关键设计决策

### 为什么不只用 LangSmith？

| 维度 | LangSmith | 自定义监控 | 两者结合 |
|------|-----------|-----------|---------|
| 可视化追踪 | ✅ 强 | ❌ | ✅ |
| 实时告警 | ⚠️ 有限 | ✅ 灵活 | ✅ |
| 成本 | 💰 按量付费 | 🆓 自控 | 📊 可控 |
| 数据隐私 | ⚠️ 云端 | ✅ 本地 | ✅ 可选 |
| 自定义指标 | ⚠️ 受限 | ✅ 完全自定义 | ✅ |

**最佳实践：两者结合使用**
- LangSmith 负责调试和回放（开发 + 采样追踪）
- 自定义 Handler 负责实时指标和告警（全量监控）

### 告警冷却机制

```
第1次告警 → 立即发送 ✅
第2次相同告警（1分钟后）→ 冷却中，跳过 ⏳
第3次相同告警（6分钟后）→ 冷却结束，发送 ✅
```

**为什么需要？** 一个 Bug 可能在几分钟内触发数百次告警，冷却机制将其合并为少量通知。
