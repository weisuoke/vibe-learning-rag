# 实战代码 - 场景4：状态流转控制

## 场景概述

**状态流转控制**是 LangGraph 中管理节点执行顺序和状态更新的核心机制。通过 **triggers**、**条件边** 和 **reducers**，可以精确控制状态在图中的流转路径。

### 核心机制

1. **Triggers（触发器）**：控制节点何时被触发执行
2. **Conditional Edges（条件边）**：根据状态动态决定下一个节点
3. **Reducers（状态归约器）**：控制状态更新的合并策略
4. **Channel 机制**：底层的状态存储和传递

### 典型应用场景

- **多步推理流程**：根据中间结果决定下一步操作
- **错误重试机制**：失败时自动重试或回退
- **并行任务协调**：多个节点并行执行后合并结果
- **动态工作流**：根据运行时状态调整执行路径

## 代码示例 1：Triggers 触发机制

这个示例展示如何使用 triggers 控制节点的触发条件。

```python
"""
场景1：Triggers 触发机制
演示：控制节点何时被触发执行
"""

from typing import TypedDict, Annotated
from operator import add

from langgraph.graph import StateGraph, START, END

# ===== 1. 定义状态 =====
class WorkflowState(TypedDict):
    """工作流状态"""
    step: int
    data: Annotated[list[str], add]  # 使用 add reducer 累积数据
    trigger_count: Annotated[int, add]  # 累积触发次数

# ===== 2. 定义节点函数 =====
def step1(state: WorkflowState) -> WorkflowState:
    """步骤1：初始化数据"""
    print(f"[step1] 执行，当前 step={state['step']}")
    return {
        "step": 1,
        "data": ["step1 完成"],
        "trigger_count": 1
    }

def step2(state: WorkflowState) -> WorkflowState:
    """步骤2：仅在 step >= 1 时触发"""
    print(f"[step2] 执行，当前 step={state['step']}")
    return {
        "step": 2,
        "data": ["step2 完成"],
        "trigger_count": 1
    }

def step3(state: WorkflowState) -> WorkflowState:
    """步骤3：仅在 step >= 2 时触发"""
    print(f"[step3] 执行，当前 step={state['step']}")
    return {
        "step": 3,
        "data": ["step3 完成"],
        "trigger_count": 1
    }

# ===== 3. 构建图（使用 triggers）=====
def create_trigger_graph():
    """创建使用 triggers 的图"""
    builder = StateGraph(WorkflowState)

    # 添加节点
    builder.add_node("step1", step1)
    builder.add_node("step2", step2)
    builder.add_node("step3", step3)

    # 添加边：定义执行顺序
    builder.add_edge(START, "step1")
    builder.add_edge("step1", "step2")
    builder.add_edge("step2", "step3")
    builder.add_edge("step3", END)

    return builder.compile()

# ===== 4. 测试 =====
if __name__ == "__main__":
    print("=== 场景1：Triggers 触发机制 ===\n")

    graph = create_trigger_graph()

    # 测试：正常执行流程
    print("--- 测试：正常执行流程 ---")
    result = graph.invoke({"step": 0, "data": [], "trigger_count": 0})
    print(f"最终状态: step={result['step']}, data={result['data']}")
    print(f"触发次数: {result['trigger_count']}\n")

    print("✅ 场景1 完成")
```

```

[step1] 执行，当前 step=0
[step2] 执行，当前 step=1
[step3] 执行，当前 step=2
最终状态: step=3, data=['step1 完成', 'step2 完成', 'step3 完成']
触发次数: 3

```

## 代码示例 2：条件触发与动态路由

这个示例展示如何使用条件边根据状态动态决定执行路径。

```python
"""
场景2：条件触发与动态路由
演示：根据状态动态决定下一个节点
"""

from typing import TypedDict, Literal

from langgraph.graph import StateGraph, START, END

# ===== 1. 定义状态 =====
class DecisionState(TypedDict):
    """决策状态"""
    input_value: int
    result: str
    path: str

# ===== 2. 定义节点函数 =====
def analyze(state: DecisionState) -> DecisionState:
    """分析输入值"""
    value = state["input_value"]
    print(f"[analyze] 分析输入值: {value}")
    return {"result": f"分析完成: {value}"}

def process_small(state: DecisionState) -> DecisionState:
    """处理小值（< 10）"""
    print(f"[process_small] 处理小值")
    return {"result": "小值处理完成", "path": "small"}

def process_medium(state: DecisionState) -> DecisionState:
    """处理中值（10-100）"""
    print(f"[process_medium] 处理中值")
    return {"result": "中值处理完成", "path": "medium"}

def process_large(state: DecisionState) -> DecisionState:
    """处理大值（> 100）"""
    print(f"[process_large] 处理大值")
    return {"result": "大值处理完成", "path": "large"}

# ===== 3. 定义路由函数 =====
def route_by_value(state: DecisionState) -> Literal["small", "medium", "large"]:
    """根据输入值路由到不同的处理节点"""
    value = state["input_value"]
    if value < 10:
        return "small"
    elif value <= 100:
        return "medium"
    else:
        return "large"

# ===== 4. 构建图 =====
def create_conditional_graph():
    """创建使用条件边的图"""
    builder = StateGraph(DecisionState)

    # 添加节点
    builder.add_node("analyze", analyze)
    builder.add_node("small", process_small)
    builder.add_node("medium", process_medium)
    builder.add_node("large", process_large)

    # 添加边
    builder.add_edge(START, "analyze")

    # 添加条件边：根据状态动态路由
    builder.add_conditional_edges(
        "analyze",
        route_by_value,
        {
            "small": "small",
            "medium": "medium",
            "large": "large"
        }
    )

    # 所有处理节点都连接到 END
    builder.add_edge("small", END)
    builder.add_edge("medium", END)
    builder.add_edge("large", END)

    return builder.compile()

# ===== 5. 测试 =====
if __name__ == "__main__":
    print("=== 场景2：条件触发与动态路由 ===\n")

    graph = create_conditional_graph()

    # 测试1：小值
    print("--- 测试1：小值 (5) ---")
    result1 = graph.invoke({"input_value": 5, "result": "", "path": ""})
    print(f"结果: {result1['result']}, 路径: {result1['path']}\n")

    # 测试2：中值
    print("--- 测试2：中值 (50) ---")
    result2 = graph.invoke({"input_value": 50, "result": "", "path": ""})
    print(f"结果: {result2['result']}, 路径: {result2['path']}\n")

    # 测试3：大值
    print("--- 测试3：大值 (200) ---")
    result3 = graph.invoke({"input_value": 200, "result": "", "path": ""})
    print(f"结果: {result3['result']}, 路径: {result3['path']}\n")

    print("✅ 场景2 完成")
```

## 代码示例 3：Reducers 状态更新策略

这个示例展示如何使用 reducers 控制状态更新的合并策略。

```python
"""
场景3：Reducers 状态更新策略
演示：控制状态更新的合并方式（覆盖 vs 累加）
"""

from typing import TypedDict, Annotated
from operator import add

from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_core.messages import HumanMessage, AIMessage

# ===== 1. 定义状态（使用不同的 reducers）=====
class ReducerState(TypedDict):
    """演示不同 reducer 的状态"""
    # 默认覆盖（无 reducer）
    last_result: str

    # 使用 add reducer 累加数值
    counter: Annotated[int, add]

    # 使用 add reducer 累加列表
    items: Annotated[list[str], add]

    # 使用 add_messages reducer 管理消息历史
    messages: Annotated[list, add_messages]

# ===== 2. 定义节点函数 =====
def node_a(state: ReducerState) -> ReducerState:
    """节点 A：更新所有字段"""
    print("[node_a] 执行")
    return {
        "last_result": "来自 node_a",
        "counter": 1,
        "items": ["item_a"],
        "messages": [AIMessage(content="node_a 的消息")]
    }

def node_b(state: ReducerState) -> ReducerState:
    """节点 B：再次更新所有字段"""
    print("[node_b] 执行")
    return {
        "last_result": "来自 node_b",  # 覆盖
        "counter": 2,  # 累加：1 + 2 = 3
        "items": ["item_b"],  # 累加：["item_a", "item_b"]
        "messages": [AIMessage(content="node_b 的消息")]  # 追加
    }

def node_c(state: ReducerState) -> ReducerState:
    """节点 C：部分更新"""
    print("[node_c] 执行")
    # 只更新部分字段
    return {
        "counter": 3,  # 累加：3 + 3 = 6
        "items": ["item_c"]  # 累加：["item_a", "item_b", "item_c"]
        # last_result 和 messages 保持不变
    }

# ===== 3. 构建图 =====
def create_reducer_graph():
    """创建演示 reducers 的图"""
    builder = StateGraph(ReducerState)

    # 添加节点
    builder.add_node("node_a", node_a)
    builder.add_node("node_b", node_b)
    builder.add_node("node_c", node_c)

    # 添加边：顺序执行
    builder.add_edge(START, "node_a")
    builder.add_edge("node_a", "node_b")
    builder.add_edge("node_b", "node_c")
    builder.add_edge("node_c", END)

    return builder.compile()

# ===== 4. 测试 =====
if __name__ == "__main__":
    print("=== 场景3：Reducers 状态更新策略 ===\n")

    graph = create_reducer_graph()

    # 初始状态
    initial_state = {
        "last_result": "初始值",
        "counter": 0,
        "items": [],
        "messages": [HumanMessage(content="用户输入")]
    }

    print("--- 初始状态 ---")
    print(f"last_result: {initial_state['last_result']}")
    print(f"counter: {initial_state['counter']}")
    print(f"items: {initial_state['items']}")
    print(f"messages: {len(initial_state['messages'])} 条\n")

    # 执行图
    result = graph.invoke(initial_state)

    print("\n--- 最终状态 ---")
    print(f"last_result: {result['last_result']} (覆盖)")
    print(f"counter: {result['counter']} (累加: 0+1+2+3)")
    print(f"items: {result['items']} (累加)")
    print(f"messages: {len(result['messages'])} 条 (追加)")
    print(f"  - {[m.content for m in result['messages']]}\n")

    print("✅ 场景3 完成")
```

## 代码示例 4：错误重试与回退机制

这个示例展示如何使用状态流转控制实现错误重试和回退。

```python
"""
场景4：错误重试与回退机制
演示：失败时自动重试或回退到安全状态
"""

from typing import TypedDict, Literal
import random

from langgraph.graph import StateGraph, START, END

# ===== 1. 定义状态 =====
class RetryState(TypedDict):
    """重试状态"""
    task: str
    retry_count: int
    max_retries: int
    success: bool
    result: str

# ===== 2. 定义节点函数 =====
def execute_task(state: RetryState) -> RetryState:
    """执行任务（可能失败）"""
    retry_count = state["retry_count"]
    print(f"[execute_task] 尝试执行任务 (第 {retry_count + 1} 次)")

    # 模拟随机失败（70% 成功率）
    success = random.random() > 0.3

    if success:
        print(f"[execute_task] ✅ 任务成功")
        return {
            "retry_count": retry_count + 1,
            "success": True,
            "result": "任务执行成功"
        }
    else:
        print(f"[execute_task] ❌ 任务失败")
        return {
            "retry_count": retry_count + 1,
            "success": False,
            "result": "任务执行失败"
        }

def handle_failure(state: RetryState) -> RetryState:
    """处理失败"""
    print(f"[handle_failure] 处理失败，准备重试")
    return {"result": "准备重试"}

def handle_max_retries(state: RetryState) -> RetryState:
    """达到最大重试次数"""
    print(f"[handle_max_retries] 达到最大重试次数，放弃任务")
    return {"result": "任务失败：超过最大重试次数"}

# ===== 3. 定义路由函数 =====
def route_after_execution(state: RetryState) -> Literal["success", "retry", "max_retries"]:
    """根据执行结果路由"""
    if state["success"]:
        return "success"
    elif state["retry_count"] >= state["max_retries"]:
        return "max_retries"
    else:
        return "retry"

# ===== 4. 构建图 =====
def create_retry_graph():
    """创建重试机制的图"""
    builder = StateGraph(RetryState)

    # 添加节点
    builder.add_node("execute", execute_task)
    builder.add_node("handle_failure", handle_failure)
    builder.add_node("handle_max_retries", handle_max_retries)

    # 添加边
    builder.add_edge(START, "execute")

    # 添加条件边：根据执行结果路由
    builder.add_conditional_edges(
        "execute",
        route_after_execution,
        {
            "success": END,  # 成功则结束
            "retry": "handle_failure",  # 失败则处理失败
            "max_retries": "handle_max_retries"  # 超过最大重试次数
        }
    )

    # 处理失败后重新执行
    builder.add_edge("handle_failure", "execute")

    # 达到最大重试次数后结束
    builder.add_edge("handle_max_retries", END)

    return builder.compile()

# ===== 5. 测试 =====
if __name__ == "__main__":
    print("=== 场景4：错误重试与回退机制 ===\n")

    graph = create_retry_graph()

    # 测试：执行任务（可能需要多次重试）
    print("--- 测试：执行任务 ---")
    result = graph.invoke({
        "task": "示例任务",
        "retry_count": 0,
        "max_retries": 3,
        "success": False,
        "result": ""
    })

    print(f"\n最终结果: {result['result']}")
    print(f"重试次数: {result['retry_count']}")
    print(f"成功: {result['success']}\n")

    print("✅ 场景4 完成")
```

[execute_task] 尝试执行任务 (第 2 次)

## 核心要点总结

### 1. Triggers vs Conditional Edges

| 特性 | Triggers | Conditional Edges |
|------|----------|-------------------|
| 用途 | 控制节点何时触发 | 控制执行路径 |
| 定义位置 | PregelNode 内部 | add_conditional_edges() |
| 灵活性 | 基于 channel 更新 | 基于状态值 |
| 使用场景 | 依赖关系管理 | 动态路由决策 |

### 2. Reducers 类型

| Reducer | 行为 | 使用场景 |
|---------|------|----------|
| 无（默认） | 覆盖 | 单一值状态 |
| `operator.add` | 累加 | 计数器、列表 |
| `add_messages` | 追加消息 | 聊天历史 |
| 自定义函数 | 自定义逻辑 | 复杂合并 |

### 3. 状态流转模式

**顺序执行**：
```python
builder.add_edge("node1", "node2")
builder.add_edge("node2", "node3")
```

**条件分支**：
```python
builder.add_conditional_edges(
    "node1",
    route_function,
    {"path_a": "node2", "path_b": "node3"}
)
```

**循环重试**：
```python
builder.add_conditional_edges(
    "execute",
    check_result,
    {"success": END, "retry": "execute"}
)
```

### 4. 最佳实践

1. **使用 Reducers 管理累积状态**
   ```python
   # ✅ 推荐：使用 reducer
   messages: Annotated[list, add_messages]

   # ❌ 不推荐：手动拼接
   def node(state):
       return {"messages": state["messages"] + [new_msg]}
   ```

2. **条件边返回明确的字符串**
   ```python
   # ✅ 推荐：使用 Literal 类型
   def route(state) -> Literal["path_a", "path_b"]:
       return "path_a" if condition else "path_b"
   ```

3. **避免无限循环**
   ```python
   # ✅ 推荐：设置最大重试次数
   if state["retry_count"] >= max_retries:
       return "stop"
   ```

### 5. 常见陷阱

1. **误区：忘记定义 reducer**
   ```python
   # ❌ 错误：期望累加但没有定义 reducer
   class State(TypedDict):
       items: list  # 会被覆盖，不会累加

   # ✅ 正确：使用 reducer
   class State(TypedDict):
       items: Annotated[list, add]
   ```

2. **误区：条件边返回未定义的路径**
   ```python
   # ❌ 错误：返回的路径未在映射中定义
   def route(state):
       return "unknown_path"  # 会报错

   # ✅ 正确：确保所有可能的返回值都在映射中
   builder.add_conditional_edges(
       "node",
       route,
       {"path_a": "node_a", "path_b": "node_b"}
   )
   ```
