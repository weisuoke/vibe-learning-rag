## 场景描述

### 什么是异步节点？

**异步节点**是使用 `async def` 定义的节点函数，可以在执行过程中进行异步操作（如 API 调用、数据库查询、I/O 操作）而不阻塞其他节点的执行。

**核心特点**：
- 使用 `async def` 定义节点函数
- 可以使用 `await` 调用异步操作
- StateGraph 自动识别并处理异步节点
- 与同步节点可以混合使用
- 使用 `ainvoke()` 或 `astream()` 执行图

### 为什么需要异步节点？

**性能优化**：
- 外部 API 调用（LLM、Embedding、向量数据库）
- 数据库查询（PostgreSQL、MongoDB）
- 文件 I/O 操作
- 网络请求

**实际场景**：
- RAG 系统中的异步 Embedding 生成
- 并行调用多个 LLM API
- 异步向量检索
- 实时数据流处理

## 核心挑战

### 挑战1：异步节点定义

**问题**：如何正确定义异步节点函数？

**解决方案**：
```python
# ✅ 正确：使用 async def
async def async_node(state: State) -> dict:
    result = await some_async_operation()
    return {"field": result}

# ❌ 错误：同步函数中使用 await
def sync_node(state: State) -> dict:
    result = await some_async_operation()  # SyntaxError
    return {"field": result}
```

### 挑战2：混合同步和异步节点

**问题**：同步节点和异步节点能否在同一个图中使用？

**解决方案**：
- StateGraph 自动处理混合节点
- 同步节点在异步上下文中自动包装
- 使用 `ainvoke()` 执行整个图

### 挑战3：性能优化

**问题**：如何最大化异步节点的性能优势？

**解决方案**：
- 识别 I/O 密集型操作
- 使用异步库（aiohttp、asyncpg、motor）
- 避免在异步节点中使用阻塞操作
- 合理设计节点粒度

## 完整代码

### 场景1：基础异步节点

```python
"""
场景1：基础异步节点
展示如何定义和使用异步节点
"""
import asyncio
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    message: str
    result: str


async def async_fetch_data(state: State) -> dict:
    """异步节点：模拟 API 调用"""
    print(f"[异步节点] 开始处理: {state['message']}")

    # 模拟异步 API 调用（1秒延迟）
    await asyncio.sleep(1)

    result = f"异步处理完成: {state['message']}"
    print(f"[异步节点] 完成处理")

    return {"result": result}


# 构建图
builder = StateGraph(State)
builder.add_node("fetch", async_fetch_data)
builder.add_edge(START, "fetch")
builder.add_edge("fetch", END)

graph = builder.compile()


# 执行异步图
async def run_basic_async():
    print("=== 场景1：基础异步节点 ===\n")

    result = await graph.ainvoke({
        "message": "Hello Async World",
        "result": ""
    })

    print(f"\n最终结果: {result['result']}")


# 运行示例
if __name__ == "__main__":
    asyncio.run(run_basic_async())
```

### 场景2：混合同步和异步节点

```python
"""
场景2：混合同步和异步节点
展示同步节点和异步节点在同一个图中的使用
"""
import asyncio
import time
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    input: str
    sync_result: str
    async_result: str
    final_result: str


def sync_process(state: State) -> dict:
    """同步节点：CPU 密集型操作"""
    print(f"[同步节点] 开始处理: {state['input']}")

    # 模拟 CPU 密集型操作
    time.sleep(0.5)

    result = f"同步处理: {state['input'].upper()}"
    print(f"[同步节点] 完成处理")

    return {"sync_result": result}


async def async_process(state: State) -> dict:
    """异步节点：I/O 密集型操作"""
    print(f"[异步节点] 开始处理: {state['sync_result']}")

    # 模拟异步 I/O 操作
    await asyncio.sleep(1)

    result = f"异步处理: {state['sync_result']}"
    print(f"[异步节点] 完成处理")

    return {"async_result": result}


def sync_combine(state: State) -> dict:
    """同步节点：结果合并"""
    print(f"[同步节点] 合并结果")

    final = f"{state['sync_result']} + {state['async_result']}"

    return {"final_result": final}


# 构建图
builder = StateGraph(State)
builder.add_node("sync_process", sync_process)
builder.add_node("async_process", async_process)
builder.add_node("sync_combine", sync_combine)

builder.add_edge(START, "sync_process")
builder.add_edge("sync_process", "async_process")
builder.add_edge("async_process", "sync_combine")
builder.add_edge("sync_combine", END)

graph = builder.compile()


# 执行混合图
async def run_mixed_nodes():
    print("=== 场景2：混合同步和异步节点 ===\n")

    start_time = time.time()

    result = await graph.ainvoke({
        "input": "test data",
        "sync_result": "",
        "async_result": "",
        "final_result": ""
    })

    elapsed = time.time() - start_time

    print(f"\n最终结果: {result['final_result']}")
    print(f"总耗时: {elapsed:.2f}秒")


# 运行示例
if __name__ == "__main__":
    asyncio.run(run_mixed_nodes())
```

```python
"""
场景3：RAG 系统异步实现
实际场景：异步 Embedding 生成和向量检索
"""
import asyncio
from typing_extensions import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
import operator


class RAGState(TypedDict):
    query: str
    documents: Annotated[list[str], operator.add]
    embeddings: list[list[float]]
    retrieved_docs: list[str]
    answer: str


async def async_load_documents(state: RAGState) -> dict:
    """异步节点：加载文档"""
    print(f"[异步加载] 加载文档...")

    # 模拟异步文档加载
    await asyncio.sleep(0.5)

    docs = [
        "LangGraph 是一个用于构建状态化工作流的框架",
        "异步节点可以提高 I/O 密集型操作的性能",
        "StateGraph 自动处理同步和异步节点"
    ]

    print(f"[异步加载] 加载了 {len(docs)} 个文档")

    return {"documents": docs}


async def async_generate_embeddings(state: RAGState) -> dict:
    """异步节点：生成 Embeddings"""
    print(f"[异步 Embedding] 生成 {len(state['documents'])} 个文档的 embeddings...")

    # 模拟异步 Embedding API 调用
    await asyncio.sleep(1)

    # 模拟 embeddings（实际应调用 OpenAI/Cohere API）
    embeddings = [[0.1, 0.2, 0.3] for _ in state['documents']]

    print(f"[异步 Embedding] 生成完成")

    return {"embeddings": embeddings}


async def async_vector_search(state: RAGState) -> dict:
    """异步节点：向量检索"""
    print(f"[异步检索] 检索相关文档: {state['query']}")

    # 模拟异步向量数据库查询
    await asyncio.sleep(0.8)

    # 模拟检索结果（实际应调用 Milvus/Chroma）
    retrieved = state['documents'][:2]

    print(f"[异步检索] 检索到 {len(retrieved)} 个相关文档")

    return {"retrieved_docs": retrieved}


async def async_generate_answer(state: RAGState) -> dict:
    """异步节点：生成答案"""
    print(f"[异步生成] 生成答案...")

    # 模拟异步 LLM API 调用
    await asyncio.sleep(1.5)

    context = "\n".join(state['retrieved_docs'])
    answer = f"基于以下上下文：\n{context}\n\n回答：{state['query']} - 异步节点可以显著提高性能"

    print(f"[异步生成] 答案生成完成")

    return {"answer": answer}


# 构建 RAG 图
builder = StateGraph(RAGState)
builder.add_node("load_docs", async_load_documents)
builder.add_node("generate_embeddings", async_generate_embeddings)
builder.add_node("vector_search", async_vector_search)
builder.add_node("generate_answer", async_generate_answer)

builder.add_edge(START, "load_docs")
builder.add_edge("load_docs", "generate_embeddings")
builder.add_edge("generate_embeddings", "vector_search")
builder.add_edge("vector_search", "generate_answer")
builder.add_edge("generate_answer", END)

rag_graph = builder.compile()


# 执行 RAG 流程
async def run_rag_async():
    print("=== 场景3：RAG 系统异步实现 ===\n")

    import time
    start_time = time.time()

    result = await rag_graph.ainvoke({
        "query": "什么是异步节点？",
        "documents": [],
        "embeddings": [],
        "retrieved_docs": [],
        "answer": ""
    })

    elapsed = time.time() - start_time

    print(f"\n{'='*60}")
    print(f"查询: {result['query']}")
    print(f"答案: {result['answer']}")
    print(f"总耗时: {elapsed:.2f}秒")
    print(f"{'='*60}")


# 运行示例
if __name__ == "__main__":
    asyncio.run(run_rag_async())
```

```python
"""
场景4：性能对比
对比同步和异步节点的性能差异
"""
import asyncio
import time
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    data: str
    result: str


# 同步版本
def sync_api_call_1(state: State) -> dict:
    time.sleep(1)
    return {"data": state["data"] + " -> API1"}


def sync_api_call_2(state: State) -> dict:
    time.sleep(1)
    return {"data": state["data"] + " -> API2"}


def sync_api_call_3(state: State) -> dict:
    time.sleep(1)
    return {"result": state["data"] + " -> API3"}


# 异步版本
async def async_api_call_1(state: State) -> dict:
    await asyncio.sleep(1)
    return {"data": state["data"] + " -> API1"}


async def async_api_call_2(state: State) -> dict:
    await asyncio.sleep(1)
    return {"data": state["data"] + " -> API2"}


async def async_api_call_3(state: State) -> dict:
    await asyncio.sleep(1)
    return {"result": state["data"] + " -> API3"}


# 构建同步图
sync_builder = StateGraph(State)
sync_builder.add_node("api1", sync_api_call_1)
sync_builder.add_node("api2", sync_api_call_2)
sync_builder.add_node("api3", sync_api_call_3)
sync_builder.add_edge(START, "api1")
sync_builder.add_edge("api1", "api2")
sync_builder.add_edge("api2", "api3")
sync_builder.add_edge("api3", END)
sync_graph = sync_builder.compile()

# 构建异步图
async_builder = StateGraph(State)
async_builder.add_node("api1", async_api_call_1)
async_builder.add_node("api2", async_api_call_2)
async_builder.add_node("api3", async_api_call_3)
async_builder.add_edge(START, "api1")
async_builder.add_edge("api1", "api2")
async_builder.add_edge("api2", "api3")
async_builder.add_edge("api3", END)
async_graph = async_builder.compile()


# 性能对比
async def run_performance_comparison():
    print("=== 场景4：性能对比 ===\n")

    # 同步执行
    print("执行同步图...")
    sync_start = time.time()
    sync_result = sync_graph.invoke({"data": "Start", "result": ""})
    sync_elapsed = time.time() - sync_start
    print(f"同步结果: {sync_result['result']}")
    print(f"同步耗时: {sync_elapsed:.2f}秒\n")

    # 异步执行
    print("执行异步图...")
    async_start = time.time()
    async_result = await async_graph.ainvoke({"data": "Start", "result": ""})
    async_elapsed = time.time() - async_start
    print(f"异步结果: {async_result['result']}")
    print(f"异步耗时: {async_elapsed:.2f}秒\n")

    # 性能提升
    improvement = ((sync_elapsed - async_elapsed) / sync_elapsed) * 100
    print(f"{'='*60}")
    print(f"性能提升: {improvement:.1f}%")
    print(f"说明: 在顺序执行场景中，异步节点的优势主要体现在 I/O 等待时的非阻塞特性")
    print(f"{'='*60}")


# 运行示例
if __name__ == "__main__":
    asyncio.run(run_performance_comparison())
```

## 关键点解析

### 1. 异步节点定义

**核心语法**：
```python
async def async_node(state: State) -> dict:
    result = await some_async_operation()
    return {"field": result}
```

**关键点**：
- 使用 `async def` 定义函数
- 可以使用 `await` 调用异步操作
- 返回值与同步节点相同（dict）
- StateGraph 自动识别异步节点

### 2. StateGraph 自动处理

**自动识别**：
- StateGraph 通过函数签名识别异步节点
- 无需额外配置或标记
- 添加方式与同步节点完全相同

**示例**：
```python
# 添加异步节点（与同步节点相同）
builder.add_node("async_node", async_node)
```

### 3. 执行方法

**异步执行**：
```python
# 使用 ainvoke 执行异步图
result = await graph.ainvoke(initial_state)

# 使用 astream 流式执行
async for chunk in graph.astream(initial_state):
    print(chunk)
```

**同步执行**：
```python
# 即使图中有异步节点，也可以使用 invoke
# StateGraph 会自动处理
result = graph.invoke(initial_state)
```

### 4. 混合节点处理

**自动包装**：
- 同步节点在异步上下文中自动包装
- 异步节点在同步上下文中通过事件循环执行
- 开发者无需关心底层实现

### 5. 性能优势

**I/O 密集型操作**：
- 外部 API 调用
- 数据库查询
- 文件读写
- 网络请求

**注意**：
- CPU 密集型操作不适合异步
- 顺序执行的异步节点性能提升有限
- 并行执行才能最大化异步优势

## 常见问题

### Q1：异步节点和同步节点可以混用吗？

**A**：可以！StateGraph 自动处理混合节点。

```python
# ✅ 完全支持
builder.add_node("sync_node", sync_function)
builder.add_node("async_node", async_function)
```

### Q2：如何在异步节点中调用同步库？

**A**：使用 `asyncio.to_thread()` 或 `loop.run_in_executor()`。

```python
async def async_node(state: State) -> dict:
    # 在线程池中运行同步函数
    result = await asyncio.to_thread(sync_blocking_function, arg1, arg2)
    return {"result": result}
```

### Q3：异步节点的错误处理？

**A**：与同步节点相同，使用 try-except。

```python
async def async_node(state: State) -> dict:
    try:
        result = await risky_async_operation()
        return {"result": result}
    except Exception as e:
        return {"error": str(e)}
```

### Q4：如何调试异步节点？

**A**：使用 `asyncio` 的调试工具。

```python
# 启用 asyncio 调试模式
import asyncio
asyncio.run(main(), debug=True)

# 或设置环境变量
# PYTHONASYNCIODEBUG=1 python script.py
```

### Q5：异步节点的性能提升有多大？

**A**：取决于 I/O 等待时间。

- **I/O 密集型**：可提升 50%-90%
- **CPU 密集型**：几乎无提升
- **混合场景**：提升 20%-50%

## 最佳实践

### 1. 何时使用异步节点

**适合场景**：
- 外部 API 调用（LLM、Embedding、向量数据库）
- 数据库查询（PostgreSQL、MongoDB）
- 文件 I/O 操作
- 网络请求

**不适合场景**：
- CPU 密集型计算
- 纯内存操作
- 简单的数据转换

### 2. 异步库选择

**推荐异步库**：
```python
# HTTP 请求
import aiohttp

# 数据库
import asyncpg  # PostgreSQL
import motor    # MongoDB

# Redis
import aioredis

# 文件 I/O
import aiofiles
```

### 3. 避免阻塞操作

**❌ 错误示例**：
```python
async def bad_async_node(state: State) -> dict:
    # 阻塞操作！
    time.sleep(1)  # 应该使用 await asyncio.sleep(1)
    result = requests.get(url)  # 应该使用 aiohttp
    return {"result": result}
```

**✅ 正确示例**：
```python
async def good_async_node(state: State) -> dict:
    await asyncio.sleep(1)
    async with aiohttp.ClientSession() as session:
        async with session.get(url) as response:
            result = await response.text()
    return {"result": result}
```

### 4. 合理设计节点粒度

**粗粒度节点**（推荐）：
```python
async def coarse_grained_node(state: State) -> dict:
    # 一个节点完成多个相关的异步操作
    docs = await load_documents()
    embeddings = await generate_embeddings(docs)
    results = await vector_search(embeddings)
    return {"results": results}
```

**细粒度节点**（不推荐）：
```python
# 过度拆分，增加调度开销
async def load_node(state: State) -> dict: ...
async def embed_node(state: State) -> dict: ...
async def search_node(state: State) -> dict: ...
```

### 5. 错误处理和重试

```python
async def robust_async_node(state: State) -> dict:
    max_retries = 3
    for attempt in range(max_retries):
        try:
            result = await api_call()
            return {"result": result}
        except Exception as e:
            if attempt == max_retries - 1:
                return {"error": str(e)}
            await asyncio.sleep(2 ** attempt)  # 指数退避
```

## 总结

### 核心要点

1. **异步节点定义**：使用 `async def`，可以 `await` 异步操作
2. **自动处理**：StateGraph 自动识别和处理异步节点
3. **混合使用**：同步和异步节点可以在同一个图中使用
4. **执行方法**：使用 `ainvoke()` 或 `astream()` 执行异步图
5. **性能优势**：I/O 密集型操作可显著提升性能
