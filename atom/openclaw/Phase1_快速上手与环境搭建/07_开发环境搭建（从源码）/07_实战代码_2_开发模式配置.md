## 二、完整 watch 脚本

### 2.1 run-node.mjs 实现

```javascript
#!/usr/bin/env node

import { spawn } from 'child_process';
import { watch } from 'chokidar';
import { fileURLToPath } from 'url';
import { dirname, join } from 'path';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);
const rootDir = join(__dirname, '..');

let gatewayProcess = null;

// 启动 Gateway
function startGateway() {
  if (gatewayProcess) {
    gatewayProcess.kill();
  }

  gatewayProcess = spawn('node', ['dist/index.js'], {
    cwd: rootDir,
    stdio: 'inherit',
    env: { ...process.env, NODE_ENV: 'development' }
  });

  gatewayProcess.on('error', (err) => {
    console.error('❌ Gateway error:', err);
  });
}

// 重新编译
function rebuild() {
  console.log('⚙️  Rebuilding...');

  const build = spawn('pnpm', ['build'], {
    cwd: rootDir,
    stdio: 'inherit'
  });

  build.on('close', (code) => {
    if (code === 0) {
      console.log('✅ Build successful');
      startGateway();
    } else {
      console.error('❌ Build failed');
    }
  });
}

// 监听文件变化
const watcher = watch('src/**/*.ts', {
  cwd: rootDir,
  ignored: /(^|[\/\\])\../,
  persistent: true,
  ignoreInitial: true
});

watcher.on('change', (path) => {
  console.log(`📝 File changed: ${path}`);
  rebuild();
});

watcher.on('ready', () => {
  console.log('👀 Watching for changes...');
  rebuild();
});

// 优雅退出
process.on('SIGINT', () => {
  console.log('\n👋 Shutting down...');
  if (gatewayProcess) {
    gatewayProcess.kill();
  }
  watcher.close();
  process.exit(0);
});
```

**数据来源**: `sourcecode/openclaw/scripts/run-node.mjs`

## 十、总结

### 核心命令

```bash
# 开发模式
pnpm dev

# 调试模式
pnpm dev:debug

# 全部开发模式
pnpm dev:all
```

### 关键配置

1. **watch 模式**: 自动监听文件变化
2. **增量编译**: 只编译变更的文件
3. **热重载**: 自动重启服务
4. **调试配置**: VSCode 调试支持
