## 场景描述

### 业务需求

构建一个分布式计数器系统，需要：
1. **多节点并发更新**：多个节点可以同时更新计数器
2. **原子性保证**：确保计数器更新的原子性
3. **多维度统计**：支持多个计数器（访问量、点赞数、评论数等）
4. **实时聚合**：实时聚合来自多个节点的计数

### 技术挑战

- 如何处理并发更新？
- 如何保证计数的准确性？
- 如何避免计数丢失？
- 如何支持多个计数器？

### 解决方案

使用 LangGraph 的 `operator.add` Reducer：
- **数值累加**：自动累加来自多个节点的数值
- **并发安全**：Reducer 保证更新的原子性
- **灵活扩展**：支持多个计数器字段

## 核心代码实现

### 1. 状态定义 (counter/state.py)

```python
"""
状态定义模块
使用 operator.add Reducer 管理计数器
"""
from typing import Annotated, TypedDict
import operator


class CounterState(TypedDict):
    """
    计数器状态

    Attributes:
        total_count: 总计数，使用 operator.add 累加
        page_views: 页面访问量
        likes: 点赞数
        comments: 评论数
    """
    total_count: Annotated[int, operator.add]
    page_views: Annotated[int, operator.add]
    likes: Annotated[int, operator.add]
    comments: Annotated[int, operator.add]


class AnalyticsState(TypedDict):
    """
    分析统计状态

    包含多个维度的计数器
    """
    # 用户行为计数
    user_actions: Annotated[int, operator.add]

    # 错误计数
    error_count: Annotated[int, operator.add]

    # 成功计数
    success_count: Annotated[int, operator.add]

    # 处理时间（毫秒）
    processing_time_ms: Annotated[int, operator.add]

    # 请求数
    request_count: Annotated[int, operator.add]


class ListAggregationState(TypedDict):
    """
    列表聚合状态

    演示 operator.add 对列表的拼接
    """
    items: Annotated[list[str], operator.add]
    numbers: Annotated[list[int], operator.add]
```

### 2. 节点函数 (counter/nodes.py)

```python
"""
节点函数模块
实现计数器的核心逻辑
"""
from counter.state import CounterState, AnalyticsState
import random
import time


def page_view_node(state: CounterState) -> dict:
    """
    页面访问节点

    模拟页面访问，增加计数
    """
    # 模拟随机访问量
    views = random.randint(1, 10)

    return {
        "total_count": views,
        "page_views": views
    }


def like_node(state: CounterState) -> dict:
    """
    点赞节点

    模拟用户点赞
    """
    # 模拟随机点赞数
    likes = random.randint(0, 5)

    return {
        "total_count": likes,
        "likes": likes
    }


def comment_node(state: CounterState) -> dict:
    """
    评论节点

    模拟用户评论
    """
    # 模拟随机评论数
    comments = random.randint(0, 3)

    return {
        "total_count": comments,
        "comments": comments
    }


def analytics_node(state: AnalyticsState) -> dict:
    """
    分析统计节点

    模拟请求处理和统计
    """
    # 模拟处理时间
    start_time = time.time()
    time.sleep(random.uniform(0.01, 0.05))
    processing_time = int((time.time() - start_time) * 1000)

    # 模拟成功/失败
    is_success = random.random() > 0.1  # 90% 成功率

    return {
        "user_actions": 1,
        "error_count": 0 if is_success else 1,
        "success_count": 1 if is_success else 0,
        "processing_time_ms": processing_time,
        "request_count": 1
    }


def batch_increment_node(state: CounterState, increment: int = 1) -> dict:
    """
    批量增加节点

    一次性增加多个计数器
    """
    return {
        "total_count": increment,
        "page_views": increment,
        "likes": increment // 2,
        "comments": increment // 3
    }


def conditional_increment_node(state: CounterState) -> dict:
    """
    条件增加节点

    根据当前状态决定是否增加
    """
    current_total = state.get("total_count", 0)

    # 如果总数小于 100，增加 10
    if current_total < 100:
        return {
            "total_count": 10,
            "page_views": 10
        }
    else:
        # 否则只增加 1
        return {
            "total_count": 1,
            "page_views": 1
        }
```

### 3. 图构建 (counter/graph.py)

```python
"""
图构建模块
定义计数器系统的工作流
"""
from langgraph.graph import StateGraph, START, END
from counter.state import CounterState, AnalyticsState
from counter.nodes import (
    page_view_node,
    like_node,
    comment_node,
    analytics_node
)


def create_counter_graph():
    """
    创建计数器图（顺序执行）

    工作流：
    START -> page_view -> like -> comment -> END
    """
    builder = StateGraph(CounterState)

    # 添加节点
    builder.add_node("page_view", page_view_node)
    builder.add_node("like", like_node)
    builder.add_node("comment", comment_node)

    # 添加边（顺序执行）
    builder.add_edge(START, "page_view")
    builder.add_edge("page_view", "like")
    builder.add_edge("like", "comment")
    builder.add_edge("comment", END)

    # 编译
    graph = builder.compile()

    return graph


def create_parallel_counter_graph():
    """
    创建并行计数器图

    工作流：
    START -> [page_view, like, comment] -> END

    所有节点并行执行，计数会自动累加
    """
    builder = StateGraph(CounterState)

    # 添加节点
    builder.add_node("page_view", page_view_node)
    builder.add_node("like", like_node)
    builder.add_node("comment", comment_node)

    # 并行执行
    builder.add_edge(START, "page_view")
    builder.add_edge(START, "like")
    builder.add_edge(START, "comment")

    # 所有节点都连接到 END
    builder.add_edge("page_view", END)
    builder.add_edge("like", END)
    builder.add_edge("comment", END)

    # 编译
    graph = builder.compile()

    return graph


def create_analytics_graph():
    """
    创建分析统计图

    工作流：
    START -> [analytics x 5] -> END

    模拟 5 个并行请求
    """
    builder = StateGraph(AnalyticsState)

    # 添加 5 个分析节点
    for i in range(5):
        builder.add_node(f"analytics_{i}", analytics_node)
        builder.add_edge(START, f"analytics_{i}")
        builder.add_edge(f"analytics_{i}", END)

    # 编译
    graph = builder.compile()

    return graph
```

## 常见问题与解决方案

### 问题 1: 初始值处理

**问题描述**：
第一次更新时，如果初始值不存在会怎样？

**解决方案**：
```python
# BinaryOperatorAggregate 会自动处理初始值
# 第一个值直接赋值，不调用 Reducer

# 示例
initial_state = {}  # 空状态
result = graph.invoke(initial_state)
# 第一个节点的值会直接赋值，后续节点才使用 operator.add
```

### 问题 2: 负数累加

**问题描述**：
如何实现计数器的减少？

**解决方案**：
```python
def decrement_node(state: CounterState) -> dict:
    """减少计数器"""
    return {
        "total_count": -1,  # 返回负数
        "page_views": -1
    }

# operator.add 会正确处理负数
# 10 + (-1) = 9
```

### 问题 3: 并发安全性

**问题描述**：
多个节点并发更新时，如何保证计数准确？

**解决方案**：
```python
# LangGraph 的 Reducer 机制保证了原子性
# 所有节点的更新会被收集，然后按顺序应用 Reducer

# 内部实现（简化版）：
def update(values: list[int]) -> int:
    result = 0
    for value in values:
        result = operator.add(result, value)
    return result
```

### 问题 4: 溢出处理

**问题描述**：
计数器可能溢出吗？

**解决方案**：
```python
def safe_increment_node(state: CounterState) -> dict:
    """安全增加，防止溢出"""
    current = state.get("total_count", 0)

    # 检查是否接近最大值
    MAX_VALUE = 2**31 - 1
    if current > MAX_VALUE - 1000:
        print("警告：计数器接近最大值")
        return {"total_count": 0}  # 不增加

    return {"total_count": 1}
```

## 性能优化建议

### 1. 批量更新

```python
# 不推荐：多次小更新
for i in range(100):
    graph.invoke({"total_count": 1})

# 推荐：批量更新
graph.invoke({"total_count": 100})
```

### 2. 减少节点数量

```python
# 不推荐：每个计数器一个节点
def node1(state): return {"counter1": 1}
def node2(state): return {"counter2": 1}
def node3(state): return {"counter3": 1}

# 推荐：合并到一个节点
def combined_node(state):
    return {
        "counter1": 1,
        "counter2": 1,
        "counter3": 1
    }
```

### 3. 使用持久化

```python
from langgraph.checkpoint.memory import MemorySaver

# 使用 Checkpointer 持久化状态
memory = MemorySaver()
graph = builder.compile(checkpointer=memory)

# 累积更新
config = {"configurable": {"thread_id": "counter-1"}}
for i in range(10):
    graph.invoke({"total_count": 1}, config)
```

## 总结

本示例展示了如何使用 `operator.add` Reducer 构建计数器系统：

1. **数值累加**：自动累加来自多个节点的数值
2. **并发安全**：Reducer 保证更新的原子性
3. **多维度统计**：支持多个计数器字段
4. **灵活扩展**：可以轻松添加新的计数器

**关键要点**：
- `operator.add` 适用于数值、字符串、列表的累加
- 第一个值直接赋值，后续值使用 Reducer
- 支持负数累加（实现减少）
- 并发更新时保证原子性
- 适用于计数器、统计、聚合等场景
