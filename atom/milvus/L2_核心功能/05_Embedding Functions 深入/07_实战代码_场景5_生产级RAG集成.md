## 架构1:基于LangChain的RAG系统

### 1.1 系统架构

```
文档输入 → 文档解析 → 文本分块 → Embedding → Milvus存储
                                                    ↓
用户查询 → Query Embedding → 向量检索 → 上下文注入 → LLM生成 → 答案输出
```

**来源**: `reference/fetch_rag_implementation.md:35-50`

### 1.2 完整实现

```python
"""
生产级RAG系统 - LangChain集成
完整的文档处理、检索和生成流程
"""

import os
from typing import List, Dict, Optional
from dataclasses import dataclass
from dotenv import load_dotenv

# LangChain imports
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.document_loaders import PyPDFLoader, TextLoader
from langchain.schema import Document
from langchain.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

# Milvus imports
from pymilvus import (
    MilvusClient,
    Function,
    FunctionType,
    CollectionSchema,
    FieldSchema,
    DataType
)

load_dotenv()

@dataclass
class RAGConfig:
    """RAG系统配置"""
    # Milvus配置
    milvus_uri: str = "http://localhost:19530"
    collection_name: str = "rag_knowledge_base"
    
    # Embedding配置
    embedding_provider: str = "voyageai"
    embedding_model: str = "voyage-3-large"
    embedding_dim: int = 1024
    
    # 分块配置
    chunk_size: int = 1000
    chunk_overlap: int = 200
    
    # 检索配置
    top_k: int = 5
    
    # LLM配置
    llm_model: str = "gpt-4"
    temperature: float = 0.0

class ProductionRAGSystem:
    """生产级RAG系统"""
    
    def __init__(self, config: RAGConfig):
        self.config = config
        self.client = MilvusClient(uri=config.milvus_uri)
        self.llm = ChatOpenAI(
            model=config.llm_model,
            temperature=config.temperature
        )
        self._init_collection()
    
    def _init_collection(self):
        """初始化Milvus Collection"""
        # 定义Embedding Function
        embedding_func = Function(
            name=f"{self.config.embedding_provider}_ef",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["text"],
            output_field_names=["vector"],
            params={
                "provider": self.config.embedding_provider,
                "model_name": self.config.embedding_model,
                "dim": self.config.embedding_dim,
                "api_key": os.getenv("VOYAGEAI_API_KEY")
            }
        )
        
        # 定义Schema
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=65535),
            FieldSchema(name="source", dtype=DataType.VARCHAR, max_length=512),
            FieldSchema(name="page", dtype=DataType.INT64),
            FieldSchema(name="vector", dtype=DataType.FLOAT_VECTOR, dim=self.config.embedding_dim)
        ]
        
        schema = CollectionSchema(
            fields=fields,
            functions=[embedding_func],
            description="Production RAG knowledge base"
        )
        
        # 创建Collection
        if self.client.has_collection(self.config.collection_name):
            self.client.drop_collection(self.config.collection_name)
        
        self.client.create_collection(
            self.config.collection_name,
            schema=schema
        )
        
        print(f"✅ Collection创建成功: {self.config.collection_name}")
    
    def ingest_documents(self, file_paths: List[str]):
        """文档导入流程"""
        print(f"\n📥 开始导入文档: {len(file_paths)}个文件")
        
        all_chunks = []
        
        for file_path in file_paths:
            print(f"\n处理文件: {file_path}")
            
            # 1. 加载文档
            if file_path.endswith('.pdf'):
                loader = PyPDFLoader(file_path)
            elif file_path.endswith('.txt'):
                loader = TextLoader(file_path)
            else:
                print(f"⚠️ 不支持的文件格式: {file_path}")
                continue
            
            documents = loader.load()
            print(f"  加载了 {len(documents)} 页")
            
            # 2. 文本分块
            text_splitter = RecursiveCharacterTextSplitter(
                chunk_size=self.config.chunk_size,
                chunk_overlap=self.config.chunk_overlap,
                separators=["\n\n", "\n", ".", "!", "?", ",", " ", ""]
            )
            
            chunks = text_splitter.split_documents(documents)
            print(f"  分块后: {len(chunks)} 个chunks")
            
            # 3. 准备数据
            for i, chunk in enumerate(chunks):
                all_chunks.append({
                    "text": chunk.page_content,
                    "source": file_path,
                    "page": chunk.metadata.get("page", 0)
                })
        
        # 4. 批量插入Milvus
        print(f"\n💾 插入Milvus: {len(all_chunks)} 个chunks")
        
        batch_size = 64
        for i in range(0, len(all_chunks), batch_size):
            batch = all_chunks[i:i + batch_size]
            self.client.insert(self.config.collection_name, batch)
            print(f"  已插入: {min(i + batch_size, len(all_chunks))}/{len(all_chunks)}")
        
        print(f"\n✅ 文档导入完成")
    
    def retrieve(self, query: str) -> List[Dict]:
        """检索相关文档"""
        # 使用Milvus的embedding function自动处理query embedding
        results = self.client.search(
            collection_name=self.config.collection_name,
            data=[query],  # Milvus会自动调用embedding function
            limit=self.config.top_k,
            output_fields=["text", "source", "page"]
        )
        
        # 格式化结果
        retrieved_docs = []
        for hits in results:
            for hit in hits:
                retrieved_docs.append({
                    "text": hit["entity"]["text"],
                    "source": hit["entity"]["source"],
                    "page": hit["entity"]["page"],
                    "score": hit["distance"]
                })
        
        return retrieved_docs
    
    def generate_answer(self, query: str, context_docs: List[Dict]) -> Dict:
        """生成答案"""
        # 构建上下文
        context = "\n\n".join([
            f"[来源: {doc['source']}, 页码: {doc['page']}]\n{doc['text']}"
            for doc in context_docs
        ])
        
        # 构建Prompt
        prompt_template = ChatPromptTemplate.from_messages([
            ("system", """你是一个专业的问答助手。请根据提供的上下文回答用户问题。

要求:
1. 仅基于提供的上下文回答
2. 如果上下文中没有相关信息,明确说明
3. 引用具体的来源和页码
4. 保持回答简洁准确"""),
            ("user", """上下文:
{context}

问题: {question}

请回答:""")
        ])
        
        # 生成答案
        messages = prompt_template.format_messages(
            context=context,
            question=query
        )
        
        response = self.llm.invoke(messages)
        
        return {
            "answer": response.content,
            "sources": [
                {"source": doc["source"], "page": doc["page"]}
                for doc in context_docs
            ]
        }
    
    def query(self, question: str) -> Dict:
        """完整的RAG查询流程"""
        print(f"\n🔍 查询: {question}")
        
        # 1. 检索
        print("  检索相关文档...")
        retrieved_docs = self.retrieve(question)
        print(f"  找到 {len(retrieved_docs)} 个相关文档")
        
        # 2. 生成答案
        print("  生成答案...")
        result = self.generate_answer(question, retrieved_docs)
        
        return result

# 使用示例
def demo_production_rag():
    """演示生产级RAG系统"""
    # 1. 初始化系统
    config = RAGConfig(
        embedding_provider="voyageai",
        embedding_model="voyage-3-large",
        embedding_dim=1024,
        chunk_size=1000,
        chunk_overlap=200,
        top_k=5
    )
    
    rag_system = ProductionRAGSystem(config)
    
    # 2. 导入文档
    documents = [
        "docs/technical_manual.pdf",
        "docs/user_guide.pdf",
        "docs/faq.txt"
    ]
    
    rag_system.ingest_documents(documents)
    
    # 3. 查询
    questions = [
        "如何配置系统参数?",
        "常见错误代码有哪些?",
        "如何进行性能优化?"
    ]
    
    for question in questions:
        result = rag_system.query(question)
        
        print(f"\n{'='*60}")
        print(f"问题: {question}")
        print(f"{'='*60}")
        print(f"\n答案:\n{result['answer']}")
        print(f"\n来源:")
        for source in result['sources']:
            print(f"  - {source['source']} (页码: {source['page']})")

if __name__ == "__main__":
    demo_production_rag()
```

## 架构2:基于LlamaIndex的RAG系统

### 2.1 系统架构

LlamaIndex提供更高级的抽象和优化:

- **查询引擎**: 优化的检索策略

**来源**: `reference/search_reddit.md:35-46`

### 2.2 完整实现

```python
"""
生产级RAG系统 - LlamaIndex集成
使用LlamaIndex的高级抽象
"""

from llama_index.core import (
    VectorStoreIndex,
    SimpleDirectoryReader,
    StorageContext,
    Settings
)
from llama_index.vector_stores.milvus import MilvusVectorStore
from llama_index.embeddings.openai import OpenAIEmbedding
from llama_index.llms.openai import OpenAI

class LlamaIndexRAGSystem:
    """基于LlamaIndex的RAG系统"""
    
    def __init__(self, config: RAGConfig):
        self.config = config
        
        # 配置全局设置
        Settings.llm = OpenAI(
            model=config.llm_model,
            temperature=config.temperature
        )
        
        Settings.embed_model = OpenAIEmbedding(
            model="text-embedding-3-large"
        )
        
        Settings.chunk_size = config.chunk_size
        Settings.chunk_overlap = config.chunk_overlap
        
        # 初始化Milvus向量存储
        self.vector_store = MilvusVectorStore(
            uri=config.milvus_uri,
            collection_name=config.collection_name,
            dim=config.embedding_dim,
            overwrite=True
        )
        
        self.storage_context = StorageContext.from_defaults(
            vector_store=self.vector_store
        )
        
        self.index = None
    
    def ingest_documents(self, directory: str):
        """从目录导入文档"""
        print(f"\n📥 从目录导入文档: {directory}")
        
        # 1. 加载文档
        documents = SimpleDirectoryReader(directory).load_data()
        print(f"  加载了 {len(documents)} 个文档")
        
        # 2. 创建索引(自动处理分块和embedding)
        self.index = VectorStoreIndex.from_documents(
            documents,
            storage_context=self.storage_context,
            show_progress=True
        )
        
        print(f"✅ 文档导入完成")
    
    def query(self, question: str) -> Dict:
        """查询"""
        if not self.index:
            raise ValueError("请先导入文档")
        
        print(f"\n🔍 查询: {question}")
        
        # 创建查询引擎
        query_engine = self.index.as_query_engine(
            similarity_top_k=self.config.top_k
        )
        
        # 执行查询
        response = query_engine.query(question)
        
        # 提取来源
        sources = []
        for node in response.source_nodes:
            sources.append({
                "source": node.metadata.get("file_name", "unknown"),
                "score": node.score
            })
        
        return {
            "answer": str(response),
            "sources": sources
        }

# 使用示例
def demo_llamaindex_rag():
    """演示LlamaIndex RAG系统"""
    config = RAGConfig(
        collection_name="llamaindex_rag",
        embedding_dim=3072,  # text-embedding-3-large
        chunk_size=1024,
        chunk_overlap=200,
        top_k=5
    )
    
    rag_system = LlamaIndexRAGSystem(config)
    
    # 导入文档
    rag_system.ingest_documents("docs/")
    
    # 查询
    result = rag_system.query("如何配置系统参数?")
    
    print(f"\n答案:\n{result['answer']}")
    print(f"\n来源:")
    for source in result['sources']:
        print(f"  - {source['source']} (相似度: {source['score']:.4f})")

if __name__ == "__main__":
    demo_llamaindex_rag()
```

## 架构3:混合检索RAG系统

### 3.1 设计思路

结合稠密向量和稀疏向量(BM25)实现混合检索:
- **稠密向量**: 语义相似度
- **稀疏向量**: 关键词匹配
- **重排序**: Cohere Rerank优化结果

**来源**: `reference/search_github.md:61-72`

### 3.2 完整实现

```python
"""
混合检索RAG系统
结合稠密向量、稀疏向量和重排序
"""

from typing import List, Dict
import numpy as np

class HybridRAGSystem:
    """混合检索RAG系统"""
    
    def __init__(self, config: RAGConfig):
        self.config = config
        self.client = MilvusClient(uri=config.milvus_uri)
        self.llm = ChatOpenAI(
            model=config.llm_model,
            temperature=config.temperature
        )
        self._init_hybrid_collection()
    
    def _init_hybrid_collection(self):
        """初始化支持混合检索的Collection"""
        # 稠密向量Embedding Function
        dense_ef = Function(
            name="dense_ef",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["text"],
            output_field_names=["dense_vector"],
            params={
                "provider": "voyageai",
                "model_name": "voyage-3-large",
                "dim": 1024,
                "api_key": os.getenv("VOYAGEAI_API_KEY")
            }
        )
        
        # 定义Schema(包含稠密和稀疏向量)
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=65535),
            FieldSchema(name="source", dtype=DataType.VARCHAR, max_length=512),
            FieldSchema(name="dense_vector", dtype=DataType.FLOAT_VECTOR, dim=1024),
            FieldSchema(name="sparse_vector", dtype=DataType.SPARSE_FLOAT_VECTOR)
        ]
        
        schema = CollectionSchema(
            fields=fields,
            functions=[dense_ef],
            description="Hybrid RAG with dense and sparse vectors"
        )
        
        collection_name = f"{self.config.collection_name}_hybrid"
        
        if self.client.has_collection(collection_name):
            self.client.drop_collection(collection_name)
        
        self.client.create_collection(collection_name, schema=schema)
        self.collection_name = collection_name
        
        print(f"✅ 混合检索Collection创建成功: {collection_name}")
    
    def _compute_bm25_vector(self, text: str) -> Dict:
        """计算BM25稀疏向量"""
        # 简化的BM25实现(实际应使用专业库)
        words = text.lower().split()
        word_freq = {}
        
        for word in words:
            word_freq[word] = word_freq.get(word, 0) + 1
        
        # 转换为稀疏向量格式
        indices = []
        values = []
        
        for word, freq in word_freq.items():
            # 使用简单的哈希作为索引
            idx = hash(word) % 10000
            indices.append(idx)
            values.append(float(freq))
        
        return {
            "indices": indices,
            "values": values
        }
    
    def ingest_documents(self, file_paths: List[str]):
        """导入文档(包含稀疏向量)"""
        print(f"\n📥 开始导入文档: {len(file_paths)}个文件")
        
        all_chunks = []
        
        for file_path in file_paths:
            # 加载和分块(同前面的实现)
            # ...
            
            # 为每个chunk计算稀疏向量
            for chunk in chunks:
                sparse_vector = self._compute_bm25_vector(chunk.page_content)
                
                all_chunks.append({
                    "text": chunk.page_content,
                    "source": file_path,
                    "sparse_vector": sparse_vector
                })
        
        # 批量插入
        batch_size = 64
        for i in range(0, len(all_chunks), batch_size):
            batch = all_chunks[i:i + batch_size]
            self.client.insert(self.collection_name, batch)
        
        print(f"✅ 文档导入完成")
    
    def hybrid_retrieve(self, query: str, alpha: float = 0.5) -> List[Dict]:
        """混合检索
        
        Args:
            query: 查询文本
            alpha: 稠密向量权重(0-1), 1-alpha为稀疏向量权重
        """
        # 1. 稠密向量检索
        dense_results = self.client.search(
            collection_name=self.collection_name,
            data=[query],
            limit=self.config.top_k * 2,
            output_fields=["text", "source"]
        )
        
        # 2. 稀疏向量检索
        query_sparse = self._compute_bm25_vector(query)
        sparse_results = self.client.search(
            collection_name=self.collection_name,
            data=[query_sparse],
            anns_field="sparse_vector",
            limit=self.config.top_k * 2,
            output_fields=["text", "source"]
        )
        
        # 3. 融合结果(RRF - Reciprocal Rank Fusion)
        combined_scores = {}
        
        for rank, hits in enumerate(dense_results[0]):
            doc_id = hits["id"]
            combined_scores[doc_id] = alpha / (rank + 60)
        
        for rank, hits in enumerate(sparse_results[0]):
            doc_id = hits["id"]
            combined_scores[doc_id] = combined_scores.get(doc_id, 0) + (1 - alpha) / (rank + 60)
        
        # 4. 排序并返回top-k
        sorted_docs = sorted(
            combined_scores.items(),
            key=lambda x: x[1],
            reverse=True
        )[:self.config.top_k]
        
        # 5. 获取文档内容
        retrieved_docs = []
        for doc_id, score in sorted_docs:
            # 从原始结果中查找文档
            for hits in dense_results[0]:
                if hits["id"] == doc_id:
                    retrieved_docs.append({
                        "text": hits["entity"]["text"],
                        "source": hits["entity"]["source"],
                        "score": score
                    })
                    break
        
        return retrieved_docs
    
    def query(self, question: str) -> Dict:
        """混合检索查询"""
        print(f"\n🔍 混合检索查询: {question}")
        
        # 1. 混合检索
        retrieved_docs = self.hybrid_retrieve(question, alpha=0.7)
        print(f"  找到 {len(retrieved_docs)} 个相关文档")
        
        # 2. 生成答案
        result = self.generate_answer(question, retrieved_docs)
        
        return result

# 使用示例
def demo_hybrid_rag():
    """演示混合检索RAG"""
    config = RAGConfig(
        collection_name="hybrid_rag",
        top_k=5
    )
    
    rag_system = HybridRAGSystem(config)
    
    # 导入文档
    rag_system.ingest_documents(["docs/manual.pdf"])
    
    # 查询
    result = rag_system.query("如何配置系统参数?")
    
    print(f"\n答案:\n{result['answer']}")

if __name__ == "__main__":
    demo_hybrid_rag()
```

### 4.1 性能优化

**索引优化**:
```python

    "index_type": "HNSW",
    "metric_type": "COSINE",
    "params": {"M": 16, "efConstruction": 256}

client.create_index(
    collection_name=collection_name,
```

**批量处理**:
- 文档导入: 批量大小64-128
- 查询处理: 并发查询支持
- 缓存策略: Redis缓存热门查询

### 4.2 监控指标

**关键指标**:
- 查询延迟 (P50, P95, P99)
- 检索准确率 (Recall@K)
- 生成质量 (BLEU, ROUGE)
- 系统吞吐量 (QPS)
- 错误率

### 4.3 成本优化

**Embedding成本**:
- 使用缓存避免重复embedding
- 选择性价比高的Provider (VoyageAI, Cohere)
- 批量处理降低API调用次数

**LLM成本**:
- 使用较小的模型 (gpt-3.5-turbo)
- 优化Prompt长度
- 实现答案缓存
