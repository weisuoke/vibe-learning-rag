## 场景描述

在实际应用中，我们经常需要同时调用多个 LLM 完成不同的任务，然后合并结果。例如：

- **内容创作**: 同时生成笑话、故事、诗歌
- **多角度分析**: 从技术、商业、用户体验角度分析产品
- **多语言翻译**: 同时翻译成多种语言
- **A/B 测试**: 使用不同 prompt 生成内容并对比

本文档展示如何使用 LangGraph 的并行执行机制优雅地实现这些场景。

---

## 核心概念

### 1. 扇出模式（Fan-out）

从一个节点分支到多个并行节点：

```python
# 从 START 扇出到三个 LLM 节点
builder.add_edge(START, "call_llm_1")
builder.add_edge(START, "call_llm_2")
builder.add_edge(START, "call_llm_3")
```

### 2. 扇入模式（Fan-in）

多个并行节点合并到一个聚合器节点：

```python
# 三个 LLM 节点扇入到聚合器
builder.add_edge("call_llm_1", "aggregator")
builder.add_edge("call_llm_2", "aggregator")
builder.add_edge("call_llm_3", "aggregator")
```

### 3. 聚合器节点

等待所有并行任务完成后，合并结果：

```python
def aggregator(state: State):
    """合并所有 LLM 的输出"""
    combined = f"主题: {state['topic']}\n\n"
    combined += f"笑话:\n{state['joke']}\n\n"
    combined += f"故事:\n{state['story']}\n\n"
    combined += f"诗歌:\n{state['poem']}"
    return {"combined_output": combined}
```

## 完整实战代码

```python
"""
LangGraph 并行 LLM 调用实战
演示：同时生成笑话、故事、诗歌，然后合并结果

环境要求:
- Python 3.13+
- langgraph>=0.2.0
- langchain-openai>=0.2.0
- python-dotenv>=1.0.0

运行前配置:
1. 复制 .env.example 为 .env
2. 设置 OPENAI_API_KEY
"""

import os
from typing import TypedDict
from dotenv import load_dotenv
from langchain_openai import ChatOpenAI
from langgraph.graph import StateGraph, START, END

# 加载环境变量
load_dotenv()

# ===== 1. 定义状态 Schema =====
class State(TypedDict):
    """
    工作流状态定义

    字段说明:
    - topic: 用户输入的主题
    - joke: 笑话生成结果
    - story: 故事生成结果
    - poem: 诗歌生成结果
    - combined_output: 合并后的最终输出
    """
    topic: str
    joke: str
    story: str
    poem: str
    combined_output: str


# ===== 2. 初始化 LLM =====
llm = ChatOpenAI(
    model="gpt-4o-mini",
    temperature=0.7,  # 创意任务使用较高温度
    max_tokens=500
)

print("=" * 60)
print("LangGraph 并行 LLM 调用实战")
print("=" * 60)
print(f"✓ LLM 模型: {llm.model_name}")
print(f"✓ Temperature: {llm.temperature}")
print()


# ===== 3. 定义并行节点 =====

def call_llm_joke(state: State):
    """
    节点1: 生成笑话

    这个节点会并行执行，不会阻塞其他节点
    """
    print(f"[节点1] 正在生成笑话... (主题: {state['topic']})")

    prompt = f"请写一个关于 {state['topic']} 的幽默笑话，要简短有趣。"
    msg = llm.invoke(prompt)

    print(f"[节点1] ✓ 笑话生成完成")
    return {"joke": msg.content}


def call_llm_story(state: State):
    """
    节点2: 生成故事

    这个节点会并行执行，不会阻塞其他节点
    """
    print(f"[节点2] 正在生成故事... (主题: {state['topic']})")

    prompt = f"请写一个关于 {state['topic']} 的简短故事，要有情节和结局。"
    msg = llm.invoke(prompt)

    print(f"[节点2] ✓ 故事生成完成")
    return {"story": msg.content}


def call_llm_poem(state: State):
    """
    节点3: 生成诗歌

    这个节点会并行执行，不会阻塞其他节点
    """
    print(f"[节点3] 正在生成诗歌... (主题: {state['topic']})")

    prompt = f"请写一首关于 {state['topic']} 的诗歌，要有韵律和意境。"
    msg = llm.invoke(prompt)

    print(f"[节点3] ✓ 诗歌生成完成")
    return {"poem": msg.content}


# ===== 4. 定义聚合器节点 =====

def aggregator(state: State):
    """
    聚合器节点: 合并所有 LLM 的输出

    关键点:
    - 等待所有并行节点完成后才执行
    - 访问所有并行节点的结果
    - 生成最终的合并输出
    """
    print("\n[聚合器] 正在合并所有结果...")

    # 合并所有内容
    combined = f"""
{'=' * 60}
主题: {state['topic']}
{'=' * 60}

【笑话】
{state['joke']}

{'─' * 60}

【故事】
{state['story']}

{'─' * 60}

【诗歌】
{state['poem']}

{'=' * 60}
"""

    print("[聚合器] ✓ 结果合并完成")
    return {"combined_output": combined}


# ===== 5. 构建工作流图 =====

print("正在构建工作流图...")

# 创建状态图
builder = StateGraph(State)

# 添加节点
builder.add_node("call_llm_joke", call_llm_joke)
builder.add_node("call_llm_story", call_llm_story)
builder.add_node("call_llm_poem", call_llm_poem)
builder.add_node("aggregator", aggregator)

# 添加扇出边: START -> 三个并行节点
builder.add_edge(START, "call_llm_joke")
builder.add_edge(START, "call_llm_story")
builder.add_edge(START, "call_llm_poem")

# 添加扇入边: 三个并行节点 -> 聚合器
builder.add_edge("call_llm_joke", "aggregator")
builder.add_edge("call_llm_story", "aggregator")
builder.add_edge("call_llm_poem", "aggregator")

# 添加结束边
builder.add_edge("aggregator", END)

# 编译图
graph = builder.compile()

print("✓ 工作流图构建完成")
print()


# ===== 6. 执行工作流 =====

print("=" * 60)
print("开始执行并行 LLM 调用")
print("=" * 60)
print()

# 输入主题
topic = "猫"

print(f"输入主题: {topic}")
print()

# 调用工作流
result = graph.invoke({"topic": topic})

# 输出结果
print("\n" + "=" * 60)
print("最终输出")
print("=" * 60)
print(result["combined_output"])


# ===== 7. 进阶示例: 多角度分析 =====

print("\n\n" + "=" * 60)
print("进阶示例: 多角度产品分析")
print("=" * 60)
print()


class AnalysisState(TypedDict):
    """产品分析状态"""
    product: str
    technical_analysis: str
    business_analysis: str
    ux_analysis: str
    final_report: str


def analyze_technical(state: AnalysisState):
    """技术角度分析"""
    print(f"[技术分析] 正在分析 {state['product']}...")

    prompt = f"""
    从技术角度分析产品: {state['product']}

    请分析:
    1. 技术架构
    2. 技术难点
    3. 技术优势

    简洁回答，3-5句话。
    """
    msg = llm.invoke(prompt)

    print("[技术分析] ✓ 完成")
    return {"technical_analysis": msg.content}


def analyze_business(state: AnalysisState):
    """商业角度分析"""
    print(f"[商业分析] 正在分析 {state['product']}...")

    prompt = f"""
    从商业角度分析产品: {state['product']}

    请分析:
    1. 市场定位
    2. 盈利模式
    3. 竞争优势

    简洁回答，3-5句话。
    """
    msg = llm.invoke(prompt)

    print("[商业分析] ✓ 完成")
    return {"business_analysis": msg.content}


def analyze_ux(state: AnalysisState):
    """用户体验角度分析"""
    print(f"[UX分析] 正在分析 {state['product']}...")

    prompt = f"""
    从用户体验角度分析产品: {state['product']}

    请分析:
    1. 用户痛点
    2. 交互设计
    3. 体验优化

    简洁回答，3-5句话。
    """
    msg = llm.invoke(prompt)

    print("[UX分析] ✓ 完成")
    return {"ux_analysis": msg.content}


def generate_report(state: AnalysisState):
    """生成综合报告"""
    print("\n[报告生成] 正在生成综合报告...")

    report = f"""
{'=' * 60}
产品分析报告: {state['product']}
{'=' * 60}

【技术分析】
{state['technical_analysis']}

{'─' * 60}

【商业分析】
{state['business_analysis']}

{'─' * 60}

【用户体验分析】
{state['ux_analysis']}

{'=' * 60}
"""

    print("[报告生成] ✓ 完成")
    return {"final_report": report}


# 构建分析工作流
analysis_builder = StateGraph(AnalysisState)

analysis_builder.add_node("analyze_technical", analyze_technical)
analysis_builder.add_node("analyze_business", analyze_business)
analysis_builder.add_node("analyze_ux", analyze_ux)
analysis_builder.add_node("generate_report", generate_report)

# 扇出
analysis_builder.add_edge(START, "analyze_technical")
analysis_builder.add_edge(START, "analyze_business")
analysis_builder.add_edge(START, "analyze_ux")

# 扇入
analysis_builder.add_edge("analyze_technical", "generate_report")
analysis_builder.add_edge("analyze_business", "generate_report")
analysis_builder.add_edge("analyze_ux", "generate_report")

analysis_builder.add_edge("generate_report", END)

analysis_graph = analysis_builder.compile()

print("✓ 分析工作流构建完成")
print()

# 执行分析
product = "ChatGPT"
print(f"分析产品: {product}")
print()

analysis_result = analysis_graph.invoke({"product": product})

print("\n" + "=" * 60)
print("分析报告")
print("=" * 60)
print(analysis_result["final_report"])
```

## 关键技术点

### 1. 并行执行机制

LangGraph 基于 **Bulk Synchronous Parallel (BSP)** 模型：

- **超步（Superstep）**: 所有从 START 出发的节点在同一个超步中并行执行
- **同步点**: 聚合器节点等待所有并行节点完成后才执行
- **自动调度**: LangGraph 自动管理并行执行和同步

### 2. 状态更新策略

每个并行节点返回部分状态更新：

```python
# 节点1只更新 joke 字段
return {"joke": msg.content}

# 节点2只更新 story 字段
return {"story": msg.content}

# 聚合器可以访问所有字段
state['joke']  # ✓ 可用
state['story']  # ✓ 可用
state['poem']  # ✓ 可用
```

### 3. 错误处理

在生产环境中，应该添加错误处理：

```python
def call_llm_with_retry(state: State, task: str):
    """带重试的 LLM 调用"""
    max_retries = 3

    for attempt in range(max_retries):
        try:
            msg = llm.invoke(f"生成{task}: {state['topic']}")
            return {task: msg.content}
        except Exception as e:
            if attempt == max_retries - 1:
                return {task: f"生成失败: {str(e)}"}
            print(f"重试 {attempt + 1}/{max_retries}...")
```

## 性能优化建议

### 1. 控制并行度

过多的并行任务会增加资源消耗：

```python
# ❌ 不推荐: 10个并行 LLM 调用
for i in range(10):
    builder.add_edge(START, f"llm_{i}")

# ✅ 推荐: 3-5个并行任务
builder.add_edge(START, "llm_1")
builder.add_edge(START, "llm_2")
builder.add_edge(START, "llm_3")
```

### 2. 使用流式输出

对于长时间运行的任务，使用流式输出提升用户体验：

```python
for chunk in graph.stream({"topic": "猫"}):
    print(chunk)
```

### 3. 缓存结果

对于相同输入，缓存 LLM 结果：

```python
from functools import lru_cache

@lru_cache(maxsize=100)
def cached_llm_call(prompt: str):
    return llm.invoke(prompt).content
```
