> 使用 BaseTool 继承实现有状态工具，展示状态管理、初始化和资源清理的完整实践

---

## 场景说明

**目标**：创建一个会话管理工具，能够在多次调用之间保持状态

**适用场景**：
- 需要维护连接池（数据库、Redis、API 客户端）
- 需要跨调用共享状态（用户会话、缓存）
- 需要复杂的初始化和清理逻辑
- 需要访问工具实例的属性

**为什么使用 BaseTool**：
- @tool 装饰器是无状态的（每次调用都是新的函数执行）
- StructuredTool 虽然可以传递外部对象，但不够灵活
- BaseTool 继承提供完整的面向对象能力

## 完整代码实现

```python
"""
有状态会话工具 - BaseTool 继承实战
演示：用户会话管理、状态持久化、资源清理
"""

from langchain.tools import BaseTool
from langchain_core.runnables import RunnableConfig
from pydantic import BaseModel, Field
from typing import Any, Optional, Literal
from datetime import datetime
import json


# ===== 1. 定义输入 Schema =====
class SessionInput(BaseModel):
    """会话操作的输入参数"""
    action: Literal["get", "set", "delete", "list"] = Field(
        description="操作类型：get(获取), set(设置), delete(删除), list(列出所有)"
    )
    key: str = Field(
        default="",
        description="会话键名（list 操作时可为空）"
    )
    value: str = Field(
        default="",
        description="会话值（仅 set 操作需要）"
    )


# ===== 2. 继承 BaseTool 实现有状态工具 =====
class SessionTool(BaseTool):
    """
    有状态的会话管理工具

    特点：
    - 维护内部状态（session_store）
    - 支持用户隔离（user_id）
    - 提供完整的 CRUD 操作
    - 记录操作历史
    """

    # ===== 工具元数据（必需） =====
    name: str = "session_manager"
    description: str = """管理用户会话数据。

    支持操作：
    - get: 获取会话值
    - set: 设置会话值
    - delete: 删除会话键
    - list: 列出所有会话键
    """
    args_schema: type[BaseModel] = SessionInput

    # ===== 工具状态（自定义属性） =====
    session_store: dict[str, Any] = {}  # 会话存储
    user_id: str = ""  # 用户 ID
    operation_history: list[dict] = []  # 操作历史
    max_history_size: int = 100  # 历史记录上限

    def __init__(self, user_id: str, **kwargs):
        """
        初始化会话工具

        Args:
            user_id: 用户唯一标识
            **kwargs: 传递给 BaseTool 的其他参数
        """
        super().__init__(**kwargs)
        self.user_id = user_id
        self.session_store = {}
        self.operation_history = []
        print(f"[SessionTool] 初始化完成，用户: {user_id}")

    def _run(
        self,
        action: str,
        key: str = "",
        value: str = "",
        run_manager: Optional[Any] = None
    ) -> str:
        """
        同步执行工具逻辑（必须实现）

        Args:
            action: 操作类型
            key: 会话键
            value: 会话值
            run_manager: 运行管理器（可选）

        Returns:
            操作结果的字符串描述
        """
        # 记录操作
        self._log_operation(action, key, value)

        # 构造完整的会话键（用户隔离）
        session_key = f"{self.user_id}:{key}" if key else ""

        # 执行操作
        if action == "get":
            return self._handle_get(session_key, key)
        elif action == "set":
            return self._handle_set(session_key, key, value)
        elif action == "delete":
            return self._handle_delete(session_key, key)
        elif action == "list":
            return self._handle_list()
        else:
            raise ValueError(f"未知操作: {action}")

    async def _arun(
        self,
        action: str,
        key: str = "",
        value: str = "",
        run_manager: Optional[Any] = None
    ) -> str:
        """
        异步执行工具逻辑（可选实现）

        对于简单的内存操作，直接调用同步版本即可
        如果涉及 I/O（如数据库、Redis），应实现真正的异步逻辑
        """
        # 简单场景：直接调用同步版本
        return self._run(action, key, value, run_manager)

    # ===== 私有辅助方法 =====

    def _handle_get(self, session_key: str, key: str) -> str:
        """处理 get 操作"""
        if session_key in self.session_store:
            value = self.session_store[session_key]
            return f"✓ 获取成功: {key} = {value}"
        else:
            return f"✗ 键不存在: {key}"

    def _handle_set(self, session_key: str, key: str, value: str) -> str:
        """处理 set 操作"""
        self.session_store[session_key] = value
        return f"✓ 设置成功: {key} = {value}"

    def _handle_delete(self, session_key: str, key: str) -> str:
        """处理 delete 操作"""
        if session_key in self.session_store:
            del self.session_store[session_key]
            return f"✓ 删除成功: {key}"
        else:
            return f"✗ 键不存在: {key}"

    def _handle_list(self) -> str:
        """处理 list 操作"""
        # 过滤出当前用户的所有键
        user_prefix = f"{self.user_id}:"
        user_keys = [
            k.replace(user_prefix, "")
            for k in self.session_store.keys()
            if k.startswith(user_prefix)
        ]

        if user_keys:
            keys_str = ", ".join(user_keys)
            return f"✓ 会话键列表 ({len(user_keys)}): {keys_str}"
        else:
            return "✓ 当前无会话数据"

    def _log_operation(self, action: str, key: str, value: str):
        """记录操作历史"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "action": action,
            "key": key,
            "value": value if action == "set" else ""
        }
        self.operation_history.append(log_entry)

        # 限制历史记录大小
        if len(self.operation_history) > self.max_history_size:
            self.operation_history = self.operation_history[-self.max_history_size:]

    # ===== 公共方法（工具特有功能） =====

    def get_history(self, limit: int = 10) -> list[dict]:
        """获取操作历史"""
        return self.operation_history[-limit:]

    def clear_session(self):
        """清空当前用户的所有会话数据"""
        user_prefix = f"{self.user_id}:"
        keys_to_delete = [k for k in self.session_store.keys() if k.startswith(user_prefix)]
        for key in keys_to_delete:
            del self.session_store[key]
        print(f"[SessionTool] 已清空用户 {self.user_id} 的会话数据")

    def export_session(self) -> str:
        """导出当前用户的会话数据为 JSON"""
        user_prefix = f"{self.user_id}:"
        user_data = {
            k.replace(user_prefix, ""): v
            for k, v in self.session_store.items()
            if k.startswith(user_prefix)
        }
        return json.dumps(user_data, indent=2, ensure_ascii=False)
```

## 关键技术点解析

### 1. BaseTool 继承结构

```python
class SessionTool(BaseTool):
    # 必需属性
    name: str = "session_manager"
    description: str = "..."
    args_schema: type[BaseModel] = SessionInput

    # 自定义状态属性
    session_store: dict[str, Any] = {}
    user_id: str = ""

    # 必需方法
    def _run(self, ...): ...

    # 可选方法
    async def _arun(self, ...): ...
```

**要点**：
- `name`、`description`、`args_schema` 是必需的
- 自定义属性用于维护状态
- `_run` 是必须实现的同步方法
- `_arun` 是可选的异步方法

### 2. 初始化和状态管理

```python
def __init__(self, user_id: str, **kwargs):
    super().__init__(**kwargs)  # 必须调用父类初始化
    self.user_id = user_id
    self.session_store = {}
    self.operation_history = []
```

**要点**：
- 必须调用 `super().__init__(**kwargs)`
- 在 `__init__` 中初始化所有状态
- 可以接受自定义参数（如 `user_id`）

### 3. 用户隔离策略

```python
# 构造带用户前缀的键
session_key = f"{self.user_id}:{key}"

# 过滤当前用户的键
user_prefix = f"{self.user_id}:"
user_keys = [k for k in self.session_store.keys() if k.startswith(user_prefix)]
```

**要点**：
- 使用键前缀实现多用户隔离
- 每个用户的工具实例独立
- 共享底层存储但逻辑隔离

### 4. 操作历史记录

```python
def _log_operation(self, action: str, key: str, value: str):
    log_entry = {
        "timestamp": datetime.now().isoformat(),
        "action": action,
        "key": key,
        "value": value if action == "set" else ""
    }
    self.operation_history.append(log_entry)

    # 限制历史记录大小
    if len(self.operation_history) > self.max_history_size:
        self.operation_history = self.operation_history[-self.max_history_size:]
```

**要点**：
- 记录每次操作的时间戳和参数
- 限制历史记录大小防止内存溢出
- 可用于审计和调试

### 5. 资源清理

```python
def clear_session(self):
    """清空当前用户的所有会话数据"""
    user_prefix = f"{self.user_id}:"
    keys_to_delete = [k for k in self.session_store.keys() if k.startswith(user_prefix)]
    for key in keys_to_delete:
        del self.session_store[key]
```

**要点**：
- 提供显式的清理方法
- 对于数据库连接等资源，应在 `__del__` 或上下文管理器中清理
- 避免资源泄漏
