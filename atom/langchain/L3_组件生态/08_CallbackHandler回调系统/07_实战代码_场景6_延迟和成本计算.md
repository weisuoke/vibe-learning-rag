# 实战代码 - 场景6：延迟和成本计算

## 场景说明

**应用场景**：生产环境的端到端性能监控，同时追踪延迟和成本，适用于需要精确控制预算和优化响应时间的 RAG 系统。

**核心价值**：
- 端到端延迟监控，识别性能瓶颈
- 实时成本追踪，避免预算超支
- 性能数据分析，辅助系统优化
- 预算管理，支持成本控制

**RAG 开发价值**：
- RAG 系统涉及多个环节（检索 + 生成），需要全链路监控
- 延迟和成本是生产环境的关键指标
- 帮助识别性能瓶颈和成本热点
- 支持 SLA 管理和预算控制

## 完整可运行代码

### 方案1：端到端性能监控

```python
"""
端到端性能监控回调
演示：同时追踪延迟和成本的完整监控系统
适用场景：生产环境的性能和成本监控
"""

import os
import time
from typing import Any, Dict, List, Optional
from dotenv import load_dotenv

from langchain_core.callbacks.base import BaseCallbackHandler
from langchain_core.outputs import LLMResult
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage

load_dotenv()


# ===== 1. 端到端性能监控回调 =====
class EndToEndMonitorCallback(BaseCallbackHandler):
    """
    端到端性能监控回调处理器

    追踪：
    - 延迟（端到端、LLM 调用、检索等）
    - Token 用量和成本
    - 调用次数和成功率
    """

    def __init__(
        self,
        prompt_cost_per_1m: float = 0.15,      # GPT-4o-mini: $0.15/1M input
        completion_cost_per_1m: float = 0.60   # GPT-4o-mini: $0.60/1M output
    ):
        super().__init__()
        # 成本配置
        self.prompt_cost_per_1m = prompt_cost_per_1m
        self.completion_cost_per_1m = completion_cost_per_1m

        # Token 统计
        self.total_prompt_tokens = 0
        self.total_completion_tokens = 0

        # 延迟统计
        self.start_time: Optional[float] = None
        self.llm_start_time: Optional[float] = None
        self.total_latency = 0.0
        self.llm_latency = 0.0

        # 调用统计
        self.call_count = 0
        self.success_count = 0
        self.error_count = 0

        # 详细记录
        self.call_details: List[Dict[str, Any]] = []

    def on_chain_start(self, serialized: Dict[str, Any], inputs: Dict[str, Any], **kwargs: Any) -> None:
        """Chain 开始时记录时间"""
        self.start_time = time.time()

    def on_llm_start(self, serialized: Dict[str, Any], prompts: List[str], **kwargs: Any) -> None:
        """LLM 开始时记录时间"""
        self.llm_start_time = time.time()

    def on_llm_end(self, response: LLMResult, **kwargs: Any) -> None:
        """LLM 结束时计算延迟和成本"""
        # 计算 LLM 延迟
        llm_latency = time.time() - self.llm_start_time if self.llm_start_time else 0
        self.llm_latency += llm_latency

        # 提取 token 用量
        prompt_tokens = 0
        completion_tokens = 0

        if response.llm_output and 'token_usage' in response.llm_output:
            usage = response.llm_output['token_usage']
            prompt_tokens = usage.get('prompt_tokens', 0)
            completion_tokens = usage.get('completion_tokens', 0)

            self.total_prompt_tokens += prompt_tokens
            self.total_completion_tokens += completion_tokens

        # 计算成本
        call_cost = (
            (prompt_tokens / 1_000_000) * self.prompt_cost_per_1m +
            (completion_tokens / 1_000_000) * self.completion_cost_per_1m
        )

        # 记录详细信息
        self.call_count += 1
        self.success_count += 1

        detail = {
            "call_number": self.call_count,
            "llm_latency_seconds": round(llm_latency, 3),
            "prompt_tokens": prompt_tokens,
            "completion_tokens": completion_tokens,
            "total_tokens": prompt_tokens + completion_tokens,
            "cost_usd": round(call_cost, 6),
            "status": "success"
        }
        self.call_details.append(detail)

        print(f"[LLM 调用 #{self.call_count}] "
              f"延迟: {llm_latency:.2f}s | "
              f"Tokens: {prompt_tokens + completion_tokens} | "
              f"成本: ${call_cost:.6f}")

    def on_llm_error(self, error: Exception, **kwargs: Any) -> None:
        """LLM 错误时记录"""
        self.call_count += 1
        self.error_count += 1

        detail = {
            "call_number": self.call_count,
            "status": "error",
            "error": str(error)
        }
        self.call_details.append(detail)

    def on_chain_end(self, outputs: Dict[str, Any], **kwargs: Any) -> None:
        """Chain 结束时计算总延迟"""
        if self.start_time:
            self.total_latency = time.time() - self.start_time

    def get_summary(self) -> Dict[str, Any]:
        """获取完整监控摘要"""
        total_cost = (
            (self.total_prompt_tokens / 1_000_000) * self.prompt_cost_per_1m +
            (self.total_completion_tokens / 1_000_000) * self.completion_cost_per_1m
        )

        avg_llm_latency = self.llm_latency / self.success_count if self.success_count > 0 else 0
        success_rate = (self.success_count / self.call_count * 100) if self.call_count > 0 else 0

        return {
            "performance": {
                "total_latency_seconds": round(self.total_latency, 3),
                "llm_latency_seconds": round(self.llm_latency, 3),
                "avg_llm_latency_seconds": round(avg_llm_latency, 3),
                "non_llm_latency_seconds": round(self.total_latency - self.llm_latency, 3)
            },
            "cost": {
                "total_tokens": self.total_prompt_tokens + self.total_completion_tokens,
                "prompt_tokens": self.total_prompt_tokens,
                "completion_tokens": self.total_completion_tokens,
                "total_cost_usd": round(total_cost, 6)
            },
            "reliability": {
                "total_calls": self.call_count,
                "success_count": self.success_count,
                "error_count": self.error_count,
                "success_rate_percent": round(success_rate, 2)
            },
            "details": self.call_details
        }


# ===== 2. 测试端到端监控 =====
print("=== 端到端性能监控测试 ===\n")

callback = EndToEndMonitorCallback()

llm = ChatOpenAI(
    model="gpt-4o-mini",
    temperature=0.7,
    callbacks=[callback]
)

# 模拟 RAG 场景：多次调用
questions = [
    "什么是向量数据库？",
    "如何选择合适的 Embedding 模型？"
]

for i, question in enumerate(questions, 1):
    print(f"\n问题 {i}: {question}")
    response = llm.invoke([HumanMessage(content=question)])
    print(f"回答: {response.content[:60]}...")

# 输出完整监控报告
print("\n" + "="*60)
print("=== 完整监控报告 ===")
summary = callback.get_summary()

print("\n【性能指标】")
for key, value in summary["performance"].items():
    print(f"  {key}: {value}")

print("\n【成本指标】")
for key, value in summary["cost"].items():
    print(f"  {key}: {value}")

print("\n【可靠性指标】")
for key, value in summary["reliability"].items():
    print(f"  {key}: {value}")
```

### 方案2：生产级监控系统（带预算控制）

```python
"""
生产级监控系统
演示：带预算控制和告警的监控系统
适用场景：生产环境的成本控制和性能优化
"""

import time
from typing import Any, Dict, List, Optional
from langchain_core.callbacks.base import BaseCallbackHandler
from langchain_core.outputs import LLMResult
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage


class ProductionMonitorCallback(BaseCallbackHandler):
    """
    生产级监控回调处理器

    功能：
    - 预算控制（超预算时抛出异常）
    - 性能告警（延迟超阈值时告警）
    - 详细监控数据
    """

    def __init__(
        self,
        max_cost_usd: float = 1.0,           # 最大成本（美元）
        max_latency_seconds: float = 5.0,   # 最大延迟（秒）
        prompt_cost_per_1m: float = 0.15,
        completion_cost_per_1m: float = 0.60
    ):
        super().__init__()
        # 预算配置
        self.max_cost = max_cost_usd
        self.max_latency = max_latency_seconds
        self.prompt_cost_per_1m = prompt_cost_per_1m
        self.completion_cost_per_1m = completion_cost_per_1m

        # 统计
        self.current_cost = 0.0
        self.total_prompt_tokens = 0
        self.total_completion_tokens = 0
        self.start_time: Optional[float] = None
        self.call_count = 0

        # 告警记录
        self.alerts: List[Dict[str, Any]] = []

    def on_chain_start(self, serialized: Dict[str, Any], inputs: Dict[str, Any], **kwargs: Any) -> None:
        """Chain 开始时记录时间"""
        self.start_time = time.time()

    def on_llm_end(self, response: LLMResult, **kwargs: Any) -> None:
        """LLM 结束时检查预算和性能"""
        self.call_count += 1

        # 提取 token 用量
        if response.llm_output and 'token_usage' in response.llm_output:
            usage = response.llm_output['token_usage']
            prompt_tokens = usage.get('prompt_tokens', 0)
            completion_tokens = usage.get('completion_tokens', 0)

            self.total_prompt_tokens += prompt_tokens
            self.total_completion_tokens += completion_tokens

            # 计算成本
            call_cost = (
                (prompt_tokens / 1_000_000) * self.prompt_cost_per_1m +
                (completion_tokens / 1_000_000) * self.completion_cost_per_1m
            )
            self.current_cost += call_cost

            # 检查预算
            if self.current_cost > self.max_cost:
                alert = {
                    "type": "budget_exceeded",
                    "call_number": self.call_count,
                    "current_cost": round(self.current_cost, 6),
                    "max_cost": self.max_cost
                }
                self.alerts.append(alert)
                raise Exception(
                    f"预算超支！当前成本: ${self.current_cost:.6f}, "
                    f"预算上限: ${self.max_cost:.6f}"
                )

            print(f"[调用 #{self.call_count}] "
                  f"成本: ${call_cost:.6f} | "
                  f"累计: ${self.current_cost:.6f}/{self.max_cost:.6f}")

    def on_chain_end(self, outputs: Dict[str, Any], **kwargs: Any) -> None:
        """Chain 结束时检查延迟"""
        if self.start_time:
            latency = time.time() - self.start_time

            # 检查延迟
            if latency > self.max_latency:
                alert = {
                    "type": "latency_exceeded",
                    "call_number": self.call_count,
                    "latency_seconds": round(latency, 3),
                    "max_latency_seconds": self.max_latency
                }
                self.alerts.append(alert)
                print(f"⚠️  延迟告警：{latency:.2f}s 超过阈值 {self.max_latency:.2f}s")

    def get_summary(self) -> Dict[str, Any]:
        """获取监控摘要"""
        budget_usage_percent = (self.current_cost / self.max_cost * 100) if self.max_cost > 0 else 0

        return {
            "budget": {
                "current_cost_usd": round(self.current_cost, 6),
                "max_cost_usd": self.max_cost,
                "budget_usage_percent": round(budget_usage_percent, 2),
                "remaining_budget_usd": round(self.max_cost - self.current_cost, 6)
            },
            "usage": {
                "total_calls": self.call_count,
                "total_tokens": self.total_prompt_tokens + self.total_completion_tokens
            },
            "alerts": self.alerts
        }


# ===== 测试生产级监控 =====
print("=== 生产级监控系统测试 ===\n")

# 设置较低的预算用于测试
callback = ProductionMonitorCallback(
    max_cost_usd=0.01,  # 0.01 美元预算
    max_latency_seconds=3.0
)

llm = ChatOpenAI(
    model="gpt-4o-mini",
    temperature=0.7,
    callbacks=[callback]
)

# 测试预算控制
try:
    for i in range(5):
        print(f"\n测试 {i+1}:")
        response = llm.invoke([HumanMessage(content="简单回答：什么是 AI？")])
        print(f"回答: {response.content[:40]}...")
except Exception as e:
    print(f"\n❌ 异常: {e}")

# 输出监控摘要
print("\n" + "="*60)
print("=== 监控摘要 ===")
summary = callback.get_summary()

print("\n【预算状态】")
for key, value in summary["budget"].items():
    print(f"  {key}: {value}")

print("\n【使用情况】")
for key, value in summary["usage"].items():
    print(f"  {key}: {value}")

if summary["alerts"]:
    print("\n【告警记录】")
    for alert in summary["alerts"]:
        print(f"  {alert}")
```

## 关键技术点

### 1. 多层延迟追踪

```python
# Chain 级别延迟
def on_chain_start(self, ...):
    self.start_time = time.time()

def on_chain_end(self, ...):
    self.total_latency = time.time() - self.start_time

# LLM 级别延迟
def on_llm_start(self, ...):
    self.llm_start_time = time.time()

def on_llm_end(self, ...):
    llm_latency = time.time() - self.llm_start_time
```

### 2. 预算控制机制

```python
# 实时检查预算
if self.current_cost > self.max_cost:
    raise Exception(f"预算超支！")
```

### 3. 性能告警

```python
# 延迟告警
if latency > self.max_latency:
    self.alerts.append({
        "type": "latency_exceeded",
        "latency_seconds": latency
    })
```

## 与 RAG 开发的联系

### 1. RAG 系统的性能瓶颈

```
总延迟 = 检索延迟 + LLM 延迟 + 其他延迟

- 检索延迟：向量数据库查询
- LLM 延迟：生成响应
- 其他延迟：网络、序列化等
```

### 2. 成本优化策略

```python
# 策略1：缓存检索结果
# 减少重复检索的成本

# 策略2：使用更便宜的模型
# GPT-4o-mini 而非 GPT-4o

# 策略3：优化 Chunk 大小
# 减少检索的文档数量
```

### 3. 生产环境 SLA 管理

```python
# SLA 要求：
# - P95 延迟 < 2s
# - 成功率 > 99%
# - 成本 < $0.01/请求

# 通过监控回调实时追踪
```
