## 场景描述

在生产环境的 RAG 系统中，我们经常需要结合多种优化策略来提升检索性能：
- **分区过滤**：只搜索相关分区
- **标量过滤**：使用元数据过滤
- **向量检索**：语义相似度搜索
- **结果重排序**：优化结果质量

本场景演示如何结合 Partition Key、标量过滤和向量检索，实现高性能的混合检索系统。

## 完整代码示例

```python
"""
Milvus 混合分区检索优化实战
场景：企业文档知识库 - 高性能混合检索

功能：
1. Partition Key + 标量过滤
2. 性能对比测试
3. 查询优化策略
4. 生产环境最佳实践

依赖：
- pymilvus
- numpy
- time
"""

from pymilvus import (
    connections,
    Collection,
    CollectionSchema,
    FieldSchema,
    DataType,
    utility
)
import numpy as np
from typing import List, Dict, Any
import time
from datetime import datetime, timedelta


class HybridSearchOptimizer:
    """混合检索优化器"""

    def __init__(self, collection_name: str = "enterprise_docs"):
        self.collection_name = collection_name
        self.collection = None

    def connect_to_milvus(self, host: str = "localhost", port: str = "19530"):
        """连接到 Milvus"""
        print(f"连接到 Milvus: {host}:{port}")
        connections.connect(alias="default", host=host, port=port)
        print("连接成功")

    def create_collection(self):
        """创建 Collection（使用 Partition Key）"""
        if utility.has_collection(self.collection_name):
            print(f"Collection '{self.collection_name}' 已存在，删除后重新创建")
            utility.drop_collection(self.collection_name)

        # 定义 Schema
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(
                name="department",
                dtype=DataType.VARCHAR,
                max_length=64,
                is_partition_key=True  # Partition Key
            ),
            FieldSchema(name="doc_type", dtype=DataType.VARCHAR, max_length=32),
            FieldSchema(name="priority", dtype=DataType.INT64),
            FieldSchema(name="created_at", dtype=DataType.INT64),
            FieldSchema(name="title", dtype=DataType.VARCHAR, max_length=512),
            FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
            FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=128)
        ]

        schema = CollectionSchema(
            fields=fields,
            description="企业文档知识库 - 混合检索优化"
        )

        self.collection = Collection(name=self.collection_name, schema=schema)
        print(f"Collection '{self.collection_name}' 创建成功")

    def create_indexes(self):
        """创建索引（向量索引 + 标量索引）"""
        print("\n创建索引...")

        # 向量索引
        vector_index_params = {
            "index_type": "IVF_FLAT",
            "metric_type": "L2",
            "params": {"nlist": 128}
        }
        self.collection.create_index(
            field_name="embedding",
            index_params=vector_index_params
        )
        print("向量索引创建成功")

        # 标量索引（加速过滤）
        self.collection.create_index(field_name="doc_type")
        self.collection.create_index(field_name="priority")
        self.collection.create_index(field_name="created_at")
        print("标量索引创建成功")

    def load_collection(self):
        """加载 Collection"""
        print("\n加载 Collection...")
        self.collection.load()
        time.sleep(2)
        print("Collection 已加载")

    def insert_sample_data(self, num_departments: int = 3, docs_per_dept: int = 100):
        """插入测试数据"""
        print(f"\n插入测试数据: {num_departments} 个部门, 每个部门 {docs_per_dept} 个文档")

        departments = [f"dept_{i}" for i in range(num_departments)]
        doc_types = ["技术文档", "产品文档", "营销文档", "财务文档"]
        priorities = [1, 2, 3]

        for dept in departments:
            data = []
            for i in range(docs_per_dept):
                doc_type = doc_types[i % len(doc_types)]
                priority = priorities[i % len(priorities)]
                created_at = int((datetime.now() - timedelta(days=i)).timestamp())

                data.append({
                    "department": dept,
                    "doc_type": doc_type,
                    "priority": priority,
                    "created_at": created_at,
                    "title": f"{dept}_{doc_type}_{i}",
                    "content": f"这是 {dept} 的 {doc_type} 内容 {i}",
                    "embedding": np.random.rand(128).tolist()
                })

            # 批量插入
            batch_data = [
                [d["department"] for d in data],
                [d["doc_type"] for d in data],
                [d["priority"] for d in data],
                [d["created_at"] for d in data],
                [d["title"] for d in data],
                [d["content"] for d in data],
                [d["embedding"] for d in data]
            ]

            result = self.collection.insert(batch_data)
            print(f"部门 '{dept}' 插入 {result.insert_count} 个文档")

    def search_baseline(self, query_embedding: List[float], top_k: int = 10):
        """基线搜索（无优化）"""
        print("\n=== 基线搜索（无优化）===")
        start_time = time.time()

        search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
        results = self.collection.search(
            data=[query_embedding],
            anns_field="embedding",
            param=search_params,
            limit=top_k,
            output_fields=["department", "doc_type", "priority"]
        )

        elapsed = time.time() - start_time
        print(f"搜索耗时: {elapsed*1000:.2f} ms")
        print(f"结果数量: {len(results[0])}")
        return results, elapsed

    def search_with_partition_key(
        self,
        query_embedding: List[float],
        department: str,
        top_k: int = 10
    ):
        """使用 Partition Key 过滤"""
        print(f"\n=== 使用 Partition Key 过滤（department={department}）===")
        start_time = time.time()

        search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
        results = self.collection.search(
            data=[query_embedding],
            anns_field="embedding",
            param=search_params,
            limit=top_k,
            expr=f'department == "{department}"',
            output_fields=["department", "doc_type", "priority"]
        )

        elapsed = time.time() - start_time
        print(f"搜索耗时: {elapsed*1000:.2f} ms")
        print(f"结果数量: {len(results[0])}")
        return results, elapsed

    def search_with_scalar_filter(
        self,
        query_embedding: List[float],
        department: str,
        doc_type: str,
        min_priority: int,
        top_k: int = 10
    ):
        """使用 Partition Key + 标量过滤"""
        print(f"\n=== 混合过滤（dept={department}, type={doc_type}, priority>={min_priority}）===")
        start_time = time.time()

        # 构建复合过滤表达式
        expr = f'department == "{department}" && doc_type == "{doc_type}" && priority >= {min_priority}'

        search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
        results = self.collection.search(
            data=[query_embedding],
            anns_field="embedding",
            param=search_params,
            limit=top_k,
            expr=expr,
            output_fields=["department", "doc_type", "priority", "title"]
        )

        elapsed = time.time() - start_time
        print(f"搜索耗时: {elapsed*1000:.2f} ms")
        print(f"结果数量: {len(results[0])}")
        return results, elapsed

    def search_with_time_range(
        self,
        query_embedding: List[float],
        department: str,
        days: int = 30,
        top_k: int = 10
    ):
        """使用 Partition Key + 时间范围过滤"""
        print(f"\n=== 时间范围过滤（dept={department}, 最近{days}天）===")
        start_time = time.time()

        # 计算时间戳
        cutoff_time = int((datetime.now() - timedelta(days=days)).timestamp())
        expr = f'department == "{department}" && created_at >= {cutoff_time}'

        search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
        results = self.collection.search(
            data=[query_embedding],
            anns_field="embedding",
            param=search_params,
            limit=top_k,
            expr=expr,
            output_fields=["department", "doc_type", "created_at"]
        )

        elapsed = time.time() - start_time
        print(f"搜索耗时: {elapsed*1000:.2f} ms")
        print(f"结果数量: {len(results[0])}")
        return results, elapsed

    def performance_comparison(self, query_embedding: List[float]):
        """性能对比测试"""
        print("\n" + "="*60)
        print("性能对比测试")
        print("="*60)

        # 1. 基线搜索
        _, time_baseline = self.search_baseline(query_embedding)

        # 2. Partition Key 过滤
        _, time_partition = self.search_with_partition_key(query_embedding, "dept_0")

        # 3. 混合过滤
        _, time_hybrid = self.search_with_scalar_filter(
            query_embedding, "dept_0", "技术文档", 2
        )

        # 4. 时间范围过滤
        _, time_range = self.search_with_time_range(query_embedding, "dept_0", 30)

        # 性能总结
        print("\n" + "="*60)
        print("性能总结")
        print("="*60)
        print(f"基线搜索:           {time_baseline*1000:.2f} ms (1.00x)")
        print(f"Partition Key 过滤: {time_partition*1000:.2f} ms ({time_baseline/time_partition:.2f}x)")
        print(f"混合过滤:           {time_hybrid*1000:.2f} ms ({time_baseline/time_hybrid:.2f}x)")
        print(f"时间范围过滤:       {time_range*1000:.2f} ms ({time_baseline/time_range:.2f}x)")


def main():
    """主函数 - 演示混合检索优化"""

    # 1. 初始化
    optimizer = HybridSearchOptimizer()
    optimizer.connect_to_milvus()
    optimizer.create_collection()

    # 2. 插入测试数据
    optimizer.insert_sample_data(num_departments=3, docs_per_dept=100)

    # 3. 创建索引
    optimizer.create_indexes()

    # 4. 加载 Collection
    optimizer.load_collection()

    # 5. 生成查询向量
    query_embedding = np.random.rand(128).tolist()

    # 6. 性能对比测试
    optimizer.performance_comparison(query_embedding)

    print("\n演示完成")


if __name__ == "__main__":
    main()
```

    expr=expr,
    ...
)

- 定期归档历史数据
- 配置合理的副本数

### 3. 常见问题

**Q: 如何选择过滤字段？**
- 选择高基数字段（如 user_id, department）
- 选择查询频繁的字段
- 避免低基数字段（如 status）

**Q: 过滤表达式过于复杂会影响性能吗？**
- 会，但影响有限
- 建议控制在 3-5 个条件以内
- 优先使用 Partition Key 过滤
