## 策略1:分类错误处理

### 1.1 错误分类

根据Milvus源码和社区实践,embedding函数错误可分为:

**1. 输入验证错误**:
- 空字符串输入
- 超过最大长度
- 无效字符编码

**2. API调用错误**:
- 认证失败 (401)
- 速率限制 (429)
- 服务不可用 (503)
- 网络超时

**3. 批量处理错误**:
- 超过MaxBatch限制
- 部分文本失败

**4. 资源错误**:
- 内存不足
- 磁盘空间不足

**来源**: `reference/source_architecture.md:82-351`

### 1.2 实现代码

```python
"""
分类错误处理
根据错误类型采取不同的处理策略
"""

import os
import time
from typing import List, Dict, Optional
from dataclasses import dataclass
from enum import Enum
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class ErrorType(Enum):
    """错误类型枚举"""
    INPUT_VALIDATION = "input_validation"
    AUTHENTICATION = "authentication"
    RATE_LIMIT = "rate_limit"
    SERVICE_UNAVAILABLE = "service_unavailable"
    NETWORK_TIMEOUT = "network_timeout"
    BATCH_SIZE_EXCEEDED = "batch_size_exceeded"
    RESOURCE_EXHAUSTED = "resource_exhausted"
    UNKNOWN = "unknown"

@dataclass
class ErrorInfo:
    """错误信息"""
    error_type: ErrorType
    message: str
    retryable: bool
    retry_after: Optional[int] = None  # 秒

class ErrorClassifier:
    """错误分类器"""
    
    @staticmethod
    def classify_error(exception: Exception) -> ErrorInfo:
        """分类错误"""
        error_msg = str(exception).lower()
        
        # 输入验证错误
        if "empty" in error_msg or "invalid input" in error_msg:
            return ErrorInfo(
                error_type=ErrorType.INPUT_VALIDATION,
                message=str(exception),
                retryable=False
            )
        
        # 认证错误
        if "401" in error_msg or "unauthorized" in error_msg or "invalid api key" in error_msg:
            return ErrorInfo(
                error_type=ErrorType.AUTHENTICATION,
                message=str(exception),
                retryable=False
            )
        
        # 速率限制
        if "429" in error_msg or "rate limit" in error_msg:
            retry_after = 60  # 默认60秒
            return ErrorInfo(
                error_type=ErrorType.RATE_LIMIT,
                message=str(exception),
                retryable=True,
                retry_after=retry_after
            )
        
        # 服务不可用
        if "503" in error_msg or "service unavailable" in error_msg:
            return ErrorInfo(
                error_type=ErrorType.SERVICE_UNAVAILABLE,
                message=str(exception),
                retryable=True,
                retry_after=30
            )
        
        # 网络超时
        if "timeout" in error_msg or "connection" in error_msg:
            return ErrorInfo(
                error_type=ErrorType.NETWORK_TIMEOUT,
                message=str(exception),
                retryable=True,
                retry_after=10
            )
        
        # 批量大小超限
        if "batch" in error_msg or "too many" in error_msg:
            return ErrorInfo(
                error_type=ErrorType.BATCH_SIZE_EXCEEDED,
                message=str(exception),
                retryable=False
            )
        
        # 资源耗尽
        if "memory" in error_msg or "disk" in error_msg:
            return ErrorInfo(
                error_type=ErrorType.RESOURCE_EXHAUSTED,
                message=str(exception),
                retryable=False
            )
        
        # 未知错误
        return ErrorInfo(
            error_type=ErrorType.UNKNOWN,
            message=str(exception),
            retryable=True,
            retry_after=5
        )
    
    @staticmethod
    def handle_error(error_info: ErrorInfo) -> str:
        """处理错误并返回建议"""
        if error_info.error_type == ErrorType.INPUT_VALIDATION:
            return "请检查输入数据,确保文本非空且格式正确"
        
        elif error_info.error_type == ErrorType.AUTHENTICATION:
            return "请检查API密钥是否正确配置"
        
        elif error_info.error_type == ErrorType.RATE_LIMIT:
            return f"触发速率限制,建议等待 {error_info.retry_after}秒后重试"
        
        elif error_info.error_type == ErrorType.SERVICE_UNAVAILABLE:
            return "服务暂时不可用,建议稍后重试或切换到备用Provider"
        
        elif error_info.error_type == ErrorType.NETWORK_TIMEOUT:
            return "网络超时,建议检查网络连接或增加超时时间"
        
        elif error_info.error_type == ErrorType.BATCH_SIZE_EXCEEDED:
            return "批量大小超限,建议减小批量大小"
        
        elif error_info.error_type == ErrorType.RESOURCE_EXHAUSTED:
            return "资源不足,建议释放资源或增加系统配置"
        
        else:
            return "未知错误,建议查看详细日志"

# 使用示例
def demo_error_classification():
    """演示错误分类"""
    classifier = ErrorClassifier()
    
    # 模拟不同类型的错误
    test_errors = [
        Exception("Invalid input: empty string provided"),
        Exception("401 Unauthorized: Invalid API key"),
        Exception("429 Too Many Requests: Rate limit exceeded"),
        Exception("503 Service Unavailable"),
        Exception("Connection timeout after 30s"),
        Exception("Batch size 200 exceeds maximum 128"),
    ]
    
    for error in test_errors:
        error_info = classifier.classify_error(error)
        suggestion = classifier.handle_error(error_info)
        
        logger.info(f"\n错误: {error}")
        logger.info(f"类型: {error_info.error_type.value}")
        logger.info(f"可重试: {error_info.retryable}")
        logger.info(f"建议: {suggestion}")

if __name__ == "__main__":
    demo_error_classification()
```

## 策略2:指数退避重试

### 2.1 设计思路

使用指数退避算法实现智能重试:
- **初始延迟**: 1秒
- **最大延迟**: 60秒
- **退避因子**: 2
- **最大重试次数**: 5次

**来源**: 业界最佳实践

### 2.2 实现代码

```python
"""
指数退避重试机制
使用tenacity库实现智能重试
"""

from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
    before_sleep_log
)
import logging

logger = logging.getLogger(__name__)

class RetryableError(Exception):
    """可重试的错误"""
    pass

class NonRetryableError(Exception):
    """不可重试的错误"""
    pass

class RobustEmbeddingClient:
    """带重试机制的Embedding客户端"""
    
    def __init__(self, provider: str, max_retries: int = 5):
        self.provider = provider
        self.max_retries = max_retries
        self.classifier = ErrorClassifier()
    
    @retry(
        stop=stop_after_attempt(5),
        wait=wait_exponential(multiplier=1, min=1, max=60),
        retry=retry_if_exception_type(RetryableError),
        before_sleep=before_sleep_log(logger, logging.WARNING)
    )
    def embed_with_retry(self, texts: List[str]) -> List[List[float]]:
        """带重试的embedding调用"""
        try:
            # 这里应该调用实际的embedding API
            # 为演示目的,我们模拟调用
            logger.info(f"调用 {self.provider} embedding API...")
            
            # 模拟随机失败
            import random
            if random.random() < 0.3:  # 30%失败率
                raise Exception("503 Service Unavailable")
            
            # 模拟成功返回
            return [[0.1, 0.2, 0.3] for _ in texts]
            
        except Exception as e:
            # 分类错误
            error_info = self.classifier.classify_error(e)
            
            logger.error(f"Embedding调用失败: {error_info.message}")
            logger.error(f"错误类型: {error_info.error_type.value}")
            
            # 根据错误类型决定是否重试
            if error_info.retryable:
                logger.warning(f"错误可重试,将在 {error_info.retry_after}秒后重试...")
                if error_info.retry_after:
                    time.sleep(error_info.retry_after)
                raise RetryableError(error_info.message)
            else:
                logger.error(f"错误不可重试: {self.classifier.handle_error(error_info)}")
                raise NonRetryableError(error_info.message)
    
    def safe_embed(self, texts: List[str]) -> Optional[List[List[float]]]:
        """安全的embedding调用(捕获所有异常)"""
        try:
            return self.embed_with_retry(texts)
        except NonRetryableError as e:
            logger.error(f"不可重试的错误: {e}")
            return None
        except Exception as e:
            logger.error(f"重试次数耗尽: {e}")
            return None

# 使用示例
def demo_retry_mechanism():
    """演示重试机制"""
    client = RobustEmbeddingClient(provider="openai", max_retries=5)
    
    texts = ["Document 1", "Document 2", "Document 3"]
    
    logger.info("开始embedding调用...")
    embeddings = client.safe_embed(texts)
    
    if embeddings:
        logger.info(f"✅ 成功获取embeddings: {len(embeddings)}条")
    else:
        logger.error("❌ Embedding调用失败")

if __name__ == "__main__":
    demo_retry_mechanism()
```

## 策略3:熔断器模式

### 3.1 设计思路

实现熔断器模式防止级联失败:
- **关闭状态**: 正常调用
- **打开状态**: 快速失败,不调用API
- **半开状态**: 尝试恢复

**状态转换**:
- 连续失败N次 → 打开
- 打开状态持续T秒 → 半开
- 半开状态成功 → 关闭
- 半开状态失败 → 打开

### 3.2 实现代码

```python
"""
熔断器模式
防止级联失败和资源耗尽
"""

from enum import Enum
from dataclasses import dataclass
import time

class CircuitState(Enum):
    """熔断器状态"""
    CLOSED = "closed"      # 正常
    OPEN = "open"          # 熔断
    HALF_OPEN = "half_open"  # 半开

@dataclass
class CircuitBreakerConfig:
    """熔断器配置"""
    failure_threshold: int = 5  # 失败阈值
    timeout: int = 60  # 熔断超时(秒)
    success_threshold: int = 2  # 半开状态成功阈值

class CircuitBreaker:
    """熔断器"""
    
    def __init__(self, config: CircuitBreakerConfig):
        self.config = config
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.success_count = 0
        self.last_failure_time = None
    
    def call(self, func, *args, **kwargs):
        """通过熔断器调用函数"""
        # 检查是否应该从打开状态转为半开状态
        if self.state == CircuitState.OPEN:
            if time.time() - self.last_failure_time >= self.config.timeout:
                logger.info("熔断器从打开状态转为半开状态")
                self.state = CircuitState.HALF_OPEN
                self.success_count = 0
            else:
                raise Exception("熔断器打开,快速失败")
        
        try:
            # 调用函数
            result = func(*args, **kwargs)
            
            # 成功处理
            self._on_success()
            
            return result
            
        except Exception as e:
            # 失败处理
            self._on_failure()
            raise
    
    def _on_success(self):
        """成功回调"""
        if self.state == CircuitState.HALF_OPEN:
            self.success_count += 1
            logger.info(f"半开状态成功计数: {self.success_count}/{self.config.success_threshold}")
            
            if self.success_count >= self.config.success_threshold:
                logger.info("熔断器从半开状态转为关闭状态")
                self.state = CircuitState.CLOSED
                self.failure_count = 0
        
        elif self.state == CircuitState.CLOSED:
            self.failure_count = 0
    
    def _on_failure(self):
        """失败回调"""
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        logger.warning(f"失败计数: {self.failure_count}/{self.config.failure_threshold}")
        
        if self.state == CircuitState.HALF_OPEN:
            logger.warning("半开状态失败,转为打开状态")
            self.state = CircuitState.OPEN
            self.success_count = 0
        
        elif self.state == CircuitState.CLOSED:
            if self.failure_count >= self.config.failure_threshold:
                logger.error("失败次数达到阈值,熔断器打开")
                self.state = CircuitState.OPEN
    
    def get_state(self) -> CircuitState:
        """获取当前状态"""
        return self.state

class CircuitBreakerEmbeddingClient:
    """带熔断器的Embedding客户端"""
    
    def __init__(self, provider: str):
        self.provider = provider
        config = CircuitBreakerConfig(
            failure_threshold=5,
            timeout=60,
            success_threshold=2
        )
        self.circuit_breaker = CircuitBreaker(config)
    
    def _embed_internal(self, texts: List[str]) -> List[List[float]]:
        """内部embedding实现"""
        # 模拟API调用
        import random
        if random.random() < 0.4:  # 40%失败率
            raise Exception("API调用失败")
        
        return [[0.1, 0.2, 0.3] for _ in texts]
    
    def embed(self, texts: List[str]) -> Optional[List[List[float]]]:
        """通过熔断器调用embedding"""
        try:
            return self.circuit_breaker.call(self._embed_internal, texts)
        except Exception as e:
            logger.error(f"Embedding调用失败: {e}")
            logger.error(f"熔断器状态: {self.circuit_breaker.get_state().value}")
            return None

# 使用示例
def demo_circuit_breaker():
    """演示熔断器"""
    client = CircuitBreakerEmbeddingClient(provider="openai")
    
    texts = ["Document 1", "Document 2"]
    
    # 模拟多次调用
    for i in range(15):
        logger.info(f"\n=== 第 {i+1} 次调用 ===")
        embeddings = client.embed(texts)
        
        if embeddings:
            logger.info("✅ 调用成功")
        else:
            logger.error("❌ 调用失败")
        
        time.sleep(1)

if __name__ == "__main__":
    demo_circuit_breaker()
```

## 策略4:优雅降级

### 4.1 设计思路

当主Provider失败时,自动降级到备用方案:
1. **主Provider**: 高质量embedding (OpenAI, VoyageAI)
2. **备用Provider**: 可靠性高 (Cohere)
3. **本地Provider**: 自托管TEI
4. **缓存**: 使用历史embedding

### 4.2 实现代码

```python
"""
优雅降级策略
多层降级确保服务可用性
"""

from typing import Optional, List, Dict
import hashlib
import json

class EmbeddingCache:
    """Embedding缓存"""
    
    def __init__(self):
        self.cache: Dict[str, List[float]] = {}
    
    def get(self, text: str) -> Optional[List[float]]:
        """获取缓存"""
        key = self._hash(text)
        return self.cache.get(key)
    
    def set(self, text: str, embedding: List[float]):
        """设置缓存"""
        key = self._hash(text)
        self.cache[key] = embedding
    
    def _hash(self, text: str) -> str:
        """计算文本哈希"""
        return hashlib.md5(text.encode()).hexdigest()

class GracefulDegradationClient:
    """优雅降级客户端"""
    
    def __init__(self):
        self.cache = EmbeddingCache()
        self.providers = [
            {"name": "voyageai", "tier": "primary"},
            {"name": "cohere", "tier": "backup"},
            {"name": "tei", "tier": "local"}
        ]
    
    def embed(self, texts: List[str]) -> List[List[float]]:
        """带降级的embedding调用"""
        results = []
        
        for text in texts:
            # 1. 尝试从缓存获取
            cached = self.cache.get(text)
            if cached:
                logger.info(f"✅ 从缓存获取: {text[:30]}...")
                results.append(cached)
                continue
            
            # 2. 尝试各个Provider
            embedding = None
            for provider in self.providers:
                try:
                    embedding = self._call_provider(provider["name"], text)
                    logger.info(f"✅ 使用 {provider['name']} ({provider['tier']})")
                    
                    # 缓存结果
                    self.cache.set(text, embedding)
                    break
                    
                except Exception as e:
                    logger.warning(f"⚠️ {provider['name']} 失败: {e}")
                    continue
            
            # 3. 所有Provider都失败
            if embedding is None:
                logger.error(f"❌ 所有Provider都失败,使用零向量")
                embedding = [0.0] * 1024  # 零向量作为最后降级
            
            results.append(embedding)
        
        return results
    
    def _call_provider(self, provider: str, text: str) -> List[float]:
        """调用Provider"""
        # 模拟API调用
        import random
        if random.random() < 0.3:
            raise Exception(f"{provider} API调用失败")
        
        return [random.random() for _ in range(1024)]

# 使用示例
def demo_graceful_degradation():
    """演示优雅降级"""
    client = GracefulDegradationClient()
    
    texts = [
        "Document 1: Important content",
        "Document 2: More content",
        "Document 1: Important content",  # 重复,应从缓存获取
    ]
    
    embeddings = client.embed(texts)
    logger.info(f"\n✅ 成功获取 {len(embeddings)} 个embeddings")

if __name__ == "__main__":
    demo_graceful_degradation()
```
