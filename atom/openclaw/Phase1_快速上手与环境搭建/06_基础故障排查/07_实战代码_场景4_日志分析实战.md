### 3.1 日志解析器

```typescript
// 日志解析器实现
import * as fs from 'fs';
import * as readline from 'readline';

interface ParsedLog {
  timestamp: Date;
  level: LogLevel;
  source: LogSource;
  message: string;
  context: Record<string, any>;
  raw: string;
}

interface LogPattern {
  pattern: RegExp;
  type: string;
  severity: 'fatal' | 'error' | 'warn' | 'info';
  extract: (match: RegExpMatchArray) => Record<string, any>;
}

class LogParser {
  private patterns: LogPattern[] = [
    {
      pattern: /EADDRINUSE.*:(\d+)/,
      type: 'port_conflict',
      severity: 'fatal',
      extract: (match) => ({ port: parseInt(match[1]) })
    },
    {
      pattern: /pairing request from (\w+) on (\w+)/,
      type: 'pairing_pending',
      severity: 'warn',
      extract: (match) => ({ sender: match[1], channel: match[2] })
    },
    {
      pattern: /unauthorized.*token/i,
      type: 'auth_failed',
      severity: 'error',
      extract: () => ({})
    },
    {
      pattern: /drop guild message \(mention required/,
      type: 'mention_gating',
      severity: 'info',
      extract: () => ({})
    },
    {
      pattern: /Gateway start blocked: set gateway\.mode=(\w+)/,
      type: 'config_error',
      severity: 'fatal',
      extract: (match) => ({ requiredMode: match[1] })
    }
  ];

  parseLine(line: string): ParsedLog | null {
    // 解析日志行格式：2026-02-24 10:30:45 ERROR [gateway] message
    const regex = /(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\s+(\w+)\s+\[(\w+)\]\s+(.+)/;
    const match = line.match(regex);

    if (!match) {
      return null;
    }

    const [, timestamp, level, source, message] = match;

    // 识别日志模式
    let context: Record<string, any> = {};
    for (const pattern of this.patterns) {
      const patternMatch = message.match(pattern.pattern);
      if (patternMatch) {
        context = {
          ...pattern.extract(patternMatch),
          type: pattern.type,
          severity: pattern.severity
        };
        break;
      }
    }

    return {
      timestamp: new Date(timestamp),
      level: level.toLowerCase() as LogLevel,
      source: source as LogSource,
      message,
      context,
      raw: line
    };
  }

  async parseFile(filePath: string): Promise<ParsedLog[]> {
    const logs: ParsedLog[] = [];
    const fileStream = fs.createReadStream(filePath);
    const rl = readline.createInterface({
      input: fileStream,
      crlfDelay: Infinity
    });

    for await (const line of rl) {
      const parsed = this.parseLine(line);
      if (parsed) {
        logs.push(parsed);
      }
    }

    return logs;
  }

  async parseStream(stream: NodeJS.ReadableStream): Promise<AsyncGenerator<ParsedLog>> {
    const rl = readline.createInterface({
      input: stream,
      crlfDelay: Infinity
    });

    async function* generator() {
      for await (const line of rl) {
        const parser = new LogParser();
        const parsed = parser.parseLine(line);
        if (parsed) {
          yield parsed;
        }
      }
    }

    return generator();
  }
}
```

### 3.2 实时日志监控

```typescript
// 实时日志监控实现
import { spawn } from 'child_process';
import { EventEmitter } from 'events';

interface MonitorConfig {
  logPath?: string;
  filters?: LogFilter[];
  alerts?: AlertRule[];
}

interface LogFilter {
  level?: LogLevel[];
  source?: LogSource[];
  pattern?: RegExp;
}

interface AlertRule {
  pattern: RegExp;
  severity: 'critical' | 'warning' | 'info';
  action: (log: ParsedLog) => void;
}

class LogMonitor extends EventEmitter {
  private parser: LogParser;
  private config: MonitorConfig;
  private process?: ReturnType<typeof spawn>;

  constructor(config: MonitorConfig = {}) {
    super();
    this.parser = new LogParser();
    this.config = config;
  }

  start(): void {
    // 启动日志跟踪
    this.process = spawn('openclaw', ['logs', '--follow']);

    this.process.stdout.on('data', (data) => {
      const lines = data.toString().split('\n');
      for (const line of lines) {
        if (!line.trim()) continue;

        const parsed = this.parser.parseLine(line);
        if (!parsed) continue;

        // 应用过滤器
        if (this.shouldFilter(parsed)) {
          continue;
        }

        // 触发事件
        this.emit('log', parsed);

        // 检查告警规则
        this.checkAlerts(parsed);
      }
    });

    this.process.stderr.on('data', (data) => {
      this.emit('error', new Error(data.toString()));
    });

    this.process.on('close', (code) => {
      this.emit('close', code);
    });
  }

  stop(): void {
    if (this.process) {
      this.process.kill();
      this.process = undefined;
    }
  }

  private shouldFilter(log: ParsedLog): boolean {
    if (!this.config.filters) {
      return false;
    }

    for (const filter of this.config.filters) {
      // 级别过滤
      if (filter.level && !filter.level.includes(log.level)) {
        return true;
      }

      // 来源过滤
      if (filter.source && !filter.source.includes(log.source)) {
        return true;
      }

      // 模式过滤
      if (filter.pattern && !filter.pattern.test(log.message)) {
        return true;
      }
    }

    return false;
  }

  private checkAlerts(log: ParsedLog): void {
    if (!this.config.alerts) {
      return;
    }

    for (const rule of this.config.alerts) {
      if (rule.pattern.test(log.message)) {
        this.emit('alert', {
          rule,
          log,
          severity: rule.severity
        });

        // 执行告警动作
        rule.action(log);
      }
    }
  }
}

// 使用示例
const monitor = new LogMonitor({
  filters: [
    {
      level: [LogLevel.ERROR, LogLevel.FATAL]
    }
  ],
  alerts: [
    {
      pattern: /EADDRINUSE/,
      severity: 'critical',
      action: (log) => {
        console.error(`[CRITICAL] Port conflict detected: ${log.message}`);
        // 发送告警通知
      }
    },
    {
      pattern: /pairing request/,
      severity: 'warning',
      action: (log) => {
        console.warn(`[WARNING] Pairing request: ${log.message}`);
      }
    }
  ]
});

monitor.on('log', (log) => {
  console.log(`[${log.level.toUpperCase()}] ${log.message}`);
});

monitor.on('alert', ({ rule, log, severity }) => {
  console.error(`[ALERT:${severity}] ${log.message}`);
});

monitor.start();
```

### 3.3 日志聚合分析

```typescript
// 日志聚合分析实现
interface LogStatistics {
  total: number;
  byLevel: Record<LogLevel, number>;
  bySource: Record<LogSource, number>;
  byType: Record<string, number>;
  errors: ParsedLog[];
  warnings: ParsedLog[];
  timeRange: {
    start: Date;
    end: Date;
  };
}

class LogAggregator {
  private logs: ParsedLog[] = [];

  add(log: ParsedLog): void {
    this.logs.push(log);
  }

  addBatch(logs: ParsedLog[]): void {
    this.logs.push(...logs);
  }

  getStatistics(): LogStatistics {
    const stats: LogStatistics = {
      total: this.logs.length,
      byLevel: {} as Record<LogLevel, number>,
      bySource: {} as Record<LogSource, number>,
      byType: {},
      errors: [],
      warnings: [],
      timeRange: {
        start: new Date(),
        end: new Date()
      }
    };

    if (this.logs.length === 0) {
      return stats;
    }

    // 初始化计数器
    for (const level of Object.values(LogLevel)) {
      stats.byLevel[level] = 0;
    }
    for (const source of Object.values(LogSource)) {
      stats.bySource[source] = 0;
    }

    // 聚合统计
    for (const log of this.logs) {
      // 按级别统计
      stats.byLevel[log.level]++;

      // 按来源统计
      stats.bySource[log.source]++;

      // 按类型统计
      const type = log.context.type || 'unknown';
      stats.byType[type] = (stats.byType[type] || 0) + 1;

      // 收集错误和警告
      if (log.level === LogLevel.ERROR || log.level === LogLevel.FATAL) {
        stats.errors.push(log);
      } else if (log.level === LogLevel.WARN) {
        stats.warnings.push(log);
      }
    }

    // 时间范围
    const timestamps = this.logs.map(l => l.timestamp.getTime());
    stats.timeRange.start = new Date(Math.min(...timestamps));
    stats.timeRange.end = new Date(Math.max(...timestamps));

    return stats;
  }

  getTopErrors(limit: number = 10): Array<{ message: string; count: number }> {
    const errorCounts = new Map<string, number>();

    for (const log of this.logs) {
      if (log.level === LogLevel.ERROR || log.level === LogLevel.FATAL) {
        const count = errorCounts.get(log.message) || 0;
        errorCounts.set(log.message, count + 1);
      }
    }

    return Array.from(errorCounts.entries())
      .map(([message, count]) => ({ message, count }))
      .sort((a, b) => b.count - a.count)
      .slice(0, limit);
  }

  getTimeline(intervalMinutes: number = 5): Array<{ time: Date; count: number }> {
    if (this.logs.length === 0) {
      return [];
    }

    const timeline: Array<{ time: Date; count: number }> = [];
    const intervalMs = intervalMinutes * 60 * 1000;

    const start = this.logs[0].timestamp.getTime();
    const end = this.logs[this.logs.length - 1].timestamp.getTime();

    for (let time = start; time <= end; time += intervalMs) {
      const count = this.logs.filter(log => {
        const logTime = log.timestamp.getTime();
        return logTime >= time && logTime < time + intervalMs;
      }).length;

      timeline.push({
        time: new Date(time),
        count
      });
    }

    return timeline;
  }

  clear(): void {
    this.logs = [];
  }
}
```

```bash
#!/bin/bash
# log-analysis.sh - OpenClaw 日志分析脚本

LOG_FILE="${1:-~/.openclaw/logs/gateway.log}"

echo "=== OpenClaw Log Analysis ==="
echo "Log file: $LOG_FILE"
echo ""

# 1. 统计日志级别分布
echo "=== Log Level Distribution ==="
grep -oE "(FATAL|ERROR|WARN|INFO|DEBUG)" "$LOG_FILE" | sort | uniq -c | sort -rn

echo ""

# 2. 统计错误类型
echo "=== Top 10 Errors ==="
grep -E "(ERROR|FATAL)" "$LOG_FILE" | \
  sed 's/.*\[.*\] //' | \
  sort | uniq -c | sort -rn | head -10

echo ""

# 3. 端口冲突检测
echo "=== Port Conflicts ==="
grep -i "EADDRINUSE" "$LOG_FILE" | tail -5

echo ""

# 4. 配对请求统计
echo "=== Pairing Requests ==="
grep "pairing request" "$LOG_FILE" | \
  grep -oE "from \w+ on \w+" | \
  sort | uniq -c | sort -rn

echo ""

# 5. 认证失败统计
echo "=== Authentication Failures ==="
grep -i "unauthorized" "$LOG_FILE" | wc -l | \
  xargs -I {} echo "Total: {} failures"

echo ""

# 6. 最近的致命错误
echo "=== Recent Fatal Errors ==="
grep "FATAL" "$LOG_FILE" | tail -5

echo ""

# 7. 时间线分析（每小时日志数）
echo "=== Hourly Log Count ==="
grep -oE "^\d{4}-\d{2}-\d{2} \d{2}" "$LOG_FILE" | \
```

### 4.2 实时日志监控脚本

```bash
#!/bin/bash
# log-monitor.sh - 实时日志监控和告警

# 颜色定义
RED='\033[0;31m'
YELLOW='\033[1;33m'
GREEN='\033[0;32m'
NC='\033[0m' # No Color

# 告警函数
alert() {
  local severity=$1
  local message=$2
  
  case $severity in
    CRITICAL)
      echo -e "${RED}[CRITICAL]${NC} $message"
      # 发送告警通知（邮件、Slack等）
      ;;
    WARNING)
      echo -e "${YELLOW}[WARNING]${NC} $message"
      ;;
    INFO)
      echo -e "${GREEN}[INFO]${NC} $message"
      ;;
  esac
}

# 监控日志
openclaw logs --follow | while read -r line; do
  # 检测端口冲突
  if echo "$line" | grep -q "EADDRINUSE"; then
    alert CRITICAL "Port conflict detected: $line"
  fi
  
  # 检测配对请求
  if echo "$line" | grep -q "pairing request"; then
    alert WARNING "Pairing request pending: $line"
  fi
  
  # 检测认证失败
  if echo "$line" | grep -qi "unauthorized"; then
    alert CRITICAL "Authentication failed: $line"
  fi
  
  # 检测配置错误
  if echo "$line" | grep -q "Gateway start blocked"; then
    alert CRITICAL "Gateway start blocked: $line"
  fi
  
  # 检测通道断开
  if echo "$line" | grep -q "channel disconnected"; then
    alert WARNING "Channel disconnected: $line"
  fi
done
```

### 4.3 日志归档脚本

```bash
#!/bin/bash
# log-archive.sh - 日志归档和清理

LOG_DIR="$HOME/.openclaw/logs"
ARCHIVE_DIR="$HOME/.openclaw/logs/archive"
DAYS_TO_KEEP=30

mkdir -p "$ARCHIVE_DIR"

echo "=== OpenClaw Log Archive ==="
echo "Log directory: $LOG_DIR"
echo "Archive directory: $ARCHIVE_DIR"
echo "Days to keep: $DAYS_TO_KEEP"
echo ""

# 1. 归档旧日志
echo "Archiving old logs..."
find "$LOG_DIR" -name "*.log" -mtime +7 -type f | while read -r file; do
  filename=$(basename "$file")
  timestamp=$(date -r "$file" +%Y%m%d)
  archive_name="${timestamp}_${filename}.gz"
  
  echo "Archiving: $filename -> $archive_name"
  gzip -c "$file" > "$ARCHIVE_DIR/$archive_name"
  rm "$file"
done

echo ""

# 2. 清理旧归档
echo "Cleaning old archives..."
find "$ARCHIVE_DIR" -name "*.gz" -mtime +$DAYS_TO_KEEP -type f -delete

echo ""

# 3. 统计
echo "=== Archive Statistics ==="
echo "Active logs: $(find "$LOG_DIR" -name "*.log" -type f | wc -l)"
echo "Archived logs: $(find "$ARCHIVE_DIR" -name "*.gz" -type f | wc -l)"
echo "Total size: $(du -sh "$LOG_DIR" | cut -f1)"
```

```typescript
// 日志解析性能优化
class OptimizedLogParser extends LogParser {
  private cache = new Map<string, ParsedLog>();

  parseLine(line: string): ParsedLog | null {
    // 使用缓存避免重复解析
    if (this.cache.has(line)) {
      return this.cache.get(line)!;
    }

    const parsed = super.parseLine(line);
    if (parsed) {
      this.cache.set(line, parsed);
      
      // 限制缓存大小
      if (this.cache.size > 10000) {
        const firstKey = this.cache.keys().next().value;
        this.cache.delete(firstKey);
      }
    }

    return parsed;
  }
}
```
