### 2. Milvus 连接池管理器

```python
# milvus_manager.py
from pymilvus import MilvusClient, DataType, Function, FunctionType
from typing import Optional, List, Dict
import logging
from contextlib import contextmanager
import time

logger = logging.getLogger(__name__)

class MilvusConnectionPool:
    """Milvus 连接池"""
    
    def __init__(self, uri: str, token: str, pool_size: int = 10):
        self.uri = uri
        self.token = token
        self.pool_size = pool_size
        self._pool: List[MilvusClient] = []
        self._initialize_pool()
    
    def _initialize_pool(self):
        """初始化连接池"""
        for _ in range(self.pool_size):
            try:
                client = MilvusClient(uri=self.uri, token=self.token)
                self._pool.append(client)
                logger.info(f"Created Milvus connection {len(self._pool)}/{self.pool_size}")
            except Exception as e:
                logger.error(f"Failed to create Milvus connection: {e}")
                raise
    
    @contextmanager
    def get_client(self):
        """获取连接（上下文管理器）"""
        if not self._pool:
            raise RuntimeError("Connection pool is empty")
        
        client = self._pool.pop()
        try:
            yield client
        finally:
            self._pool.append(client)
    
    def close_all(self):
        """关闭所有连接"""
        for client in self._pool:
            try:
                client.close()
            except Exception as e:
                logger.error(f"Error closing connection: {e}")
        self._pool.clear()
        logger.info("All connections closed")

class MilvusManager:
    """Milvus 管理器"""
    
    def __init__(self, pool: MilvusConnectionPool, collection_name: str):
        self.pool = pool
        self.collection_name = collection_name
        self._ensure_collection()
    
    def _ensure_collection(self):
        """确保 Collection 存在"""
        with self.pool.get_client() as client:
            if not client.has_collection(self.collection_name):
                self._create_collection(client)
    
    def _create_collection(self, client: MilvusClient):
        """创建 Collection"""
        schema = client.create_schema(auto_id=False)
        schema.add_field(field_name="id", datatype=DataType.INT64, is_primary=True)
        schema.add_field(field_name="text", datatype=DataType.VARCHAR, 
                         max_length=2000, enable_analyzer=True)
        schema.add_field(field_name="text_dense", datatype=DataType.FLOAT_VECTOR, dim=384)
        schema.add_field(field_name="text_sparse", datatype=DataType.SPARSE_FLOAT_VECTOR)
        
        bm25_function = Function(
            name="text_bm25_emb",
            input_field_names=["text"],
            output_field_names=["text_sparse"],
            function_type=FunctionType.BM25,
        )
        schema.add_function(bm25_function)
        
        index_params = client.prepare_index_params()
        index_params.add_index(field_name="text_dense", index_type="AUTOINDEX", metric_type="IP")
        index_params.add_index(field_name="text_sparse", index_type="SPARSE_INVERTED_INDEX",
                               metric_type="BM25")
        
        client.create_collection(
            collection_name=self.collection_name,
            schema=schema,
            index_params=index_params
        )
        logger.info(f"Collection '{self.collection_name}' created")
    
    def hybrid_search_with_retry(
        self,
        query_text: str,
        query_dense: List[float],
        top_k: int = 5,
        max_retries: int = 3
    ) -> List[Dict]:
        """带重试的混合检索"""
        from pymilvus import AnnSearchRequest, RRFRanker
        
        for attempt in range(max_retries):
            try:
                with self.pool.get_client() as client:
                    request_dense = AnnSearchRequest(
                        data=[query_dense],
                        anns_field="text_dense",
                        limit=top_k * 2
                    )
                    
                    request_sparse = AnnSearchRequest(
                        data=[query_text],
                        anns_field="text_sparse",
                        limit=top_k * 2
                    )
                    
                    ranker = RRFRanker(k=60)
                    
                    results = client.hybrid_search(
                        collection_name=self.collection_name,
                        reqs=[request_dense, request_sparse],
                        ranker=ranker,
                        limit=top_k,
                        output_fields=["text"]
                    )
                    
                    retrieved_docs = []
                    for hits in results:
                        for hit in hits:
                            retrieved_docs.append({
                                "text": hit["entity"]["text"],
                                "score": hit["distance"]
                            })
                    
                    return retrieved_docs
                    
            except Exception as e:
                logger.warning(f"Search attempt {attempt + 1} failed: {e}")
                if attempt == max_retries - 1:
                    raise
                time.sleep(0.5 * (attempt + 1))  # 指数退避
        
        return []
```

### 3. Redis 缓存管理器

```python
# cache_manager.py
import redis
import json
import hashlib
from typing import Optional, Any
import logging

logger = logging.getLogger(__name__)

class CacheManager:
    """Redis 缓存管理器"""
    
    def __init__(self, redis_url: str, ttl: int = 3600):
        self.redis_client = redis.from_url(redis_url, decode_responses=True)
        self.ttl = ttl
    
    def _generate_key(self, query: str, top_k: int) -> str:
        """生成缓存键"""
        key_str = f"{query}:{top_k}"
        return f"search:{hashlib.md5(key_str.encode()).hexdigest()}"
    
    def get(self, query: str, top_k: int) -> Optional[Any]:
        """获取缓存"""
        try:
            key = self._generate_key(query, top_k)
            cached = self.redis_client.get(key)
            if cached:
                logger.info(f"Cache hit for query: {query[:50]}...")
                return json.loads(cached)
            return None
        except Exception as e:
            logger.error(f"Cache get error: {e}")
            return None
    
    def set(self, query: str, top_k: int, value: Any):
        """设置缓存"""
        try:
            key = self._generate_key(query, top_k)
            self.redis_client.setex(
                key,
                self.ttl,
                json.dumps(value, ensure_ascii=False)
            )
            logger.info(f"Cached result for query: {query[:50]}...")
        except Exception as e:
            logger.error(f"Cache set error: {e}")
    
    def clear_all(self):
        """清空所有缓存"""
        try:
            keys = self.redis_client.keys("search:*")
            if keys:
                self.redis_client.delete(*keys)
                logger.info(f"Cleared {len(keys)} cache entries")
        except Exception as e:
            logger.error(f"Cache clear error: {e}")
```

```python
# main.py
from fastapi import FastAPI, HTTPException, Depends
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel, Field
from typing import List, Optional
from sentence_transformers import SentenceTransformer
import logging
from contextlib import asynccontextmanager

from config import settings
from milvus_manager import MilvusConnectionPool, MilvusManager
from cache_manager import CacheManager

# 配置日志
logging.basicConfig(
    level=settings.LOG_LEVEL,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

# 全局资源
milvus_pool: Optional[MilvusConnectionPool] = None
milvus_manager: Optional[MilvusManager] = None
cache_manager: Optional[CacheManager] = None
embedding_model: Optional[SentenceTransformer] = None

@asynccontextmanager
async def lifespan(app: FastAPI):
    """应用生命周期管理"""
    global milvus_pool, milvus_manager, cache_manager, embedding_model
    
    # 启动时初始化
    logger.info("Initializing application...")
    
    try:
        # 初始化 Milvus
        milvus_pool = MilvusConnectionPool(
            uri=settings.MILVUS_URI,
            token=settings.MILVUS_TOKEN,
            pool_size=settings.MILVUS_POOL_SIZE
        )
        milvus_manager = MilvusManager(milvus_pool, settings.MILVUS_COLLECTION)
        
        # 初始化 Redis
        cache_manager = CacheManager(settings.REDIS_URL, settings.CACHE_TTL)
        
        # 加载 Embedding 模型
        embedding_model = SentenceTransformer('sentence-transformers/all-MiniLM-L6-v2')
        
        logger.info("Application initialized successfully")
        
    except Exception as e:
        logger.error(f"Failed to initialize application: {e}")
        raise
    
    yield
    
    # 关闭时清理
    logger.info("Shutting down application...")
    if milvus_pool:
        milvus_pool.close_all()
    logger.info("Application shutdown complete")

# 创建 FastAPI 应用
app = FastAPI(
    title="Production Hybrid Search API",
    description="生产级混合检索系统",
    version="1.0.0",
    lifespan=lifespan
)

# CORS 配置
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 请求/响应模型
class SearchRequest(BaseModel):
    query: str = Field(..., description="搜索查询", min_length=1, max_length=500)
    top_k: int = Field(5, description="返回结果数量", ge=1, le=100)
    use_cache: bool = Field(True, description="是否使用缓存")

class SearchResult(BaseModel):
    text: str
    score: float

class SearchResponse(BaseModel):
    query: str
    results: List[SearchResult]
    total: int
    cached: bool = False

# API 端点
@app.get("/health")
async def health_check():
    """健康检查"""
    try:
        # 检查 Milvus 连接
        with milvus_pool.get_client() as client:
            client.list_collections()
        
        # 检查 Redis 连接
        cache_manager.redis_client.ping()
        
        return {
            "status": "healthy",
            "milvus": "connected",
            "redis": "connected",
            "model": "loaded"
        }
    except Exception as e:
        logger.error(f"Health check failed: {e}")
        raise HTTPException(status_code=503, detail=f"Service unhealthy: {str(e)}")

@app.post("/search", response_model=SearchResponse)
async def search(request: SearchRequest):
    """混合检索"""
    try:
        # 检查缓存
        if request.use_cache:
            cached_result = cache_manager.get(request.query, request.top_k)
            if cached_result:
                return SearchResponse(
                    query=request.query,
                    results=[SearchResult(**r) for r in cached_result],
                    total=len(cached_result),
                    cached=True
                )
        
        # 生成查询向量
        query_dense = embedding_model.encode([request.query])[0].tolist()
        
        # 执行混合检索
        results = milvus_manager.hybrid_search_with_retry(
            query_text=request.query,
            query_dense=query_dense,
            top_k=request.top_k
        )
        
        # 缓存结果
        if request.use_cache:
            cache_manager.set(request.query, request.top_k, results)
        
        return SearchResponse(
            query=request.query,
            results=[SearchResult(**r) for r in results],
            total=len(results),
            cached=False
        )
        
    except Exception as e:
        logger.error(f"Search error: {e}")
        raise HTTPException(status_code=500, detail=f"Search failed: {str(e)}")

@app.post("/cache/clear")
async def clear_cache():
    """清空缓存"""
    try:
        cache_manager.clear_all()
        return {"status": "success", "message": "Cache cleared"}
    except Exception as e:
        logger.error(f"Cache clear error: {e}")
        raise HTTPException(status_code=500, detail=f"Failed to clear cache: {str(e)}")

@app.get("/metrics")
async def get_metrics():
    """获取系统指标"""
    try:
        # 这里可以集成 Prometheus 指标
        return {
            "milvus_pool_size": len(milvus_pool._pool),
            "cache_keys": len(cache_manager.redis_client.keys("search:*"))
        }
    except Exception as e:
        logger.error(f"Metrics error: {e}")
        raise HTTPException(status_code=500, detail=f"Failed to get metrics: {str(e)}")

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(
        "main:app",
        host=settings.API_HOST,
        port=settings.API_PORT,
        workers=settings.API_WORKERS,
        log_level=settings.LOG_LEVEL.lower()
    )
```

## 性能测试

```python
# load_test.py
import asyncio
import aiohttp
import time
from typing import List

async def send_request(session: aiohttp.ClientSession, query: str):
    """发送单个请求"""
    async with session.post(
        "http://localhost:8000/search",
        json={"query": query, "top_k": 5}
    ) as response:
        return await response.json()

async def load_test(num_requests: int = 100, concurrency: int = 10):
    """负载测试"""
    queries = [f"test query {i}" for i in range(num_requests)]
    
    start_time = time.time()
    
    async with aiohttp.ClientSession() as session:
        tasks = []
        for query in queries:
            task = send_request(session, query)
            tasks.append(task)
            
            if len(tasks) >= concurrency:
                await asyncio.gather(*tasks)
                tasks = []
        
        if tasks:
            await asyncio.gather(*tasks)
    
    duration = time.time() - start_time
    qps = num_requests / duration
    
    print(f"Total requests: {num_requests}")
    print(f"Duration: {duration:.2f}s")
    print(f"QPS: {qps:.2f}")

if __name__ == "__main__":
    asyncio.run(load_test(num_requests=1000, concurrency=50))
```
