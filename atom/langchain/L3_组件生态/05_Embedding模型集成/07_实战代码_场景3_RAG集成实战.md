# 实战代码：场景3 - RAG 集成实战

## 场景概述

本文档展示如何将 LangChain Embedding 模型集成到完整的 RAG（检索增强生成）系统中，涵盖从文档加载、分块、向量化、存储到检索和生成的完整流程。

**核心流程**：
```
文档加载 → 文本分块 → 向量化 → 存储到 Chroma → 语义检索 → LLM 生成答案
```

**技术栈**：
- Embeddings: OpenAI text-embedding-3-large
- Vector Store: Chroma
- LLM: OpenAI GPT-4
- Framework: LangChain

## 示例 1：文档加载与分块

### 代码实现

```python
"""
场景1：文档加载与分块
功能：加载 PDF 文档并进行智能分块
"""

from langchain_community.document_loaders import PyPDFLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from typing import List
from langchain.schema import Document

def load_and_split_documents(
    pdf_path: str,
    chunk_size: int = 1000,
    chunk_overlap: int = 200
) -> List[Document]:
    """
    加载 PDF 文档并分块

    Args:
        pdf_path: PDF 文件路径
        chunk_size: 每个块的字符数
        chunk_overlap: 块之间的重叠字符数

    Returns:
        分块后的文档列表
    """
    # 1. 加载 PDF 文档
    print(f"📄 加载文档: {pdf_path}")
    loader = PyPDFLoader(pdf_path)
    documents = loader.load()

    print(f"✅ 加载完成，共 {len(documents)} 页")

    # 2. 创建文本分割器
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=chunk_size,
        chunk_overlap=chunk_overlap,
        length_function=len,
        separators=["\n\n", "\n", " ", ""]  # 优先按段落分割
    )

    # 3. 分割文档
    print(f"✂️  分块中 (chunk_size={chunk_size}, overlap={chunk_overlap})...")
    splits = text_splitter.split_documents(documents)

    print(f"✅ 分块完成，共 {len(splits)} 个块")

    # 4. 显示示例
    if splits:
        print(f"\n📝 第一个块预览:")
        print(f"内容: {splits[0].page_content[:200]}...")
        print(f"元数据: {splits[0].metadata}")

    return splits


# 使用示例
if __name__ == "__main__":
    # 加载并分块文档
    splits = load_and_split_documents(
        pdf_path="data/sample.pdf",
        chunk_size=1000,
        chunk_overlap=200
    )

    # 统计信息
    total_chars = sum(len(doc.page_content) for doc in splits)
    avg_chars = total_chars / len(splits) if splits else 0

    print(f"\n📊 统计信息:")
    print(f"  - 总块数: {len(splits)}")
    print(f"  - 总字符数: {total_chars}")
    print(f"  - 平均每块字符数: {avg_chars:.0f}")
```

### 关键点说明

1. **RecursiveCharacterTextSplitter**：
   - 递归分割，优先按段落（`\n\n`）分割
   - 保持语义完整性
   - 支持自定义分隔符优先级

2. **chunk_overlap**：
   - 块之间重叠，避免语义断裂
   - 推荐值：chunk_size 的 10-20%

3. **元数据保留**：
   - 自动保留页码、来源等信息
   - 便于后续溯源

## 示例 2：文档向量化与存储（Chroma）

"""
场景2：文档向量化与存储
功能：使用 OpenAI Embeddings 向量化文档并存储到 Chroma
"""

import os
from dotenv import load_dotenv
from langchain_openai import OpenAIEmbeddings
from langchain_community.vectorstores import Chroma
from langchain.schema import Document
from typing import List

# 加载环境变量
load_dotenv()

def create_vector_store(
    documents: List[Document],
    collection_name: str = "rag_documents",
    persist_directory: str = "./chroma_db"
) -> Chroma:
    """
    创建向量存储

    Args:
        documents: 文档列表
        collection_name: 集合名称
        persist_directory: 持久化目录

    Returns:
        Chroma 向量存储实例
    """
    # 1. 初始化 Embeddings
    print("🔧 初始化 OpenAI Embeddings...")
    embeddings = OpenAIEmbeddings(
        model="text-embedding-3-large",
        # 可选：使用自定义 API 端点
        # openai_api_base="https://your-proxy.com/v1"
    )

    # 2. 创建向量存储
    print(f"💾 创建向量存储 (collection: {collection_name})...")
    vectorstore = Chroma.from_documents(
        documents=documents,
        embedding=embeddings,
        collection_name=collection_name,
        persist_directory=persist_directory
    )

    print(f"✅ 向量存储创建完成，已存储 {len(documents)} 个文档")

    # 3. 持久化
    vectorstore.persist()
    print(f"💾 已持久化到: {persist_directory}")

    return vectorstore


def load_existing_vector_store(
    collection_name: str = "rag_documents",
    persist_directory: str = "./chroma_db"
) -> Chroma:
    """
    加载已存在的向量存储

    Args:
        collection_name: 集合名称
        persist_directory: 持久化目录

    Returns:
        Chroma 向量存储实例
    """
    print(f"📂 加载向量存储 (collection: {collection_name})...")

    embeddings = OpenAIEmbeddings(model="text-embedding-3-large")

    vectorstore = Chroma(
        collection_name=collection_name,
        embedding_function=embeddings,
        persist_directory=persist_directory
    )

    print("✅ 向量存储加载完成")
    return vectorstore

## 示例 3：语义检索

### 代码实现

```python
"""
场景3：语义检索
功能：基于向量相似度的语义检索
"""

from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings
from typing import List, Tuple
from langchain.schema import Document

def semantic_search(
    vectorstore: Chroma,
    query: str,
    k: int = 4,
    score_threshold: float = 0.0
) -> List[Tuple[Document, float]]:
    """
    语义检索

    Args:
        vectorstore: 向量存储
        query: 查询文本
        k: 返回结果数量
        score_threshold: 相似度阈值（0-1）

    Returns:
        (文档, 相似度分数) 列表
    """
    print(f"🔍 检索查询: {query}")
    print(f"📊 参数: k={k}, threshold={score_threshold}")

    # 相似度检索（带分数）
    results = vectorstore.similarity_search_with_score(
        query=query,
        k=k
    )

    # 过滤低分结果
    filtered_results = [
        (doc, score) for doc, score in results
        if score >= score_threshold
    ]

    print(f"✅ 找到 {len(filtered_results)} 个相关文档")

    # 显示结果
    for i, (doc, score) in enumerate(filtered_results, 1):
        print(f"\n📄 结果 {i} (相似度: {score:.4f}):")
        print(f"内容: {doc.page_content[:150]}...")
        print(f"元数据: {doc.metadata}")

    return filtered_results


def advanced_search(
    vectorstore: Chroma,
    query: str,
    k: int = 4,
    filter_dict: dict = None
) -> List[Document]:
    """
    高级检索（带元数据过滤）

    Args:
        vectorstore: 向量存储
        query: 查询文本
        k: 返回结果数量
        filter_dict: 元数据过滤条件

    Returns:
        文档列表
    """
    print(f"🔍 高级检索: {query}")
    if filter_dict:
        print(f"🔧 过滤条件: {filter_dict}")

    # 带过滤的检索
    results = vectorstore.similarity_search(
        query=query,
        k=k,
        filter=filter_dict
    )

    print(f"✅ 找到 {len(results)} 个文档")
    return results


# 使用示例
if __name__ == "__main__":
    from dotenv import load_dotenv
    load_dotenv()

    # 加载向量存储
    embeddings = OpenAIEmbeddings(model="text-embedding-3-large")
    vectorstore = Chroma(
        collection_name="demo_collection",
        embedding_function=embeddings,
        persist_directory="./chroma_db"
    )

    # 示例1：基础语义检索
    print("=" * 60)
    print("示例1：基础语义检索")
    print("=" * 60)
    results = semantic_search(
        vectorstore=vectorstore,
        query="什么是 RAG？",
        k=2
    )

    # 示例2：带阈值的检索
    print("\n" + "=" * 60)
    print("示例2：带阈值的检索")
    print("=" * 60)
    results = semantic_search(
        vectorstore=vectorstore,
        query="向量数据库",
        k=3,
        score_threshold=0.5
    )

    # 示例3：带元数据过滤的检索
    print("\n" + "=" * 60)
    print("示例3：元数据过滤检索")
    print("=" * 60)
    results = advanced_search(
        vectorstore=vectorstore,
        query="LangChain",
        k=2,
        filter_dict={"page": 1}  # 只检索第1页的内容
    )
```

### 关键点说明

1. **相似度计算**：
   - Chroma 默认使用余弦相似度
   - 分数越高越相似（0-1 范围）

2. **Top-K 检索**：
   - `k` 参数控制返回数量
   - 推荐值：3-5 个文档

3. **元数据过滤**：
   - 支持按来源、页码等过滤
   - 提高检索精准度

## 示例 4：完整 RAG 问答管道

### 代码实现

```python
"""
场景4：完整 RAG 问答管道
功能：端到端的 RAG 系统，从文档加载到答案生成
"""

import os
from dotenv import load_dotenv
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_community.vectorstores import Chroma
from langchain_community.document_loaders import PyPDFLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.chains import RetrievalQA
from langchain.prompts import PromptTemplate
from langchain.schema.runnable import RunnablePassthrough
from langchain.schema.output_parser import StrOutputParser

# 加载环境变量
load_dotenv()


class RAGSystem:
    """完整的 RAG 系统"""

    def __init__(
        self,
        collection_name: str = "rag_system",
        persist_directory: str = "./chroma_db"
    ):
        """初始化 RAG 系统"""
        self.collection_name = collection_name
        self.persist_directory = persist_directory

        # 初始化组件
        self.embeddings = OpenAIEmbeddings(model="text-embedding-3-large")
        self.llm = ChatOpenAI(model="gpt-4", temperature=0)
        self.vectorstore = None
        self.qa_chain = None

        print("✅ RAG 系统初始化完成")

    def load_documents(self, pdf_path: str):
        """加载并处理文档"""
        print(f"\n📄 加载文档: {pdf_path}")

        # 1. 加载 PDF
        loader = PyPDFLoader(pdf_path)
        documents = loader.load()

        # 2. 分块
        text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=1000,
            chunk_overlap=200
        )
        splits = text_splitter.split_documents(documents)

        print(f"✅ 文档处理完成: {len(splits)} 个块")

        # 3. 创建向量存储
        print("💾 创建向量存储...")
        self.vectorstore = Chroma.from_documents(
            documents=splits,
            embedding=self.embeddings,
            collection_name=self.collection_name,
            persist_directory=self.persist_directory
        )
        self.vectorstore.persist()

        print("✅ 向量存储创建完成")

        # 4. 构建 QA 链
        self._build_qa_chain()

    def load_existing_store(self):
        """加载已存在的向量存储"""
        print(f"\n📂 加载向量存储...")

        self.vectorstore = Chroma(
            collection_name=self.collection_name,
            embedding_function=self.embeddings,
            persist_directory=self.persist_directory
        )

        print("✅ 向量存储加载完成")

        # 构建 QA 链
        self._build_qa_chain()

    def _build_qa_chain(self):
        """构建问答链"""
        # 创建检索器
        retriever = self.vectorstore.as_retriever(
            search_kwargs={"k": 4}
        )

        # 定义 Prompt 模板
        template = """使用以下上下文回答问题。如果不知道答案，就说不知道，不要编造答案。

上下文：
{context}

问题：{question}

答案："""

        prompt = PromptTemplate(
            template=template,
            input_variables=["context", "question"]
        )

        # 构建 LCEL 链
        self.qa_chain = (
            {
                "context": retriever | self._format_docs,
                "question": RunnablePassthrough()
            }
            | prompt
            | self.llm
            | StrOutputParser()
        )

        print("✅ QA 链构建完成")

    def _format_docs(self, docs):
        """格式化检索到的文档"""
        return "\n\n".join(doc.page_content for doc in docs)

    def ask(self, question: str) -> str:
        """提问"""
        if not self.qa_chain:
            raise ValueError("请先加载文档或向量存储")

        print(f"\n❓ 问题: {question}")
        print("🤔 思考中...")

        # 执行问答
        answer = self.qa_chain.invoke(question)

        print(f"\n💡 答案: {answer}")
        return answer

    def ask_with_sources(self, question: str):
        """提问并返回来源"""
        if not self.vectorstore:
            raise ValueError("请先加载文档或向量存储")

        print(f"\n❓ 问题: {question}")

        # 1. 检索相关文档
        docs = self.vectorstore.similarity_search(question, k=4)

        # 2. 生成答案
        context = self._format_docs(docs)
        answer = self.qa_chain.invoke(question)

        # 3. 显示结果
        print(f"\n💡 答案: {answer}")
        print(f"\n📚 来源文档:")
        for i, doc in enumerate(docs, 1):
            print(f"  {i}. {doc.metadata.get('source', 'Unknown')} "
                  f"(页码: {doc.metadata.get('page', 'N/A')})")

        return {
            "answer": answer,
            "sources": docs
        }


# 使用示例
if __name__ == "__main__":
    # 创建 RAG 系统
    rag = RAGSystem(collection_name="my_rag_system")

    # 方式1：加载新文档
    # rag.load_documents("data/sample.pdf")

    # 方式2：加载已存在的向量存储
    rag.load_existing_store()

    # 问答示例
    print("\n" + "=" * 60)
    print("RAG 问答系统")
    print("=" * 60)

    # 示例1：基础问答
    rag.ask("什么是 LangChain？")

    # 示例2：带来源的问答
    result = rag.ask_with_sources("RAG 的工作原理是什么？")

    # 示例3：多轮对话
    questions = [
        "什么是向量数据库？",
        "Chroma 有什么特点？",
        "如何选择 Embedding 模型？"
    ]

    for q in questions:
        rag.ask(q)
        print("\n" + "-" * 60)
```

### 关键点说明

1. **LCEL 链式调用**：
   - 使用 `|` 操作符连接组件
   - 清晰的数据流
   - 易于调试和扩展

2. **Prompt 工程**：
   - 明确指示不要编造答案
   - 提供上下文和问题
   - 控制输出格式

3. **来源追溯**：
   - 保留文档元数据
   - 显示答案来源
   - 提高可信度

## 性能优化建议

### 1. 使用缓存

```python
from langchain.embeddings import CacheBackedEmbeddings
from langchain.storage import LocalFileStore

# 创建缓存
store = LocalFileStore("./embeddings_cache")
cached_embeddings = CacheBackedEmbeddings.from_bytes_store(
    OpenAIEmbeddings(),
    store,
    namespace="openai-embeddings"
)

# 使用缓存的 Embeddings
vectorstore = Chroma.from_documents(
    documents=splits,
    embedding=cached_embeddings
)
```

### 2. 批量处理

```python
# 批量嵌入文档
batch_size = 100
for i in range(0, len(documents), batch_size):
    batch = documents[i:i+batch_size]
    vectorstore.add_documents(batch)
```

### 3. 异步处理

```python
import asyncio

async def async_rag_query(rag_system, question):
    """异步问答"""
    return await rag_system.qa_chain.ainvoke(question)

# 并发处理多个问题
questions = ["问题1", "问题2", "问题3"]
results = await asyncio.gather(*[
    async_rag_query(rag, q) for q in questions
])
```
