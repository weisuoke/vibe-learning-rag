## 场景1：简单比较查询

### 数据准备

```python
from pymilvus import MilvusClient, DataType
from sentence_transformers import SentenceTransformer
import random

# 1. 初始化
client = MilvusClient("http://localhost:19530")
model = SentenceTransformer('all-MiniLM-L6-v2')

# 2. 创建 schema
schema = client.create_schema(auto_id=True)
schema.add_field("id", DataType.INT64, is_primary=True)
schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
schema.add_field("age", DataType.INT32)
schema.add_field("city", DataType.VARCHAR, max_length=100)
schema.add_field("salary", DataType.DOUBLE)

# 3. 创建集合
collection_name = "users_basic"
if client.has_collection(collection_name):
    client.drop_collection(collection_name)

client.create_collection(
    collection_name=collection_name,
    schema=schema
)

# 4. 准备数据
cities = ["北京", "上海", "深圳", "广州", "杭州"]
data = []

for i in range(10000):
    text = f"User {i}"
    vector = model.encode(text).tolist()
    data.append({
        "vector": vector,
        "age": random.randint(20, 60),
        "city": random.choice(cities),
        "salary": random.uniform(5000, 50000)
    })

# 5. 插入数据
client.insert(collection_name, data)
print(f"插入 {len(data)} 条数据")
```

### 创建索引

```python
# 准备索引参数
index_params = client.prepare_index_params()

# 向量索引
index_params.add_index(
    field_name="vector",
    index_type="AUTOINDEX",
    metric_type="COSINE"
)

# 标量索引 - age
index_params.add_index(
    field_name="age",
    index_type="INVERTED"
)

# 标量索引 - city
index_params.add_index(
    field_name="city",
    index_type="BITMAP"  # 低基数字段使用 BITMAP
)

# 标量索引 - salary
index_params.add_index(
    field_name="salary",
    index_type="INVERTED"
)

# 创建索引
client.create_index(collection_name, index_params)
print("索引创建完成")
```

### 测试1：有索引 vs 无索引

```python
import time

# 创建无索引集合
collection_name_no_index = "users_no_index"

    client.drop_collection(collection_name_no_index)

    collection_name=collection_name_no_index,

print("\n性能对比：")
print(f"有索引：{time_with_index:.2f} ms")
```

## 常见问题

### Q1: 查询返回空结果

**问题**：
```python
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="age > 100",  # 没有数据满足条件
    limit=10
)
# 返回空结果
```

**解决方案**：
```python
# 1. 先不加过滤，确认数据存在
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    limit=10,
    output_fields=["age"]
)
print("数据范围：", [hit['entity']['age'] for hits in results for hit in hits])

# 2. 调整过滤条件
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="age > 20",  # 调整为合理范围
    limit=10
)
```

### Q2: 索引创建失败

**问题**：
```python
# 错误：字段不存在
index_params.add_index("non_existent_field", "INVERTED")
# MilvusException: field not found
```

**解决方案**：
```python
# 检查 schema
schema_info = client.describe_collection(collection_name)
print("字段列表：", [field['name'] for field in schema_info['fields']])

# 确认字段名称正确
index_params.add_index("age", "INVERTED")  # 使用正确的字段名
```

### Q3: 字符串比较错误

**问题**：
```python
# 错误：字符串需要用引号
filter = "city == 北京"  # 语法错误
```

**解决方案**：
```python
# 正确：字符串用单引号或双引号
filter = "city == '北京'"  # 正确
# 或
filter = 'city == "北京"'  # 正确
```

### Q4: 性能慢

**问题**：
```python
# 查询很慢（> 1000ms）
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="age > 30",
    limit=10
)
```

**解决方案**：
```python
# 1. 检查是否创建了索引
index_info = client.list_indexes(collection_name)
print("索引列表：", index_info)

# 2. 如果没有索引，创建索引
if "age" not in [idx['field_name'] for idx in index_info]:
    index_params = client.prepare_index_params()
    index_params.add_index("age", "INVERTED")
    client.create_index(collection_name, index_params)

# 3. 重新查询
results = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter="age > 30",
    limit=10
)
```

## 完整示例代码

### 保存为 `basic_scalar_filtering.py`

```python
from pymilvus import MilvusClient, DataType
from sentence_transformers import SentenceTransformer
import random
import time

def main():
    # 1. 初始化
    client = MilvusClient("http://localhost:19530")
    model = SentenceTransformer('all-MiniLM-L6-v2')

    # 2. 创建集合
    collection_name = "users_demo"
    if client.has_collection(collection_name):
        client.drop_collection(collection_name)

    schema = client.create_schema(auto_id=True)
    schema.add_field("id", DataType.INT64, is_primary=True)
    schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
    schema.add_field("age", DataType.INT32)
    schema.add_field("city", DataType.VARCHAR, max_length=100)
    schema.add_field("salary", DataType.DOUBLE)

    client.create_collection(collection_name=collection_name, schema=schema)

    # 3. 准备数据
    cities = ["北京", "上海", "深圳", "广州", "杭州"]
    data = []
    for i in range(10000):
        text = f"User {i}"
        vector = model.encode(text).tolist()
        data.append({
            "vector": vector,
            "age": random.randint(20, 60),
            "city": random.choice(cities),
            "salary": random.uniform(5000, 50000)
        })

    # 4. 插入数据
    client.insert(collection_name, data)
    print(f"插入 {len(data)} 条数据")

    # 5. 创建索引
    index_params = client.prepare_index_params()
    index_params.add_index("vector", "AUTOINDEX", "COSINE")
    index_params.add_index("age", "INVERTED")
    index_params.add_index("city", "BITMAP")
    index_params.add_index("salary", "INVERTED")
    client.create_index(collection_name, index_params)
    print("索引创建完成")

    # 6. 查询示例
    query_vector = model.encode("User query").tolist()

    # 简单比较
    print("\n=== 简单比较查询 ===")
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="age > 30",
        limit=5,
        output_fields=["age", "city", "salary"]
    )
    for hits in results:
        for hit in hits:
            print(f"Age: {hit['entity']['age']}, City: {hit['entity']['city']}")

    # 范围查询
    print("\n=== 范围查询 ===")
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="age >= 25 AND age <= 35",
        limit=5,
        output_fields=["age", "city", "salary"]
    )
    for hits in results:
        for hit in hits:
            print(f"Age: {hit['entity']['age']}")

    # IN 操作符
    print("\n=== IN 操作符 ===")
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="city IN ['北京', '上海', '深圳']",
        limit=5,
        output_fields=["age", "city", "salary"]
    )
    for hits in results:
        for hit in hits:
            print(f"City: {hit['entity']['city']}")

    # 7. 性能测试
    print("\n=== 性能测试 ===")
    start = time.time()
    client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter="age > 30 AND city == '北京'",
        limit=10
    )
    elapsed = (time.time() - start) * 1000
    print(f"查询延迟：{elapsed:.2f} ms")

if __name__ == "__main__":
    main()
```

## 性能优化技巧

### 1. 索引选择

```python
# 低基数字段（< 100个值）→ BITMAP
index_params.add_index("city", "BITMAP")  # 只有5个城市

# 中高基数字段 → INVERTED
index_params.add_index("age", "INVERTED")  # 年龄范围广

# 不确定 → AUTOINDEX
index_params.add_index("field", "AUTOINDEX")  # 让 Milvus 自动选择
```

### 2. 查询优化

```python
# 选择性高的条件在前
filter = "city == '北京' AND age > 30"  # city 选择性高（20%），先执行

# 使用 IN 替代多个 OR
filter = "city IN ['北京', '上海', '深圳']"  # 推荐
# 而不是：city == '北京' OR city == '上海' OR city == '深圳'
```

### 3. 批量操作

```python
# 批量插入
client.insert(collection_name, data_list)  # 一次插入多条

# 批量查询
query_vectors = [vector1, vector2, vector3]
results = client.search(
    collection_name=collection_name,
    data=query_vectors,  # 一次查询多个向量
    limit=10
)
```

## 类比理解

### 前端开发类比

**标量过滤 = JavaScript 数组过滤**：
```javascript
// JavaScript
const users = [
  { age: 25, city: '北京', salary: 10000 },
  { age: 35, city: '上海', salary: 20000 }
];

// 过滤：age > 30
const filtered = users.filter(u => u.age > 30);

// Milvus（等价）
filter = "age > 30"
```

**索引 = Map 查找**：
```javascript
// 无索引：线性查找 O(n)
const result = users.find(u => u.age === 30);

// 有索引：哈希表查找 O(1)
const ageMap = new Map();
ageMap.set(30, [user1, user2]);
const result = ageMap.get(30);
```

### 日常生活类比

**标量过滤 = 餐厅筛选**：
```
找餐厅的条件：
- 价格：100-500元（age >= 100 AND age <= 500）
- 距离：< 3km（distance < 3）
- 营业中（status == 'open'）

组合：
"在满足价格、距离、营业状态的餐厅中，找口味最相似的"
= 标量过滤 + 向量检索
```
