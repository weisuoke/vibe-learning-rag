## 4.1 最基础的循环：条件边回连

### 概念说明

**循环是什么？**

LangGraph 中的循环就是"边指向图中已经存在的节点"。最常见的方式是用条件边（conditional edge）决定：继续循环，还是走向 END。

**核心模式**：

```
A → B → (条件判断) → 回到 A（继续）
                   → END（结束）
```

**关键**：路由函数返回节点名就继续循环，返回 `END` 就结束。

### 代码示例

```python
"""
最基础的循环：A → B → A → B → ... → END
节点 A 追加数据，节点 B 追加数据，条件边决定是否继续
"""

import operator
from typing import Annotated, Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    aggregate: Annotated[list[str], operator.add]


def node_a(state: State) -> dict:
    print(f'Node A sees {state["aggregate"]}')
    return {"aggregate": ["A"]}


def node_b(state: State) -> dict:
    print(f'Node B sees {state["aggregate"]}')
    return {"aggregate": ["B"]}


def route(state: State) -> Literal["b", "__end__"]:
    """路由函数：列表长度 < 7 就继续循环，否则结束"""
    if len(state["aggregate"]) < 7:
        return "b"
    else:
        return END


# 构建图
builder = StateGraph(State)
builder.add_node("a", node_a)
builder.add_node("b", node_b)

# 边的连接：START → a，a 通过条件边决定去 b 还是 END，b 固定回到 a
builder.add_edge(START, "a")
builder.add_conditional_edges("a", route)
builder.add_edge("b", "a")  # 关键：b 回连到 a，形成循环

graph = builder.compile()

# 执行
result = graph.invoke({"aggregate": []})
print(f"\n最终结果: {result['aggregate']}")
```

**运行输出**：
```
Node A sees []
Node B sees ['A']
Node A sees ['A', 'B']
Node B sees ['A', 'B', 'A']
Node A sees ['A', 'B', 'A', 'B']
Node B sees ['A', 'B', 'A', 'B', 'A']
Node A sees ['A', 'B', 'A', 'B', 'A', 'B']
Node B sees ['A', 'B', 'A', 'B', 'A', 'B', 'A']

最终结果: ['A', 'B', 'A', 'B', 'A', 'B', 'A', 'B']
```

### 要点

- `add_edge("b", "a")` 是形成循环的关键——让 b 执行完后回到 a
- 路由函数 `route` 是循环的"开关"——返回 `"b"` 继续，返回 `END` 停止
- `Annotated[list, operator.add]` 让每次迭代的结果追加而非覆盖

## 4.2 设置递归限制

### 概念说明

**为什么需要递归限制？**

如果路由函数的逻辑有 bug，循环可能永远不会停止。递归限制是最后的安全网——超过指定步数就强制停止并抛出 `GraphRecursionError`。

**核心语法**：
```python
# 通过 config 设置递归限制（默认值为 25）
graph.invoke(inputs, config={"recursion_limit": 10})
```

### 代码示例

```python
"""
演示递归限制：故意创建一个不会自然终止的循环
"""

import operator
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.errors import GraphRecursionError


class State(TypedDict):
    count: Annotated[list[str], operator.add]


def loop_node(state: State) -> dict:
    return {"count": ["loop"]}


def always_continue(state: State) -> str:
    """故意永远不返回 END（模拟 bug）"""
    return "loop_node"


builder = StateGraph(State)
builder.add_node("loop_node", loop_node)
builder.add_edge(START, "loop_node")
builder.add_conditional_edges("loop_node", always_continue)

graph = builder.compile()

# 设置较小的递归限制
try:
    result = graph.invoke(
        {"count": []},
        config={"recursion_limit": 5}
    )
except GraphRecursionError as e:
    print(f"捕获到递归限制错误: {e}")
    print("提示: 检查你的路由函数是否有正确的终止条件")
```

**运行输出**：
```
捕获到递归限制错误: Recursion limit of 5 reached without hitting a stop condition.
You can increase the limit by setting the `recursion_limit` config key.
提示: 检查你的路由函数是否有正确的终止条件
```

### 要点

- 默认递归限制是 25，对大多数场景足够
- 简单循环可以设为 3-5，复杂 Agent 可以设为 50-100
- `GraphRecursionError` 继承自 Python 的 `RecursionError`，可以用 try/except 捕获
- 递归限制是"兜底保护"，不应该作为正常的终止条件

## 4.3 状态驱动的终止条件

### 概念说明

**比递归限制更好的方式：用状态字段控制循环终止。**

两种常见模式：
- **计数器模式**：`state["count"] >= max_count` 时停止
- **质量模式**：`state["score"] >= threshold` 时停止

### 代码示例：计数器模式

```python
"""
计数器模式：用状态中的计数器控制循环次数
比递归限制更精确、更可控
"""

import operator
from typing import Annotated, Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    results: Annotated[list[str], operator.add]
    attempt: int  # 计数器（LastValue，每次覆盖）
    max_attempts: int


def process_node(state: State) -> dict:
    """处理节点：每次尝试生成一个结果"""
    attempt = state["attempt"] + 1
    result = f"尝试#{attempt}: 处理完成"
    print(f"  {result}")
    return {
        "results": [result],
        "attempt": attempt,
    }


def should_retry(state: State) -> Literal["process", "__end__"]:
    """计数器终止条件"""
    if state["attempt"] < state["max_attempts"]:
        return "process"
    print(f"  达到最大尝试次数 {state['max_attempts']}，停止")
    return END


builder = StateGraph(State)
builder.add_node("process", process_node)
builder.add_edge(START, "process")
builder.add_conditional_edges("process", should_retry)

graph = builder.compile()

result = graph.invoke({
    "results": [],
    "attempt": 0,
    "max_attempts": 3,
})

print(f"\n所有结果: {result['results']}")
```

**运行输出**：
```
  尝试#1: 处理完成
  尝试#2: 处理完成
  尝试#3: 处理完成
  达到最大尝试次数 3，停止

所有结果: ['尝试#1: 处理完成', '尝试#2: 处理完成', '尝试#3: 处理完成']
```

### 代码示例：质量门控模式

```python
"""
质量门控模式：评估输出质量，不达标就重新生成
典型场景：自我修正 Agent（Generator-Critic 循环）
"""

import operator
import random
from typing import Annotated, Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END


class State(TypedDict):
    drafts: Annotated[list[str], operator.add]
    current_draft: str
    score: float
    iteration: int


def generate_node(state: State) -> dict:
    """生成器：产出一个草稿"""
    iteration = state.get("iteration", 0) + 1
    draft = f"草稿v{iteration}: 这是第{iteration}次生成的内容"
    print(f"  生成: {draft}")
    return {
        "drafts": [draft],
        "current_draft": draft,
        "iteration": iteration,
    }


def evaluate_node(state: State) -> dict:
    """评估器：给草稿打分（模拟）"""
    # 模拟：每次迭代分数递增，第3次达标
    score = min(0.3 * state["iteration"], 1.0)
    print(f"  评分: {score:.1f}")
    return {"score": score}


def should_regenerate(state: State) -> Literal["generate", "__end__"]:
    """质量门控：分数 >= 0.8 就通过"""
    if state["score"] >= 0.8:
        print(f"  ✓ 质量达标，结束循环")
        return END
    print(f"  ✗ 质量不达标，重新生成")
    return "generate"


builder = StateGraph(State)
builder.add_node("generate", generate_node)
builder.add_node("evaluate", evaluate_node)

builder.add_edge(START, "generate")
builder.add_edge("generate", "evaluate")
builder.add_conditional_edges("evaluate", should_regenerate)

graph = builder.compile()

result = graph.invoke({
    "drafts": [],
    "current_draft": "",
    "score": 0.0,
    "iteration": 0,
})

print(f"\n最终草稿: {result['current_draft']}")
print(f"最终分数: {result['score']}")
print(f"总迭代次数: {result['iteration']}")
```

**运行输出**：
```
  生成: 草稿v1: 这是第1次生成的内容
  评分: 0.3
  ✗ 质量不达标，重新生成
  生成: 草稿v2: 这是第2次生成的内容
  评分: 0.6
  ✗ 质量不达标，重新生成
  生成: 草稿v3: 这是第3次生成的内容
  评分: 0.9
  ✓ 质量达标，结束循环

最终草稿: 草稿v3: 这是第3次生成的内容
最终分数: 0.9
总迭代次数: 3
```

### 要点

- 计数器模式适合"固定次数"的场景（重试、批处理）
- 质量门控适合"达标即停"的场景（自我修正、优化）
- 两种模式可以组合：质量达标 OR 超过最大次数，先到先停

## 4.4 RemainingSteps 优雅降级

### 概念说明

**问题**：递归限制触发时会抛出 `GraphRecursionError`，程序直接崩溃。

**更好的方式**：用 `RemainingSteps` 让节点感知"还剩几步"，在快要到限制时主动结束，而不是等着被强制终止。

**核心语法**：
```python
from langgraph.managed.is_last_step import RemainingSteps

class State(TypedDict):
    remaining_steps: RemainingSteps  # 自动注入，无需手动设置
```

### 代码示例

```python
"""
RemainingSteps 优雅降级：
不是等到 GraphRecursionError 崩溃，而是提前感知并优雅退出
"""

import operator
from typing import Annotated, Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.managed.is_last_step import RemainingSteps


class State(TypedDict):
    messages: Annotated[list[str], operator.add]
    remaining_steps: RemainingSteps  # LangGraph 自动填充，不需要手动赋值


def agent_node(state: State) -> dict:
    """Agent 节点：感知剩余步数"""
    remaining = state["remaining_steps"]
    print(f"  Agent: 剩余步数={remaining}")

    if remaining <= 2:
        # 快要到限制了，返回当前最佳结果
        return {"messages": ["[优雅降级] 接近步数限制，返回当前结果"]}

    # 正常处理
    return {"messages": [f"正常处理（剩余{remaining}步）"]}


def route(state: State) -> Literal["agent", "__end__"]:
    """路由：剩余步数 <= 2 时主动结束"""
    if state["remaining_steps"] <= 2:
        return END
    return "agent"


builder = StateGraph(State)
builder.add_node("agent", agent_node)
builder.add_edge(START, "agent")
builder.add_conditional_edges("agent", route)

graph = builder.compile()

# 设置较小的递归限制来观察效果
result = graph.invoke(
    {"messages": []},
    config={"recursion_limit": 6}
)

print(f"\n最终消息:")
for msg in result["messages"]:
    print(f"  - {msg}")
print("（没有抛出 GraphRecursionError！）")
```

**运行输出**：
```
  Agent: 剩余步数=7
  Agent: 剩余步数=5
  Agent: 剩余步数=3
  Agent: 剩余步数=1

最终消息:
  - 正常处理（剩余7步）
  - 正常处理（剩余5步）
  - 正常处理（剩余3步）
  - [优雅降级] 接近步数限制，返回当前结果
（没有抛出 GraphRecursionError！）
```

### 要点

- `RemainingSteps` 是 LangGraph 的"管理值"（Managed Value），由框架自动注入
- 不需要手动计算或维护，只需在 State 中声明即可
- 推荐在 `remaining_steps <= 2` 时开始降级（留 2 步余量给收尾逻辑）
- 这是官方推荐的"主动式"递归处理方式，优于被动等待 `GraphRecursionError`

## 4.5 Command 快速自循环

### 概念说明

**Command 是什么？**

`Command` 允许节点在返回值中同时做两件事：
1. 更新状态（`update`）
2. 指定下一个节点（`goto`）

当 `goto` 指向自身时，就形成了自循环——节点自己决定是否再执行一次。

**核心语法**：
```python
from langgraph.types import Command

# 自循环：goto 指向当前节点自己
return Command(update={"key": "value"}, goto="self_node_name")

# 结束：goto 指向 END
return Command(update={"key": "value"}, goto=END)
```

### 代码示例

```python
"""
Command 自循环：节点自己决定是否继续迭代
适合"状态更新 + 路由决策"需要在同一个地方完成的场景
"""

from typing import Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import Command


class State(TypedDict):
    value: int
    history: list[str]


def doubler_node(state: State) -> Command[Literal["doubler", "__end__"]]:
    """
    翻倍节点：每次将 value 翻倍
    value < 100 时继续自循环，否则结束
    """
    current = state["value"]
    new_value = current * 2
    history = state.get("history", []) + [f"{current} → {new_value}"]

    print(f"  翻倍: {current} → {new_value}")

    if new_value < 100:
        # 继续自循环
        return Command(
            update={"value": new_value, "history": history},
            goto="doubler",  # 指向自己
        )
    else:
        # 达到目标，结束
        return Command(
            update={"value": new_value, "history": history},
            goto=END,
        )


builder = StateGraph(State)
builder.add_node("doubler", doubler_node)
builder.add_edge(START, "doubler")

graph = builder.compile()

result = graph.invoke({"value": 3, "history": []})

print(f"\n最终值: {result['value']}")
print(f"历史: {' → '.join(result['history'])}")
```

**运行输出**：
```
  翻倍: 3 → 6
  翻倍: 6 → 12
  翻倍: 12 → 24
  翻倍: 24 → 48
  翻倍: 48 → 96
  翻倍: 96 → 192

最终值: 192
历史: 3 → 6 → 6 → 12 → 12 → 24 → 24 → 48 → 48 → 96 → 96 → 192
```

### Command vs 条件边：何时用哪个？

| 场景 | 推荐方式 |
|------|----------|
| 只需要路由，不需要更新状态 | 条件边 `add_conditional_edges` |
| 需要同时更新状态 + 路由 | `Command` |
| 多智能体交接 | `Command`（传递信息给下一个 Agent） |
| 简单的 A → B 循环 | 条件边（更清晰） |
| 节点自循环 | `Command`（更简洁） |

### 要点

- `Command` 的 `goto` 可以指向任何节点，包括自身
- `Command` 不会覆盖静态边——如果节点有 `add_edge` 定义的静态边，两者都会执行
- 返回类型注解 `Command[Literal["node_a", "node_b"]]` 是必须的，用于图的编译和渲染
