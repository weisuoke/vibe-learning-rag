## 场景说明

本示例演示 Milvus 2.6 中复杂查询与过滤的高级用法：
- **Query vs Search**：精确查询与向量检索的区别
- **复杂过滤表达式**：多条件组合查询
- **混合检索**：向量检索 + 标量过滤
- **性能对比**：不同查询方式的性能差异

适用场景：
- RAG 系统的精确文档查询
- 基于元数据的过滤检索
- 多条件组合查询
- 性能优化和调优

## 完整代码

```python
"""
Milvus 2.6 复杂查询与过滤实战
演示：Query vs Search、复杂过滤表达式、混合检索
"""

from pymilvus import MilvusClient
import numpy as np
import time
from typing import List, Dict

# ===== 1. 连接 Milvus =====
print("=== 1. 连接 Milvus ===")
client = MilvusClient(uri="http://localhost:19530")
print("✓ 连接成功")

# ===== 2. 创建 Collection =====
print("\n=== 2. 创建 Collection ===")
collection_name = "complex_query_demo"

# 删除已存在的 Collection
if client.has_collection(collection_name):
    client.drop_collection(collection_name)
    print(f"✓ 删除旧 Collection: {collection_name}")

# 创建新 Collection
client.create_collection(
    collection_name=collection_name,
    dimension=128,
    metric_type="L2",
    auto_id=False,
)
print(f"✓ 创建 Collection: {collection_name}")

# ===== 3. 准备测试数据 =====
print("\n=== 3. 准备测试数据 ===")

rng = np.random.default_rng(seed=42)
data = []

# 生成 1000 条测试数据
categories = ["tech", "business", "science", "health", "education"]
tags = ["python", "java", "javascript", "go", "rust", "ai", "ml", "data"]
authors = ["Alice", "Bob", "Charlie", "David", "Eve"]

for i in range(1, 1001):
    data.append({
        "id": i,
        "vector": rng.random(128).tolist(),
        "title": f"Document {i}",
        "content": f"This is the content of document {i}",
        "category": categories[i % len(categories)],
        "tags": [tags[i % len(tags)], tags[(i + 1) % len(tags)]],
        "author": authors[i % len(authors)],
        "score": float(i % 100),
        "views": int(i * 10),
        "published_year": 2020 + (i % 6),
        "is_featured": i % 10 == 0
    })

# 批量插入
result = client.insert(collection_name=collection_name, data=data)
print(f"✓ 插入数据: {result['insert_count']} 条")

# ===== 4. Query vs Search 对比 =====
print("\n=== 4. Query vs Search 对比 ===")

# 4.1 Query：精确查询（基于主键）
print("\n4.1 Query：精确查询（基于主键）")
start_time = time.time()
query_result = client.query(
    collection_name=collection_name,
    ids=[1, 100, 500, 1000],
    output_fields=["id", "title", "category", "score"]
)
query_time = time.time() - start_time

print(f"✓ Query 查询到 {len(query_result)} 条数据，耗时: {query_time*1000:.2f}ms")
for item in query_result[:2]:
    print(f"  - ID: {item['id']}, Title: {item['title']}, Score: {item['score']}")

# 4.2 Search：向量检索
print("\n4.2 Search：向量检索")
query_vector = rng.random(128).tolist()
start_time = time.time()
search_result = client.search(
    collection_name=collection_name,
    data=[query_vector],
    limit=4,
    output_fields=["id", "title", "category", "score"]
)
search_time = time.time() - start_time

print(f"✓ Search 检索到 {len(search_result[0])} 条数据，耗时: {search_time*1000:.2f}ms")
for hit in search_result[0][:2]:
    print(f"  - ID: {hit['id']}, Title: {hit['entity']['title']}, "
          f"Distance: {hit['distance']:.4f}")

print(f"\n性能对比: Query {query_time*1000:.2f}ms vs Search {search_time*1000:.2f}ms")

# ===== 5. 复杂过滤表达式 =====
print("\n=== 5. 复杂过滤表达式 ===")

# 5.1 单条件过滤
print("\n5.1 单条件过滤")
result = client.query(
    collection_name=collection_name,
    filter='category == "tech"',
    output_fields=["id", "title", "category"],
    limit=3
)
print(f"✓ 查询到 {len(result)} 条 tech 类别的数据")

# 5.2 多条件 AND 组合
print("\n5.2 多条件 AND 组合")
result = client.query(
    collection_name=collection_name,
    filter='category == "tech" and score >= 50 and score <= 80',
    output_fields=["id", "title", "category", "score"],
    limit=5
)
print(f"✓ 查询到 {len(result)} 条数据:")
for item in result[:3]:
    print(f"  - ID: {item['id']}, Category: {item['category']}, Score: {item['score']}")

# 5.3 多条件 OR 组合
print("\n5.3 多条件 OR 组合")
result = client.query(
    collection_name=collection_name,
    filter='category == "tech" or category == "science"',
    output_fields=["id", "title", "category"],
    limit=5
)
print(f"✓ 查询到 {len(result)} 条数据:")
for item in result[:3]:
    print(f"  - ID: {item['id']}, Category: {item['category']}")

# 5.4 IN 运算符
print("\n5.4 IN 运算符")
result = client.query(
    collection_name=collection_name,
    filter='category in ["tech", "business", "science"]',
    output_fields=["id", "title", "category"],
    limit=5
)
print(f"✓ 查询到 {len(result)} 条数据:")
for item in result[:3]:
    print(f"  - ID: {item['id']}, Category: {item['category']}")

# 5.5 范围查询
print("\n5.5 范围查询")
result = client.query(
    collection_name=collection_name,
    filter='views >= 5000 and views <= 10000',
    output_fields=["id", "title", "views"],
    limit=5
)
print(f"✓ 查询到 {len(result)} 条数据:")
for item in result[:3]:
    print(f"  - ID: {item['id']}, Views: {item['views']}")

# 5.6 布尔字段过滤
print("\n5.6 布尔字段过滤")
result = client.query(
    collection_name=collection_name,
    filter='is_featured == true',
    output_fields=["id", "title", "is_featured"],
    limit=5
)
print(f"✓ 查询到 {len(result)} 条精选文档:")
for item in result[:3]:
    print(f"  - ID: {item['id']}, Title: {item['title']}")

# 5.7 复杂嵌套条件
print("\n5.7 复杂嵌套条件")
result = client.query(
    collection_name=collection_name,
    filter='(category == "tech" and score >= 70) or (category == "science" and score >= 80)',
    output_fields=["id", "title", "category", "score"],
    limit=5
)
print(f"✓ 查询到 {len(result)} 条数据:")
for item in result[:3]:
    print(f"  - ID: {item['id']}, Category: {item['category']}, Score: {item['score']}")

# ===== 6. 混合检索（向量 + 标量过滤）=====
print("\n=== 6. 混合检索（向量 + 标量过滤）===")

# 6.1 基础混合检索
print("\n6.1 基础混合检索")
query_vector = rng.random(128).tolist()
result = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter='category == "tech" and score >= 50',
    limit=5,
    output_fields=["id", "title", "category", "score"]
)
print(f"✓ 检索到 {len(result[0])} 条数据:")
for hit in result[0][:3]:
    print(f"  - ID: {hit['id']}, Category: {hit['entity']['category']}, "
          f"Score: {hit['entity']['score']}, Distance: {hit['distance']:.4f}")

# 6.2 多条件混合检索
print("\n6.2 多条件混合检索")
result = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter='category in ["tech", "science"] and score >= 60 and published_year >= 2023',
    limit=5,
    output_fields=["id", "title", "category", "score", "published_year"]
)
print(f"✓ 检索到 {len(result[0])} 条数据:")
for hit in result[0][:3]:
    print(f"  - ID: {hit['id']}, Category: {hit['entity']['category']}, "
          f"Year: {hit['entity']['published_year']}, Distance: {hit['distance']:.4f}")

# 6.3 精选文档混合检索
print("\n6.3 精选文档混合检索")
result = client.search(
    collection_name=collection_name,
    data=[query_vector],
    filter='is_featured == true and views >= 500',
    limit=5,
    output_fields=["id", "title", "views", "is_featured"]
)
print(f"✓ 检索到 {len(result[0])} 条精选文档:")
for hit in result[0][:3]:
    print(f"  - ID: {hit['id']}, Views: {hit['entity']['views']}, "
          f"Distance: {hit['distance']:.4f}")

# ===== 7. 性能对比测试 =====
print("\n=== 7. 性能对比测试 ===")

def benchmark_query(description, query_func, iterations=10):
    """性能测试函数"""
    times = []
    for _ in range(iterations):
        start = time.time()
        query_func()
        times.append(time.time() - start)
    avg_time = sum(times) / len(times)
    print(f"{description}: {avg_time*1000:.2f}ms (平均)")
    return avg_time

# 7.1 Query 性能测试
print("\n7.1 Query 性能测试")
benchmark_query(
    "Query - 主键查询",
    lambda: client.query(collection_name, ids=[1, 100, 500], output_fields=["id"])
)
benchmark_query(
    "Query - 简单过滤",
    lambda: client.query(collection_name, filter='category == "tech"', limit=10, output_fields=["id"])
)
benchmark_query(
    "Query - 复杂过滤",
    lambda: client.query(
        collection_name,
        filter='(category == "tech" and score >= 70) or (category == "science" and score >= 80)',
        limit=10,
        output_fields=["id"]
    )
)

# 7.2 Search 性能测试
print("\n7.2 Search 性能测试")
query_vector = rng.random(128).tolist()
benchmark_query(
    "Search - 纯向量检索",
    lambda: client.search(collection_name, data=[query_vector], limit=10)
)
benchmark_query(
    "Search - 简单过滤混合检索",
    lambda: client.search(
        collection_name,
        data=[query_vector],
        filter='category == "tech"',
        limit=10
    )
)
benchmark_query(
    "Search - 复杂过滤混合检索",
    lambda: client.search(
        collection_name,
        data=[query_vector],
        filter='category in ["tech", "science"] and score >= 60',
        limit=10
    )
)

# ===== 8. 清理资源 =====
print("\n=== 8. 清理资源 ===")
client.drop_collection(collection_name)
print(f"✓ 删除 Collection: {collection_name}")

print("\n=== 完成 ===")
```

## 关键知识点

### 1. Query vs Search 的区别

| 特性 | Query | Search |
|------|-------|--------|
| **用途** | 精确查询 | 相似度检索 |
| **基于** | 主键或标量字段 | 向量距离 |
| **返回** | 精确匹配的数据 | Top-K 相似数据 |
| **性能** | 更快（索引查询） | 较慢（向量计算） |
| **排序** | 无排序或按字段排序 | 按相似度排序 |
| **适用场景** | 已知ID或条件查询 | 语义搜索、推荐 |

### 2. 过滤表达式语法

**比较运算符**：
```python
# 等于
filter='category == "tech"'

# 不等于
filter='category != "tech"'

# 大于/小于
filter='score > 50'
filter='score >= 50'
filter='score < 100'
filter='score <= 100'
```

**逻辑运算符**：
```python
# AND
filter='category == "tech" and score >= 50'

# OR
filter='category == "tech" or category == "science"'

# NOT
filter='not (category == "tech")'
```

**成员运算符**：
```python
# IN
filter='category in ["tech", "business", "science"]'

# NOT IN
filter='category not in ["health", "education"]'
```

**嵌套条件**：
```python
filter='(category == "tech" and score >= 70) or (category == "science" and score >= 80)'
```

### 3. 混合检索的优势

**优势**：
- 结合语义相似度和精确条件
- 提高检索准确性
- 减少无关结果
- 支持复杂业务逻辑

**性能影响**：
- 过滤条件越复杂，性能越低
- 建议为常用过滤字段创建索引
- 合理使用 limit 限制返回数量

### 4. 性能优化建议

**Query 优化**：
```python
# 1. 只返回需要的字段
client.query(
    collection_name,
    filter='category == "tech"',
    output_fields=["id", "title"],  # 不要返回 vector
    limit=100
)

# 2. 使用主键查询（最快）
client.query(collection_name, ids=[1, 2, 3])

# 3. 简化过滤条件
# 不推荐：复杂嵌套
filter='((a == 1 and b == 2) or (c == 3 and d == 4)) and e == 5'
# 推荐：简化逻辑
filter='category == "tech" and score >= 50'
```

**Search 优化**：
```python
# 1. 减小 limit
client.search(collection_name, data=[vector], limit=10)  # 而非 100

# 2. 使用简单过滤条件
filter='category == "tech"'  # 而非复杂嵌套

# 3. 考虑使用 Partition
client.search(
    collection_name,
    data=[vector],
    partition_names=["tech_partition"],  # 减少搜索范围
    limit=10
)
```

## 最佳实践

### 1. 选择合适的查询方式

```python
# 场景1：已知文档ID → 使用 Query
doc = client.query(collection_name, ids=[doc_id])

# 场景2：基于元数据查询 → 使用 Query
docs = client.query(
    collection_name,
    filter='author == "Alice" and published_year == 2024'
)

# 场景3：语义搜索 → 使用 Search
results = client.search(
    collection_name,
    data=[query_embedding],
    limit=10
)

# 场景4：语义搜索 + 元数据过滤 → 使用 Search + filter
results = client.search(
    collection_name,
    data=[query_embedding],
    filter='category == "tech" and score >= 70',
    limit=10
)
```

### 2. 过滤表达式最佳实践

```python
# ✅ 推荐：简洁明了
filter='category == "tech" and score >= 50'

# ❌ 不推荐：过于复杂
filter='((category == "tech" and score >= 50 and views > 1000) or (category == "science" and score >= 80 and views > 2000)) and published_year >= 2023 and is_featured == true'

# ✅ 推荐：拆分复杂查询
# 先用简单条件过滤，再在应用层进一步筛选
results = client.query(
    collection_name,
    filter='category in ["tech", "science"] and score >= 50',
    limit=100
)
# 应用层进一步过滤
filtered = [r for r in results if r['views'] > 1000 and r['is_featured']]
```

### 3. 字段类型注意事项

```python
# 字符串：必须加引号
filter='category == "tech"'  # ✅
filter='category == tech'    # ❌

# 数字：不加引号
filter='score >= 50'         # ✅
filter='score >= "50"'       # ❌

# 布尔：使用 true/false
filter='is_featured == true'  # ✅
filter='is_featured == True'  # ❌
filter='is_featured == 1'     # ❌
```

## 常见问题

### Q1: Query 和 Search 可以同时使用吗？

不能直接同时使用，但可以组合：
```python
# 方案1：先 Query 后 Search
ids = [doc['id'] for doc in client.query(collection_name, filter='category == "tech"')]
results = client.search(collection_name, data=[vector], filter=f'id in {ids}')

# 方案2：使用 Search + filter（推荐）
results = client.search(
    collection_name,
    data=[vector],
    filter='category == "tech"',
    limit=10
)
```

### Q2: 过滤表达式支持正则表达式吗？

Milvus 2.6 不直接支持正则表达式，但可以使用：
- `like` 运算符（部分匹配）
- 应用层正则过滤

### Q3: 如何优化复杂过滤查询的性能？

1. 为常用过滤字段创建标量索引
2. 使用 Partition 减少搜索范围
3. 简化过滤条件
4. 使用 limit 限制返回数量
