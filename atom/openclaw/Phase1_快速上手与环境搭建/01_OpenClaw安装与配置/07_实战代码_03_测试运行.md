/**
 * OpenClaw 测试与验证脚本
 *
 * 功能：
 * 1. Gateway 状态检测
 * 2. 健康检查
 * 3. 测试消息发送
 * 4. 日志查看
 * 5. 故障诊断
 *
 * 使用方法：
 *   ts-node test-openclaw.ts
 */

import { execSync } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import * as https from 'https';

// ===== 类型定义 =====

interface HealthCheckResult {
  healthy: boolean;
  status?: string;
  version?: string;
  uptime?: number;
  error?: string;
}

interface TestResult {
  name: string;
  passed: boolean;
  duration: number;
  error?: string;
  details?: any;
}

interface DiagnosticReport {
  timestamp: string;
  environment: {
    platform: string;
    nodeVersion: string;
    openclawVersion?: string;
  };
  gateway: {
    running: boolean;
    healthy: boolean;
    port: number;
  };
  configuration: {
    exists: boolean;
    valid: boolean;
    channels: string[];
  };
  tests: TestResult[];
  recommendations: string[];
}

// ===== Gateway 状态检测 =====

/**
 * 检查 Gateway 是否正在运行
 */
function isGatewayRunning(): boolean {
  try {
    if (os.platform() === 'win32') {
      execSync('tasklist | findstr "openclaw"', { stdio: 'ignore' });
    } else {
      execSync('pgrep -f "openclaw gateway"', { stdio: 'ignore' });
    }
    return true;
  } catch {
    return false;
  }
}

/**
 * 获取 Gateway 进程信息
 */
function getGatewayProcessInfo(): { pid: number; uptime: number } | null {
  try {
    if (os.platform() === 'win32') {
      // Windows: 使用 tasklist
      const output = execSync('tasklist /FI "IMAGENAME eq node.exe" /FO CSV', {
        encoding: 'utf-8'
      });
      // 解析输出获取 PID
      const lines = output.split('\n');
      for (const line of lines) {
        if (line.includes('openclaw')) {
          const match = line.match(/"(\d+)"/);
          if (match) {
            return { pid: parseInt(match[1]), uptime: 0 };
          }
        }
      }
    } else {
      // Unix/Linux/macOS: 使用 ps
      const output = execSync('ps aux | grep "openclaw gateway" | grep -v grep', {
        encoding: 'utf-8'
      });

      const match = output.match(/\s+(\d+)\s+/);
      if (match) {
        const pid = parseInt(match[1]);

        // 获取进程启动时间
        const startTime = execSync(`ps -p ${pid} -o lstart=`, {
          encoding: 'utf-8'
        }).trim();

        const start = new Date(startTime);
        const uptime = Math.floor((Date.now() - start.getTime()) / 1000);

        return { pid, uptime };
      }
    }
  } catch {
    // 进程不存在
  }

  return null;
}

// ===== 健康检查 =====

/**
 * 执行健康检查
 */
async function healthCheck(port: number = 3000): Promise<HealthCheckResult> {
  return new Promise((resolve) => {
    const url = `http://localhost:${port}/health`;

    https.get(url, (res) => {
      let data = '';

      res.on('data', (chunk) => {
        data += chunk;
      });

      res.on('end', () => {
        try {
          const health = JSON.parse(data);
          resolve({
            healthy: health.status === 'ok',
            status: health.status,
            version: health.version,
            uptime: health.uptime
          });
        } catch (error) {
          resolve({
            healthy: false,
            error: '无法解析健康检查响应'
          });
        }
      });
    }).on('error', (error) => {
      resolve({
        healthy: false,
        error: `连接失败: ${error.message}`
      });
    });
  });
}

// ===== 测试消息发送 =====

/**
 * 发送测试消息
 */
async function sendTestMessage(
  port: number = 3000,
  model: string = 'claude-3-5-sonnet-20241022'
): Promise<TestResult> {
  const startTime = Date.now();

  return new Promise((resolve) => {
    const postData = JSON.stringify({
      model,
      max_tokens: 100,
      messages: [
        {
          role: 'user',
          content: 'Hello! Please respond with "OpenClaw is working!"'
        }
      ]
    });

    const options = {
      hostname: 'localhost',
      port,
      path: '/v1/messages',
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Content-Length': Buffer.byteLength(postData)
      }
    };

    const req = https.request(options, (res) => {
      let data = '';

      res.on('data', (chunk) => {
        data += chunk;
      });

      res.on('end', () => {
        const duration = Date.now() - startTime;

        try {
          const response = JSON.parse(data);

          if (response.content && response.content[0]?.text) {
            resolve({
              name: 'Test Message',
              passed: true,
              duration,
              details: {
                model,
                response: response.content[0].text
              }
            });
          } else {
            resolve({
              name: 'Test Message',
              passed: false,
              duration,
              error: '响应格式不正确'
            });
          }
        } catch (error: any) {
          resolve({
            name: 'Test Message',
            passed: false,
            duration,
            error: `解析响应失败: ${error.message}`
          });
        }
      });
    });

    req.on('error', (error) => {
      resolve({
        name: 'Test Message',
        passed: false,
        duration: Date.now() - startTime,
        error: `请求失败: ${error.message}`
      });
    });

    req.write(postData);
    req.end();
  });
}

// ===== 配置验证 =====

/**
 * 验证配置文件
 */
function validateConfiguration(): {
  exists: boolean;
  valid: boolean;
  channels: string[];
  errors: string[];
} {
  const configPath = path.join(os.homedir(), '.openclaw', 'settings.json');

  if (!fs.existsSync(configPath)) {
    return {
      exists: false,
      valid: false,
      channels: [],
      errors: ['配置文件不存在']
    };
  }

  try {
    const configData = fs.readFileSync(configPath, 'utf-8');
    const config = JSON.parse(configData);

    const errors: string[] = [];
    const channels: string[] = [];

    // 验证 gateway 配置
    if (!config.gateway) {
      errors.push('缺少 gateway 配置');
    } else {
      if (!config.gateway.port) {
        errors.push('缺少 gateway.port');
      }
    }

    // 验证 channels 配置
    if (!config.channels || Object.keys(config.channels).length === 0) {
      errors.push('缺少 channels 配置');
    } else {
      for (const [name, channelConfig] of Object.entries(config.channels)) {
        if ((channelConfig as any).enabled) {
          channels.push(name);

          if (!(channelConfig as any).apiKey) {
            errors.push(`通道 ${name} 缺少 apiKey`);
          }
        }
      }
    }

    return {
      exists: true,
      valid: errors.length === 0,
      channels,
      errors
    };
  } catch (error: any) {
    return {
      exists: true,
      valid: false,
      channels: [],
      errors: [`配置文件格式错误: ${error.message}`]
    };
  }
}

// ===== 日志查看 =====

/**
 * 读取最新日志
 */
function readRecentLogs(lines: number = 50): string[] {
  const logPath = path.join(os.homedir(), '.openclaw', 'gateway.log');

  if (!fs.existsSync(logPath)) {
    return ['日志文件不存在'];
  }

  try {
    const logData = fs.readFileSync(logPath, 'utf-8');
    const logLines = logData.split('\n').filter(line => line.trim());

    return logLines.slice(-lines);
  } catch (error: any) {
    return [`读取日志失败: ${error.message}`];
  }
}

/**
 * 分析日志中的错误
 */
function analyzeLogsForErrors(): { errors: string[]; warnings: string[] } {
  const logs = readRecentLogs(100);
  const errors: string[] = [];
  const warnings: string[] = [];

  for (const log of logs) {
    try {
      const entry = JSON.parse(log);

      if (entry.level === 'error') {
        errors.push(entry.message || log);
      } else if (entry.level === 'warn') {
        warnings.push(entry.message || log);
      }
    } catch {
      // 非 JSON 格式的日志
      if (log.toLowerCase().includes('error')) {
        errors.push(log);
      } else if (log.toLowerCase().includes('warn')) {
        warnings.push(log);
      }
    }
  }

  return { errors, warnings };
}

// ===== 完整测试套件 =====

/**
 * 运行完整测试套件
 */
async function runTestSuite(): Promise<TestResult[]> {
  const results: TestResult[] = [];

  console.log('🧪 Running test suite...\n');

  // 1. 测试：OpenClaw 命令可用
  console.log('Test 1: OpenClaw command availability');
  const startTime1 = Date.now();
  try {
    execSync('openclaw --version', { stdio: 'ignore' });
    results.push({
      name: 'OpenClaw Command',
      passed: true,
      duration: Date.now() - startTime1
    });
    console.log('✅ Passed\n');
  } catch (error: any) {
    results.push({
      name: 'OpenClaw Command',
      passed: false,
      duration: Date.now() - startTime1,
      error: 'openclaw 命令不可用'
    });
    console.log('❌ Failed\n');
  }

  // 2. 测试：配置文件有效
  console.log('Test 2: Configuration validity');
  const startTime2 = Date.now();
  const configValidation = validateConfiguration();
  results.push({
    name: 'Configuration',
    passed: configValidation.valid,
    duration: Date.now() - startTime2,
    error: configValidation.errors.join(', '),
    details: { channels: configValidation.channels }
  });
  console.log(configValidation.valid ? '✅ Passed\n' : '❌ Failed\n');

  // 3. 测试：Gateway 运行状态
  console.log('Test 3: Gateway running');
  const startTime3 = Date.now();
  const running = isGatewayRunning();
  results.push({
    name: 'Gateway Running',
    passed: running,
    duration: Date.now() - startTime3,
    error: running ? undefined : 'Gateway 未运行'
  });
  console.log(running ? '✅ Passed\n' : '❌ Failed\n');

  // 4. 测试：健康检查
  if (running) {
    console.log('Test 4: Health check');
    const health = await healthCheck();
    results.push({
      name: 'Health Check',
      passed: health.healthy,
      duration: 0,
      error: health.error,
      details: { status: health.status, version: health.version }
    });
    console.log(health.healthy ? '✅ Passed\n' : '❌ Failed\n');

    // 5. 测试：发送消息
    if (health.healthy && configValidation.channels.length > 0) {
      console.log('Test 5: Send test message');
      const testMessage = await sendTestMessage(3000, configValidation.channels[0]);
      results.push(testMessage);
      console.log(testMessage.passed ? '✅ Passed\n' : '❌ Failed\n');
    }
  }

  return results;
}

// ===== 诊断报告 =====

/**
 * 生成诊断报告
 */
async function generateDiagnosticReport(): Promise<DiagnosticReport> {
  console.log('🔍 Generating diagnostic report...\n');

  // 环境信息
  const nodeVersion = process.version;
  let openclawVersion: string | undefined;
  try {
    openclawVersion = execSync('openclaw --version', {
      encoding: 'utf-8'
    }).trim();
  } catch {
    // OpenClaw 未安装
  }

  // Gateway 状态
  const running = isGatewayRunning();
  const processInfo = getGatewayProcessInfo();
  let healthy = false;
  let port = 3000;

  if (running) {
    const health = await healthCheck();
    healthy = health.healthy;
  }

  // 配置验证
  const configValidation = validateConfiguration();

  // 运行测试
  const tests = await runTestSuite();

  // 日志分析
  const logAnalysis = analyzeLogsForErrors();

  // 生成建议
  const recommendations: string[] = [];

  if (!openclawVersion) {
    recommendations.push('安装 OpenClaw: npm install -g openclaw@latest');
  }

  if (!configValidation.exists) {
    recommendations.push('运行配置向导: openclaw onboard');
  } else if (!configValidation.valid) {
    recommendations.push('修复配置文件: ~/.openclaw/settings.json');
    recommendations.push(`配置错误: ${configValidation.errors.join(', ')}`);
  }

  if (!running) {
    recommendations.push('启动 Gateway: openclaw gateway start');
  } else if (!healthy) {
    recommendations.push('检查 Gateway 日志: tail -f ~/.openclaw/gateway.log');
    recommendations.push('重启 Gateway: openclaw gateway stop && openclaw gateway start');
  }

  if (logAnalysis.errors.length > 0) {
    recommendations.push(`发现 ${logAnalysis.errors.length} 个错误，请检查日志`);
  }

  return {
    timestamp: new Date().toISOString(),
    environment: {
      platform: os.platform(),
      nodeVersion,
      openclawVersion
    },
    gateway: {
      running,
      healthy,
      port
    },
    configuration: {
      exists: configValidation.exists,
      valid: configValidation.valid,
      channels: configValidation.channels
    },
    tests,
    recommendations
  };
}

/**
 * 打印诊断报告
 */
function printDiagnosticReport(report: DiagnosticReport): void {
  console.log('');
  console.log('='.repeat(60));
  console.log('OpenClaw Diagnostic Report');
  console.log('='.repeat(60));
  console.log('');

  // 时间戳
  console.log(`Generated: ${report.timestamp}\n`);

  // 环境信息
  console.log('Environment:');
  console.log(`  Platform: ${report.environment.platform}`);
  console.log(`  Node.js: ${report.environment.nodeVersion}`);
  console.log(`  OpenClaw: ${report.environment.openclawVersion || 'Not installed'}\n`);

  // Gateway 状态
  console.log('Gateway:');
  console.log(`  Running: ${report.gateway.running ? '✅' : '❌'}`);
  console.log(`  Healthy: ${report.gateway.healthy ? '✅' : '❌'}`);
  console.log(`  Port: ${report.gateway.port}\n`);

  // 配置状态
  console.log('Configuration:');
  console.log(`  Exists: ${report.configuration.exists ? '✅' : '❌'}`);
  console.log(`  Valid: ${report.configuration.valid ? '✅' : '❌'}`);
  console.log(`  Channels: ${report.configuration.channels.join(', ') || 'None'}\n`);

  // 测试结果
  console.log('Test Results:');
  for (const test of report.tests) {
    console.log(`  ${test.passed ? '✅' : '❌'} ${test.name} (${test.duration}ms)`);
    if (test.error) {
      console.log(`     Error: ${test.error}`);
    }
  }
  console.log('');

  // 建议
  if (report.recommendations.length > 0) {
    console.log('Recommendations:');
    for (const rec of report.recommendations) {
      console.log(`  • ${rec}`);
    }
    console.log('');
  }

  console.log('='.repeat(60));
}

// ===== 主函数 =====

/**
 * 主测试流程
 */
async function main(): Promise<void> {
  const args = process.argv.slice(2);

  if (args.includes('--help') || args.includes('-h')) {
    console.log('OpenClaw Test & Diagnostic Tool\n');
    console.log('Usage:');
    console.log('  ts-node test-openclaw.ts [options]\n');
    console.log('Options:');
    console.log('  --test, -t       Run test suite only');
    console.log('  --health, -h     Run health check only');
    console.log('  --logs, -l       View recent logs');
    console.log('  --diagnose, -d   Generate diagnostic report (default)');
    console.log('  --help           Show this help message');
    return;
  }

  if (args.includes('--test') || args.includes('-t')) {
    // 只运行测试
    await runTestSuite();
  } else if (args.includes('--health')) {
    // 只运行健康检查
    const health = await healthCheck();
    console.log('Health Check:');
    console.log(`  Status: ${health.status || 'unknown'}`);
    console.log(`  Healthy: ${health.healthy ? '✅' : '❌'}`);
    if (health.version) console.log(`  Version: ${health.version}`);
    if (health.error) console.log(`  Error: ${health.error}`);
  } else if (args.includes('--logs') || args.includes('-l')) {
    // 查看日志
    const logs = readRecentLogs(50);
    console.log('Recent Logs:\n');
    logs.forEach(log => console.log(log));
  } else {
    // 生成完整诊断报告（默认）
    const report = await generateDiagnosticReport();
    printDiagnosticReport(report);

    // 保存报告到文件
    const reportPath = path.join(os.homedir(), '.openclaw', 'diagnostic-report.json');
    fs.writeFileSync(reportPath, JSON.stringify(report, null, 2));
    console.log(`\nReport saved to: ${reportPath}\n`);
  }
}

// 运行主函数
if (require.main === module) {
  main().catch((error) => {
    console.error('Error:', error.message);
    process.exit(1);
  });
}

// 导出函数供其他模块使用
export {
  isGatewayRunning,
  healthCheck,
  sendTestMessage,
  validateConfiguration,
  readRecentLogs,
  runTestSuite,
  generateDiagnosticReport
};

## 实际应用场景

### 场景 1：CI/CD 健康检查

```yaml
# .github/workflows/health-check.yml
name: Health Check

on:
  schedule:
    - cron: '*/5 * * * *'  # 每 5 分钟

jobs:
  health-check:
    runs-on: ubuntu-latest
    steps:
      - name: Health Check
        run: npx tsx test-openclaw.ts --health

      - name: Alert on Failure
        if: failure()
        uses: actions/github-script@v6
        with:
          script: |
            github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: 'OpenClaw Gateway Health Check Failed',
              body: 'Gateway is not healthy. Please investigate.'
            })
```

### 场景 2：监控脚本

```bash
#!/bin/bash
# monitor-openclaw.sh

while true; do
  npx tsx test-openclaw.ts --health > /dev/null 2>&1

  if [ $? -ne 0 ]; then
    echo "$(date): Gateway unhealthy, restarting..."
    openclaw gateway stop
    sleep 5
    openclaw gateway start --daemon
  fi

  sleep 60  # 每分钟检查一次
done
```
