## 概述

**本场景涵盖**:
- DirectoryLoader 基础用法
- 文件类型过滤
- 并行批量加载
- 错误处理与容错

- 大规模文档处理策略

### 自定义并行加载

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
from pathlib import Path
from typing import List

def load_single_file(file_path: Path) -> List[Document]:
    """加载单个文件"""
    suffix = file_path.suffix.lower()

    try:
        if suffix == ".txt":
            from langchain_community.document_loaders import TextLoader
            loader = TextLoader(str(file_path))
        elif suffix == ".pdf":
            from langchain_community.document_loaders import PyPDFLoader
            loader = PyPDFLoader(str(file_path))
        elif suffix in [".docx", ".doc"]:
            from langchain_community.document_loaders import Docx2txtLoader
            loader = Docx2txtLoader(str(file_path))
        else:
            from langchain_community.document_loaders import UnstructuredFileLoader
            loader = UnstructuredFileLoader(str(file_path))

        return loader.load()
    except Exception as e:
        print(f"加载 {file_path} 失败: {e}")
        return []

def parallel_load_directory(
    directory: str,
    extensions: List[str] = [".txt", ".pdf", ".docx"],
    max_workers: int = 4
) -> List[Document]:
    """并行加载目录中的所有文档"""
    # 查找所有匹配的文件
    files = []
    for ext in extensions:
        files.extend(Path(directory).rglob(f"*{ext}"))

    print(f"找到 {len(files)} 个文件")

    all_documents = []

    # 使用线程池并行加载
    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        # 提交所有任务
        future_to_file = {
            executor.submit(load_single_file, file): file
            for file in files
        }

        # 使用 tqdm 显示进度
        from tqdm import tqdm
        for future in tqdm(as_completed(future_to_file), total=len(files)):
            file = future_to_file[future]
            try:
                docs = future.result()
                all_documents.extend(docs)
            except Exception as e:
                print(f"处理 {file} 时出错: {e}")

    return all_documents

# 使用
documents = parallel_load_directory("./docs", max_workers=4)
print(f"并行加载了 {len(documents)} 个文档")
```

### 自定义错误处理

```python
import logging
from typing import List, Tuple

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def load_with_error_tracking(
    directory: str,
    extensions: List[str] = [".txt", ".pdf", ".docx"]
) -> Tuple[List[Document], List[str]]:
    """加载文档并跟踪错误"""
    files = []
    for ext in extensions:
        files.extend(Path(directory).rglob(f"*{ext}"))

    all_documents = []
    failed_files = []

    for file in tqdm(files, desc="加载文档"):
        try:
            docs = load_single_file(file)
            all_documents.extend(docs)
            logger.info(f"✓ 成功加载 {file.name}")
        except Exception as e:
            failed_files.append(str(file))
            logger.error(f"✗ 加载 {file.name} 失败: {e}")

    # 生成报告
    print(f"\n加载报告:")
    print(f"- 成功: {len(files) - len(failed_files)}/{len(files)}")
    print(f"- 失败: {len(failed_files)}/{len(files)}")

    if failed_files:
        print(f"\n失败的文件:")
        for file in failed_files:
            print(f"  - {file}")

    return all_documents, failed_files

# 使用
documents, failed_files = load_with_error_tracking("./docs")
```

### 使用 tqdm 进度条

```python
from tqdm import tqdm

def load_with_progress(directory: str) -> List[Document]:
    """带进度条的文档加载"""
    files = list(Path(directory).rglob("*.{txt,pdf,docx}"))
    all_documents = []

    # 使用 tqdm 显示进度
    for file in tqdm(files, desc="加载文档", unit="file"):
        try:
            docs = load_single_file(file)
            all_documents.extend(docs)
        except Exception as e:
            tqdm.write(f"加载 {file} 失败: {e}")

    return all_documents

# 使用
documents = load_with_progress("./docs")
```

### 详细进度信息

```python
from tqdm import tqdm
import time

def load_with_detailed_progress(directory: str) -> List[Document]:
    """带详细进度信息的文档加载"""
    files = list(Path(directory).rglob("*.*"))
    all_documents = []

    # 创建进度条
    pbar = tqdm(total=len(files), desc="加载文档")

    for file in files:
        start_time = time.time()
        try:
            docs = load_single_file(file)
            all_documents.extend(docs)

            # 更新进度条描述
            elapsed = time.time() - start_time
            pbar.set_postfix({
                "文件": file.name[:20],
                "文档数": len(docs),
                "耗时": f"{elapsed:.2f}s"
            })
        except Exception as e:
            pbar.write(f"✗ {file.name}: {e}")

        pbar.update(1)

    pbar.close()
    return all_documents

# 使用
documents = load_with_detailed_progress("./docs")
```

### 分批处理

```python
from typing import Iterator

def batch_load_documents(
    directory: str,
    batch_size: int = 100
) -> Iterator[List[Document]]:
    """分批加载文档，避免内存溢出"""
    files = list(Path(directory).rglob("*.*"))

    for i in range(0, len(files), batch_size):
        batch_files = files[i:i + batch_size]
        batch_documents = []

        for file in batch_files:
            try:
                docs = load_single_file(file)
                batch_documents.extend(docs)
            except Exception as e:
                print(f"加载 {file} 失败: {e}")

        yield batch_documents

# 使用 - 分批处理
for batch_idx, batch_docs in enumerate(batch_load_documents("./docs", batch_size=100)):
    print(f"批次 {batch_idx + 1}: 加载了 {len(batch_docs)} 个文档")

    # 处理这一批文档（例如：向量化、存储）
    # process_batch(batch_docs)
```

def stream_load_documents(directory: str) -> Iterator[Document]:
    """流式加载文档，逐个返回"""
    files = Path(directory).rglob("*.*")

    for file in files:
        try:
            docs = load_single_file(file)
            for doc in docs:
                yield doc
        except Exception as e:

## 场景 7: RAG 管道集成

### 完整的批量文档 RAG 系统

```python
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_community.vectorstores import Chroma
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.chains import RetrievalQA

def build_rag_from_directory(
    directory: str,
    collection_name: str = "documents"
) -> RetrievalQA:
    """从目录构建完整的 RAG 系统"""

    # 1. 批量加载文档
    print("1. 批量加载文档...")
    documents = parallel_load_directory(directory, max_workers=4)
    print(f"   加载了 {len(documents)} 个文档")

    # 2. 文本分块
    print("2. 文本分块...")
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=1000,
        chunk_overlap=200
    )
    chunks = text_splitter.split_documents(documents)
    print(f"   分块后: {len(chunks)} 个 chunks")

    # 3. 创建向量存储
    print("3. 创建向量存储...")
    embeddings = OpenAIEmbeddings()
    vectorstore = Chroma.from_documents(
        documents=chunks,
        embedding=embeddings,
        collection_name=collection_name
    )

    # 4. 创建检索器
    print("4. 创建检索器...")
    retriever = vectorstore.as_retriever(
        search_type="similarity",
        search_kwargs={"k": 3}
    )

    # 5. 创建 QA 链
    print("5. 创建 QA 链...")
    llm = ChatOpenAI(model="gpt-4", temperature=0)
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        chain_type="stuff",
        retriever=retriever,
        return_source_documents=True
    )

    print("✓ RAG 系统构建完成")
    return qa_chain

# 使用
qa_chain = build_rag_from_directory("./docs")

# 查询
query = "文档中提到的主要内容是什么？"
result = qa_chain.invoke({"query": query})

print(f"问题: {query}")
print(f"答案: {result['result']}")
print(f"\n来源文档:")
for doc in result['source_documents']:
    print(f"- {doc.metadata.get('source', 'unknown')}")
```

## 场景 8: 增量更新

### 检测新文件并增量加载

```python
import json
from datetime import datetime

class IncrementalDocumentLoader:
    """增量文档加载器"""

    def __init__(self, directory: str, state_file: str = ".loader_state.json"):
        self.directory = directory
        self.state_file = state_file
        self.loaded_files = self._load_state()

    def _load_state(self) -> set:
        """加载已处理文件的状态"""
        if Path(self.state_file).exists():
            with open(self.state_file, 'r') as f:
                return set(json.load(f))
        return set()

    def _save_state(self):
        """保存已处理文件的状态"""
        with open(self.state_file, 'w') as f:
            json.dump(list(self.loaded_files), f)

    def load_new_documents(self) -> List[Document]:
        """只加载新文档"""
        all_files = set(str(f) for f in Path(self.directory).rglob("*.*"))
        new_files = all_files - self.loaded_files

        print(f"发现 {len(new_files)} 个新文件")

        new_documents = []
        for file_path in tqdm(new_files, desc="加载新文档"):
            try:
                docs = load_single_file(Path(file_path))
                new_documents.extend(docs)
                self.loaded_files.add(file_path)
            except Exception as e:
                print(f"加载 {file_path} 失败: {e}")

        # 保存状态
        self._save_state()

        return new_documents

# 使用
loader = IncrementalDocumentLoader("./docs")

# 第一次加载
documents = loader.load_new_documents()
print(f"首次加载: {len(documents)} 个文档")

# 后续增量加载（只加载新文件）
new_documents = loader.load_new_documents()
print(f"增量加载: {len(new_documents)} 个新文档")
```

## 总结

### 批量加载策略对比

| 策略 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| DirectoryLoader | 简单易用 | 功能有限 | 小规模文档 |
| 并行加载 | 速度快 | 资源占用高 | 中等规模文档 |
| 分批处理 | 内存友好 | 实现复杂 | 大规模文档 |
| 流式处理 | 内存占用最小 | 无法批量操作 | 超大规模文档 |
| 增量更新 | 避免重复加载 | 需要状态管理 | 持续更新场景 |

### 最佳实践

1. **选择合适的策略**:
   - 小规模（<1000 文件）→ DirectoryLoader
   - 中等规模（1000-10000 文件）→ 并行加载
   - 大规模（>10000 文件）→ 分批/流式处理

2. **错误处理**:
   - 使用 `silent_errors=True` 跳过错误文件
   - 记录失败文件用于后续处理
   - 添加重试机制

3. **性能优化**:
   - 使用多线程并行加载
   - 控制并发数避免资源耗尽
   - 使用进度条跟踪进度

4. **内存管理**:
   - 大规模文档使用分批处理
   - 流式处理避免内存溢出
   - 及时释放不需要的对象
