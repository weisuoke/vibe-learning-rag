## 实战1：InMemoryChatMessageHistory 基础用法

### 代码实现

```python
"""
实战1：InMemoryChatMessageHistory 基础用法
演示如何手动管理对话历史
"""

from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage

def demo_basic_history():
    """演示基础的历史管理"""
    print("=== 实战1：InMemoryChatMessageHistory 基础用法 ===\n")

    # 1. 创建历史存储
    history = InMemoryChatMessageHistory()
    print("1. 创建历史存储")

    # 2. 添加系统消息
    history.add_message(SystemMessage(content="You are a helpful assistant."))
    print("2. 添加系统消息")

    # 3. 添加用户消息
    history.add_message(HumanMessage(content="Hello! My name is Alice."))
    print("3. 添加用户消息: 'Hello! My name is Alice.'")

    # 4. 添加 AI 响应
    history.add_message(AIMessage(content="Hello Alice! How can I help you today?"))
    print("4. 添加 AI 响应: 'Hello Alice! How can I help you today?'")

    # 5. 继续对话
    history.add_message(HumanMessage(content="What's my name?"))
    print("5. 添加用户消息: 'What's my name?'")

    history.add_message(AIMessage(content="Your name is Alice."))
    print("6. 添加 AI 响应: 'Your name is Alice.'")

    # 6. 获取所有消息
    print("\n7. 获取所有消息:")
    for i, msg in enumerate(history.messages, 1):
        print(f"   [{i}] {msg.__class__.__name__}: {msg.content}")

    # 7. 获取消息数量
    print(f"\n8. 消息总数: {len(history.messages)}")

    # 8. 清空历史
    history.clear()
    print(f"9. 清空历史后消息数: {len(history.messages)}")

if __name__ == "__main__":
    demo_basic_history()
```

## 实战2：RunnableWithMessageHistory 集成

### 代码实现

```python
"""
实战2：RunnableWithMessageHistory 集成
演示如何使用 RunnableWithMessageHistory 自动管理历史
"""

from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_openai import ChatOpenAI
from dotenv import load_dotenv

# 加载环境变量
load_dotenv()

def demo_runnable_with_history():
    """演示 RunnableWithMessageHistory 的基础用法"""
    print("=== 实战2：RunnableWithMessageHistory 集成 ===\n")

    # 1. 创建存储字典
    store = {}
    print("1. 创建存储字典")

    # 2. 定义获取历史的函数
    def get_session_history(session_id: str) -> InMemoryChatMessageHistory:
        if session_id not in store:
            print(f"   创建新会话: {session_id}")
            store[session_id] = InMemoryChatMessageHistory()
        else:
            print(f"   使用现有会话: {session_id}")
        return store[session_id]

    # 3. 创建聊天模型
    chat_model = ChatOpenAI(model="gpt-4", temperature=0)
    print("2. 创建聊天模型: gpt-4")

    # 4. 包装模型
    conversation = RunnableWithMessageHistory(
        chat_model,
        get_session_history,
    )
    print("3. 包装模型为 RunnableWithMessageHistory")

    # 5. 第一轮对话
    print("\n4. 第一轮对话:")
    config = {"configurable": {"session_id": "user_alice"}}
    response1 = conversation.invoke("Hi, I'm Alice. I love pizza.", config=config)
    print(f"   用户: Hi, I'm Alice. I love pizza.")
    print(f"   AI: {response1.content}")

    # 6. 第二轮对话
    print("\n5. 第二轮对话:")
    response2 = conversation.invoke("What's my name?", config=config)
    print(f"   用户: What's my name?")
    print(f"   AI: {response2.content}")

    # 7. 第三轮对话
    print("\n6. 第三轮对话:")
    response3 = conversation.invoke("What food do I like?", config=config)
    print(f"   用户: What food do I like?")
    print(f"   AI: {response3.content}")

    # 8. 查看历史
    print("\n7. 查看历史:")
    history = store["user_alice"]
    print(f"   消息总数: {len(history.messages)}")
    for i, msg in enumerate(history.messages, 1):
        print(f"   [{i}] {msg.__class__.__name__}: {msg.content[:50]}...")

if __name__ == "__main__":
    demo_runnable_with_history()
```

## 实战3：Session 管理实现

### 代码实现

```python
"""
实战3：Session 管理实现
演示如何管理多个用户的对话会话
"""

from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_openai import ChatOpenAI
from dotenv import load_dotenv

load_dotenv()

def demo_session_management():
    """演示多用户 Session 管理"""
    print("=== 实战3：Session 管理实现 ===\n")

    # 1. 创建存储字典
    store = {}

    # 2. 定义获取历史的函数
    def get_session_history(session_id: str) -> InMemoryChatMessageHistory:
        if session_id not in store:
            store[session_id] = InMemoryChatMessageHistory()
        return store[session_id]

    # 3. 创建对话系统
    conversation = RunnableWithMessageHistory(
        ChatOpenAI(model="gpt-4", temperature=0),
        get_session_history,
    )

    # 4. 用户1的对话
    print("1. 用户1 (Alice) 的对话:")
    config_alice = {"configurable": {"session_id": "user_alice"}}

    response1 = conversation.invoke("Hi, I'm Alice. I'm 25 years old.", config_alice)
    print(f"   Alice: Hi, I'm Alice. I'm 25 years old.")
    print(f"   AI: {response1.content}")

    response2 = conversation.invoke("What's my age?", config_alice)
    print(f"   Alice: What's my age?")
    print(f"   AI: {response2.content}")

    # 5. 用户2的对话
    print("\n2. 用户2 (Bob) 的对话:")
    config_bob = {"configurable": {"session_id": "user_bob"}}

    response3 = conversation.invoke("Hi, I'm Bob. I'm 30 years old.", config_bob)
    print(f"   Bob: Hi, I'm Bob. I'm 30 years old.")
    print(f"   AI: {response3.content}")

    response4 = conversation.invoke("What's my age?", config_bob)
    print(f"   Bob: What's my age?")
    print(f"   AI: {response4.content}")

    # 6. 用户1继续对话
    print("\n3. 用户1 (Alice) 继续对话:")
    response5 = conversation.invoke("What's my name and age?", config_alice)
    print(f"   Alice: What's my name and age?")
    print(f"   AI: {response5.content}")

    # 7. 查看所有会话
    print("\n4. 查看所有会话:")
    for session_id, history in store.items():
        print(f"   {session_id}: {len(history.messages)} 条消息")

if __name__ == "__main__":
    demo_session_management()
```

## 实战5：基础错误处理

### 代码实现

```python
"""
实战5：基础错误处理
演示如何处理常见错误
"""

from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_openai import ChatOpenAI
from dotenv import load_dotenv

load_dotenv()

def demo_error_handling():
    """演示基础错误处理"""
    print("=== 实战5：基础错误处理 ===\n")

    # 1. 创建对话系统
    store = {}

    def get_session_history(session_id: str) -> InMemoryChatMessageHistory:
        if session_id not in store:
            store[session_id] = InMemoryChatMessageHistory()
        return store[session_id]

    conversation = RunnableWithMessageHistory(
        ChatOpenAI(model="gpt-4", temperature=0),
        get_session_history,
    )

    # 2. 错误1：忘记传递 config
    print("1. 错误1：忘记传递 config")
    try:
        response = conversation.invoke("Hello!")
        print(f"   响应: {response.content}")
    except KeyError as e:
        print(f"   ❌ 错误: {e}")
        print(f"   原因: 必须传递 config 对象")

    # 3. 正确做法
    print("\n2. 正确做法：传递 config")
    try:
        config = {"configurable": {"session_id": "user_123"}}
        response = conversation.invoke("Hello!", config=config)
        print(f"   ✅ 响应: {response.content}")
    except Exception as e:
        print(f"   ❌ 错误: {e}")

    # 4. 错误2：Session ID 不唯一
    print("\n3. 错误2：Session ID 不唯一")
    config_shared = {"configurable": {"session_id": "default"}}

    # 用户1
    response1 = conversation.invoke("Hi, I'm Alice.", config_shared)
    print(f"   用户1: Hi, I'm Alice.")
    print(f"   AI: {response1.content}")

    # 用户2（使用相同的 session_id）
    response2 = conversation.invoke("Hi, I'm Bob.", config_shared)
    print(f"   用户2: Hi, I'm Bob.")
    print(f"   AI: {response2.content}")

    # 用户1继续对话
    response3 = conversation.invoke("What's my name?", config_shared)
    print(f"   用户1: What's my name?")
    print(f"   AI: {response3.content}")
    print(f"   ❌ 问题: AI 可能混淆了用户身份")

    # 5. 正确做法：每个用户使用唯一的 session_id
    print("\n4. 正确做法：每个用户使用唯一的 session_id")
    config_alice = {"configurable": {"session_id": "user_alice"}}
    config_bob = {"configurable": {"session_id": "user_bob"}}

    # 用户1
    response4 = conversation.invoke("Hi, I'm Alice.", config_alice)
    print(f"   用户1: Hi, I'm Alice.")
    print(f"   AI: {response4.content}")

    # 用户2
    response5 = conversation.invoke("Hi, I'm Bob.", config_bob)
    print(f"   用户2: Hi, I'm Bob.")
    print(f"   AI: {response5.content}")

    # 用户1继续对话
    response6 = conversation.invoke("What's my name?", config_alice)
    print(f"   用户1: What's my name?")
    print(f"   AI: {response6.content}")
    print(f"   ✅ 正确: AI 正确识别了用户身份")

if __name__ == "__main__":
    demo_error_handling()
```

## 完整示例：简单聊天机器人

### 代码实现

```python
"""
完整示例：简单聊天机器人
一个完整的命令行聊天机器人示例
"""

from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_openai import ChatOpenAI
from dotenv import load_dotenv

load_dotenv()

def create_chatbot():
    """创建聊天机器人"""
    # 1. 创建存储
    store = {}

    # 2. 定义获取历史的函数
    def get_session_history(session_id: str) -> InMemoryChatMessageHistory:
        if session_id not in store:
            store[session_id] = InMemoryChatMessageHistory()
        return store[session_id]

    # 3. 创建对话系统
    conversation = RunnableWithMessageHistory(
        ChatOpenAI(model="gpt-4", temperature=0.7),
        get_session_history,
    )

    return conversation, store

def main():
    """主函数"""
    print("=== 简单聊天机器人 ===")
    print("输入 'quit' 退出，输入 'history' 查看历史\n")

    # 创建聊天机器人
    conversation, store = create_chatbot()

    # 获取用户名
    username = input("请输入您的名字: ")
    config = {"configurable": {"session_id": f"user_{username}"}}

    print(f"\n欢迎, {username}! 开始对话吧。\n")

    # 对话循环
    while True:
        # 获取用户输入
        user_input = input(f"{username}: ")

        # 处理特殊命令
        if user_input.lower() == "quit":
            print("再见!")
            break

        if user_input.lower() == "history":
            history = store[f"user_{username}"]
            print(f"\n对话历史 ({len(history.messages)} 条消息):")
            for i, msg in enumerate(history.messages, 1):
                print(f"  [{i}] {msg.__class__.__name__}: {msg.content}")
            print()
            continue

        # 调用聊天机器人
        try:
            response = conversation.invoke(user_input, config=config)
            print(f"AI: {response.content}\n")
        except Exception as e:
            print(f"错误: {e}\n")

if __name__ == "__main__":
    main()
```

## 关键要点总结

### 1. InMemoryChatMessageHistory

**核心方法**：
- `add_message(message)` - 添加单条消息
- `messages` - 获取所有消息
- `clear()` - 清空历史

**适用场景**：
- 开发和测试
- 短期对话
- 不需要持久化

### 2. RunnableWithMessageHistory

**核心要素**：
- 存储字典 `store = {}`
- 获取函数 `get_session_history(session_id)`
- 配置对象 `config = {"configurable": {"session_id": "..."}}`

**优势**：
- 自动管理历史
- 支持多用户
- 统一接口

### 3. Session 管理

**关键点**：
- 每个用户使用唯一的 session_id
- 通过 config 传递 session_id
- 支持多用户并发

### 4. 错误处理

**常见错误**：
- 忘记传递 config
- Session ID 不唯一
- 没有错误处理

**解决方案**：
- 始终传递 config
- 使用唯一的 session_id
- 添加 try-except
