# 图可视化 - 实战代码: 场景4 RAG工作流可视化

## 场景描述

在开发 RAG (Retrieval-Augmented Generation) 工作流时,需要可视化复杂的检索、评分、生成流程,理解条件路由和循环逻辑。

## 完整代码

```python
"""
场景4: RAG 工作流可视化
演示: 可视化完整的 RAG 工作流,包括检索、评分、查询改写、生成等步骤
"""

from langgraph.graph import StateGraph, START, END
from typing import TypedDict, Literal
from IPython.display import Image, display, Markdown

# ===== 1. 定义状态 =====
class RAGState(TypedDict):
    """RAG 工作流状态"""
    query: str
    documents: list[str]
    relevant_docs: list[str]
    answer: str
    retry_count: int
    max_retries: int

# ===== 2. 定义节点函数 =====
def retrieve(state: RAGState) -> RAGState:
    """检索节点"""
    print(f"[Retrieve] 检索查询: {state['query']}")
    # 模拟检索
    documents = [
        f"文档1: {state['query']} 的基础信息",
        f"文档2: {state['query']} 的详细说明",
        f"文档3: {state['query']} 的实例",
        f"文档4: 无关文档"
    ]
    return {
        **state,
        "documents": documents
    }

def grade_documents(state: RAGState) -> RAGState:
    """文档评分节点"""
    print(f"[Grade] 评分 {len(state['documents'])} 个文档")
    # 模拟评分 - 过滤无关文档
    relevant_docs = [doc for doc in state['documents'] if "无关" not in doc]
    return {
        **state,
        "relevant_docs": relevant_docs
    }

def decide_to_generate(state: RAGState) -> Literal["generate", "transform_query"]:
    """决策: 是否有足够的相关文档"""
    if len(state['relevant_docs']) >= 2:
        print("[Decision] 有足够的相关文档,进入生成")
        return "generate"
    else:
        print("[Decision] 相关文档不足,需要改写查询")
        return "transform_query"

def transform_query(state: RAGState) -> RAGState:
    """查询改写节点"""
    print(f"[Transform] 改写查询: {state['query']}")
    new_query = f"{state['query']} (改写 {state['retry_count'] + 1})"
    return {
        **state,
        "query": new_query,
        "retry_count": state['retry_count'] + 1
    }

def check_retry_limit(state: RAGState) -> Literal["retrieve", "generate"]:
    """检查重试次数"""
    if state['retry_count'] < state['max_retries']:
        print(f"[Retry] 重试次数: {state['retry_count']}/{state['max_retries']}")
        return "retrieve"
    else:
        print(f"[Retry] 达到最大重试次数,强制生成")
        return "generate"

def generate(state: RAGState) -> RAGState:
    """生成答案节点"""
    doc_count = len(state['relevant_docs'])
    print(f"[Generate] 基于 {doc_count} 个文档生成答案")
    
    if doc_count > 0:
        answer = f"基于 {doc_count} 个相关文档,关于 '{state['query']}' 的答案是..."
    else:
        answer = f"抱歉,没有找到关于 '{state['query']}' 的相关信息"
    
    return {
        **state,
        "answer": answer
    }

# ===== 3. 创建 RAG 工作流 =====
print("=== 创建 RAG 工作流 ===")

workflow = StateGraph(RAGState)

# 添加节点
workflow.add_node("retrieve", retrieve)
workflow.add_node("grade_documents", grade_documents)
workflow.add_node("transform_query", transform_query)
workflow.add_node("generate", generate)

# 添加边
workflow.add_edge(START, "retrieve")
workflow.add_edge("retrieve", "grade_documents")

# 添加条件边: 评分后决定是否生成
workflow.add_conditional_edges(
    "grade_documents",
    decide_to_generate,
    {
        "generate": "generate",
        "transform_query": "transform_query"
    }
)

# 添加条件边: 查询改写后检查重试次数
workflow.add_conditional_edges(
    "transform_query",
    check_retry_limit,
    {
        "retrieve": "retrieve",
        "generate": "generate"
    }
)

workflow.add_edge("generate", END)

# 编译图
app = workflow.compile()
print("RAG 工作流创建完成")

# ===== 4. 可视化 RAG 工作流 =====
print("\n=== 可视化 RAG 工作流 ===")

# 在 Jupyter 中显示
display(Markdown("## RAG 工作流结构"))
display(Image(app.get_graph().draw_mermaid_png()))

# ===== 5. 分析图结构 =====
print("\n=== 分析图结构 ===")

graph = app.get_graph()
print(f"节点数量: {len(graph.nodes)}")
print(f"边数量: {len(graph.edges)}")

print("\n节点列表:")
for node in graph.nodes:
    print(f"  - {node.id}")

print("\n边列表:")
for edge in graph.edges:
    edge_type = "条件边" if edge.conditional else "普通边"
    label = f" ({edge.data})" if edge.data else ""
    print(f"  - {edge.source} -> {edge.target} [{edge_type}]{label}")

# ===== 6. 测试场景1: 成功检索 =====
print("\n=== 测试场景1: 成功检索 ===")

test_input_1 = {
    "query": "LangGraph",
    "documents": [],
    "relevant_docs": [],
    "answer": "",
    "retry_count": 0,
    "max_retries": 3
}

result_1 = app.invoke(test_input_1)
print(f"\n最终结果:")
print(f"  查询: {result_1['query']}")
print(f"  相关文档数: {len(result_1['relevant_docs'])}")
print(f"  答案: {result_1['answer']}")
print(f"  重试次数: {result_1['retry_count']}")

# ===== 7. 测试场景2: 需要查询改写 =====
print("\n=== 测试场景2: 需要查询改写 ===")

# 修改 retrieve 函数模拟返回少量文档
def retrieve_few(state: RAGState) -> RAGState:
    """检索节点 - 返回少量文档"""
    print(f"[Retrieve] 检索查询: {state['query']}")
    documents = [f"文档1: {state['query']} 的信息"]
    return {**state, "documents": documents}

# 重新创建工作流
workflow_2 = StateGraph(RAGState)
workflow_2.add_node("retrieve", retrieve_few)
workflow_2.add_node("grade_documents", grade_documents)
workflow_2.add_node("transform_query", transform_query)
workflow_2.add_node("generate", generate)

workflow_2.add_edge(START, "retrieve")
workflow_2.add_edge("retrieve", "grade_documents")
workflow_2.add_conditional_edges(
    "grade_documents",
    decide_to_generate,
    {"generate": "generate", "transform_query": "transform_query"}
)
workflow_2.add_conditional_edges(
    "transform_query",
    check_retry_limit,
    {"retrieve": "retrieve", "generate": "generate"}
)
workflow_2.add_edge("generate", END)

app_2 = workflow_2.compile()

test_input_2 = {
    "query": "复杂查询",
    "documents": [],
    "relevant_docs": [],
    "answer": "",
    "retry_count": 0,
    "max_retries": 2
}

result_2 = app_2.invoke(test_input_2)
print(f"\n最终结果:")
print(f"  查询: {result_2['query']}")
print(f"  相关文档数: {len(result_2['relevant_docs'])}")
print(f"  答案: {result_2['answer']}")
print(f"  重试次数: {result_2['retry_count']}")

# ===== 8. 可视化不同配置 =====
print("\n=== 可视化不同配置 ===")

display(Markdown("### 配置1: 标准 RAG 工作流"))
display(Image(app.get_graph().draw_mermaid_png()))

display(Markdown("### 配置2: 带查询改写的 RAG 工作流"))
display(Image(app_2.get_graph().draw_mermaid_png()))

# ===== 9. 保存可视化图表 =====
print("\n=== 保存可视化图表 ===")

# 保存 PNG
png_data = app.get_graph().draw_mermaid_png()
with open("rag_workflow.png", "wb") as f:
    f.write(png_data)
print("PNG 已保存: rag_workflow.png")

# 保存 Mermaid 文本
mermaid_text = app.get_graph().draw_mermaid()
with open("rag_workflow.mmd", "w") as f:
    f.write(mermaid_text)
print("Mermaid 文本已保存: rag_workflow.mmd")

# ===== 10. 生成文档 =====
print("\n=== 生成文档 ===")

documentation = f"""
# RAG 工作流文档

## 工作流结构

```mermaid
{mermaid_text}
```

## 节点说明

- **retrieve**: 检索相关文档
- **grade_documents**: 评分文档相关性
- **transform_query**: 改写查询以提高检索质量
- **generate**: 基于相关文档生成答案

## 条件路由

1. **grade_documents → generate/transform_query**: 
   - 如果有足够的相关文档 (≥2),进入生成
   - 否则,进入查询改写

2. **transform_query → retrieve/generate**:
   - 如果未达到最大重试次数,重新检索
   - 否则,强制生成答案

## 使用示例

```python
result = app.invoke({{
    "query": "LangGraph",
    "documents": [],
    "relevant_docs": [],
    "answer": "",
    "retry_count": 0,
    "max_retries": 3
}})
```

## 性能指标

- 节点数: {len(graph.nodes)}
- 边数: {len(graph.edges)}
- 条件边数: {sum(1 for e in graph.edges if e.conditional)}
"""

with open("rag_workflow_doc.md", "w") as f:
    f.write(documentation)
print("文档已保存: rag_workflow_doc.md")

print("\n=== RAG 工作流可视化完成 ===")
```

## 关键点说明

### 1. RAG 工作流结构

```python
# 核心流程
START → retrieve → grade_documents → [decision] → generate/transform_query → END

# 循环逻辑
transform_query → [check_retry] → retrieve/generate
```

### 2. 条件路由

```python

)
```

### 3. 状态管理

```python
class RAGState(TypedDict):

## 常见问题

### Q1: 如何可视化循环逻辑?

**A**: 使用条件边实现循环:
```python
workflow.add_conditional_edges(
    "transform_query",
    check_retry_limit,
    {
        "retrieve": "retrieve",  # 循环回到 retrieve
        "generate": "generate"
    }
)
```

### Q2: 如何调试条件路由?

**A**: 在决策函数中添加日志:
```python
def decide_to_generate(state):
    if len(state['relevant_docs']) >= 2:
        print("[Decision] 进入生成")
        return "generate"
    else:
        print("[Decision] 进入查询改写")
        return "transform_query"
```

### Q3: 如何优化 RAG 工作流?

**A**:
- 调整相关文档阈值
- 优化查询改写策略
- 增加缓存机制
- 并行检索多个来源
