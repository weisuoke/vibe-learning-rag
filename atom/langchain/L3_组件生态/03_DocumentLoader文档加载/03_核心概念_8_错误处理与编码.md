## 常见错误类型

### 1. 编码错误

**问题**: 文件编码与预期不符

```python
# 错误示例
loader = TextLoader("file.txt")
docs = loader.load()
# UnicodeDecodeError: 'utf-8' codec can't decode byte 0xff

# 原因: 文件是 GBK 编码,但默认使用 UTF-8 解码
```

**解决方案**:

```python
# 方案 1: 指定正确的编码
loader = TextLoader("file.txt", encoding="gbk")
docs = loader.load()

# 方案 2: 自动检测编码
import chardet

with open("file.txt", "rb") as f:
    raw_data = f.read()
    result = chardet.detect(raw_data)
    encoding = result["encoding"]

loader = TextLoader("file.txt", encoding=encoding)
docs = loader.load()

# 方案 3: 使用错误处理策略
loader = TextLoader("file.txt", encoding="utf-8", errors="ignore")
docs = loader.load()  # 忽略无法解码的字符
```

### 2. 文件不存在

**问题**: 文件路径错误或文件已删除

```python
# 错误示例
loader = PyPDFLoader("nonexistent.pdf")
docs = loader.load()
# FileNotFoundError: [Errno 2] No such file or directory: 'nonexistent.pdf'
```

**解决方案**:

```python
import os

# 方案 1: 检查文件是否存在
file_path = "document.pdf"
if os.path.exists(file_path):
    loader = PyPDFLoader(file_path)
    docs = loader.load()
else:
    print(f"文件不存在: {file_path}")

# 方案 2: 使用 try-except
try:
    loader = PyPDFLoader(file_path)
    docs = loader.load()
except FileNotFoundError:
    print(f"文件不存在: {file_path}")
    docs = []
```

### 3. 格式不支持

**问题**: 文件格式损坏或不支持

```python
# 错误示例
loader = PyPDFLoader("corrupted.pdf")
docs = loader.load()
# PDFSyntaxError: PDF file is corrupted
```

**解决方案**:

```python
# 方案 1: 捕获特定错误
from pypdf.errors import PdfReadError

try:
    loader = PyPDFLoader("document.pdf")
    docs = loader.load()
except PdfReadError as e:
    print(f"PDF 文件损坏: {e}")
    docs = []

# 方案 2: 尝试多个加载器
loaders = [
    PyPDFLoader("document.pdf"),
    PDFPlumberLoader("document.pdf"),
    UnstructuredPDFLoader("document.pdf")
]

docs = []
for loader in loaders:
    try:
        docs = loader.load()
        break  # 成功则退出
    except Exception as e:
        print(f"{loader.__class__.__name__} 失败: {e}")
        continue
```

### 4. 内存溢出

**问题**: 文件过大导致内存不足

```python
# 错误示例
loader = PyPDFLoader("huge_file.pdf")  # 10GB
docs = loader.load()
# MemoryError: Unable to allocate memory
```

**解决方案**:

```python
# 方案 1: 使用懒加载
loader = PyPDFLoader("huge_file.pdf")
for doc in loader.lazy_load():
    process(doc)  # 逐个处理,释放内存

# 方案 2: 分批处理
def batch_process(loader, batch_size=10):
    batch = []
    for doc in loader.lazy_load():
        batch.append(doc)
        if len(batch) >= batch_size:
            process_batch(batch)
            batch = []  # 清空批次
    if batch:
        process_batch(batch)

batch_process(loader)
```

### 5. 权限错误

**问题**: 没有读取文件的权限

```python
# 错误示例
loader = TextLoader("/root/protected.txt")
docs = loader.load()
# PermissionError: [Errno 13] Permission denied: '/root/protected.txt'
```

**解决方案**:

```python
# 方案 1: 检查权限
import os

file_path = "/root/protected.txt"
if os.access(file_path, os.R_OK):
    loader = TextLoader(file_path)
    docs = loader.load()
else:
    print(f"没有读取权限: {file_path}")

# 方案 2: 捕获权限错误
try:
    loader = TextLoader(file_path)
    docs = loader.load()
except PermissionError:
    print(f"没有读取权限: {file_path}")
    docs = []
```

### 常见编码问题

**问题 1: 中文乱码**

```python
# 错误: 使用错误的编码
loader = TextLoader("chinese.txt", encoding="utf-8")
docs = loader.load()
print(docs[0].page_content)
# 输出: ���������� (乱码)

# 正确: 使用正确的编码
loader = TextLoader("chinese.txt", encoding="gbk")
docs = loader.load()
print(docs[0].page_content)
# 输出: 这是中文内容
```

**问题 2: BOM 标记**

```python
# 问题: UTF-8 BOM 导致的额外字符
with open("file.txt", "r", encoding="utf-8") as f:
    content = f.read()
    print(repr(content[:3]))
    # 输出: '\ufeffHello' (BOM 标记)

# 解决: 使用 utf-8-sig 编码
loader = TextLoader("file.txt", encoding="utf-8-sig")
docs = loader.load()
# BOM 标记被自动移除
```

**问题 3: 混合编码**

```python
# 问题: 文件包含多种编码
# 解决: 使用 errors 参数

# 忽略无法解码的字符
loader = TextLoader("mixed.txt", encoding="utf-8", errors="ignore")

# 替换无法解码的字符
loader = TextLoader("mixed.txt", encoding="utf-8", errors="replace")

# 使用 XML 字符引用
loader = TextLoader("mixed.txt", encoding="utf-8", errors="xmlcharrefreplace")
```

## 错误恢复策略

### 1. 重试机制

```python
import time

def load_with_retry(loader, max_retries=3, delay=1):
    """带重试的加载"""
    for attempt in range(max_retries):
        try:
            return loader.load()
        except Exception as e:
            if attempt < max_retries - 1:
                print(f"加载失败 (尝试 {attempt + 1}/{max_retries}): {e}")
                time.sleep(delay)
            else:
                print(f"加载失败,已达最大重试次数: {e}")
                raise

# 使用
loader = WebBaseLoader("https://example.com")
docs = load_with_retry(loader)
```

### 2. 降级策略

```python
def load_with_fallback(file_path):
    """带降级的加载"""
    # 尝试顺序: PyPDF -> PDFPlumber -> Unstructured
    loaders = [
        ("PyPDF", PyPDFLoader(file_path)),
        ("PDFPlumber", PDFPlumberLoader(file_path)),
        ("Unstructured", UnstructuredPDFLoader(file_path))
    ]

    for name, loader in loaders:
        try:
            docs = loader.load()
            print(f"使用 {name} 加载成功")
            return docs
        except Exception as e:
            print(f"{name} 加载失败: {e}")
            continue

    # 所有加载器都失败
    raise Exception(f"无法加载文件: {file_path}")

# 使用
docs = load_with_fallback("document.pdf")
```

### 3. 部分成功处理

```python
def load_directory_safe(directory):
    """安全地加载目录中的所有文件"""
    loader = DirectoryLoader(directory)

    successful = []
    failed = []

    for file_path in loader.file_paths:
        try:
            file_loader = TextLoader(file_path)
            docs = file_loader.load()
            successful.extend(docs)
        except Exception as e:
            failed.append((file_path, str(e)))

    # 返回成功和失败的信息
    return {
        "successful": successful,
        "failed": failed,
        "success_rate": len(successful) / (len(successful) + len(failed))
    }

# 使用
result = load_directory_safe("docs/")
print(f"成功: {len(result['successful'])} 个文档")
print(f"失败: {len(result['failed'])} 个文件")
for file_path, error in result["failed"]:
    print(f"  - {file_path}: {error}")
```

## 实际应用

### 场景 1: 批量文档加载

```python
from pathlib import Path

def load_documents_safe(directory):
    """安全地批量加载文档"""
    directory = Path(directory)
    results = {
        "successful": [],
        "failed": [],
        "skipped": []
    }

    for file_path in directory.rglob("*"):
        if not file_path.is_file():
            continue

        # 检查文件大小
        if file_path.stat().st_size > 100 * 1024 * 1024:  # 100MB
            results["skipped"].append((str(file_path), "文件过大"))
            continue

        # 根据扩展名选择加载器
        ext = file_path.suffix.lower()
        try:
            if ext == ".pdf":
                loader = PyPDFLoader(str(file_path))
            elif ext == ".txt":
                loader = TextLoader(str(file_path), encoding="utf-8", errors="ignore")
            elif ext == ".docx":
                loader = UnstructuredWordDocumentLoader(str(file_path))
            else:
                results["skipped"].append((str(file_path), "不支持的格式"))
                continue

            docs = loader.load()
            results["successful"].extend(docs)

        except Exception as e:
            results["failed"].append((str(file_path), str(e)))

    return results

# 使用
results = load_documents_safe("docs/")
print(f"成功: {len(results['successful'])} 个文档")
print(f"失败: {len(results['failed'])} 个文件")
print(f"跳过: {len(results['skipped'])} 个文件")
```

### 场景 2: 网络文档加载

```python
import requests
from requests.adapters import HTTPAdapter
from requests.packages.urllib3.util.retry import Retry

def load_web_with_retry(url, max_retries=3):
    """带重试的网络文档加载"""
    # 配置重试策略
    session = requests.Session()
    retry = Retry(
        total=max_retries,
        backoff_factor=1,
        status_forcelist=[500, 502, 503, 504]
    )
    adapter = HTTPAdapter(max_retries=retry)
    session.mount("http://", adapter)
    session.mount("https://", adapter)

    try:
        # 加载网页
        response = session.get(url, timeout=10)
        response.raise_for_status()

        # 创建 Blob
        blob = Blob.from_data(
            data=response.text,
            metadata={"source": url}
        )

        # 解析 HTML
        parser = BSHTMLParser()
        docs = list(parser.lazy_parse(blob))

        return docs

    except requests.exceptions.RequestException as e:
        print(f"网络请求失败: {e}")
        return []

# 使用
docs = load_web_with_retry("https://example.com")
```

### 场景 3: 编码自动检测

```python
import chardet

def load_text_auto_encoding(file_path):
    """自动检测编码并加载文本"""
    # 读取文件的前 10000 字节用于检测
    with open(file_path, "rb") as f:
        raw_data = f.read(10000)

    # 检测编码
    result = chardet.detect(raw_data)
    encoding = result["encoding"]
    confidence = result["confidence"]

    print(f"检测到编码: {encoding} (置信度: {confidence:.2%})")

    # 如果置信度低,使用默认编码
    if confidence < 0.7:
        print("置信度较低,使用 UTF-8")
        encoding = "utf-8"

    # 加载文档
    try:
        loader = TextLoader(file_path, encoding=encoding, errors="replace")
        docs = loader.load()
        return docs
    except Exception as e:
        print(f"加载失败: {e}")
        # 降级到 UTF-8
        loader = TextLoader(file_path, encoding="utf-8", errors="ignore")
        return loader.load()

# 使用
docs = load_text_auto_encoding("unknown_encoding.txt")
```

## 常见误区

### ❌ 误区 1: 忽略错误

**错误**: 不处理异常,导致程序崩溃

```python
# ✗ 错误: 不处理异常
loader = PyPDFLoader("file.pdf")
docs = loader.load()  # 可能崩溃

# ✓ 正确: 处理异常
try:
    loader = PyPDFLoader("file.pdf")
    docs = loader.load()
except Exception as e:
    print(f"加载失败: {e}")
    docs = []
```

### ❌ 误区 2: 过度捕获

**错误**: 捕获所有异常,隐藏真正的问题

```python
# ✗ 错误: 捕获所有异常
try:
    loader = PyPDFLoader("file.pdf")
    docs = loader.load()
except:  # 捕获所有异常
    docs = []  # 隐藏了真正的问题

# ✓ 正确: 捕获特定异常
try:
    loader = PyPDFLoader("file.pdf")
    docs = loader.load()
except FileNotFoundError:
    print("文件不存在")
    docs = []
except PdfReadError:
    print("PDF 文件损坏")
    docs = []
```

### ❌ 误区 3: 默认编码

**错误**: 总是使用默认编码

```python
# ✗ 错误: 假设所有文件都是 UTF-8
loader = TextLoader("file.txt")  # 可能是 GBK

# ✓ 正确: 检测或指定编码
import chardet

with open("file.txt", "rb") as f:
    result = chardet.detect(f.read())
    encoding = result["encoding"]

loader = TextLoader("file.txt", encoding=encoding)
```

## 最佳实践

### 1. 分层错误处理

```python
# 底层: 捕获特定错误
def load_file(file_path):
    try:
        loader = TextLoader(file_path)
        return loader.load()
    except FileNotFoundError:
        raise FileNotFoundError(f"文件不存在: {file_path}")
    except UnicodeDecodeError as e:
        raise ValueError(f"编码错误: {e}")

# 中层: 重试和降级
def load_with_retry(file_path):
    for attempt in range(3):
        try:
            return load_file(file_path)
        except Exception as e:
            if attempt < 2:
                time.sleep(1)
                continue
            raise

# 上层: 批量处理
def load_directory(directory):
    results = {"successful": [], "failed": []}
    for file_path in Path(directory).rglob("*"):
        try:
            docs = load_with_retry(file_path)
            results["successful"].extend(docs)
        except Exception as e:
            results["failed"].append((file_path, str(e)))
    return results
```

### 4. 超时控制

```python
import signal

class TimeoutError(Exception):
    pass

def timeout_handler(signum, frame):
    raise TimeoutError("操作超时")

def load_with_timeout(loader, timeout=30):
    """带超时的加载"""
    # 设置超时
    signal.signal(signal.SIGALRM, timeout_handler)
    signal.alarm(timeout)

    try:
        docs = loader.load()
        signal.alarm(0)  # 取消超时
        return docs
    except TimeoutError:
        print(f"加载超时 ({timeout}秒)")
        return []
```

## 总结

### 常见错误

| 错误类型 | 原因 | 解决方案 |
|---------|------|----------|
| 编码错误 | 编码不匹配 | 检测或指定编码 |
| 文件不存在 | 路径错误 | 检查文件存在 |
| 格式不支持 | 文件损坏 | 尝试多个加载器 |
| 内存溢出 | 文件过大 | 使用懒加载 |
| 权限错误 | 无读取权限 | 检查权限 |

### 处理策略

| 策略 | 说明 |
|------|------|
| 重试机制 | 失败后重试 |
| 降级策略 | 尝试备用方案 |
| 部分成功 | 记录失败项 |
| 日志记录 | 记录错误信息 |
