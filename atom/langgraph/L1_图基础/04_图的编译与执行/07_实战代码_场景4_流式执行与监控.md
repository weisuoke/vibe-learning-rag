```python
"""
场景4：流式执行与监控
演示 LangGraph 的流式执行和监控机制

来源：

- Context7: langgraph_examples_01.md

import os
import time
from typing import TypedDict, Annotated
from operator import add
from datetime import datetime
from dotenv import load_dotenv

from langgraph.graph import StateGraph, START, END

# 加载环境变量
load_dotenv()

# ===== 1. 定义状态 Schema =====
print("=== 1. 定义状态 Schema ===\n")


class ProcessState(TypedDict):
    """处理状态"""
    task_id: str
    input_data: str
    processed_data: str
    results: list[str]
    logs: Annotated[list[str], add]
    progress: int


print("状态 Schema 定义完成\n")


# ===== 2. 定义节点函数（模拟耗时操作）=====
print("=== 2. 定义节点函数 ===\n")


def load_data_node(state: ProcessState) -> dict:
    """加载数据节点"""
    print(f"[load_data_node] 开始加载数据...")
    time.sleep(1)  # 模拟耗时操作

    return {
        "processed_data": f"loaded_{state['input_data']}",
        "logs": [f"{datetime.now()}: 数据加载完成"],
        "progress": 25
    }


def transform_node(state: ProcessState) -> dict:
    """转换数据节点"""
    print(f"[transform_node] 开始转换数据...")
    time.sleep(1)

    return {
        "processed_data": f"transformed_{state['processed_data']}",
        "logs": [f"{datetime.now()}: 数据转换完成"],
        "progress": 50
    }


def analyze_node(state: ProcessState) -> dict:
    """分析数据节点"""
    print(f"[analyze_node] 开始分析数据...")
    time.sleep(1)

    return {
        "results": ["分析结果1", "分析结果2", "分析结果3"],
        "logs": [f"{datetime.now()}: 数据分析完成"],
        "progress": 75
    }


def finalize_node(state: ProcessState) -> dict:
    """完成节点"""
    print(f"[finalize_node] 完成处理...")
    time.sleep(0.5)

    return {
        "logs": [f"{datetime.now()}: 处理完成"],
        "progress": 100
    }


print("节点函数定义完成\n")


# ===== 3. 构建图 =====
print("=== 3. 构建图 ===\n")

workflow = StateGraph(ProcessState)
workflow.add_node("load", load_data_node)
workflow.add_node("transform", transform_node)
workflow.add_node("analyze", analyze_node)
workflow.add_node("finalize", finalize_node)

workflow.add_edge(START, "load")
workflow.add_edge("load", "transform")
workflow.add_edge("transform", "analyze")
workflow.add_edge("analyze", "finalize")
workflow.add_edge("finalize", END)

app = workflow.compile()

print("图编译完成\n")


# ===== 4. 场景A：stream_mode="updates"（仅输出更新）=====
print("=== 4. 场景A：stream_mode='updates' ===\n")

inputs_a = {
    "task_id": "task-001",
    "input_data": "raw_data",
    "processed_data": "",
    "results": [],
    "logs": [],
    "progress": 0
}

print("执行流式输出（仅显示更新）...")
print("-" * 60)

# 来源: Context7 langgraph_invoke_01.md
for chunk in app.stream(inputs_a, stream_mode="updates"):
    for node_name, updates in chunk.items():
        print(f"\n节点 '{node_name}' 更新:")
        for key, value in updates.items():
            print(f"  {key}: {value}")
    print("-" * 60)

print("\n场景A完成\n")


# ===== 5. 场景B：stream_mode="values"（输出完整状态）=====

inputs_b = {
    "task_id": "task-002",
    "input_data": "raw_data",
    "processed_data": "",
    "results": [],
    "logs": [],
    "progress": 0
}

print("执行流式输出（显示完整状态）...")
print("-" * 60)

# 来源: Context7 langgraph_invoke_01.md
for chunk in app.stream(inputs_b, stream_mode="values"):

    print(f"  任务ID: {chunk['task_id']}")
    print(f"  进度: {chunk['progress']}%")
    print(f"  处理数据: {chunk['processed_data']}")
    print(f"  日志条数: {len(chunk['logs'])}")
    print("-" * 60)

print("\n场景B完成\n")


# ===== 6. 场景C：实时进度监控 =====

def progress_monitor(stream_output):
    """
    进度监控器

    实际应用：
    - Web应用的进度条
    - CLI工具的进度显示
    - 日志记录系统
    """
    print("进度监控器启动...")
    print("\n进度条:")

    for chunk in stream_output:
        if "progress" in chunk:
            progress = chunk["progress"]
            bar_length = 50
            filled = int(bar_length * progress / 100)
            bar = "█" * filled + "░" * (bar_length - filled)
            print(f"\r[{bar}] {progress}%", end="", flush=True)

    print("\n\n进度监控完成")


inputs_c = {
    "task_id": "task-003",
    "input_data": "raw_data",
    "processed_data": "",
    "results": [],
    "logs": [],
    "progress": 0
}

# 使用进度监控器
progress_monitor(app.stream(inputs_c, stream_mode="values"))

print()

class LogCollector:
    """
    日志收集器

    实际应用：
    - 集中式日志管理
    - 错误追踪
    - 性能分析
    """

    def __init__(self):
        self.logs = []
        self.node_timings = {}

    def collect(self, stream_output):
        """收集日志和性能数据"""
        print("日志收集器启动...\n")

        for chunk in stream_output:
            # 收集日志
            if "logs" in chunk and chunk["logs"]:
                self.logs.extend(chunk["logs"])

            # 记录节点执行
            for node_name in chunk.keys():
                if node_name not in ["task_id", "input_data", "processed_data", "results", "logs", "progress"]:
                    if node_name not in self.node_timings:
                        self.node_timings[node_name] = datetime.now()

        print("日志收集完成\n")

    def report(self):
        """生成报告"""
        print("=" * 60)
        print("执行报告")
        print("=" * 60)

        print(f"\n总日志条数: {len(self.logs)}")
        print("\n日志详情:")
        for log in self.logs:
            print(f"  - {log}")

        print(f"\n执行的节点数: {len(self.node_timings)}")
        print("=" * 60)

class MultiModeMonitor:
    """
    多模式监控器

    实际应用：
    - 调试复杂工作流
    - 性能优化
    - 问题诊断
    """

    def __init__(self):
        self.updates_count = 0
        self.states_count = 0

    def monitor(self, app, inputs):
        """同时监控更新和状态"""
        print("多模式监控器启动...\n")

        # 使用 updates 模式
        print("监控模式1：节点更新")
        print("-" * 60)

        for chunk in app.stream(inputs, stream_mode="updates"):
            self.updates_count += 1
            for node_name, updates in chunk.items():
                print(f"节点 '{node_name}' 更新了 {len(updates)} 个字段")

        print(f"\n总更新次数: {self.updates_count}")
        print()

        # 重新执行，使用 values 模式
        print("监控模式2：完整状态")
        print("-" * 60)

        for chunk in app.stream(inputs, stream_mode="values"):
            self.states_count += 1
            print(f"状态快照 #{self.states_count}: 进度 {chunk['progress']}%")

        print(f"\n总状态快照数: {self.states_count}")
        print()

class WebAPIProgressHandler:
    """
    Web API 进度处理器

    实际应用：
    - FastAPI/Flask 的 SSE（Server-Sent Events）
    - WebSocket 实时通信
    - 长时间运行任务的进度反馈
    """

    def __init__(self):
        self.events = []

    def stream_to_client(self, app, inputs):
        """
        模拟向客户端流式发送进度

        在实际应用中，这会通过 SSE 或 WebSocket 发送
        """
        print("模拟 Web API 流式响应...\n")

        for chunk in app.stream(inputs, stream_mode="values"):
            # 构造 SSE 事件
            event = {
                "type": "progress",
                "data": {
                    "task_id": chunk["task_id"],
                    "progress": chunk["progress"],
                    "status": "processing" if chunk["progress"] < 100 else "completed"
                }
            }

            self.events.append(event)

            # 模拟发送到客户端
            print(f"发送事件: {event}")

        print(f"\n总共发送了 {len(self.events)} 个事件")

class PerformanceMonitor:
    """
    性能监控器

    实际应用：
    - 识别性能瓶颈
    - 优化节点执行
    - 资源使用分析
    """

    def __init__(self):
        self.node_times = {}
        self.start_time = None

    def monitor(self, app, inputs):
        """监控执行性能"""
        print("性能监控器启动...\n")

        self.start_time = time.time()
        last_time = self.start_time

        for chunk in app.stream(inputs, stream_mode="updates"):
            current_time = time.time()
            elapsed = current_time - last_time

            for node_name in chunk.keys():
                if node_name not in self.node_times:
                    self.node_times[node_name] = []
                self.node_times[node_name].append(elapsed)

            last_time = current_time

        total_time = time.time() - self.start_time

        print("性能报告:")
        print("-" * 60)
        print(f"总执行时间: {total_time:.2f}秒\n")

        print("各节点执行时间:")
        for node_name, times in self.node_times.items():
            avg_time = sum(times) / len(times)
            print(f"  {node_name}: {avg_time:.2f}秒")

        print("-" * 60)
```

## 关键知识点

### 1. 流模式使用

```python
# updates 模式：仅输出更新
for chunk in app.stream(inputs, stream_mode="updates"):
    for node_name, updates in chunk.items():
        print(f"{node_name}: {updates}")

# values 模式：输出完整状态
for chunk in app.stream(inputs, stream_mode="values"):
    print(f"完整状态: {chunk}")
```

### 2. 进度监控实现

```python
for chunk in app.stream(inputs, stream_mode="values"):
    progress = chunk.get("progress", 0)
    print(f"进度: {progress}%")
```

### 3. 日志收集

```python
logs = []
for chunk in app.stream(inputs, stream_mode="values"):
    if "logs" in chunk:
```
