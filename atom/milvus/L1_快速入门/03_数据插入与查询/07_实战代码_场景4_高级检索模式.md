### 2. 创建混合搜索 Collection

```python
def create_hybrid_collection():
    """
    创建支持混合搜索的 Collection
    包含 Dense 和 Sparse 向量
    """
    collection_name = "hybrid_collection"

    if milvus_client.has_collection(collection_name):
        milvus_client.drop_collection(collection_name)

    schema = milvus_client.create_schema()

    # 基础字段
    schema.add_field("id", DataType.INT64, is_primary=True, auto_id=True)
    schema.add_field("document", DataType.VARCHAR, max_length=5000)
    schema.add_field("category", DataType.VARCHAR, max_length=100)
    schema.add_field("score", DataType.FLOAT)

    # Dense 向量（语义理解）
    schema.add_field("dense_vector", DataType.FLOAT_VECTOR, dim=1536)

    # Sparse 向量（关键词匹配）
    schema.add_field("sparse_vector", DataType.SPARSE_FLOAT_VECTOR)

    # 配置 Dense Embedding Function
    schema.add_function(
        Function(
            name="dense_embedding",
            function_type=FunctionType.TEXTEMBEDDING,
            input_field_names=["document"],
            output_field_names=["dense_vector"],
            params={
                "provider": "openai",
                "model_name": "text-embedding-3-small",
                "api_key": OPENAI_API_KEY
            }
        )
    )

    # 配置 Sparse Embedding Function (BM25)
    schema.add_function(
        Function(
            name="sparse_embedding",
            function_type=FunctionType.BM25,
            input_field_names=["document"],
            output_field_names=["sparse_vector"]
        )
    )

    # 创建索引
    index_params = milvus_client.prepare_index_params()

    # Dense 向量索引
    index_params.add_index(
        field_name="dense_vector",
        index_type="AUTOINDEX",
        metric_type="COSINE"
    )

    # Sparse 向量索引
    index_params.add_index(
        field_name="sparse_vector",
        index_type="SPARSE_INVERTED_INDEX",
        metric_type="IP"
    )

    milvus_client.create_collection(
        collection_name=collection_name,
        schema=schema,
        index_params=index_params
    )

    print(f"✅ 混合搜索 Collection 创建成功: {collection_name}")
    return collection_name
```

### 4. 混合搜索（Dense + Sparse）

```python
def hybrid_search_demo(collection_name, query_text):
    """
    混合搜索演示
    结合 Dense（语义）和 Sparse（关键词）向量
    """
    print(f"\n【混合搜索】")
    print(f"查询: {query_text}")

    # 创建 Dense 搜索请求
    dense_req = AnnSearchRequest(
        data=[query_text],
        anns_field="dense_vector",
        param={"metric_type": "COSINE"},
        limit=50  # 候选数量
    )

    # 创建 Sparse 搜索请求
    sparse_req = AnnSearchRequest(
        data=[query_text],
        anns_field="sparse_vector",
        param={"metric_type": "IP"},
        limit=50  # 候选数量
    )

    # 执行混合搜索（使用 RRF 重排序）
    results = milvus_client.hybrid_search(
        collection_name=collection_name,
        reqs=[dense_req, sparse_req],
        ranker=RRFRanker(),
        limit=3,  # 最终返回数量
        output_fields=["document", "category", "score"]
    )

    print("\n结果:")
    for i, hit in enumerate(results[0], 1):
        print(f"{i}. 相似度: {hit['distance']:.4f}")
        print(f"   类别: {hit['entity']['category']}")
        print(f"   文档: {hit['entity']['document'][:60]}...")

    return results[0]
```

### 5. 加权混合搜索

```python
def weighted_hybrid_search(collection_name, query_text, dense_weight=0.7):
    """
    加权混合搜索
    可以控制 Dense 和 Sparse 的权重
    """
    print(f"\n【加权混合搜索】")
    print(f"查询: {query_text}")
    print(f"权重: Dense {dense_weight}, Sparse {1-dense_weight}")

    dense_req = AnnSearchRequest(
        data=[query_text],
        anns_field="dense_vector",
        param={"metric_type": "COSINE"},
        limit=50
    )

    sparse_req = AnnSearchRequest(
        data=[query_text],
        anns_field="sparse_vector",
        param={"metric_type": "IP"},
        limit=50
    )

    # 使用加权重排序
    results = milvus_client.hybrid_search(
        collection_name=collection_name,
        reqs=[dense_req, sparse_req],
        ranker=WeightedRanker(weights=[dense_weight, 1-dense_weight]),
        limit=3,
        output_fields=["document", "category"]
    )

    print("\n结果:")
    for i, hit in enumerate(results[0], 1):
        print(f"{i}. 相似度: {hit['distance']:.4f}")
        print(f"   文档: {hit['entity']['document'][:60]}...")

    return results[0]
```

## 重排序策略对比

### RRFRanker（推荐）

**优势**:
- 无需调参
- 鲁棒性好
- 适合通用场景

**使用**:
```python
ranker = RRFRanker()
```

### WeightedRanker

**优势**:
- 可控制权重
- 适合已知重要性

**使用**:
```python
# Dense 更重要
ranker = WeightedRanker(weights=[0.7, 0.3])

# Sparse 更重要（关键词查询）
ranker = WeightedRanker(weights=[0.3, 0.7])
```

## 常见问题

### Q1: 混合搜索比单一搜索慢多少？

**答案**: 约 2 倍时间，但检索质量显著提升。

```
单一搜索: ~10ms
混合搜索: ~20ms
```

### Q2: 如何选择候选数量？

**推荐值**:
```python
# 最终返回 10 个
dense_req = AnnSearchRequest(..., limit=50)   # 候选 50
sparse_req = AnnSearchRequest(..., limit=50)  # 候选 50

# 候选数量 = 最终返回数量 × 5-10
```

### Q3: 过滤会降低性能吗？

**答案**: 取决于过滤比例。

- 过滤掉 90% 数据：性能提升
- 过滤掉 10% 数据：性能下降
- 建议：使用分区代替过滤
