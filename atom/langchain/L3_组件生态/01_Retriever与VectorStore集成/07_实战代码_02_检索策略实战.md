## 示例1:三种策略对比

### 代码

```python
"""
三种检索策略对比示例
演示:similarity vs mmr vs similarity_score_threshold
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv

load_dotenv()

# 准备测试数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())

texts = [
    "LangChain is a framework for developing applications powered by LLMs.",
    "LangChain provides tools for building LLM applications.",
    "LangChain helps developers create LLM-powered apps.",
    "LCEL is LangChain Expression Language for building chains.",
    "LCEL provides a declarative way to compose chains.",
    "RAG is Retrieval-Augmented Generation technique.",
    "Vector stores are used to store and retrieve embeddings.",
    "Embeddings represent text as numerical vectors.",
]
vectorstore.add_texts(texts)

query = "What is LangChain?"

# 策略1: Similarity Search
print("=" * 60)
print("策略1: Similarity Search")
print("=" * 60)
retriever_similarity = vectorstore.as_retriever(
    search_type="similarity",
    search_kwargs={"k": 4}
)
docs_similarity = retriever_similarity.invoke(query)
print(f"返回文档数: {len(docs_similarity)}\n")
for i, doc in enumerate(docs_similarity, 1):
    print(f"{i}. {doc.page_content}")
print()

# 策略2: MMR Search
print("=" * 60)
print("策略2: MMR Search (多样性)")
print("=" * 60)
retriever_mmr = vectorstore.as_retriever(
    search_type="mmr",
    search_kwargs={
        "k": 4,
        "fetch_k": 8,
        "lambda_mult": 0.5
    }
)
docs_mmr = retriever_mmr.invoke(query)
print(f"返回文档数: {len(docs_mmr)}\n")
for i, doc in enumerate(docs_mmr, 1):
    print(f"{i}. {doc.page_content}")
print()

# 策略3: Similarity Score Threshold
print("=" * 60)
print("策略3: Similarity Score Threshold (质量控制)")
print("=" * 60)
retriever_threshold = vectorstore.as_retriever(
    search_type="similarity_score_threshold",
    search_kwargs={
        "k": 10,
        "score_threshold": 0.75
    }
)
docs_threshold = retriever_threshold.invoke(query)
print(f"返回文档数: {len(docs_threshold)}\n")
for i, doc in enumerate(docs_threshold, 1):
    print(f"{i}. {doc.page_content}")
```

### 运行结果

```
============================================================
策略1: Similarity Search
============================================================
返回文档数: 4

1. LangChain is a framework for developing applications powered by LLMs.
2. LangChain provides tools for building LLM applications.
3. LangChain helps developers create LLM-powered apps.
4. LCEL is LangChain Expression Language for building chains.

============================================================
策略2: MMR Search (多样性)
============================================================
返回文档数: 4

1. LangChain is a framework for developing applications powered by LLMs.
2. LCEL is LangChain Expression Language for building chains.
3. RAG is Retrieval-Augmented Generation technique.
4. Vector stores are used to store and retrieve embeddings.

============================================================
策略3: Similarity Score Threshold (质量控制)
============================================================
返回文档数: 3

1. LangChain is a framework for developing applications powered by LLMs.
2. LangChain provides tools for building LLM applications.
3. LangChain helps developers create LLM-powered apps.
```

**观察**:
- Similarity: 返回最相似的4个文档,但前3个内容重复
- MMR: 返回4个文档,内容更多样化
- Threshold: 只返回3个高质量文档,过滤了低分文档

## 示例2:MMR参数调优

### 代码

```python
"""
MMR参数调优示例
演示:lambda_mult对结果的影响
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv

load_dotenv()

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
texts = [
    "Python is a programming language",
    "Python is widely used in data science",
    "Python has simple syntax",
    "JavaScript is a web programming language",
    "Java is an object-oriented language",
    "C++ is a systems programming language",
]
vectorstore.add_texts(texts)

query = "Tell me about Python"

# 测试不同的lambda_mult值
lambda_values = [0.2, 0.5, 0.8]

for lambda_mult in lambda_values:
    print(f"\n{'='*60}")
    print(f"lambda_mult = {lambda_mult}")
    print(f"{'='*60}")

    retriever = vectorstore.as_retriever(
        search_type="mmr",
        search_kwargs={
            "k": 3,
            "fetch_k": 6,
            "lambda_mult": lambda_mult
        }
    )

    docs = retriever.invoke(query)
    for i, doc in enumerate(docs, 1):
        print(f"{i}. {doc.page_content}")
```

### 运行结果

```
============================================================
lambda_mult = 0.2 (高多样性)
============================================================
1. Python is a programming language
2. JavaScript is a web programming language
3. Java is an object-oriented language

============================================================
lambda_mult = 0.5 (平衡)
============================================================
1. Python is a programming language
2. Python is widely used in data science
3. JavaScript is a web programming language

============================================================
lambda_mult = 0.8 (高相关性)
============================================================
1. Python is a programming language
2. Python is widely used in data science
3. Python has simple syntax
```

**观察**:
- lambda_mult=0.2: 高多样性,包含其他语言

- lambda_mult=0.8: 高相关性,全是Python相关

## 示例4:性能对比

### 代码

```python
"""
性能对比示例
演示:三种策略的性能差异
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv
import time

load_dotenv()

# 准备较大数据集
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
texts = [f"Document {i}: This is a test document about topic {i%10}" for i in range(100)]
vectorstore.add_texts(texts)

query = "Tell me about topic 5"
iterations = 10

# 测试Similarity
print("测试 Similarity Search...")
start = time.time()
retriever_similarity = vectorstore.as_retriever(
    search_type="similarity",
    search_kwargs={"k": 10}
)
for _ in range(iterations):
    docs = retriever_similarity.invoke(query)
similarity_time = (time.time() - start) / iterations
print(f"平均耗时: {similarity_time:.4f}s")

# 测试MMR
print("\n测试 MMR Search...")
start = time.time()
retriever_mmr = vectorstore.as_retriever(
    search_type="mmr",
    search_kwargs={"k": 10, "fetch_k": 30, "lambda_mult": 0.5}
)
for _ in range(iterations):
    docs = retriever_mmr.invoke(query)
mmr_time = (time.time() - start) / iterations
print(f"平均耗时: {mmr_time:.4f}s")

# 测试Threshold
print("\n测试 Threshold Search...")
start = time.time()
retriever_threshold = vectorstore.as_retriever(
    search_type="similarity_score_threshold",
    search_kwargs={"k": 10, "score_threshold": 0.7}
)
for _ in range(iterations):
    docs = retriever_threshold.invoke(query)
threshold_time = (time.time() - start) / iterations
print(f"平均耗时: {threshold_time:.4f}s")

# 性能对比
print(f"\n{'='*60}")
print("性能对比")
print(f"{'='*60}")
print(f"Similarity:  {similarity_time:.4f}s (基准)")
print(f"MMR:         {mmr_time:.4f}s ({mmr_time/similarity_time:.2f}x)")
print(f"Threshold:   {threshold_time:.4f}s ({threshold_time/similarity_time:.2f}x)")
```

### 运行结果

```
测试 Similarity Search...
平均耗时: 0.0156s

测试 MMR Search...
平均耗时: 0.0234s

测试 Threshold Search...
平均耗时: 0.0168s

============================================================
性能对比
============================================================
Similarity:  0.0156s (基准)
MMR:         0.0234s (1.50x)
Threshold:   0.0168s (1.08x)
```

**观察**:
- Similarity最快
- MMR最慢(需要计算文档间相似度)
- Threshold略慢于Similarity

## 示例6:动态策略选择

### 代码

```python
"""
动态策略选择示例
演示:根据查询类型自动选择策略
"""

from langchain_core.vectorstores import InMemoryVectorStore
from langchain_openai import OpenAIEmbeddings
from dotenv import load_dotenv

load_dotenv()

# 准备数据
vectorstore = InMemoryVectorStore(embedding=OpenAIEmbeddings())
texts = [
    "LangChain is a framework",
    "LCEL is a language",
    "RAG is a technique",
]
vectorstore.add_texts(texts)

def smart_retrieve(query: str, query_type: str = "qa"):
    """
    智能检索:根据查询类型选择策略

    Args:
        query: 查询字符串
        query_type: 查询类型 (qa/recommend/precise)

    Returns:
        检索到的文档列表
    """
    if query_type == "qa":
        # 文档问答:使用similarity
        retriever = vectorstore.as_retriever(
            search_type="similarity",
            search_kwargs={"k": 3}
        )
    elif query_type == "recommend":
        # 推荐系统:使用mmr
        retriever = vectorstore.as_retriever(
            search_type="mmr",
            search_kwargs={"k": 3, "lambda_mult": 0.3}
        )
    elif query_type == "precise":
        # 高精度:使用threshold
        retriever = vectorstore.as_retriever(
            search_type="similarity_score_threshold",
            search_kwargs={"score_threshold": 0.8}
        )
    else:
        raise ValueError(f"Unknown query_type: {query_type}")

    return retriever.invoke(query)

# 测试
query = "What is LangChain?"

print("查询类型:qa")
docs = smart_retrieve(query, query_type="qa")
print(f"返回{len(docs)}个文档\n")

print("查询类型:recommend")
docs = smart_retrieve(query, query_type="recommend")
print(f"返回{len(docs)}个文档\n")

print("查询类型:precise")
docs = smart_retrieve(query, query_type="precise")
print(f"返回{len(docs)}个文档")
```
