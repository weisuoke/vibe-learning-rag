# 实战代码场景5：多租户元数据管理

> 本文档演示 Milvus 2.6 中三种多租户隔离策略的实现与对比

---

## 一、场景描述

### 1.1 业务需求

**场景**：构建 SaaS 型 RAG 知识库平台，不同企业客户（租户）有独立的文档数据和元数据需求。

**核心挑战**：
- 数据隔离：租户间数据不能互相访问
- 元数据差异：不同租户有不同的业务元数据字段
- 可扩展性：租户数量从几十到上万不等
- 成本控制：资源开销随租户数量线性增长不可接受

### 1.2 技术目标

1. **实现三种隔离策略**：Collection 隔离、Partition 隔离、元数据过滤
2. **动态元数据扩展**：每个租户可以有不同的元数据字段
3. **权限控制**：查询时只返回当前租户的数据

### 1.3 适用规模

| 策略 | 租户数量 | 资源开销 | 隔离性 |
|------|---------|---------|--------|
| Collection 隔离 | < 100 | 高 | 完全隔离 |
| Partition 隔离 | 100 - 10K | 中 | 较强 |
| 元数据过滤 | > 10K | 低 | 依赖过滤 |

## 二、完整实战代码

```python
"""
多租户元数据管理实战
演示：三种隔离策略的创建、插入、查询和对比
"""

import numpy as np
import time
from pymilvus import MilvusClient, DataType

# ===== 配置 =====
MILVUS_URI = "http://localhost:19530"
DIM = 128
client = MilvusClient(MILVUS_URI)
np.random.seed(42)


# ===== 工具函数 =====
def generate_docs(tenant_id: str, num: int, start_id: int = 0) -> list[dict]:
    """生成租户文档数据（带动态元数据）"""
    docs = []
    for i in range(num):
        doc = {
            "id": start_id + i,
            "vector": np.random.randn(DIM).astype(np.float32).tolist(),
            "text": f"tenant_{tenant_id}_doc_{i}",
            "tenant_id": tenant_id,
            "created_at": int(time.time()) - np.random.randint(0, 86400 * 30),
        }
        # 不同租户有不同的动态元数据
        if tenant_id == "tech_corp":
            doc["api_version"] = f"v{np.random.choice([1, 2, 3])}"
            doc["tech_stack"] = np.random.choice(["python", "java", "go"])
        elif tenant_id == "legal_firm":
            doc["approval_status"] = np.random.choice(["pending", "approved", "rejected"])
            doc["case_type"] = np.random.choice(["contract", "patent", "litigation"])
        elif tenant_id == "finance_co":
            doc["fiscal_year"] = np.random.choice([2024, 2025, 2026])
            doc["budget_code"] = f"BU-{np.random.randint(100, 999)}"
        docs.append(doc)
    return docs


query_vector = np.random.randn(DIM).astype(np.float32).tolist()


# ============================================================
# 策略1：Collection 级别隔离（< 100 租户）
# ============================================================
print("=" * 60)
print("策略1：Collection 级别隔离")
print("=" * 60)

tenants_strategy1 = ["tech_corp", "legal_firm", "finance_co"]

for tenant in tenants_strategy1:
    coll_name = f"mt_collection_{tenant}"
    if client.has_collection(coll_name):
        client.drop_collection(coll_name)

    # 每个租户独立的 Collection（启用动态字段）
    schema = client.create_schema(auto_id=False, enable_dynamic_field=True)
    schema.add_field("id", DataType.INT64, is_primary=True)
    schema.add_field("vector", DataType.FLOAT_VECTOR, dim=DIM)
    schema.add_field("text", DataType.VARCHAR, max_length=512)
    schema.add_field("tenant_id", DataType.VARCHAR, max_length=64)
    schema.add_field("created_at", DataType.INT64)

    index_params = client.prepare_index_params()
    index_params.add_index("vector", index_type="HNSW", metric_type="COSINE",
                           params={"M": 16, "efConstruction": 200})

    client.create_collection(coll_name, schema=schema, index_params=index_params)

    # 插入数据
    docs = generate_docs(tenant, num=100, start_id=0)
    client.insert(coll_name, docs)
    print(f"  租户 {tenant}: 创建 Collection，插入 100 条")

# 查询特定租户
results = client.search(
    collection_name="mt_collection_tech_corp",
    data=[query_vector],
    limit=5,
    output_fields=["text", "tenant_id", "api_version", "tech_stack"]
)
print(f"\n  查询 tech_corp 结果:")
for r in results[0][:3]:
    print(f"    id={r['id']}, api_version={r['entity'].get('api_version')}, "
          f"tech_stack={r['entity'].get('tech_stack')}")


# ============================================================
# 策略2：Partition 级别隔离（100 - 10K 租户）
# ============================================================
print("\n" + "=" * 60)
print("策略2：Partition 级别隔离")
print("=" * 60)

partition_coll = "mt_partition_demo"
if client.has_collection(partition_coll):
    client.drop_collection(partition_coll)

# 所有租户共享一个 Collection
schema = client.create_schema(auto_id=False, enable_dynamic_field=True)
schema.add_field("id", DataType.INT64, is_primary=True)
schema.add_field("vector", DataType.FLOAT_VECTOR, dim=DIM)
schema.add_field("text", DataType.VARCHAR, max_length=512)
schema.add_field("tenant_id", DataType.VARCHAR, max_length=64)
schema.add_field("created_at", DataType.INT64)

index_params = client.prepare_index_params()
index_params.add_index("vector", index_type="HNSW", metric_type="COSINE",
                       params={"M": 16, "efConstruction": 200})

client.create_collection(partition_coll, schema=schema, index_params=index_params)

# 为每个租户创建 Partition
global_id = 0
for tenant in tenants_strategy1:
    partition_name = f"tenant_{tenant}"
    client.create_partition(partition_coll, partition_name)

    docs = generate_docs(tenant, num=100, start_id=global_id)
    client.insert(partition_coll, docs, partition_name=partition_name)
    global_id += 100
    print(f"  租户 {tenant}: 创建 Partition，插入 100 条")

# 查询特定租户（指定 Partition）
results = client.search(
    collection_name=partition_coll,
    data=[query_vector],
    limit=5,
    partition_names=["tenant_tech_corp"],  # 只搜索该租户的 Partition
    output_fields=["text", "tenant_id", "api_version"]
)
print(f"\n  查询 tech_corp Partition 结果:")
for r in results[0][:3]:
    print(f"    id={r['id']}, tenant={r['entity'].get('tenant_id')}, "
          f"api_version={r['entity'].get('api_version')}")


# ============================================================
# 策略3：元数据过滤隔离（> 10K 租户）
# ============================================================
print("\n" + "=" * 60)
print("策略3：元数据过滤隔离")
print("=" * 60)

filter_coll = "mt_filter_demo"
if client.has_collection(filter_coll):
    client.drop_collection(filter_coll)

# 所有租户共享一个 Collection，用 tenant_id 过滤
schema = client.create_schema(auto_id=False, enable_dynamic_field=True)
schema.add_field("id", DataType.INT64, is_primary=True)
schema.add_field("vector", DataType.FLOAT_VECTOR, dim=DIM)
schema.add_field("text", DataType.VARCHAR, max_length=512)
schema.add_field("tenant_id", DataType.VARCHAR, max_length=64)
schema.add_field("created_at", DataType.INT64)

index_params = client.prepare_index_params()
index_params.add_index("vector", index_type="HNSW", metric_type="COSINE",
                       params={"M": 16, "efConstruction": 200})
# 为 tenant_id 创建标量索引（关键！）
index_params.add_index("tenant_id", index_type="INVERTED")

client.create_collection(filter_coll, schema=schema, index_params=index_params)

# 所有租户数据插入同一个 Collection
global_id = 0
for tenant in tenants_strategy1:
    docs = generate_docs(tenant, num=100, start_id=global_id)
    client.insert(filter_coll, docs)
    global_id += 100
    print(f"  租户 {tenant}: 插入 100 条（共享 Collection）")

# 查询特定租户（使用 filter）
results = client.search(
    collection_name=filter_coll,
    data=[query_vector],
    limit=5,
    filter='tenant_id == "tech_corp"',  # 元数据过滤
    output_fields=["text", "tenant_id", "api_version"]
)
print(f"\n  查询 tech_corp（filter）结果:")
for r in results[0][:3]:
    print(f"    id={r['id']}, tenant={r['entity'].get('tenant_id')}, "
          f"api_version={r['entity'].get('api_version')}")


# ============================================================
# 对比总结
# ============================================================
print("\n" + "=" * 60)
print("三种策略对比总结")
print("=" * 60)

print("""
┌──────────────┬──────────────┬──────────────┬──────────────┐
│ 特性         │ Collection   │ Partition    │ 元数据过滤   │
├──────────────┼──────────────┼──────────────┼──────────────┤
│ 租户数量     │ < 100        │ 100 - 10K    │ > 10K        │
│ 隔离性       │ 完全隔离     │ 较强         │ 依赖过滤     │
│ 资源开销     │ 高           │ 中           │ 低           │
│ 索引共享     │ 不共享       │ 共享         │ 共享         │
│ 删除租户     │ drop集合     │ drop分区     │ 按条件删除   │
│ 动态元数据   │ ✅ 支持      │ ✅ 支持      │ ✅ 支持      │
│ 查询方式     │ 指定集合名   │ 指定分区名   │ filter条件   │
└──────────────┴──────────────┴──────────────┴──────────────┘
""")


# ===== 清理 =====
for tenant in tenants_strategy1:
    client.drop_collection(f"mt_collection_{tenant}")
client.drop_collection(partition_coll)
client.drop_collection(filter_coll)
print("✅ 测试完成，所有 Collection 已清理")
```

## 四、生产环境建议

### 4.1 策略选择决策树

```
租户数量？
  ├─ < 100 → Collection 隔离
  │          优点：完全隔离，独立管理
  │          缺点：资源开销大
  │
  ├─ 100 - 10K → Partition 隔离
  │              优点：共享索引，资源适中
  │              缺点：Partition 数量有上限（建议 < 4096）
  │
  └─ > 10K → 元数据过滤
             优点：无租户数量限制，资源最小
             缺点：必须为 tenant_id 创建 INVERTED 索引
```

### 4.2 关键注意事项

- **元数据过滤必须加索引**：`tenant_id` 字段必须创建 INVERTED 索引，否则全表扫描
- **动态字段不要用于聚合**：group_by 对缺失动态字段有 bug（Issue #47438）
- **租户删除策略**：Collection 隔离直接 drop；Partition 隔离 drop partition；元数据过滤需要按条件 delete
