## 为什么需要异步工具？

### 与 RAG/Agent 开发的关系

在实际的 RAG 和 Agent 系统中，工具经常需要调用外部服务：

1. **RAG 系统中的异步场景**：
   - 调用 Embedding API（OpenAI、Cohere）
   - 查询向量数据库（Milvus、Pinecone）
   - 调用 Rerank API（Cohere Rerank）
   - 并发处理多个文档

2. **Agent 系统中的异步场景**：
   - 调用天气 API、地图 API
   - 查询数据库（PostgreSQL、MongoDB）
   - 调用第三方服务（支付、邮件）
   - 并发执行多个工具

**异步的优势**：
- **性能提升**：I/O 等待时可以处理其他任务
- **并发能力**：同时调用多个 API
- **资源利用**：避免线程阻塞

```python
"""
场景 2.1：创建基础异步天气工具
演示：异步函数装饰、HTTP 请求、错误处理
"""

import os
import asyncio
import httpx
from langchain.tools import tool
from dotenv import load_dotenv

# 加载环境变量
load_dotenv()


# ===== 1. 最简单的异步工具 =====
@tool
async def get_weather_simple(city: str) -> str:
    """Get current weather for a city.

    Args:
        city: City name, e.g., "Beijing", "New York"

    Returns:
        Weather information as text
    """
    # 模拟 API 调用（使用 httpbin.org 测试）
    async with httpx.AsyncClient() as client:
        try:
            # 模拟延迟
            await asyncio.sleep(0.5)

            # 实际项目中这里会调用真实的天气 API
            # response = await client.get(f"https://api.weather.com/v1/current?q={city}")

            # 模拟返回数据
            return f"Weather in {city}: 22°C, Sunny"

        except httpx.HTTPError as e:
            return f"Error fetching weather: {str(e)}"


# ===== 2. 测试异步工具 =====
async def test_simple_weather():
    print("=== 测试 1：基础异步天气工具 ===\n")

    cities = ["Beijing", "Shanghai", "Guangzhou"]

    # 串行调用
    print("串行调用（慢）：")
    import time
    start = time.time()
    for city in cities:
        result = await get_weather_simple.ainvoke({"city": city})
        print(f"{city}: {result}")
    print(f"耗时: {time.time() - start:.2f}秒\n")

    # 并发调用
    print("并发调用（快）：")
    start = time.time()
    tasks = [get_weather_simple.ainvoke({"city": city}) for city in cities]
    results = await asyncio.gather(*tasks)
    for city, result in zip(cities, results):
        print(f"{city}: {result}")
    print(f"耗时: {time.time() - start:.2f}秒\n")


# 运行测试
if __name__ == "__main__":
    asyncio.run(test_simple_weather())
```

```

Guangzhou: Weather in Guangzhou: 22°C, Sunny
```

### 场景 2.2：复杂 Schema 定义

```python
"""
场景 2.2：使用 Pydantic 定义复杂输入 schema
演示：Literal 类型、默认值、Field 描述
"""

import os
import asyncio
import httpx
from typing import Literal
from pydantic import BaseModel, Field
from langchain.tools import tool
from dotenv import load_dotenv

load_dotenv()


# ===== 1. 定义复杂输入 Schema =====
class WeatherInput(BaseModel):
    """Input schema for weather queries."""

    location: str = Field(
        description="City name or coordinates (e.g., 'Beijing' or '39.9,116.4')"
    )

    units: Literal["celsius", "fahrenheit"] = Field(
        default="celsius",
        description="Temperature unit preference"
    )

    include_forecast: bool = Field(
        default=False,
        description="Include 5-day forecast in addition to current weather"
    )

    language: Literal["en", "zh", "es", "fr"] = Field(
        default="en",
        description="Response language"
    )


# ===== 2. 创建工具 =====
@tool(args_schema=WeatherInput)
async def get_weather(
    location: str,
    units: str = "celsius",
    include_forecast: bool = False,
    language: str = "en"
) -> str:
    """Get current weather and optional forecast for a location.

    Supports multiple units and languages.
    """
    async with httpx.AsyncClient() as client:
        try:
            # 模拟 API 调用
            await asyncio.sleep(0.3)

            # 构造响应
            temp = 22 if units == "celsius" else 72
            unit_symbol = "°C" if units == "celsius" else "°F"

            result = f"Current weather in {location}: {temp}{unit_symbol}, Sunny"

            if include_forecast:
                result += "\n\nForecast (next 5 days):"
                result += "\n- Day 1: 24°C, Partly Cloudy"
                result += "\n- Day 2: 23°C, Rainy"
                result += "\n- Day 3: 25°C, Sunny"
                result += "\n- Day 4: 26°C, Sunny"
                result += "\n- Day 5: 24°C, Cloudy"

            return result

        except Exception as e:
            return f"Error: {str(e)}"


# ===== 3. 测试复杂 Schema =====
async def test_complex_schema():
    print("=== 测试 2：复杂 Schema 定义 ===\n")

    test_cases = [
        {
            "location": "Beijing",
            "units": "celsius",
            "include_forecast": False,
            "language": "en"
        },
        {
            "location": "New York",
            "units": "fahrenheit",
            "include_forecast": True,
            "language": "en"
        },
    ]

    for i, test in enumerate(test_cases, 1):
        print(f"测试用例 {i}:")
        print(f"输入: {test}")
        result = await get_weather.ainvoke(test)
        print(f"输出:\n{result}\n")
        print("-" * 60 + "\n")

    # 查看 Schema
    print("=== 工具 Schema ===")
    import json
    print(json.dumps(get_weather.args_schema.model_json_schema(), indent=2))


if __name__ == "__main__":
    asyncio.run(test_complex_schema())
```

3. **详细描述**：帮助 LLM 理解参数用途
4. **required 字段**：只有 location 是必填的

### 场景 2.3：生产级错误处理

```python
"""
场景 2.3：生产级错误处理
演示：重试机制、超时控制、详细错误信息
"""

import os
import asyncio
import httpx
from typing import Literal
from pydantic import BaseModel, Field
from langchain.tools import tool, ToolException
from dotenv import load_dotenv

load_dotenv()


# ===== 1. 配置类 =====
class WeatherAPIConfig:
    """Weather API configuration."""
    BASE_URL = "https://api.openweathermap.org/data/2.5"
    TIMEOUT = 10.0  # 秒
    MAX_RETRIES = 3
    RETRY_DELAY = 1.0  # 秒


# ===== 2. 输入 Schema =====
class WeatherInput(BaseModel):
    location: str = Field(description="City name")
    units: Literal["metric", "imperial"] = Field(default="metric")


# ===== 3. 带错误处理的工具 =====
@tool(args_schema=WeatherInput)
async def get_weather_robust(location: str, units: str = "metric") -> str:
    """Get current weather with robust error handling.

    Features:
    - Automatic retry on failure
    - Timeout protection
    - Detailed error messages
    """
    api_key = os.getenv("WEATHER_API_KEY", "demo_key")

    # 重试逻辑
    for attempt in range(1, WeatherAPIConfig.MAX_RETRIES + 1):
        try:
            async with httpx.AsyncClient(timeout=WeatherAPIConfig.TIMEOUT) as client:
                # 实际 API 调用（使用 OpenWeatherMap）
                url = f"{WeatherAPIConfig.BASE_URL}/weather"
                params = {
                    "q": location,
                    "units": units,
                    "appid": api_key
                }

                response = await client.get(url, params=params)

                # 检查 HTTP 状态码
                if response.status_code == 401:
                    raise ToolException("Invalid API key. Please check WEATHER_API_KEY in .env")
                elif response.status_code == 404:
                    raise ToolException(f"City '{location}' not found")
                elif response.status_code != 200:
                    raise httpx.HTTPStatusError(
                        f"API returned status {response.status_code}",
                        request=response.request,
                        response=response
                    )

                # 解析响应
                data = response.json()
                temp = data["main"]["temp"]
                description = data["weather"][0]["description"]
                humidity = data["main"]["humidity"]

                unit_symbol = "°C" if units == "metric" else "°F"
                return (
                    f"Weather in {location}:\n"
                    f"- Temperature: {temp}{unit_symbol}\n"
                    f"- Condition: {description}\n"
                    f"- Humidity: {humidity}%"
                )

        except httpx.TimeoutException:
            if attempt < WeatherAPIConfig.MAX_RETRIES:
                print(f"Timeout on attempt {attempt}, retrying...")
                await asyncio.sleep(WeatherAPIConfig.RETRY_DELAY)
                continue
            raise ToolException(f"API timeout after {WeatherAPIConfig.MAX_RETRIES} attempts")

        except httpx.HTTPStatusError as e:
            if attempt < WeatherAPIConfig.MAX_RETRIES and e.response.status_code >= 500:
                print(f"Server error on attempt {attempt}, retrying...")
                await asyncio.sleep(WeatherAPIConfig.RETRY_DELAY)
                continue
            raise ToolException(f"API error: {str(e)}")

        except ToolException:
            # 直接抛出 ToolException（不重试）
            raise

        except Exception as e:
            raise ToolException(f"Unexpected error: {str(e)}")

    raise ToolException("Max retries exceeded")


# ===== 4. 测试错误处理 =====
async def test_error_handling():
    print("=== 测试 3：错误处理 ===\n")

    test_cases = [
        {"location": "Beijing", "units": "metric"},  # 正常情况
        {"location": "InvalidCityXYZ123", "units": "metric"},  # 城市不存在
    ]

    for i, test in enumerate(test_cases, 1):
        print(f"测试用例 {i}: {test['location']}")
        try:
            result = await get_weather_robust.ainvoke(test)
            print(f"✓ 成功: {result}\n")
        except ToolException as e:
            print(f"✗ 工具异常: {e}\n")
        except Exception as e:
            print(f"✗ 未知异常: {e}\n")


if __name__ == "__main__":
    asyncio.run(test_error_handling())
```

**错误处理策略**：

1. **ToolException**：
   - 专门用于工具错误
   - Agent 捕获后可以继续执行
   - 不会中断整个 Agent 流程

2. **重试机制**：
   - 网络错误：自动重试
   - 服务器错误（5xx）：自动重试
   - 客户端错误（4xx）：不重试

3. **超时控制**：
   - 设置合理的超时时间
   - 避免长时间等待

```python
"""
场景 2.4：将异步天气工具集成到 Agent
演示：Agent 如何调用异步工具
"""

import os
import asyncio
from langchain.tools import tool
from langchain_openai import ChatOpenAI
from langchain.agents import AgentExecutor, create_tool_calling_agent
from langchain_core.prompts import ChatPromptTemplate
from pydantic import BaseModel, Field
from typing import Literal
from dotenv import load_dotenv

load_dotenv()


# ===== 1. 定义工具 =====
class WeatherInput(BaseModel):
    location: str = Field(description="City name")
    units: Literal["celsius", "fahrenheit"] = Field(default="celsius")


@tool(args_schema=WeatherInput)
async def get_weather(location: str, units: str = "celsius") -> str:
    """Get current weather for a location."""
    await asyncio.sleep(0.3)  # 模拟 API 延迟
    temp = 22 if units == "celsius" else 72
    unit_symbol = "°C" if units == "celsius" else "°F"
    return f"Weather in {location}: {temp}{unit_symbol}, Sunny"


# ===== 2. 创建 Agent =====
llm = ChatOpenAI(
    model="gpt-4o-mini",
    temperature=0,
    api_key=os.getenv("OPENAI_API_KEY")
)

prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful weather assistant. Use the weather tool to answer questions about weather."),
    ("human", "{input}"),
    ("placeholder", "{agent_scratchpad}"),
])

tools = [get_weather]
agent = create_tool_calling_agent(llm, tools, prompt)
agent_executor = AgentExecutor(
    agent=agent,
    tools=tools,
    verbose=True,
)


# ===== 3. 测试 Agent =====
async def test_agent():
    print("=== 测试 4：Agent 调用异步工具 ===\n")

    questions = [
        "What's the weather like in Beijing?",
        "Tell me the weather in New York in Fahrenheit",
        "Compare the weather in Tokyo and London",
    ]

    for question in questions:
        print(f"问题: {question}")
        result = await agent_executor.ainvoke({"input": question})
        print(f"回答: {result['output']}\n")
        print("-" * 60 + "\n")


if __name__ == "__main__":
    asyncio.run(test_agent())
```

## 核心知识点总结

### 1. 异步工具的创建方式

```python
# 方式 1：async def + @tool
@tool
async def async_tool(param: str) -> str:
    async with httpx.AsyncClient() as client:
        response = await client.get(f"https://api.example.com?q={param}")
        return response.text

# 方式 2：StructuredTool.from_function
from langchain.tools import StructuredTool

async def async_func(param: str) -> str:
    # ... async implementation

tool = StructuredTool.from_function(
    name="async_tool",
    description="Async tool",
    coroutine=async_func,  # 注意：使用 coroutine 参数
)
```

### 2. 异步工具的调用方式

```python
# 同步调用（自动回退到线程池）
result = tool.invoke({"param": "value"})

# 异步调用（推荐）
result = await tool.ainvoke({"param": "value"})

# 并发调用多个工具
results = await asyncio.gather(
    tool1.ainvoke({"param": "value1"}),
    tool2.ainvoke({"param": "value2"}),
)
```

### 3. 错误处理最佳实践

```python
from langchain.tools import ToolException

@tool
async def safe_tool(param: str) -> str:
    try:
        # 业务逻辑
        result = await external_api_call(param)
        return result
    except APIError as e:
        # 抛出 ToolException，Agent 可以继续
        raise ToolException(f"API failed: {e}")
    except Exception as e:
        # 未知错误也包装为 ToolException
        raise ToolException(f"Unexpected error: {e}")
```

## 最佳实践

### ✅ 推荐做法

1. **使用 httpx 而非 requests**
```python
# ✓ 推荐：httpx 支持异步
async with httpx.AsyncClient() as client:
    response = await client.get(url)

# ✗ 避免：requests 不支持异步
import requests
response = requests.get(url)  # 会阻塞
```

2. **设置合理的超时**
```python
async with httpx.AsyncClient(timeout=10.0) as client:
    response = await client.get(url)
```

3. **实现重试机制**
```python
for attempt in range(MAX_RETRIES):
    try:
        result = await api_call()
        return result
    except TransientError:
        if attempt < MAX_RETRIES - 1:
            await asyncio.sleep(RETRY_DELAY)
            continue
        raise
```

4. **使用环境变量管理密钥**
```python
from dotenv import load_dotenv
load_dotenv()

api_key = os.getenv("API_KEY")
if not api_key:
    raise ValueError("API_KEY not found in environment")
```

### ❌ 避免的做法

1. **在异步函数中使用同步 I/O**
```python
@tool
async def bad_tool(param: str) -> str:
    # ✗ 避免：time.sleep 会阻塞事件循环
    import time
    time.sleep(1)

    # ✓ 推荐：使用 asyncio.sleep
    await asyncio.sleep(1)
```

2. **硬编码 API 密钥**
```python
# ✗ 避免
api_key = "sk-1234567890abcdef"

# ✓ 推荐
api_key = os.getenv("API_KEY")
```

3. **忽略错误处理**
```python
# ✗ 避免
@tool
async def unsafe_tool(param: str) -> str:
    response = await client.get(url)  # 可能抛出异常
    return response.text

# ✓ 推荐
@tool
async def safe_tool(param: str) -> str:
    try:
        response = await client.get(url)
        return response.text
    except httpx.HTTPError as e:
        raise ToolException(f"API error: {e}")
```
