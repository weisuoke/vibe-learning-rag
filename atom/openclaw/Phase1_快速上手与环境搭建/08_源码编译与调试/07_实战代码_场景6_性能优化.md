### 2.2 缓存优化

**问题**：重复计算 embedding

**优化**：使用 LRU 缓存

```typescript
// src/infra/cache.ts
import LRU from 'lru-cache'

// Embedding 缓存
const embeddingCache = new LRU<string, number[]>({
  max: 1000,
  ttl: 1000 * 60 * 60 // 1 小时
})

export async function generateEmbedding(text: string): Promise<number[]> {
  // 检查缓存
  const cached = embeddingCache.get(text)
  if (cached) {
    return cached
  }

  // 生成 embedding
  const embedding = await openai.embeddings.create({
    model: 'text-embedding-3-small',
    input: text
  })

  const vector = embedding.data[0].embedding

  // 缓存结果
  embeddingCache.set(text, vector)

  return vector
}
```

```typescript
const pool = new Pool({
```

```typescript
export async function processLargeFile(path: string) {
```

### 3.3 对象池

**问题**：频繁创建和销毁对象

**优化**：使用对象池

```typescript
// src/infra/pool.ts
class ObjectPool<T> {
  private pool: T[] = []
  private factory: () => T
  private reset: (obj: T) => void

  constructor(factory: () => T, reset: (obj: T) => void, size: number) {
    this.factory = factory
    this.reset = reset

    // 预创建对象
    for (let i = 0; i < size; i++) {
      this.pool.push(factory())
    }
  }

  acquire(): T {
    return this.pool.pop() || this.factory()
  }

  release(obj: T) {
    this.reset(obj)
    this.pool.push(obj)
  }
}

// 使用示例
const bufferPool = new ObjectPool(
  () => Buffer.alloc(1024),
  (buf) => buf.fill(0),
  10
)

// 获取 buffer
const buf = bufferPool.acquire()

// 使用 buffer
// ...

// 释放 buffer
bufferPool.release(buf)
```

### 4.3 请求合并

**问题**：多个小请求

**优化**：合并为批量请求

```typescript
// src/infra/batch.ts
class BatchProcessor<T, R> {
  private queue: Array<{
    input: T
    resolve: (result: R) => void
    reject: (error: Error) => void
  }> = []

  private timer: NodeJS.Timeout | null = null
  private processor: (inputs: T[]) => Promise<R[]>

  constructor(processor: (inputs: T[]) => Promise<R[]>, delay: number = 10) {
    this.processor = processor
  }

  async process(input: T): Promise<R> {
    return new Promise((resolve, reject) => {
      this.queue.push({ input, resolve, reject })

      if (!this.timer) {
        this.timer = setTimeout(() => this.flush(), 10)
      }
    })
  }

  private async flush() {
    const batch = this.queue.splice(0)
    this.timer = null

    try {
      const inputs = batch.map(item => item.input)
      const results = await this.processor(inputs)

      batch.forEach((item, index) => {
        item.resolve(results[index])
      })
    } catch (error) {
      batch.forEach(item => {
        item.reject(error as Error)
      })
    }
  }
}

// 使用示例
const embeddingBatch = new BatchProcessor(
  async (texts: string[]) => {
    const response = await openai.embeddings.create({
      model: 'text-embedding-3-small',
      input: texts
    })
    return response.data.map(d => d.embedding)
  }
)

// 单个请求会自动合并
const embedding1 = await embeddingBatch.process('text1')
const embedding2 = await embeddingBatch.process('text2')
const embedding3 = await embeddingBatch.process('text3')
// 实际只发送一个批量请求
```

### 5.2 查询优化

```typescript
// 优化前：N+1 查询
export async function getSessions(userId: string) {
  const sessions = await db.query(
    'SELECT * FROM sessions WHERE user_id = $1',
    [userId]
  )

  for (const session of sessions) {
    session.messages = await db.query(
      'SELECT * FROM messages WHERE session_id = $1',
      [session.id]
    )
  }

  return sessions
}

// 优化后：JOIN 查询
export async function getSessionsOptimized(userId: string) {
  const result = await db.query(`
    SELECT
      s.*,
      json_agg(m.*) as messages
    FROM sessions s
    LEFT JOIN messages m ON m.session_id = s.id
    WHERE s.user_id = $1
    GROUP BY s.id
  `, [userId])

  return result.rows
}
```

## 核心洞察

### 1. 性能优化需要数据支持

**表面**：感觉慢就优化

**实际**：
- 使用 profiler 找到瓶颈
- 测量优化前后的性能
- 避免过早优化
- 关注关键路径

---

### 2. 缓存是性能优化的关键

**表面**：缓存所有东西

**实际**：
- 缓存计算密集型操作
- 设置合理的 TTL
- 监控缓存命中率
- 避免缓存雪崩

---

### 3. 并发不是银弹

**表面**：并发越多越快

**实际**：
- 并发需要资源
- 过度并发导致资源竞争
- 需要限流和熔断
- 监控系统负载

---

## 一句话总结

**OpenClaw 性能优化的关键是：使用 profiler 找到瓶颈，启用缓存和连接池，优化数据库查询，监控关键指标。**
