# 实战代码场景4：Int8向量实战

> 本文档演示 Milvus 2.6 Int8 向量的量化、创建、搜索和性能对比

---

## 一、场景描述

### 1.1 业务需求

**场景**：RAG 知识库存储百万级文档向量，内存成本过高，需要通过 Int8 量化降低存储成本。

**核心挑战**：
- Float32 向量占用大量内存（100万条 768维 = 2.87GB）
- 需要在内存节省和检索质量之间找到平衡
- 量化过程不能简单截断，需要保留向量间的相对关系

**业务价值**：
- 内存节省 75%（2.87GB → 0.72GB）
- QPS 提升约 4 倍
- 召回率下降 5-10%（大多数 RAG 场景可接受）

### 1.2 技术目标

1. **实现 SQ8 标量量化**：Float32 → Int8 的正确量化方法
2. **创建 Int8 向量 Collection**：正确配置维度和索引
3. **性能对比测试**：Float32 vs Int8 的内存和搜索对比
4. **生产环境注意事项**：维度限制、精度评估

### 1.3 预期效果

- 量化后向量保留原始向量的相对关系
- Int8 Collection 创建和搜索正常工作
- 内存占用降低约 75%

---

## 二、技术方案

### 2.1 SQ8 量化原理

```
Float32 向量: [0.123, -0.456, 0.789, ...]
                    │
                    ▼ SQ8 标量量化
            ┌───────────────────┐
            │ 1. 找每维度min/max │
            │ 2. 线性映射到[-128,127] │
            │ 3. 四舍五入取整    │
            └───────────────────┘
                    │
                    ▼
Int8 向量:   [12, -46, 79, ...]

内存: 4 bytes/dim → 1 byte/dim（节省75%）
```

### 2.2 关键限制

- **维度必须是 8 的倍数**（SIMD 优化要求，Issue #43466）
- 支持的度量类型：COSINE、L2、IP
- 不能与动态 Schema 的向量字段混用（Int8 是固定字段）

## 三、完整实战代码

```python
"""
Int8 向量实战
演示：SQ8量化、Collection创建、搜索对比
"""

import numpy as np
import time
from pymilvus import MilvusClient, DataType

# ===== 配置 =====
MILVUS_URI = "http://localhost:19530"
DIM = 128          # 维度（必须是8的倍数）
NUM_VECTORS = 10000  # 向量数量
TOP_K = 10

client = MilvusClient(MILVUS_URI)


# ===== 1. SQ8 标量量化实现 =====
print("=" * 60)
print("1. SQ8 标量量化")
print("=" * 60)


def sq8_quantize(vectors: np.ndarray) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
    """
    SQ8 标量量化：将 Float32 向量线性映射到 Int8 范围

    返回：(量化后的Int8向量, 每维度min值, 每维度scale值)
    """
    # 计算每个维度的 min 和 max
    v_min = vectors.min(axis=0)  # shape: (dim,)
    v_max = vectors.max(axis=0)  # shape: (dim,)

    # 计算缩放因子
    scale = (v_max - v_min) / 255.0
    scale[scale == 0] = 1.0  # 避免除零

    # 线性映射到 [0, 255]，再偏移到 [-128, 127]
    quantized = np.round((vectors - v_min) / scale - 128).astype(np.int8)

    return quantized, v_min, scale


def sq8_dequantize(quantized: np.ndarray, v_min: np.ndarray, scale: np.ndarray) -> np.ndarray:
    """反量化：Int8 → Float32（用于验证精度损失）"""
    return (quantized.astype(np.float32) + 128) * scale + v_min


# 生成测试向量
np.random.seed(42)
float32_vectors = np.random.randn(NUM_VECTORS, DIM).astype(np.float32)

# 量化
int8_vectors, v_min, scale = sq8_quantize(float32_vectors)

# 验证精度
reconstructed = sq8_dequantize(int8_vectors, v_min, scale)
mse = np.mean((float32_vectors - reconstructed) ** 2)
print(f"量化前后均方误差 (MSE): {mse:.6f}")
print(f"Float32 内存: {float32_vectors.nbytes / 1024 / 1024:.2f} MB")
print(f"Int8 内存:    {int8_vectors.nbytes / 1024 / 1024:.2f} MB")
print(f"内存节省:     {(1 - int8_vectors.nbytes / float32_vectors.nbytes) * 100:.0f}%")


# ===== 2. 创建 Float32 Collection（对照组） =====
print("\n" + "=" * 60)
print("2. 创建 Float32 Collection（对照组）")
print("=" * 60)

float32_collection = "int8_demo_float32"
if client.has_collection(float32_collection):
    client.drop_collection(float32_collection)

schema_f32 = client.create_schema(auto_id=False)
schema_f32.add_field("id", DataType.INT64, is_primary=True)
schema_f32.add_field("vector", DataType.FLOAT_VECTOR, dim=DIM)
schema_f32.add_field("text", DataType.VARCHAR, max_length=256)

index_params_f32 = client.prepare_index_params()
index_params_f32.add_index(field_name="vector", index_type="HNSW",
                           metric_type="COSINE", params={"M": 16, "efConstruction": 200})

client.create_collection(float32_collection, schema=schema_f32, index_params=index_params_f32)

# 插入 Float32 数据
f32_data = [
    {"id": i, "vector": float32_vectors[i].tolist(), "text": f"doc_{i}"}
    for i in range(NUM_VECTORS)
]
client.insert(float32_collection, f32_data)
print(f"Float32 Collection 插入 {NUM_VECTORS} 条数据")


# ===== 3. 创建 Int8 Collection =====
print("\n" + "=" * 60)
print("3. 创建 Int8 Collection")
print("=" * 60)

int8_collection = "int8_demo_int8"
if client.has_collection(int8_collection):
    client.drop_collection(int8_collection)

schema_i8 = client.create_schema(auto_id=False)
schema_i8.add_field("id", DataType.INT64, is_primary=True)
schema_i8.add_field("vector", DataType.INT8_VECTOR, dim=DIM)  # Int8向量
schema_i8.add_field("text", DataType.VARCHAR, max_length=256)

index_params_i8 = client.prepare_index_params()
index_params_i8.add_index(field_name="vector", index_type="HNSW",
                          metric_type="COSINE", params={"M": 16, "efConstruction": 200})

client.create_collection(int8_collection, schema=schema_i8, index_params=index_params_i8)

# 插入 Int8 数据
i8_data = [
    {"id": i, "vector": int8_vectors[i].tolist(), "text": f"doc_{i}"}
    for i in range(NUM_VECTORS)
]
client.insert(int8_collection, i8_data)
print(f"Int8 Collection 插入 {NUM_VECTORS} 条数据")


# ===== 4. 搜索对比 =====
print("\n" + "=" * 60)
print("4. 搜索对比")
print("=" * 60)

# 生成查询向量
query_float32 = np.random.randn(1, DIM).astype(np.float32)
query_int8, _, _ = sq8_quantize(query_float32)

# Float32 搜索
start = time.time()
results_f32 = client.search(
    float32_collection, data=query_float32.tolist(),
    limit=TOP_K, output_fields=["text"],
    search_params={"metric_type": "COSINE", "params": {"ef": 100}}
)
time_f32 = time.time() - start

# Int8 搜索
start = time.time()
results_i8 = client.search(
    int8_collection, data=query_int8.tolist(),
    limit=TOP_K, output_fields=["text"],
    search_params={"metric_type": "COSINE", "params": {"ef": 100}}
)
time_i8 = time.time() - start

print(f"Float32 搜索耗时: {time_f32*1000:.2f} ms")
print(f"Int8 搜索耗时:    {time_i8*1000:.2f} ms")

# 对比结果
f32_ids = set(r["id"] for r in results_f32[0])
i8_ids = set(r["id"] for r in results_i8[0])
overlap = len(f32_ids & i8_ids)
print(f"\nTop-{TOP_K} 结果重叠: {overlap}/{TOP_K} ({overlap/TOP_K*100:.0f}%)")

print("\nFloat32 Top-5 结果:")
for r in results_f32[0][:5]:
    print(f"  id={r['id']}, distance={r['distance']:.4f}")

print("\nInt8 Top-5 结果:")
for r in results_i8[0][:5]:
    print(f"  id={r['id']}, distance={r['distance']:.4f}")


# ===== 5. 清理 =====
print("\n" + "=" * 60)
print("5. 总结")
print("=" * 60)

print(f"""
┌─────────────────────────────────────────┐
│           Int8 vs Float32 对比           │
├──────────────┬──────────┬───────────────┤
│ 指标         │ Float32  │ Int8          │
├──────────────┼──────────┼───────────────┤
│ 每维度字节数 │ 4 bytes  │ 1 byte        │
│ {DIM}维向量  │ {DIM*4} B│ {DIM} B       │
│ {NUM_VECTORS}条内存│ {float32_vectors.nbytes/1024/1024:.1f} MB │ {int8_vectors.nbytes/1024/1024:.1f} MB │
│ 内存节省     │ -        │ 75%           │
│ 搜索耗时     │ {time_f32*1000:.1f}ms │ {time_i8*1000:.1f}ms │
│ Top-{TOP_K}重叠│ -      │ {overlap/TOP_K*100:.0f}%  │
└──────────────┴──────────┴───────────────┘
""")

# 清理
client.drop_collection(float32_collection)
client.drop_collection(int8_collection)
print("✅ 测试完成，Collection 已清理")
```

## 五、生产环境注意事项

### 5.1 维度对齐

```python
# ✅ 正确：维度是 8 的倍数
dim = 128   # 8 × 16
dim = 256   # 8 × 32
dim = 768   # 8 × 96
dim = 1024  # 8 × 128

# ❌ 错误：维度不是 8 的倍数
dim = 100   # 报错
dim = 300   # 报错

# 解决方案：填充到最近的 8 的倍数
def align_dim(dim: int) -> int:
    return ((dim + 7) // 8) * 8

align_dim(100)  # → 104
align_dim(300)  # → 304
```

### 5.2 量化质量评估

```python
def evaluate_quantization(original: np.ndarray, quantized_reconstructed: np.ndarray, k: int = 10):
    """评估量化对检索质量的影响"""
    from sklearn.metrics.pairwise import cosine_similarity

    # 随机选择查询向量
    query_idx = np.random.choice(len(original), 100)

    recall_sum = 0
    for idx in query_idx:
        # 原始向量的 Top-K
        sim_orig = cosine_similarity([original[idx]], original)[0]
        top_k_orig = set(np.argsort(sim_orig)[-k-1:-1])

        # 量化后的 Top-K
        sim_quant = cosine_similarity([quantized_reconstructed[idx]], quantized_reconstructed)[0]
        top_k_quant = set(np.argsort(sim_quant)[-k-1:-1])

        recall_sum += len(top_k_orig & top_k_quant) / k

    recall = recall_sum / len(query_idx)
    print(f"Recall@{k}: {recall:.4f} ({recall*100:.1f}%)")
    return recall
```
