# 图可视化 - 实战代码: 场景7 文档生成导出

## 场景描述

在项目开发完成后,需要为团队生成完整的工作流文档,包括图表、说明、代码示例等,用于知识传递和维护。

## 完整代码

```python
"""
场景7: 文档生成导出
演示: 为 LangGraph 工作流生成完整的文档,包括图表、说明、代码示例
"""

from langgraph.graph import StateGraph, START, END
from typing import TypedDict
from datetime import datetime
import os

# ===== 1. 定义状态 =====
class State(TypedDict):
    """工作流状态定义"""
    query: str
    documents: list[str]
    answer: str

# ===== 2. 创建示例工作流 =====
print("=== 创建示例工作流 ===")

def retrieve(state: State) -> State:
    """检索节点"""
    documents = [f"文档{i}: {state['query']}" for i in range(1, 4)]
    return {**state, "documents": documents}

def process(state: State) -> State:
    """处理节点"""
    return {**state, "documents": state['documents'][:2]}

def generate(state: State) -> State:
    """生成节点"""
    answer = f"基于 {len(state['documents'])} 个文档的答案"
    return {**state, "answer": answer}

workflow = StateGraph(State)
workflow.add_node("retrieve", retrieve)
workflow.add_node("process", process)
workflow.add_node("generate", generate)

workflow.add_edge(START, "retrieve")
workflow.add_edge("retrieve", "process")
workflow.add_edge("process", "generate")
workflow.add_edge("generate", END)

app = workflow.compile()
print("工作流创建完成")

# ===== 3. 生成 Mermaid 文本 =====
print("\n=== 生成 Mermaid 文本 ===")

mermaid_text = app.get_graph().draw_mermaid()
print(f"Mermaid 文本长度: {len(mermaid_text)} 字符")

# 保存 Mermaid 文本
with open("workflow.mmd", "w") as f:
    f.write(mermaid_text)
print("Mermaid 文本已保存: workflow.mmd")

# ===== 4. 生成 PNG 图像 =====
print("\n=== 生成 PNG 图像 ===")

try:
    png_data = app.get_graph().draw_mermaid_png()
    with open("workflow.png", "wb") as f:
        f.write(png_data)
    print(f"PNG 图像已保存: workflow.png ({len(png_data)} 字节)")
except Exception as e:
    print(f"PNG 生成失败: {e}")

# ===== 5. 生成 Markdown 文档 =====
print("\n=== 生成 Markdown 文档 ===")

# 分析图结构
graph = app.get_graph()
node_count = len(graph.nodes)
edge_count = len(graph.edges)
conditional_edges = [e for e in graph.edges if e.conditional]

# 生成文档
markdown_doc = f"""# 工作流文档

**生成时间**: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}

## 概述

本文档描述了一个 LangGraph 工作流的结构和使用方法。

## 工作流结构

### 图表

```mermaid
{mermaid_text}
```

### 统计信息

- **节点数量**: {node_count}
- **边数量**: {edge_count}
- **条件边数量**: {len(conditional_edges)}

### 节点列表

"""

# 添加节点说明
for node in graph.nodes:
    if node.id not in ("__start__", "__end__"):
        markdown_doc += f"- **{node.id}**: 工作流节点\n"

markdown_doc += f"""

### 边列表

"""

# 添加边说明
for edge in graph.edges:
    edge_type = "条件边" if edge.conditional else "普通边"
    label = f" ({edge.data})" if edge.data else ""
    markdown_doc += f"- `{edge.source}` → `{edge.target}` [{edge_type}]{label}\n"

markdown_doc += f"""

## 使用方法

### 安装依赖

```bash
pip install langgraph
```

### 基础使用

```python

from typing import TypedDict

# 定义状态
class State(TypedDict):
    query: str
    documents: list[str]
    answer: str

# 创建工作流
workflow = StateGraph(State)
workflow.add_node("retrieve", retrieve)
workflow.add_node("process", process)
workflow.add_node("generate", generate)

workflow.add_edge(START, "retrieve")
workflow.add_edge("retrieve", "process")
workflow.add_edge("process", "generate")
workflow.add_edge("generate", END)

# 编译并执行
app = workflow.compile()
result = app.invoke({{"query": "test", "documents": [], "answer": ""}})
```

### 可视化

## 节点说明

### retrieve 节点

**功能**: 检索相关文档

**输入**: 
- `query`: 查询文本

**输出**:
- `documents`: 检索到的文档列表

### process 节点

**功能**: 处理和过滤文档

**输入**:
- `documents`: 原始文档列表

**输出**:
- `documents`: 处理后的文档列表

### generate 节点

**功能**: 基于文档生成答案

**输入**:
- `documents`: 处理后的文档列表

**输出**:
- `answer`: 生成的答案

## 配置选项

import zipfile

def create_documentation_package(app, package_name="workflow_docs.zip"):
    """创建文档包"""
    # 生成所有文档
    generate_documentation(app, "temp_docs")
    
    # 创建 ZIP 包
    with zipfile.ZipFile(package_name, 'w', zipfile.ZIP_DEFLATED) as zipf:
        for root, dirs, files in os.walk("temp_docs"):
            for file in files:
                file_path = os.path.join(root, file)
                arcname = os.path.relpath(file_path, "temp_docs")
                zipf.write(file_path, arcname)
    
    print(f"文档包已创建: {package_name}")
    
    # 清理临时文件
    import shutil
    shutil.rmtree("temp_docs")
    print("临时文件已清理")
