---

## 场景描述

**场景**：构建一个自适应 RAG 系统，根据文档相关性和生成质量动态调整执行路径。

**特点**：
- 多层条件路由
- 状态驱动决策
- 循环与回退机制
- 适合生产级 RAG 系统

## 完整代码

```python
"""
LangGraph 实战代码 - 场景4：复杂状态路由
演示：构建自适应 RAG 系统
"""

from langgraph.graph import StateGraph, START, END
from typing import Literal
from typing_extensions import TypedDict


# ===== 1. 定义状态 =====
class RAGState(TypedDict):
    """RAG 系统状态"""
    query: str                  # 用户查询
    documents: list[str]        # 检索到的文档
    relevance_score: float      # 文档相关性分数
    answer: str                 # 生成的答案
    quality_score: float        # 答案质量分数
    retry_count: int            # 重试次数
    route_history: list[str]    # 路由历史
    log: list[str]              # 操作日志


# ===== 2. 定义节点函数 =====
def retrieve_documents(state: RAGState) -> dict:
    """检索文档"""
    print(f"[检索文档] 查询: {state['query']}")

    # 模拟文档检索
    documents = [
        f"文档1: 关于 {state['query']} 的相关内容...",
        f"文档2: {state['query']} 的详细说明...",
        f"文档3: {state['query']} 的补充信息..."
    ]

    print(f"[检索文档] 检索到 {len(documents)} 个文档")

    return {
        "documents": documents,
        "route_history": state.get("route_history", []) + ["retrieve"],
        "log": state.get("log", []) + ["文档检索完成"]
    }


def grade_documents(state: RAGState) -> dict:
    """评估文档相关性"""
    print("[评估文档] 开始评估文档相关性...")

    # 模拟相关性评分（基于查询长度和文档数量）
    query_length = len(state["query"])
    doc_count = len(state["documents"])

    relevance_score = min(1.0, (query_length * doc_count) / 100)

    print(f"[评估文档] 相关性分数: {relevance_score:.2f}")

    return {
        "relevance_score": relevance_score,
        "route_history": state.get("route_history", []) + ["grade"],
        "log": state.get("log", []) + [f"文档评估完成，相关性: {relevance_score:.2f}"]
    }


def transform_query(state: RAGState) -> dict:
    """转换查询"""
    print("[转换查询] 文档相关性低，转换查询...")

    # 模拟查询转换
    original_query = state["query"]
    transformed_query = f"{original_query} (改写)"

    print(f"[转换查询] 原查询: {original_query}")
    print(f"[转换查询] 新查询: {transformed_query}")

    return {
        "query": transformed_query,
        "retry_count": state.get("retry_count", 0) + 1,
        "route_history": state.get("route_history", []) + ["transform"],
        "log": state.get("log", []) + ["查询已转换"]
    }


def web_search(state: RAGState) -> dict:
    """网络搜索"""
    print("[网络搜索] 文档不足，进行网络搜索...")

    # 模拟网络搜索
    web_docs = [
        f"网络文档1: {state['query']} 的最新信息...",
        f"网络文档2: {state['query']} 的权威解释..."
    ]

    print(f"[网络搜索] 搜索到 {len(web_docs)} 个网络文档")

    return {
        "documents": state.get("documents", []) + web_docs,
        "route_history": state.get("route_history", []) + ["web_search"],
        "log": state.get("log", []) + ["网络搜索完成"]
    }


def generate_answer(state: RAGState) -> dict:
    """生成答案"""
    print("[生成答案] 开始生成答案...")

    # 模拟答案生成
    doc_count = len(state["documents"])
    answer = f"基于 {doc_count} 个文档，{state['query']} 的答案是..."

    # 模拟质量评分
    quality_score = min(1.0, doc_count / 5)

    print(f"[生成答案] 答案: {answer}")
    print(f"[生成答案] 质量分数: {quality_score:.2f}")

    return {
        "answer": answer,
        "quality_score": quality_score,
        "route_history": state.get("route_history", []) + ["generate"],
        "log": state.get("log", []) + [f"答案生成完成，质量: {quality_score:.2f}"]
    }


# ===== 3. 定义路由函数 =====
def route_after_grading(state: RAGState) -> Literal["transform_query", "generate", END]:
    """评估后的路由决策"""
    relevance_score = state["relevance_score"]
    retry_count = state.get("retry_count", 0)

    print(f"[路由决策] 相关性: {relevance_score:.2f}, 重试次数: {retry_count}")

    # 检查重试次数限制
    if retry_count >= 2:
        print("[路由决策] → END (达到重试上限)")
        return END

    # 相关性高，直接生成
    if relevance_score > 0.7:
        print("[路由决策] → generate (相关性高)")
        return "generate"

    # 相关性低，转换查询
    else:
        print("[路由决策] → transform_query (相关性低)")
        return "transform_query"


def route_after_generation(state: RAGState) -> Literal["web_search", "generate", END]:
    """生成后的路由决策"""
    quality_score = state["quality_score"]
    retry_count = state.get("retry_count", 0)

    print(f"[路由决策] 质量: {quality_score:.2f}, 重试次数: {retry_count}")

    # 检查重试次数限制
    if retry_count >= 2:
        print("[路由决策] → END (达到重试上限)")
        return END

    # 质量高，结束
    if quality_score > 0.8:
        print("[路由决策] → END (质量高)")
        return END

    # 质量中等，网络搜索
    elif quality_score > 0.5:
        print("[路由决策] → web_search (质量中等)")
        return "web_search"

    # 质量低，重新生成
    else:
        print("[路由决策] → generate (质量低，重新生成)")
        return "generate"


# ===== 4. 构建图 =====
def create_adaptive_rag_graph():
    """创建自适应 RAG 图"""
    # 创建状态图
    graph = StateGraph(RAGState)

    # 添加节点
    graph.add_node("retrieve", retrieve_documents)
    graph.add_node("grade", grade_documents)
    graph.add_node("transform", transform_query)
    graph.add_node("web_search", web_search)
    graph.add_node("generate", generate_answer)

    # 添加边
    graph.add_edge(START, "retrieve")
    graph.add_edge("retrieve", "grade")

    # 第一层条件路由：评估后的决策
    graph.add_conditional_edges(
        "grade",
        route_after_grading,
        {
            "transform_query": "transform",
            "generate": "generate",
            END: END
        }
    )

    # 转换查询后重新检索
    graph.add_edge("transform", "retrieve")

    # 第二层条件路由：生成后的决策
    graph.add_conditional_edges(
        "generate",
        route_after_generation,
        {
            "web_search": "web_search",
            "generate": "generate",
            END: END
        }
    )

    # 网络搜索后重新生成
    graph.add_edge("web_search", "generate")

    # 编译图
    app = graph.compile()

    return app


# ===== 5. 测试不同场景 =====
def test_high_relevance():
    """测试高相关性场景"""
    print("=" * 60)
    print("测试场景 1：高相关性文档")
    print("=" * 60)
    print()

    app = create_adaptive_rag_graph()

    initial_state = {
        "query": "什么是 LangGraph 的条件边？这是一个详细的查询",
        "documents": [],
        "relevance_score": 0.0,
        "answer": "",
        "quality_score": 0.0,
        "retry_count": 0,
        "route_history": [],
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("最终结果：")
    print(f"  查询: {result['query']}")
    print(f"  答案: {result['answer']}")
    print(f"  路由历史: {' → '.join(result['route_history'])}")
    print()


def test_low_relevance():
    """测试低相关性场景"""
    print("=" * 60)
    print("测试场景 2：低相关性文档")
    print("=" * 60)
    print()

    app = create_adaptive_rag_graph()

    initial_state = {
        "query": "LangGraph",
        "documents": [],
        "relevance_score": 0.0,
        "answer": "",
        "quality_score": 0.0,
        "retry_count": 0,
        "route_history": [],
        "log": []
    }

    result = app.invoke(initial_state)

    print()
    print("最终结果：")
    print(f"  查询: {result['query']}")
    print(f"  答案: {result['answer']}")
    print(f"  路由历史: {' → '.join(result['route_history'])}")
    print()


# ===== 6. 主函数 =====
def main():
    """主函数"""
    print("=" * 60)
    print("LangGraph 实战 - 场景4：复杂状态路由")
    print("=" * 60)
    print()

    # 测试两种场景
    test_high_relevance()
    test_low_relevance()


if __name__ == "__main__":
    main()
```

## 代码详解

### 1. 多层条件路由

```python
# 第一层：评估后的决策
graph.add_conditional_edges(
    "grade",
    route_after_grading,
    {
        "transform_query": "transform",
        "generate": "generate",
        END: END
    }
)

# 第二层：生成后的决策
graph.add_conditional_edges(
    "generate",
    route_after_generation,
    {
        "web_search": "web_search",
        "generate": "generate",
        END: END
    }
)
```

**关键点**：
- **多层决策**：两个条件路由点
- **状态驱动**：基于 relevance_score 和 quality_score
- **循环机制**：transform → retrieve → grade 形成循环
- **退出条件**：retry_count 限制循环次数

---

### 2. 执行流程图

```
START
  ↓
retrieve (检索文档)
  ↓
grade (评估文档)
  ↓
route_after_grading
  ↓
┌─────┼─────┐
↓     ↓     ↓
transform generate END
(转换)  (生成)  (结束)
↓       ↓
retrieve route_after_generation
  ↓       ↓
grade   ┌─────┼─────┐
        ↓     ↓     ↓
    web_search generate END
    (搜索)  (重生成) (结束)
        ↓       ↓
    generate   (循环)
```

**执行路径**：
- **高相关性**：retrieve → grade → generate → web_search → generate → END
- **低相关性**：retrieve → grade → transform → retrieve → grade → ... → END

## 实战应用场景

### 场景 1：生产级 RAG 系统

```python
class ProductionRAGState(TypedDict):
    query: str
    documents: list[str]
    relevance_scores: list[float]
    answer: str
    confidence: float
    sources: list[str]
    retry_count: int

def route_production_rag(state: ProductionRAGState) -> Literal["rerank", "generate", "fallback", END]:
    """生产级 RAG 路由"""
    avg_relevance = sum(state["relevance_scores"]) / len(state["relevance_scores"])

    if avg_relevance > 0.8:
        return "generate"
    elif avg_relevance > 0.5:
        return "rerank"
    elif state["retry_count"] < 3:
        return "fallback"
    else:
        return END
```

**应用**：企业级知识库问答系统

### 场景 2：多模态 RAG

```python
class MultimodalRAGState(TypedDict):
    query: str
    text_docs: list[str]
    image_docs: list[str]
    video_docs: list[str]
    modality_scores: dict[str, float]
    answer: str

def route_multimodal(state: MultimodalRAGState) -> Literal["text_gen", "image_gen", "video_gen", "hybrid_gen"]:
    """多模态路由"""
    scores = state["modality_scores"]

    if scores["text"] > 0.8:
        return "text_gen"
    elif scores["image"] > 0.8:
        return "image_gen"
    elif scores["video"] > 0.8:
        return "video_gen"
    else:
        return "hybrid_gen"
```

**应用**：多模态内容检索系统

## 总结

**复杂状态路由的特点**：
1. **多层条件路由**：多个决策点
2. **状态驱动决策**：基于多个状态值
3. **循环与回退**：支持重试和回退
4. **退出条件**：防止无限循环

**适用场景**：
- 生产级 RAG 系统
- 自适应工作流
- 多模态处理
- 复杂决策流程

**关键代码**：
```python
# 多层条件路由
graph.add_conditional_edges("node1", route1)
graph.add_conditional_edges("node2", route2)

# 循环路由
graph.add_edge("transform", "retrieve")  # 形成循环

# 退出条件
def route_func(state):
    if state["retry_count"] >= 3:
        return END
    return "retry"
```
