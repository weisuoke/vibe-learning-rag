# 实战代码 - 场景4：本地LLM流式输出

## 场景说明

**应用场景**：使用本地 LLM（如 Llama.cpp）实现流式输出，适用于私有化部署、离线环境或对数据隐私有严格要求的场景。

**核心挑战**：
- 本地 LLM 的流式输出配置与云端 API 不同
- 需要处理模型加载、内存管理
- 异步流式输出的队列管理
- FastAPI 集成的复杂性

**RAG 开发价值**：
- 私有化 RAG 系统的核心能力
- 降低 API 调用成本
- 数据不出本地，保护隐私
- 适合企业内部知识库问答

## 完整可运行代码

### 方案1：基础本地 LLM 流式输出（Panel UI）

```python
"""
本地 LLM 流式输出 - 基础方案
演示：使用 Llama.cpp 实现本地模型的流式输出
适用场景：开发测试、本地演示
"""

import os
from operator import itemgetter
from typing import Iterator

from huggingface_hub import hf_hub_download
from langchain.memory import ConversationSummaryBufferMemory
from langchain_core.output_parsers import StrOutputParser
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.runnables import RunnableLambda, RunnablePassthrough
from langchain_community.llms.llamacpp import LlamaCpp

# ===== 1. 配置本地模型 =====
print("=== 配置本地 LLM ===")

# 从 HuggingFace 下载量化模型（GGUF 格式）
REPO_ID = "TheBloke/zephyr-7B-beta-GGUF"
FILENAME = "zephyr-7b-beta.Q5_K_M.gguf"  # Q5_K_M 量化版本，平衡质量和速度
SYSTEM_PROMPT = "你是一个专业的 RAG 系统助手，擅长回答知识库相关问题。"


def load_llm(repo_id: str = REPO_ID, filename: str = FILENAME, **kwargs):
    """加载本地 LLM 模型"""
    print(f"正在下载模型: {repo_id}/{filename}")
    model_path = hf_hub_download(repo_id=repo_id, filename=filename)
    print(f"模型路径: {model_path}")

    llm = LlamaCpp(
        model_path=model_path,
        streaming=True,  # 启用流式输出
        n_gpu_layers=1,  # 使用 GPU 加速（如果可用）
        temperature=0.75,  # 控制随机性
        max_tokens=2000,  # 最大生成长度
        top_p=1,  # 核采样参数
        verbose=False,  # 关闭详细日志
        **kwargs
    )
    return llm


# ===== 2. 初始化模型和记忆 =====
print("\n=== 初始化模型 ===")

model = load_llm()
memory = ConversationSummaryBufferMemory(
    return_messages=True,
    llm=model,
    max_token_limit=500  # 限制记忆长度
)

# ===== 3. 构建 LCEL 链 =====
print("\n=== 构建 LCEL 链 ===")

prompt = ChatPromptTemplate.from_messages([
    ("system", SYSTEM_PROMPT),
    MessagesPlaceholder(variable_name="history"),
    ("human", "{input}"),
])

output_parser = StrOutputParser()

chain = (
    RunnablePassthrough.assign(
        history=RunnableLambda(memory.load_memory_variables) | itemgetter("history")
    )
    | prompt
    | model
    | output_parser
)

print("LCEL 链构建完成")


# ===== 4. 流式输出函数 =====
def stream_chat(user_input: str) -> Iterator[str]:
    """
    流式输出聊天响应

    Args:
        user_input: 用户输入

    Yields:
        str: 每个生成的 token
    """
    print(f"\n用户: {user_input}")
    print("助手: ", end="", flush=True)

    full_response = ""
    inputs = {"input": user_input}

    # 流式生成
    for token in chain.stream(inputs):
        full_response += token
        print(token, end="", flush=True)
        yield token

    print()  # 换行

    # 保存对话历史
    memory.save_context(inputs, {"output": full_response})


# ===== 5. 测试流式输出 =====
print("\n=== 测试流式输出 ===")

# 测试问题1：基础问答
question1 = "什么是 RAG 系统？"
list(stream_chat(question1))

# 测试问题2：带上下文的问答
question2 = "它有哪些核心组件？"
list(stream_chat(question2))

# 测试问题3：技术细节
question3 = "如何优化检索性能？"
list(stream_chat(question3))


# ===== 6. 查看对话历史 =====
print("\n=== 对话历史 ===")
history = memory.load_memory_variables({})
print(f"历史消息数量: {len(history['history'])}")
for i, msg in enumerate(history['history']):
    print(f"{i+1}. {msg.type}: {msg.content[:50]}...")
```

### 方案2：FastAPI 集成本地 LLM 流式输出（生产级）

```python
"""
本地 LLM 流式输出 - FastAPI 集成方案
演示：将本地模型集成到 FastAPI，实现 SSE 流式响应
适用场景：生产环境、API 服务
"""

import asyncio
import os
from typing import AsyncIterable, Awaitable

import uvicorn
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from langchain.callbacks import AsyncIteratorCallbackHandler
from langchain_community.llms.llamacpp import LlamaCpp
from langchain.schema import HumanMessage
from pydantic import BaseModel
from huggingface_hub import hf_hub_download

app = FastAPI(title="本地 LLM 流式 API")


# ===== 1. 模型配置 =====
REPO_ID = "TheBloke/zephyr-7B-beta-GGUF"
FILENAME = "zephyr-7b-beta.Q5_K_M.gguf"


def get_model_path() -> str:
    """获取或下载模型路径"""
    return hf_hub_download(repo_id=REPO_ID, filename=FILENAME)


# ===== 2. 异步流式消息生成器 =====
async def send_message(message: str) -> AsyncIterable[str]:
    """
    异步生成流式响应

    Args:
        message: 用户消息

    Yields:
        str: SSE 格式的 token
    """
    # 创建异步回调处理器
    callback = AsyncIteratorCallbackHandler()

    # 初始化本地 LLM（每次请求新建实例，避免并发冲突）
    model = LlamaCpp(
        model_path=get_model_path(),
        streaming=True,
        verbose=False,
        callbacks=[callback],
        n_gpu_layers=1,
        temperature=0.7,
        max_tokens=1000,
    )

    async def wrap_done(fn: Awaitable, event: asyncio.Event):
        """包装异步任务，确保完成时设置事件"""
        try:
            await fn
        except Exception as e:
            print(f"生成错误: {e}")
        finally:
            # 通知迭代器停止
            event.set()

    # 在后台启动生成任务
    task = asyncio.create_task(
        wrap_done(
            model.agenerate(prompts=[message]),
            callback.done
        )
    )

    # 异步迭代生成的 token
    async for token in callback.aiter():
        # 使用 SSE 格式
        yield f"data: {token}\n\n"

    await task


# ===== 3. API 端点 =====
class StreamRequest(BaseModel):
    """流式请求体"""
    message: str
    max_tokens: int = 1000
    temperature: float = 0.7


@app.get("/health")
async def health_check():
    """健康检查"""
    return {"status": "ok", "model": FILENAME}


@app.post("/stream")
async def stream_endpoint(body: StreamRequest):
    """
    流式生成端点

    Args:
        body: 请求体，包含用户消息

    Returns:
        StreamingResponse: SSE 流式响应
    """
    return StreamingResponse(
        send_message(body.message),
        media_type="text/event-stream"
    )


@app.post("/chat")
async def chat_endpoint(body: StreamRequest):
    """
    非流式生成端点（用于对比）

    Args:
        body: 请求体

    Returns:
        dict: 完整响应
    """
    model = LlamaCpp(
        model_path=get_model_path(),
        streaming=False,
        verbose=False,
        n_gpu_layers=1,
        temperature=body.temperature,
        max_tokens=body.max_tokens,
    )

    response = model.invoke(body.message)
    return {"response": response}


# ===== 4. 启动服务 =====
if __name__ == "__main__":
    print("=== 启动本地 LLM 流式 API ===")
    print(f"模型: {FILENAME}")
    print("访问 http://localhost:8000/docs 查看 API 文档")

    uvicorn.run(
        app,
        host="0.0.0.0",
        port=8000,
        log_level="info"
    )
```

```python
"""
本地 LLM 流式输出 - 客户端测试
演示：如何调用流式 API
"""

import requests


def test_stream_api(message: str, url: str = "http://localhost:8000/stream"):
    """
    测试流式 API

    Args:
        message: 用户消息
        url: API 地址
    """
    print(f"用户: {message}")
    print("助手: ", end="", flush=True)

    # 发送流式请求
    with requests.post(
        url,
        json={"message": message},
        stream=True  # 启用流式接收
    ) as response:
        # 逐行读取 SSE 数据
        for line in response.iter_lines():
            if line:
                # 解析 SSE 格式：data: <token>
                decoded_line = line.decode("utf-8")
                if decoded_line.startswith("data: "):
                    token = decoded_line[6:]  # 去掉 "data: " 前缀
                    print(token, end="", flush=True)

    print()  # 换行


# ===== 测试用例 =====
```

## 关键技术点

### 1. 本地 LLM 配置要点

```python
# 关键参数说明
llm = LlamaCpp(
    model_path=model_path,      # 模型文件路径
    streaming=True,              # 必须启用流式输出
    n_gpu_layers=1,              # GPU 加速层数（0=纯CPU）
    temperature=0.7,             # 控制随机性（0-1）
    max_tokens=1000,             # 最大生成长度
    top_p=1,                     # 核采样参数
    verbose=False,               # 关闭详细日志
    callbacks=[callback],        # 流式回调处理器
)
```

**常见问题**：
- **内存不足**：使用更小的量化版本（Q4_K_M 而非 Q5_K_M）
- **速度慢**：增加 `n_gpu_layers` 或使用更小的模型
- **输出质量差**：调整 `temperature` 和 `top_p` 参数

### 2. 异步流式输出模式

```python
# 核心模式：AsyncIteratorCallbackHandler
callback = AsyncIteratorCallbackHandler()

# 后台任务 + 异步迭代
task = asyncio.create_task(wrap_done(model.agenerate(...), callback.done))
async for token in callback.aiter():
    yield f"data: {token}\n\n"
await task
```

**关键点**：
- 每个请求新建 `CallbackHandler` 实例（避免并发冲突）
- 使用 `wrap_done` 确保任务完成时设置事件
- SSE 格式：`data: <token>\n\n`

### 3. 模型选择建议

| 模型大小 | 量化版本 | 内存占用 | 速度 | 质量 | 适用场景 |
|---------|---------|---------|------|------|---------|
| 7B | Q4_K_M | ~4GB | 快 | 中 | 开发测试 |
| 7B | Q5_K_M | ~5GB | 中 | 高 | 生产环境 |
| 13B | Q4_K_M | ~8GB | 中 | 高 | 高质量需求 |
| 13B | Q5_K_M | ~10GB | 慢 | 很高 | 专业应用 |

### 4. 性能优化技巧

```python
# 1. 使用 GPU 加速

# 2. 批量处理
n_batch=512  # 增加批处理大小

# 3. 线程优化
```

## 与 RAG 开发的联系

**场景**：企业内部知识库，数据不能上传到云端

```python
# 完整的私有化 RAG 管道
from langchain.vectorstores import Chroma
from langchain.embeddings import HuggingFaceEmbeddings

# 本地 Embedding 模型
embeddings = HuggingFaceEmbeddings(
    model_name="sentence-transformers/all-MiniLM-L6-v2"
)

# 本地向量数据库
vectorstore = Chroma(
    persist_directory="./chroma_db",
    embedding_function=embeddings
)

llm = LlamaCpp(model_path=model_path, streaming=True)

# RAG 链
from langchain.chains import RetrievalQA
qa_chain = RetrievalQA.from_chain_type(
    llm=llm,
    retriever=vectorstore.as_retriever(),
    return_source_documents=True
)
```

- **云端 API**：$0.002/1K tokens（GPT-3.5）

**适用场景**：
- 高频调用（每天 >10万次）
- 长期运行（>6个月）
- 大规模部署

## 常见问题与解决方案

### 问题1：模型加载失败

**错误**：`OSError: [Errno 28] No space left on device`

**解决**：
```python
# 指定缓存目录
os.environ["HF_HOME"] = "/path/to/large/disk"
```

### 问题2：流式输出卡住

**原因**：回调处理器未正确设置

**解决**：
```python
# 确保使用 AsyncIteratorCallbackHandler
callback = AsyncIteratorCallbackHandler()
model = LlamaCpp(..., callbacks=[callback])

# 确保调用 callback.done.set()
finally:
    callback.done.set()
```

### 问题3：并发请求冲突

**错误**：多个请求共享同一个模型实例导致输出混乱

**解决**：
```python
# 每个请求新建模型实例
async def send_message(message: str):
    model = LlamaCpp(...)  # 新建实例
    # ...
```
