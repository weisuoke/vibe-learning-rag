# 实战代码:场景3 - 泛型类型实战

## 场景概述

本文档展示如何在 LangGraph 中使用泛型类型系统，包括 StateT、ContextT、InputT、OutputT 的实际应用，以及如何实现类型安全的节点定义。

**适用场景**：
- 需要分离输入输出类型
- 需要不可变的运行时上下文
- 需要类型安全的节点定义
- 复杂的多类型参数系统

## 核心代码示例

### 示例1：基础泛型类型定义

```python
"""
基础泛型类型定义示例
展示 StateT, InputT, OutputT 的基本用法
"""
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END

# 定义内部状态
class InternalState(TypedDict):
    """内部状态 - 用于节点间通信"""
    messages: list[str]
    context: dict
    step_count: int

# 定义输入类型
class InputState(TypedDict):
    """输入类型 - 用户提供的数据"""
    query: str
    user_id: str

# 定义输出类型
class OutputState(TypedDict):
    """输出类型 - 返回给用户的数据"""
    response: str
    metadata: dict

# 创建图 - 使用泛型类型参数
graph = StateGraph[InternalState, None, InputState, OutputState](
    state_schema=InternalState,
    input_schema=InputState,
    output_schema=OutputState
)

def process_input(state: InternalState) -> InternalState:
    """处理输入 - 节点接收内部状态"""
    print(f"处理查询: {state.get('messages', [])}")
    return {
        "messages": state.get("messages", []) + ["processed"],
        "context": state.get("context", {}),
        "step_count": state.get("step_count", 0) + 1
    }

def generate_output(state: InternalState) -> InternalState:
    """生成输出"""
    return {
        "messages": state["messages"] + ["generated"],
        "context": state["context"],
        "step_count": state["step_count"] + 1
    }

graph.add_node("process", process_input)
graph.add_node("generate", generate_output)
graph.add_edge(START, "process")
graph.add_edge("process", "generate")
graph.add_edge("generate", END)

app = graph.compile()

# 测试：输入类型自动转换为内部状态
result = app.invoke({
    "query": "Hello",
    "user_id": "user_123"
})

print(f"结果类型: {type(result)}")
print(f"结果内容: {result}")
# 输出会根据 output_schema 进行转换
```

### 示例2：context_schema 使用

```python
"""
context_schema 使用示例
展示如何使用不可变的运行时上下文
"""
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END

# 定义状态
class State(TypedDict):
    messages: list[str]
    result: str

# 定义上下文 - 不可变的运行时信息
class Context(TypedDict):
    user_id: str
    session_id: str
    api_key: str
    config: dict

# 创建图 - 使用 context_schema
graph = StateGraph[State, Context](
    state_schema=State,
    context_schema=Context
)

def process_with_context(state: State, *, config: dict) -> State:
    """处理节点 - 可以访问上下文"""
    # 从 config 中获取上下文
    context = config.get("configurable", {})
    user_id = context.get("user_id", "unknown")
    api_key = context.get("api_key", "")

    print(f"用户ID: {user_id}")
    print(f"API密钥: {api_key[:10]}...")

    return {
        "messages": state["messages"] + [f"processed by {user_id}"],
        "result": state.get("result", "")
    }

def generate_with_context(state: State, *, config: dict) -> State:
    """生成节点 - 使用上下文信息"""
    context = config.get("configurable", {})
    session_id = context.get("session_id", "unknown")

    return {
        "messages": state["messages"],
        "result": f"Generated for session {session_id}"
    }

graph.add_node("process", process_with_context)
graph.add_node("generate", generate_with_context)
graph.add_edge(START, "process")
graph.add_edge("process", "generate")
graph.add_edge("generate", END)

app = graph.compile()

# 测试：传递上下文
result = app.invoke(
    {"messages": ["Hello"], "result": ""},
    config={
        "configurable": {
            "user_id": "user_123",
            "session_id": "session_456",
            "api_key": "sk-1234567890abcdef",
            "config": {"temperature": 0.7}
        }
    }
)

print(f"最终结果: {result}")
```

### 示例3：input_schema 和 output_schema 分离

```python
"""
input_schema 和 output_schema 分离示例
展示如何定义不同的输入输出类型
"""
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from datetime import datetime

# 内部状态 - 完整的状态信息
class InternalState(TypedDict):
    raw_query: str
    processed_query: str
    retrieved_docs: list[str]
    response: str
    metadata: dict
    timestamp: str

# 输入状态 - 用户提供的最小信息
class InputState(TypedDict):
    query: str

# 输出状态 - 返回给用户的信息
class OutputState(TypedDict):
    response: str
    sources: list[str]

# 创建图
graph = StateGraph[InternalState, None, InputState, OutputState](
    state_schema=InternalState,
    input_schema=InputState,
    output_schema=OutputState
)

def preprocess(state: InternalState) -> InternalState:
    """预处理 - 扩展输入状态"""
    return {
        "raw_query": state.get("raw_query", ""),
        "processed_query": state["raw_query"].strip().lower(),
        "retrieved_docs": [],
        "response": "",
        "metadata": {},
        "timestamp": datetime.now().isoformat()
    }

def retrieve(state: InternalState) -> InternalState:
    """检索文档"""
    docs = [
        f"文档1: 关于 {state['processed_query']}",
        f"文档2: 更多关于 {state['processed_query']}"
    ]
    return {
        "raw_query": state["raw_query"],
        "processed_query": state["processed_query"],
        "retrieved_docs": docs,
        "response": state["response"],
        "metadata": state["metadata"],
        "timestamp": state["timestamp"]
    }

def generate(state: InternalState) -> InternalState:
    """生成回答"""
    response = f"基于 {len(state['retrieved_docs'])} 个文档的回答"
    return {
        "raw_query": state["raw_query"],
        "processed_query": state["processed_query"],
        "retrieved_docs": state["retrieved_docs"],
        "response": response,
        "metadata": {"doc_count": len(state["retrieved_docs"])},
        "timestamp": state["timestamp"]
    }

graph.add_node("preprocess", preprocess)
graph.add_node("retrieve", retrieve)
graph.add_node("generate", generate)
graph.add_edge(START, "preprocess")
graph.add_edge("preprocess", "retrieve")
graph.add_edge("retrieve", "generate")
graph.add_edge("generate", END)

app = graph.compile()

# 测试：输入只需要 query，输出只包含 response 和 sources
result = app.invoke({"query": "LangGraph 是什么"})

print(f"输入类型: InputState (只有 query)")
print(f"输出类型: OutputState (只有 response 和 sources)")
print(f"实际输出: {result}")
# 注意：输出会根据 output_schema 进行过滤
```

### 示例4：类型安全的节点定义

```python
"""
类型安全的节点定义示例
展示如何使用类型推断和显式类型定义
"""
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END

# 定义多个状态类型
class BaseState(TypedDict):
    """基础状态"""
    id: str
    status: str

class ProcessingState(TypedDict):
    """处理状态"""
    id: str
    status: str
    data: list[str]

class ResultState(TypedDict):
    """结果状态"""
    id: str
    status: str
    result: str

# 创建图
graph = StateGraph[BaseState](state_schema=BaseState)

# 方式1：类型推断 - 从函数签名推断输入类型
def process_node(state: ProcessingState) -> ProcessingState:
    """处理节点 - 自动推断输入类型为 ProcessingState"""
    print(f"处理数据: {state.get('data', [])}")
    return {
        "id": state["id"],
        "status": "processing",
        "data": state.get("data", []) + ["processed"]
    }

# 方式2：显式指定 input_schema
def result_node(state: BaseState) -> ResultState:
    """结果节点 - 显式指定输入类型"""
    return {
        "id": state["id"],
        "status": "completed",
        "result": "success"
    }

# 添加节点 - 自动推断类型
graph.add_node("process", process_node)

# 添加节点 - 显式指定类型
graph.add_node("result", result_node, input_schema=ResultState)

graph.add_edge(START, "process")
graph.add_edge("process", "result")
graph.add_edge("result", END)

app = graph.compile()

# 测试
result = app.invoke({
    "id": "task_123",
    "status": "pending"
})

print(f"最终状态: {result}")
```

### 示例5：完整的泛型类型系统

```python
"""
完整的泛型类型系统示例
展示 StateT, ContextT, InputT, OutputT 的综合应用
"""
from typing_extensions import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END

# 定义所有类型
class InternalState(TypedDict):
    """内部状态 - 完整的状态信息"""
    query: str
    documents: list[str]
    response: str
    metadata: dict

class Context(TypedDict):
    """上下文 - 不可变的运行时信息"""
    user_id: str
    api_key: str
    model: str

class Input(TypedDict):
    """输入 - 用户提供的数据"""
    question: str

class Output(TypedDict):
    """输出 - 返回给用户的数据"""
    answer: str
    confidence: float

# 创建图 - 使用所有泛型参数
graph = StateGraph[InternalState, Context, Input, Output](
    state_schema=InternalState,
    context_schema=Context,
    input_schema=Input,
    output_schema=Output
)

def retrieve_docs(state: InternalState, *, config: dict) -> InternalState:
    """检索文档 - 使用上下文"""
    context = config.get("configurable", {})
    user_id = context.get("user_id", "unknown")

    print(f"为用户 {user_id} 检索文档")

    docs = [
        f"文档1: 关于 {state['query']}",
        f"文档2: 更多信息"
    ]

    return {
        "query": state["query"],
        "documents": docs,
        "response": "",
        "metadata": {"user_id": user_id}
    }

def generate_response(state: InternalState, *, config: dict) -> InternalState:
    """生成回答 - 使用上下文"""
    context = config.get("configurable", {})
    model = context.get("model", "default")

    response = f"使用 {model} 模型生成的回答，基于 {len(state['documents'])} 个文档"

    return {
        "query": state["query"],
        "documents": state["documents"],
        "response": response,
        "metadata": {**state["metadata"], "model": model}
    }

graph.add_node("retrieve", retrieve_docs)
graph.add_node("generate", generate_response)
graph.add_edge(START, "retrieve")
graph.add_edge("retrieve", "generate")
graph.add_edge("generate", END)

app = graph.compile()

# 测试：完整的类型系统
result = app.invoke(
    # 输入类型：Input
    {"question": "什么是 LangGraph"},
    # 上下文：Context
    config={
        "configurable": {
            "user_id": "user_123",
            "api_key": "sk-xxx",
            "model": "gpt-4"
        }
    }
)

print(f"输入类型: Input (question)")
print(f"输出类型: Output (answer, confidence)")
print(f"实际输出: {result}")
```

## 完整实战：RAG 系统的泛型类型设计

```python
"""
完整实战：RAG 系统的泛型类型设计
展示在实际 RAG 系统中如何使用泛型类型系统
"""
from typing_extensions import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
from typing import Literal

# 内部状态 - 完整的 RAG 流程状态
class RAGInternalState(TypedDict):
    """RAG 内部状态"""
    original_query: str
    processed_query: str
    retrieved_docs: list[dict]
    reranked_docs: list[dict]
    context: str
    response: str
    metadata: dict
    step: Literal["init", "retrieve", "rerank", "generate", "done"]

# 上下文 - 不可变的配置信息
class RAGContext(TypedDict):
    """RAG 上下文"""
    user_id: str
    session_id: str
    api_key: str
    embedding_model: str
    llm_model: str
    top_k: int
    temperature: float

# 输入 - 用户查询
class RAGInput(TypedDict):
    """RAG 输入"""
    query: str
    filters: dict  # 可选的过滤条件

# 输出 - 返回结果
class RAGOutput(TypedDict):
    """RAG 输出"""
    answer: str
    sources: list[dict]
    confidence: float
    metadata: dict

# 创建 RAG 图
graph = StateGraph[RAGInternalState, RAGContext, RAGInput, RAGOutput](
    state_schema=RAGInternalState,
    context_schema=RAGContext,
    input_schema=RAGInput,
    output_schema=RAGOutput
)

def preprocess_query(state: RAGInternalState, *, config: dict) -> RAGInternalState:
    """预处理查询"""
    context = config.get("configurable", {})
    user_id = context.get("user_id", "unknown")

    processed = state["original_query"].strip().lower()

    return {
        "original_query": state["original_query"],
        "processed_query": processed,
        "retrieved_docs": [],
        "reranked_docs": [],
        "context": "",
        "response": "",
        "metadata": {"user_id": user_id},
        "step": "retrieve"
    }

def retrieve_documents(state: RAGInternalState, *, config: dict) -> RAGInternalState:
    """检索文档"""
    context = config.get("configurable", {})
    top_k = context.get("top_k", 5)
    embedding_model = context.get("embedding_model", "default")

    print(f"使用 {embedding_model} 检索 top {top_k} 文档")

    # 模拟检索
    docs = [
        {"id": i, "content": f"文档{i}: 关于 {state['processed_query']}", "score": 0.9 - i * 0.1}
        for i in range(top_k)
    ]

    return {
        **state,
        "retrieved_docs": docs,
        "step": "rerank"
    }

def rerank_documents(state: RAGInternalState, *, config: dict) -> RAGInternalState:
    """重排序文档"""
    # 模拟重排序
    reranked = sorted(state["retrieved_docs"], key=lambda x: x["score"], reverse=True)[:3]

    return {
        **state,
        "reranked_docs": reranked,
        "step": "generate"
    }

def generate_answer(state: RAGInternalState, *, config: dict) -> RAGInternalState:
    """生成回答"""
    context = config.get("configurable", {})
    llm_model = context.get("llm_model", "default")
    temperature = context.get("temperature", 0.7)

    # 构建上下文
    context_str = "\n".join([doc["content"] for doc in state["reranked_docs"]])

    # 模拟生成
    response = f"基于 {len(state['reranked_docs'])} 个文档生成的回答 (模型: {llm_model}, 温度: {temperature})"

    return {
        **state,
        "context": context_str,
        "response": response,
        "metadata": {
            **state["metadata"],
            "model": llm_model,
            "doc_count": len(state["reranked_docs"])
        },
        "step": "done"
    }

# 构建图
graph.add_node("preprocess", preprocess_query)
graph.add_node("retrieve", retrieve_documents)
graph.add_node("rerank", rerank_documents)
graph.add_node("generate", generate_answer)

graph.add_edge(START, "preprocess")
graph.add_edge("preprocess", "retrieve")
graph.add_edge("retrieve", "rerank")
graph.add_edge("rerank", "generate")
graph.add_edge("generate", END)

app = graph.compile()

# 测试：完整的 RAG 流程
result = app.invoke(
    # 输入：RAGInput
    {
        "query": "什么是 LangGraph 的状态管理",
        "filters": {"category": "documentation"}
    },
    # 上下文：RAGContext
    config={
        "configurable": {
            "user_id": "user_123",
            "session_id": "session_456",
            "api_key": "sk-xxx",
            "embedding_model": "text-embedding-3-small",
            "llm_model": "gpt-4",
            "top_k": 5,
            "temperature": 0.7
        }
    }
)

print(f"\n=== RAG 系统执行结果 ===")
print(f"输入类型: RAGInput")
print(f"输出类型: RAGOutput")
print(f"回答: {result.get('answer', 'N/A')}")
print(f"来源数量: {len(result.get('sources', []))}")
print(f"置信度: {result.get('confidence', 0)}")
print(f"元数据: {result.get('metadata', {})}")
```

## 类型推断机制

```python
"""
类型推断机制示例
展示 LangGraph 如何自动推断节点的输入类型
"""
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
import inspect

# 定义多个状态类型
class State1(TypedDict):
    value: int

class State2(TypedDict):
    value: int
    name: str

class State3(TypedDict):
    value: int
    name: str
    data: list

# 创建图
graph = StateGraph[State1](state_schema=State1)

# 节点1：自动推断输入类型为 State2
def node1(state: State2) -> State2:
    """LangGraph 会从函数签名推断输入类型"""
    print(f"Node1 输入类型: State2")
    return {"value": state["value"] + 1, "name": state.get("name", "default")}

# 节点2：自动推断输入类型为 State3
def node2(state: State3) -> State3:
    """LangGraph 会从函数签名推断输入类型"""
    print(f"Node2 输入类型: State3")
    return {
        "value": state["value"] + 1,
        "name": state["name"],
        "data": state.get("data", []) + ["processed"]
    }

# 节点3：显式指定输入类型
def node3(state: State1) -> State1:
    """显式指定输入类型"""
    print(f"Node3 输入类型: State1 (显式指定)")
    return {"value": state["value"] + 1}

# 添加节点 - 自动推断
graph.add_node("node1", node1)  # 推断为 State2
graph.add_node("node2", node2)  # 推断为 State3

# 添加节点 - 显式指定
graph.add_node("node3", node3, input_schema=State1)

graph.add_edge(START, "node1")
graph.add_edge("node1", "node2")
graph.add_edge("node2", "node3")
graph.add_edge("node3", END)

app = graph.compile()

# 测试
result = app.invoke({"value": 0})
print(f"最终结果: {result}")

# 查看推断的类型
print(f"\n=== 类型推断信息 ===")
for node_name, node_spec in graph.nodes.items():
    sig = inspect.signature(node_spec.action)
    params = list(sig.parameters.values())
    if params:
        first_param = params[0]
        print(f"{node_name}: 推断类型 = {first_param.annotation}")
```

## 最佳实践总结

### 何时使用泛型类型

**推荐场景**：
- 需要分离输入输出类型（API 边界清晰）
- 需要不可变的运行时上下文（配置、认证信息）
- 需要类型安全的节点定义（避免类型错误）
- 复杂的多阶段处理流程

**不推荐场景**：
- 简单的单一状态流程
- 输入输出类型相同
- 不需要上下文信息

### 关键要点

1. **StateT**：内部状态，节点间通信使用
2. **ContextT**：不可变上下文，通过 config 传递
3. **InputT**：输入类型，自动转换为内部状态
4. **OutputT**：输出类型，从内部状态提取
5. **类型推断**：从函数签名自动推断，也可显式指定

### 类型系统设计原则

1. **最小输入**：输入类型只包含必需字段
2. **完整内部**：内部状态包含所有处理信息
3. **精简输出**：输出类型只包含用户需要的字段
4. **不可变上下文**：上下文信息不应在节点中修改
