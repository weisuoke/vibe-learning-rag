### RAG 查询函数

```python
def rag_query(question, category=None, language=None, difficulty=None, min_rating=None, top_k=3):
    """RAG 混合检索查询"""

    # 1. 生成查询向量
    query_vector = model.encode(question).tolist()

    # 2. 构建过滤条件
    filters = []
    if category:
        filters.append(f"metadata['category'] == '{category}'")
    if language:
        filters.append(f"metadata['language'] == '{language}'")
    if difficulty:
        filters.append(f"metadata['difficulty'] == '{difficulty}'")
    if min_rating:
        filters.append(f"metadata['rating'] >= {min_rating}")

    filter_expr = " AND ".join(filters) if filters else None

    # 3. 执行混合检索
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter=filter_expr,
        limit=top_k,
        output_fields=["text", "metadata"]
    )

    # 4. 提取上下文
    contexts = []
    for hits in results:
        for hit in hits:
            contexts.append({
                "text": hit['entity']['text'],
                "metadata": hit['entity']['metadata'],
                "distance": hit['distance']
            })

    # 5. 构建 Prompt
    context_text = "\n\n".join([f"文档{i+1}: {ctx['text']}" for i, ctx in enumerate(contexts)])

    prompt = f"""基于以下文档回答问题：

{context_text}

问题：{question}

请基于上述文档内容回答问题。如果文档中没有相关信息，请说明。"""

    # 6. 调用 LLM
    response = openai_client.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "system", "content": "你是一个helpful的助手，基于提供的文档回答问题。"},
            {"role": "user", "content": prompt}
        ]
    )

    answer = response.choices[0].message.content

    return {
        "question": question,
        "answer": answer,
        "contexts": contexts,
        "filter": filter_expr
    }

# 测试查询
result = rag_query(
    question="What is machine learning?",
    category="AI",
    language="en",
    min_rating=4.0
)

print(f"问题: {result['question']}")
print(f"过滤条件: {result['filter']}")
print(f"\n检索到的文档:")
for i, ctx in enumerate(result['contexts']):
    print(f"  {i+1}. {ctx['text'][:50]}... (距离: {ctx['distance']:.4f})")
print(f"\n回答: {result['answer']}")
```

```python
def customer_service_query(user_question, category=None, language=None, priority=None):
    """智能客服查询"""

    # 生成查询向量
    query_vector = model.encode(user_question).tolist()

    # 构建过滤条件
    filters = []
    if category:
        filters.append(f"metadata['category'] == '{category}'")
    if language:
        filters.append(f"metadata['language'] == '{language}'")
    if priority:
        filters.append(f"metadata['priority'] == '{priority}'")

    filter_expr = " AND ".join(filters) if filters else None

    # 执行检索
    results = client.search(
        collection_name=collection_name,
        data=[query_vector],
        filter=filter_expr,
        limit=3,
        output_fields=["question", "answer", "metadata"]
    )

    # 提取最佳答案
    best_match = None
    for hits in results:
        if hits:
            best_match = hits[0]
            break

    if best_match:
        return {
            "user_question": user_question,
            "matched_question": best_match['entity']['question'],
            "answer": best_match['entity']['answer'],
            "confidence": 1 - best_match['distance'],
            "metadata": best_match['entity']['metadata']
        }
    else:
        return {
            "user_question": user_question,
            "answer": "抱歉，我没有找到相关答案。请联系人工客服。",
            "confidence": 0
        }

# 测试查询

)

print(f"答案: {result['answer']}")
```

## 场景3：多语言文档检索

### 完整实现

```python
class MultilingualRAG:
    """多语言 RAG 系统"""

    def __init__(self, collection_name="multilingual_docs"):
        self.client = MilvusClient("http://localhost:19530")
        self.model = SentenceTransformer('all-MiniLM-L6-v2')
        self.collection_name = collection_name
        self._setup_collection()

    def _setup_collection(self):
        """设置集合和索引"""
        if self.client.has_collection(self.collection_name):
            self.client.drop_collection(self.collection_name)

        schema = self.client.create_schema(auto_id=True)
        schema.add_field("id", DataType.INT64, is_primary=True)
        schema.add_field("vector", DataType.FLOAT_VECTOR, dim=384)
        schema.add_field("text", DataType.VARCHAR, max_length=2000)
        schema.add_field("metadata", DataType.JSON)

        self.client.create_collection(
            collection_name=self.collection_name,
            schema=schema
        )

        # 创建索引
        index_params = self.client.prepare_index_params()
        index_params.add_index("vector", "AUTOINDEX", "COSINE")
        index_params.add_index(
            field_name="metadata",
            index_name="language_index",
            index_type="BITMAP",
            params={"json_path": "metadata['language']", "json_cast_type": "varchar"}
        )
        index_params.add_index(
            field_name="metadata",
            index_name="category_index",
            index_type="INVERTED",
            params={"json_path": "metadata['category']", "json_cast_type": "varchar"}
        )

        self.client.create_index(self.collection_name, index_params)

    def add_documents(self, documents):
        """添加文档"""
        data = []
        for doc in documents:
            vector = self.model.encode(doc["text"]).tolist()
            data.append({
                "vector": vector,
                "text": doc["text"],
                "metadata": doc["metadata"]
            })

        self.client.insert(self.collection_name, data)
        return len(data)

    def search(self, query, language=None, category=None, top_k=5):
        """搜索文档"""
        query_vector = self.model.encode(query).tolist()

        filters = []
        if language:
            filters.append(f"metadata['language'] == '{language}'")
        if category:
            filters.append(f"metadata['category'] == '{category}'")

        filter_expr = " AND ".join(filters) if filters else None

        results = self.client.search(
            collection_name=self.collection_name,
            data=[query_vector],
            filter=filter_expr,
            limit=top_k,
            output_fields=["text", "metadata"]
        )

        documents = []
        for hits in results:
            for hit in hits:
                documents.append({
                    "text": hit['entity']['text'],
                    "metadata": hit['entity']['metadata'],
                    "score": 1 - hit['distance']
                })

        return documents

# 使用示例
rag = MultilingualRAG()

# 添加文档

    {

        "metadata": {"language": "en", "category": "programming"}

    print(f"文本: {doc['text']}")
    print(f"得分: {doc['score']:.4f}\n")
```

### 批量查询优化

```python
def batch_rag_query(questions, category=None, language=None):
    """批量 RAG 查询"""

    # 批量生成向量
    query_vectors = [model.encode(q).tolist() for q in questions]

    # 构建过滤条件
    filters = []
    if category:
        filters.append(f"metadata['category'] == '{category}'")
    if language:
        filters.append(f"metadata['language'] == '{language}'")

    filter_expr = " AND ".join(filters) if filters else None

    # 批量检索
    results = client.search(
        collection_name=collection_name,
        data=query_vectors,
        filter=filter_expr,
        limit=3,
        output_fields=["text", "metadata"]
    )

    # 处理结果
    answers = []
    for i, hits in enumerate(results):
        contexts = [hit['entity']['text'] for hit in hits]
        answers.append({
            "question": questions[i],
            "contexts": contexts
        })

    return answers

# 测试批量查询

    "What is Python?",
    "What is machine learning?",
    "What is deep learning?"
]

results = batch_rag_query(questions, category="AI", language="en")
for result in results:
    print(f"问题: {result['question']}")
    print(f"上下文数量: {len(result['contexts'])}\n")
```

```python
        {

            "metadata": {"category": "tech", "language": "en", "rating": 4.5}

            "metadata": {"category": "AI", "language": "en", "rating": 4.8}
        }
    ]

    data = []
    for doc in documents:
        vector = model.encode(doc["text"]).tolist()
        data.append({
            "vector": vector,
            "text": doc["text"],

    # 创建索引
    index_params = client.prepare_index_params()

    query = "What is machine learning?"
    query_vector = model.encode(query).tolist()

        data=[query_vector],

        limit=3,

    print(f"延迟: {elapsed:.2f} ms\n")

    for hits in results:
        for hit in hits:
            print(f"文本: {hit['entity']['text']}")
            print(f"类别: {hit['entity']['metadata']['category']}")
            print(f"得分: {1 - hit['distance']:.4f}\n")

if __name__ == "__main__":
    main()
```

## 常见问题

### Q1: 如何平衡向量检索和标量过滤？

**答案**：
```python
# 策略1：先过滤再检索（高选择性）
filter = "metadata['category'] == 'AI' AND metadata['rating'] >= 4.5"
# 选择性高（< 10%），使用 Pre-filtering

# 策略2：先检索再过滤（低选择性）
filter = "metadata['language'] == 'en'"
# 选择性低（> 50%），使用 Post-filtering

# Milvus 会自动选择最优策略
```

```python
# 方案1：使用多语言 Embedding 模型
```

## 类比理解

### 前端开发类比

**RAG 混合检索 = 多条件搜索 + 排序**：
```javascript
// JavaScript
const results = products
  .filter(p => p.category === 'electronics' && p.rating >= 4.0)  // 标量过滤
  .sort((a, b) => similarity(a, query) - similarity(b, query))  // 向量检索
  .slice(0, 10);  // Top-10

// Milvus RAG
filter = "metadata['category'] == 'electronics' AND metadata['rating'] >= 4.0"
// + 向量检索
```

### 日常生活类比

**RAG 混合检索 = 图书馆找书**：
```
1. 标量过滤：按分类、语言、出版年份筛选
2. 向量检索：在筛选结果中找内容最相关的
3. 返回结果：Top-K 最相关的书

= RAG 混合检索
```
